**渲染层工作原理**

```javascript
// 小程序渲染层核心实现
class MiniProgramRenderer {
  constructor(pageId) {
    this.pageId = pageId;
    this.virtualDOM = null;
    this.pendingUpdate = false;
  }

  // 接收逻辑层数据更新
  onSetData(data) {
    // 1. 合并数据
    this.mergeData(data);
    // 2. 执行虚拟DOM diff
    const patches = this.diff(this.virtualDOM, this.newVirtualDOM);
    // 3. 批量更新DOM
    this.applyPatches(patches);
  }

  // 虚拟DOM diff算法
  diff(oldVNode, newVNode) {
    if (!oldVNode) {
      return { type: 'INSERT', node: newVNode };
    }
    if (!newVNode) {
      return { type: 'REMOVE', node: oldVNode };
    }
    if (oldVNode.type !== newVNode.type) {
      return { type: 'REPLACE', oldNode: oldVNode, newNode: newVNode };
    }
    if (oldVNode.type === 'TEXT') {
      if (oldVNode.content !== newVNode.content) {
        return { type: 'UPDATE_TEXT', content: newVNode.content };
      }
      return null;
    }
    // 元素节点diff
    return this.diffElement(oldVNode, newVNode);
  }

  diffElement(oldEl, newEl) {
    const patches = [];
    // 比较属性
    const propsDiff = this.diffProps(oldEl.props, newEl.props);
    if (propsDiff.length > 0) {
      patches.push({ type: 'UPDATE_PROPS', props: propsDiff });
    }
    // 比较子节点
    const childrenDiff = this.diffChildren(oldEl.children, newEl.children);
    if (childrenDiff.length > 0) {
      patches.push({ type: 'UPDATE_CHILDREN', children: childrenDiff });
    }
    return patches;
  }
}
```

```javascript
Page({

    list: []
  },

  // 不推荐 - 每次都重新渲染整个列表
  updateListBad(newItems) {
    this.setData({ list: newItems });
  },

  // 优化 - 只更新变化的项
  updateListGood(newItems) {
    const oldList = this.data.list;
    const updates = {};

    newItems.forEach((item, index) => {
      if (JSON.stringify(item) !== JSON.stringify(oldList[index])) {
```

**长列表优化实战**

```javascript
// 分页长列表组件
const virtualListBehavior = Behavior({
  data: {
    visibleItems: [],
    startIndex: 0,
    endIndex: 20,
    scrollTop: 0
  },

  methods: {
    initVirtualList(totalCount, itemHeight) {
      this._totalCount = totalCount;
      this._itemHeight = itemHeight;
      this._bufferSize = 5;
      this.updateVisibleItems(0);
    },

    onScroll(e) {
      const scrollTop = e.detail.scrollTop;
      const startIndex = Math.floor(scrollTop / this._itemHeight);
      const visibleCount = Math.ceil(this.getViewportHeight() / this._itemHeight);

      const newStartIndex = Math.max(0, startIndex - this._bufferSize);
      const newEndIndex = Math.min(
        this._totalCount,
        startIndex + visibleCount + this._bufferSize
      );

      if (newStartIndex !== this.data.startIndex ||
          newEndIndex !== this.data.endIndex) {
        this.updateVisibleItems(newStartIndex, newEndIndex, scrollTop);
      }
    },

    updateVisibleItems(startIndex, endIndex, scrollTop) {
      const items = this.getItemsByRange(startIndex, endIndex);
      this.setData({
        visibleItems: items,
        startIndex: startIndex,
        endIndex: endIndex,
        scrollTop: scrollTop,
        totalHeight: this._totalCount * this._itemHeight,
        offsetY: startIndex * this._itemHeight
      });
    },

    getItemsByRange(start, end) {
      // 实际实现中从数据源获取指定范围的数据
      return this._allItems.slice(start, end);
    },

    getViewportHeight() {
      return wx.getSystemInfoSync().windowHeight;
    }
  }
});

// 使用虚拟列表的页面
Page({
```

```javascript
      // WebP格式支持检测与转换

        optimizedUrl = this.convertToWebP(src);
      }

      // 添加图片压缩参数
      if (this.needCompress(src)) {
        optimizedUrl = this.addCompressParams(src);
      }

      this.setData({
        optimizedSrc: optimizedUrl,
        loading: true,
        error: false
      });
    },

    supportWebP() {
      const systemInfo = wx.getSystemInfoSync();
      // iOS 14+ 和 Android 5.0+ 支持WebP
      return true; // 实际需要根据systemInfo判断
    },

    convertToWebP(url) {
      // 假设有图片服务器支持格式转换
      if (url.includes('?')) {
        return `${url}&format=webp`;
      }
      return `${url}?format=webp`;
    },

    addCompressParams(url) {
      const params = '&quality=80&width=750';
      if (url.includes('?')) {
        return `${url}${params}`;
      }
      return `${url}?imageView${params}`;
    },

    needCompress(url) {
      // 根据URL特征判断是否需要压缩
      return !url.includes('cloud://');
    },

    onImageLoad(e) {
      this.setData({ loading: false });
      this.triggerEvent('load', e.detail);
    },

    onImageError(e) {
      this.setData({
        loading: false,
        error: true
      });
      this.triggerEvent('error', e.detail);
    }
  }
});
```

**请求封装与拦截器**

```javascript
// utils/request.js
class Request {
  constructor() {
    this.baseURL = 'https://api.weixin.com';
    this.timeout = 10000;
    this.interceptors = {
      request: [],
      response: []
    };
  }

  // 添加请求拦截器
  useRequestInterceptor(interceptor) {
    this.interceptors.request.push(interceptor);
  }

  // 添加响应拦截器
  useResponseInterceptor(interceptor) {
    this.interceptors.response.push(interceptor);
  }

  // 执行请求
  request(options) {
    const config = {
      url: options.url,
      method: options.method || 'GET',
      data: options.data || {},
      header: options.header || {},
      timeout: options.timeout || this.timeout
    };

    // 处理URL
    if (!config.url.startsWith('http')) {
      config.url = this.baseURL + config.url;
    }

    // 执行请求拦截器
    let finalConfig = config;
    for (const interceptor of this.interceptors.request) {
      finalConfig = interceptor(finalConfig) || finalConfig;
    }

    return new Promise((resolve, reject) => {
      wx.request({
        ...finalConfig,
        success: (res) => {
          // 执行响应拦截器
          let finalResponse = res;
          for (const interceptor of this.interceptors.response) {
            finalResponse = interceptor(finalResponse) || finalResponse;
          }

          if (finalResponse.statusCode >= 200 && finalResponse.statusCode < 300) {
            resolve(finalResponse.data);
          } else {
            reject(finalResponse);
          }
        },
        fail: (err) => {
          reject(err);
        }
      });
    });
  }

  // 快捷方法
  get(url, params, options = {}) {
    return this.request({
      url,
      method: 'GET',
      data: params,
      ...options
    });
  }

  post(url, data, options = {}) {
    return this.request({
      url,
      method: 'POST',
      data,
      ...options
    });
  }

  put(url, data, options = {}) {
    return this.request({
      url,
      method: 'PUT',
      data,
      ...options
    });
  }

  delete(url, data, options = {}) {
    return this.request({
      url,
      method: 'DELETE',
      data,
      ...options
    });
  }
}
```

```javascript
function debounce(fn, delay = 300) {
  let timer = null;
  return function(...args) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

function throttle(fn, delay = 300) {
  let lastTime = 0;
  let timer = null;
  return function(...args) {
    const now = Date.now();
    if (now - lastTime >= delay) {
      fn.apply(this, args);
      lastTime = now;
    } else {
      if (timer) clearTimeout(timer);
      timer = setTimeout(() => {
        fn.apply(this, args);
        lastTime = Date.now();
      }, delay - (now - lastTime));
    }
  };
}

      this.render();

    const startIndex = Math.floor(this.scrollTop / this.options.itemHeight);

      end: Math.min(

            <div style="height: ${this.options.itemHeight}px;">
```

```javascript
// 简化版MVVM框架实现
class MVVM {
  constructor(options) {
    this.$options = options;
    this.$data = options.data;
    this.$el = options.el;

    // 数据代理
    this.proxyData();

    // 数据监听
    this.observe(this.$data);

    // 模板编译
    this.$compile = new Compile(this.$el, this);
  }

  proxyData() {
    Object.keys(this.$data).forEach(key => {
      Object.defineProperty(this, key, {
        enumerable: true,
        configurable: true,
        get: () => this.$data[key],
        set: (val) => {
          this.$data[key] = val;
        }
      });
    });
  }

  observe(data) {
    if (!data || typeof data !== 'object') return;

    Object.keys(data).forEach(key => {
      let value = data[key];
      const dep = new Dep();

      Object.defineProperty(data, key, {
        enumerable: true,
        configurable: true,
        get() {
          if (Dep.target) {
            dep.addSub(Dep.target);
          }
          return value;
        },
        set(val) {
          if (value === val) return;
          value = val;
          dep.notify();
        }
      });

      if (typeof value === 'object') {
        this.observe(value);
      }
    });
  }
}

// 观察者
class Watcher {
  constructor(vm, exp, callback) {
    this.vm = vm;
    this.exp = exp;
    this.callback = callback;
    this.value = this.get();
  }

  get() {
    Dep.target = this;
    const value = this.vm.$data[this.exp];
    Dep.target = null;
    return value;
  }

  update() {
    const newValue = this.vm.$data[this.exp];
    const oldValue = this.value;
    if (newValue !== oldValue) {
      this.callback(newValue, oldValue);
      this.value = newValue;
    }
  }
}

// 模板编译器
class Compile {
  constructor(el, vm) {
    this.el = document.querySelector(el);
    this.vm = vm;
    this.fragment = null;
    this.init();
  }

  init() {
    this.fragment = this.nodeToFragment(this.el);
    this.compileElement(this.fragment);
    this.el.appendChild(this.fragment);
  }

  nodeToFragment(node) {
    const fragment = document.createDocumentFragment();
    let child;
    while (child = node.firstChild) {
      fragment.appendChild(child);
    }
    return fragment;
  }

  compileElement(node) {
    const childNodes = node.childNodes;
    Array.from(childNodes).forEach(node => {
      if (this.isElementNode(node)) {
        this.compile(node);
      } else if (this.isTextNode(node)) {
        this.compileText(node);
      }

      if (node.childNodes && node.childNodes.length) {
        this.compileElement(node);
      }
    });
  }

  isElementNode(node) {
    return node.nodeType === 1;
  }

  isTextNode(node) {
    return node.nodeType === 3;
  }

  compile(node) {
    const attrs = node.attributes;
    Array.from(attrs).forEach(attr => {
      const name = attr.name;
      if (this.isDirective(name)) {
        const exp = attr.value;
        const dir = name.substring(2);

        if (this.isEventDirective(dir)) {
          this.compileEvent(node, exp, dir);
        } else {
          this.compileModel(node, exp, dir);
        }
      }
    });
  }

  isDirective(attr) {
    return attr.startsWith('v-');
  }

  isEventDirective(dir) {
    return dir.startsWith('on:');
  }

  compileEvent(node, exp, dir) {
    const eventType = dir.substring(3);
    const method = this.vm.$options.methods && this.vm.$options.methods[exp];

    if (eventType && method) {
      node.addEventListener(eventType, method.bind(this.vm));
    }
  }

  compileModel(node, exp, dir) {
    const self = this;
    new Watcher(this.vm, exp, function(newValue) {
      node.value = newValue;
    });

    node.addEventListener('input', function(e) {
      self.vm.$data[exp] = e.target.value;
    });

    node.value = this.vm.$data[exp];
  }

  compileText(node) {
    const reg = /\{\{((.|\n)+?)\}\}/g;
    const text = node.textContent;

    if (reg.test(text)) {
      node.textContent = text.replace(reg, function(_, exp) {
        new Watcher(this.vm, exp.trim(), function(newValue) {
          node.textContent = newValue;
        });
        return this.vm.$data[exp.trim()];
      }.bind(this));
    }
  }
}
```

```javascript
class Store {
  constructor(options) {
    this._state = options.state || {};
    this._mutations = options.mutations || {};
    this._actions = options.actions || {};
    this._getters = options.getters || {};

    // 绑定this
    this.commit = this.commit.bind(this);
    this.dispatch = this.dispatch.bind(this);

    // 实现getters
    const self = this;
    this._computed = {};
    Object.keys(this._getters).forEach(key => {
      Object.defineProperty(self._computed, key, {
        get() {
          return self._getters[key](self.state);
        }
      });
    });
  }

  get state() {
    return this._state;
  }

  get getters() {
    return this._computed;
  }

  commit(type, payload) {
    const mutation = this._mutations[type];
    if (!mutation) {
      throw new Error(`Mutation ${type} not found`);
    }
    mutation(this._state, payload);
  }

  dispatch(type, payload) {
    const action = this._actions[type];
    if (!action) {
      throw new Error(`Action ${type} not found`);
    }
    // actions中可以通过context访问state、commit、dispatch
    const context = {
      state: this._state,
      commit: this.commit,
      dispatch: this.dispatch,
      getters: this.getters
    };
    return action(context, payload);
  }
}
```

```javascript
class MicroFrontend {
  constructor() {
    this.apps = new Map();
    this.currentApp = null;
  }

  // 注册子应用
  registerApp(name, config) {
    this.apps.set(name, {
      name,
      entry: config.entry,
      container: config.container,
      activeWhen: config.activeWhen,
      props: config.props || {}
    });
  }

  // 启动
  start() {
    // 监听路由变化
    window.addEventListener('popstate', () => this.handleRouterChange());
    this.handleRouterChange();
  }

  // 处理路由变化
  handleRouterChange() {
    const currentPath = window.location.pathname;

    for (const [name, app] of this.apps.entries()) {
      if (this.matchPath(currentPath, app.activeWhen)) {
        if (this.currentApp !== name) {
          this.loadApp(name);
        }
        return;
      }
    }

    // 没有匹配的应用
    if (this.currentApp) {
      this.unmountApp(this.currentApp);
      this.currentApp = null;
    }
  }

  matchPath(pathname, activeWhen) {
    if (typeof activeWhen === 'string') {
      return pathname.startsWith(activeWhen);
    }
    if (activeWhen instanceof RegExp) {
      return activeWhen.test(pathname);
    }
    if (typeof activeWhen === 'function') {
      return activeWhen(pathname);
    }
    return false;
  }

  // 加载应用
  async loadApp(name) {
    const app = this.apps.get(name);

    // 卸载当前应用
    if (this.currentApp && this.currentApp !== name) {
      await this.unmountApp(this.currentApp);
    }

    // 加载新应用
    const container = document.querySelector(app.container);
    container.innerHTML = '<div>Loading...</div>';

    try {
      // 加载脚本和样式
      await this.loadResources(app.entry);

      // 获取应用入口
      const entry = await this.getEntry(app.entry);

      // 挂载应用
      const mount = entry.mount;
      await mount(app.container, app.props);

      this.currentApp = name;
    } catch (error) {
      console.error(`Failed to load app ${name}:`, error);
    }
  }

  // 加载资源
  async loadResources(entry) {
    const html = await fetch(entry).then(res => res.text());

    // 解析HTML获取资源
    const parser = new DOMParser();
    const doc = parser.parseFromString(html, 'text/html');

    // 加载样式
    const styles = doc.querySelectorAll('link[rel="stylesheet"]');
    await Promise.all(Array.from(styles).map(style => {
      return new Promise((resolve, reject) => {
        const link = document.createElement('link');
        link.rel = 'stylesheet';
        link.href = new URL(style.href, entry).href;
        link.onload = resolve;
        link.onerror = reject;
        document.head.appendChild(link);
      });
    }));

    // 加载脚本
    const scripts = doc.querySelectorAll('script');
    await Promise.all(Array.from(scripts).map(script => {
      return new Promise((resolve, reject) => {
        const s = document.createElement('script');
        s.src = new URL(script.src, entry).href;
        s.onload = resolve;
        s.onerror = reject;
        document.head.appendChild(s);
      });
    }));

    return html;
  }

  // 获取应用入口

    // 假设入口导出 mount 和 unmount 函数

  }

    if (window.__micro_frontend_entry__ && window.__micro_frontend_entry__.unmount) {
      await window.__micro_frontend_entry__.unmount(app.container);
    }

  }
```

```javascript
// 手写Promise

class MyPromise {
  constructor(executor) {
    this.state = 'pending';
    this.value = undefined;
    this.reason = undefined;
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];

    const resolve = (value) => {
      if (this.state === 'pending') {
        this.state = 'fulfilled';
        this.value = value;
        this.onFulfilledCallbacks.forEach(cb => cb(this.value));
      }
    };

    const reject = (reason) => {
      if (this.state === 'pending') {
        this.state = 'rejected';
        this.reason = reason;
        this.onRejectedCallbacks.forEach(cb => cb(this.reason));
      }
    };

    try {
      executor(resolve, reject);
    } catch (error) {
      reject(error);
    }
  }

  then(onFulfilled, onRejected) {
    // 返回新的Promise
    return new MyPromise((resolve, reject) => {
      if (this.state === 'fulfilled') {
        try {
          const result = onFulfilled ? onFulfilled(this.value) : this.value;
          resolve(result);
        } catch (error) {
          reject(error);
        }
      }

      if (this.state === 'rejected') {
        try {
          const result = onRejected ? onRejected(this.reason) : this.reason;
          reject(result);
        } catch (error) {
          reject(error);
        }
      }

      if (this.state === 'pending') {
        if (onFulfilled) {
          this.onFulfilledCallbacks.push((value) => {
            try {
              const result = onFulfilled(value);
              resolve(result);
            } catch (error) {
              reject(error);
            }
          });
        }

        if (onRejected) {
          this.onRejectedCallbacks.push((reason) => {
            try {
              const result = onRejected(reason);
              resolve(result);
            } catch (error) {
              reject(error);
            }
          });
        }
      }
    });
  }

  catch(onRejected) {
    return this.then(null, onRejected);
  }

  finally(onFinally) {
    return this.then(
      (value) => {
        onFinally();
        return value;
      },
      (reason) => {
        onFinally();
        throw reason;
      }
    );
  }

  static resolve(value) {
    return new MyPromise((resolve) => resolve(value));
  }

  static reject(reason) {
    return new MyPromise((_, reject) => reject(reason));
  }

  static all(promises) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(promises)) {
        return reject(new TypeError('Arguments must be an array'));
      }

      const results = [];
      let count = 0;

      if (promises.length === 0) {
        return resolve(results);
      }

      promises.forEach((promise, index) => {
        MyPromise.resolve(promise).then(
          (value) => {
            results[index] = value;
            count++;
            if (count === promises.length) {
              resolve(results);
            }
          },
          (reason) => {
            reject(reason);
          }
        );
      });
    });
  }

  static race(promises) {
    return new MyPromise((resolve, reject) => {
      promises.forEach((promise) => {
        MyPromise.resolve(promise).then(resolve, reject);
      });
    });
  }

  static allSettled(promises) {
    return new MyPromise((resolve) => {
      const results = [];
      let count = 0;

      if (promises.length === 0) {
        return resolve(results);
      }

      promises.forEach((promise, index) => {
```

```javascript
// 防抖 (Debounce) - 等待停止后执行

function debounce(fn, delay, immediate = false) {
  let timer = null;

  return function(...args) {
    const context = this;

    // 立即执行模式
    if (immediate && !timer) {
      fn.apply(context, args);
    }

    // 清除之前的定时器
    if (timer) {
      clearTimeout(timer);
    }

    // 设置新的定时器
    timer = setTimeout(() => {
      if (!immediate) {
        fn.apply(context, args);
      }
      timer = null;
    }, delay);
  };
}

// 节流 (Throttle) - 固定时间间隔执行

function throttle(fn, delay) {
  let lastTime = 0;
  let timer = null;

  return function(...args) {
    const context = this;
    const now = Date.now();

    // 如果距离上次执行超过delay，立即执行
    if (now - lastTime >= delay) {
      fn.apply(context, args);
      lastTime = now;
    } else {
      // 否则等delay时间后执行
      if (timer) {
        clearTimeout(timer);
      }
      timer = setTimeout(() => {
        fn.apply(context, args);
        lastTime = Date.now();
        timer = null;
      }, delay - (now - lastTime));
    }
  };
}

  };
```

```javascript
function deepClone(target) {

  if (target === null || typeof target !== 'object') {

  }

  // 处理日期
  if (target instanceof Date) {

function deepCloneWithCycle(target) {
  const hash = new WeakMap();

  function clone(target) {
    if (target === null || typeof target !== 'object') {
      return target;
    }

    // 解决循环引用
    if (hash.has(target)) {
      return hash.get(target);
    }

    // 处理日期
    if (target instanceof Date) {
      const result = new Date(target.getTime());
      hash.set(target, result);
      return result;
    }

    // 处理正则
    if (target instanceof RegExp) {
      const result = new RegExp(target.source, target.flags);
      hash.set(target, result);
      return result;
    }

    // 处理数组
    if (Array.isArray(target)) {
      const result = [];
      hash.set(target, result);
      for (const item of target) {
        result.push(clone(item));
      }
      return result;
    }

    // 处理对象
    const result = {};
    hash.set(target, result);
    for (const key in target) {
      if (target.hasOwnProperty(key)) {
        result[key] = clone(target[key]);
      }
    }

    return result;
  }

  return clone(target);
}

// 3. 完整版深拷贝（包含更多类型）
function deepCloneComplete(target, hash = new WeakMap()) {
  // 基础类型直接返回
  if (target === null || typeof target !== 'object') {
    return target;
  }

  // 避免重复引用
  if (hash.has(target)) {
    return hash.get(target);
  }

  // 处理Date
  if (target instanceof Date) {
    const date = new Date(target.getTime());
    hash.set(target, date);
    return date;
  }

  // 处理RegExp
  if (target instanceof RegExp) {
    const regex = new RegExp(target.source, target.flags);
    hash.set(target, regex);
    return regex;
  }

  // 处理Set
  if (target instanceof Set) {
    const set = new Set();
    hash.set(target, set);
    target.forEach(value => {
      set.add(deepCloneComplete(value, hash));
    });
    return set;
  }

  // 处理Map
  if (target instanceof Map) {
    const map = new Map();
    hash.set(target, map);
    target.forEach((value, key) => {
      map.set(key, deepCloneComplete(value, hash));
    });
    return map;
  }

  // 处理Array和Object
  const isArray = Array.isArray(target);
  const clone = isArray ? [] : {};
  hash.set(target, clone);

  if (isArray) {
    target.forEach((item, index) => {
      clone[index] = deepCloneComplete(item, hash);
    });
  } else {
    Object.keys(target).forEach(key => {
      clone[key] = deepCloneComplete(target[key], hash);
    });
  }

  return clone;
}
```

```javascript
class TreeNode {
  constructor(value) {

    this.left = null;
    this.right = null;
  }

// 前序遍历 - 根 -> 左 -> 右
function preOrder(root) {
  const result = [];

  function traverse(node) {
    if (!node) return;

    result.push(node.value);  // 访问根
    traverse(node.left);      // 遍历左子树
    traverse(node.right);     // 遍历右子树
  }

  traverse(root);
  return result;
}

// 非递归前序遍历
function preOrderIterative(root) {
  const result = [];
  const stack = [];
  let current = root;

  while (current || stack.length) {
    while (current) {
      result.push(current.value);
      stack.push(current);
      current = current.left;
    }

    current = stack.pop();
    current = current.right;
  }

  return result;
}

// 中序遍历 - 左 -> 根 -> 右
function inOrder(root) {
  const result = [];

  function traverse(node) {
    if (!node) return;

    traverse(node.left);
    result.push(node.value);
    traverse(node.right);
  }

  traverse(root);
  return result;
}

// 非递归中序遍历
function inOrderIterative(root) {
  const result = [];
  const stack = [];
  let current = root;

  while (current || stack.length) {
    while (current) {
      stack.push(current);
      current = current.left;
    }

    current = stack.pop();
    result.push(current.value);
    current = current.right;
  }

  return result;
}

function postOrder(root) {

  function traverse(node) {
    if (!node) return;

    traverse(node.left);
    traverse(node.right);
    result.push(node.value);
  }

  traverse(root);
  return result;
}

// 非递归后序遍历
function postOrderIterative(root) {
  const result = [];
  const stack = [];
  let current = root;
  let lastVisited = null;

  while (current || stack.length) {
    while (current) {
      stack.push(current);
      current = current.left;
    }

    const peekNode = stack[stack.length - 1];

    if (peekNode.right && peekNode.right !== lastVisited) {
      current = peekNode.right;
    } else {
      result.push(peekNode.value);
      lastVisited = stack.pop();
    }
  }

  return result;
}

// 层序遍历 (BFS)
function levelOrder(root) {
  if (!root) return [];

  const result = [];
  const queue = [root];

  while (queue.length) {
    const level = [];
    const levelSize = queue.length;

    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift();
      level.push(node.value);

      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }

    result.push(level);
  }

  return result;
}

// 二叉树构建
function buildTree(preorder, inorder) {
  if (!preorder.length || !inorder.length) return null;

  const rootValue = preorder[0];
  const root = new TreeNode(rootValue);

  const rootIndexInorder = inorder.indexOf(rootValue);

  const leftInorder = inorder.slice(0, rootIndexInorder);
  const rightInorder = inorder.slice(rootIndexInorder + 1);

  const leftPreorder = preorder.slice(1, 1 + leftInorder.length);
  const rightPreorder = preorder.slice(1 + leftInorder.length);

  root.left = buildTree(leftPreorder, leftInorder);
  root.right = buildTree(rightPreorder, rightInorder);

  return root;
}
```

```javascript
// 冒泡排序
function bubbleSort(arr) {
  const n = arr.length;

  for (let i = 0; i < n - 1; i++) {
    let swapped = false;

    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        swapped = true;
      }
    }

    // 如果没有交换，说明已经有序
    if (!swapped) break;
  }

  return arr;
}

// 快速排序
function quickSort(arr, left = 0, right = arr.length - 1) {
  if (left >= right) return arr;

  const pivotIndex = partition(arr, left, right);
  quickSort(arr, left, pivotIndex - 1);
  quickSort(arr, pivotIndex + 1, right);

  return arr;
}

function partition(arr, left, right) {
  const pivot = arr[right];
  let i = left - 1;

  for (let j = left; j < right; j++) {
    if (arr[j] <= pivot) {
      i++;
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
  }

  [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
  return i + 1;
}

// 归并排序
function mergeSort(arr) {
  if (arr.length <= 1) return arr;

  const mid = Math.floor(arr.length / 2);
  const left = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));

  return merge(left, right);
}

function merge(left, right) {
  const result = [];
  let i = 0, j = 0;

  while (i < left.length && j < right.length) {
    if (left[i] <= right[j]) {
      result.push(left[i++]);
    } else {
      result.push(right[j++]);
    }
  }

  return result.concat(left.slice(i)).concat(right.slice(j));
}

// 堆排序
function heapSort(arr) {
  const n = arr.length;

  // 构建最大堆
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i);
  }

  // 排序
  for (let i = n - 1; i > 0; i--) {
    [arr[0], arr[i]] = [arr[i], arr[0]];
    heapify(arr, i, 0);
  }

  return arr;
}

function heapify(arr, n, i) {
  let largest = i;
  const left = 2 * i + 1;
  const right = 2 * i + 2;

  if (left < n && arr[left] > arr[largest]) {
    largest = left;
  }

  if (right < n && arr[right] > arr[largest]) {
    largest = right;
  }

  if (largest !== i) {
    [arr[i], arr[largest]] = [arr[largest], arr[i]];
    heapify(arr, n, largest);
  }
}

// 计数排序（适用于整数）
function countingSort(arr) {
  if (arr.length === 0) return arr;

  const min = Math.min(...arr);
  const max = Math.max(...arr);
  const range = max - min + 1;

  const count = new Array(range).fill(0);
  const output = new Array(arr.length);

  // 计数
  for (const num of arr) {
    count[num - min]++;
  }

  // 累加
  for (let i = 1; i < range; i++) {
    count[i] += count[i - 1];
  }

  // 排序
  for (let i = arr.length - 1; i >= 0; i--) {
    output[count[arr[i] - min] - 1] = arr[i];
    count[arr[i] - min]--;
  }

  return output;
}
```

```javascript
// LRU (Least Recently Used) 缓存实现

class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();
  }

  // 获取值
  get(key) {
    if (!this.cache.has(key)) {
      return -1;
    }

    // 将访问的元素移到末尾（最近使用）
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);

    return value;
  }

  // 设置值
  put(key, value) {
    // 如果key已存在，删除
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }
    // 如果容量已满，删除最旧的元素（Map的第一个元素）
    else if (this.cache.size >= this.capacity) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    // 添加新元素
    this.cache.set(key, value);
  }

  // 获取所有数据（用于调试）
  getAll() {
    return Array.from(this.cache.entries());
  }
}

// 使用双向链表优化（更高效）
class LRUCacheOptimized {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();

    // 虚拟头尾节点
    this.head = { key: null, value: null, prev: null, next: null };
    this.tail = { key: null, value: null, prev: null, next: null };
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }

  // 移动节点到头部（最近使用）
  moveToHead(node) {
    this.removeNode(node);
    this.addToHead(node);
  }

  // 添加节点到头部
  addToHead(node) {
    node.prev = this.head;
    node.next = this.head.next;
    this.head.next.prev = node;
    this.head.next = node;
  }

  // 移除节点
  removeNode(node) {
    node.prev.next = node.next;
    node.next.prev = node.prev;
  }

  // 获取值
  get(key) {
    const node = this.cache.get(key);
    if (!node) return -1;

    this.moveToHead(node);
    return node.value;
  }

  // 设置值
  put(key, value) {
    let node = this.cache.get(key);

    if (node) {
      // 更新值并移到头部
      node.value = value;
      this.moveToHead(node);
    } else {
      // 创建新节点
      node = { key, value, prev: null, next: null };
      this.cache.set(key, node);
      this.addToHead(node);

      // 超过容量删除尾部节点
      if (this.cache.size > this.capacity) {
        const tailNode = this.tail.prev;
        this.removeNode(tailNode);
        this.cache.delete(tailNode.key);
      }
    }
  }

  // 打印缓存内容
  print() {
    const result = [];
    let node = this.head.next;
    while (node !== this.tail) {
      result.push({ key: node.key, value: node.value });
      node = node.next;
    }
    return result;
  }
}
```

```javascript
Component({
  properties: {
    value: {
      type: Number,
      value: 0

      type: Number,
      value: 0
    },

      type: Number,
      value: 999
    },

      type: Number,
      value: 1
    },
    disabled: {
      type: Boolean,
      value: false
    },
    size: {
      type: String,
      value: 'normal' // small, normal, large
    }
  },

  data: {
    localValue: 0,
    minusDisabled: false,
    plusDisabled: false
  },

  observers: {
    'value': function(newVal) {
      this.setData({
        localValue: newVal
      });
      this.updateDisabledState(newVal);
    }
  },

  lifetimes: {
    attached() {
      this.setData({
        localValue: this.data.value
      });
      this.updateDisabledState(this.data.value);
    }
  },

  methods: {
    updateDisabledState(value) {
      this.setData({
        minusDisabled: value <= this.data.min,
        plusDisabled: value >= this.data.max
      });
    },

    onMinus() {
      if (this.data.disabled || this.data.minusDisabled) return;

      const newValue = this.data.localValue - this.data.step;
      this.handleChange(newValue);
    },

    onPlus() {
      if (this.data.disabled || this.data.plusDisabled) return;

      const newValue = this.data.localValue + this.data.step;
      this.handleChange(newValue);
    },

    onInput(e) {
      let value = parseInt(e.detail.value) || this.data.min;
      value = Math.max(this.data.min, Math.min(value, this.data.max));
      this.handleChange(value, true);
    },

    onBlur(e) {
      let value = parseInt(e.detail.value) || this.data.min;
      value = Math.max(this.data.min, Math.min(value, this.data.max));
      this.handleChange(value);
    },

    handleChange(value, isInput = false) {
      this.setData({ localValue: value });
      this.updateDisabledState(value);

      if (!isInput) {
        this.triggerEvent('change', { value });
      }
    },

    onMinusTap() {
      this.onMinus();
    },

    onPlusTap() {
      this.onPlus();
    }
  }
});
```

```javascript
// scripts/build.js - 小程序构建脚本
const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

const ROOT_DIR = path.resolve(__dirname, '..');
const SRC_DIR = path.join(ROOT_DIR, 'src');
const DIST_DIR = path.join(ROOT_DIR, 'dist');

// 构建配置
const config = {
  // 需要复制到输出目录的文件
  copyFiles: [
    'project.config.json',
    'sitemap.json'
  ],

  // 需要编译的目录
  srcDirs: [
    'components',
    'pages',
    'utils',
    'services',
    'store',
    'styles',
    'assets'
  ],

  // 需要排除的文件
  excludeFiles: [
    '**/*.test.js',
    '**/*.spec.js',
    '**/.*'
  ]
};

// 日志函数
const log = {
  info: (...args) => console.log('\x1b[34m[INFO]\x1b[0m', ...args),
  success: (...args) => console.log('\x1b[32m[SUCCESS]\x1b[0m', ...args),
  error: (...args) => console.log('\x1b[31m[ERROR]\x1b[0m', ...args),
  warn: (...args) => console.log('\x1b[33m[WARN]\x1b[0m', ...args)
};

// 检查文件是否应该排除
function shouldExclude(filePath) {
  return config.excludeFiles.some(pattern => {
    if (pattern.includes('**')) {
      const regex = new RegExp(
        pattern.replace('**', '.*').replace('*', '[^/]+')
      );
      return regex.test(filePath);
    }
    return filePath.includes(pattern);
  });
}

// 复制文件
function copyFile(src, dest) {
  const destDir = path.dirname(dest);
  if (!fs.existsSync(destDir)) {
    fs.mkdirSync(destDir, { recursive: true });
  }
  fs.copyFileSync(src, dest);
  log.info(`Copied: ${src} -> ${dest}`);
}

// 复制目录
function copyDir(src, dest) {
  if (!fs.existsSync(src)) return;

  if (!fs.existsSync(dest)) {
    fs.mkdirSync(dest, { recursive: true });
  }

  const entries = fs.readdirSync(src, { withFileTypes: true });

  for (const entry of entries) {
    const srcPath = path.join(src, entry.name);
    const destPath = path.join(dest, entry.name);

    if (shouldExclude(srcPath)) {
      continue;
    }

    if (entry.isDirectory()) {
      copyDir(srcPath, destPath);
    } else {
      copyFile(srcPath, destPath);
    }
  }
}

// 清理输出目录
function cleanDist() {
  if (fs.existsSync(DIST_DIR)) {
    fs.rmSync(DIST_DIR, { recursive: true });
  }
  fs.mkdirSync(DIST_DIR, { recursive: true });
  log.info('Cleaned dist directory');
}

// 处理页面和组件的依赖
function processDependencies() {
  const jsonFiles = [];

  // 递归查找所有json文件
  function findJsonFiles(dir) {
    const entries = fs.readdirSync(dir, { withFileTypes: true });

    for (const entry of entries) {
      const fullPath = path.join(dir, entry.name);

      if (entry.isDirectory()) {
        findJsonFiles(fullPath);
      } else if (entry.name.endsWith('.json')) {
        jsonFiles.push(fullPath);
      }
    }
  }

  findJsonFiles(SRC_DIR);

  // 提取所有JS文件引用
  const jsFiles = new Set();
  const pages = new Set();
  const usingComponents = {};

  for (const jsonFile of jsonFiles) {
    try {
      const content = JSON.parse(fs.readFileSync(jsonFile, 'utf-8'));

      // 处理页面配置
      if (content.pages) {
        for (const page of content.pages) {
          pages.add(page);
          jsFiles.add(path.join(SRC_DIR, 'pages', page + '.js'));
        }
      }

      // 处理组件
      if (content.usingComponents) {
        const dir = path.dirname(jsonFile);
        for (const [name, component] of Object.entries(content.usingComponents)) {
          // 处理相对路径
          if (component.startsWith('./') || component.startsWith('../')) {
            const componentPath = path.resolve(dir, component);
            jsFiles.add(componentPath + '.js');
          }
        }
      }
    } catch (error) {
      // 忽略错误
    }
  }

  return { pages, jsFiles: Array.from(jsFiles) };
}

// 主构建流程
async function build() {
  const startTime = Date.now();

  log.info('Starting build...');
  log.info(`Source: ${SRC_DIR}`);
  log.info(`Destination: ${DIST_DIR}`);

  try {
    // 1. 清理输出目录
    cleanDist();

    // 2. 复制静态配置文件
    for (const file of config.copyFiles) {
      const srcPath = path.join(ROOT_DIR, file);
      const destPath = path.join(DIST_DIR, file);
      if (fs.existsSync(srcPath)) {
        copyFile(srcPath, destPath);
      }
    }

    // 3. 复制源代码目录
    for (const dir of config.srcDirs) {
      const srcPath = path.join(SRC_DIR, dir);
      const destPath = path.join(DIST_DIR, dir);

      if (fs.existsSync(srcPath)) {
        copyDir(srcPath, destPath);
      }
    }

    // 4. 处理依赖
    const { pages, jsFiles } = processDependencies();

    // 5. 输出构建信息
    log.success('Build completed!');
    log.info(`Total pages: ${pages.size}`);
    log.info(`Total JS files: ${jsFiles.length}`);
    log.info(`Time: ${Date.now() - startTime}ms`);

  } catch (error) {
    log.error('Build failed:', error.message);
    process.exit(1);
  }
}

// 执行构建
build();
```

```javascript
    let left = 0;
    let right = arr.length - 1;

    while (left <= right) {
      const mid = Math.floor((left + right) / 2);
      if (arr[mid] === target) return mid;
      if (arr[mid] < target) left = mid + 1;
      else right = mid - 1;
    }
    return -1;
  },

  // O(n) - 线性时间

    let sum = 0;
    for (const item of arr) {
      sum += item;
    }
```
