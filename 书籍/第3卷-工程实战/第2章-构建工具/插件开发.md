#### 2.9.2 完整的自定义 Loader 示例

```javascript
// markdown-loader

const marked = require('marked');
const frontMatter = require('front-matter');
const hljs = require('highlight.js');

/**
 * Markdown 转 HTML 的自定义 Loader
 */
module.exports = function(source) {
  const options = this.getOptions() || {};

  // 解析 Front Matter
  const { attributes, body } = frontMatter(source);

  // 配置 marked
  marked.setOptions({
    highlight: (code, lang) => {
      if (lang && hljs.getLanguage(lang)) {
        try {
          return hljs.highlight(code, { language: lang }).value;
        } catch (e) {}
      }
      return hljs.highlightAuto(code).value;
    }
  });

  // 转换为 HTML
  const html = marked(body);

  // 生成导出的 JavaScript 代码
  const result = `
    export const attributes = ${JSON.stringify(attributes)};
    export const html = ${JSON.stringify(html)};
    export const raw = ${JSON.stringify(body)};
  `;

  return result;
};
```

#### 2.10.1 Plugin 基础

```javascript
// my-plugin/index.js

/**
 * 自定义 Webpack Plugin 基础模板
 */
class MyPlugin {
  /**
   * @param {object} options - 插件选项
   */
  constructor(options = {}) {
    this.options = options;
  }

  /**
   * 应用插件
   * @param {Compiler} compiler - Webpack 编译器
   */
  apply(compiler) {
    // 监听编译开始
    compiler.hooks.run.tap('MyPlugin', (compilation) => {
      console.log('Webpack 开始编译...');
    });

    // 监听编译完成
    compiler.hooks.done.tap('MyPlugin', (stats) => {
      console.log('Webpack 编译完成!');
    });

    // 监听 emit 事件（输出前）
    compiler.hooks.emit.tap('MyPlugin', (compilation) => {
      // 添加额外文件
      compilation.assets['manifest.json'] = {
        source: () => JSON.stringify({
          timestamp: Date.now(),
          hash: compilation.hash
        }),
        size: () => 50
      };
    });

    // 监听 compilation 事件
    compiler.hooks.compilation.tap('MyPlugin', (compilation) => {
      compilation.hooks.optimizeModules.tap('MyPlugin', (modules) => {
        console.log(`优化了 ${modules.length} 个模块`);
      });
    });
  }
}

module.exports = MyPlugin;
```

#### 2.10.2 完整的自定义 Plugin 示例

```javascript
// 生成文件列表插件

class FileListPlugin {
  constructor(options = {}) {
    this.options = {
      filename: 'filelist.md',
      title: 'Build Files',
      ...options
    };
  }

  apply(compiler) {
    compiler.hooks.emit.tap('FileListPlugin', (compilation) => {
      // 获取所有输出文件
      const assets = Object.keys(compilation.assets);

      // 生成文件列表内容
      let content = `# ${this.options.title}\n\n`;
      content += `生成时间: ${new Date().toLocaleString()}\n\n`;
      content += `## 文件列表\n\n`;
      content += `| 文件名 | 大小 |\n`;
      content += `|--------|------|\n`;

      assets.forEach((filename) => {
        const asset = compilation.assets[filename];
        const size = asset.size();
        content += `| ${filename} | ${this.formatSize(size)} |\n`;
      });

      // 添加到输出
      compilation.assets[this.options.filename] = {
        source: () => content,
        size: () => content.length
      };
    });
  }

  formatSize(bytes) {
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(2) + ' KB';
    return (bytes / (1024 * 1024)).toFixed(2) + ' MB';
  }
}

// 使用
// webpack.config.js
const FileListPlugin = require('./my-plugin');

module.exports = {
  plugins: [
    new FileListPlugin({
      filename: 'FILELIST.md',
      title: '构建产物列表'
    })
  ]
};
```

#### 2.10.3 常用 Hooks

```javascript
class AdvancedPlugin {
  apply(compiler) {
    // 同步 Hooks
    compiler.hooks.afterPlugins.tap('AdvancedPlugin', () => {
      console.log('所有插件加载完成');
    });

    compiler.hooks.compile.tap('AdvancedPlugin', () => {
      console.log('开始编译');
    });

    compiler.hooks.emit.tap('AdvancedPlugin', (compilation) => {
      // 输出文件生成前
    });

    // 异步 Hooks
    compiler.hooks.beforeRun.async('AdvancedPlugin', (compilation, callback) => {
      // 编译运行前
      setTimeout(() => {
        console.log('beforeRun 完成');
        callback();
      }, 100);
    });

    compiler.hooks.done.tapAsync('AdvancedPlugin', (stats, callback) => {
      // 编译完成后
      console.log('编译完成');
      callback();
    });

    // 监听特定模块
    compiler.hooks.compilation.tap('AdvancedPlugin', (compilation) => {
      compilation.hooks.succeedModule.tap('AdvancedPlugin', (module) => {
        console.log('模块编译成功:', module.resource);
      });

      compilation.hooks.failedModule.tap('AdvancedPlugin', (module, error) => {
        console.log('模块编译失败:', module.resource, error);
      });
    });
  }
}
```

#### 2.12.1 环境变量注入插件

```javascript
class EnvironmentPlugin {
  constructor(options) {
    this.env = options || process.env;
  }

  apply(compiler) {
    compiler.hooks.compilation.tap('EnvironmentPlugin', (compilation) => {
      compilation.hooks.optimizeModules.tap('EnvironmentPlugin', (modules) => {
        modules.forEach((module) => {
          if (module.resource && module.resource.endsWith('.js')) {
            // 替换环境变量
            const source = module.originalSource();
            if (source) {
              let code = source.source();
              Object.keys(this.env).forEach((key) => {
                const value = JSON.stringify(this.env[key]);
                code = code.replace(
                  new RegExp(`process\\.env\\.${key}`, 'g'),
                  value
                );
              });
              module.updateHash(code);
            }
          }
        });
      });
    });
  }
}
```

#### 2.12.2 自动生成组件文档插件

```javascript
class ComponentDocsPlugin {
  constructor(options) {
    this.options = {
      output: 'component-docs.md',
      ...options
    };
  }

  apply(compiler) {
    compiler.hooks.emit.tapAsync('ComponentDocsPlugin', (compilation, callback) => {
      const docs = [];

      // 遍历所有模块
      compilation.modules.forEach((module) => {
        if (module.resource && module.resource.endsWith('.vue')) {
          const source = module.originalSource();
          if (source) {
            // 提取组件信息
            const content = source.source();
            const nameMatch = content.match(/name:\s*['"]([^'"]+)['"]/);
            const propsMatch = content.match(/props:\s*{([^}]+)}/);
            const eventsMatch = content.match(/emits:\s*{([^}]+)}/);

            if (nameMatch) {
              docs.push({
                name: nameMatch[1],
                props: propsMatch ? propsMatch[1].trim() : 'None',
                events: eventsMatch ? eventsMatch[1].trim() : 'None'
              });
            }
          }
        }
      });

      // 生成文档
      let docContent = '# Component Documentation\n\n';
      docs.forEach((doc) => {
        docContent += `## ${doc.name}\n\n`;
        docContent += `### Props\n${doc.props}\n\n`;
        docContent += `### Events\n${doc.events}\n\n`;
      });

      // 添加到输出
      compilation.assets[this.options.output] = {
        source: () => docContent,
        size: () => docContent.length
      };

      callback();
    });
  }
}
```
