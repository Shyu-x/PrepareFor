### 1.3 窗口层级管理（Z-Index）

```typescript
// 使用全局 Z-Index 计数器
const MAX_BASE_ZINDEX = 10000;

const useWindowManager = create<WindowManagerState>()((set, get) => ({
  windows: [],
  activeWindowId: null,

  bringToFront: (windowId: string) => {
    set(state => {
      const maxZ = Math.max(...state.windows.map(w => w.zIndex), MAX_BASE_ZINDEX);

      return {
        activeWindowId: windowId,
        windows: state.windows.map(w =>
          w.id === windowId
            ? { ...w, zIndex: maxZ + 1 }
            : w
        ),
      };
    });
  },

  // 最小化时记录当前层级
  minimize: (windowId: string) => {
    set(state => ({
      windows: state.windows.map(w =>
        w.id === windowId
          ? { ...w, minimized: true, prevZIndex: w.zIndex }
          : w
      ),
    }));
  },

  // 恢复时回到之前的层级
  restore: (windowId: string) => {
    set(state => ({
      windows: state.windows.map(w =>
        w.id === windowId
          ? { ...w, minimized: false, zIndex: w.prevZIndex || MAX_BASE_ZINDEX + 1 }
          : w
      ),
    }));
  },
}));
```

### 2.1 大文件处理与分块存储

```typescript
// 分块存储实现
const CHUNK_SIZE = 1024 * 1024; // 1MB per chunk

class ChunkedFileStorage {
  async saveLargeFile(path: string, content: ArrayBuffer): Promise<string> {
    const fileId = crypto.randomUUID();
    const totalChunks = Math.ceil(content.byteLength / CHUNK_SIZE);

    // 保存元数据
    await this.db.write({
      id: fileId,
      path,
      type: 'file',
      totalChunks,
      totalSize: content.byteLength,
      chunkSize: CHUNK_SIZE,
    });

    // 分块写入
    for (let i = 0; i < totalChunks; i++) {
      const start = i * CHUNK_SIZE;
      const end = Math.min(start + CHUNK_SIZE, content.byteLength);
      const chunk = content.slice(start, end);

      await this.db.write({
        id: `${fileId}_chunk_${i}`,
        parentId: fileId,
        chunkIndex: i,
        data: chunk,
      });
    }

    return fileId;
  }

  async readLargeFile(fileId: string): Promise<ArrayBuffer> {
    const metadata = await this.db.read(fileId);
    const chunks: Uint8Array[] = [];

    for (let i = 0; i < metadata.totalChunks; i++) {
      const chunk = await this.db.read(`${fileId}_chunk_${i}`);
      chunks.push(new Uint8Array(chunk.data));
    }

    // 合并
    const totalLength = chunks.reduce((sum, c) => sum + c.byteLength, 0);
    const result = new Uint8Array(totalLength);
    let offset = 0;
    for (const chunk of chunks) {
      result.set(chunk, offset);
      offset += chunk.byteLength;
    }

    return result.buffer;
  }
}
```

### 2.2 离线支持与数据同步

```typescript
// 离线操作队列
interface SyncOperation {
  id: string;
  type: 'create' | 'update' | 'delete';
  path: string;
  data?: any;
  timestamp: number;
  status: 'pending' | 'syncing' | 'completed' | 'failed';
}

class OfflineSyncManager {
  private operationQueue: SyncOperation[] = [];
  private isOnline = navigator.onLine;

  constructor() {
    window.addEventListener('online', () => this.sync());
    window.addEventListener('offline', () => this.handleOffline());
  }

  // 添加操作到队列
  async queueOperation(operation: Omit<SyncOperation, 'id' | 'timestamp' | 'status'>) {
    const op: SyncOperation = {
      ...operation,
      id: crypto.randomUUID(),
      timestamp: Date.now(),
      status: 'pending',
    };

    this.operationQueue.push(op);

    if (this.isOnline) {
      await this.sync();
    }
  }

  // 同步操作
  async sync() {
    const pendingOps = this.operationQueue.filter(op => op.status === 'pending');

    for (const op of pendingOps) {
      try {
        op.status = 'syncing';
        await this.executeRemoteOperation(op);
        op.status = 'completed';
      } catch (error) {
        op.status = 'failed';
        // 重试逻辑
      }
    }

    // 清理已完成操作
    this.operationQueue = this.operationQueue.filter(op => op.status !== 'completed');
  }

  // 冲突检测
  private async detectConflict(op: SyncOperation, remoteMeta: FileMetadata): Promise<boolean> {
    const localMeta = await this.getLocalMetadata(op.path);
    if (!localMeta) return false;
    return localMeta.updatedAt > remoteMeta.updatedAt && op.timestamp < remoteMeta.updatedAt;
  }
}
```

### 3.1 PTY 多进程管理

```typescript
// PTY 进程池实现
class PTYProcessPool {
  private pool: Map<string, Terminal> = new Map();
  private maxProcesses = 100;
  private idleTimeout = 5 * 60 * 1000; // 5分钟

  async acquire(sessionId: string, options: PTYOptions): Promise<Terminal> {
    if (this.pool.has(sessionId)) {
      return this.pool.get(sessionId)!;
    }

    // 超过最大进程数，清理空闲进程
    if (this.pool.size >= this.maxProcesses) {
      await this.cleanupIdleProcesses();
    }

    const process = createTerminal(options.shell, [], {
      cols: options.cols || 80,
      rows: options.rows || 24,
      cwd: options.cwd || process.env.HOME,
    });

    this.pool.set(sessionId, process);
    this.setupIdleTimeout(sessionId);

    return process;
  }

  private setupIdleTimeout(sessionId: string) {
    const process = this.pool.get(sessionId);
    if (!process) return;

    let idleTimer: NodeJS.Timeout;

    // 有数据活动时重置定时器
    process.onData(() => {
      clearTimeout(idleTimer);
      this.setupIdleTimeout(sessionId);
    });

    // 空闲超时释放
    idleTimer = setTimeout(() => {
      this.release(sessionId);
    }, this.idleTimeout);
  }

  async release(sessionId: string) {
    const process = this.pool.get(sessionId);
    if (process) {
      process.kill();
      this.pool.delete(sessionId);
    }
  }
}
```

### 5.1 技术难点分析

### 5.1 浏览器渲染原理与窗口系统

#### 5.1.1 CSS Transform vs Position

| 属性 | 触发重排 | 触发重绘 | GPU加速 |
|------|----------|----------|---------|
| `left/top` | ✅ | ✅ | ❌ |
| `transform` | ❌ | ❌ | ✅ |

**浏览器渲染流程**:
1. **DOM 树构建** -> 2. **CSS 树构建** -> 3. **渲染树** -> 4. **布局(Layout)** -> 5. **绘制(Paint)** -> 6. **合成(Composite)**

使用 `left/top` 修改位置会触发步骤 4（布局），这是最耗时的操作。使用 `transform: translate()` 只触发步骤 6（合成），因为它在一个独立的合成层中运行。

#### 5.1.2 Z-Index 合成层

**浏览器分层机制**:
- 某些 CSS 属性会创建新的合成层（Compositing Layer）
- 合成层独立于主线程渲染，不受 JS 阻塞影响
- 使用 `will-change: transform` 可以主动创建合成层

**内存考虑**:
- 每个合成层占用额外 GPU 内存
- 过度使用会导致内存压力，反而降低性能

### 5.2 虚拟文件系统架构

#### 5.2.1 IndexedDB 存储机制

**IndexedDB 架构**:
```
Database
  ┗━━ ObjectStore (类似表)
        ┗━━ Index (索引)
              ┗━━ Record (记录)
```

**事务机制**:
- IndexedDB 所有操作都在事务中执行
- 只读事务可以并发，写事务会阻塞
- 使用事务可以保证数据一致性

```typescript
// 权限级别
enum Permission {
  READ = 'read',      // 只读
  WRITE = 'write',    // 读写
  ADMIN = 'admin',    // 管理
}

// 文件访问控制
interface AccessControl {
  ownerId: string;
  permissions: {
    [userId: string]: Permission[];
  };
  publicLevel: Permission;
}

// 权限检查
function checkAccess(ac: AccessControl, userId: string, required: Permission): boolean {
  if (ac.ownerId === userId) return true;
  if (ac.permissions[userId]?.includes(required)) return true;
  if (ac.permissions[userId]?.includes(Permission.ADMIN)) return true;
  if (ac.publicLevel === required) return true;
  return false;
}
```
