### 1.1 Scene3D 主场景组件

**文件路径**: `frontend/src/components/game/Scene3D.tsx`

Scene3D 是整个 3D 游戏的核心容器，使用 React Three Fiber 构建完整的 3D 游戏环境。

```typescript
import { Canvas } from '@react-three/fiber';
import { Stars, PerspectiveCamera } from '@react-three/drei';
import { useMemo } from 'react';
import * as THREE from 'three';

export default function Scene3D({
  gameState,
  onCardClick,
  onDrawClick,
  onColorSelect,
  onUnoClick
}: Scene3DProps) {
  // 响应式相机配置
  const cameraConfig = useMemo(() => {
    const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
    const isPortrait = window.innerHeight > window.innerWidth;
    const playerCount = gameState?.players.length || 4;

    let cameraPos: [number, number, number];
    let cameraFov: number;

    if (isMobile) {
      if (isPortrait) {
        cameraPos = [0, 55, 70];
        cameraFov = 60;
      } else {
        cameraPos = [0, 35, 48];
        cameraFov = 45;
      }
    } else {
      if (playerCount >= 10) {
        cameraPos = [0, 28, 45];
        cameraFov = 28;
      } else if (playerCount >= 6) {
        cameraPos = [0, 25, 40];
        cameraFov = 36;
      } else {
        cameraPos = [0, 22, 35];
        cameraFov = 35;
      }
    }

    return { position: cameraPos, fov: cameraFov };
  }, [gameState?.players.length]);

  return (
    <Canvas shadows>
      {/* 相机系统 */}
      <PerspectiveCamera
        makeDefault
        position={cameraConfig.position}
        fov={cameraConfig.fov}
      />

      {/* 环境系统 */}
      <Stars radius={100} depth={50} count={5000} factor={4} fade speed={1} />
      <color attach="background" args={['#0a0a1a']} />
      <fog attach="fog" args={['#0a0a1a', 30, 80]} />

      {/* 灯光系统 */}
      <ambientLight intensity={1.2} />
      <pointLight position={[0, 30, 0]} intensity={150} castShadow />
      <spotLight
        position={[0, 50, 0]}
        angle={0.5}
        penumbra={0.6}
        intensity={200}
        color="#fef3c7"
      />

      {/* 游戏桌面 */}
      <GameTable gameState={gameState} />

      {/* 玩家区域 */}
      {gameState?.players.map((player, index) => (
        <PlayerArea
          key={player.id}
          player={player}
          position={calculatePlayerPosition(index, gameState.players.length)}
          isCurrentPlayer={gameState.currentPlayerIndex === index}
        />
      ))}

      {/* 卡牌堆和弃牌堆 */}
      <Deck3D onClick={onDrawClick} />
      <DiscardPile3D cards={gameState?.discardPile || []} />

      {/* 特效层 */}
      <ParticleEffects />
    </Canvas>
  );
}

// 计算玩家在 3D 空间中的位置
function calculatePlayerPosition(index: number, totalPlayers: number): [number, number, number] {
  const radius = 15;
  const angle = (index / totalPlayers) * Math.PI * 2 - Math.PI / 2;
  return [
    Math.cos(angle) * radius,
    0,
    Math.sin(angle) * radius
  ];
}
```

**技术难点**:
1. **性能优化**: 3D 场景中需要同时渲染大量卡牌和粒子效果，需要使用 `useMemo` 缓存计算结果
2. **响应式适配**: 不同设备需要不同的相机参数，需要动态计算
3. **阴影渲染**: 使用 `castShadow` 和 `receiveShadow` 需要合理设置光源位置

### 1.2 Card3D 3D 卡牌组件

使用 Three.js 的 BoxGeometry 创建 3D 卡牌模型，结合 React Spring 实现动画效果。

```typescript
import { useRef, useState } from 'react';
import { useFrame } from '@react-three/fiber';
import { useSpring, animated } from '@react-spring/three';
import * as THREE from 'three';

interface Card3DProps {
  card: UnoCard;
  position: [number, number, number];
  rotation?: [number, number, number];
  isPlayable?: boolean;
  isHovered?: boolean;
  isHinted?: boolean;
  onClick?: () => void;
}

export function Card3D({
  card,
  position,
  rotation = [0, 0, 0],
  isPlayable = false,
  isHovered = false,
  isHinted = false,
  onClick
}: Card3DProps) {
  const meshRef = useRef<THREE.Mesh>(null);
  const [hovered, setHovered] = useState(false);

  // 使用 React Spring 实现动画
  const { springPos, springScale, springRot } = useSpring({
    springPos: [
      position[0],
      position[1] + (hovered ? 2.5 : (isHinted ? 0.8 : 0)),
      position[2] + (hovered ? 1.5 : 0)
    ] as [number, number, number],
    springScale: hovered ? 1.25 : 1,
    springRot: rotation as [number, number, number],
    config: { mass: 1, tension: 350, friction: 35 }
  });

  return (
    <animated.group
      position={springPos}
      scale={springScale}
      rotation={springRot}
    >
      <mesh
        ref={meshRef}
        onClick={(e) => {
          e.stopPropagation();
          if (isPlayable) onClick?.();
        }}
        castShadow
        receiveShadow
      >
        {/* 卡牌主体 */}
        <boxGeometry args={[2.5, 3.5, 0.18]} />
        <meshStandardMaterial color={COLOR_MAP[card.color]} roughness={0.4} metalness={0.1} />
      </mesh>
    </animated.group>
  );
}
```

```typescript
export function ConfettiParticles({ active, count = 200 }: ConfettiParticlesProps) {
  const pointsRef = useRef<THREE.Points>(null);

  const { positions, velocities, colors, sizes } = useMemo(() => {
    const positions = new Float32Array(count * 3);
    const velocities = new Float32Array(count * 3);
    const colors = new Float32Array(count * 3);
    const sizes = new Float32Array(count);

    for (let i = 0; i < count; i++) {
      const i3 = i * 3;
      positions[i3] = (Math.random() - 0.5) * 10;
      positions[i3 + 1] = Math.random() * 20 + 10;
      positions[i3 + 2] = (Math.random() - 0.5) * 10;
      velocities[i3 + 1] = -Math.random() * 0.2 - 0.05;
      // ...
    }
    return { positions, velocities, colors, sizes };
  }, [count]);

  useFrame((_, delta) => {
    if (!active || !pointsRef.current) return;
    // 更新粒子位置...
  });

  return (
    <points ref={pointsRef}>
      <bufferGeometry>
        <bufferAttribute attach="attributes-position" count={count} array={positions} itemSize={3} />
      </bufferGeometry>
      <pointsMaterial size={0.3} vertexColors transparent opacity={0.8} sizeAttenuation blending={THREE.AdditiveBlending} />
    </points>
  );
}
```

### 2.1 Scene2D 主场景组件

```typescript
export default function Scene2D({
  gameState,
  playerId,
  onCardClick,
  onDrawClick,
  onColorSelect,
  onUnoClick
}: Scene2DProps) {
  const config = useMemo(() => {
    if (isMobile) {
      return {
        cardSize: 'small' as const,
        handHeight: isPortraitMode ? 'h-20' : 'h-24',
        handCardOffset: 4,
        playerRadius: isPortraitMobile ? 20 : 28,
      };
    } else {
      return {
        cardSize: 'large' as const,
        handHeight: 'h-52',
        handCardOffset: 8,
        playerRadius: 46,
      };
    }
  }, [isMobile, isTablet, isPortrait]);

  const getPlayerPositions = (count: number, myIndex: number) => {
    const positions = [];
    const radius = config.playerRadius;
    const startAngle = -140;

    for (let i = 0; i < count; i++) {
      if (i === myIndex) {
        positions.push({ x: 50, y: 85, rotation: 0 });
      } else {
        // 其他玩家在上半圆弧分布
        const otherPlayers = [];
        for (let j = 0; j < count; j++) {
          if (j !== myIndex) otherPlayers.push(j);
        }
        const angle = startAngle + (otherPlayers.indexOf(i) * 160 / Math.max(1, otherPlayers.length - 1));
        positions.push({
          x: 50 + radius * Math.cos(angle * Math.PI / 180),
          y: 40 + radius * Math.sin(angle * Math.PI / 180) * 0.5,
          rotation: 0,
        });
      }
    }
    return positions;
  };

  return (
    <div className="relative w-full h-full">
      <Table2D />
      <Deck2D onClick={onDrawClick} remaining={gameState?.deck.length || 0} />
      <DiscardPile2D topCard={gameState?.discardPile.slice(-1)[0]} />
      <DirectionIndicator2D direction={gameState?.direction} />
      {gameState?.players.map((player, index) => (
        <PlayerArea2D
          key={player.id}
          player={player}
          position={getPlayerPositions(gameState.players.length, playerIndex)[index]}
          isCurrentPlayer={gameState.currentPlayerIndex === index}
        />
      ))}
      <Hand2D
        cards={myHand}
        onCardClick={onCardClick}
        canPlayCard={isMyTurn}
      />
      {needsColorPicker && <ColorPicker2D onSelect={onColorSelect} />}
      {shouldShowUnoButton && <UnoButton2D onClick={onUnoClick} />}
    </div>
  );
}
```

```typescript
class GameSocketContext {
  private socket: Socket | null = null;

  connect(url: string) {
    this.socket = io(url, {
      transports: ['websocket'],
      reconnection: true,
    });

    this.socket.on('connect', () => console.log('Connected'));
    this.socket.on('gameStateUpdate', (state) => {
      useGameStore.getState().setGameState(state);
    });

      useGameStore.getState().addNotification({
        type: 'info',

      });
    });
```

### 4.1 AI 决策算法

```typescript
class AiService {
  getBestMove(hand: Card[], topCard: Card, currentColor: CardColor): Card | null {
    const playableCards = hand.filter(card => this.isPlayable(card, topCard, currentColor));
    if (playableCards.length === 0) return null;

    // 简单难度：随机选择
    if (difficulty === 'EASY') {
      return playableCards[Math.floor(Math.random() * playableCards.length)];
    }

    // 中等难度：优先普通牌
    if (difficulty === 'MEDIUM') {
      const colorCount = this.countColors(hand);
      const bestColor = Object.entries(colorCount).sort((a, b) => b[1] - a[1])[0]?.[0];
      return playableCards.find(c => c.color === bestColor && c.type !== 'WILD') || playableCards[0];
    }

    // 困难难度：攻击策略
    if (difficulty === 'HARD') {
      // 攻击下家
      if (hand.length <= 3) {
        const actionCards = playableCards.filter(c => c.type !== 'NUMBER');
        if (actionCards.length > 0) return actionCards[0];
      }
      // 清理最多颜色
      const colorCount = this.countColors(hand);
      const bestColor = Object.entries(colorCount).sort((a, b) => b[1] - a[1])[0]?.[0];
      return playableCards.find(c => c.color === bestColor) || playableCards[0];
    }
  }

  private isPlayable(card: Card, topCard: Card, currentColor: CardColor): boolean {
    return card.color === CardColor.WILD ||
           card.color === currentColor ||
           (card.type === topCard.type && card.type !== 'NUMBER') ||
           (card.type === 'NUMBER' && topCard.type === 'NUMBER' && card.value === topCard.value);
  }
}
```

### 5.1 音频管理器

```typescript
class AudioManager {
  private context: AudioContext | null = null;
  private sounds: Map<string, AudioBuffer> = new Map();

  async init() {
    this.context = new (window.AudioContext || (window as any).webkitAudioContext)();
    await this.loadSounds();
  }

  async loadSounds() {
    const soundFiles = ['login', 'match-success', 'game-start', 'play-card', 'shout-uno', 'challenge', 'draw-card', 'win', 'lose', 'game-over'];
    for (const sound of soundFiles) {
      const response = await fetch(`/sounds/${sound}.mp3`);
      const arrayBuffer = await response.arrayBuffer();
      const audioBuffer = await this.context!.decodeAudioData(arrayBuffer);
      this.sounds.set(sound, audioBuffer);
    }
  }

  play(soundName: string, volume: number = 1) {
    if (!this.context || !this.sounds.has(soundName)) return;

    const source = this.context.createBufferSource();
    const gainNode = this.context.createGain();
    source.buffer = this.sounds.get(soundName)!;
    gainNode.gain.value = volume;
    source.connect(gainNode);
    gainNode.connect(this.context.destination);
    source.start();
  }
}

export const audioManager = new AudioManager();
```
