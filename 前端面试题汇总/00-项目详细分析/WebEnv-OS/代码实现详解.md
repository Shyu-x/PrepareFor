```tsx
"use client";

import { useState, useRef, useEffect, ReactNode } from "react";

interface WindowProps {
  title: string;
  children: ReactNode;
  initialPosition?: { x: number; y: number };

  visible?: boolean;
  onClose?: () => void;
  onMinimize?: () => void;
  onMaximize?: () => void;
  resizable?: boolean;
  draggable?: boolean;
  zIndex?: number;
  className?: string;
  style?: React.CSSProperties;
  onMouseDown?: () => void;
  windowControls?: "windows" | "macos" | "linux";
}

export default function Window({
  title,
  children,
  initialPosition = { x: 100, y: 100 },
  initialSize = { width: 600, height: 400 },
  visible = true,
  onClose,
  onMinimize,
  onMaximize,
  resizable = true,
  draggable = true,
  zIndex = 1000,
  className = "",
  style = {},
  onMouseDown,
  minSize = { width: 400, height: 300 },

  // 状态管理
  const [position, setPosition] = useState(initialPosition);
  const [size, setSize] = useState(initialSize);
  const [isDragging, setIsDragging] = useState(false);
  const [isResizing, setIsResizing] = useState(false);
  const [isMaximized, setIsMaximized] = useState(false);
  const [dragStart, setDragStart] = useState({ x: 0, y: 0 });
  const [resizeStart, setResizeStart] = useState({ x: 0, y: 0 });
  const [resizeStartSize, setResizeStartSize] = useState(initialSize);
  const windowRef = useRef<HTMLDivElement>(null);
```

**技术选型理由**:
- 使用 `useState` 管理窗口的本地状态（位置、大小、拖拽状态等）
- 使用 `useRef` 引用 DOM 元素，用于计算位置和大小
- 使用 `useEffect` 监听全局鼠标事件，实现拖拽和调整大小

#### 1.1.2 窗口拖拽实现代码

```tsx
// 拖拽开始
const handleDragStart = (e: React.MouseEvent) => {
  if (!draggable || isMaximized) return;
  e.preventDefault();
  setIsDragging(true);
  // 记录鼠标相对于窗口左上角的偏移量
  setDragStart({ x: e.clientX - position.x, y: e.clientY - position.y });
};

// 拖拽移动
const handleDragMove = (e: MouseEvent) => {
  if (!isDragging) return;
  setPosition({
    x: e.clientX - dragStart.x,
    y: e.clientY - dragStart.y,
  });
};

// 拖拽结束
const handleDragEnd = () => {
  setIsDragging(false);
};
```

**业务场景说明**:
- 当用户点击窗口标题栏并拖动时，触发拖拽
- 拖拽过程中，窗口位置实时更新
- 最大化的窗口不允许拖拽

**技术细节**:
- 使用 `e.clientX - dragStart.x` 计算新位置，确保拖拽的平滑性
- `dragStart` 保存了鼠标点击位置相对于窗口左上角的偏移，避免窗口跳动

#### 1.1.3 窗口调整大小实现代码

```tsx
// 调整大小开始
const handleResizeStart = (e: React.MouseEvent, direction: string) => {
  if (!resizable || isMaximized) return;
  e.preventDefault();
  e.stopPropagation();
  setIsResizing(true);
  setResizeStart({ x: e.clientX, y: e.clientY });
  setResizeStartSize(size);
};

// 调整大小移动
const handleResizeMove = (e: MouseEvent) => {
  if (!isResizing) return;
  const deltaX = e.clientX - resizeStart.x;
  const deltaY = e.clientY - resizeStart.y;

  setSize({
    width: Math.max(minSize.width, resizeStartSize.width + deltaX),
    height: Math.max(minSize.height, resizeStartSize.height + deltaY),
  });
};

// 调整大小结束
const handleResizeEnd = () => {
  setIsResizing(false);
};
```

**技术细节**:
- 使用 `minSize` 限制最小宽高度，防止窗口过小
- 支持右下角、右边、底部三个方向的调整大小

#### 1.1.4 最大化/还原实现

```tsx
// 最大化/还原
const handleMaximize = () => {
  if (isMaximized) {
    // 还原到原始位置和大小
    setPosition(initialPosition);
    setSize(initialSize);
  } else {
    // 最大化：移动到左上角，大小为视口大小减去任务栏
    setPosition({ x: 0, y: 0 });
    setSize({
      width: window.innerWidth,
      height: window.innerHeight - 28 - 70  // 减去 MenuBar 和 Dock 高度
    });
  }
  setIsMaximized(!isMaximized);
  onMaximize?.();
};
```

#### 1.1.5 全局鼠标事件监听

```tsx
// 监听全局鼠标事件
useEffect(() => {
  const handleMouseMove = (e: MouseEvent) => {
    handleDragMove(e);
    handleResizeMove(e);
  };

  const handleMouseUp = () => {
    handleDragEnd();
    handleResizeEnd();
  };

  if (isDragging || isResizing) {
    document.addEventListener("mousemove", handleMouseMove);
    document.addEventListener("mouseup", handleMouseUp);
  }

  return () => {
    document.removeEventListener("mousemove", handleMouseMove);
    document.removeEventListener("mouseup", handleMouseUp);
  };
}, [isDragging, isResizing, dragStart, resizeStart]);
```

**技术选型理由**:
- 使用 `document` 上的鼠标事件而非组件级别的事件，确保拖拽流畅
- 在 `useEffect` 中根据拖拽/调整大小的不同状态动态添加/移除事件监听
- 清理函数确保组件卸载时事件监听被正确移除，防止内存泄漏

```tsx
export default function WindowManager({
  initialWindows = [],
  onWindowClose,
  onWindowMinimize,
  onWindowMaximize,
  onWindowsChange,
  onReady,
  className = "",
  windowControls = "windows",
}: WindowManagerProps) {
  const [windows, setWindows] = useState<WindowItem[]>(initialWindows);
  const [activeWindowId, setActiveWindowId] = useState<string | null>(null);

  // 创建新窗口
  const createWindow = useCallback(
    (
      id: string,
      title: string,
      content: ReactNode,
      options?: Partial<WindowItem>,
    ) => {
      setWindows((prev) => {
        const newWindow: WindowItem = {
          id,
          appId: options?.appId,
          title,
          content,
          visible: true,
          minimized: false,
          maximized: false,
          position: options?.position || {
            x: 100 + prev.length * 20,  // 错开显示
            y: 100 + prev.length * 20,
          },
          size: options?.size || { width: 600, height: 400 },
          zIndex: 100 + prev.length,
          windowControls: options?.windowControls || windowControls,
          ...options,
        };

        // 新窗口立即激活
        const maxZIndex = Math.max(...prev.map((w) => w.zIndex), 100);
        newWindow.zIndex = maxZIndex + 1;

        return [...prev, newWindow];
      });

      setActiveWindowId(id);
    },
    [windowControls],
  );
```

#### 1.2.3 窗口的 CRUD 操作

```tsx
// 关闭窗口
const closeWindow = useCallback(
  (id: string) => {
    setWindows((prev) => prev.filter((w) => w.id !== id));
    onWindowClose?.(id);
  },
  [onWindowClose],
);

// 最小化窗口
const minimizeWindow = useCallback(
  (id: string) => {
    setWindows((prev) =>
      prev.map((w) => (w.id === id ? { ...w, minimized: true } : w)),
    );
    onWindowMinimize?.(id);
  },
  [onWindowMinimize],
);

// 最大化窗口
const maximizeWindow = useCallback(
  (id: string) => {
    setWindows((prev) =>
      prev.map((w) => (w.id === id ? { ...w, maximized: !w.maximized } : w)),
    );
    onWindowMaximize?.(id);
  },
  [onWindowMaximize],
);

// 激活窗口（置顶）
const activateWindow = useCallback((id: string) => {
  setActiveWindowId(id);
  setWindows((prev) => {
    const maxZIndex = Math.max(...prev.map((w) => w.zIndex), 100);

    // 如果已经是最顶层，则不更新
    const current = prev.find(w => w.id === id);
    if (current && current.zIndex === maxZIndex) return prev;

    return prev.map((w) =>
      w.id === id ? { ...w, zIndex: maxZIndex + 1 } : w,
    );
  });
}, []);

// 恢复窗口（从最小化）
const restoreWindow = useCallback(
  (id: string) => {
    setWindows((prev) =>
      prev.map((w) =>
        w.id === id ? { ...w, minimized: false, visible: true } : w,
      ),
    );
    activateWindow(id);
  },
  [activateWindow],
);
```

#### 1.2.4 z-index 层级管理算法

```tsx
// 激活窗口时更新 z-index
const activateWindow = useCallback((id: string) => {
  setActiveWindowId(id);
  setWindows((prev) => {
    // 找出当前最大 z-index
    const maxZIndex = Math.max(...prev.map((w) => w.zIndex), 100);

    // 如果已经是顶层窗口，不做处理（避免不必要的渲染）
    const current = prev.find(w => w.id === id);
    if (current && current.zIndex === maxZIndex) return prev;

    // 将目标窗口的 z-index 设为最大，其他窗口保持不变
    return prev.map((w) =>
      w.id === id ? { ...w, zIndex: maxZIndex + 1 } : w,
    );
  });
}, []);
```

**技术选型理由**:
- 使用简单的递增 z-index 算法，简化层级管理
- 优化：如果窗口已经在最顶层，则不触发重新渲染

**业务场景说明**:
- 点击窗口时，该窗口会置顶显示
- 新创建的窗口自动获得最高层级

```tsx
/**
 * Dock栏组件 - 应用快速启动器
 * 类似macOS的Dock栏，支持动画和预览
 */

"use client";

import { useState, useRef } from "react";
import { Tooltip, Badge, Dropdown } from "antd";

interface DockApp {
  id: string;
  name: string;
  icon: React.ReactNode;
  onClick: () => void;
  badge?: number;
  openCount?: number;
  isActive?: boolean;
  onCloseAll?: () => void;
}

export default function Dock({
  apps,
  position = "bottom",
  className = "",
}: DockProps) {
  const [hoveredApp, setHoveredApp] = useState<string | null>(null);

  // 计算缩放效果 - 仿 macOS 效果
  const calculateScale = (index: number, hoveredIndex: number | null) => {
    if (hoveredIndex === null) return 1;
    const distance = Math.abs(index - hoveredIndex);
    if (distance === 0) return 1.5;   // 当前 hover 的图标放大 1.5 倍
    if (distance === 1) return 1.2;   // 相邻图标放大 1.2 倍
    if (distance === 2) return 1.1;    // 相隔一个的图标放大 1.1 倍
    return 1;
  };

  return (
    <div className={`dock glass border-white/10 shadow-2xl ${className}`} style={getDockStyle()}>
      {apps.map((app, index) => {
        const isHovered = hoveredApp === app.id;
        const currentScale = calculateScale(
          index,
          apps.findIndex((a) => a.id === hoveredApp),
        );
        const hasOpen = (app.openCount ?? 0) > 0;

        return (
          <Tooltip key={app.id} title={app.name} placement="top">
            <Dropdown menu={...}>
              <div
                className="relative flex items-center justify-center rounded-2xl transition-all cursor-pointer group"
                style={{
                  width: 48,
                  height: 48,
                  transform: `scale(${currentScale})`,
                  transformOrigin: 'bottom center'  // 从底部中心缩放
                }}
                onMouseEnter={() => setHoveredApp(app.id)}
                onMouseLeave={() => setHoveredApp(null)}
                onClick={app.onClick}
              >
                {/* 应用图标 */}
                <div className="flex items-center justify-center text-white/90 drop-shadow-lg">
                  {app.icon}
                </div>

                {/* 活跃指示器 */}
                {hasOpen && (
                  <div className="absolute -bottom-1 w-1.5 h-1.5 rounded-full bg-primary glow-primary" />
                )}

                {/* 多窗口计数徽章 */}
                {(app.openCount ?? 0) > 1 && (
                  <Badge count={app.openCount} ... />
                )}
              </div>
            </Dropdown>
          </Tooltip>
        );
      })}
    </div>
  );
}
```

```tsx
            path: "/tmp",
            name: "tmp",

            writable: true
        });

        console.log("[VFS] 初始化 ZenFS...");

        try {
            await configure({
                mounts: {
                    "/": { backend: IndexedDB, name: "webenv-root" },

            await this.ensureFramework();

            console.error("[VFS] 配置失败:", e);
            throw e;
        }
    }
```

#### 2.1.3 IndexedDB 存储实现

**文件位置**: `D:\Develeping\webEnv\webenv-os\src\lib\storage\db.ts`

```tsx
/**
 * IndexedDB 数据库封装
 * 提供基础的数据库操作
 */

export const DB_NAME = "webenv-os-db";
export const DB_VERSION = 1;
export const STORES = {
  FILES: "files",     // 存储文件元数据
  CONTENT: "content", // 存储文件内容 (分离存储以提高列表性能)
};

export class IndexedDBAdapter {
  private db: IDBDatabase | null = null;

  // 初始化数据库
  async init(): Promise<void> {
    if (this.db) return;

    return new Promise((resolve, reject) => {
      this.openRequest = indexedDB.open(DB_NAME, DB_VERSION);

      this.openRequest.onupgradeneeded = (event) => {
        const db = (event.target as IDBOpenDBRequest).result;

        // 文件元数据存储
        if (!db.objectStoreNames.contains(STORES.FILES)) {
          const fileStore = db.createObjectStore(STORES.FILES, { keyPath: "id" });
          fileStore.createIndex("parentId", "parentId", { unique: false });
          fileStore.createIndex("path", "path", { unique: true }); // 路径索引
        }

        // 内容存储
        if (!db.objectStoreNames.contains(STORES.CONTENT)) {
          db.createObjectStore(STORES.CONTENT, { keyPath: "id" });
        }
      };

      this.openRequest.onsuccess = (event) => {
        this.db = (event.target as IDBOpenDBRequest).result;
        resolve();
      };

      this.openRequest.onerror = (event) => {
        console.error("IndexedDB init error:", event);
        reject((event.target as IDBOpenDBRequest).error);
      };
    });
  }

  // 通过索引获取
  async getByIndex<T>(storeName: string, indexName: string, value: string): Promise<T[]> {
    await this.init();
    return new Promise((resolve, reject) => {
      const tx = this.getTransaction([storeName], "readonly");
      const store = tx.objectStore(storeName);
      const index = store.index(indexName);
      const request = index.getAll(value);

      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }

  // 事务支持
  async transaction(
      storeNames: string[],
      mode: IDBTransactionMode,
      callback: (stores: Record<string, IDBObjectStore>) => Promise<void>
  ): Promise<void> {
      await this.init();
      return new Promise((resolve, reject) => {
          const tx = this.getTransaction(storeNames, mode);
          const stores: Record<string, IDBObjectStore> = {};
          storeNames.forEach(name => {
              stores[name] = tx.objectStore(name);
          });

          callback(stores)
              .then(() => {})
              .catch(err => {
                  tx.abort();
                  reject(err);
              });

          tx.oncomplete = () => resolve();
          tx.onerror = () => reject(tx.error);
      });
  }
}

export const db = new IndexedDBAdapter();
```

**技术选型理由**:
- 使用 IndexedDB 存储文件元数据和内容，充分利用浏览器的本地存储能力
- 分离元数据和内容存储：元数据存储用于快速列表，内容存储用于实际文件内容
- 创建索引（parentId、path）加速查询

```tsx
/**
 * 统一文件系统适配器
 * 远程工作区逻辑：所有文件先同步到本地 IndexedDB，后续操作在本地进行并自动同步到远程
 * 本地工作区逻辑：直接操作本地文件系统
 * 特性：基于哈希的智能变更检测
 */
export class UnifiedFSAdapter implements IFileSystemAdapter {
    private workspaceId: string;
    private localAdapter: LocalFSAdapter;
    private changeTracker: ReturnType<typeof getChangeTracker>;
    private initialized: boolean = false;
    private syncInterval: NodeJS.Timeout | null = null;

    constructor(workspaceId: string, isRemote: boolean = true) {
        this.workspaceId = workspaceId || "default";
        this.localAdapter = new LocalFSAdapter();
        this.changeTracker = getChangeTracker(this.workspaceId);
    }

    /**
     * 初始化：同步远程文件到本地
     */
    async init(): Promise<void> {
        if (this.initialized) return;

        console.log(`[UnifiedFSAdapter] 初始化工作区: ${this.workspaceId}`);

        // 加载变更追踪器数据
        await this.changeTracker.loadFromStorage();

        try {
            // 检查网络状态
            if (!navigator.onLine) {
                console.warn("[UnifiedFSAdapter] 当前离线，使用本地缓存");
                this.initialized = true;
                return;
            }

            // 从远程获取完整文件树
            const tree = await filesApi.getFileTree(this.workspaceId);
            if (tree && tree.length > 0) {
                // 递归同步文件到本地（使用智能变更检测）
                await this.syncTreeToLocal(tree);
            }

            this.initialized = true;

            // 启动定时同步（检测远程变更）
            this.startSyncInterval();
        } catch (error) {
            console.error("[UnifiedFSAdapter] 初始化失败，使用本地缓存:", error);
            this.initialized = true;
        }
    }

    /**
     * 递归同步文件树到本地（带变更检测）
     */
    private async syncTreeToLocal(nodes: FileItem[], forceSync: boolean = false): Promise<number> {
        let syncedCount = 0;
        for (const node of nodes) {
            try {
                if (node.type === "folder") {
                    await this.localAdapter.createDirectory(node.path);
                    if (node.children && node.children.length > 0) {
                        syncedCount += await this.syncTreeToLocal(node.children, forceSync);
                    }
                } else {
                    // 文件：检查是否需要同步
                    const localExists = await this.checkLocalExists(node.path);
                    const metadata = this.changeTracker.getMetadata(node.path);

                    const needsSync = forceSync || !localExists ||
                        (nodeAny.updatedAt && metadata && nodeAny.updatedAt > metadata.timestamp);

                    if (needsSync) {
                        const content = await filesApi.getFileContent(node.path, this.workspaceId);
                        await this.localAdapter.createFile(node.path, content.content || "");
                        // 记录元数据
                        await this.changeTracker.recordFile(node.path, content.content || "");
                        syncedCount++;
                    }
                }
            } catch (e) {
                console.warn(`[UnifiedFSAdapter] 同步节点失败: ${node.path}`, e);
            }
        }
        return syncedCount;
    }
}
```

**业务场景说明**:
- 远程工作区：文件先同步到本地 IndexedDB，后续在本地操作，最后自动同步到远程
- 本地工作区：直接操作本地文件系统
- 智能变更检测：基于文件内容的哈希值判断是否需要同步

```tsx
import { mutate } from "swr";
import { useEffect, useState, useCallback, useRef } from "react";
import { WorkspaceStorageManager } from "@/lib/storage/VirtualFileSystem";

export function useFileSystem(rootPath: string = "default") {
  const [files, setFiles] = useState<FileNode[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [isError, setIsError] = useState<any>(null);
  const adapterRef = useRef<ReturnType<typeof WorkspaceStorageManager.getAdapter> | null>(null);

  // 获取适配器
  useEffect(() => {
    adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
  }, [rootPath]);

  // 加载文件树
  const loadFileTree = useCallback(async () => {
    setIsLoading(true);
    setIsError(null);
    try {
      if (!adapterRef.current) {
        adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
      }
      const fileTree = await adapterRef.current.readdir("/");
      setFiles(fileTree);
    } catch (err) {
      console.error("加载文件树失败:", err);
      setIsError(err);
      setFiles([]);
    } finally {
      setIsLoading(false);
    }
  }, [rootPath]);

  // 刷新文件树
  const refresh = useCallback(async () => {
    await loadFileTree();
    // 清除 SWR 缓存
    mutate(`file://${rootPath}`);
  }, [loadFileTree, rootPath]);

  // 读取文件
  const readFile = useCallback(async (path: string): Promise<string> => {
    if (!adapterRef.current) {
      adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
    }
    return await adapterRef.current.readFile(path);
  }, [rootPath]);

  // 写入文件
  const writeFile = useCallback(async (path: string, content: string): Promise<void> => {
    if (!adapterRef.current) {
      adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
    }
    await adapterRef.current.writeFile(path, content);
    await refresh();
  }, [rootPath, refresh]);

  // 创建文件/目录
  const createFile = useCallback(async (path: string, content: string = ""): Promise<void> => {
    if (!adapterRef.current) {
      adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
    }
    await adapterRef.current.createFile(path, content);
    await refresh();
  }, [rootPath, refresh]);

  const createDirectory = useCallback(async (path: string): Promise<void> => {
    if (!adapterRef.current) {
      adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
    }
    await adapterRef.current.createDirectory(path);
    await refresh();
  }, [rootPath, refresh]);

  // 删除节点
  const deleteNode = useCallback(async (path: string): Promise<void> => {
    if (!adapterRef.current) {
      adapterRef.current = WorkspaceStorageManager.getAdapter(rootPath);
    }
    await adapterRef.current.deleteNode(path);
    await refresh();
  }, [rootPath, refresh]);

  return {
    files, isLoading, isError, refresh,
    readFile, writeFile, createFile, createDirectory, deleteNode, mountLocal
  };
```

function detectCollision(
  movingWindow: WindowState,
  otherWindows: WindowState[]
): { direction: 'left' | 'right' | 'top' | 'bottom' | null; distance: number } {
  const movingRect = {
    left: movingWindow.position.x,
    right: movingWindow.position.x + movingWindow.size.width,
    top: movingWindow.position.y,
    bottom: movingWindow.position.y + movingWindow.size.height,
  };

  let minDistance = Infinity;
  let collisionDirection: 'left' | 'right' | 'top' | 'bottom' | null = null;

  for (const other of otherWindows) {
    const otherRect = {
      left: other.position.x,
      right: other.position.x + other.size.width,
      top: other.position.y,
      bottom: other.position.y + other.size.height,
    };

    // 检测是否重叠
    const isOverlapping = !(
      movingRect.right <= otherRect.left ||
      movingRect.left >= otherRect.right ||
      movingRect.bottom <= otherRect.top ||
      movingRect.top >= otherRect.bottom
    );

    if (isOverlapping) {
      // 计算重叠区域的面积
      const overlapX = Math.min(movingRect.right, otherRect.right) - Math.max(movingRect.left, otherRect.left);
      const overlapY = Math.min(movingRect.bottom, otherRect.bottom) - Math.max(movingRect.top, otherRect.top);
      const overlapArea = overlapX * overlapY;

      if (overlapArea < minDistance) {
        minDistance = overlapArea;
        // 确定推开方向：向重叠区域最小的方向推开
        if (overlapX < overlapY) {
          collisionDirection = movingRect.left < otherRect.left ? 'left' : 'right';
        } else {
          collisionDirection = movingRect.top < otherRect.top ? 'top' : 'bottom';
        }
      }
    }
  }

  return { direction: collisionDirection, distance: minDistance };
}

#### A.2 窗口对齐辅助线算法

当窗口拖拽到接近屏幕边缘或其他窗口边缘时，显示对齐辅助线：

```typescript
/**
 * 窗口对齐辅助线检测
 * 当窗口接近边缘时，返回对齐位置
 */
function calculateSnapPoints(
  window: WindowState,
  screenBounds: { width: number; height: number; taskbarHeight: number },
  otherWindows: WindowState[],
  snapThreshold: number = 10
): { x?: number; y?: number } {
  const snapPoints: { x?: number; y?: number } = {};
  const { position, size } = window;

  // 屏幕边缘对齐
  // 左边缘对齐
  if (Math.abs(position.x) < snapThreshold) {
    snapPoints.x = 0;
  }
  // 右边缘对齐
  if (Math.abs(position.x + size.width - screenBounds.width) < snapThreshold) {
    snapPoints.x = screenBounds.width - size.width;
  }
  // 顶部对齐（减去任务栏）
  if (Math.abs(position.y - screenBounds.taskbarHeight) < snapThreshold) {
    snapPoints.y = screenBounds.taskbarHeight;
  }
  // 底部对齐
  if (Math.abs(position.y + size.height - screenBounds.height) < snapThreshold) {
    snapPoints.y = screenBounds.height - size.height;
  }

  // 窗口之间对齐
  for (const other of otherWindows) {
    const otherRight = other.position.x + other.size.width;
    const otherBottom = other.position.y + other.size.height;

    // 与其他窗口左边缘对齐
    if (Math.abs(position.x - other.position.x) < snapThreshold) {
      snapPoints.x = other.position.x;
    }
    // 与其他窗口右边缘对齐
    if (Math.abs(position.x - otherRight) < snapThreshold) {
      snapPoints.x = otherRight;
    }
    // 与其他窗口顶部对齐
    if (Math.abs(position.y - other.position.y) < snapThreshold) {
      snapPoints.y = other.position.y;
    }
    // 与其他窗口底部对齐
    if (Math.abs(position.y - otherBottom) < snapThreshold) {
      snapPoints.y = otherBottom;
    }
  }

  return snapPoints;
}
```

```typescript
/**
 * IndexedDB 存储优化实现
 * 使用事务和索引提升性能
 */
class OptimizedIndexedDB {
  private dbName: string;
  private version: number;
  private db: IDBDatabase | null = null;

  async init(): Promise<void> {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(this.dbName, this.version);

      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };

      request.onupgradeneeded = (event) => {
        const db = (event.target as IDBOpenDBRequest).result;

        // 创建对象存储
        if (!db.objectStoreNames.contains('files')) {
          const fileStore = db.createObjectStore('files', { keyPath: 'id' });

          // 创建索引加速查询
          fileStore.createIndex('path', 'path', { unique: false });
          fileStore.createIndex('parentId', 'parentId', { unique: false });
          fileStore.createIndex('type', 'type', { unique: false });
          fileStore.createIndex('modifiedAt', 'modifiedAt', { unique: false });
        }

        // 元数据存储
        if (!db.objectStoreNames.contains('metadata')) {
          const metaStore = db.createObjectStore('metadata', { keyPath: 'key' });
          metaStore.createIndex('key', 'key', { unique: true });
        }
      };
    });
  }

  // 批量写入优化：使用事务批量操作
  async batchWrite(files: FileRecord[]): Promise<void> {
    if (!this.db) throw new Error('Database not initialized');

    return new Promise((resolve, reject) => {
      const transaction = this.db!.transaction(['files'], 'readwrite');
      const store = transaction.objectStore('files');

      for (const file of files) {
        store.put(file);
      }

      transaction.oncomplete = () => resolve();
      transaction.onerror = () => reject(transaction.error);
    });
  }

  // 范围查询：使用索引查询特定目录下的所有文件
  async queryByDirectory(parentId: string): Promise<FileRecord[]> {
    if (!this.db) throw new Error('Database not initialized');

    return new Promise((resolve, reject) => {
      const transaction = this.db!.transaction(['files'], 'readonly');
      const store = transaction.objectStore('files');
      const index = store.index('parentId');
      const request = index.getAll(parentId);

      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }

  // 游标遍历：用于大数据量的增量处理
  async *iterateFiles(batchSize: number = 100): AsyncGenerator<FileRecord[]> {
    if (!this.db) throw new Error('Database not initialized');

    let cursor: IDBCursorWithValue | null = null;
    const transaction = this.db.transaction(['files'], 'readonly');
    const store = transaction.objectStore('files');

    let batch: FileRecord[] = [];
    let request = store.openCursor();

    while (true) {
      await new Promise((resolve, reject) => {
        request.onsuccess = () => {
          cursor = request.result;
          resolve();
        };
        request.onerror = () => reject(request.error);
      });

      if (!cursor) break;

      batch.push(cursor.value);

      if (batch.length >= batchSize) {
        yield batch;
        batch = [];
      }

      cursor.continue();
    }

    if (batch.length > 0) {
      yield batch;
    }
  }
}
```

#### B.2 文件内容分块存储

对于大文件，需要分块存储以避免 IndexedDB 的大小限制：

```typescript
/**
 * 大文件分块存储
 * IndexedDB 单个对象大小有限制，需要分块存储
 */
class ChunkedFileStorage {
  private db: OptimizedIndexedDB;
  private chunkSize: number = 1024 * 1024; // 1MB per chunk

  async saveLargeFile(path: string, content: ArrayBuffer): Promise<string> {
    const fileId = crypto.randomUUID();
    const chunks: ArrayBuffer[] = [];
    const totalChunks = Math.ceil(content.byteLength / this.chunkSize);

    // 分块
    for (let i = 0; i < totalChunks; i++) {
      const start = i * this.chunkSize;
      const end = Math.min(start + this.chunkSize, content.byteLength);
      const chunk = content.slice(start, end);
      chunks.push(chunk);
    }

    // 保存元数据
    await this.db.write({
      id: fileId,
      path,
      type: 'file',
      totalChunks,
      totalSize: content.byteLength,
      chunkSize: this.chunkSize,
      createdAt: Date.now(),
    });

    // 保存每个块
    for (let i = 0; i < chunks.length; i++) {
      await this.db.write({
        id: `${fileId}_chunk_${i}`,
        parentId: fileId,
        chunkIndex: i,
        data: chunks[i],
      });
    }

    return fileId;
  }

  async readLargeFile(fileId: string): Promise<ArrayBuffer> {
    const metadata = await this.db.read(fileId);
    const chunks: ArrayBuffer[] = [];

    for (let i = 0; i < metadata.totalChunks; i++) {
      const chunk = await this.db.read(`${fileId}_chunk_${i}`);
      chunks.push(chunk.data);
    }

    // 合并所有块
    const totalLength = chunks.reduce((sum, chunk) => sum + chunk.byteLength, 0);
    const result = new Uint8Array(totalLength);

    let offset = 0;
    for (const chunk of chunks) {
      result.set(new Uint8Array(chunk), offset);
      offset += chunk.byteLength;
    }

    return result.buffer;
  }
}
```

```typescript
/**
 * ANSI 转义序列解析器
 * 用于处理终端颜色、格式等
 */
class ANSIParser {
  private currentState: 'text' | 'escape' | 'csi' | 'osc' = 'text';
  private buffer: string = '';
  private params: number[] = [];

  parse(input: string): string {
    let output = '';

    for (const char of input) {
      switch (this.currentState) {
        case 'text':
          if (char === '\x1b') {  // ESC
            output += this.buffer;
            this.buffer = '';
            this.currentState = 'escape';
          } else {
            this.buffer += char;
          }
          break;

        case 'escape':
          if (char === '[') {
            this.params = [];
            this.currentState = 'csi';
          } else if (char === ']') {
            this.currentState = 'osc';
          } else {
            this.currentState = 'text';
          }
          break;

        case 'csi':
          if (char >= '0' && char <= '9') {
            // 收集参数
            const lastParam = this.params.length - 1;
            if (lastParam >= 0) {
              this.params[lastParam] = this.params[lastParam] * 10 + parseInt(char);
            } else {
              this.params.push(parseInt(char));
            }
          } else if (char === ';') {
            this.params.push(0);
          } else if (char >= '@' && char <= '~') {
            // CSI 序列结束，处理命令
            this.handleCSI(char, this.params);
            this.currentState = 'text';
          }
          break;

        case 'osc':
          // OSC 序列 (操作系统命令)
          if (char === '\x07' || char === '\\') {
            // OSC 结束
            this.currentState = 'text';
          }
          break;
      }
    }

    output += this.buffer;
    return output;
  }

  private handleCSI(command: string, params: number[]) {
    switch (command) {
      case 'm':  // SGR - 选择图形再现（颜色/样式）
        this.applyStyle(params);
        break;
      case 'H':  // CUP - 光标位置
        this.setCursorPosition(params[0], params[1]);
        break;
      case 'J':  // ED - 清除屏幕
        this.clearScreen(params[0]);
        break;
      case 'K':  // EL - 清除行
        this.clearLine(params[0]);
        break;
      // ... 其他 CSI 命令
    }
  }
}
```

### Q1: 窗口拖拽性能如何优化？

**详细回答**:
1. **使用 transform 代替 left/top**: CSS transform 使用 GPU 加速，避免触发重排（reflow）
2. **requestAnimationFrame**: 将位置更新放在下一帧执行，合并多次移动为一次渲染
3. **事件委托**: 将 mousemove/mouseup 事件绑定到 document 而非窗口元素，避免快速移动时鼠标离开窗口元素
4. **will-change**: 对窗口元素使用 `will-change: transform` 提示浏览器进行优化

```typescript
// 优化后的拖拽实现
const handleDragMove = useCallback((e: MouseEvent) => {
  if (!isDragging) return;

  requestAnimationFrame(() => {
    setPosition(prev => ({
      x: Math.max(0, Math.min(e.clientX - dragStart.x, bounds.width - size.width)),
      y: Math.max(taskbarHeight, Math.min(e.clientY - dragStart.y, bounds.height - size.height))
    }));
  });
}, [isDragging, dragStart, size, bounds, taskbarHeight]);
```

### Q2: 如何实现离线文件系统？

**详细回答**:
1. **Service Worker**: 拦截请求，优先返回缓存
2. **IndexedDB**: 本地存储文件内容和元数据
3. **Sync Queue**: 离线时的操作记录到队列，在线时同步
4. **冲突解决**: 使用时间戳或 CRDT 解决并发修改冲突

```typescript
// 离线同步队列
class OfflineSyncQueue {
  private queue: SyncOperation[] = [];

  async addOperation(operation: SyncOperation) {
    this.queue.push(operation);
    await this.persist();
  }

  async sync() {
    if (!navigator.onLine) return;

    for (const op of this.queue) {
      try {
        await this.executeOperation(op);
        this.queue.shift();  // 成功后移除
      } catch (error) {
        // 重试或标记失败
        break;
      }
    }
    await this.persist();
  }
}
```
