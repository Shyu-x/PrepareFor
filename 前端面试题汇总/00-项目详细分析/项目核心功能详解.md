### 1.1 窗口管理系统底层原理

#### 1.1.1 拖拽性能优化的底层逻辑

**问题本质**：
浏览器渲染流程中，修改 `left/top` 属性会触发 **Layout（布局）** → **Paint（绘制）** → **Composite（合成）** 三个阶段。而使用 `transform` 只触发最后一个阶段。

**浏览器渲染流水线**：
```
DOM Change
    ↓
Style Calculation (样式计算)
    ↓
Layout (布局) ← left/top 修改会触发这里 ❌
    ↓
Paint (绘制) ← background-color 等修改会触发这里 ❌
    ↓
Composite (合成) ← transform 修改只触发这里 ✅
    ↓
Display
```

**代码实现**：
```typescript
// ❌ 错误实现：触发重排
const handleDragMove = (e: MouseEvent) => {
  setPosition({
    x: e.clientX - dragStart.x,
    y: e.clientY - dragStart.y
  });
};

// 样式会是这样的：
// position: absolute;
// left: ${x}px;  ← 触发 Layout
// top: ${y}px;   ← 触发 Layout

// ✅ 正确实现：使用 transform + requestAnimationFrame
const handleDragMove = useCallback((e: MouseEvent) => {
  if (!isDragging) return;

  // 使用 requestAnimationFrame 合并多次移动为单次渲染
  requestAnimationFrame(() => {
    const newX = Math.max(0, Math.min(
      e.clientX - dragStart.x,
      window.innerWidth - windowSize.width
    ));
    const newY = Math.max(taskbarHeight, Math.min(
      e.clientY - dragStart.y,
      window.innerHeight - windowSize.height
    ));

    setPosition({ x: newX, y: newY });
  });
}, [isDragging, dragStart, windowSize, taskbarHeight]);

// 渲染样式：
// position: absolute;
// transform: translate(${x}px, ${y}px);  ← 只触发 Composite
// will-change: transform;  ← 提示浏览器创建合成层
```

**GPU 合成层原理**：
- 某些 CSS 属性会创建独立的合成层（Compositing Layer）
- 合成层在 GPU 上运行，不占用主线程
- 使用 `will-change: transform` 主动创建合成层

```css
/* 合成层创建条件 */
.window-dragging {
  will-change: transform; /* 主动创建 */
  transform: translateZ(0); /* 硬件加速 */
}

/* 注意：过度使用会导致内存问题 */
```

#### 1.1.2 窗口层级管理的数学原理

**Z-Index 栈算法**：
```typescript
// 窗口层级管理器
class WindowManager {
  private baseZIndex = 1000;
  private windowZIndices = new Map<string, number>();

  // 激活窗口：将其置顶
  activateWindow(windowId: string) {
    // 获取当前最大 Z-Index
    const maxZ = Math.max(
      this.baseZIndex,
      ...Array.from(this.windowZIndices.values())
    );

    // 新窗口 = 最大值 + 1
    this.windowZIndices.set(windowId, maxZ + 1);

    // 更新所有窗口的 Z-Index 顺序
    this.normalizeZIndices();
  }

  // 规范化：确保 Z-Index 是连续的 1, 2, 3...
  private normalizeZIndices() {
    const sorted = Array.from(this.windowZIndices.entries())
      .sort((a, b) => a[1] - b[1]);

    sorted.forEach(([id], index) => {
      this.windowZIndices.set(id, this.baseZIndex + index);
    });
  }
}
```

**窗口坐标计算**：

```typescript
// 窗口位置限制算法
const constrainPosition = (
  position: { x: number; y: number },
  size: { width: number; height: number },
  screen: { width: number; height: number; taskbarHeight: number }
) => {
  return {
    x: Math.max(0, Math.min(position.x, screen.width - size.width)),
    y: Math.max(screen.taskbarHeight, Math.min(position.y, screen.height - size.height))
  };
};

// 窗口对齐辅助线检测
const snapToEdges = (
  window: { x: number; y: number; width: number; height: number },
  threshold: number = 10
): { x?: number; y?: number } => {
  const snaps: { x?: number; y?: number } = {};

  // 左边缘对齐
  if (Math.abs(window.x) < threshold) snaps.x = 0;
  // 右边缘对齐
  if (Math.abs(window.x + window.width - windowWidth) < threshold)
    snaps.x = windowWidth - window.width;
  // 上边缘对齐
  if (Math.abs(window.y - taskbarHeight) < threshold)
    snaps.y = taskbarHeight;

  return snaps;
};
```

### 1.2 虚拟文件系统底层原理

#### 1.2.1 IndexedDB 存储机制

**IndexedDB vs LocalStorage**：
| 特性 | LocalStorage | IndexedDB |
|------|--------------|-----------|
| 容量 | 5-10MB | 50MB-无限制 |
| 数据类型 | 仅字符串 | 任何结构化数据 |
| API | 同步 | 异步 |
| 索引 | 不支持 | 支持 |
| 事务 | 不支持 | 支持 |

**IndexedDB 架构**：
```
┌─────────────────────────────────────────┐
│              IndexedDB                   │
├─────────────────────────────────────────┤
│  Database (数据库)                       │
│    └── ObjectStore (对象仓库，类似表)     │
│         ├── Index 1 (索引)              │
│         ├── Index 2 (索引)              │
│         └── Record (记录)                │
└─────────────────────────────────────────┘
```

**核心代码实现**：
```typescript
// 完整的 IndexedDB 封装
class FileStorage {
  private dbName = 'webenv-files';
  private dbVersion = 1;
  private db: IDBDatabase | null = null;

  async init(): Promise<void> {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(this.dbName, this.dbVersion);

      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };

      // 数据库升级回调（首次打开或版本升级）
      request.onupgradeneeded = (event) => {
        const db = (event.target as IDBOpenDBRequest).result;

        // 创建文件存储
        if (!db.objectStoreNames.contains('files')) {
          const fileStore = db.createObjectStore('files', { keyPath: 'id' });

          // 创建索引加速查询
          fileStore.createIndex('path', 'path', { unique: false });
          fileStore.createIndex('parentId', 'parentId', { unique: false });
          fileStore.createIndex('type', 'type', { unique: false });
          fileStore.createIndex('modifiedAt', 'modifiedAt', { unique: false });
        }

        // 创建元数据存储
        if (!db.objectStoreNames.contains('metadata')) {
          db.createObjectStore('metadata', { keyPath: 'key' });
        }
      };
    });
  }

  // 读取文件
  async readFile(path: string): Promise<FileRecord | null> {
    return new Promise((resolve, reject) => {
      if (!this.db) throw new Error('Database not initialized');

      const transaction = this.db.transaction(['files'], 'readonly');
      const store = transaction.objectStore('files');
      const index = store.index('path');
      const request = index.get(path);

      request.onsuccess = () => resolve(request.result || null);
      request.onerror = () => reject(request.error);
    });
  }

  // 写入文件
  async writeFile(record: FileRecord): Promise<void> {
    return new Promise((resolve, reject) => {
      if (!this.db) throw new Error('Database not initialized');

      const transaction = this.db.transaction(['files'], 'readwrite');
      const store = transaction.objectStore('files');
      const request = store.put(record);

      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  }

  // 事务批量写入（性能更好）
  async batchWrite(records: FileRecord[]): Promise<void> {
    return new Promise((resolve, reject) => {
      if (!this.db) throw new Error('Database not initialized');

      const transaction = this.db.transaction(['files'], 'readwrite');
      const store = transaction.objectStore('files');

      records.forEach(record => store.put(record));

      transaction.oncomplete = () => resolve();
      transaction.onerror = () => reject(transaction.error);
    });
  }

  // 范围查询：获取目录下所有文件
  async readDirectory(parentId: string): Promise<FileRecord[]> {
    return new Promise((resolve, reject) => {
      if (!this.db) throw new Error('Database not initialized');

      const transaction = this.db.transaction(['files'], 'readonly');
      const store = transaction.objectStore('files');
      const index = store.index('parentId');
      const request = index.getAll(parentId);

      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }
}
```

#### 1.2.2 大文件分块存储原理

**问题**：IndexedDB 对单个对象大小有限制（通常 50MB-1GB），大文件需要分块存储。

**分块存储算法**：
```typescript
const CHUNK_SIZE = 1024 * 1024; // 1MB per chunk

class ChunkedFileStorage {
  private storage: FileStorage;

  // 分块写入
  async writeLargeFile(file: File): Promise<string> {
    const fileId = crypto.randomUUID();
    const totalChunks = Math.ceil(file.size / CHUNK_SIZE);

    // 读取文件内容
    const arrayBuffer = await file.arrayBuffer();

    // 分块写入
    for (let i = 0; i < totalChunks; i++) {
      const start = i * CHUNK_SIZE;
      const end = Math.min(start + CHUNK_SIZE, arrayBuffer.byteLength);
      const chunk = arrayBuffer.slice(start, end);

      await this.storage.writeFile({
        id: `${fileId}_chunk_${i}`,
        parentId: fileId,
        chunkIndex: i,
        data: chunk,
        type: 'chunk'
      });
    }

    // 写入元数据
    await this.storage.writeFile({
      id: fileId,
      path: file.name,
      type: 'file',
      totalChunks,
      totalSize: file.size,
      chunkSize: CHUNK_SIZE,
      createdAt: Date.now()
    });

    return fileId;
  }

  // 分块读取
  async readLargeFile(fileId: string): Promise<Blob> {
    const metadata = await this.storage.readFile(fileId);
    if (!metadata || metadata.type !== 'file') {
      throw new Error('File not found');
    }

    const chunks: Uint8Array[] = [];

    for (let i = 0; i < metadata.totalChunks; i++) {
      const chunk = await this.storage.readFile(`${fileId}_chunk_${i}`);
      if (chunk?.data) {
        chunks.push(new Uint8Array(chunk.data));
      }
    }

    // 合并所有块
    const totalLength = chunks.reduce((sum, chunk) => sum + chunk.length, 0);
    const result = new Uint8Array(totalLength);

    let offset = 0;
    for (const chunk of chunks) {
      result.set(chunk, offset);
      offset += chunk.length;
    }

    return new Blob([result]);
  }
}
```

#### 1.2.3 离线同步机制

**离线操作队列**：
```typescript
// 离线操作记录
interface SyncOperation {
  id: string;
  type: 'create' | 'update' | 'delete';
  path: string;
  content?: ArrayBuffer;
  timestamp: number;
  retryCount: number;
}

class OfflineSyncManager {
  private operationQueue: SyncOperation[] = [];
  private isOnline = navigator.onLine;
  private syncInProgress = false;

  constructor() {
    // 监听网络状态变化
    window.addEventListener('online', () => {
      this.isOnline = true;
      this.sync(); // 联网后自动同步
    });

    window.addEventListener('offline', () => {
      this.isOnline = false;
    });
  }

  // 添加操作到队列
  async queueOperation(operation: Omit<SyncOperation, 'id' | 'timestamp' | 'retryCount'>) {
    const op: SyncOperation = {
      ...operation,
      id: crypto.randomUUID(),
      timestamp: Date.now(),
      retryCount: 0
    };

    this.operationQueue.push(op);

    // 持久化到 IndexedDB
    await this.persistQueue();

    // 如果在线，立即尝试同步
    if (this.isOnline) {
      this.sync();
    }
  }

  // 同步所有操作
  async sync() {
    if (this.syncInProgress || !this.isOnline) return;

    this.syncInProgress = true;

    try {
      const pendingOps = this.operationQueue.filter(op => op.retryCount < 3);

      for (const op of pendingOps) {
        try {
          await this.executeRemoteOperation(op);

          // 成功后从队列移除
          this.operationQueue = this.operationQueue.filter(o => o.id !== op.id);
        } catch (error) {
          op.retryCount++;
          console.error(`Sync failed for ${op.path}, retry ${op.retryCount}`);
        }
      }
    } finally {
      this.syncInProgress = false;
      await this.persistQueue();
    }
  }

  // 冲突检测与解决
  private async detectConflict(op: SyncOperation, remoteMeta: FileRecord): Promise<boolean> {
    const localMeta = await this.storage.readFile(op.path);
    if (!localMeta) return false;

    // 如果远程版本比本地新，且本地不是最新修改，则冲突
    return remoteMeta.modifiedAt > localMeta.modifiedAt &&
           op.timestamp < remoteMeta.modifiedAt;
  }
}
```

```typescript
// NestJS WebSocket Gateway
@WebSocketGateway({ cors: { origin: '*' } })
export class TerminalGateway {
  private sessions = new Map<string, TerminalSession>();

  @SubscribeMessage('terminal:create')
  handleCreateTerminal(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: { cols?: number; rows?: number }
  ) {
    const sessionId = `term_${Date.now()}`;

    // 创建 PTY 进程
    const ptyProcess = pty.spawn('bash', [], {
      name: 'xterm-256color',
      cols: data.cols || 80,
      rows: data.rows || 24,
      cwd: process.env.HOME || '/home/user',
      env: process.env as { [key: string]: string }
    });

    const session: TerminalSession = {
      id: sessionId,
      ptyProcess,
      socketId: client.id
    };

    this.sessions.set(sessionId, session);

    // PTY 数据输出 -> 前端
    ptyProcess.onData((data: string) => {
      if (client.connected) {
        client.emit('terminal:data', { sessionId, data });
      }
    });

    // 进程退出处理
    ptyProcess.onExit(({ exitCode, signal }) => {
      this.sessions.delete(sessionId);
      if (client.connected) {
        client.emit('terminal:exit', { sessionId, exitCode, signal });
      }
    });

    return { sessionId };
  }

  @SubscribeMessage('terminal:input')
  handleInput(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: { sessionId: string; input: string }
  ) {
    const session = this.sessions.get(data.sessionId);
    if (session) {
      session.ptyProcess.write(data.input);
    }
  }

  @SubscribeMessage('terminal:resize')
  handleResize(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: { sessionId: string; cols: number; rows: number }
  ) {
    const session = this.sessions.get(data.sessionId);
    if (session) {
      session.ptyProcess.resize(data.cols, data.rows);
    }
  }
}
```

#### 1.3.2 ANSI 转义序列解析

**终端如何显示颜色和样式**：
```typescript
// ANSI 转义序列解析器
class ANSIParser {
  private state: 'text' | 'escape' | 'csi' = 'text';
  private buffer = '';
  private params: number[] = [];
  private styles: string[] = [];

  // ANSI 颜色代码映射
  private colorMap: Record<number, string> = {
    30: '#0d0d0d', // black
    31: '#cd3131', // red
    32: '#0dbc79', // green
    33: '#e5e510', // yellow
    34: '#2472c8', // blue
    35: '#bc3fbc', // magenta
    36: '#11a8cd', // cyan
    37: '#e5e5e5', // white
  };

  // 解析 ANSI 序列
  parse(input: string): string {
    let output = '';

    for (const char of input) {
      switch (this.state) {
        case 'text':
          if (char === '\x1b') { // ESC
            output += this.buffer;
            this.buffer = '';
            this.state = 'escape';
          } else {
            this.buffer += char;
          }
          break;

        case 'escape':
          if (char === '[') {
            this.params = [];
            this.state = 'csi';
          } else {
            this.state = 'text';
          }
          break;

        case 'csi':
          if (char >= '0' && char <= '9') {
            const lastParam = this.params.length - 1;
            if (lastParam >= 0) {
              this.params[lastParam] = this.params[lastParam] * 10 + parseInt(char);
            } else {
              this.params.push(parseInt(char));
            }
          } else if (char === ';') {
            this.params.push(0);
          } else if (char >= '@' && char <= '~') {
            // CSI 序列结束，处理命令
            this.handleCSI(char);
            this.state = 'text';
          }
          break;
      }
    }

    output += this.buffer;
    return output;
  }

  // 处理 CSI 命令
  private handleCSI(command: string, params: number[] = this.params) {
    switch (command) {
      case 'm': // SGR - 图形再现（颜色/样式）
        this.handleSGR(params);
        break;
      case 'H': // CUP - 光标位置
        this.handleCursorPosition(params);
        break;
      case 'J': // ED - 清除屏幕
        this.handleClearScreen(params[0]);
        break;
      case 'K': // EL - 清除行
        this.handleClearLine(params[0]);
        break;
    }
  }

  private handleSGR(params: number[]) {
    if (params.length === 0 || params[0] === 0) {
      // 重置
      this.styles = [];
    } else if (params[0] === 1) {
      // 加粗
      this.styles.push('font-weight: bold');
    } else if (params[0] === 38 && params[1] === 5) {
      // 256 色
      const colorCode = params[2];
      this.styles.push(`color: ${this.color256(colorCode)}`);
    } else if (params[0] >= 30 && params[0] <= 37) {
      // 标准颜色
      this.styles.push(`color: ${this.colorMap[params[0]]}`);
    }
  }
}
```

```typescript
    // 创建房间配置
    const roomOptions = {
      adaptiveStream: true,      // 自适应网络
      dynacast: true,          // 动态码率

      },

    await this.publishLocalTracks();
  }

  // 发布本地音视频
  private async publishLocalTracks(): Promise<void> {
    if (!this.room) return;

    // 获取媒体设备
    const stream = await navigator.mediaDevices.getUserMedia({
      audio: true,
      video: true
    });

    // 发布音频
    const audioTrack = stream.getAudioTracks()[0];
    if (audioTrack) {
      const publication = await this.room.localParticipant.publishTrack(audioTrack, {
        name: 'audio',
        simulcast: true
      });
      this.localTracks.set('audio', publication.track!);
    }

    // 发布视频
    const videoTrack = stream.getVideoTracks()[0];
    if (videoTrack) {
      const publication = await this.room.localParticipant.publishTrack(videoTrack, {
        name: 'video',
        simulcast: true
      });
      this.localTracks.set('video', publication.track!);
    }
  }

  // 屏幕共享
  async startScreenShare(): Promise<void> {
    const stream = await navigator.mediaDevices.getDisplayMedia({
      video: { cursor: 'always' },
      audio: false
    });

    const screenTrack = stream.getVideoTracks()[0];

    await this.room?.localParticipant.publishTrack(screenTrack, {
      name: 'screen',
      simulcast: false
    });

    // 监听停止
    screenTrack.onended = () => {
      this.stopScreenShare();
    };
  }

  // 切换音频
  async toggleAudio(enabled: boolean): Promise<void> {
    const audioPub = this.room?.localParticipant.audioTracks.values().next().value;
    if (!audioPub) return;

    }
```

**网格布局计算**：
```typescript
// 视频布局计算器
function calculateVideoLayout(
  containerWidth: number,
  containerHeight: number,
  participantCount: number
): VideoTile[] {
  const tiles: VideoTile[] = [];
  const padding = 8;
  const aspectRatio = 16 / 9;

  // 计算最佳列数（正方形布局）
  const columns = Math.ceil(Math.sqrt(participantCount));
  const rows = Math.ceil(participantCount / columns);

  // 计算单个瓦片尺寸
  const tileWidth = (containerWidth - padding * (columns + 1)) / columns;
  const tileHeight = tileWidth / aspectRatio;

  // 限制最大高度
  const maxHeight = (containerHeight - padding * (rows + 1)) / rows;
  const actualHeight = Math.min(tileHeight, maxHeight);
  const actualWidth = actualHeight * aspectRatio;

  // 计算居中偏移
  const totalWidth = columns * actualWidth + (columns - 1) * padding;
  const totalHeight = rows * actualHeight + (rows - 1) * padding;
  const offsetX = (containerWidth - totalWidth) / 2;
  const offsetY = (containerHeight - totalHeight) / 2;

  // 生成每个瓦片的位置
  for (let i = 0; i < participantCount; i++) {
    const col = i % columns;
    const row = Math.floor(i / columns);

    tiles.push({
      id: `participant-${i}`,
      x: offsetX + col * (actualWidth + padding),
      y: offsetY + row * (actualHeight + padding),
      width: actualWidth,
      height: actualHeight,
      zIndex: i
    });
  }

  return tiles;
}

// 演讲者视图布局
function calculateSpeakerLayout(
  containerWidth: number,
  containerHeight: number,
  speakerId: string,
  otherIds: string[]
): VideoTile[] {
  const tiles: VideoTile[] = [];
  const padding = 8;

  // 演讲者大画面 (75% 高度)
  const mainHeight = containerHeight * 0.75;
  const mainWidth = containerWidth - padding * 2;

  tiles.push({
    id: speakerId,
    x: padding,
    y: padding,
    width: mainWidth,
    height: mainHeight,
    zIndex: 100
  });

  // 其他参与者小画面
  if (otherIds.length > 0) {
    const thumbWidth = 160;
    const thumbHeight = thumbWidth * 9 / 16;
    const thumbsPerRow = Math.floor((containerWidth - padding * 2) / (thumbWidth + padding));

    otherIds.forEach((id, index) => {
      const col = index % thumbsPerRow;
      const row = Math.floor(index / thumbsPerRow);

      tiles.push({
        id,
        x: padding + col * (thumbWidth + padding),
        y: mainHeight + padding + row * (thumbHeight + padding),
        width: thumbWidth,
        height: thumbHeight,
        zIndex: index
      });
    });
  }

  return tiles;
}
```
