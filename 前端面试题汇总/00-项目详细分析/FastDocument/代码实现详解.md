## 1. 富文本编辑器具体实现

### 1.1 TipTap 编辑器集成

> **技术选型理由**: 项目采用自定义块编辑器架构而非直接使用 TipTap，核心原因是需要更精细的块级控制、虚拟滚动支持以及与协作系统的深度集成。自定义编辑器采用 Block 架构，每个内容块独立管理，便于实现拖拽排序、分块加载等高级功能。

#### 编辑器组件完整代码

核心编辑器位于 `D:\Develeping\FastDocument\frontend\src\components\Editor.tsx`，采用块架构设计：

```tsx
"use client";

import React, { useRef, useEffect, useState, useCallback } from "react";
import { useDocumentStore, Block } from "@/store/documentStore";
import { socketClient } from "@/lib/socket";
import { exportDocument, importDocument } from "@/lib/export";
import { useTheme } from "@/components/ThemeProvider";
import { DocumentMenuBar } from "@/components/DocumentMenuBar";
import { DocumentOutline } from "@/components/DocumentOutline";
import { BubbleMenu } from "@/components/BubbleMenu";
import { VirtualEditor } from "./VirtualEditor";
import { motion, AnimatePresence } from "framer-motion";

/**
 * 文档编辑器主组件
 * 根据文档大小自动选择普通编辑器或虚拟滚动编辑器
 *
 * 业务场景:
 * - 小型文档 (<100 块): 使用普通渲染，完整的动画效果
 * - 大型文档 (>=100 块): 使用虚拟滚动，优化性能
 */
export const Editor: React.FC = () => {
  const {
    id: docId,
    blocks,
    title,
    updateTitle,
    addBlock,
    remoteUpdateBlock,
    setUserTyping,
    removeUserTyping,
    fetchDocument,
    setOnlineStatus,
    setOnlineUsers,
    onlineUsers
  } = useDocumentStore();
  const { isMounted, theme } = useTheme();
  const isDark = isMounted && theme.isDark;
  const [showOutline, setShowOutline] = useState(false);

  // 判断是否有协作者（仅显示聊天和评论功能）
  const hasCollaborators = onlineUsers && onlineUsers.length > 0;

  // 大文件阈值：超过 100 个块使用虚拟滚动编辑器
  const VIRTUAL_EDITOR_THRESHOLD = 100;
  const useVirtualEditor = blocks.length > VIRTUAL_EDITOR_THRESHOLD;

  // 初始数据获取 & Socket 连接
  useEffect(() => {
    fetchDocument(docId || "demo-doc");
    const currentUserName = typeof window !== 'undefined'
      ? (localStorage.getItem('username') || "Guest")
      : "Guest";

    // 建立 WebSocket 连接
    socketClient.connect((status) => setOnlineStatus(status));
    socketClient.joinDocument(docId, currentUserName);

    // 监听远程块更新
    socketClient.onBlockUpdated((data: any) => {
      if (data.docId === docId) {
        remoteUpdateBlock(data.blockId, data.content, data.properties, data.type);
      }
    });

    return () => { socketClient.disconnect(); };
  }, [docId]);

  // 大文件使用虚拟滚动编辑器
  if (useVirtualEditor) {
    return <VirtualEditor />;
  }

  return (
    <div className={`relative min-h-screen ${isDark ? 'bg-[#0B0F1A]' : 'bg-white'}`}>
      <DocumentMenuBar onToggleOutline={() => setShowOutline(!showOutline)} showOutline={showOutline} />

      <main className={`mx-auto max-w-4xl px-12 pt-32 pb-40 transition-all duration-300 ${showOutline ? 'pr-80' : ''}`}>
        <EditorToolbar />

        {/* Document Title */}
        <input
          type="text"
          className={`w-full text-5xl font-bold border-none outline-none mb-12 bg-transparent placeholder:opacity-40 ${isDark ? 'text-white' : 'text-gray-900'}`}
          placeholder="Untitled"
          value={title}
          onChange={(e) => updateTitle(e.target.value)}
        />

        {/* Blocks Area - 使用 AnimatePresence 实现块添加/删除动画 */}
        <div className="space-y-1">
          <AnimatePresence mode="popLayout">
            {blocks.map((block, index) => (
              <BlockRenderer key={block.id} block={block} index={index} />
            ))}
          </AnimatePresence>
        </div>

        {/* Empty State / Add Block Area */}
        <div
          className="mt-4 opacity-50 hover:opacity-100 transition-opacity cursor-text"
          onClick={() => addBlock("text")}
        >
          <div className="flex items-center gap-2 text-gray-400">
            <Plus size={18} />
            <span>Click to add a block</span>
          </div>
        </div>

        {/* 协作功能 - 仅在有协作者时显示 */}
        {hasCollaborators && (
          <>
            <ChatDrawer />
            <CommentPanel />
          </>
        )}

        {/* 浮动气泡菜单用于文本选择 */}
        <BubbleMenu onFormat={handleFormat} />
      </main>

      <DocumentOutline isVisible={showOutline} onClose={() => setShowOutline(false)} />
    </div>
  );
};
```

#### 快捷键处理

块编辑器支持丰富的快捷键操作：

```tsx
const handleKeyDown = (e: React.KeyboardEvent<HTMLElement>) => {
  // Slash 命令菜单导航
  if (slashMenuVisible) {
    if (e.key === "ArrowDown") {
      e.preventDefault();
      setSelectedSlashIndex((prev) => Math.min(prev + 1, slashCommands.length - 1));
      return;
    }
    if (e.key === "ArrowUp") {
      e.preventDefault();
      setSelectedSlashIndex((prev) => Math.max(prev - 1, 0));
      return;
    }
    if (e.key === "Enter") {
      e.preventDefault();
      const cmd = slashCommands[selectedSlashIndex];
      updateBlock(block.id, "", {}, cmd.key as Block["type"]);
      socketClient.updateBlock(docId, block.id, "", cmd.key as Block["type"]);
      setSlashMenuVisible(false);
      return;
    }
    if (e.key === "Escape") {
      setSlashMenuVisible(false);
      return;
    }
  }

  // Enter: 在当前块下方创建新块
  if (e.key === "Enter" && !e.shiftKey) {
    e.preventDefault();
    addBlock("text", block.id);
    setSlashMenuVisible(false);
  }

  // Backspace: 空块时删除
  if (e.key === "Backspace" && block.content === "") {
    e.preventDefault();
    removeBlock(block.id);
  }

  // Tab: 缩进（未来扩展）
  if (e.key === "Tab") {
    e.preventDefault();
    indentBlock(block.id, e.shiftKey ? 'decrease' : 'increase');
  }
};
```

#### Markdown 触发器

```tsx
// 在 BlockRenderer 组件中
const checkMarkdownTrigger = (content: string) => {
  const patterns: Record<string, Block['type']> = {
    "# ": "h1",           // # 标题 -> H1
    "## ": "h2",          // ## 标题 -> H2
    "### ": "h3",         // ### 标题 -> H3
    "- ": "todo",         // - 待办 -> 待办事项
    "[] ": "todo",        // [ ] 待办 -> 待办事项
    "> ": "callout",      // > 引用 -> 提示框
    "---": "divider",     // --- -> 分隔线
    "```": "code"         // ``` -> 代码块
  };

  // 检查精确匹配（如 ---）
  if (patterns[content]) {
    transformBlock(block.id, patterns[content]);
    if (patterns[content] === 'divider' || patterns[content] === 'code') {
      updateBlock(block.id, "", {}, patterns[content]);
      socketClient.updateBlock(docId, block.id, "", patterns[content]);
    }
    return true;
  }

  // 检查前缀匹配
  for (const [prefix, type] of Object.entries(patterns)) {
    if (content.startsWith(prefix) && content.length > prefix.length) {
      const newContent = content.slice(prefix.length);
      updateBlock(block.id, newContent, {}, type);
      socketClient.updateBlock(docId, block.id, newContent, type);
      return true;
    }
  }
  return false;
};
```

### 1.3 虚拟滚动编辑器

对于大型文档（>100 块），使用虚拟滚动优化性能：

```tsx
// VirtualEditor.tsx
const ITEM_HEIGHT = 60;      // 估计的块高度
const OVERSCAN = 5;          // overscan 数量（预渲染区域）

/**
 * 虚拟滚动编辑器组件
 * 适用于大文件文档（>100 个块）
 */
export const VirtualEditor: React.FC = () => {
  const [scrollTop, setScrollTop] = useState(0);
  const [containerHeight, setContainerHeight] = useState(800);
  const containerRef = useRef<HTMLDivElement>(null);

  // 计算可见块
  const { visibleItems, totalHeight, startIndex, endIndex } = useMemo(() => {
    const totalHeight = blocks.length * ITEM_HEIGHT;
    const startIdx = Math.max(0, Math.floor(scrollTop / ITEM_HEIGHT) - OVERSCAN);
    const endIdx = Math.min(
      blocks.length - 1,
      Math.ceil((scrollTop + containerHeight) / ITEM_HEIGHT) + OVERSCAN
    );

    const visible = [];
    for (let i = startIdx; i <= endIdx; i++) {
      visible.push({ block: blocks[i], index: i });
    }

    return { visibleItems: visible, totalHeight, startIndex: startIdx, endIndex: endIdx };

  useEffect(() => {
```

```typescript
// 检测冲突
const detectConflict = (blockId: string, newContent: string) => {
  const { blocks, pendingConflicts } = get();
  const block = blocks.find(b => b.id === blockId);

  if (!block) return;

  // 简单冲突检测：本地和远程内容不同
  // 在实际生产环境中，应该使用时间戳或版本号来判断
  const hasConflict = pendingConflicts.some(c => c.blockId === blockId);

  if (hasConflict && block.content !== newContent) {
    // 添加到待解决冲突列表
    set(state => ({
      pendingConflicts: [
        ...state.pendingConflicts,
        {
          blockId,
          localContent: block.content,
          remoteContent: newContent,
          timestamp: Date.now()
        }
      ]
    }));
    return true;
  }
  return false;
};

// 解决冲突
const resolveConflict = (blockId: string, useLocal: boolean) => {
  const { pendingConflicts, blocks } = get();
  const conflict = pendingConflicts.find(c => c.blockId === blockId);

  if (!conflict) return;

  const newContent = useLocal ? conflict.localContent : conflict.remoteContent;

  // 更新块内容
  set(state => ({
    blocks: state.blocks.map(b =>
      b.id === blockId ? { ...b, content: newContent } : b
    ),
    pendingConflicts: state.pendingConflicts.filter(c => c.blockId !== blockId)
  }));

  // 同步到服务器
  socketClient.updateBlock(get().id, blockId, newContent);
};
```

```typescript
export interface MediaStreamConfig {
  video?: boolean | MediaTrackConstraints;
  audio?: boolean | MediaTrackConstraints;
  screenShare?: boolean;
}

class WebRTCManager {
  private localStream: MediaStream | null = null;
  private screenStream: MediaStream | null = null;
  private mediaRecorder: MediaRecorder | null = null;
  private recordedChunks: Blob[] = [];

  // 事件回调
  onStreamReady?: (stream: MediaStream) => void;
  onScreenStreamReady?: (stream: MediaStream) => void;
  onRecordingStarted?: () => void;
  onRecordingStopped?: (blob: Blob, duration: number) => void;
  onError?: (error: Error) => void;

  /**
   * 获取本地媒体流（摄像头和麦克风）
   */
  async getLocalStream(config: MediaStreamConfig = {}): Promise<MediaStream> {
    try {
      const constraints: MediaStreamConstraints = {
        video: config.video ?? true,
        audio: config.audio ?? true,
      };

      this.localStream = await navigator.mediaDevices.getUserMedia(constraints);
      this.onStreamReady?.(this.localStream);
      return this.localStream;
    } catch (error) {
      const err = error as Error;
      console.error("获取媒体设备失败:", err);
      this.onError?.(err);
      throw err;
    }
  }

  /**
   * 获取屏幕共享流
   */
  async getScreenStream(): Promise<MediaStream> {
    try {
      this.screenStream = await navigator.mediaDevices.getDisplayMedia({
        video: true,
        audio: true,
      });

      // 监听屏幕共享停止
      this.screenStream.getVideoTracks()[0].onended = () => {
        this.stopScreenShare();
      };

      this.onScreenStreamReady?.(this.screenStream);
      return this.screenStream;
    } catch (error) {
      const err = error as Error;
      console.error("获取屏幕共享失败:", err);
      this.onError?.(err);
      throw err;
    }
  }

  /**
   * 切换摄像头
   */
  async switchCamera(): Promise<void> {
    if (!this.localStream) return;

    const videoTracks = this.localStream.getVideoTracks();
    if (videoTracks.length === 0) return;

    // 获取所有视频设备
    const devices = await navigator.mediaDevices.enumerateDevices();
    const videoDevices = devices.filter((d) => d.kind === "videoinput");

    if (videoDevices.length < 2) return;

    const currentDeviceId = videoTracks[0].getSettings().deviceId;
    const currentIndex = videoDevices.findIndex((d) => d.deviceId === currentDeviceId);
    const nextIndex = (currentIndex + 1) % videoDevices.length;

    // 切换到下一个摄像头
    const newStream = await navigator.mediaDevices.getUserMedia({
      video: { deviceId: { exact: videoDevices[nextIndex].deviceId } },
      audio: false,
    });

    // 替换视频轨道
    const newVideoTrack = newStream.getVideoTracks()[0];
    const oldVideoTrack = videoTracks[0];

    this.localStream.removeTrack(oldVideoTrack);
    this.localStream.addTrack(newVideoTrack);
    oldVideoTrack.stop();
  }

  /**
   * 启用/禁用视频
   */
  toggleVideo(enabled: boolean): void {
    if (!this.localStream) return;
    this.localStream.getVideoTracks().forEach((track) => {
      track.enabled = enabled;
    });
  }

  /**
   * 启用/禁用音频
   */
  toggleAudio(enabled: boolean): void {
    if (!this.localStream) return;
    this.localStream.getAudioTracks().forEach((track) => {
      track.enabled = enabled;
    });
  }

  /**
   * 开始录制
   */
  startRecording(includeScreen: boolean = false): void {
    if (this.mediaRecorder?.state === "recording") return;

    let streamToRecord: MediaStream;

    if (includeScreen && this.screenStream) {
      // 合并屏幕和音频
      streamToRecord = this.screenStream.clone();
      if (this.localStream) {
        this.localStream.getAudioTracks().forEach((track) => {
          streamToRecord.addTrack(track);
        });
      }
    } else if (this.localStream) {
      streamToRecord = this.localStream.clone();
    } else {
      return;
    }

    // 选择支持的 MIME 类型
    let mimeType = "video/webm;codecs=vp9,opus";
    if (!MediaRecorder.isTypeSupported(mimeType)) {
      mimeType = "video/webm;codecs=vp8,opus";
      if (!MediaRecorder.isTypeSupported(mimeType)) {
        mimeType = "video/webm";
      }
    }

    this.recordedChunks = [];
    this.mediaRecorder = new MediaRecorder(streamToRecord, { mimeType });

    this.mediaRecorder.ondataavailable = (event) => {
      if (event.data.size > 0) {
        this.recordedChunks.push(event.data);
      }
    };

    this.mediaRecorder.onstop = () => {
      const blob = new Blob(this.recordedChunks, { type: mimeType });
      const duration = Date.now() - this.recordingStartTime;
      this.onRecordingStopped?.(blob, duration);
    };

    this.mediaRecorder.start(1000); // 每秒收集数据
    this.recordingStartTime = Date.now();
    this.onRecordingStarted?.();
  }

  /**
   * 停止录制
   */
  stopRecording(): void {
    if (this.mediaRecorder?.state === "recording") {
      this.mediaRecorder.stop();
    }
  }

  /**
   * 获取设备列表
   */
  async getDevices(): Promise<{
    videoInputs: MediaDeviceInfo[];
    audioInputs: MediaDeviceInfo[];
    audioOutputs: MediaDeviceInfo[];
  }> {
    const devices = await navigator.mediaDevices.enumerateDevices();
    return {
      videoInputs: devices.filter((d) => d.kind === "videoinput"),
      audioInputs: devices.filter((d) => d.kind === "audioinput"),
      audioOutputs: devices.filter((d) => d.kind === "audiooutput"),
    };
  }

  /**
   * 清理所有资源
   */
  dispose(): void {
    this.stopRecording();
    this.stopScreenShare();

    if (this.localStream) {
      this.localStream.getTracks().forEach((track) => track.stop());
      this.localStream = null;
    }
  }
}
```

```typescript
      history: [],
      historyIndex: -1,
      canUndo: false,
      canRedo: false,

      // 大文件优化：分块加载状态
      chunkedLoading: {
        loadedChunks: new Map(),
        totalBlocks: 0,
        chunkSize: 50,
        lastSyncTime: null,
      },

      loadBlocksChunked: async (offset: number, limit: number) => {
        const { id, chunkedLoading } = get();
        set({ isLoadingChunk: true });

        try {
          const res = await api.get(`/documents/${id}/blocks`, {
            params: { offset, limit, includeTotal: true },
          });

          const newChunks = new Map(chunkedLoading.loadedChunks);
          newChunks.set(offset, res.data.blocks);

          set({
            chunkedLoading: {
              ...chunkedLoading,
              loadedChunks: newChunks,
              totalBlocks: res.data.total || chunkedLoading.totalBlocks,
            },
            blocks: Array.from(newChunks.values()).flat(),
            isLoadingChunk: false,
          });
        } catch (error) {
          console.error("分块加载失败:", error);
          set({ isLoadingChunk: false });
        }

      // 更新块
      updateBlock: (id: string, content: string, properties?: Block["properties"], type?: Block["type"]) => {
        get().saveHistory();
        set((state) => ({
          blocks: state.blocks.map((b) =>
            b.id === id ? { ...b, content, type: type || b.type, properties: { ...b.properties, ...properties } } : b
          ),
        }));
      },

      addBlock: (type: Block["type"], afterId?: string) => {
        get().saveHistory();
        set((state) => {
          const newBlock: Block = {
            id: Math.random().toString(36).substr(2, 9),
            type,
            content: ""
          };
          if (!afterId) return { blocks: [...state.blocks, newBlock] };
          const index = state.blocks.findIndex((b) => b.id === afterId);
          const newBlocks = [...state.blocks];
          newBlocks.splice(index + 1, 0, newBlock);
          return { blocks: newBlocks };
        });
      },

      // 删除块
      removeBlock: (id: string) => {
        get().saveHistory();
        set((state) => ({
          blocks: state.blocks.filter((b) => b.id !== id),
        }));
      },

      // 移动块
      moveBlock: (fromIndex: number, toIndex: number) => {
        get().saveHistory();
        set((state) => {
          const newBlocks = [...state.blocks];
          const [removed] = newBlocks.splice(fromIndex, 1);
          newBlocks.splice(toIndex, 0, removed);
          return { blocks: newBlocks };
        });
      },

      // 保存历史记录（撤销/重做）
      saveHistory: () => {
        const { blocks, title, history, historyIndex } = get();
        const newHistory = history.slice(0, historyIndex + 1);
        newHistory.push({ blocks: JSON.parse(JSON.stringify(blocks)), title });

        // 限制历史记录长度
        if (newHistory.length > 50) {
          newHistory.shift();
        }

        set({
          history: newHistory,
          historyIndex: newHistory.length - 1,
          canUndo: newHistory.length > 1,
          canRedo: false
        });
      },

      // 撤销
      undo: () => {
        const { history, historyIndex } = get();
        if (historyIndex > 0) {
          const prevState = history[historyIndex - 1];
          set({
            blocks: prevState.blocks,
            title: prevState.title,
            historyIndex: historyIndex - 1,
            canUndo: historyIndex - 1 > 0,
            canRedo: true
          });
        }
      },

      // 重做
      redo: () => {
        const { history, historyIndex } = get();
        if (historyIndex < history.length - 1) {
          const nextState = history[historyIndex + 1];
          set({
            blocks: nextState.blocks,
            title: nextState.title,
            historyIndex: historyIndex + 1,
            canUndo: true,
            canRedo: historyIndex + 1 < history.length - 1
          });
        }
      },
    }),
    {
      name: "fastdoc-main-storage",
      storage: createJSONStorage(() => localStorage),
      partialize: (state) => ({
        title: state.title,
        blocks: state.blocks,
        id: state.id
      }),
    }
  )
);
```

```typescript
const ITEM_HEIGHT = 60;      // 固定块高度
const OVERSCAN = 5;          // 预渲染数量

// 计算可见块
const { visibleItems, totalHeight, startIndex, endIndex } = useMemo(() => {
  const totalHeight = blocks.length * ITEM_HEIGHT;

  // 计算起始索引（考虑 overscan）
  const startIdx = Math.max(0, Math.floor(scrollTop / ITEM_HEIGHT) - OVERSCAN);

  // 计算结束索引
  const endIdx = Math.min(
    blocks.length - 1,
    Math.ceil((scrollTop + containerHeight) / ITEM_HEIGHT) + OVERSCAN
  );

  // 提取可见块
  const visible = [];
  for (let i = startIdx; i <= endIdx; i++) {
    visible.push({ block: blocks[i], index: i });
  }

  return { visibleItems: visible, totalHeight, startIndex: startIdx, endIndex: endIdx };
}, [blocks, scrollTop, containerHeight]);

// 渲染虚拟滚动区域
<div style={{ height: totalHeight, position: 'relative' }}>
  {visibleItems.map(({ block, index }) => (
    <div
      key={block.id}
      style={{
        position: 'absolute',
        top: index * ITEM_HEIGHT,
        height: ITEM_HEIGHT,
        width: '100%',
      }}
    >
      <BlockRenderer block={block} index={index} />
    </div>
  ))}
</div>
```

```typescript
// 导出为 Markdown
export function blocksToMarkdown(blocks: Block[], title: string): string {
  let md = `# ${title}\n\n`;

  for (const block of blocks) {
    switch (block.type) {
      case "h1":
        md += `# ${block.content}\n\n`;
        break;
      case "h2":
        md += `## ${block.content}\n\n`;
        break;
      case "todo":
        md += `- [${block.properties?.checked ? "x" : " "}] ${block.content}\n`;
        break;
      case "code":
        md += `\`\`\`\n${block.content}\n\`\`\`\n\n`;
        break;
      case "image":
        md += `![${block.content}](${block.properties?.url || ""})\n\n`;
        break;
      // ... 更多类型
      default:
        md += `${block.content}\n\n`;
    }
  }

  return md;
}

// 导出为 DOCX
export async function blocksToDocx(blocks: Block[], title: string): Promise<Blob> {
  const children: Paragraph[] = [];

  // 添加标题
  children.push(
    new Paragraph({
      text: title,
      heading: HeadingLevel.TITLE,
      spacing: { after: 400 }
    })
  );

  // 转换每个块
  for (const block of blocks) {
    const paragraph = blockToDocxParagraph(block);
    if (paragraph) {
      children.push(paragraph);
    }
  }

  const doc = new Document({
    sections: [{
      properties: {},
      children
    }]
  });

  return await Packer.toBlob(doc);
}

export async function importDocument(file: File): Promise<{ title: string; blocks: Block[] }> {
  const extension = file.name.split(".").pop()?.toLowerCase();

  switch (extension) {
    case "md":
    case "markdown":
      return await parseMarkdown(file);
    case "docx":
      return await parseDocx(file);
    case "txt":
      return await parseTxt(file);
    case "html":
```

```tsx
const BubbleMenu: React.FC<{ onFormat: (format: string) => void }> = ({ onFormat }) => {
  const [position, setPosition] = useState({ top: 0, left: 0 });
  const [visible, setVisible] = useState(false);

  useEffect(() => {
    const handleSelectionChange = () => {
      const selection = window.getSelection();
      if (selection && selection.toString().length > 0) {
        const range = selection.getRangeAt(0);
        const rect = range.getBoundingClientRect();

        setPosition({
          top: rect.top - 50,
          left: rect.left + (rect.width / 2),
        });
        setVisible(true);
      } else {
        setVisible(false);
      }
    };

    document.addEventListener('selectionchange', handleSelectionChange);
    return () => document.removeEventListener('selectionchange', handleSelectionChange);
  }, []);

  if (!visible) return null;

  return (
    <motion.div
      initial={{ opacity: 0, y: 10 }}
      animate={{ opacity: 1, y: 0 }}
      className="fixed bg-white shadow-lg rounded-lg flex gap-1 p-1 z-50"
      style={{ top: position.top, left: position.left }}
    >
      <button onClick={() => onFormat('bold')}><Bold size={16} /></button>
      <button onClick={() => onFormat('italic')}><Italic size={16} /></button>
      <button onClick={() => onFormat('underline')}><Underline size={16} /></button>
```
