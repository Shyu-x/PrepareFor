## 第一部分：3D 渲染核心实现

### 1.1 Scene3D 主场景组件

**文件路径**: `frontend/src/components/game/Scene3D.tsx`

**核心实现逻辑**:

Scene3D 是整个 3D 游戏的核心容器，使用 React Three Fiber 构建完整的 3D 游戏环境。

```typescript
import { Canvas } from '@react-three/fiber';
import { Stars, PerspectiveCamera } from '@react-three/drei';
import { useMemo } from 'react';
import * as THREE from 'three';

// 主场景组件
export default function Scene3D({
  gameState,
  onCardClick,
  onDrawClick,
  onColorSelect,
  onUnoClick
}: Scene3DProps) {
  // 响应式相机配置
  const cameraConfig = useMemo(() => {
    const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
    const isPortrait = window.innerHeight > window.innerWidth;
    const playerCount = gameState?.players.length || 4;

    let cameraPos: [number, number, number];
    let cameraFov: number;

    if (isMobile) {
      if (isPortrait) {
        cameraPos = [0, 55, 70];
        cameraFov = 60;
      } else {
        cameraPos = [0, 35, 48];
        cameraFov = 45;
      }
    } else {
      if (playerCount >= 10) {
        cameraPos = [0, 28, 45];
        cameraFov = 28;
      } else if (playerCount >= 6) {
        cameraPos = [0, 25, 40];
        cameraFov = 36;
      } else {
        cameraPos = [0, 22, 35];
        cameraFov = 35;
      }
    }

    return { position: cameraPos, fov: cameraFov };
  }, [gameState?.players.length]);

  return (
    <Canvas shadows>
      {/* 相机系统 */}
      <PerspectiveCamera
        makeDefault
        position={cameraConfig.position}
        fov={cameraConfig.fov}
      />

      {/* 环境系统 */}
      <Stars radius={100} depth={50} count={5000} factor={4} fade speed={1} />
      <color attach="background" args={['#0a0a1a']} />
      <fog attach="fog" args={['#0a0a1a', 30, 80]} />

      {/* 灯光系统 */}
      <ambientLight intensity={1.2} />
      <pointLight position={[0, 30, 0]} intensity={150} castShadow />
      <spotLight
        position={[0, 50, 0]}
        angle={0.5}
        penumbra={0.6}
        intensity={200}
        color="#fef3c7"
      />

      {/* 游戏桌面 */}
      <GameTable gameState={gameState} />

      {/* 玩家区域 */}
      {gameState?.players.map((player, index) => (
        <PlayerArea
          key={player.id}
          player={player}
          position={calculatePlayerPosition(index, gameState.players.length)}
          isCurrentPlayer={gameState.currentPlayerIndex === index}
        />
      ))}

      {/* 卡牌堆和弃牌堆 */}
      <Deck3D onClick={onDrawClick} />
      <DiscardPile3D cards={gameState?.discardPile || []} />

      {/* 特效层 */}
      <ParticleEffects />
    </Canvas>
  );
}

// 计算玩家在 3D 空间中的位置
function calculatePlayerPosition(index: number, totalPlayers: number): [number, number, number] {
  const radius = 15;
  const angle = (index / totalPlayers) * Math.PI * 2 - Math.PI / 2;
  return [
    Math.cos(angle) * radius,
    0,
    Math.sin(angle) * radius
  ];
}
```

**技术难点**:
1. **性能优化**: 3D 场景中需要同时渲染大量卡牌和粒子效果，需要使用 `useMemo` 缓存计算结果
2. **响应式适配**: 不同设备需要不同的相机参数，需要动态计算
3. **阴影渲染**: 使用 `castShadow` 和 `receiveShadow` 需要合理设置光源位置

**工作量分布**:
- 相机系统配置: 30%
- 环境系统搭建: 25%
- 灯光系统调试: 20%
- 响应式适配: 25%

### 1.2 Card3D 3D 卡牌组件

**文件路径**: `frontend/src/components/game/Card3D.tsx`

**核心实现逻辑**:

使用 Three.js 的 BoxGeometry 创建 3D 卡牌模型，结合 React Spring 实现动画效果。

```typescript
import { useRef, useState } from 'react';
import { useFrame } from '@react-three/fiber';
import { useSpring, animated } from '@react-spring/three';
import * as THREE from 'three';
import { CardColor, CardType } from '@/types/game';

interface Card3DProps {
  card: UnoCard;
  position: [number, number, number];
  rotation?: [number, number, number];
  isPlayable?: boolean;
  isHovered?: boolean;
  isHinted?: boolean;
  onClick?: () => void;
}

// 卡牌颜色映射
const COLOR_MAP: Record<CardColor, string> = {
  [CardColor.RED]: '#FF6B6B',
  [CardColor.GREEN]: '#51CF66',
  [CardColor.BLUE]: '#339AF0',
  [CardColor.YELLOW]: '#FFD43B',
  [CardColor.WILD]: '#868E96',
};

export function Card3D({
  card,
  position,
  rotation = [0, 0, 0],
  isPlayable = false,
  isHovered = false,
  isHinted = false,
  onClick
}: Card3DProps) {
  const meshRef = useRef<THREE.Mesh>(null);
  const [hovered, setHovered] = useState(false);

  // 使用 React Spring 实现动画
  const { springPos, springScale, springRot } = useSpring({
    springPos: [
      position[0],
      position[1] + (hovered ? 2.5 : (isHinted ? 0.8 : 0)),
      position[2] + (hovered ? 1.5 : 0)
    ] as [number, number, number],
    springScale: hovered ? 1.25 : 1,
    springRot: rotation as [number, number, number],
    config: { mass: 1, tension: 350, friction: 35 }
  });

  // 卡牌几何尺寸
  const cardWidth = 2.5;
  const cardHeight = 3.5;
  const cardDepth = 0.18;

  return (
    <animated.group
      position={springPos}
      scale={springScale}
      rotation={springRot}
    >
      <mesh
        ref={meshRef}
        onClick={(e) => {
          e.stopPropagation();
          if (isPlayable) onClick?.();
        }}
        onPointerOver={(e) => {
          e.stopPropagation();
          if (isPlayable) setHovered(true);
        }}
        onPointerOut={() => setHovered(false)}
        castShadow
        receiveShadow
      >
        {/* 卡牌主体 */}
        <boxGeometry args={[cardWidth, cardHeight, cardDepth]} />

        {/* 材质 - 使用 MeshStandardMaterial 实现真实光照 */}
        <meshStandardMaterial
          color={COLOR_MAP[card.color]}
          roughness={0.4}
          metalness={0.1}
        />

        {/* 卡牌正面内容 */}
        <CardFace card={card} position={[0, 0, cardDepth / 2 + 0.01]} />

        {/* 卡牌背面 */}
        <CardBack position={[0, 0, -cardDepth / 2 - 0.01]} />
      </mesh>

      {/* 发光效果 - 可出牌时显示 */}
      {isPlayable && (
        <pointLight
          position={[0, 0, 2]}
          intensity={hovered ? 2 : 0.5}
          color="#ffffff"
          distance={5}
        />
      )}
    </animated.group>
  );
}

function CardFace({ card, position }: { card: UnoCard; position: [number, number, number] }) {
  const color = card.color === CardColor.WILD ? '#1a1a1a' : COLOR_MAP[card.color];

  return (
    <group position={position}>
      {/* 数字/符号显示 */}
      {card.type === CardType.NUMBER && (
        <mesh position={[0, 0, 0.01]}>
          <planeGeometry args={[1, 1.5]} />
          <meshBasicMaterial color="#ffffff" transparent opacity={0.9} />
          {/* 这里可以使用 Text 组件渲染数字 */}
        </mesh>
      )}

      {card.type === CardType.SKIP && (
        <SkipSymbol position={[0, 0, 0.01]} color={color} />
      )}

      {card.type === CardType.DRAW_TWO && (
        <DrawTwoSymbol position={[0, 0, 0.01]} color={color} />
      )}

      {card.type === CardType.WILD && (
        <WildSymbol position={[0, 0, 0.01]} />
      )}

      {card.type === CardType.WILD_DRAW_FOUR && (
        <WildDrawFourSymbol position={[0, 0, 0.01]} />

    </group>
  );
}
```

### 1.3 粒子特效系统

**文件路径**: `frontend/src/components/game/particles/ConfettiParticles.tsx`

**核心实现逻辑**:

使用 Three.js 的 Points 和 BufferGeometry 实现高性能粒子特效。

```typescript
import { useRef, useMemo } from 'react';
import { useFrame } from '@react-three/fiber';
import * as THREE from 'three';

interface ConfettiParticlesProps {
  active: boolean;
  count?: number;
}

// 彩带粒子系统
export function ConfettiParticles({ active, count = 200 }: ConfettiParticlesProps) {
  const pointsRef = useRef<THREE.Points>(null);

  // 使用 useMemo 缓存粒子数据，避免每帧重新创建
  const { positions, velocities, colors, sizes } = useMemo(() => {
    const positions = new Float32Array(count * 3);
    const velocities = new Float32Array(count * 3);
    const colors = new Float32Array(count * 3);
    const sizes = new Float32Array(count);

    const colorPalette = [
      new THREE.Color('#ff6b6b'),
      new THREE.Color('#4ecdc4'),
      new THREE.Color('#ffe66d'),
      new THREE.Color('#95e1d3'),
      new THREE.Color('#f38181'),
    ];

    for (let i = 0; i < count; i++) {
      const i3 = i * 3;

      // 随机初始位置 - 从中心向外扩散
      positions[i3] = (Math.random() - 0.5) * 10;
      positions[i3 + 1] = Math.random() * 20 + 10;
      positions[i3 + 2] = (Math.random() - 0.5) * 10;

      // 随机速度
      velocities[i3] = (Math.random() - 0.5) * 0.1;
      velocities[i3 + 1] = -Math.random() * 0.2 - 0.05;
      velocities[i3 + 2] = (Math.random() - 0.5) * 0.1;

      // 随机颜色
      const color = colorPalette[Math.floor(Math.random() * colorPalette.length)];
      colors[i3] = color.r;
      colors[i3 + 1] = color.g;
      colors[i3 + 2] = color.b;

      // 随机大小
      sizes[i] = Math.random() * 0.3 + 0.1;
    }

    return { positions, velocities, colors, sizes };
  }, [count]);

  // 动画循环 - 每帧更新粒子位置
  useFrame((_, delta) => {
    if (!active || !pointsRef.current) return;

    const positionAttribute = pointsRef.current.geometry.attributes.position;

    for (let i = 0; i < count; i++) {
      const i3 = i * 3;

      // 更新位置
      positionAttribute.array[i3] += velocities[i3];
      positionAttribute.array[i3 + 1] += velocities[i3 + 1];
      positionAttribute.array[i3 + 2] += velocities[i3 + 2];

      // 边界检查 - 粒子落下后重置到顶部
      if (positionAttribute.array[i3 + 1] < -10) {
        positionAttribute.array[i3] = (Math.random() - 0.5) * 10;
        positionAttribute.array[i3 + 1] = 20 + Math.random() * 10;
        positionAttribute.array[i3 + 2] = (Math.random() - 0.5) * 10;
      }
    }

    positionAttribute.needsUpdate = true;
  });

  if (!active) return null;

  return (
    <points ref={pointsRef}>
      <bufferGeometry>
        <bufferAttribute
          attach="attributes-position"
          count={count}
          array={positions}
          itemSize={3}
        />
        <bufferAttribute
          attach="attributes-color"
          count={count}
          array={colors}
          itemSize={3}
        />
      </bufferGeometry>
      <pointsMaterial
        size={0.3}
        vertexColors
        transparent
        opacity={0.8}
        sizeAttenuation
        blending={THREE.AdditiveBlending}
      />
    </points>
  );
}

// 魔法光球特效
export function MagicOrbs({ active }: { active: boolean }) {
  const orbsRef = useRef<THREE.Group>(null);

  const orbs = useMemo(() => {
    return Array.from({ length: 8 }, (_, i) => ({
      position: [
        Math.cos((i / 8) * Math.PI * 2) * 12,
        Math.random() * 5 + 5,
        Math.sin((i / 8) * Math.PI * 2) * 12
      ] as [number, number, number],
      color: ['#ff6b6b', '#4ecdc4', '#ffe66d', '#a855f7'][i % 4],
      speed: Math.random() * 0.5 + 0.5,
      offset: Math.random() * Math.PI * 2,
    }));
  }, []);

  useFrame(({ clock }) => {
    if (!orbsRef.current) return;

    orbsRef.current.children.forEach((orb, i) => {
      const orbData = orbs[i];
      const t = clock.getElapsedTime() * orbData.speed + orbData.offset;

      orb.position.y = orbData.position[1] + Math.sin(t) * 2;
      orb.position.x = orbData.position[0] + Math.cos(t * 0.5) * 0.5;
    });
  });

  if (!active) return null;

  return (
    <group ref={orbsRef}>
      {orbs.map((orb, i) => (
        <mesh key={i} position={orb.position}>
          <sphereGeometry args={[0.5, 16, 16]} />
          <meshBasicMaterial color={orb.color} transparent opacity={0.6} />
          <pointLight color={orb.color} intensity={2} distance={10} />
        </mesh>
      ))}
    </group>
  );
}
```

**技术难点**:
1. **性能优化**: 使用 BufferAttribute 直接操作底层数据，避免 React 渲染开销
2. **动画算法**: 粒子运动需要模拟重力、空气阻力等物理效果
3. **内存管理**: 大量粒子需要合理管理，避免内存泄漏

### 2.1 Hand2D 手牌组件

**文件路径**: `frontend/src/components/game/2d/Hand2D.tsx`

**核心实现逻辑**:

手牌组件需要处理卡牌重叠、展开动画、点击检测等复杂交互。

```typescript
import { useRef, useState, useEffect, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { useGameStore } from '@/store/useGameStore';
import Card2D from './Card2D';
import CardBack2D from './CardBack2D';
import { cn } from '@/lib/utils';

interface Hand2DProps {
  cards: UnoCard[];
  playerId: string;
  onCardClick?: (card: UnoCard, index: number) => void;
  isCurrentPlayer?: boolean;
}

export default function Hand2D({
  cards,
  playerId,
  onCardClick,
  isCurrentPlayer = false
}: Hand2DProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [containerWidth, setContainerWidth] = useState(0);
  const [hoveredIndex, setHoveredIndex] = useState<number | null>(null);

  const { gameState } = useGameStore();

  // 响应式配置
  const config = useMemo(() => {
    const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);
    return {
      cardWidth: isMobile ? 60 : 100,
      cardHeight: isMobile ? 84 : 140,
      cardOffset: isMobile ? 20 : 35,
      expandedOffset: isMobile ? 8 : 15,
    };
  }, []);

  // 监听容器宽度变化
  useEffect(() => {
    if (!containerRef.current) return;

    const observer = new ResizeObserver((entries) => {
      setContainerWidth(entries[0].contentRect.width);
    });

    observer.observe(containerRef.current);
    return () => observer.disconnect();
  }, []);

  // 计算卡牌重叠量
  const getOverlap = useCallback(() => {
    const cardCount = cards.length;
    if (cardCount === 0) return 0;

    const availableWidth = containerWidth - config.cardWidth;
    const minOverlap = config.cardWidth * 0.25;
    const maxOverlap = config.cardWidth * 0.7;
    const slotWidth = availableWidth / cardCount;
    const calculatedOverlap = config.cardWidth - slotWidth;

    return Math.max(minOverlap, Math.min(maxOverlap, calculatedOverlap));
  }, [cards.length, containerWidth, config]);

  // 处理卡牌点击
  const handleCardClick = (card: UnoCard, index: number) => {
    // 检查是否可以出这张牌
    const canPlay = checkCanPlayCard(card, gameState);
    if (canPlay) {
      onCardClick?.(card, index);
    }
  };

  return (
    <div
      ref={containerRef}
      className={cn(
        "absolute bottom-0 left-0 right-0 flex justify-center",
        isCurrentPlayer ? "cursor-pointer" : "cursor-default"
      )}
      style={{
        height: config.cardHeight + 20,
        padding: '0 10px'
      }}
    >
      <div className="relative flex items-end justify-center h-full">
        {cards.map((card, index) => {
          const overlap = getOverlap();
          const isHovered = hoveredIndex === index;

          // 计算卡牌位置 - 从中心向两侧展开
          const totalWidth = config.cardWidth + (cards.length - 1) * (config.cardWidth - overlap);
          const startX = -totalWidth / 2 + config.cardWidth / 2;
          const x = startX + index * (config.cardWidth - overlap);

          return (
            <motion.div
              key={card.id}
              initial={{ y: 50, opacity: 0 }}
              animate={{
                y: isHovered ? -20 : 0,
                x: x,
                scale: isHovered ? 1.1 : 1,
                zIndex: isHovered ? 100 : index
              }}
              transition={{ type: "spring", stiffness: 300, damping: 25 }}
              onMouseEnter={() => setHoveredIndex(index)}
              onMouseLeave={() => setHoveredIndex(null)}
              className="absolute"
              style={{
                width: config.cardWidth,
                height: config.cardHeight,
                transform: `translateX(${x}px)`
              }}
            >
              <Card2D
                card={card}
                size="medium"
                isPlayable={isCurrentPlayer && checkCanPlayCard(card, gameState)}
                onClick={() => handleCardClick(card, index)}
              />
            </motion.div>
          );
        })}
      </div>
    </div>
  );
}

// 检查卡牌是否可以出
function checkCanPlayCard(card: UnoCard, gameState: GameState | null): boolean {
  if (!gameState) return false;

  const topCard = gameState.discardPile[gameState.discardPile.length - 1];
  if (!topCard) return true;

  // 万能牌可以匹配任何颜色
  if (card.color === CardColor.WILD) return true;

  // 颜色匹配
  if (card.color === topCard.color) return true;

  // 类型匹配
  if (card.type === topCard.type) return true;

  return false;
}
```

**技术难点**:
1. **自适应重叠算法**: 根据容器宽度和卡牌数量动态计算重叠量，确保卡牌不溢出
2. **动画性能**: 使用 Framer Motion 的 `transform` 而不是 `left/top` 实现动画，避免重排
3. **点击检测**: 需要准确判断点击的是哪张卡牌，特别是重叠时

**工作量分布**:
- 响应式布局: 30%
- 动画系统: 25%
- 交互逻辑: 25%
- 状态管理: 20%

```typescript
        type: 'error',

  const emit = useCallback((event: string, ...args: any[]) => {

  }, []);

  // 封装 off 方法
  const off = useCallback((event: string, callback?: (...args: any[]) => void) => {
    if (callback) {
      socketRef.current?.off(event, callback);
    } else {
      socketRef.current?.off(event);
    }
  }, []);

  return (
    <GameSocketContext.Provider value={{ socket: socketRef.current, isConnected, emit, on, off }}>
      {children}
    </GameSocketContext.Provider>
  );
}
```

## 第五部分：后端游戏逻辑

### 5.1 GameService 核心服务

**文件路径**: `backend/src/game/game/game.service.ts`

**核心实现逻辑**:

```typescript
import { Injectable } from '@nestjs/common';
import { v4 as uuidv4 } from 'uuid';
import { Card, CardColor, CardType, GameState, Player } from './types';

@Injectable()
export class GameService {
  // 存储所有游戏房间
  private rooms: Map<string, GameState> = new Map();

  // 创建房间
  createRoom(roomId?: string): GameState {
    const id = roomId || uuidv4();

    const room: GameState = {
      id,
      players: [],
      currentPlayerIndex: 0,
      direction: 1,
      drawPile: [],
      discardPile: [],
      currentColor: null,
      currentWildColor: null,
      status: 'waiting',
      createdAt: Date.now(),
      settings: {
        maxPlayers: 10,
        initialCards: 7,
        winScore: 500,
      },
    };

    this.rooms.set(id, room);
    return room;
  }

  // 生成 UNO 卡牌堆
  private generateUnoDeck(deckCount: number = 1): Card[] {
    const deck: Card[] = [];
    const colors = [CardColor.RED, CardColor.GREEN, CardColor.BLUE, CardColor.YELLOW];

    for (let d = 0; d < deckCount; d++) {
      // 数字牌 0-9 (0 只有一张，1-9 各两张)
      for (const color of colors) {
        deck.push({
          id: uuidv4(),
          color,
          type: CardType.NUMBER,
          value: 0,
        });

        for (let i = 1; i <= 9; i++) {
          deck.push({
            id: uuidv4(),
            color,
            type: CardType.NUMBER,
            value: i,
          });
          deck.push({
            id: uuidv4(),
            color,
            type: CardType.NUMBER,
            value: i,
          });
        }
      }

      // 功能牌 (每种颜色各2张)
      for (const color of colors) {
        for (let i = 0; i < 2; i++) {
          deck.push({
            id: uuidv4(),
            color,
            type: CardType.SKIP,
          });
          deck.push({
            id: uuidv4(),
            color,
            type: CardType.REVERSE,
          });
          deck.push({
            id: uuidv4(),
            color,
            type: CardType.DRAW_TWO,
          });
        }
      }

      // 万能牌 (每种4张)
      for (let i = 0; i < 4; i++) {
        deck.push({
          id: uuidv4(),
          color: CardColor.WILD,
          type: CardType.WILD,
        });
        deck.push({
          id: uuidv4(),
          color: CardColor.WILD,
          type: CardType.WILD_DRAW_FOUR,
        });
      }
    }

    // 洗牌
    return this.shuffleDeck(deck);
  }

  // 洗牌算法 (Fisher-Yates)
  private shuffleDeck(deck: Card[]): Card[] {
    for (let i = deck.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [deck[i], deck[j]] = [deck[j], deck[i]];
    }
    return deck;
  }

  // 开始游戏
  startGame(roomId: string): GameState | null {
    const room = this.rooms.get(roomId);
    if (!room || room.players.length < 2) {
      return null;
    }

    // 生成卡牌堆
    room.drawPile = this.generateUnoDeck(2); // 2 副牌
    room.discardPile = [];

    // 给每个玩家发牌
    for (const player of room.players) {
      player.cards = room.drawPile.splice(0, 7);
      player.hasShoutedUno = false;
    }

    // 翻开第一张牌
    const firstCard = room.drawPile.pop()!;
    room.discardPile.push(firstCard);

    // 设置当前颜色
    if (firstCard.color === CardColor.WILD) {
      room.currentColor = null;
      room.currentWildColor = this.selectRandomColor();
    } else {
      room.currentColor = firstCard.color;
    }

    // 随机选择起始玩家
    room.currentPlayerIndex = Math.floor(Math.random() * room.players.length);

    room.status = 'playing';
    room.startedAt = Date.now();

    return room;
  }

  // 玩家出牌
  playCard(roomId: string, playerId: string, cardId: string, chosenColor?: CardColor): {
    success: boolean;
    error?: string;
    nextPlayerIndex?: number;
  } {
    const room = this.rooms.get(roomId);
    if (!room) {
      return { success: false, error: 'Room not found' };
    }

    const player = room.players.find(p => p.id === playerId);
    if (!player) {
      return { success: false, error: 'Player not found' };
    }

    // 检查是否轮到该玩家
    const currentPlayer = room.players[room.currentPlayerIndex];
    if (currentPlayer.id !== playerId) {
      return { success: false, error: 'Not your turn' };
    }

    // 找到要出的卡牌
    const cardIndex = player.cards.findIndex(c => c.id === cardId);
    if (cardIndex === -1) {
      return { success: false, error: 'Card not found' };
    }

    const card = player.cards[cardIndex];
    const topCard = room.discardPile[room.discardPile.length - 1];

    // 检查出牌合法性
    if (!this.isValidMove(card, topCard, room.currentColor)) {
      return { success: false, error: 'Invalid move' };
    }

    // 移除卡牌
    player.cards.splice(cardIndex, 1);

    // 添加到弃牌堆
    room.discardPile.push(card);

    // 处理万能牌颜色选择
    if (card.color === CardColor.WILD) {
      room.currentWildColor = chosenColor || this.selectRandomColor();
    } else {
      room.currentColor = card.color;
    }

    // 检查是否获胜
    if (player.cards.length === 0) {
      return this.handleRoundEnd(roomId);
    }

    // 处理特殊卡牌效果
    let nextIndex = room.currentPlayerIndex;
    switch (card.type) {
      case CardType.SKIP:
        nextIndex = this.getNextPlayerIndex(room, 2); // 跳过一个人
        break;
      case CardType.REVERSE:
        room.direction *= -1;
        nextIndex = room.players.length > 2
          ? this.getNextPlayerIndex(room, 1)
          : this.getNextPlayerIndex(room, 2);
        break;
      case CardType.DRAW_TWO:
        this.drawCards(room, this.getNextPlayerIndex(room, 1), 2);
        nextIndex = this.getNextPlayerIndex(room, 2);
        break;
      case CardType.WILD_DRAW_FOUR:
        this.drawCards(room, this.getNextPlayerIndex(room, 1), 4);
        nextIndex = this.getNextPlayerIndex(room, 2);
        break;
      default:
        nextIndex = this.getNextPlayerIndex(room, 1);
    }

    room.currentPlayerIndex = nextIndex;

    return { success: true, nextPlayerIndex: nextIndex };
  }

  // 检查出牌是否合法
  private isValidMove(card: Card, topCard: Card, currentColor: CardColor | null): boolean {
    // 万能牌总是可以出
    if (card.color === CardColor.WILD) {
      return true;
    }

    // 颜色匹配
    if (card.color === currentColor) {
      return true;
    }

    // 类型匹配
    if (card.type === topCard.type) {
      return true;
    }

    return false;
  }

  // 获取下一个玩家索引
  private getNextPlayerIndex(room: GameState, steps: number): number {
    const count = room.players.length;
    let index = room.currentPlayerIndex;

    for (let i = 0; i < steps; i++) {
      index = (index + room.direction + count) % count;
    }

    return index;
  }

  // 抽牌
  drawCard(roomId: string, playerId: string): Card | null {
    const room = this.rooms.get(roomId);
    if (!room) return null;

    const currentPlayer = room.players[room.currentPlayerIndex];
    if (currentPlayer.id !== playerId) return null;

    // 牌堆空了则重新洗牌
    if (room.drawPile.length === 0) {
      this.reshuffleDeck(room);
    }

    const card = room.drawPile.pop()!;
    currentPlayer.cards.push(card);

    return card;
  }

  // 重新洗牌
  private reshuffleDeck(room: GameState): void {
    if (room.discardPile.length <= 1) return;

    const topCard = room.discardPile.pop()!;
    const otherCards = room.discardPile;

    // 保留当前颜色
    const currentColor = room.currentColor;
    const currentWildColor = room.currentWildColor;

    // 洗牌
    room.drawPile = this.shuffleDeck(otherCards);
    room.discardPile = [topCard];

    // 恢复颜色设置
    room.currentColor = currentColor;
    room.currentWildColor = currentWildColor;
  }

  // 处理回合结束
  private handleRoundEnd(roomId: string): { success: boolean; nextPlayerIndex?: number } {
    const room = this.rooms.get(roomId);
    if (!room) {
      return { success: false };
    }

    // 计算得分
    const winner = room.players.find(p => p.cards.length === 0)!;

    for (const player of room.players) {
      let score = 0;
      for (const card of player.cards) {
        if (card.type === CardType.NUMBER) {
          score += card.value || 0;
        } else if (card.color === CardColor.WILD) {
          score += 50;
        } else {
          score += 20;
        }
      }
      room.scores[winner.id] = (room.scores[winner.id] || 0) + score;
    }

    // 检查是否达到胜利分数
    if (Object.values(room.scores).some(s => s >= room.settings.winScore)) {
      room.status = 'game_end';
    } else {
      room.status = 'round_end';
    }

    return { success: true };
  }

  // 选择随机颜色
  private selectRandomColor(): CardColor {
    const colors = [CardColor.RED, CardColor.GREEN, CardColor.BLUE, CardColor.YELLOW];
    return colors[Math.floor(Math.random() * colors.length)];
  }
}
```

### 5.2 AI 服务实现

**文件路径**: `backend/src/game/ai.service.ts`

**核心实现逻辑**:

```typescript
import { Injectable } from '@nestjs/common';
import { Card, CardColor, CardType, Player, GameState } from './types';

type AIDifficulty = 'easy' | 'medium' | 'hard';

@Injectable()
export class AiService {
  // AI 决策入口
  makeDecision(
    aiPlayer: Player,
    gameState: GameState,
    difficulty: AIDifficulty
  ): { action: 'play' | 'draw'; cardId?: string; color?: CardColor } {
    const playableCards = this.getPlayableCards(aiPlayer, gameState);

    switch (difficulty) {
      case 'easy':
        return this.easyAI(aiPlayer, gameState, playableCards);
      case 'medium':
        return this.mediumAI(aiPlayer, gameState, playableCards);
      case 'hard':
        return this.hardAI(aiPlayer, gameState, playableCards);
    }
  }

  // 简单难度 - 随机选择
  private easyAI(
    player: Player,
    gameState: GameState,
    playableCards: Card[]
  ): { action: 'play' | 'draw'; cardId?: string; color?: CardColor } {
    if (playableCards.length === 0) {
      return { action: 'draw' };
    }

    // 随机选择一张可出的牌
    const card = playableCards[Math.floor(Math.random() * playableCards.length)];
    const color = this.selectRandomColor();

    return { action: 'play', cardId: card.id, color };
  }

  // 中等难度 - 策略选择
  private mediumAI(
    player: Player,
    gameState: GameState,
    playableCards: Card[]
  ): { action: 'play' | 'draw'; cardId?: string; color?: CardColor } {
    if (playableCards.length === 0) {
      return { action: 'draw' };
    }

    // 优先出普通牌，保留万能牌
    const normalCards = playableCards.filter(c => c.color !== CardColor.WILD);
    const wildCards = playableCards.filter(c => c.color === CardColor.WILD);

    // 统计手牌中各颜色数量
    const colorCounts = this.countColors(player.cards);

    // 优先出数量最多的颜色的牌
    const maxColor = Object.entries(colorCounts)
      .filter(([color]) => color !== CardColor.WILD)
      .sort((a, b) => b[1] - a[1])[0];

    // 尝试找出符合优先颜色的牌
    if (maxColor) {
      const priorityCard = normalCards.find(c => c.color === maxColor[0]);
      if (priorityCard) {
        return { action: 'play', cardId: priorityCard.id, color: maxColor[0] as CardColor };
      }
    }

    // 如果没有优先牌，出第一张普通牌
    if (normalCards.length > 0) {
      return { action: 'play', cardId: normalCards[0].id, color: normalCards[0].color };
    }

    // 只有万能牌
    return {
      action: 'play',
      cardId: wildCards[0].id,
      color: this.selectMostCommonColor(player.cards)
    };
  }

  // 困难难度 - 高级策略
  private hardAI(
    player: Player,
    gameState: GameState,
    playableCards: Card[]
  ): { action: 'play' | 'draw'; cardId?: string; color?: CardColor } {
    if (playableCards.length === 0) {
      return { action: 'draw' };
    }

    // 统计手牌
    const colorCounts = this.countColors(player.cards);
    const cardCount = player.cards.length;

    // 如果只剩1-3张牌，优先出功能牌（攻击性策略）
    if (cardCount <= 3) {
      const actionCards = playableCards.filter(c =>
        c.type === CardType.SKIP ||
        c.type === CardType.REVERSE ||
        c.type === CardType.DRAW_TWO ||
        c.type === CardType.WILD_DRAW_FOUR
      );

      if (actionCards.length > 0) {
        return {
          action: 'play',
          cardId: actionCards[0].id,
          color: this.selectMostCommonColor(player.cards)
        };
      }
    }

    // 优先清理数量最多的颜色
    const maxColor = Object.entries(colorCounts)
      .filter(([color]) => color !== CardColor.WILD)
      .sort((a, b) => b[1] - a[1])[0];

    if (maxColor && maxColor[1] > 1) {
      const priorityCard = playableCards.find(c => c.color === maxColor[0]);
      if (priorityCard) {
        return { action: 'play', cardId: priorityCard.id, color: maxColor[0] as CardColor };
      }
    }

    // 保留万能牌到最后
    const normalCards = playableCards.filter(c => c.color !== CardColor.WILD);
    if (normalCards.length > 0) {
      return {
        action: 'play',
        cardId: normalCards[0].id,
        color: normalCards[0].color || this.selectMostCommonColor(player.cards)
      };
    }

    // 不得不使用万能牌
    return {
      action: 'play',
      cardId: playableCards[0].id,
      color: this.selectMostCommonColor(player.cards)
    };
  }

  // 获取可出的卡牌
  private getPlayableCards(player: Player, gameState: GameState): Card[] {
    const topCard = gameState.discardPile[gameState.discardPile.length - 1];
    const currentColor = gameState.currentWildColor || gameState.currentColor;

    return player.cards.filter(card => {
      // 万能牌总是可以出
      if (card.color === CardColor.WILD) return true;

      // 颜色匹配
      if (card.color === currentColor) return true;

      // 类型匹配
      if (topCard && card.type === topCard.type) return true;

      return false;
    });
  }

  // 统计颜色数量
  private countColors(cards: Card[]): Record<string, number> {
    const counts: Record<string, number> = {};

    for (const card of cards) {
      counts[card.color] = (counts[card.color] || 0) + 1;
    }

    return counts;
  }

  // 选择最常见的颜色
  private selectMostCommonColor(cards: Card[]): CardColor {
    const counts = this.countColors(cards);

    const sorted = Object.entries(counts)
      .filter(([color]) => color !== CardColor.WILD)
      .sort((a, b) => b[1] - a[1]);

    if (sorted.length > 0) {
      return sorted[0][0] as CardColor;
    }

    return this.selectRandomColor();
  }

  // 随机选择颜色
  private selectRandomColor(): CardColor {
    const colors = [CardColor.RED, CardColor.GREEN, CardColor.BLUE, CardColor.YELLOW];
    return colors[Math.floor(Math.random() * colors.length)];
  }
}
```

**技术难点**:
1. **策略算法**: 不同难度需要不同的决策算法，需要反复测试平衡性
2. **状态评估**: 困难难度需要评估全局状态，计算最优解

### 6.1 AudioManager 音频管理器

**文件路径**: `frontend/src/utils/audioManager.ts`

**核心实现逻辑**:

```typescript
import { Howl, Howler } from 'howler';

type SoundType =
  | 'cardPlay'
  | 'cardDraw'
  | 'unoShout'
  | 'unoCatch'
  | 'skip'
  | 'reverse'
  | 'drawTwo'
  | 'wild'
  | 'win'
  | 'lose'
  | 'buttonClick';

class AudioManager {
  private sounds: Map<SoundType, Howl> = new Map();
  private masterVolume: number = 0.7;
  private initialized: boolean = false;

  // 初始化音频资源
  async init(): Promise<void> {
    if (this.initialized) return;

    const soundFiles: Record<SoundType, string> = {
      cardPlay: '/sounds/card-play.mp3',
      cardDraw: '/sounds/card-draw.mp3',
      unoShout: '/sounds/uno-shout.mp3',
      unoCatch: '/sounds/uno-catch.mp3',
      skip: '/sounds/skip.mp3',
      reverse: '/sounds/reverse.mp3',
      drawTwo: '/sounds/draw-two.mp3',
      wild: '/sounds/wild.mp3',
      win: '/sounds/win.mp3',
      lose: '/sounds/lose.mp3',
      buttonClick: '/sounds/button-click.mp3',
    };

    // 预加载所有音效
    const loadPromises = Object.entries(soundFiles).map(([type, src]) => {
      return new Promise<void>((resolve) => {
        const sound = new Howl({
          src: [src],
          volume: this.masterVolume,
          onload: () => {
            this.sounds.set(type as SoundType, sound);
            resolve();
          },
          onloaderror: () => {
            console.warn(`Failed to load sound: ${src}`);
            resolve(); // 不阻塞加载
          },
        });
      });
    });

    await Promise.all(loadPromises);
    this.initialized = true;
  }

  // 播放音效
  play(soundType: SoundType): void {
    const sound = this.sounds.get(soundType);
    if (!sound) {
      console.warn(`Sound not found: ${soundType}`);
      return;
    }

    // 如果正在播放，重新播放
    if (sound.playing()) {
      sound.stop();
    }

    sound.play();
  }

  // 设置音量
  setVolume(volume: number): void {
    this.masterVolume = Math.max(0, Math.min(1, volume));
    Howler.volume(this.masterVolume);
  }

  // 静音
  mute(muted: boolean): void {
    Howler.mute(muted);
  }

  // 停止所有音效
  stopAll(): void {
    Howler.stop();
  }
}

// 单例实例
export const audioManager = new AudioManager();
```

## 第七部分：技术难点总结

### 7.1 主要技术挑战

| 挑战 | 描述 | 解决方案 |
|------|------|----------|
| **3D 性能优化** | 同时渲染大量 3D 对象导致帧率下降 | 使用 `useMemo` 缓存计算结果，使用 `BufferGeometry` 优化粒子系统 |
| **响应式适配** | 不同设备需要不同的布局和相机参数 | 建立设备检测和配置系统，动态调整 |
| **状态同步** | 服务器和客户端状态需要保持一致 | 使用 Socket.io 实时同步，建立状态机 |
| **AI 决策** | 需要实现不同难度的游戏 AI | 分层策略设计，简单/中等/困难三种算法 |
| **卡牌动画** | 需要流畅的卡牌移动和展开效果 | 使用 React Spring + Framer Motion |
| **音效管理** | 需要预加载和及时播放音效 | 使用 Howler.js 封装音频管理器 |
| **游戏规则** | UNO 完整规则实现复杂 | 建立状态机处理各种游戏场景 |
