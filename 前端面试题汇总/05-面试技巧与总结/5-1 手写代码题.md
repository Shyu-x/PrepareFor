## 一、异步与 Promise 进阶

### 1.1 带并发限制的异步调度器 (Scheduler)
**题目**：实现一个 `Scheduler` 类，使同时运行的任务不超过 `limit` 个。
**考察点**：队列控制、Promise 状态流转。

```javascript
class Scheduler {
  constructor(limit = 2) {
    this.limit = limit;
    this.running = 0;
    this.queue = [];
  }

  async add(promiseCreator) {
    // 如果当前正在运行的任务数已满，则阻塞在此处，等待 resolve
    if (this.running >= this.limit) {
      await new Promise(resolve => this.queue.push(resolve));
    }
    
    this.running++;
    try {
      return await promiseCreator();
    } finally {
      this.running--;
      // 任务完成后，从队列拿出一个 resolve 执行，通知下一个 add 继续
      if (this.queue.length > 0) {
        this.queue.shift()();
      }
    }
  }
}

// 测试
const timeout = (ms) => new Promise(resolve => setTimeout(resolve, ms));
const scheduler = new Scheduler(2);
const addTask = (time, order) => {
  scheduler.add(() => timeout(time)).then(() => console.log(order));
};
addTask(1000, '1'); addTask(500, '2'); addTask(300, '3'); addTask(400, '4');
// 输出: 2, 3, 1, 4 (1和2先跑，2跑完后3进，3跑完后4进，1最后跑完)
```

### 2.1 工业级深拷贝 (Deep Clone)
**要求**：处理循环引用、Date、RegExp、Symbol、不可枚举属性。

```javascript
function deepClone(obj, cache = new WeakMap()) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (obj instanceof Date) return new Date(obj);
  if (obj instanceof RegExp) return new RegExp(obj);
  
  // 处理循环引用
  if (cache.has(obj)) return cache.get(obj);
  
  // 获取所有属性，包括 Symbol
  const allKeys = Reflect.ownKeys(obj);
  const cloneObj = Array.isArray(obj) ? [] : {};
  
  cache.set(obj, cloneObj);
  
  allKeys.forEach(key => {
    cloneObj[key] = deepClone(obj[key], cache);
  });
  
  return cloneObj;
}
```

## 三、函数式编程

### 3.2 增强版节流 (Throttle)
**要求**：支持 `leading` (立即执行) 和 `trailing` (结束后执行)。

```javascript
function throttle(fn, delay, options = {}) {
  let timer = null;
  let last = 0;
  const { leading = true, trailing = false } = options;

  const throttled = function(...args) {
    const now = Date.now();

    if (!last && !leading) last = now;

    if (now - last >= delay) {
      if (timer) {
        clearTimeout(timer);
        timer = null;
      }
      fn.apply(this, args);
      last = now;
    } else if (trailing && !timer) {
      timer = setTimeout(() => {
        last = leading ? Date.now() : 0;
        timer = null;
        fn.apply(this, args);
      }, delay - (now - last));
    }
  };

  throttled.cancel = () => {
    clearTimeout(timer);
    timer = null;
    last = 0;
  };

  return throttled;
}
```

```javascript
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function(...args2) {
      return curried.apply(this, args.concat(args2));
    };
  };
```

## 五、事件总线 (Event Bus)

### 5.1 基础事件总线
**要求**：实现发布订阅模式，支持事件监听、触发、移除。

```javascript
class EventBus {
  constructor() {
    this.events = {};
  }

  on(event, callback) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    this.events[event].push(callback);
    // 返回取消订阅函数
    return () => this.off(event, callback);
  }

  off(event, callback) {
    if (!this.events[event]) return;
    this.events[event] = this.events[event].filter(cb => cb !== callback);
  }

  emit(event, ...args) {
    if (!this.events[event]) return;
    this.events[event].forEach(cb => cb(...args));
  }

  once(event, callback) {
    const wrapper = (...args) => {
      callback(...args);
      this.off(event, wrapper);
    };
    this.on(event, wrapper);
  }
}
```

## 六、LRU 缓存

### 6.1 实现 LRU 缓存
**要求**：实现一个最近最少使用缓存，支持 `get` 和 `put` 操作。

```javascript
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();
  }

  get(key) {
    if (!this.cache.has(key)) return -1;

    // 移到末尾（最近使用）
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);
    return value;
  }

  put(key, value) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else if (this.cache.size >= this.capacity) {
      // 删除最旧的（第一个）
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    this.cache.set(key, value);
  }
}
```

```javascript
class LazyMan {
  constructor(name) {
    this.tasks = [];

    const task = this.tasks.shift();

  }
```

### 8.1 JSONP 封装
**要求**：实现 `jsonp` 函数，支持 Promise 风格调用。

```javascript
function jsonp(url, params = {}) {
  return new Promise((resolve, reject) => {
    const callbackName = `jsonp_${Date.now()}`;
    const script = document.createElement('script');

    // 构建URL
    const queryString = Object.entries(params)
      .map(([key, value]) => `${key}=${value}`)
      .join('&');
    const fullUrl = `${url}?${queryString}&callback=${callbackName}`;

    // 全局回调
    window[callbackName] = (data) => {
      resolve(data);
      cleanup();
    };

    script.onerror = (err) => {
      reject(err);
      cleanup();
    };

    script.src = fullUrl;
    document.body.appendChild(script);

    function cleanup() {
      delete window[callbackName];
      script.remove();
    }
  });
}
```

```javascript
// createStore
function createStore(reducer) {
  let state;
  let listeners = [];

  const getState = () => state;

  const dispatch = (action) => {
    state = reducer(state, action);
    listeners.forEach(listener => listener());
  };

  const subscribe = (listener) => {
    listeners.push(listener);
    return () => {
      listeners = listeners.filter(l => l !== listener);
    };
  };

  dispatch({ type: '@@INIT' });

  return { getState, dispatch, subscribe };
}

// combineReducers
function combineReducers(reducers) {
  return (state = {}, action) => {
    const nextState = {};
    for (const key in reducers) {
      nextState[key] = reducers[key](state[key], action);
    }
    return nextState;
  };
}
```

```javascript
function useWindowDrag(initialPosition) {
  const [position, setPosition] = useState(initialPosition);
  const [isDragging, setIsDragging] = useState(false);
  const dragStartRef = useRef({ x: 0, y: 0 });

      setPosition({ x: constrainedX, y: constrainedY });

      setIsDragging(false);
    };

    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
```

### 10.2 节流 Scheduler 改进版
**题目**：实现带优先级的任务调度器。

```javascript
class PriorityScheduler {
  constructor(limit = 2) {
    this.limit = limit;
    this.running = 0;
    this.queue = [];
  }

  async add(promiseCreator, priority = 0) {
    if (this.running >= this.limit) {
      await new Promise(resolve => {
        this.queue.push({ resolve, priority, promiseCreator });
        // 按优先级排序
        this.queue.sort((a, b) => b.priority - a.priority);
      });
    }

    this.running++;
    try {
      return await promiseCreator();
    } finally {
      this.running--;
      if (this.queue.length > 0) {
        const { resolve, promiseCreator } = this.queue.shift();
        resolve(promiseCreator());
      }
    }
  }
}
```

### 10.4 异步并发控制器
**题目**：实现可控制并发数的异步任务池。

```javascript
class TaskPool {
  constructor(concurrency) {
    this.concurrency = concurrency;
    this.running = 0;
    this.queue = [];
  }

  addTask(task) {
    return new Promise((resolve, reject) => {
      this.queue.push({ task, resolve, reject });
      this.process();
    });
  }

  async process() {
    while (this.running < this.concurrency && this.queue.length > 0) {
      const { task, resolve, reject } = this.queue.shift();
      this.running++;

      try {
        const result = await task();
        resolve(result);
      } catch (err) {
        reject(err);
      } finally {
        this.running--;
        this.process();
      }
    }
  }
}

const pool = new TaskPool(3);
```

```javascript
function myInstanceof(left, right) {
  if (typeof left !== 'object' || left === null) return false;

  let proto = Object.getPrototypeOf(left);
  const prototype = right.prototype;

  while (proto) {
    if (proto === prototype) return true;
    proto = Object.getPrototypeOf(proto);
  }

  return false;
}
```

### 13.3 手写 Promise
**题目**：实现一个简化的 Promise。

```javascript
class MyPromise {
  constructor(executor) {
    this.status = 'pending';
    this.value = undefined;
    this.reason = undefined;
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];

    const resolve = (value) => {
      if (this.status === 'pending') {
        this.status = 'fulfilled';
        this.value = value;
        this.onFulfilledCallbacks.forEach(cb => cb(this.value));
      }
    };

    const reject = (reason) => {
      if (this.status === 'pending') {
        this.status = 'rejected';
        this.reason = reason;
        this.onRejectedCallbacks.forEach(cb => cb(this.reason));
      }
    };

    try {
      executor(resolve, reject);
    } catch (err) {
      reject(err);
    }
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      if (this.status === 'fulfilled') {
        try {
          const result = onFulfilled(this.value);
          resolve(result);
        } catch (err) {
          reject(err);
        }
      } else if (this.status === 'rejected') {
        try {
          const result = onRejected(this.reason);
          resolve(result);
        } catch (err) {
          reject(err);
        }
      } else {
        this.onFulfilledCallbacks.push((value) => {
          try {
            const result = onFulfilled(value);
            resolve(result);
          } catch (err) {
            reject(err);
          }
        });
        this.onRejectedCallbacks.push((reason) => {
          try {
            const result = onRejected(reason);
            resolve(result);
          } catch (err) {
            reject(err);
          }
        });
      }
    });
  }

  catch(onRejected) {
    return this.then(null, onRejected);
  }

  finally(onFinally) {
    return this.then(
      (value) => {
        onFinally();
        return value;
      },
      (reason) => {
        onFinally();
        throw reason;
      }
    );
  }

  static resolve(value) {
    if (value instanceof MyPromise) {
      return value;
    }
    return new MyPromise(resolve => resolve(value));
  }

  static reject(reason) {
    return new MyPromise((_, reject) => reject(reason));
  }

  static all(promises) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(promises)) {
        return reject(new TypeError('Promise.all requires an array'));
      }

      const results = [];
      let count = 0;

      if (promises.length === 0) {
        return resolve(results);
      }

      promises.forEach((promise, index) => {
        MyPromise.resolve(promise).then(
          value => {
            results[index] = value;
            count++;
            if (count === promises.length) {
              resolve(results);
            }
          },
          reject
        );
      });
    });
  }

  static race(promises) {
    return new MyPromise((resolve, reject) => {
      promises.forEach(promise => {
        MyPromise.resolve(promise).then(resolve, reject);
      });
    });
  }
}
```

```javascript
function bubbleSort(arr) {
  const n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];

    }
  }
```

### 14.2 快速排序

```javascript
function quickSort(arr) {
  if (arr.length <= 1) return arr;

  const pivotIndex = Math.floor(arr.length / 2);
  const pivot = arr.splice(pivotIndex, 1)[0];

  const left = [];
  const right = [];

  for (let i = 0; i < arr.length; i++) {
    if (arr[i] < pivot) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }

  return [...quickSort(left), pivot, ...quickSort(right)];
}
```

```javascript
function binarySearch(arr, target) {
  let left = 0;
  let right = arr.length - 1;

  while (left <= right) {
    const mid = Math.floor((left + right) / 2);

    if (arr[mid] === target) {
      return mid;
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;

}
```

```javascript
          settledCount++;
          if (settledCount === promises.length) {

          }
        }
      );
```

### 16.2 手写深比较

```javascript
function deepEqual(a, b, visited = new WeakMap()) {
  // 相同引用
  if (a === b) return true;

  // 排除 null 和 undefined
  if (a == null || b == null) return false;

  // 类型不同
  if (typeof a !== typeof b) return false;

  // 基本类型
  if (typeof a !== 'object') return a === b;

  // 循环引用检测
  if (visited.has(a)) return visited.get(a) === b;
  visited.set(a, b);

  // 数组
  if (Array.isArray(a) !== Array.isArray(b)) return false;

  if (Array.isArray(a)) {
    if (a.length !== b.length) return false;
    return a.every((item, i) => deepEqual(item, b[i], visited));
  }

  // 对象
  const keysA = Object.keys(a);
  const keysB = Object.keys(b);

  if (keysA.length !== keysB.length) return false;

  return keysA.every(key => deepEqual(a[key], b[key], visited));
}
```

```javascript
  if (!depsMap) {
    depsMap = new Map();
    targetMap.set(target, depsMap);
  }

  if (!effects) {
    effects = new Set();
    depsMap.set(key, effects);

  effects.add(currentEffect);
```

### 13.1 实现 LRU 缓存

```javascript
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();
  }

  get(key) {
    if (!this.cache.has(key)) return -1;

    // 将key移到末尾（最新使用）
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);
    return value;
  }

  put(key, value) {
    // 如果key已存在，删除
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }
    // 如果容量已满，删除最旧的（第一个）
    else if (this.cache.size >= this.capacity) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    this.cache.set(key, value);
  }
}
```

```javascript
class EventEmitter {
  constructor() {
    this.events = {};
  }

  on(event, callback) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    this.events[event].push(callback);
    return () => this.off(event, callback);
  }

  once(event, callback) {
    const wrapper = (...args) => {
      callback(...args);
      this.off(event, wrapper);
    };
    return this.on(event, wrapper);
  }

  off(event, callback) {
    if (!this.events[event]) return;
    this.events[event] = this.events[event].filter(cb => cb !== callback);
  }

  emit(event, ...args) {
    if (!this.events[event]) return;
    this.events[event].forEach(cb => cb(...args));
  }

  clear(event) {
    if (event) {
      delete this.events[event];
    } else {
      this.events = {};
    }
  }
}
```

### 14.1 二分查找
**题目**：实现二分查找算法。

```javascript
// 基础版
function binarySearch(arr, target) {
  let left = 0;
  let right = arr.length - 1;

  while (left <= right) {
    const mid = Math.floor((left + right) / 2);

    if (arr[mid] === target) {
      return mid;
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  return -1;
}

// 查找左边界
function lowerBound(arr, target) {
  let left = 0;
  let right = arr.length;

  while (left < right) {
    const mid = Math.floor((left + right) / 2);
    if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid;
    }
  }

  return left;
}

// 查找右边界
function upperBound(arr, target) {
  let left = 0;
  let right = arr.length;

  while (left < right) {
    const mid = Math.floor((left + right) / 2);
```

```javascript
  const pivot = arr[right];

      [arr[i], arr[j]] = [arr[j], arr[i]];
      i++;
    }

  [arr[i], arr[right]] = [arr[right], arr[i]];
  return i;

        [arr[i], arr[j]] = [arr[j], arr[i]];
        i++;
      }
    }

    [arr[i], arr[high]] = [arr[high], arr[i]];
    sort(low, i - 1);
```

```javascript
  const result = [];
  let i = 0, j = 0;

  while (i < left.length && j < right.length) {
    if (left[i] <= right[j]) {
      result.push(left[i++]);
    } else {
      result.push(right[j++]);
    }
  }

      i++;
    } else {
      const temp = arr[j];
      // 移动元素
      for (let k = j; k > i; k--) {
        arr[k] = arr[k - 1];
      }
      arr[i] = temp;
      i++;
      mid++;
      j++;
    }
```

```javascript
function heapSort(arr) {
  const n = arr.length;

  // 构建最大堆
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(arr, n, i);
  }

  // 排序
  for (let i = n - 1; i > 0; i--) {
    [arr[0], arr[i]] = [arr[i], arr[0]];
    heapify(arr, i, 0);
  }

  return arr;
}

function heapify(arr, n, i) {
  let largest = i;
  const left = 2 * i + 1;
  const right = 2 * i + 2;

  if (left < n && arr[left] > arr[largest]) {
    largest = left;
  }

  if (right < n && arr[right] > arr[largest]) {
    largest = right;
  }

  if (largest !== i) {
    [arr[i], arr[largest]] = [arr[largest], arr[i]];
    heapify(arr, n, largest);
  }
}

// 最小堆
class MinHeap {
  constructor() {
    this.heap = [];
  }

  push(val) {
    this.heap.push(val);
    this.bubbleUp(this.heap.length - 1);
  }

  pop() {
    if (this.heap.length === 0) return null;
    if (this.heap.length === 1) return this.heap.pop();

    const min = this.heap[0];
    this.heap[0] = this.heap.pop();
    this.bubbleDown(0);
    return min;
  }

  bubbleUp(index) {
    while (index > 0) {
      const parentIndex = Math.floor((index - 1) / 2);
      if (this.heap[parentIndex] <= this.heap[index]) break;
      [this.heap[parentIndex], this.heap[index]] = [this.heap[index], this.heap[parentIndex]];
      index = parentIndex;
    }
  }

  bubbleDown(index) {
    const n = this.heap.length;
    while (true) {
      const left = 2 * index + 1;
      const right = 2 * index + 2;
      let smallest = index;

      if (left < n && this.heap[left] < this.heap[smallest]) {
        smallest = left;
      }
      if (right < n && this.heap[right] < this.heap[smallest]) {
        smallest = right;
      }

      if (smallest === index) break;
      [this.heap[index], this.heap[smallest]] = [this.heap[smallest], this.heap[index]];
      index = smallest;
    }
  }
}
```

```javascript
function knapSack(weights, values, capacity) {
  const n = weights.length;
  const dp = Array(n + 1).fill(null).map(() => Array(capacity + 1).fill(0));

  for (let i = 1; i <= n; i++) {
    for (let w = 0; w <= capacity; w++) {
      if (weights[i - 1] <= w) {
        dp[i][w] = Math.max(
          dp[i - 1][w],
          dp[i - 1][w - weights[i - 1]] + values[i - 1]

      }
    }
  }

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
```

```javascript
// 邻接表表示的图
function dfs(graph, start) {
  const visited = new Set();
  const result = [];

  function traverse(node) {
    if (visited.has(node)) return;

    visited.add(node);
    result.push(node);

    for (const neighbor of graph[node] || []) {
      traverse(neighbor);
    }
  }

  traverse(start);
  return result;
}

// 迭代版
function dfsIterative(graph, start) {
  const visited = new Set();
  const result = [];
  const stack = [start];

  while (stack.length) {
    const node = stack.pop();

    if (visited.has(node)) continue;

    visited.add(node);
    result.push(node);

    const neighbors = graph[node] || [];
    for (let i = neighbors.length - 1; i >= 0; i--) {
      if (!visited.has(neighbors[i])) {
        stack.push(neighbors[i]);
      }
    }
  }

  return result;
}

// 岛屿数量（经典DFS）
function numIslands(grid) {
  if (!grid || !grid.length) return 0;

  let count = 0;
  const rows = grid.length;
  const cols = grid[0].length;

  function dfs(r, c) {
    if (r < 0 || r >= rows || c < 0 || c >= cols || grid[r][c] === '0') {
      return;
    }

    grid[r][c] = '0'; // 标记已访问

    dfs(r + 1, c);
    dfs(r - 1, c);
    dfs(r, c + 1);
    dfs(r, c - 1);
  }

  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      if (grid[r][c] === '1') {
        count++;
        dfs(r, c);
      }
    }
  }

  return count;
}
```

```javascript
  let listeners = [];

    return state;
```

### 15.5 实现深比较函数
**题目**：实现类似 lodash 的 deepEqual。

```javascript
function deepEqual(a, b) {
  // 相同引用
  if (a === b) return true;

  // 类型检查
  if (typeof a !== 'object' || typeof b !== 'object' || a === null || b === null) {
    return a !== a && b !== b; // 处理 NaN
  }

  // 数组检查
  const isArrayA = Array.isArray(a);
  const isArrayB = Array.isArray(b);

  if (isArrayA !== isArrayB) return false;

  // 键数量检查
  const keysA = Object.keys(a);
  const keysB = Object.keys(b);

  if (keysA.length !== keysB.length) return false;

  // 递归比较
  for (const key of keysA) {
    if (!keysB.includes(key)) return false;
    if (!deepEqual(a[key], b[key])) return false;
  }

  return true;
}

// 完整版（包含 Date, RegExp, Error 等）
function isEqual(a, b) {
  // 基础类型
  if (a === b) return true;
  if (a === null || b === null) return a === b;
  if (typeof a !== 'object' || typeof b !== 'object') return a === a && b === b;

  // Date
  if (a instanceof Date && b instanceof Date) {
    return a.getTime() === b.getTime();
  }

  // RegExp
  if (a instanceof RegExp && b instanceof RegExp) {
    return a.toString() === b.toString();
  }

  // Array
  if (Array.isArray(a) !== Array.isArray(b)) return false;

  // Object keys
  const keysA = Object.keys(a);
  const keysB = Object.keys(b);

  if (keysA.length !== keysB.length) return false;

  return keysA.every(key => isEqual(a[key], b[key]));
}
```

```javascript
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function (...args2) {
      return curried.apply(this, args.concat(args2));
    };
  };
}

// 箭头函数版
const curry = fn => {
  const arity = fn.length;
  return function curried(...args) {
    return args.length >= arity
      ? fn(...args)
      : (...more) => curried(...args, ...more);
  };
};

// 高级版：支持占位符
function curryWithPlaceholder(fn) {
  const Placeholder = Symbol('placeholder');

  return function curried(...args) {
    const newArgs = args.map(arg =>
      arg === Placeholder ? arguments[curried.position++] : arg
    );

    if (newArgs.filter(a => a !== Placeholder).length >= fn.length) {
      return fn.apply(this, newArgs);
    }

    function next(...moreArgs) {
      let idx = 0;
      const combined = newArgs.map(arg =>
        arg === Placeholder && idx < moreArgs.length
          ? moreArgs[idx++]
          : arg
      );
      return curried(...combined, ...moreArgs.slice(idx));
    }

    next.position = 0;
    next._ = Placeholder;
    return next;
  };
}
```

```javascript
async function concurrencyLimit(tasks, limit = 3) {
  const results = [];
  const executing = [];

  for (const task of tasks) {
    const p = Promise.resolve(task()).then(result => {
      results.push(result);
    });

    executing.push(p);

    if (executing.length >= limit) {
      await Promise.race(executing);
      executing.splice(
        executing.findIndex(e => e === p),
        1
      );
    }
  }

  return Promise.all(executing).then(() => results);
}

    results.push(...batchResults);

    if (i + batchSize < items.length) {

  }
```

### 15.10 实现消息订阅系统
**题目**：实现 EventBus 或类似 Pub/Sub。

```javascript
class EventBus {
  constructor() {
    this.events = new Map();
    this.middlewares = [];
  }

  on(event, callback, context = null) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }

    const handler = { callback, context };
    this.events.get(event).push(handler);

    // 返回取消订阅函数
    return () => this.off(event, callback);
  }

  once(event, callback, context = null) {
    const wrapper = (...args) => {
      callback.apply(context, args);
      this.off(event, wrapper);
    };
    return this.on(event, wrapper, context);
  }

  off(event, callback) {
    if (!this.events.has(event)) return;

    const handlers = this.events.get(event);
    const index = handlers.findIndex(h => h.callback === callback);

    if (index !== -1) {
      handlers.splice(index, 1);
    }

    if (handlers.length === 0) {
      this.events.delete(event);
    }
  }

  emit(event, ...args) {
    if (!this.events.has(event)) return;

    const handlers = this.events.get(event);

    // 复制一份避免在循环中修改导致的问题
    [...handlers].forEach(({ callback, context }) => {
      try {
        callback.apply(context, args);
      } catch (error) {
        console.error(`Error in event ${event}:`, error);
      }
    });
  }

  // 中间件
  use(middleware) {
    this.middlewares.push(middleware);
  }

  // 异步事件
  emitAsync(event, ...args) {
    if (!this.events.has(event)) return Promise.resolve();

    const handlers = this.events.get(event);

    return Promise.all(
      handlers.map(async ({ callback, context }) => {
        await callback.apply(context, args);
      })
    );
  }

  // 清除所有事件
  clear(event) {
    if (event) {
      this.events.delete(event);
    } else {
      this.events.clear();
    }
  }
}

// 使用

bus.on('user:login', (user) => {
  console.log('User logged in:', user.name);
});

  console.log('Analytics: login event');
}, { context: 'analytics' });
```

## 十七、JavaScript 内置函数手动实现（详细原理篇）

### 17.1 Object 方法实现

#### 17.1.1 实现 Object.create

**题目**：手动实现 Object.create 方法。

**核心原理**：

Object.create 是 ES5 引入的方法，用于创建一个新对象，并指定其原型。它的实现原理基于 JavaScript 的原型链继承机制。

**原型链继承原理**：

在 JavaScript 中，每个对象都有一个内部属性 `[[Prototype]]`（浏览器中通过 `__proto__` 访问），指向它的原型对象。当访问对象的属性时，如果对象本身没有该属性，JavaScript 会沿着原型链向上查找，直到找到或者到达原型链顶端。

```
instance --__proto__--> proto --__proto__--> Object.prototype --__proto__--> null
```

**实现步骤详解**：

1. 创建一个空函数 F，这个函数没有任何属性和方法
2. 将函数 F 的 prototype 属性设置为传入的原型对象 proto
3. 使用 new F() 创建实例，此时实例的 __proto__ 会指向 F.prototype
4. 由于 F.prototype 就是 proto，所以实例的原型链就链接到了 proto

```javascript
function createObject(proto) {
  // 创建一个新函数，将它的原型指向传入的对象
  function F() {}
  F.prototype = proto;
  return new F();
}

// 测试
const obj = { name: 'test' };
const instance = createObject(obj);
console.log(instance.__proto__ === obj); // true
console.log(Object.getPrototypeOf(instance) === obj); // true
```

**为什么不能直接 return { __proto__: proto }？**

虽然现代浏览器支持这种写法，但这种方式创建的 对象特性与 Object.create 不完全一致：
- Object.create(null) 可以创建没有原型的纯净对象
- Object.create 可以通过第二个参数定义属性描述符

**完整版（带属性描述符）**：

```javascript
function myCreate(proto, propertiesObject) {
  // 参数验证：proto 必须是对象或 null
  if (typeof proto !== 'object' && proto !== null) {
    throw new TypeError('Object prototype may only be an Object or null');
  }

  // 创建一个空函数作为构造函数
  function F() {}
  // 将构造函数的 prototype 指向原型对象

  return new F();
}
```

#### 17.1.2 实现 Object.keys / Object.values / Object.entries

**核心原理**：

这三个方法都是遍历对象自身（不含继承）可枚举属性的方法。它们的区别在于返回值的形式不同：
- Object.keys：返回属性名数组
- Object.values：返回属性值数组
- Object.entries：返回 [key, value] 键值对数组

**什么是可枚举属性？**

对象属性有一个 enumerable 特性，默认为 true。通过 Object.defineProperty 可以设置为 false 使其不可枚举。

```javascript
const obj = { a: 1, b: 2, c: 3 };
console.log(Object.keys(obj));   // ['a', 'b', 'c']
console.log(Object.values(obj));  // [1, 2, 3]
console.log(Object.entries(obj)); // [['a', 1], ['b', 2], ['c', 3]]
```

**实现 Object.keys**：

```javascript
// Object.keys - 返回对象自身的可枚举属性名
function objectKeys(obj) {
  // 参数校验：null 和 undefined 需要特殊处理
  if (obj === null || obj === undefined) {
    throw new TypeError('Cannot convert undefined or null to object');
  }

  const result = [];
  // for...in 会遍历对象自身的和继承的可枚举属性
  for (const key in obj) {
    // hasOwnProperty 检查是否是对象自身的属性（不含继承）
    if (obj.hasOwnProperty(key)) {
      result.push(key);
    }
  }
  return result;
}
```

**for...in 的特性**：

1. 遍历对象自身和继承的可枚举属性
2. 遍历数组时包括索引（字符串形式）
3. 遍历顺序：整数键按数值排序，其他按插入顺序

**实现 Object.values**：

```javascript
// Object.values - 返回对象自身的可枚举属性值
function objectValues(obj) {
  if (obj === null || obj === undefined) {
    throw new TypeError('Cannot convert undefined or null to object');
  }

  const result = [];
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      result.push(obj[key]);
    }
  }
  return result;
}
```

**实现 Object.entries**：

```javascript
// Object.entries - 返回对象自身的可枚举属性 [key, value] 对
function objectEntries(obj) {
  if (obj === null || obj === undefined) {
    throw new TypeError('Cannot convert undefined or null to object');
  }

  const result = [];
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      result.push([key, obj[key]]);
    }
  }
  return result;
}
```

**为什么要用 hasOwnProperty？**

因为 for...in 会遍历继承的属性，如果对象继承自其他对象，也会遍历继承来的属性：

```javascript
const parent = { a: 1 };
const child = Object.create(parent);
child.b = 2;

console.log(Object.keys(child)); // ['b'] - 不包含继承的 a
// 如果用 for...in + hasOwnProperty，效果相同
```

#### 17.1.3 实现 Object.freeze / Object.seal

**核心原理**：

这两个方法都用于"锁定"对象，但程度不同：

- Object.freeze：最严格，对象完全不可变
  - 不能添加新属性
  - 不能删除现有属性
  - 不能修改属性值
  - 不能修改属性的描述符

- Object.seal：相对宽松
  - 不能添加新属性
  - 不能删除现有属性
  - 可以修改现有属性的值
  - 可以修改 writable 以外的属性描述符

**冻结原理详解**：

```javascript
// Object.freeze - 冻结对象
function myFreeze(obj) {
  // 基本类型直接返回
  if (typeof obj !== 'object' || obj === null) {
    return obj;
  }

  // 获取对象的所有属性描述符
  const propNames = Object.getOwnPropertyNames(obj);
  const propSymbols = Object.getOwnPropertySymbols(obj);

  // 遍历所有属性
  [...propNames, ...propSymbols].forEach(name => {
    const descriptor = Object.getOwnPropertyDescriptor(obj, name);
    // 如果属性是对象类型，需要递归冻结
    if (descriptor && !descriptor.writable) {
      if (typeof descriptor.value === 'object') {
        myFreeze(descriptor.value);
      }
      // 设置属性为不可写、不可配置
      Object.defineProperty(obj, name, {
        ...descriptor,
        writable: false,
        configurable: false
      });
    }
  });

  // 禁止扩展：阻止添加新属性
  return Object.preventExtensions(obj);
}
```

**preventExtensions 原理**：

Object.preventExtensions() 将对象的 [[Extensible]] 内部属性设置为 false，之后无法向对象添加新属性。但已有的属性仍然可以修改和删除。

**Object.seal 实现**：

```javascript
// Object.seal - 密封对象
function mySeal(obj) {
  if (typeof obj !== 'object' || obj === null) {
    return obj;
  }

  const propNames = Object.getOwnPropertyNames(obj);
  const propSymbols = Object.getOwnPropertySymbols(obj);

  [...propNames, ...propSymbols].forEach(name => {
    const descriptor = Object.getOwnPropertyDescriptor(obj, name);
    // 只能修改 configurable 为 false
    if (descriptor && descriptor.configurable) {
      Object.defineProperty(obj, name, {
        ...descriptor,
        configurable: false
      });
    }
  });

  return Object.preventExtensions(obj);
}
```

**属性描述符详解**：

对象属性实际上由多个特性组成，分为数据描述符和存取描述符：

```javascript
// 数据描述符
{
  value: 123,           // 属性值
  writable: true,       // 是否可写
  enumerable: true,     // 是否可枚举
  configurable: true   // 是否可配置
}

// 存取描述符（ accessor descriptor）
{
  get: function() {},   // getter 函数
  set: function() {},   // setter 函数
  enumerable: true,
  configurable: true
}
```

#### 17.1.4 实现 Object.assign

**核心原理**：

Object.assign 用于将所有可枚举的自有属性从一个或多个源对象复制到目标对象，返回目标对象。

**实现要点**：

1. 第一个参数 target 不能是 null/undefined（需要转换为对象）
2. 后续的 source 对象会按顺序依次合并
3. 后面的 source 会覆盖前面的相同属性
4. Symbol 属性也会被复制

```javascript
// Object.assign - 合并对象
function myAssign(target, ...sources) {
  // 将第一个参数转换为对象（null/undefined 需要特殊处理）
  if (target === null || target === undefined) {
    target = {};
  }

  // 转换为 Object 类型，确保有对象方法可用
  const to = Object(target);

  // 遍历所有源对象
  for (let i = 0; i < sources.length; i++) {
    const source = sources[i];
    // 跳过 null/undefined 源
    if (source !== null && source !== undefined) {
      // 遍历源对象的所有属性
      for (const key in source) {
        // 只复制自有属性
        if (source.hasOwnProperty(key)) {
          to[key] = source[key];
        }
      }
      // Symbol 属性也会被复制
      Object.getOwnPropertySymbols(source).forEach(sym => {
        // 检查是否可枚举
        if (Object.propertyIsEnumerable.call(source, sym)) {
          to[sym] = source[sym];
        }
      });
    }
  }

  return to;
}
```

**为什么要转换 target？**

```javascript
Object.assign(1, { a: 1 }); // Number{1} 被转换为对象
// 结果：{ a: 1 }

// 因为原始类型会被包装为对象
```

**浅拷贝特性**：

Object.assign 只做浅拷贝，如果源对象的属性值是对象引用，只会复制引用：

```javascript
const source = { a: { b: 1 } };
const target = Object.assign({}, source);
target.a.b = 2;
console.log(source.a.b); // 2 - 被修改了
```

#### 17.1.5 实现 Object.is

**核心原理**：

Object.is 用于判断两个值是否"相同"，它的判断规则比 === 更严格：

```javascript
console.log(Object.is(1, 1));           // true - 同值
console.log(Object.is(NaN, NaN));       // true - NaN 被视为相同
console.log(Object.is(0, -0));         // false - 0 和 -0 被视为不同
console.log(Object.is('1', 1));        // false - 类型不同
console.log(Object.is(null, undefined)); // false - null 和 undefined 不同
```

**与 === 的区别**：

1. NaN === NaN 返回 false，但 Object.is(NaN, NaN) 返回 true
2. 0 === -0 返回 true，但 Object.is(0, -0) 返回 false

**实现原理**：

```javascript
// Object.is - 判断两个值是否相同
function myIs(value1, value2) {
  // 相同值的情况
  if (value1 === value2) {
    // 0 和 -0 在严格相等中相等，但在 Object.is 中被认为不同
    // 通过 1/value1 === 1/value2 来区分
    return value1 !== 0 || 1 / value1 === 1 / value2;
  }

  // NaN 和 NaN 相同
  // NaN !== NaN 为 true，所以需要特殊处理
  return value1 !== value1 && value2 !== value2;
}
```

**关键点解释**：

1. `value1 !== 0` 检查 value1 是否为 0（可以是 +0 或 -0）
2. 如果 value1 是 0，那么 `1/value1` 会区分 +0 和 -0：
   - `1 / (+0)` = +Infinity
   - `1 / (-0)` = -Infinity
3. 所以 `1 / value1 === 1 / value2` 只有在两个 0 同号时才为 true

**NaN 判断技巧**：

`value !== value` 是判断 NaN 的经典技巧，因为 NaN 是唯一与自身不相等的值。

#### 17.1.6 实现 Object.getPrototypeOf / Object.setPrototypeOf

**核心原理**：

- Object.getPrototypeOf：返回对象的原型
- Object.setPrototypeOf：设置对象的原型（不推荐使用，会影响性能）

```javascript
function myGetPrototypeOf(obj) {
  // null 和 undefined 直接返回
  if (obj === null) return null;

  // 基本类型返回对应的包装对象的原型
  if (typeof obj !== 'object' && typeof obj !== 'function') {
    return Object.prototype;
  }

  // 优先使用 __proto__
  if (obj.__proto__) {
    return obj.__proto__;
  }

  // 使用 constructor.prototype（更标准的方式）
  if (obj.constructor && obj.constructor.prototype) {
    return obj.constructor.prototype;
  }

  // 最后尝试 Object.getPrototypeOf
  return Object.getPrototypeOf(obj);
}
```

**原型链顶端**：

所有对象的原型链最终都会指向 Object.prototype，除了 Object.create(null) 创建的对象：

```javascript
const obj = {};
console.log(Object.getPrototypeOf(obj) === Object.prototype); // true

const arr = [];
console.log(Object.getPrototypeOf(arr) === Array.prototype); // true
console.log(Object.getPrototypeOf(Array.prototype) === Object.prototype); // true

// 原型链：arr -> Array.prototype -> Object.prototype -> null
```

### 17.2 Array 方法实现

#### 17.2.1 实现 Array.prototype.map

**核心原理**：

map 方法创建一个新数组，其结果是该数组中的每个元素都调用一次提供的函数后的返回值。

**关键特性**：

1. 不改变原数组
2. 返回新数组
3. 回调函数接收三个参数：当前元素、索引、原数组
4. 跳过稀疏数组的空位
5. thisArg 可以指定回调函数的 this

```javascript
// Array.prototype.map - 返回新数组
Array.prototype.myMap = function(callback, thisArg) {
  // 回调函数类型检查
  if (typeof callback !== 'function') {
    throw new TypeError(callback + ' is not a function');
  }

  const result = [];
  const arr = this;
  const len = arr.length;

  for (let i = 0; i < len; i++) {
    // in 操作符会检查索引是否在数组中存在（包括稀疏数组的空位）
    if (i in arr) {
      const value = arr[i];
      const mappedValue = callback.call(thisArg, value, i, arr);
      result[i] = mappedValue;
    }
  }

  return result;
}
```

**为什么要用 i in arr？**

JavaScript 数组是稀疏的，可以有不连续的元素：

```javascript
const arr = [1, , 3]; // arr[1] 是空位
console.log(1 in arr); // false - 索引 1 不存在

// for 循环会跳过空位，for...in 也会跳过
// 但 map 不应该为不存在的索引调用回调
```

**map 的数学定义**：

```
map :: (a -> b) -> [a] -> [b]
```

将一个函数 f 映射到列表 L，得到新列表 [f(a₁), f(a₂), ..., f(aₙ)]

#### 17.2.2 实现 Array.prototype.filter

**核心原理**：

filter 方法创建一个新数组，包含通过所提供函数测试的所有元素。

```javascript
// Array.prototype.filter - 过滤数组
Array.prototype.myFilter = function(callback, thisArg) {
  if (typeof callback !== 'function') {
    throw new TypeError(callback + ' is not a function');
  }

  const result = [];
  const arr = this;
  const len = arr.length;

  for (let i = 0; i < len; i++) {
    if (i in arr) {
      const value = arr[i];
      // 回调返回 true 时保留该元素
      if (callback.call(thisArg, value, i, arr)) {
        result.push(value);
      }
    }
  }

  return result;
}
```

**filter 与 map 的区别**：

- map：转换每个元素，返回新数组（元素数量不变）
- filter：过滤元素，返回可能变短的数组

#### 17.2.3 实现 Array.prototype.reduce

**核心原理**：

reduce 是最强大和复杂的数组方法，它对数组中的每个元素执行一个 reducer 函数，将其结果汇总为单个返回值。

**执行过程**：

```
[1, 2, 3, 4].reduce((acc, cur) => acc + cur, 0)

第1次: acc = 0,  cur = 1  -> acc = 1
第2次: acc = 1,  cur = 2  -> acc = 3
第3次: acc = 3,  cur = 3  -> acc = 6
第4次: acc = 6,  cur = 4  -> acc = 10

结果: 10
```

```javascript
// Array.prototype.reduce - 归约数组
Array.prototype.myReduce = function(callback, initialValue) {
  if (typeof callback !== 'function') {
    throw new TypeError(callback + ' is not a function');
  }

  const arr = this;
  const len = arr.length;

  // 没有初始值且数组为空会报错
  if (len === 0 && initialValue === undefined) {
    throw new TypeError('Reduce of empty array with no initial value');
  }

  let accumulator;
  let startIndex;

  // 有初始值：从索引 0 开始
  if (initialValue !== undefined) {
    accumulator = initialValue;
    startIndex = 0;
  } else {
    // 没有初始值：找到第一个非空索引的值作为初始值
    let i = 0;
    // 跳过稀疏数组的空位
    while (i < len && !(i in arr)) {
      i++;
    }
    if (i === len) {
      throw new TypeError('Reduce of empty array with no initial value');
    }
    accumulator = arr[i];
    startIndex = i + 1;
  }

  // 从 startIndex 开始遍历
  for (let i = startIndex; i < len; i++) {
    if (i in arr) {
      accumulator = callback(accumulator, arr[i], i, arr);
    }
  }

  return accumulator;
}
```

**没有初始值的情况**：

```javascript
// 有初始值
[1, 2, 3].reduce((a, b) => a + b, 0); // 6

// 没有初始值：取第一个元素为初始值，从第二个元素开始
[1, 2, 3].reduce((a, b) => a + b); // 6 (1+2+3)

// 稀疏数组
[, 2, 3].reduce((a, b) => a + b); // 5 (2+3，跳过第一个空位)
```

**reduce 的常见用途**：

```javascript
// 求和
[1, 2, 3].reduce((a, b) => a + b, 0); // 6

// 扁平化
[[1, 2], [3, 4], [5]].reduce((a, b) => a.concat(b), []); // [1,2,3,4,5]

// 计数的 Map
['a', 'b', 'a', 'c'].reduce((acc, cur) => {
  acc[cur] = (acc[cur] || 0) + 1;
  return acc;
}, {}); // { a: 2, b: 1, c: 1 }

// 分组
[{ type: 'a', val: 1 }, { type: 'b', val: 2 }, { type: 'a', val: 3 }]
  .reduce((acc, cur) => {
    (acc[cur.type] = acc[cur.type] || []).push(cur);
    return acc;
  }, {});
// { a: [{type:'a',val:1}, {type:'a',val:3}], b: [{type:'b',val:2}] }
```

#### 17.2.4 实现 Promise.all

**核心原理**：

Promise.all 接收一个 Promise 数组（或可迭代对象），返回一个新的 Promise：
- 所有 Promise 都成功时，返回包含所有结果的数组
- 任意一个 Promise 失败时，立即 reject 第一个失败的原因

**实现要点**：

1. 处理非 Promise 值：使用 Promise.resolve() 包装
2. 计数：跟踪有多少 Promise 已经 settled
3. 竞态处理：第一个 reject 立即触发

```javascript
// Promise.all - 所有 promise 都 resolve 时返回结果数组
function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    // 参数类型检查
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Promise.all requires an array'));
    }

    const results = [];
    let resolvedCount = 0;

    // 空数组直接返回
    if (promises.length === 0) {
      return resolve(results);
    }

    // 遍历所有 promise
    promises.forEach((promise, index) => {
      // Promise.resolve 处理非 promise 值
      Promise.resolve(promise).then(
        value => {
          // 保存结果，保持数组顺序
          results[index] = value;
          resolvedCount++;

          // 所有 promise 都 resolved 后 resolve
          if (resolvedCount === promises.length) {
            resolve(results);
          }
        },
        reason => {
          // 任意一个 reject，立即 reject
          reject(reason);
        }
      );
    });
  });
}
```

**为什么需要 results[index] 而不是 push？**

因为 Promise 可能以任意顺序完成，必须按原始数组顺序保存结果：

```javascript
const p1 = new Promise(r => setTimeout(() => r(1), 100));
const p2 = new Promise(r => setTimeout(() => r(2), 10));
const p3 = new Promise(r => setTimeout(() => r(3), 50));

Promise.all([p1, p2, p3]).then(console.log); // [1, 2, 3] - 按顺序
```

#### 17.2.5 实现 Promise.race

**核心原理**：

Promise.race 返回一个新的 Promise，只要任意一个 Promise settled（无论是 resolved 还是 rejected），就采用其结果。

```javascript
// Promise.race - 返回最快 settle 的结果
function promiseRace(promises) {
  return new Promise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Promise.race requires an array'));
    }

    // 任意一个 settled，就采用其结果
    promises.forEach(promise => {
      Promise.resolve(promise).then(
        value => resolve(value),
        reason => reject(reason)
      );
    });
  });
}
```

**常见用途**：

1. 实现超时功能：
```javascript
function withTimeout(promise, ms) {
  return Promise.race([
    promise,
    new Promise((_, reject) =>
      setTimeout(() => reject(new Error('Timeout')), ms)
    )
  ]);
}
```

2. 实现竞态条件处理：
```javascript
// 多个请求竞争，只取第一个结果
Promise.race([fetch('/api1'), fetch('/api2')])
  .then(result => console.log(result));
```

#### 17.2.6 完整版 Promise 实现原理

**Promise 的三种状态**：

```
                resolve(value)
                  ↓
        ┌───────────────────┐
        │    fulfilled     │
        │ (resolved)      │
        └───────────────────┘
              ↑
              │ reject
              │
┌───────┐  ┌───────────────────┐  ┌───────────────┐
│pending │─→│    pending       │─→│  rejected     │
└───────┘  └───────────────────┘  └───────────────┘
```

**状态转换规则**：

1. pending → fulfilled：调用 resolve()
2. pending → rejected：调用 reject()
3. 一旦状态改变，就不能再改变

**then 方法的实现原理**：

```javascript
then(onFulfilled, onRejected) {
  // 返回新的 Promise，实现链式调用
  const promise = new MyPromise((resolve, reject) => {
    // 根据当前状态处理回调
    if (this.state === 'fulfilled') {
      // 异步执行，确保 promise 已创建
      setTimeout(() => {
        try {
          const result = onFulfilled ? onFulfilled(this.value) : this.value;
          resolve(result);
        } catch (error) {
          reject(error);
        }
      }, 0);
    } else if (this.state === 'rejected') {
      setTimeout(() => {
        try {
          if (onRejected) {
            const result = onRejected(this.value);
            resolve(result); // 错误处理后可以恢复
          } else {
            reject(this.value); // 传播错误
          }
        } catch (error) {
          reject(error);
        }
      }, 0);
    } else {
      // pending 状态：加入回调队列
      this.onFulfilledCallbacks.push(() => {
        try {
          const result = onFulfilled ? onFulfilled(this.value) : this.value;
          resolve(result);
        } catch (error) {
          reject(error);
        }
      });
      this.onRejectedCallbacks.push(() => {
        try {
          if (onRejected) {
            const result = onRejected(this.value);
            resolve(result);
          } else {
            reject(this.value);
          }
        } catch (error) {
          reject(error);
        }
      });
    }
  });

  return promise;
}
```

**Promise 解决过程（Promise Resolution Procedure）**：

这是 Promise 规范中最复杂的部分，处理 Promise 与其他值的嵌套：

```javascript
_resolve(value) {
  // 如果已经 settled，不做处理
  if (this.state !== 'pending') return;

  // 处理 Promise 值（thenable）
  if (value && (typeof value === 'object' || typeof value === 'function')) {
    let called = false;
    try {
      // 尝试提取 then 方法
      const then = value.then;
      if (typeof then === 'function') {
        // 如果 value 是 Promise，使用 value.then
        // 使用 call 确保 this 指向正确
        then.call(
          value,
          y => {
            if (called) return;
            called = true;
            this._resolve(y); // 递归解析
          },
          r => {
            if (called) return;
            called = true;
            this._reject(r);
          }
        );
        return;
      }
    } catch (error) {
      if (!called) {
        this._reject(error);
      }
      return;
    }
  }

  // 普通值：直接 fulfill
  this.state = 'fulfilled';
  this.value = value;
  this._notifyFulfilled();
}
```

**为什么需要 called 标志？**

防止 Promise 的回调被多次调用：

```javascript
const promise = new Promise((resolve, reject) => {
  // 恶意代码可能多次调用 resolve
  resolve(1);
  resolve(2);
  reject(3);
});
// 如果没有 called 标志，可能触发多次状态转换，导致错误
```

### 17.3 Function 方法实现

#### 17.3.1 实现 Function.prototype.bind

**核心原理**：

bind 方法创建一个新的函数（称为绑定函数），该函数的 this 值被设置为 bind 的第一个参数，其余参数作为新函数的参数。

**bind 的特性**：

1. 返回一个新函数
2. 永久绑定 this 和部分参数
3. 支持构造函数调用（此时忽略 this 绑定）

```javascript
// Function.prototype.bind - 返回一个新函数，绑定 this 和参数
Function.prototype.myBind = function(thisArg, ...args) {
  if (typeof this !== 'function') {
    throw new TypeError('Function.prototype.bind called on non-function');
  }

  const fn = this;

  // 返回的新函数
  function bound(...newArgs) {
    // 判断是否作为构造函数调用（通过 new）
    // 使用 instanceof 检查 this 是否是 bound 的实例
    const isConstructor = this instanceof bound;
    // 使用 apply 处理：构造函数调用忽略 thisArg，普通调用使用 thisArg
    return fn.apply(isConstructor ? this : thisArg, [...args, ...newArgs]);
  }

  // 保持原型链
  bound.prototype = Object.create(fn.prototype);

  return bound;
}
```

**为什么需要 isConstructor 检查？**

```javascript
const obj = { name: 'test' };

function greet(greeting) {
  return `${greeting}, ${this.name}`;
}

const boundGreet = greet.myBind(obj, 'Hello');

// 普通调用
boundGreet('!'); // 'Hello, test!'

// 构造函数调用

// 构造函数调用时，this 应该是新创建的 instance
// 而不是 bind 时绑定的 obj
```

#### 17.3.2 实现 Function.prototype.call

**核心原理**：

call 方法调用一个函数，其 this 值被指定为 call 的第一个参数，其余参数作为被调用函数的参数。

```javascript
// Function.prototype.call - 调用函数，指定 this
Function.prototype.myCall = function(thisArg, ...args) {
  if (typeof this !== 'function') {
    throw new TypeError('Function.prototype.call called on non-function');
  }

  // 处理 null 和 undefined：指向全局对象（window 或 global）
  // Object() 包装确保 thisArg 是对象
  thisArg = thisArg === null || thisArg === undefined ? window : Object(thisArg);

  // 使用 Symbol 避免属性名冲突
  const fn = Symbol('fn');
  thisArg[fn] = this;

  // 调用函数并获取结果
  const result = thisArg[fn](...args);

  // 清理：删除临时添加的属性
  delete thisArg[fn];

  return result;
}
```

**为什么用 Symbol 作为属性名？**

Symbol 是唯一的，可以避免与对象原有属性冲突：

```javascript
const obj = { name: 'test' };
function greet() { return this.name; }

// 用普通字符串作为属性名可能覆盖原有属性
obj.fn = greet;
obj.fn(); // 'test'
delete obj.fn;

// 用 Symbol 更安全
const fn = Symbol('fn');
obj[fn] = greet;
obj[fn](); // 'test'
```

#### 17.3.3 实现 Function.prototype.apply

**核心原理**：

apply 与 call 功能相同，区别在于参数形式：apply 接收一个数组（或类数组对象）作为参数列表。

```javascript
// Function.prototype.apply - 调用函数，指定 this，参数为数组
Function.prototype.myApply = function(thisArg, argsArray) {
  if (typeof this !== 'function') {
    throw new TypeError('Function.prototype.apply called on non-function');
  }

  // 处理参数数组
  if (argsArray !== undefined && argsArray !== null) {
    if (!Array.isArray(argsArray)) {
      throw new TypeError('Second argument must be an array or null');
    }
  }

  thisArg = thisArg === null || thisArg === undefined ? window : Object(thisArg);

  const fn = Symbol('fn');
  thisArg[fn] = this;

  let result;
  if (argsArray) {
    // 使用展开运算符将数组展开为参数列表
    result = thisArg[fn](...argsArray);
  } else {
    result = thisArg[fn]();
  }

  delete thisArg[fn];

  return result;
}
```

**apply 的特殊用途**：

1. 数组展开为参数：
```javascript
const arr = [1, 2, 3];
Math.max.apply(null, arr); // 3
Math.max(...arr); // ES6 等效写法
```

2. 数组方法借用：
```javascript
// 类数组转换为数组
const arrayLike = { 0: 'a', 1: 'b', length: 2 };
const arr = Array.prototype.slice.myApply(arrayLike, [0, 2]); // ['a', 'b']
```

```javascript
// Object.freeze - 冻结对象
function myFreeze(obj) {
  if (typeof obj !== 'object' || obj === null) {
    return obj;
  }

  // 获取对象的所有属性描述符
  const propNames = Object.getOwnPropertyNames(obj);
  const propSymbols = Object.getOwnPropertySymbols(obj);

  // 遍历所有属性
  [...propNames, ...propSymbols].forEach(name => {
    const descriptor = Object.getOwnPropertyDescriptor(obj, name);
    if (descriptor && !descriptor.writable) {
      // 递归冻结
      if (typeof descriptor.value === 'object') {
        myFreeze(descriptor.value);
      }
      // 设置属性为不可写、不可配置
      Object.defineProperty(obj, name, {
        ...descriptor,
        writable: false,
        configurable: false
      });
    }
  });

  // 禁止扩展
  return Object.preventExtensions(obj);
}

// Object.seal - 密封对象
function mySeal(obj) {
  if (typeof obj !== 'object' || obj === null) {
    return obj;
  }

  const propNames = Object.getOwnPropertyNames(obj);
  const propSymbols = Object.getOwnPropertySymbols(obj);

  [...propNames, ...propSymbols].forEach(name => {
    const descriptor = Object.getOwnPropertyDescriptor(obj, name);
    if (descriptor && descriptor.configurable) {
      Object.defineProperty(obj, name, {
        ...descriptor,
        configurable: false
      });
    }
  });

  return Object.preventExtensions(obj);
}
```

```javascript
Array.prototype.myReduce = function(callback, initialValue) {
  if (typeof callback !== 'function') {
    throw new TypeError(callback + ' is not a function');
  }

  const arr = this;
  const len = arr.length;

  // 没有初始值且数组为空会报错
  if (len === 0 && initialValue === undefined) {
    throw new TypeError('Reduce of empty array with no initial value');
  }

  let accumulator;
  let startIndex;

  if (initialValue !== undefined) {
    accumulator = initialValue;
    startIndex = 0;
  } else {
    // 找到第一个非空索引的值作为初始值
    let i = 0;
    while (i < len && !(i in arr)) {
      i++;
    }
    if (i === len) {
      throw new TypeError('Reduce of empty array with no initial value');
    }
    accumulator = arr[i];
    startIndex = i + 1;
  }

  for (let i = startIndex; i < len; i++) {
    if (i in arr) {
      accumulator = callback(accumulator, arr[i], i, arr);
    }
  }

  return accumulator;
};
```

```javascript
// Array.prototype.slice - 截取数组
Array.prototype.mySlice = function(begin = 0, end = undefined) {
  const arr = this;
  const len = arr.length;

  // 处理负数索引
  let start = begin;
  if (start < 0) {
    start = len + start;
    if (start < 0) start = 0;
  }

  let stop = end;
  if (stop === undefined) {
    stop = len;
  } else if (stop < 0) {
    stop = len + stop;
    if (stop < 0) stop = 0;
  }

  const result = [];
  for (let i = start; i < stop && i < len; i++) {
    if (i in arr) {
      result.push(arr[i]);
    }
  }

  return result;
};

// Array.prototype.splice - 删除/替换/插入元素
Array.prototype.mySplice = function(start, deleteCount, ...items) {
  const arr = this;
  const len = arr.length;

  // 处理负数索引
  if (start < 0) {
    start = len + start;
    if (start < 0) start = 0;
  }

  if (start > len) {
    start = len;
  }

  // 处理 deleteCount
  let count = deleteCount;
  if (count === undefined) {
    count = len - start;
  } else if (count < 0) {
    count = 0;
  } else {
    count = Math.min(count, len - start);
  }

  // 删除的元素
  const deleted = [];
  for (let i = start; i < start + count; i++) {
    if (i in arr) {
      deleted.push(arr[i]);
    }
  }

  // 剩余元素
  const remaining = [];
  for (let i = start + count; i < len; i++) {
    if (i in arr) {
      remaining.push(arr[i]);
    }
  }

  // 替换：删除部分 + 插入部分 + 剩余部分
  const newLen = len - count + items.length;
  for (let i = 0; i < items.length; i++) {
    arr[start + i] = items[i];
  }
  for (let i = 0; i < remaining.length; i++) {
    arr[start + items.length + i] = remaining[i];
  }

  // 调整数组长度
  arr.length = newLen;

  return deleted;
};
```

```javascript
String.prototype.myReplace = function(searchValue, replaceValue) {
  const str = String(this);

  if (typeof searchValue === 'string') {
    // 字符串搜索
    const index = str.indexOf(searchValue);
    if (index === -1) return str;

    } else if (typeof replaceValue === 'function') {

    const match = str.match(searchValue);
    if (!match) return str;

    if (typeof replaceValue === 'string') {
      return str.replace(searchValue, replaceValue);
    } else if (typeof replaceValue === 'function') {
      return str.replace(searchValue, replaceValue);
    }
  }

  return str;
};
```

```javascript
String.prototype.mySplit = function(separator, limit) {

  const result = [];

  if (separator === '') {
    // 空分隔符：返回所有字符
    const len = str.length;
    for (let i = 0; i < len && result.length < (limit || Infinity); i++) {
      result.push(str[i]);
    }
    return result;
  }

  if (separator === undefined) {
    // 无分隔符：返回整个字符串
    return limit === undefined ? [str] : [str.slice(0, limit)];
  }

  let lastIndex = 0;
  let index;

  while ((index = str.indexOf(separator, lastIndex)) !== -1) {
    result.push(str.slice(lastIndex, index));
    lastIndex = index + separator.length;

    if (limit !== undefined && result.length >= limit) {
      return result;
    }
  }

  // 加上最后一部分
  result.push(str.slice(lastIndex));

  return result;
};

// 测试
console.log('a,b,c'.mySplit(','));    // ['a', 'b', 'c']
console.log('hello world'.mySplit(' ')); // ['hello', 'world']
console.log('abc'.mySplit(''));      // ['a', 'b', 'c']
```

```javascript
function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Promise.all requires an array'));

      );
    });
  });
}
```

```javascript
// Promise.allSettled - 所有 promise 都 settled（无论 resolve/reject）后返回结果
function promiseAllSettled(promises) {
  return new Promise((resolve) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Promise.allSettled requires an array'));
    }

    const results = [];
    let settledCount = 0;

    if (promises.length === 0) {
      return resolve(results);
    }

    const processResult = (index, status, valueOrReason) => {
      results[index] = {

      Promise.resolve(promise).then(
        value => processResult(index, 'fulfilled', value),
        reason => processResult(index, 'rejected', reason)
      );
    });
  });
}
```

```javascript
function promiseAny(promises) {
  return new Promise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Promise.any requires an array'));
    }

    const errors = [];
    let rejectedCount = 0;

    if (promises.length === 0) {
      return reject(new AggregateError([], 'All promises were rejected'));
    }

    promises.forEach((promise, index) => {
      Promise.resolve(promise).then(
        value => resolve(value),
        reason => {
          errors[index] = reason;
          rejectedCount++;

          if (rejectedCount === promises.length) {
            reject(new AggregateError(errors, 'All promises were rejected'));
          }
        }
      );
    });
  });
}
```

#### 17.4.6 完整版 Promise 实现

```javascript
// 完整版 Promise 实现
class MyPromise {
  constructor(executor) {
    if (typeof executor !== 'function') {
      throw new TypeError('Promise resolver ' + executor + ' is not a function');
    }

    this.state = 'pending'; // pending, fulfilled, rejected
    this.value = undefined;
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];

    try {
      executor(
        value => this._resolve(value),
        reason => this._reject(reason)
      );
    } catch (error) {
      this._reject(error);
    }
  }

  _resolve(value) {
    // 如果已经 settled，不做处理
    if (this.state !== 'pending') return;

    // 处理 Promise 值（thenable）
    if (value && (typeof value === 'object' || typeof value === 'function')) {
      let called = false;
      try {
        const then = value.then;
        if (typeof then === 'function') {
          then.call(
            value,
            y => {
              if (called) return;
              called = true;
              this._resolve(y);
            },
            r => {
              if (called) return;
              called = true;
              this._reject(r);
            }
          );
          return;
        }
      } catch (error) {
        if (!called) {
          this._reject(error);
        }
        return;
      }
    }

    this.state = 'fulfilled';
    this.value = value;
    this._notifyFulfilled();
  }

  _reject(reason) {
    if (this.state !== 'pending') return;

    this.state = 'rejected';
    this.value = reason;
    this._notifyRejected();
  }

  _notifyFulfilled() {
    this.onFulfilledCallbacks.forEach(callback => {
      try {
        callback(this.value);
      } catch (error) {
        // 忽略回调中的错误
      }
    });
    this.onFulfilledCallbacks = [];
  }

  _notifyRejected() {
    this.onRejectedCallbacks.forEach(callback => {
      try {
        callback(this.value);
      } catch (error) {
        // 忽略回调中的错误
      }
    });
    this.onRejectedCallbacks = [];
  }

  then(onFulfilled, onRejected) {
    const promise = new MyPromise((resolve, reject) => {
      const handleFulfilled = () => {
        try {
          const result = onFulfilled ? onFulfilled(this.value) : this.value;
          resolve(result);
        } catch (error) {
          reject(error);
        }
      };

      const handleRejected = () => {
        try {
          if (onRejected) {
            const result = onRejected(this.value);
            resolve(result);
          } else {
            reject(this.value);
          }
        } catch (error) {
          reject(error);
        }
      };

      if (this.state === 'fulfilled') {
        setTimeout(handleFulfilled, 0);
      } else if (this.state === 'rejected') {
        setTimeout(handleRejected, 0);
      } else {
        this.onFulfilledCallbacks.push(handleFulfilled);
        this.onRejectedCallbacks.push(handleRejected);
      }
    });

    return promise;
  }

  catch(onRejected) {
    return this.then(undefined, onRejected);
  }

  finally(onFinally) {
    return this.then(
      value => {
        onFinally();
        return value;
      },
      reason => {
        onFinally();
        throw reason;
      }
    );
  }

  static resolve(value) {
    if (value instanceof MyPromise) {
      return value;
    }
    return new MyPromise(resolve => resolve(value));
  }

  static reject(reason) {
    return new MyPromise((_, reject) => reject(reason));
  }

  static all(promises) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(promises)) {
        return reject(new TypeError('Promise.all requires an array'));
      }

      const results = [];
      let count = 0;

      if (promises.length === 0) {
        return resolve(results);
      }

      promises.forEach((promise, index) => {
        MyPromise.resolve(promise).then(
          value => {
            results[index] = value;
            count++;
            if (count === promises.length) {
              resolve(results);
            }
          },
          reject
        );
      });
    });
  }

  static race(promises) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(promises)) {
        return reject(new TypeError('Promise.race requires an array'));
      }

      promises.forEach(promise => {
        MyPromise.resolve(promise).then(resolve, reject);
      });
    });
  }

  static allSettled(promises) {
    return new MyPromise(resolve => {
      if (!Array.isArray(promises)) {
        return reject(new TypeError('Promise.allSettled requires an array'));
      }

      const results = [];
      let count = 0;

      if (promises.length === 0) {
        return resolve(results);
      }

      const processResult = (index, status, valueOrReason) => {
        results[index] = {
          status,
          ...(status === 'fulfilled' ? { value: valueOrReason } : { reason: valueOrReason })
        };

        count++;
        if (count === promises.length) {
          resolve(results);
        }
      };

      promises.forEach((promise, index) => {
        MyPromise.resolve(promise).then(
          value => processResult(index, 'fulfilled', value),
          reason => processResult(index, 'rejected', reason)
        );
      });
    });
  }
}
```

```javascript
Function.prototype.myBind = function(thisArg, ...args) {
  if (typeof this !== 'function') {
    throw new TypeError('Function.prototype.bind called on non-function');
  }

  const fn = this;

  // 返回的新函数
  function bound(...newArgs) {
    // 构造函数调用时，忽略 thisArg
    const isConstructor = this instanceof bound;
    // 使用 apply 处理 new 操作符的情况
    return fn.apply(isConstructor ? this : thisArg, [...args, ...newArgs]);
  }

  // 保持原型链
  bound.prototype = Object.create(fn.prototype);

  return bound;
};
```

```javascript
Function.prototype.myCall = function(thisArg, ...args) {
```

```javascript
Function.prototype.myApply = function(thisArg, argsArray) {
  if (typeof this !== 'function') {
    throw new TypeError('Function.prototype.apply called on non-function');
  }

  // 处理参数
  if (argsArray !== undefined && argsArray !== null) {
    if (!Array.isArray(argsArray)) {
      throw new TypeError('Second argument must be an array or null');
    }
  }

  thisArg = thisArg === null || thisArg === undefined ? window : Object(thisArg);

  const fn = Symbol('fn');
  thisArg[fn] = this;

  let result;
  if (argsArray) {
    result = thisArg[fn](...argsArray);
  } else {
    result = thisArg[fn]();
  }

  delete thisArg[fn];

  return result;
};
```

#### 17.7.1 实现 JSON.stringify

```javascript
// 简化版 JSON.stringify
function jsonStringify(value) {
  if (value === undefined) return undefined;
  if (value === null) return 'null';

  if (typeof value === 'number' || typeof value === 'boolean') {
    return String(value);
  }

  if (typeof value === 'string') {
    // 转义特殊字符
    return '"' + value
      .replace(/\\/g, '\\\\')
      .replace(/"/g, '\\"')
      .replace(/\n/g, '\\n')
      .replace(/\r/g, '\\r')
      .replace(/\t/g, '\\t') + '"';
  }

  if (Array.isArray(value)) {
    const arr = value.map(item => jsonStringify(item));
    return '[' + arr.join(',') + ']';
  }

  if (typeof value === 'object') {
    const keys = Object.keys(value);
    const pairs = keys.map(key => {
      const val = jsonStringify(value[key]);
      if (val === undefined) return undefined;
      return jsonStringify(key) + ':' + val;
    }).filter(Boolean);
    return '{' + pairs.join(',') + '}';
  }

  return undefined;
}

// 测试
console.log(jsonStringify({ a: 1, b: 'hello' })); // {"a":1,"b":"hello"}
console.log(jsonStringify([1, 2, { c: 3 }])); // [1,2,{"c":3}]
```
