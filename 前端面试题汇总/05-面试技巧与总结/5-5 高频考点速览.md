### 1.2 原型与继承

**原型链**：
- 每个对象都有 `__proto__` 属性
- 构造函数有 `prototype` 属性
- 原型链顶端是 `Object.prototype`
- `Object.prototype.__proto__` 是 null

**继承实现方式**：

1. **原型链继承**：`Child.prototype = new Parent()`
2. **构造函数继承**：`Parent.call(this, args)`
3. **组合继承**：原型链 + 构造函数
4. **ES6 class**：`class Child extends Parent`

### 1.3 作用域与闭包

**作用域类型**：
- 全局作用域
- 函数作用域
- 块级作用域 (let, const)

**闭包**：
- 定义：函数能够访问外部作用域变量的现象
- 作用：私有化数据、函数工厂
- 注意点：内存泄漏风险

**经典面试题**：

```javascript
for (var i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 100);
}
// 输出：5, 5, 5, 5, 5

// 解决方式1：使用let
for (let i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 100);
}
// 输出：0, 1, 2, 3, 4
```

### 1.4 事件循环

**宏任务队列**：setTimeout, setInterval, I/O, UI渲染
**微任务队列**：Promise, MutationObserver, process.nextTick

**执行顺序**：
1. 同步代码
2. 清空微任务队列
3. 渲染（如需要）
4. 执行下一个宏任务

**async/await**：
- async 函数返回 Promise
- await 等待 Promise resolve
- await 后的代码相当于 Promise.then

### 1.5 this 指向

| 场景 | this指向 |
|------|---------|
| 普通函数调用 | window/undefined |
| 对象方法 | 调用该方法的对象 |
| 构造函数 | 实例对象 |
| call/apply/bind | 第一个参数 |
| 箭头函数 | 定义时的外层this |

### 1.6 浅拷贝与深拷贝

**浅拷贝**：
```javascript
const shallow = { ...obj };
const shallow2 = Object.assign({}, obj);
```

**深拷贝**：
```javascript
// 简单版 - 有缺陷（不能复制函数、Symbol）
const deep = JSON.parse(JSON.stringify(obj));

// 现代API
const deep2 = structuredClone(obj); // 原生支持

// 完整版
function deepClone(obj, hash = new WeakMap()) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (hash.has(obj)) return hash.get(obj);

  const clone = Array.isArray(obj) ? [] : {};
  hash.set(obj, clone);

  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key], hash);
    }
  }
  return clone;
}
```

## 九、手写代码

### 9.1 防抖与节流

```javascript
// 防抖 - debounce
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

// 节流 - throttle
function throttle(fn, delay) {
  let last = 0;
  return function(...args) {
    const now = Date.now();
    if (now - last >= delay) {
      last = now;
      fn.apply(this, args);
    }
  };
}
```

### 9.2 深拷贝

```javascript
function deepClone(obj, hash = new WeakMap()) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (hash.has(obj)) return hash.get(obj);

  const clone = Array.isArray(obj) ? [] : {};
  hash.set(obj, clone);

  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key], hash);
    }
  }
  return clone;
}
```

### 9.3 Promise.all / Promise.race

```javascript
Promise.myAll = function(promises) {
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;
    promises.forEach((p, i) => {
      Promise.resolve(p).then(val => {
        results[i] = val;
        count++;
        if (count === promises.length) resolve(results);
      }, reject);
    });
  });
};

Promise.myRace = function(promises) {
  return new Promise((resolve, reject) => {
    promises.forEach(p => Promise.resolve(p).then(resolve, reject));
  });
};
```

### 9.4 new 操作符

```javascript
function myNew(constructor, ...args) {
  const obj = Object.create(constructor.prototype);
  const result = constructor.apply(obj, args);
  return result instanceof Object ? result : obj;
}
```

### 9.5 call/apply/bind

```javascript
Function.prototype.myCall = function(context, ...args) {
  const fn = Symbol('fn');
  context[fn] = this;
  const result = context[fn](...args);
  delete context[fn];
  return result;
};

Function.prototype.myBind = function(context, ...args) {
  return (...newArgs) => this.call(context, ...args, ...newArgs);
};
```

```javascript
function VirtualList({ items, rowHeight }) {
  const [scrollTop, setScrollTop] = useState(0);
  const containerHeight = 500;

  const startIndex = Math.floor(scrollTop / rowHeight);
  const endIndex = startIndex + Math.ceil(containerHeight / rowHeight) + 1;

    <div
      style={{ height: containerHeight, overflow: 'auto' }}

              height: rowHeight
            }}
          >

    </div>
  );
}
```

### 19.3 实现图片懒加载

```javascript
function LazyImage({ src, placeholder }) {
  const [loaded, setLoaded] = useState(false);
  const [inView, setInView] = useState(false);
  const imgRef = useRef();

  useEffect(() => {
    const observer = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        setInView(true);
        observer.disconnect();
      }
    });

    observer.observe(imgRef.current);
    return () => observer.disconnect();
  }, []);

  return (
    <div ref={imgRef}>
      {inView && (
        <img
          src={src}
          onLoad={() => setLoaded(true)}
          style={{ opacity: loaded ? 1 : 0 }}
        />
      )}
      {!loaded && placeholder}
    </div>
  );
}
```

### 26.2 Promise 原理

```javascript
class MyPromise {
  constructor(executor) {
    this.state = 'pending';
    this.value = undefined;
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];

    const resolve = (value) => {
      if (this.state === 'pending') {
        this.state = 'fulfilled';
        this.value = value;
        this.onFulfilledCallbacks.forEach(cb => cb(value));
      }
    };

    const reject = (reason) => {
      if (this.state === 'pending') {
        this.state = 'rejected';
        this.value = reason;
        this.onRejectedCallbacks.forEach(cb => cb(reason));
      }
    };

    try {
      executor(resolve, reject);
    } catch (error) {
      reject(error);
    }
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      if (this.state === 'fulfilled') {
        const result = onFulfilled(this.value);
        resolve(result);
      } else if (this.state === 'rejected') {
        const result = onRejected(this.value);
        resolve(result);
      } else {
        this.onFulfilledCallbacks.push(value => {
          const result = onFulfilled(value);
          resolve(result);
        });
        this.onRejectedCallbacks.push(reason => {
          const result = onRejected(reason);
          resolve(result);
        });
      }
    });
  }
}
```

### 50.1 手写 Promise.all

```javascript
function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;

    promises.forEach((p, i) => {
      Promise.resolve(p).then(
        val => {
          results[i] = val;
          count++;
          if (count === promises.length) resolve(results);
        },
        reject
      );
    });
  });

```

### 50.2 手写防抖节流

```javascript
// 防抖
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

// 节流
function throttle(fn, delay) {
  let last = 0;
  return function(...args) {
    const now = Date.now();
    if (now - last >= delay) {
      last = now;
      fn.apply(this, args);
    }
  };
}
```

### 50.3 手写深拷贝

```javascript
function deepClone(obj, hash = new WeakMap()) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (hash.has(obj)) return hash.get(obj);

  const clone = Array.isArray(obj) ? [] : {};
  hash.set(obj, clone);

  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key], hash);
    }
  }
  return clone;
}
```

### 43.1 XSS 攻击与防御

#### XSS 攻击原理

**存储型 XSS**：
1. 攻击者将恶意代码提交到数据库
2. 用户访问页面
3. 服务器返回包含恶意代码的内容
4. 浏览器执行恶意代码

**反射型 XSS**：
1. 攻击者构造包含恶意代码的 URL
2. 用户点击链接
3. 服务器将 URL 参数原样返回
4. 浏览器执行恶意代码

**DOM 型 XSS**：
1. 攻击者构造包含恶意代码的 URL
2. 用户点击链接
3. 前端 JavaScript 解析 URL 并操作 DOM
4. 恶意代码被执行

#### 防御措施

**输入过滤**：
```javascript
function escapeHtml(str) {
  const map = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#039;'
  };
  return str.replace(/[&<>"']/g, m => map[m]);
}
```

**输出编码**：
- HTML 编码
- URL 编码
- JavaScript 编码

**Content Security Policy**：

### 43.2 CSRF 攻击与防御

#### CSRF 攻击原理

1. 用户登录网站 A
2. 用户被诱导访问网站 B
3. 网站 B 向网站 A 发送请求
4. 浏览器自动携带网站 A 的 Cookie
5. 网站 A 认为是用户合法请求

#### 防御措施

**CSRF Token**：
```javascript
// 服务端生成
const token = crypto.randomBytes(32).toString('hex');
// 存储到 session 或 Redis

// 前端发送请求时携带
fetch('/api/action', {
  method: 'POST',
  headers: {
    'X-CSRF-Token': token
  },
  body: JSON.stringify(data)
});
```

**SameSite Cookie**：
```
Set-Cookie: session=xxx; SameSite=Strict
Set-Cookie: session=xxx; SameSite=Lax
```

**Referer 验证**：
- 验证请求来源
- 不能完全依赖

### 43.3 点击劫持与防御

#### 攻击原理

1. 攻击者创建一个透明 iframe
2. 将目标网站嵌入其中
3. 在透明层上覆盖诱骗内容
4. 用户点击实际触发目标网站操作

#### 防御措施

**X-Frame-Options**：
```
X-Frame-Options: DENY
X-Frame-Options: SAMEORIGIN
```

**Content Security Policy**：
```
Content-Security-Policy: frame-ancestors 'self'
```

## 四十四、浏览器原理深入

### 44.1 浏览器渲染流程

#### 详细流程

```
1. HTML 解析 → DOM 树
2. CSS 解析 → CSSOM 树
3. DOM + CSSOM → Render 树
4. Layout（布局） → 绘制顺序
5. Paint（绘制） → 分层
6. Composite（合成） → 渲染到屏幕
```

#### 重排与重绘

**触发重排的操作**：
- 添加/删除元素
- 改变元素尺寸
- 改变元素位置
- 获取布局信息（offsetWidth、getComputedStyle 等）

**触发重绘的操作**：
- 改变颜色、背景
- 改变 visibility
- 改变边框样式

### 44.2 JavaScript 引擎

#### V8 引擎原理

**主要组件**：
1. **Parser**：解析源代码生成 AST
2. **Ignition**：解释器，执行字节码
3. **TurboFan**：编译器，优化热点代码
4. **Orinoco**：垃圾回收器

**执行流程**：
```
源代码 → Parser → AST → Ignition（字节码）→ 执行
                  ↓
            TurboFan（优化编译器）
                  ↓
                优化机器码
```

### 44.3 垃圾回收

#### 垃圾回收算法

**标记清除**：
1. 标记所有可达对象
2. 清除未标记对象
3. 整理内存碎片

**标记整理**：
1. 标记所有可达对象
2. 整理存活对象到一端
3. 清除边界外对象

**分代回收**：
- **新生代**：短生命周期对象
- **老生代**：长生命周期对象
