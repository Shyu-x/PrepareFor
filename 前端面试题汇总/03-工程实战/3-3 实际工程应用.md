## 一、性能优化：5s 优化到 1s 以内

### 1.1 如何实现首屏加载极致优化？

**策略清单：**
1.  **资源瘦身**：
    *   **Gzip/Brotli 压缩**。
    *   **Tree Shaking** 剔除无用代码。
    *   **图片优化**：WebP 格式、CDN 裁剪、懒加载。
2.  **减少阻塞**：
    *   **代码分割 (Code Splitting)**：使用 React.lazy 或 dynamic import 减小主 Bundle。
    *   **CSS 放在头部，JS 放在尾部 (defer/async)**。
3.  **缓存利用**：
    *   **强缓存 + 协商缓存**。
    *   **Service Worker** 实现离线缓存。
4.  **渲染提速**：
    *   **SSR (服务端渲染)** 或 **SSG (静态生成)**。
    *   **骨架屏 (Skeleton Screen)** 提升感知速度。

```javascript
// 图片优化工具函数

/**
 * 生成 CDN 图片 URL
 * @param {string} url 原始图片 URL
 * @param {object} options 裁剪选项
 * @returns {string} 优化后的 URL
 */
export function getOptimizedImageUrl(url, options = {}) {
  const {
    width,
    height,
    quality = 80,
    format = 'webp',
    mode = 'fill', // fill, fit, limit, scale
  } = options;

  // 如果不是 CDN 图片，直接返回原图
  if (!url.includes('cdn.example.com')) {
    return url;
  }

  const params = new URLSearchParams();
  if (width) params.append('w', width);
  if (height) params.append('h', height);
  if (quality) params.append('q', quality);
  if (format) params.append('fm', format);
  if (mode) params.append('m', mode);

  const paramString = params.toString();
  return paramString ? `${url}?${paramString}` : url;
}

/**
 * 响应式图片组件 - React 示例
 */
export const ResponsiveImage = ({ src, srcSet, alt, sizes, className }) => {
  const generateSrcSet = (url) => {
    const widths = [320, 640, 960, 1280, 1920];
    return widths
      .map(w => `${getOptimizedImageUrl(url, { width: w })} ${w}w`)
      .join(', ');
  };

  return (
    <img
      src={getOptimizedImageUrl(src, { width: 640 })}
      srcSet={generateSrcSet(src)}
      sizes={sizes || '(max-width: 640px) 100vw, (max-width: 1280px) 50vw, 33vw'}
      alt={alt}
      className={className}
      loading="lazy"
      decoding="async"
    />
  );
};

/**
 * 图片懒加载组件 - Intersection Observer 实现
 */
export class LazyImage extends React.Component {
  constructor(props) {
    super(props);
    this.imageRef = React.createRef();
    this.state = { isLoaded: false, isInView: false };
  }

  componentDidMount() {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            this.setState({ isInView: true });
            observer.disconnect();
          }
        });
      },
      { rootMargin: '50px' }
    );

    if (this.imageRef.current) {
      observer.observe(this.imageRef.current);
    }
  }

  render() {
    const { src, alt, placeholder, ...props } = this.props;
    const { isLoaded, isInView } = this.state;

    return (
      <div ref={this.imageRef} style={{ position: 'relative' }}>
        {!isLoaded && (
          <div style={{
            position: 'absolute',
            top: 0,
            left: 0,
            width: '100%',
            height: '100%',
            background: '#f0f0f0'
          }}>
            {placeholder}
          </div>
        )}
        {isInView && (
          <img
            src={src}
            alt={alt}
            onLoad={() => this.setState({ isLoaded: true })}
            {...props}
          />
        )}
      </div>
    );
  }
}
```

```javascript
 * 性能监控 Hook - React

import { useEffect, useState } from 'react';

export function usePerformanceMetrics() {
  const [metrics, setMetrics] = useState({});

  useEffect(() => {
    if (!('PerformanceObserver' in window)) return;

    // 观察 LCP
    const lcpObserver = new PerformanceObserver((list) => {
      const entries = list.getEntries();
      const lastEntry = entries[entries.length - 1];
      setMetrics(prev => ({
        ...prev,
        LCP: lastEntry.renderTime || lastEntry.loadTime,
      }));
    });
    lcpObserver.observe({ entryType: 'largest-contentful-paint' });

    // 观察 CLS
    const clsObserver = new PerformanceObserver((list) => {
      let clsScore = 0;
      for (const entry of list.getEntries()) {
        if (!entry.hadRecentInput) {
          clsScore += entry.value;
        }
      }
      setMetrics(prev => ({ ...prev, CLS: clsScore }));
    });
    clsObserver.observe({ entryType: 'layout-shift' });

    // 观察 FID
    const fidObserver = new PerformanceObserver((list) => {
      const firstEntry = list.getEntries()[0];
      if (firstEntry) {
        setMetrics(prev => ({
          ...prev,
          FID: firstEntry.processingStart - firstEntry.startTime,
        }));
      }
    });
    fidObserver.observe({ entryType: 'first-input' });

    // 获取其他性能指标
    const paintEntries = performance.getEntriesByType('paint');
    paintEntries.forEach(entry => {
      if (entry.name === 'first-contentful-paint') {
        setMetrics(prev => ({ ...prev, FCP: entry.startTime }));
      }
    });

    return () => {
      lcpObserver.disconnect();
      clsObserver.disconnect();
      fidObserver.disconnect();
    };
  }, []);
```

## 二、架构设计：可撤销/重做 (Undo/Redo)

### 2.1 方案选型

**方案对比：**
1.  **状态快照流 (Snapshot)**：
    *   原理：每次操作保存一个完整的 state 到历史栈。
    *   缺点：内存开销大。
    *   适用场景：状态较小、操作不频繁的场景。
2.  **命令模式 (Command Pattern)**：
    *   原理：将操作封装为对象，包含 `execute()` 和 `undo()` 方法。
    *   优点：节省空间，可精确回滚。
    *   适用场景：复杂操作、状态较大的场景。

### 2.2 命令模式实现 - 完整代码

```typescript
// command.ts - 命令实现

import { Command, SerializableCommand, DocumentState } from './types';

/**
 * 插入文本命令
 */
export class InsertTextCommand implements SerializableCommand {
  type = 'INSERT_TEXT';
  timestamp = Date.now();

  constructor(
    private documentId: string,
    private position: number,
    private text: string,
    private previousState?: DocumentState
  ) {}

  get payload() {
    return {
      documentId: this.documentId,
      position: this.position,
      text: this.text,
    };
  }

  execute(state: EditorState): EditorState {
    const doc = state.documents[this.documentId];
    if (!doc) return state;

    const newContent =
      doc.content.slice(0, this.position) +
      this.text +
      doc.content.slice(this.position);

    return {
      ...state,
      documents: {
        ...state.documents,
        [this.documentId]: {
          ...doc,
          content: newContent,
          selection: {
            start: this.position + this.text.length,
            end: this.position + this.text.length,
          },
          history: [...doc.history, doc.content],
        },
      },
    };
  }

  undo(state: EditorState): EditorState {
    const doc = state.documents[this.documentId];
    if (!doc || doc.history.length === 0) return state;

    const previousContent = doc.history[doc.history.length - 1];
    const newHistory = doc.history.slice(0, -1);

    return {
      ...state,
      documents: {
        ...state.documents,
        [this.documentId]: {
          ...doc,
          content: previousContent,
          history: newHistory,
          selection: {
            start: this.position,
            end: this.position,
          },
        },
      },
    };
  }

  redo(state: EditorState): EditorState {
    return this.execute(state);
  }

  static fromJSON(json: any): InsertTextCommand {
    return new InsertTextCommand(
      json.payload.documentId,
      json.payload.position,
      json.payload.text
    );
  }
}

export class DeleteTextCommand implements SerializableCommand {
  type = 'DELETE_TEXT';
  timestamp = Date.now();

  constructor(
    private documentId: string,
    private position: number,
    private length: number,
    private deletedText?: string
  ) {}

  get payload() {
    return {
      documentId: this.documentId,
      position: this.position,
      length: this.length,
      deletedText: this.deletedText,
    };
  }

  execute(state: EditorState): EditorState {
    const doc = state.documents[this.documentId];
    if (!doc) return state;

    const deletedText = doc.content.slice(
      this.position,
      this.position + this.length
    );
    const newContent =
      doc.content.slice(0, this.position) +
      doc.content.slice(this.position + this.length);

    return {
      ...state,
      documents: {
        ...state.documents,
        [this.documentId]: {
          ...doc,
          content: newContent,
          selection: {
            start: this.position,
            end: this.position,
          },
          history: [...doc.history, doc.content],
        },
      },
    };
  }

  undo(state: EditorState): EditorState {
    const doc = state.documents[this.documentId];
    if (!doc || doc.history.length === 0) return state;

    const previousContent = doc.history[doc.history.length - 1];
    const newHistory = doc.history.slice(0, -1);

    return {
      ...state,
      documents: {
        ...state.documents,
        [this.documentId]: {
          ...doc,
          content: previousContent,
          history: newHistory,
          selection: {
            start: this.position,
            end: this.position + (this.deletedText?.length || this.length),

        },
      },
    };
  }

  redo(state: EditorState): EditorState {
    return this.execute(state);
  }

      json.payload.position,
      json.payload.length,
      json.payload.deletedText
```

```typescript
// history-manager.ts - 历史管理器

import { Command, SerializableCommand, EditorState } from './types';
import {
  InsertTextCommand,
  DeleteTextCommand,
  FormatCommand,
} from './command';

/**
 * 命令注册表
 */
const CommandRegistry: Record<string, any> = {
  INSERT_TEXT: InsertTextCommand,
  DELETE_TEXT: DeleteTextCommand,
  FORMAT: FormatCommand,
};

/**
 * 历史记录管理器
 */
export class HistoryManager {
  private undoStack: SerializableCommand[] = [];
  private redoStack: SerializableCommand[] = [];
  private maxHistorySize: number;
  private state: EditorState;
  private listeners: Set<(state: EditorState) => void> = new Set();

  constructor(initialState: EditorState, maxHistorySize = 100) {
    this.state = initialState;
    this.maxHistorySize = maxHistorySize;
  }

  /**
   * 执行命令
   */
  executeCommand(command: Command): void {
    // 执行命令
    this.state = command.execute(this.state);

    // 将命令推入撤销栈
    if (this.isUndoableCommand(command)) {
      this.undoStack.push(command as SerializableCommand);
      // 限制历史记录大小
      if (this.undoStack.length > this.maxHistorySize) {
        this.undoStack.shift();
      }
    }

    // 清空重做栈（执行新命令后重做栈应该被清空）
    this.redoStack = [];

    // 通知监听器
    this.notifyListeners();
  }

  /**
   * 撤销
   */
  undo(): boolean {
    if (!this.canUndo()) return false;

    const command = this.undoStack.pop()!;
    this.state = command.undo(this.state);
    this.redoStack.push(command);

    this.notifyListeners();
    return true;
  }

  /**
   * 重做
   */
  redo(): boolean {
    if (!this.canRedo()) return false;

    const command = this.redoStack.pop()!;
    this.state = command.redo(this.state);
    this.undoStack.push(command);

    this.notifyListeners();
    return true;
  }

  /**
   * 是否可以撤销
   */
  canUndo(): boolean {
    return this.undoStack.length > 0;
  }

  /**
   * 是否可以重做
   */
  canRedo(): boolean {
    return this.redoStack.length > 0;
  }

  /**
   * 获取当前状态
   */
  getState(): EditorState {
    return this.state;
  }

  /**
   * 获取撤销栈长度
   */
  getUndoStackSize(): number {
    return this.undoStack.length;
  }

  /**
   * 获取重做栈长度
   */
  getRedoStackSize(): number {
    return this.redoStack.length;
  }

  /**
   * 订阅状态变化
   */
  subscribe(listener: (state: EditorState) => void): () => void {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }

  /**
   * 通知所有监听器
   */
  private notifyListeners(): void {
    this.listeners.forEach((listener) => listener(this.state));
  }

  /**
   * 判断命令是否可撤销
   */
  private isUndoableCommand(command: Command): boolean {
    return 'type' in command && 'payload' in command;
  }

  /**
   * 序列化历史记录（用于持久化）
   */
  serialize(): string {
    return JSON.stringify({
      undoStack: this.undoStack,
      redoStack: this.redoStack,
    });
  }

  /**
   * 反序列化历史记录
   */
  static deserialize(json: string, initialState: EditorState): HistoryManager {
    const data = JSON.parse(json);
    const manager = new HistoryManager(initialState);

    // 反序列化命令
    manager.undoStack = data.undoStack.map((cmd: any) => {
      const CommandClass = CommandRegistry[cmd.type];
      if (!CommandClass) {
        console.warn(`Unknown command type: ${cmd.type}`);
        return null;
      }
      return CommandClass.fromJSON(cmd);
    }).filter(Boolean);

    manager.redoStack = data.redoStack.map((cmd: any) => {
      const CommandClass = CommandRegistry[cmd.type];
      if (!CommandClass) return null;
      return CommandClass.fromJSON(cmd);
    }).filter(Boolean);

    return manager;
  }

  /**
   * 跳转到指定历史点
   */
  jumpTo(index: number): void {
    if (index < 0 || index >= this.undoStack.length) return;

    // 计算需要撤销/重做的次数
    const steps = this.undoStack.length - 1 - index;

    if (steps > 0) {
      // 撤销多次
      for (let i = 0; i < steps; i++) {
        this.undo();
      }
    } else if (steps < 0) {
      // 重做多次
      for (let i = 0; i < Math.abs(steps); i++) {
        this.redo();
      }
    }
  }

  /**
   * 清除历史记录
   */
  clear(): void {
    this.undoStack = [];
    this.redoStack = [];
    this.notifyListeners();
  }
}
```

```typescript
    historyRef.current!.executeCommand(command);
  }, []);

    canUndo,
    canRedo,
```

```typescript
// snapshot-history.ts - 快照模式实现

import { useState, useCallback, useRef, useEffect } from 'react';

/**
 * 快照历史管理器
 * 每次操作保存完整的应用状态快照
 */
export class SnapshotHistoryManager<T> {
  private past: T[] = [];
  private future: T[] = [];
  private maxSize: number;
  private current: T;

  constructor(initialState: T, maxSize = 50) {
    this.current = initialState;
    this.maxSize = maxSize;
  }

  /**
   * 保存状态快照
   */
  snapshot(state: T): void {
    // 将当前状态推入历史
    this.past.push(this.current);

    // 限制历史大小
    if (this.past.length > this.maxSize) {
      this.past.shift();
    }

    // 更新当前状态
    this.current = state;

    // 清空未来状态（保存新快照后不能重做）
    this.future = [];
  }

  /**
   * 撤销
   */
  undo(): T | null {
    if (this.past.length === 0) return null;

    // 将当前状态推入未来
    this.future.push(this.current);

    // 弹出上一个状态
    const previousState = this.past.pop()!;
    this.current = previousState;

    return this.current;
  }

  /**
   * 重做
   */
  redo(): T | null {
    if (this.future.length === 0) return null;

    // 将当前状态推入历史
    this.past.push(this.current);

    // 弹出下一个状态
    const nextState = this.future.pop()!;
    this.current = nextState;

    return this.current;
  }

  /**
   * 获取当前状态
   */
  getState(): T {
    return this.current;
  }

  /**
   * 是否可以撤销
   */
  canUndo(): boolean {
    return this.past.length > 0;
  }

  /**
   * 是否可以重做
   */
  canRedo(): boolean {
    return this.future.length > 0;
  }

  /**
   * 获取历史长度
   */
  getHistoryLength(): number {
    return this.past.length;
  }

  /**
   * 跳转到指定历史索引
   */
  goTo(index: number): T | null {
    if (index < 0 || index >= this.past.length) return null;

    // 计算需要移动的步数
    const steps = this.past.length - 1 - index;

    if (steps > 0) {
      // 向后移动（撤销）
      for (let i = 0; i < steps; i++) {
        this.future.push(this.past.pop()!);
        this.current = this.past[this.past.length - 1] || this.current;
      }
    }

    return this.current;
  }

  /**
   * 清除历史
   */
  clear(): void {
    this.past = [];
    this.future = [];
  }

  /**
   * 序列化
   */
  serialize(): string {
    return JSON.stringify({
      past: this.past,
      future: this.future,
      current: this.current,
    });
  }

  /**
   * 反序列化
   */
  static deserialize<T>(json: string): SnapshotHistoryManager<T> {
    const data = JSON.parse(json);
    const manager = new SnapshotHistoryManager<T>(data.current);
    manager.past = data.past;
    manager.future = data.future;
    return manager;
  }
}

/**
 * 使用快照历史的 React Hook
 */
export function useSnapshotHistory<T>(
  initialState: T,
  options: {
    maxSize?: number;
    enableDebounce?: boolean;
    debounceMs?: number;
  } = {}
) {
  const { maxSize = 50, enableDebounce = true, debounceMs = 500 } = options;

  const managerRef = useRef<SnapshotHistoryManager<T> | null>(null);
  const debounceRef = useRef<NodeJS.Timeout | null>(null);

  if (!managerRef.current) {
    managerRef.current = new SnapshotHistoryManager(initialState, maxSize);
  }

  const [state, setState] = useState<T>(initialState);
  const [canUndo, setCanUndo] = useState(false);
  const [canRedo, setCanRedo] = useState(false);

  const updateState = useCallback((newState: T) => {
    setState(newState);
    setCanUndo(managerRef.current!.canUndo());
    setCanRedo(managerRef.current!.canRedo());
  }, []);

  /**
   * 更新状态并保存快照
   */
  const setStateWithSnapshot = useCallback(
    (newState: T | ((prev: T) => T)) => {
      const resolvedState =
        typeof newState === 'function'
          ? (newState as (prev: T) => T)(managerRef.current!.getState())
          : newState;

      if (enableDebounce) {
        // 防抖保存
        if (debounceRef.current) {
          clearTimeout(debounceRef.current);
        }
        debounceRef.current = setTimeout(() => {
          managerRef.current!.snapshot(resolvedState);
          updateState(resolvedState);
        }, debounceMs);
      } else {
        managerRef.current!.snapshot(resolvedState);
        updateState(resolvedState);
      }
    },
    [enableDebounce, debounceMs, updateState]
  );

  /**
   * 撤销
   */
  const undo = useCallback(() => {
    const previousState = managerRef.current!.undo();
    if (previousState !== null) {
      updateState(previousState);
      return true;
    }
    return false;
  }, [updateState]);

  /**
   * 重做
   */
  const redo = useCallback(() => {
    const nextState = managerRef.current!.redo();
    if (nextState !== null) {
      updateState(nextState);
      return true;
    }
    return false;
  }, [updateState]);

  /**
   * 清除历史
   */
  const clearHistory = useCallback(() => {
    managerRef.current!.clear();
    setCanUndo(false);
    setCanRedo(false);
  }, []);

  // 清理防抖定时器
  useEffect(() => {
    return () => {
      if (debounceRef.current) {
        clearTimeout(debounceRef.current);
      }
    };
  }, []);

  return {
    state,
    setState: setStateWithSnapshot,
    undo,
    redo,
    canUndo,
    canRedo,
    clearHistory,
    historyLength: managerRef.current.getHistoryLength(),
  };
}
```

```typescript
   */
  track(eventName: string, properties: Record<string, any> = {}): void {
    const event: TrackEvent = {
      event: eventName as EventType,
      timestamp: Date.now(),
      userId: this.userId,
      sessionId: this.sessionId,
      url: window.location.href,
      title: document.title,
      device: this.deviceInfo,
      properties: {
        ...this.config,
        ...properties,
      },
    };

    // 加入队列
    this.queue.push(event);

    // 检查是否需要立即上传
    if (this.queue.length >= this.config.batchSize) {
      this.upload();
    }

    // 调试模式打印
    if (this.config.debug) {
      console.log('[Tracker]', event);
    }
  }

  /**
   * 点击事件
   */
  trackClick(
    element: HTMLElement,
    properties: Record<string, any> = {}
  ): void {
    this.track(EventType.CLICK, {
      ...properties,
      element: this.getElementInfo(element),
    });
  }

  /**
   * 获取元素信息
   */
  private getElementInfo(element: HTMLElement): any {
    return {
      tagName: element.tagName.toLowerCase(),
      id: element.id || undefined,
      className: element.className || undefined,
      text: element.textContent?.slice(0, 100) || undefined,
      href: (element as HTMLAnchorElement).href || undefined,
      // 生成唯一路径
      path: this.generateElementPath(element),
    };
  }

  /**
   * 生成元素路径
   */
  private generateElementPath(element: HTMLElement): string {
    const path: string[] = [];
    let current: HTMLElement | null = element;

    while (current && current !== document.body) {
      let selector = current.tagName.toLowerCase();

      if (current.id) {
        selector += `#${current.id}`;
        path.unshift(selector);
        break;
      } else if (current.className) {
        const classes = current.className
          .split(' ')
          .filter((c) => c)
          .slice(0, 2)
          .join('.');
        if (classes) {
          selector += `.${classes}`;
        }
      }

      // 添加索引
      const parent = current.parentElement;
      if (parent) {
        const children = Array.from(parent.children).filter(
          (child) => child.tagName === current!.tagName
        );
        if (children.length > 1) {
          const index = children.indexOf(current) + 1;
          selector += `:nth-of-type(${index})`;
        }
      }

      path.unshift(selector);
      current = current.parentElement;
    }

    return path.join(' > ');
  }

  /**
   * 性能追踪
   */
  private trackPerformance(): void {
    // 等待页面加载完成
    window.addEventListener('load', () => {
      setTimeout(() => {
        const timing = performance.timing;
        const navigation = performance.getEntriesByType(
          'navigation'
        )[0] as PerformanceNavigationTiming;

        this.track(EventType.PERFORMANCE, {
          // 页面加载时间
          loadTime: timing.loadEventEnd - timing.navigationStart,
          // DOM 构建时间
          domReadyTime: timing.domContentLoadedEventEnd - timing.navigationStart,
          // 首次绘制时间
          firstPaint: (performance as any).getEntriesByType('paint')[0]?.startTime,
          // 首屏渲染时间
          firstContentfulPaint: (performance as any)
            .getEntriesByType('paint')
            ?.find((e: any) => e.name === 'first-contentful-paint')?.startTime,
          // DNS 解析时间
          dnsTime: timing.domainLookupEnd - timing.domainLookupStart,
          // TCP 连接时间
          tcpTime: timing.connectEnd - timing.connectStart,
          // SSL 建立时间
          sslTime: timing.secureConnectionStart
            ? timing.connectEnd - timing.secureConnectionStart
            : 0,
          // 请求响应时间
          responseTime: timing.responseEnd - timing.requestStart,
          // 内容传输时间
          transferTime: timing.responseEnd - timing.responseStart,
          // DOM 解析时间
          domParseTime: timing.domInteractive - timing.domLoading,
          // 资源加载时间
          resourceLoadTime: timing.loadEventEnd - timing.domContentLoadedEventEnd,
          // 页面大小
          pageSize: Math.round(
            (navigation?.transferSize || 0) / 1024
          ),
        });
      }, 0);
    });
  }

  /**
   * 错误追踪
   */
  private trackError(): void {
    // 监听 JS 错误
    window.addEventListener('error', (event) => {
      this.track(EventType.ERROR, {
        type: 'javascript',
        message: event.message,
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno,
        stack: (event as any).error?.stack,
      });
    });

    // 监听 Promise 拒绝
    window.addEventListener('unhandledrejection', (event) => {
      this.track(EventType.ERROR, {
        type: 'promise',
        message: event.reason?.message || event.reason,
        stack: event.reason?.stack,
      });
    });

    // 监听资源加载失败
    window.addEventListener('error', (event) => {
      if (event.target !== window) {
        const target = event.target as HTMLScriptElement | HTMLLinkElement;
        this.track(EventType.ERROR, {
          type: 'resource',
          url: target.src || target.href,
          tagName: target.tagName,
        });
      }
    }, true);
  }

  /**
   * 监听页面跳转
   */
  private setupPageViewListeners(): void {
    // 监听 popstate（浏览器前进后退）
    window.addEventListener('popstate', () => {
      this.trackPageView();
    });

    // 覆盖 pushState
    const originalPushState = window.history.pushState;
    window.history.pushState = (...args) => {
      originalPushState.apply(window.history, args);
      this.trackPageView();
    };

    // 覆盖 replaceState
    const originalReplaceState = window.history.replaceState;
    window.history.replaceState = (...args) => {
      originalReplaceState.apply(window.history, args);
      this.trackPageView();
    };
  }

  /**
   * 监听网络状态
   */
  private setupNetworkListeners(): void {
    window.addEventListener('online', () => {
      this.isOnline = true;
      this.upload();
    });

    window.addEventListener('offline', () => {
      this.isOnline = false;
    });
  }

  /**
   * 启动定时上传
   */
  private startUploadTimer(): void {
    this.uploadTimer = setInterval(() => {
      if (this.queue.length > 0 && this.isOnline) {
        this.upload();
      }
    }, this.config.uploadInterval);
  }

  /**
   * 页面卸载时上传
   */
  private setupBeforeUnload(): void {
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden' && this.queue.length > 0) {
        this.upload(true);
      }
    });

    window.addEventListener('beforeunload', () => {
      if (this.queue.length > 0) {
        this.upload(true);
      }
    });
  }

  /**
   * 上传数据
   */
  private upload(isImmediate = false): void {
    if (this.queue.length === 0) return;

    const data = this.queue.splice(0, this.config.batchSize);

    // 使用 sendBeacon
    if (isImmediate && navigator.sendBeacon) {
      const blob = new Blob([JSON.stringify(data)], {
        type: 'application/json',
      });
      navigator.sendBeacon(this.config.serverUrl, blob);
      return;
    }

    // 使用 fetch
    fetch(this.config.serverUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(data),
      keepalive: true,
    }).catch((error) => {
      // 上传失败，放回队列
      this.queue.unshift(...data);
      if (this.config.debug) {
        console.error('[Tracker] Upload failed:', error);
      }
    });
  }

  /**
   * 销毁
   */
  destroy(): void {
    if (this.uploadTimer) {
      clearInterval(this.uploadTimer);
    }
    // 上传剩余数据
    this.upload(true);
  }
}
```

```typescript
// visual-tracker.ts - 可视化埋点实现

import { Tracker, TrackEvent } from './tracker';

/**
 * 可视化埋点配置
 */
export interface VisualTrackerConfig {
  // 埋点 SDK 实例
  tracker: Tracker;
  // 埋点标记类名
  markerClassName: string;
  // 自动标记的元素选择器
  autoTrackSelectors: string[];
  // 排除的元素选择器
  excludeSelectors: string[];
  // 是否启用
  enabled: boolean;
}

/**
 * 埋点标记
 */
interface TrackMarker {
  // 事件名称
  eventName: string;
  // 事件属性
  properties: Record<string, any>;
}

/**
 * 可视化埋点类
 */
export class VisualTracker {
  private config: VisualTrackerConfig;
  private markers: Map<HTMLElement, TrackMarker> = new Map();
  private clickHandler: (event: MouseEvent) => void;

  constructor(config: VisualTrackerConfig) {
    this.config = {
      markerClassName: 'track-marker',
      autoTrackSelectors: ['button', 'a', '.trackable'],
      excludeSelectors: ['[data-track-ignore]'],
      enabled: true,
      ...config,
    };

    this.clickHandler = this.handleClick.bind(this);
    this.init();
  }

  /**
   * 初始化
   */
  private init(): void {
    if (!this.config.enabled) return;

    // 标记已存在的元素
    this.markElements();

    // 监听 DOM 变化，标记新元素
    this.observeDOM();

    // 绑定点击事件
    document.addEventListener('click', this.clickHandler, true);
  }

  /**
   * 标记元素
   */
  private markElements(): void {
    const selector = this.config.autoTrackSelectors.join(',');
    const elements = document.querySelectorAll(selector);

    elements.forEach((element) => {
      if (this.shouldTrack(element as HTMLElement)) {
        this.markElement(element as HTMLElement);
      }
    });
  }

  /**
   * 标记单个元素
   */
  markElement(element: HTMLElement, eventName?: string, properties?: Record<string, any>): void {
    // 生成事件名称
    const event = eventName || this.generateEventName(element);

    // 生成属性
    const props = properties || this.generateProperties(element);

    // 存储标记
    this.markers.set(element, {
      eventName: event,
      properties: props,
    });

    // 添加标记类名
    element.classList.add(this.config.markerClassName);
  }

  /**
   * 移除标记
   */
  unmarkElement(element: HTMLElement): void {
    this.markers.delete(element);
    element.classList.remove(this.config.markerClassName);
  }

  /**
   * 判断是否应该追踪
   */
  private shouldTrack(element: HTMLElement): boolean {
    // 检查排除选择器
    for (const selector of this.config.excludeSelectors) {
      if (element.matches(selector) || element.closest(selector)) {
        return false;
      }
    }

    // 检查是否有 data-track-ignore 属性
    if (element.hasAttribute('data-track-ignore')) {
      return false;
    }

    return true;
  }

  /**
   * 生成事件名称
   */
  private generateEventName(element: HTMLElement): string {
    const tagName = element.tagName.toLowerCase();

    // 优先使用 data-track-event 属性
    if (element.hasAttribute('data-track-event')) {
      return element.getAttribute('data-track-event')!;
    }

    // 生成默认事件名
    const id = element.id ? `#${element.id}` : '';
    const className = element.className
      .split(' ')
      .filter((c) => c && c !== this.config.markerClassName)
      .slice(0, 2)
      .join('.');

    return `${tagName}${id}${className ? '.' + className : ''}_click`;
  }

  /**
   * 生成事件属性
   */
  private generateProperties(element: HTMLElement): Record<string, any> {
    const props: Record<string, any> = {};

    // 从 data-track-properties 读取 JSON
    if (element.hasAttribute('data-track-properties')) {
      try {
        const json = element.getAttribute('data-track-properties');
        Object.assign(props, JSON.parse(json!));
      } catch (e) {
        console.warn('[VisualTracker] Invalid JSON in data-track-properties');
      }
    }

    // 添加元素信息
    if (element.id) {
      props.elementId = element.id;
    }

    if (element.className) {
      props.elementClass = element.className;
    }

    if (element.textContent) {
      props.elementText = element.textContent.slice(0, 50);
    }

    if (element.tagName === 'A') {
      const anchor = element as HTMLAnchorElement;
      props.href = anchor.href;
      props.target = anchor.target;
    }

    return props;
  }

  /**
   * 处理点击事件
   */
  private handleClick(event: MouseEvent): void {
    const target = event.target as HTMLElement;

    // 找到最近的已标记元素
    const markedElement = target.closest(`.${this.config.markerClassName}`) as HTMLElement;

    if (markedElement && this.markers.has(markedElement)) {
      const marker = this.markers.get(markedElement)!;

      // 上报事件
      this.config.tracker.track(marker.eventName, {
        ...marker.properties,
        // 添加点击位置信息
        clickX: event.clientX,
        clickY: event.clientY,
        // 添加时间戳
        timestamp: Date.now(),
      });

      event.stopPropagation();
    }
  }

  /**
   * 监听 DOM 变化
   */
  private observeDOM(): void {
    const observer = new MutationObserver((mutations) => {
      mutations.forEach((mutation) => {
        // 处理新增节点
        mutation.addedNodes.forEach((node) => {
          if (node.nodeType === Node.ELEMENT_NODE) {
            const element = node as HTMLElement;
            const selector = this.config.autoTrackSelectors.join(',');

            if (element.matches(selector) && this.shouldTrack(element)) {
              this.markElement(element);
            }

            // 检查子元素
            const children = element.querySelectorAll(selector);
            children.forEach((child) => {
              if (this.shouldTrack(child as HTMLElement)) {
                this.markElement(child as HTMLElement);
              }
            });
          }
        });

        // 处理移除节点
        mutation.removedNodes.forEach((node) => {
          if (node.nodeType === Node.ELEMENT_NODE) {
            const element = node as HTMLElement;
            this.unmarkElement(element);
          }
        });
      });
    });

    observer.observe(document.body, {
      childList: true,
      subtree: true,
    });
  }

  /**
   * 销毁
   */
  destroy(): void {
    document.removeEventListener('click', this.clickHandler, true);
    this.markers.clear();
  }
}
```

```tsx
// useTrack.ts - React 埋点 Hook

import { useEffect, useCallback, useRef } from 'react';
import { track, trackPageView, TrackEvent } from './tracker';

/**
 * 埋点 Hook 配置
 */
export interface UseTrackOptions {
  // 页面名称
  pageName?: string;
  // 自动追踪页面浏览
  autoTrackPageView?: boolean;
  // 自动追踪性能
  autoTrackPerformance?: boolean;
  // 自动追踪错误
  autoTrackError?: boolean;
}

/**
 * 埋点 Hook
 */
export function useTrack(options: UseTrackOptions = {}) {
  const {
    pageName,
    autoTrackPageView = true,
    autoTrackPerformance = true,
    autoTrackError = true,
  } = options;

  const pageNameRef = useRef(pageName);

  // 更新页面名称
  useEffect(() => {
    if (pageName) {
      pageNameRef.current = pageName;
    }
  }, [pageName]);

  // 自动追踪页面浏览
  useEffect(() => {
    if (autoTrackPageView && pageNameRef.current) {
      trackPageView({
        pageName: pageNameRef.current,
      });
    }
  }, [autoTrackPageView]);

  /**
   * 追踪事件
   */
  const trackEvent = useCallback(
    (eventName: string, properties?: Record<string, any>) => {
      track(eventName, {
        ...properties,
        pageName: pageNameRef.current,
      });
    },
    []
  );

  /**
   * 追踪点击
   */
  const trackClick = useCallback(
    (element: HTMLElement, properties?: Record<string, any>) => {
      track('click', {
        ...properties,
        pageName: pageNameRef.current,
      });
    },
    []
  );

  /**
   * 追踪曝光
   */
  const trackExposure = useCallback(
    (elementId: string, properties?: Record<string, any>) => {
      track('exposure', {
        ...properties,
        elementId,
        pageName: pageNameRef.current,
      });
    },
    []
  );

  return {
    track: trackEvent,
    trackClick,
    trackExposure,
  };
}

/**
 * 自动曝光追踪 Hook
 */
export function useExposureTracker(
  ref: React.RefObject<HTMLElement>,
  eventName: string,
  properties?: Record<string, any>
) {
  const hasTracked = useRef(false);

  useEffect(() => {
    const element = ref.current;
    if (!element || hasTracked.current) return;

    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting && !hasTracked.current) {
            hasTracked.current = true;
            track(eventName, {
              ...properties,
              exposureTime: Date.now(),
            });
            observer.disconnect();
          }
        });
      },
      {
        threshold: 0.5, // 50% 可见
      }
    );

    observer.observe(element);

    return () => {
      observer.disconnect();
    };
  }, [ref, eventName, properties]);
}

/**
 * 自动点击追踪 Hook
 */
export function useClickTracker(
  ref: React.RefObject<HTMLElement>,
  eventName: string,
  properties?: Record<string, any>
) {
  useEffect(() => {
    const element = ref.current;
    if (!element) return;

    const handleClick = () => {
      track(eventName, properties);
    };

    element.addEventListener('click', handleClick);

    return () => {
      element.removeEventListener('click', handleClick);
    };
  }, [ref, eventName, properties]);
}
```

### 4.2 路由权限管理实现

```typescript
// permission-manager.ts - 权限管理

import {
  Permission,
  Role,
  User,
  RouteConfig,
  RouteMeta,
  PermissionType,
} from './permission';

/**
 * 权限管理器
 */
export class PermissionManager {
  private user: User | null = null;
  private permissionCache: Set<string> = new Set();
  private roleCache: Set<string> = new Set();

  /**
   * 设置当前用户
   */
  setUser(user: User | null): void {
    this.user = user;
    this.refreshCache();
  }

  /**
   * 获取当前用户
   */
  getUser(): User | null {
    return this.user;
  }

  /**
   * 刷新权限缓存
   */
  private refreshCache(): void {
    this.permissionCache.clear();
    this.roleCache.clear();

    if (!this.user) return;

    // 缓存角色
    this.user.roles.forEach((role) => {
      this.roleCache.add(role.code);
    });

    // 缓存权限
    this.user.permissions.forEach((perm) => {
      this.permissionCache.add(perm.code);
    });
  }

  /**
   * 检查是否有权限
   */
  hasPermission(permission: string | string[]): boolean {
    if (!this.user) return false;

    // 超级管理员拥有所有权限
    if (this.isAdmin()) return true;

    const permissions = Array.isArray(permission)
      ? permission
      : [permission];

    return permissions.some((perm) => this.permissionCache.has(perm));
  }

  /**
   * 检查是否有角色
   */
  hasRole(role: string | string[]): boolean {
    if (!this.user) return false;

    // 超级管理员拥有所有角色
    if (this.isAdmin()) return true;

    const roles = Array.isArray(role) ? role : [role];

    return roles.some((r) => this.roleCache.has(r));
  }

  /**
   * 是否为管理员
   */
  isAdmin(): boolean {
    if (!this.user) return false;

    return (
      this.user.roles.some((role) => role.code === 'admin') ||
      this.user.permissions.some((perm) => perm.code === '*')
    );
  }

  /**
   * 过滤路由 - 返回有权限的路由
   */
  filterRoutes(routes: RouteConfig[]): RouteConfig[] {
    return routes
      .map((route) => this.filterRoute(route))
      .filter((route) => route !== null) as RouteConfig[];
  }

  /**
   * 过滤单个路由
   */
  private filterRoute(route: RouteConfig): RouteConfig | null {
    // 检查角色权限
    if (route.meta?.roles && route.meta.roles.length > 0) {
      const hasRole = route.meta.roles.some((role) =>
        this.roleCache.has(role)
      );
      if (!hasRole) return null;
    }

    // 检查权限代码
    if (route.meta?.permission) {
      const permission = route.meta.permission;
      const hasPermission = this.hasPermission(permission);
      if (!hasPermission) return null;
    }

    // 过滤子路由
    const filteredChildren = route.children
      ?.map((child) => this.filterRoute(child))
      .filter((child) => child !== null) as RouteConfig[] | undefined;

    const filteredRoute: RouteConfig = {
      ...route,
      children: filteredChildren,
    };

    // 如果是叶子路由且没有权限，返回 null
    if (
      !filteredRoute.component &&
      (!filteredChildren || filteredChildren.length === 0)
    ) {
      return route.meta?.visible === false ? null : route;
    }

    return filteredRoute;
  }

  /**
   * 获取用户可访问的菜单
   */
  getAccessibleMenus(routes: RouteConfig[]): RouteConfig[] {
    const accessibleRoutes = this.filterRoutes(routes);

    // 过滤掉隐藏的菜单
    return accessibleRoutes.filter((route) => {
      if (route.meta?.visible === false) return false;
      return true;
    });
  }

  /**
   * 权限检查函数 - 用于 Vue/React 组件
   */
  permissionChecker(permission: string | string[]): boolean {
    return this.hasPermission(permission);
  }

  /**
   * 角色检查函数
   */
  roleChecker(role: string | string[]): boolean {
    return this.hasRole(role);
  }
}
```

```typescript
      } else {
        next(`/login?redirect=${to.path}`);
      }
      return;
    }

    // 已登录，尝试访问登录页
    if (to.path === '/login') {
      next({ path: '/' });
      return;
    }

    // 获取用户信息
    const user = store?.getters['user/userInfo'];

    // 用户信息不存在，尝试获取
    if (!user) {

        await store.dispatch('user/getUserInfo');

        return;

    const hasPermission = checkRoutePermission(to);

      next({ path: '/403', replace: true });
      return;

    if (!hasRole) {
      return false;
    }

  // 检查权限代码
  const requiredPermission = to.meta?.permission;
  if (requiredPermission) {
    const hasPermission = permissionManager.hasPermission(requiredPermission);

      return false;
    }
```

```typescript
    this.sessionId = this.generateSessionId();

    // 获取设备信息
    this.deviceInfo = this.getDeviceInfo();

    // 获取页面信息
    this.pageInfo = this.getPageInfo();

    // 初始化
    this.init();
  }

  /**
   * 生成会话 ID
   */
  private generateSessionId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  /**
   * 获取设备信息
   */
  private getDeviceInfo(): DeviceInfo {
    const ua = navigator.userAgent;
    let os = 'Unknown';
    let osVersion = 'Unknown';

    if (/Windows/.test(ua)) {
      os = 'Windows';
      osVersion = /Windows NT ([\d.]+)/.exec(ua)?.[1] || 'Unknown';
    } else if (/Mac OS X/.test(ua)) {
      os = 'macOS';
      osVersion = /Mac OS X ([\d._]+)/.exec(ua)?.[1]?.replace(/_/g, '.') || 'Unknown';
    } else if (/Android/.test(ua)) {
      os = 'Android';
      osVersion = /Android ([\d.]+)/.exec(ua)?.[1] || 'Unknown';
    } else if (/iPhone|iPad|iPod/.test(ua)) {
      os = 'iOS';
      osVersion = /OS ([\d_]+)/.exec(ua)?.[1]?.replace(/_/g, '.') || 'Unknown';
    }

    let browser = 'Unknown';
    let browserVersion = 'Unknown';
    if (/Chrome/.test(ua) && !/Edge/.test(ua)) {
      browser = 'Chrome';
      browserVersion = /Chrome\/([\d.]+)/.exec(ua)?.[1] || 'Unknown';
    } else if (/Safari/.test(ua) && !/Chrome/.test(ua)) {
      browser = 'Safari';
      browserVersion = /Version\/([\d.]+)/.exec(ua)?.[1] || 'Unknown';
    } else if (/Firefox/.test(ua)) {
      browser = 'Firefox';
      browserVersion = /Firefox\/([\d.]+)/.exec(ua)?.[1] || 'Unknown';
    } else if (/Edge/.test(ua)) {
      browser = 'Edge';
      browserVersion = /Edge\/([\d.]+)/.exec(ua)?.[1] || 'Unknown';
    }

    let deviceType: 'mobile' | 'tablet' | 'desktop' = 'desktop';
    if (/Mobile|Android|iPhone|iPod/.test(ua)) {
      deviceType = 'mobile';
    } else if (/Tablet|iPad/.test(ua)) {
      deviceType = 'tablet';
    }

    const connection = (navigator as any).connection;

    return {
      os,
      osVersion,
      browser,
      browserVersion,
      screenWidth: window.screen.width,
      screenHeight: window.screen.height,
      deviceType,
      language: navigator.language,
      networkType: connection?.effectiveType,
      memory: (navigator as any).deviceMemory,
    };
  }

  /**
   * 获取页面信息
   */
  private getPageInfo(): PageInfo {
    return {
      url: window.location.href,
      title: document.title,
      referrer: document.referrer,
      startTime: Date.now(),
    };
  }

  /**
   * 初始化
   */
  private init(): void {
    if (!this.config.enabled) return;

    // 采样判断
    if (Math.random() > this.config.sampleRate) {
      console.log('[Monitor] Sampling - disabled');
      return;
    }

    // 自动上报错误
    if (this.config.autoReportError) {
      this.setupErrorListeners();
    }

    // 自动上报性能
    if (this.config.autoReportPerformance) {
      this.setupPerformanceListeners();
    }

    // 自动上报 API
    if (this.config.autoReportAPI) {
      this.setupAPIListeners();
    }

    // 监听网络状态
    this.setupNetworkListeners();

    // 启动定时上传
    this.startUploadTimer();

    // 页面卸载时上传
    this.setupBeforeUnload();
  }

  /**
   * 设置错误监听
   */
  private setupErrorListeners(): void {
    // 监听 JS 错误
    window.addEventListener('error', (event) => {
      // 排除资源加载错误
      if (event.target !== window) {
        return;
      }

      const errorData: JSErrorData = {
        type: 'error',
        message: event.message,
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno,
        stack: (event as any).error?.stack,
      };

      this.report(MetricType.JS_ERROR, errorData);
    });

    // 监听 Promise 拒绝
    window.addEventListener('unhandledrejection', (event) => {
      const errorData: JSErrorData = {
        type: 'unhandledrejection',
        message: event.reason?.message || String(event.reason),
        stack: event.reason?.stack,
      };

      this.report(MetricType.UNHANDLED_REJECTION, errorData);
    });
  }

  /**
   * 设置性能监听
   */
  private setupPerformanceListeners(): void {
    // 页面加载完成后上报性能数据
    window.addEventListener('load', () => {
      setTimeout(() => {
        this.reportPerformance();
      }, 0);
    });

    // 监听 LCP 变化
    if ('PerformanceObserver' in window) {
      try {
        const lcpObserver = new PerformanceObserver((list) => {
          const entries = list.getEntries();
          const lastEntry = entries[entries.length - 1] as any;
          this.report(MetricType.PERFORMANCE, {
            url: window.location.href,
            timestamp: Date.now(),
            lcp: lastEntry.renderTime || lastEntry.loadTime,
          });
        });
        lcpObserver.observe({ entryType: 'largest-contentful-paint' });
      } catch (e) {
        console.warn('[Monitor] LCP observer not supported');
      }

      // 监听 CLS
      try {
        let clsScore = 0;
        const clsObserver = new PerformanceObserver((list) => {
          for (const entry of list.getEntries()) {
            if (!(entry as any).hadRecentInput) {
              clsScore += (entry as any).value;
            }
          }
        });
        clsObserver.observe({ entryType: 'layout-shift' });
      } catch (e) {
        console.warn('[Monitor] CLS observer not supported');
      }
    }
  }

  /**
   * 上报性能数据
   */
  private reportPerformance(): void {
    const timing = performance.timing;
    const navigation = performance.getEntriesByType(
      'navigation'
    )[0] as PerformanceNavigationTiming;

    const perfData: PerformanceData = {
      url: window.location.href,
      timestamp: Date.now(),
      loadTime: timing.loadEventEnd - timing.navigationStart,
      domReadyTime: timing.domContentLoadedEventEnd - timing.navigationStart,
      ttfb: timing.responseStart - timing.requestStart,
      dnsTime: timing.domainLookupEnd - timing.domainLookupStart,
      tcpTime: timing.connectEnd - timing.connectStart,
      responseTime: timing.responseEnd - timing.requestStart,
      domParseTime: timing.domInteractive - timing.domLoading,
      pageSize: Math.round((navigation?.transferSize || 0) / 1024),
    };

    // 检查是否超过阈值
    if (perfData.loadTime && perfData.loadTime > this.config.performanceThreshold) {
      this.report(MetricType.PERFORMANCE, perfData);
    }
  }

  /**
   * 设置 API 监听
   */
  private setupAPIListeners(): void {
    // 拦截 fetch
    const originalFetch = window.fetch;
    window.fetch = async (...args) => {
      const startTime = Date.now();
      const url = typeof args[0] === 'string' ? args[0] : args[0]?.url;
      const method = args[1]?.method || 'GET';

      try {
        const response = await originalFetch(...args);
        const duration = Date.now() - startTime;
        const clone = response.clone();

        // 读取响应大小
        let responseSize = 0;
        try {
          const blob = await clone.blob();
          responseSize = Math.round(blob.size / 1024);
        } catch (e) {
          // 忽略
        }

        const apiData: APIRequestData = {
          url,
          method,
          status: response.status,
          duration,
          responseSize,
          success: response.ok,
          timestamp: startTime,
        };

        // 慢请求上报
        if (duration > this.config.slowRequestThreshold) {
          this.report(MetricType.SLOW_REQUEST, apiData);
        }

        return response;
      } catch (error: any) {
        const duration = Date.now() - startTime;

        const apiData: APIRequestData = {
          url,
          method,
          status: 0,
          duration,
          success: false,
          errorMessage: error?.message || 'Network Error',
          timestamp: startTime,
        };

        this.report(MetricType.API_ERROR, apiData);
        throw error;
      }
    };

    // 拦截 XMLHttpRequest
    const originalXHROpen = XMLHttpRequest.prototype.open;
    const originalXHRSend = XMLHttpRequest.prototype.send;

    XMLHttpRequest.prototype.open = function (
      method: string,
      url: string,
      ...rest: any[]
    ) {
      (this as any)._monitor_url = url;
      (this as any)._monitor_method = method;
      (this as any)._monitor_startTime = Date.now();
      return originalXHROpen.apply(this, [method, url, ...rest]);
    };

    XMLHttpRequest.prototype.send = function (...args: any[]) {
      const xhr = this as XMLHttpRequest;
      const startTime = (xhr as any)._monitor_startTime;

      xhr.addEventListener('loadend', () => {
        const duration = Date.now() - startTime;
        const responseHeaders = xhr.getAllResponseHeaders();
        const contentLength = responseHeaders.match(/content-length:\s*(\d+)/i);

        const apiData: APIRequestData = {
          url: (xhr as any)._monitor_url,
          method: (xhr as any)._monitor_method,
          status: xhr.status,
          duration,
          responseSize: contentLength
            ? Math.round(parseInt(contentLength[1]) / 1024)
            : 0,
          success: xhr.status >= 200 && xhr.status < 300,
          timestamp: startTime,
        };

        // 慢请求或错误上报
        if (duration > this.config.slowRequestThreshold) {
          this.report(MetricType.SLOW_REQUEST, apiData);
        } else if (!apiData.success) {
          this.report(MetricType.API_ERROR, apiData);
        }
      });

      return originalXHRSend.apply(this, args);
    };
  }

  /**
   * 设置网络监听
   */
  private setupNetworkListeners(): void {
    window.addEventListener('online', () => {
      this.isOnline = true;
      this.upload();
    });

    window.addEventListener('offline', () => {
      this.isOnline = false;
    });
  }

  /**
   * 启动定时上传
   */
  private startUploadTimer(): void {
    this.uploadTimer = setInterval(() => {
      if (this.queue.length > 0 && this.isOnline) {
        this.upload();
      }
    }, this.config.uploadInterval);
  }

  /**
   * 页面卸载时上传
   */
  private setupBeforeUnload(): void {
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden' && this.queue.length > 0) {
        this.upload(true);
      }
    });

    window.addEventListener('beforeunload', () => {
      if (this.queue.length > 0) {
        this.upload(true);
      }
    });
  }

  /**
   * 上报事件
   */
  report(type: MetricType, data: any): void {
    const event: MonitorEvent = {
      id: this.generateEventId(),
      type,
      data,
      userId: this.userId,
      sessionId: this.sessionId,
      device: this.deviceInfo,
      page: this.pageInfo,
      timestamp: Date.now(),
    };

    this.queue.push(event);

    // 检查是否需要立即上传
    if (this.queue.length >= this.config.batchSize) {
      this.upload();
    }
  }

  /**
   * 生成事件 ID
   */
  private generateEventId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  /**
   * 设置用户 ID
   */
  setUserId(userId: string): void {
    this.userId = userId;
  }

  /**
   * 清除用户 ID
   */
  clearUserId(): void {
    this.userId = undefined;
  }

  /**
   * 上传数据
   */
  private upload(isImmediate = false): void {
    if (this.queue.length === 0) return;

    const data = this.queue.splice(0, this.config.batchSize);

    // 使用 sendBeacon
    if (isImmediate && navigator.sendBeacon) {
      const blob = new Blob([JSON.stringify(data)], {
        type: 'application/json',
      });
      navigator.sendBeacon(this.config.serverUrl, blob);
      return;
    }

    // 使用 fetch
    fetch(this.config.serverUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(data),
      keepalive: true,
    }).catch((error) => {
      // 上传失败，放回队列
      this.queue.unshift(...data);
      console.error('[Monitor] Upload failed:', error);
    });
  }

  /**
   * 销毁
   */
  destroy(): void {
    if (this.uploadTimer) {
      clearInterval(this.uploadTimer);
    }
    this.upload(true);
  }
}

// 创建全局实例
let monitorInstance: Monitor | null = null;
```

```typescript
import { InjectionKey } from 'vue';

// 模块类型
import { userStore, UserState } from './modules/user';
import { appStore, AppState } from './modules/app';

 * 根状态类型

  app: AppState;
  permission: PermissionState;

// 创建 injection key
export const storeKey: InjectionKey<Store<RootState>> = Symbol('store');

/**
 * 创建 store
 */
export function createAppStore() {
  const store = createStore<RootState>({
    // 开启严格模式
    strict: process.env.NODE_ENV !== 'production',

    // 状态
    state: {
      user: {},
      app: {},
      permission: {},
    },

    // 模块
    modules: {
      user: userStore,
      app: appStore,
      permission: permissionStore,
    },

    // 插件
    plugins: [
      // 持久化插件
      createPersistPlugin(),
      // 日志插件（开发环境）
      ...(process.env.NODE_ENV !== 'production'
        ? [createLoggerPlugin()]
        : []),
    ],
  });

  return store;
}

/**
 * 持久化插件
 */
function createPersistPlugin() {
  return (store: Store<RootState>) => {
    // 读取本地存储
    const storedState = localStorage.getItem('vuex-persist');
    if (storedState) {
      try {
        const parsed = JSON.parse(storedState);
        // 恢复需要持久化的状态
        if (parsed.user) {
          store.commit('user/RESTORE_STATE', parsed.user);
        }
      } catch (e) {
        console.error('Failed to restore vuex state:', e);
      }
    }

    // 监听状态变化，保存到本地存储
    store.subscribe((mutation, state) => {
      const persistState = {
        user: {
          token: state.user.token,
          userInfo: state.user.userInfo,
        },
      };
      localStorage.setItem('vuex-persist', JSON.stringify(persistState));
    });
  };
}

/**
 * 日志插件（开发环境）
 */
function createLoggerPlugin() {
  return (store: Store<RootState>) => {
    store.subscribe((mutation, state) => {
      console.log('[Vuex Mutation]', mutation.type, mutation.payload);
      console.log('[Vuex State]', state);
    });
  };
}

// 重新导出 store 实例
export const store = createAppStore();

// 类型化的 useStore
export function useStore(): Store<RootState> {
  return useVuexStore(storeKey);
}
```

```typescript
// store/modules/user.ts - 用户状态模块

import { Module } from 'vuex';
import { RootState } from '../index';
import { User, LoginParams, LoginResult } from '@/types/user';
import { userService } from '@/services/user';

/**
 * 用户状态
 */
export interface UserState {
  // 用户信息
  userInfo: User | null;
  // Token
  token: string | null;
  // 用户角色
  roles: string[];
  // 用户权限
  permissions: string[];
  // 加载状态
  loading: boolean;
}

/**
 * 用户模块
 */
export const userStore: Module<UserState, RootState> = {
  namespaced: true,

  state: {
    userInfo: null,
    token: localStorage.getItem('token') || null,
    roles: [],
    permissions: [],
    loading: false,
  },

  getters: {
    // 获取用户信息
    userInfo: (state) => state.userInfo,
    // 获取 Token
    token: (state) => state.token,
    // 是否已登录
    isLoggedIn: (state) => !!state.token,
    // 获取用户角色
    roles: (state) => state.roles,
    // 获取用户权限
    permissions: (state) => state.permissions,
    // 获取用户名称
    username: (state) => state.userInfo?.username || '',
    // 获取用户头像
    avatar: (state) => state.userInfo?.avatar || '',
    // 是否是管理员
    isAdmin: (state) => state.roles.includes('admin'),
  },

  mutations: {
    // 设置用户信息
    SET_USER_INFO(state, userInfo: User | null) {
      state.userInfo = userInfo;
    },

    // 设置 Token
    SET_TOKEN(state, token: string | null) {
      state.token = token;
      if (token) {
        localStorage.setItem('token', token);
      } else {
        localStorage.removeItem('token');
      }
    },

    // 设置角色
    SET_ROLES(state, roles: string[]) {
      state.roles = roles;
    },

    // 设置权限
    SET_PERMISSIONS(state, permissions: string[]) {
      state.permissions = permissions;
    },

    // 设置加载状态
    SET_LOADING(state, loading: boolean) {
      state.loading = loading;
    },

    // 恢复状态（持久化）
    RESTORE_STATE(state, savedState: Partial<UserState>) {
      if (savedState.token) {
        state.token = savedState.token;
      }
      if (savedState.userInfo) {
        state.userInfo = savedState.userInfo;
      }
    },
  },

  actions: {
    /**
     * 登录
     */
    async login({ commit }, params: LoginParams) {
      commit('SET_LOADING', true);
      try {
        const result = await userService.login(params);
        commit('SET_TOKEN', result.token);
        commit('SET_USER_INFO', result.user);
        return result;
      } finally {
        commit('SET_LOADING', false);
      }
    },

    /**
     * 获取用户信息
     */
    async getUserInfo({ commit, state }) {
      if (!state.token) {
        throw new Error('No token');
      }

      commit('SET_LOADING', true);
      try {
        const userInfo = await userService.getUserInfo();
        // 从用户信息中提取角色和权限
        const roles = userInfo.roles?.map((r) => r.code) || [];
        const permissions = userInfo.permissions?.map((p) => p.code) || [];

        commit('SET_USER_INFO', userInfo);
```

```typescript
        style={{
          position: 'absolute',
          top: i * itemHeight,
          height: itemHeight,
          left: 0,
          right: 0,

      </div>
    );

      ref={containerRef}
      style={{
        height,

        position: 'relative',
      }}
```

```typescript
/**
 * 缓存工具 - 内存缓存 + 持久化
 */
class CacheManager<T> {
  private cache: Map<string, { value: T; expire: number }> = new Map();
  private storageKey: string;
  private defaultExpire: number;

  constructor(storageKey: string, defaultExpire = 5 * 60 * 1000) {
    this.storageKey = storageKey;
    this.defaultExpire = defaultExpire;
    this.loadFromStorage();
  }

  /**
   * 设置缓存
   */
  set(key: string, value: T, expire?: number): void {
    const expireTime = Date.now() + (expire || this.defaultExpire);
    this.cache.set(key, { value, expire: expireTime });
    this.saveToStorage();
  }

  /**
   * 获取缓存
   */
  get(key: string): T | null {
    const item = this.cache.get(key);
    if (!item) return null;

    // 检查过期
    if (Date.now() > item.expire) {
      this.cache.delete(key);
      this.saveToStorage();
      return null;
    }

    return item.value;
  }

  /**
   * 删除缓存
   */
  delete(key: string): void {
    this.cache.delete(key);
    this.saveToStorage();
  }

  /**
   * 清空缓存
   */
  clear(): void {
    this.cache.clear();
    this.saveToStorage();
  }

  /**
   * 从存储加载
   */
  private loadFromStorage(): void {
    try {
      const stored = localStorage.getItem(this.storageKey);
      if (stored) {
        const data = JSON.parse(stored);
        Object.entries(data).forEach(([key, value]: [string, any]) => {
          if (value.expire > Date.now()) {
            this.cache.set(key, value);
          }
        });
      }
    } catch (e) {
      console.error('Failed to load cache from storage:', e);
    }
  }

  /**
   * 保存到存储
   */
  private saveToStorage(): void {
    try {
      const data: Record<string, any> = {};
      this.cache.forEach((value, key) => {
        data[key] = value;
      });
      localStorage.setItem(this.storageKey, JSON.stringify(data));
    } catch (e) {
      console.error('Failed to save cache to storage:', e);
    }
  }
}

/**
 * 防抖函数
 */
export function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number,
  immediate = false
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout | null = null;

  return function (this: any, ...args: Parameters<T>) {
    const context = this;

    const later = () => {
      timeout = null;
      if (!immediate) {
        func.apply(context, args);
      }
    };

    const callNow = immediate && !timeout;

    if (timeout) {
      clearTimeout(timeout);
    }

    timeout = setTimeout(later, wait);

    if (callNow) {
      func.apply(context, args);
    }
  };
}

/**
 * 节流函数
 */
export function throttle<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout | null = null;
  let previous = 0;

  return function (this: any, ...args: Parameters<T>) {
    const context = this;
    const now = Date.now();

    if (!previous) {
      previous = now;
    }

    const remaining = wait - (now - previous);

    if (remaining <= 0 || remaining > wait) {
      if (timeout) {
        clearTimeout(timeout);
        timeout = null;
      }
      previous = now;
      func.apply(context, args);
    } else if (!timeout) {
      timeout = setTimeout(() => {
        previous = Date.now();
        timeout = null;
        func.apply(context, args);
      }, remaining);
    }
  };
}

export function memoize<T extends (...args: any[]) => any>(
  func: T
): T {
  const cache = new Map<string, ReturnType<T>>();

  return ((...args: Parameters<T>) => {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      return cache.get(key)!;
    }
    const result = func(...args);
    cache.set(key, result);
    return result;
  }) as T;
}
```

```typescript
   * 追踪点击
   */

      const target = event.target as HTMLElement;
      const trackingData = {
        element: target.tagName.toLowerCase(),
        id: target.id,
        class: target.className,
        text: target.textContent?.slice(0, 50),
        x: event.clientX,
        y: event.clientY,
      };

      this.pushEvent('click', trackingData);
    });
  }

  /**
   * 追踪滚动
   */
  private trackScroll(): void {
    let scrollStart = 0;
    let maxScroll = 0;

    const handleScroll = () => {
      const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
      const scrollHeight = document.documentElement.scrollHeight - window.innerHeight;

      // 记录最大滚动深度
      maxScroll = Math.max(maxScroll, scrollTop / scrollHeight);

      // 记录滚动开始
      if (scrollTop > 0 && scrollStart === 0) {
        scrollStart = scrollTop;
      }

      // 滚动到底部
      if (scrollTop + window.innerHeight >= document.documentElement.scrollHeight - 50) {
        this.pushEvent('scroll-bottom', { maxScroll });
      }
    };

    window.addEventListener('scroll', handleScroll, { passive: true });
  }

  /**
   * 追踪页面停留
   */
  private trackPageStay(): void {
    const startTime = Date.now();
    const pageUrl = window.location.href;

    const handleBeforeUnload = () => {
      const stayTime = Date.now() - startTime;
      this.pushEvent('page-stay', {
        pageUrl,
        stayTime,
        timestamp: startTime,
      });
      this.report();
    };

    window.addEventListener('beforeunload', handleBeforeUnload);
  }

   * 添加事件
   */
  private pushEvent(type: string, data: any): void {

      data,
```

```typescript
/**
 * XSS 防护工具

export class XSSSanitizer {
  /**
   * HTML 实体编码映射
   */
  private static entityMap: Record<string, string> = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#39;',
    '/': '&#x2F;',
    '`': '&#x60;',
    '=': '&#x3D;',
  };

  /**
   * 转义 HTML 特殊字符
   */
  static escapeHtml(str: string): string {
    return str.replace(/[&<>"`=/]/g, (char) => this.entityMap[char] || char);
  }

  /**
   * 转义 HTML 属性
   */
  static escapeHtmlAttr(str: string): string {
    return str.replace(/["'&<>]/g, (char) => this.entityMap[char] || char);
  }

  /**
   * 转义 JavaScript 字符串
   */
  static escapeJs(str: string): string {
    return str
      .replace(/\\/g, '\\\\')
      .replace(/'/g, "\\'")
      .replace(/"/g, '\\"')
      .replace(/\n/g, '\\n')
      .replace(/\r/g, '\\r')
      .replace(/\u2028/g, '\\u2028')
      .replace(/\u2029/g, '\\u2029');
  }

  /**
   * 转义 URL 参数
   */
  static escapeUrlParam(str: string): string {
    return encodeURIComponent(str)
      .replace(/[!'()*]/g, (c) => `%${c.charCodeAt(0).toString(16).toUpperCase()}`);
  }

  /**
   * 清理富文本 HTML
   */
  static sanitizeHtml(dirty: string): string {
    // 允许的标签
    const allowedTags = [
      'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
      'p', 'br', 'hr',
      'ul', 'ol', 'li',
      'strong', 'em', 'b', 'i', 'u', 's',
      'a', 'img',
      'table', 'thead', 'tbody', 'tr', 'th', 'td',
      'blockquote', 'pre', 'code',
    ];

    // 允许的属性
    const allowedAttrs = {
      'a': ['href', 'title', 'target'],
      'img': ['src', 'alt', 'title', 'width', 'height'],
      'td': ['colspan', 'rowspan'],
      'th': ['colspan', 'rowspan'],
    };

    // 创建 DOM 解析器
    const doc = new DOMParser().parseFromString(dirty, 'text/html');

    // 遍历所有元素
    const walk = (node: Node) => {
      if (node.nodeType === Node.TEXT_NODE) {
        return node.textContent || '';
      }

      if (node.nodeType !== Node.ELEMENT_NODE) {
        return '';
      }

      const element = node as Element;
      const tagName = element.tagName.toLowerCase();

      // 检查是否允许的标签
      if (!allowedTags.includes(tagName)) {
        return element.textContent || '';
      }

      // 检查允许的属性
      const allowed = allowedAttrs[tagName as keyof typeof allowedAttrs] || [];
      let attrs = '';

      for (const attr of Array.from(element.attributes)) {
        if (allowed.includes(attr.name)) {
          // 检查 href/src 属性
          if (attr.name === 'href' || attr.name === 'src') {
            const value = attr.value.trim();
            // 检查危险协议
            if (
              value.startsWith('javascript:') ||
              value.startsWith('data:') ||
              value.startsWith('vbscript:')
            ) {
              continue;
            }
          }
          attrs += ` ${attr.name}="${this.escapeHtmlAttr(attr.value)}"`;
        }
      }

      // 处理子元素
      const children = Array.from(element.childNodes)
        .map((child) => walk(child))
        .join('');

      return `<${tagName}${attrs}>${children}</${tagName}>`;
    };

    return Array.from(doc.body.childNodes)
      .map((node) => walk(node))
      .join('');
  }

  /**
   * 验证 URL
   */
  static validateUrl(url: string): boolean {
    try {
      const parsed = new URL(url);
      // 只允许 http 和 https
      return ['http:', 'https:'].includes(parsed.protocol);
    } catch {
      return false;
    }
  }

  /**
   * 安全地设置 HTML
   */
  static setInnerHTML(element: HTMLElement, html: string): void {
    // 使用 DOMPurify 等专业库更好
    element.innerHTML = this.sanitizeHtml(html);
  }
}

    };
```

```typescript
class AuthService {
  private tokenKey = 'auth_token';
  private refreshTokenKey = 'refresh_token';
  private tokenExpireKey = 'token_expire';

  /**
   * 保存认证信息
   */
  saveAuth(auth: {
    token: string;
    refreshToken: string;
    expiresIn: number;
  }): void {
    const expireTime = Date.now() + auth.expiresIn * 1000;

    localStorage.setItem(this.tokenKey, auth.token);
    localStorage.setItem(this.refreshTokenKey, auth.refreshToken);
    localStorage.setItem(this.tokenExpireKey, expireTime.toString());

    // 设置 Token 过期前自动刷新
    this.scheduleTokenRefresh(auth.expiresIn);
  }

  /**
   * 获取访问 Token
   */
  getToken(): string | null {
    const token = localStorage.getItem(this.tokenKey);
    const expireTime = parseInt(localStorage.getItem(this.tokenExpireKey) || '0');

    // Token 即将过期，尝试刷新
    if (token && expireTime - Date.now() < 5 * 60 * 1000) {
      this.refreshToken();
    }

    return token;
  }

  /**
   * 获取刷新 Token
   */
  getRefreshToken(): string | null {
    return localStorage.getItem(this.refreshTokenKey);
  }

  /**
   * 检查是否已登录
   */
  isAuthenticated(): boolean {
    const token = this.getToken();
    const expireTime = parseInt(localStorage.getItem(this.tokenExpireKey) || '0');

    return !!token && expireTime > Date.now();
  }

  /**
   * 刷新 Token
   */
  async refreshToken(): Promise<boolean> {
    const refreshToken = this.getRefreshToken();
    if (!refreshToken) {
      this.logout();
      return false;
    }

    try {
      const response = await fetch('/api/auth/refresh', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ refreshToken }),
      });

      if (response.ok) {
        const auth = await response.json();
        this.saveAuth(auth);
        return true;
      }
    } catch (error) {
      console.error('Token refresh failed:', error);
    }

    this.logout();
    return false;
  }

  /**
   * 计划 Token 刷新
   */
  private scheduleTokenRefresh(expiresIn: number): void {
    // 在 Token 过期前 5 分钟刷新
    const refreshTime = expiresIn * 1000 - 5 * 60 * 1000;

    setTimeout(() => {
      this.refreshToken();
    }, Math.max(refreshTime, 0));
  }

  /**
   * 登出
   */
  logout(): void {
    localStorage.removeItem(this.tokenKey);
    localStorage.removeItem(this.refreshTokenKey);
    localStorage.removeItem(this.tokenExpireKey);

    // 跳转到登录页
    window.location.href = '/login';
  }
}

export function parseJWT(token: string): {
  payload: any;
  expired: boolean;
  expiresAt: Date | null;
} {
  try {
    const parts = token.split('.');
    if (parts.length !== 3) {
      throw new Error('Invalid token format');

    const payload = JSON.parse(atob(parts[1]));
    const expiresAt = payload.exp ? new Date(payload.exp * 1000) : null;

    return { payload, expired, expiresAt };

  if (requiredRoles.length > 0 && !hasRole(userRoles, requiredRoles)) {
    return <Navigate to="/403" replace />;
  }

  if (
    requiredPermissions.length > 0 &&
    !hasPermission(userPermissions, requiredPermissions)
  ) {
    return <Navigate to="/403" replace />;
  }

  return <>{children}</>;
};

/**
 * 路由守卫 - Vue
 */
export function createAuthGuard(router: any) {
  router.beforeEach(async (to: any, from: any, next: any) => {
    // 页面标题
    document.title = to.meta?.title || '应用';

    // 检查登录状态
    if (!authService.isAuthenticated()) {
      // 免登录页面
      if (to.meta?.public) {
        return next();
      }
      return next({
        path: '/login',
        query: { redirect: to.fullPath },
      });
    }

    // 需要权限的页面
    if (to.meta?.permission) {
      const userPermissions: string[] = [];
      if (!hasPermission(userPermissions, to.meta.permission)) {
        return next({ path: '/403' });
      }
    }

    next();
  });
}
```
