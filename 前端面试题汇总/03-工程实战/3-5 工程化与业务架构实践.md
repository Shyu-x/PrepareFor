## 一、TOB 与 TOC 业务特点与架构差异

### 1.1 TOB（To Business）业务特点

TOB（面向企业）业务是指为企业客户提供产品或服务的业务模式。其特点如下：

| 特性 | 说明 |
|------|------|
| **用户群体** | 企业员工、管理者、决策者 |
| **使用场景** | 办公、管理、数据分析 |
| **功能复杂度** | 高，功能繁多且相互关联 |
| **交互方式** | 鼠标操作为主，操作密集型 |
| **性能要求** | 强调数据处理能力和稳定性 |
| **安全性** | 高，涉及企业敏感数据 |
| **定制化** | 高，需要支持客户化配置 |
| **上线周期** | 长，测试严格 |

**典型 TOB 产品：**
- 企业管理系统（ERP、CRM、OA）
- 财务系统
- 人力资源管理系统
- 数据分析平台
- SaaS 产品

```typescript
class OfflineManager {
  private cacheName = 'toc-app-cache-v1';
  private staticResources = [
    '/',
    '/index.html',
    '/static/js/main.js',
    '/static/css/main.css',
    '/manifest.json'
  ];

  async cacheStaticResources(): Promise<void> {
    const cache = await caches.open(this.cacheName);
    await cache.addAll(this.staticResources);
  }

  async handleFetch(event: FetchEvent): Promise<Response> {
    const url = new URL(event.request.url);

    // 网络优先策略（API 请求）
    if (url.pathname.startsWith('/api/')) {
      try {
        const response = await fetch(event.request);
        return response;
      } catch {
        // 网络失败返回缓存
        return caches.match(event.request) || new Response(
          JSON.stringify({ error: 'Offline' }),
          { headers: { 'Content-Type': 'application/json' } }
        );
      }
    }

    // 缓存优先策略（静态资源）
    const cachedResponse = await caches.match(event.request);
    if (cachedResponse) {
      // 后台更新缓存
      this.backgroundUpdate(event.request);
      return cachedResponse;
    }

    return fetch(event.request);
  }

  private async backgroundUpdate(request: Request): Promise<void> {
    const cache = await caches.open(this.cacheName);
    try {
      const response = await fetch(request);
      await cache.put(request, response);
    } catch {}
  }
}

class PushNotificationService {
  async requestPermission(): Promise<boolean> {
    if (!('Notification' in window)) return false;

    const permission = await Notification.requestPermission();
    return permission === 'granted';
  }

  async subscribe(): Promise<PushSubscription | null> {
    const registration = await navigator.serviceWorker.ready;

    const subscription = await registration.pushManager.subscribe({
      userVisibleOnly: true,
      applicationServerKey: this.urlBase64ToUint8Array(
        process.env.VITE_VAPID_PUBLIC_KEY!
      )
    });

    // 发送到服务器保存
    await this.saveSubscription(subscription);

    return subscription;
  }

  private urlBase64ToUint8Array(base64String: string): Uint8Array {
    const padding = '='.repeat((4 - base64String.length % 4) % 4);
    const base64 = (base64String + padding)
      .replace(/-/g, '+')
      .replace(/_/g, '/');

    const rawData = window.atob(base64);
    const outputArray = new Uint8Array(rawData.length);

    for (let i = 0; i < rawData.length; ++i) {
      outputArray[i] = rawData.charCodeAt(i);
    }

    return outputArray;
  }

  private async saveSubscription(subscription: PushSubscription): Promise<void> {
    await fetch('/api/push/subscribe', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(subscription)
    });
  }
}
```

##### 一、React 在协作白板应用中的深入原理

**1. React Fiber 与时间分片机制**

白板类应用面临大量状态更新（如笔迹绘制、多人光标移动），传统 React 渲染机制可能导致卡顿。React Fiber 通过时间分片解决此问题：

```typescript
// React Fiber 时间分片在白板应用中的应用
import { useCallback, useRef, useEffect } from 'react';

// 高频更新场景：光标位置同步
function useCursorSync(socket, roomId) {
  const cursorRef = useRef({ x: 0, y: 0 });
  const rafRef = useRef<number>();
  const pendingUpdate = useRef(false);

  // 使用 useCallback 稳定函数引用
  const updateCursor = useCallback((x: number, y: number) => {
    cursorRef.current = { x, y };

    // 标记有待更新
    pendingUpdate.current = true;

    // 使用 requestAnimationFrame 批量更新
    if (!rafRef.current) {
      rafRef.current = requestAnimationFrame(() => {
        if (pendingUpdate.current) {
          socket.emit('cursorMove', {
            roomId,
            position: cursorRef.current,
            timestamp: Date.now()
          });
          pendingUpdate.current = false;
        }
        rafRef.current = undefined;
      });
    }
  }, [socket, roomId]);

  // 清理
  useEffect(() => {
    return () => {
      if (rafRef.current) {
        cancelAnimationFrame(rafRef.current);
      }
    };
  }, []);

  return updateCursor;
}

// 面试问题：为什么白板应用需要手动实现节流/时间分片？
// 答案：React 的批量更新机制在某些场景下不够精细，特别是：
// 1. 原生事件（touch/mouse）不触发 React 合成事件批处理
// 2. 60fps 的高频更新需要更细粒度的控制
// 3. 避免阻塞主线程影响触控响应
```

```typescript
// 白板操作的优先级划分
const PRIORITIES = {

  REMOTE_OPERATION: ImmediatePriority,

function scheduleDrawingOperation(drawData) {
```

```typescript
class VirtualizedCanvas {
  private viewport: Viewport;
  private elements: CanvasElement[] = [];
  private renderCache: Map<string, HTMLCanvasElement> = new Map();

  // 计算可见区域元素
  getVisibleElements(): CanvasElement[] {
    const margin = 100; // 视口外 100px 也渲染
    const visibleRect = {
      x: this.viewport.x - margin,
      y: this.viewport.y - margin,
      width: this.viewport.width + margin * 2,
      height: this.viewport.height + margin * 2,
    };

    return this.elements.filter(el => {
      return this.isIntersect(el.boundingBox, visibleRect);
    });
  }

  // 离屏 Canvas 缓存复杂元素
  getCachedElement(element: CanvasElement): HTMLCanvasElement {
    const cacheKey = element.id;

    if (!this.renderCache.has(cacheKey)) {
      const offscreen = document.createElement('canvas');
      offscreen.width = element.width;
      offscreen.height = element.height;
      const ctx = offscreen.getContext('2d')!;
      this.renderElement(ctx, element);
      this.renderCache.set(cacheKey, offscreen);
    }

    return this.renderCache.get(cacheKey)!;
  }

  // 只渲染可见元素
  render() {
    const visibleElements = this.getVisibleElements();
    const ctx = this.canvas.getContext('2d')!;

    ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);

    for (const element of visibleElements) {
      if (element.isComplex) {
        // 复杂元素使用缓存
        const cached = this.getCachedElement(element);
        ctx.drawImage(cached, element.x, element.y);
      } else {
        // 简单元素直接渲染
        this.renderElement(ctx, element);
      }
    }
  }
}
```

```typescript
    // 验证码防刷
    if (this.saleConfig.requireCaptcha) {
      const captcha = await this.showCaptcha();
      if (!captcha) return { success: false, error: 'CAPTCHA_REQUIRED' };
    }

    // 排队机制
    return this.requestQueue.add(async () => {
      try {
        const result = await this.seckillApi.submit({
          skuId,
          quantity,
          token: this.saleConfig.seckillToken,
          timestamp: Date.now(),
          sign: this.generateSign({ skuId, quantity })
        });

        if (result.code === 'SUCCESS') {
          // 标记本地库存
          this.localCount++;
          return { success: true, orderId: result.orderId };
        } else if (result.code === 'STOCK_EMPTY') {
          this.localCount = -1; // 标记已售罄
        }

        return { success: false, error: result.code };
      } catch (error) {
        // 失败重试（最多3次）
        return this.retrySubmit(skuId, quantity, 3);
      }
    });
  }

  private localCountCheck(quantity: number): boolean {
    if (this.localCount === -1) return false; // 已售罄
    const remaining = this.saleConfig.totalStock - this.localCount;
    return remaining >= quantity;
  }

  // 绝望式防刷：服务端返回售罄后本地标记
  markSoldOut(): void {
    this.localCount = -1;
  }
}
```

```typescript
// 1. 即时通讯架构
interface IMConfig {
  wsUrl: string;
  heartbeatInterval: number;
  reconnectionDelay: number;
  maxRetries: number;
}

class IMClient {
  private ws: WebSocket | null = null;
  private config: IMConfig;
  private messageQueue: Message[] = [];
  private pendingAcks: Map<string, number>;
  private sequence: number = 0;
  private heartbeatTimer: number = 0;
  private reconnectTimer: number = 0;

  constructor(config: IMConfig) {
    this.config = config;
    this.pendingAcks = new Map();
    this.connect();
  }

  // 建立连接
  private connect(): void {
    this.ws = new WebSocket(this.config.wsUrl);

    this.ws.onopen = () => {
      console.log('IM Connected');
      this.startHeartbeat();
      this.flushMessageQueue();
      this.syncOfflineMessages();
    };

    this.ws.onmessage = (event) => {
      this.handleMessage(JSON.parse(event.data));
    };

    this.ws.onclose = () => {
      this.handleDisconnect();
    };

    this.ws.onerror = (error) => {
      console.error('IM Error:', error);
    };
  }

  // 心跳保活
  private startHeartbeat(): void {
    this.heartbeatTimer = window.setInterval(() => {
      this.send({
        type: 'HEARTBEAT',
        timestamp: Date.now(),
        sequence: this.sequence++
      });
    }, this.config.heartbeatInterval);
  }

  // 消息发送（支持离线消息队列）
  send(message: Message): void {
    const seq = message.sequence || this.sequence++;

    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({ ...message, sequence: seq }));
      // 等待 ACK
      this.pendingAcks.set(seq, Date.now());
    } else {
      // 离线队列
      this.messageQueue.push({ ...message, sequence: seq });
    }
  }

  // 发送消息（带重试）
  async sendWithRetry(message: Message, retries: number = 3): Promise<boolean> {
    for (let i = 0; i < retries; i++) {
      try {
        await this.sendAsync(message);
        return true;
      } catch (error) {
        await this.delay(1000 * Math.pow(2, i)); // 指数退避
      }
    }
    return false;
  }

  // 消息处理
  private handleMessage(data: IMMessage): void {
    switch (data.type) {
      case 'ACK':
        this.pendingAcks.delete(data.sequence);
        break;
      case 'MESSAGE':
        this.handleNewMessage(data.message);
        break;
      case 'NOTIFICATION':
        this.handleNotification(data.notification);
        break;
      case 'SYNC':
        this.handleSync(data.sync);
        break;
    }
  }

  // 处理新消息
  private handleNewMessage(message: Message): void {
    // 存储本地
    this.saveToLocal(message);
    // 更新会话列表
    this.updateConversation(message.conversationId);
    // 发送已读回执
    this.sendReceipt(message.id, 'READ');
    // 触发 UI 更新
    this.eventBus.emit('newMessage', message);
  }

  // 断线重连
  private handleDisconnect(): void {
    clearInterval(this.heartbeatTimer);

    // 指数退避重连
    let delay = this.config.reconnectionDelay;
    let retries = 0;

    const reconnect = () => {
      if (retries >= this.config.maxRetries) {
        console.error('Max retries reached');
        return;
      }

      console.log(`Reconnecting in ${delay}ms...`);
      this.reconnectTimer = window.setTimeout(() => {
        this.connect();
        retries++;
        delay *= 2;
        reconnect();
      }, delay);
    };

    reconnect();
  }
}

interface FeedItem {

class SocialFeed {
  private items: FeedItem[] = [];
  private seenIds: Set<string> = new Set();
  private cursor: string = '';
  private loading: boolean = false;
  private mergeStrategy: 'timeline' | 'rank' = 'rank';

  // 混合拉取算法
  async fetchFeed(pullToRefresh: boolean = false): Promise<FeedItem[]> {
    if (this.loading) return [];
    this.loading = true;

    try {
      // 并行请求多个来源
      const [followingFeed, recommendFeed, hotFeed] = await Promise.all([
        api.feed.getFollowing({ cursor: pullToRefresh ? '' : this.cursor }),
        api.feed.getRecommend({ limit: 10 }),
        api.feed.getHot({ limit: 5 })
      ]);

      // 合并策略
      let newItems: FeedItem[];
      if (this.mergeStrategy === 'rank') {
        newItems = this.mergeAndRank(followingFeed, recommendFeed, hotFeed);
      } else {
        newItems = [...followingFeed.items, ...recommendFeed.items, ...hotFeed.items];
      }

      // 去重
      newItems = newItems.filter(item => !this.seenIds.has(item.id));
      newItems.forEach(item => this.seenIds.add(item.id));

      if (pullToRefresh) {
        this.items = newItems;
      } else {
        this.items.push(...newItems);
      }

      this.cursor = followingFeed.cursor;

      // 曝光追踪
      this.trackImpression(newItems);

      return newItems;
    } finally {
      this.loading = false;
    }
  }

  // 智能合并排序
  private mergeAndRank(...feeds: FeedItem[][]): FeedItem[] {
    // 权重计算
    const score = (item: FeedItem, source: string): number => {
      const timeScore = 1 / (Date.now() - item.timestamp + 1);
      const interactionScore = item.interactions.likes / 10000;
      const sourceScore = { following: 0.4, recommend: 0.3, hot: 0.3 }[source] || 0;

      return timeScore * 0.3 + interactionScore * 0.4 + sourceScore * 0.3;
    };

    // 合并并排序
    const allItems = feeds.flat();
    return allItems.sort((a, b) => {
      const scoreA = score(a, 'following');
      const scoreB = score(b, 'following');
      return scoreB - scoreA;
    });
  }

  // 点赞/评论交互（乐观更新）
  async interact(itemId: string, type: 'like' | 'comment', data?: any): Promise<void> {
    const item = this.items.find(i => i.id === itemId);
    if (!item) return;

    // 乐观更新
    if (type === 'like') {
      item.interactions.isLiked = !item.interactions.isLiked;
      item.interactions.likes += item.interactions.isLiked ? 1 : -1;
    }

    this.eventBus.emit('interactionUpdate', { itemId, type, data });

    try {
      await api.feed.interact(itemId, type, data);
    } catch (error) {
      // 回滚
      if (type === 'like') {
        item.interactions.isLiked = !item.interactions.isLiked;
        item.interactions.likes += item.interactions.isLiked ? 1 : -1;
      }
      throw error;
    }
  }
}
```

```typescript
class CoursewareRenderer {
  private container: HTMLElement;
  private currentPage: number = 1;
  private totalPages: number = 0;
  private renderCache: Map<number, HTMLElement> = new Map();
  private preRenderRange: number = 3;

  constructor(container: HTMLElement) {
    this.container = container;
    this.init();
  }

  async loadCourseware(url: string): Promise<void> {
    const extension = url.split('.').pop()?.toLowerCase();

    switch (extension) {
      case 'pdf':
        await this.loadPDF(url);
        break;
      case 'ppt':
      case 'pptx':
        await this.loadPPT(url);
        break;
      case 'doc':
      case 'docx':
        await this.loadWord(url);
        break;
      default:
        throw new Error('Unsupported format');
    }

    // 预渲染
    this.prerenderPages();
  }

  // PDF 渲染（使用 pdf.js）
  private async loadPDF(url: string): Promise<void> {
    const pdf = await pdfjsLib.getDocument(url).promise;
    this.totalPages = pdf.numPages;

    // 渲染首页
    await this.renderPage(1);

    // 预渲染后续页
    this.preloadPages(2, this.preRenderRange + 1);
  }

  async renderPage(pageNum: number): Promise<void> {
    // 检查缓存
    if (this.renderCache.has(pageNum)) {
      const cached = this.renderCache.get(pageNum)!;
      this.container.innerHTML = '';
      this.container.appendChild(cached);
      this.currentPage = pageNum;
      return;
    }

    // 渲染新页
    const page = await this.renderPageContent(pageNum);
    this.renderCache.set(pageNum, page);

    this.container.innerHTML = '';
    this.container.appendChild(page);
    this.currentPage = pageNum;

    // 清理旧缓存
    this.cleanCache(pageNum);
  }

  // 页面预渲染
  private prerenderPages(): void {
    const startPage = this.currentPage + 1;
    const endPage = Math.min(this.currentPage + this.preRenderRange, this.totalPages);

    for (let i = startPage; i <= endPage; i++) {
      if (!this.renderCache.has(i)) {
        this.preloadPages(i, i);
      }
    }
  }

  // 异步预加载
  private async preloadPages(start: number, end: number): Promise<void> {
    for (let i = start; i <= end; i++) {
      if (i > 0 && i <= this.totalPages && !this.renderCache.has(i)) {
        try {
          const page = await this.renderPageContent(i);
          this.renderCache.set(i, page);
        } catch (error) {
          console.error(`Failed to preload page ${i}:`, error);
        }
      }
    }
  }

  // 翻页
  async nextPage(): Promise<void> {
    if (this.currentPage < this.totalPages) {
      await this.renderPage(this.currentPage + 1);
      this.prerenderPages();
    }
  }

  async prevPage(): Promise<void> {
    if (this.currentPage > 1) {
      await this.renderPage(this.currentPage - 1);
      this.prerenderPages();
    }
  }

  // 内存管理
  private cleanCache(currentPage: number): void {
    const keepRange = 2;
    this.renderCache.forEach((_, pageNum) => {
      if (Math.abs(pageNum - currentPage) > keepRange) {
        this.renderCache.delete(pageNum);
      }
    });
  }
}
```

#### 1.2.6 TOC 业务性能监控与优化实践

在 TOC 业务场景中，用户体验直接决定了产品的留存和转化。性能监控是保障优质用户体验的重要环节，需要建立完善的指标体系和监控方案。

##### 1.2.6.1 性能指标体系

TOC 业务主要关注以下核心性能指标（Core Web Vitals）：

| 指标 | 全称 | 定义 | 优化目标 |
|------|------|------|----------|
| FCP | First Contentful Paint | 首次内容绘制时间 | < 1.8s |
| LCP | Largest Contentful Paint | 最大内容绘制时间 | < 2.5s |
| FID | First Input Delay | 首次输入延迟 | < 100ms |
| TTI | Time to Interactive | 首次可交互时间 | < 3.8s |
| CLS | Cumulative Layout Shift | 累积布局偏移 | < 0.1 |
| TBT | Total Blocking Time | 总阻塞时间 | < 200ms |

```typescript
// 性能指标采集实现
class PerformanceMonitor {
  private metrics: PerformanceMetrics = {
    fp: 0,
    fcp: 0,
    lcp: 0,
    fid: 0,
    tti: 0,
    cls: 0,
    tbt: 0
  };

  constructor() {
    this.initObservers();
  }

  private initObservers(): void {
    // FCP 首次内容绘制
    new PerformanceObserver((entryList) => {
      const entries = entryList.getEntries();
      this.metrics.fcp = entries[0].startTime;
    }).observe({ type: 'paint', buffered: true });

    // LCP 最大内容绘制
    new PerformanceObserver((entryList) => {
      const entries = entryList.getEntries();
      const lastEntry = entries[entries.length - 1] as PerformanceEntry & { renderTime: number };
      this.metrics.lcp = lastEntry.renderTime;
    }).observe({ type: 'largest-contentful-paint', buffered: true });

    // FID 首次输入延迟
    new PerformanceObserver((entryList) => {
      const entries = entryList.getEntries();
      const firstEntry = entries[0] as PerformanceEventTiming;
      this.metrics.fid = firstEntry.processingStart - firstEntry.startTime;
    }).observe({ type: 'first-input', buffered: true });

    // CLS 累积布局偏移
    new PerformanceObserver((entryList) => {
      let clsScore = 0;
      for (const entry of entryList.getEntries()) {
        const layoutShift = entry as LayoutShift;
        if (!layoutShift.hadRecentInput) {
          clsScore += layoutShift.value;
        }
      }
      this.metrics.cls = clsScore;
    }).observe({ type: 'layout-shift', buffered: true });

    // TTI 可交互时间
    this.calculateTTI();
  }

          });
        }
      }
```

```typescript
        });

        this.buffer = [];
```

```typescript
        message: event.message,
        filename: event.filename,

        stack: event.reason?.stack
      });
    });

    // 资源加载失败
    window.addEventListener('error', (event) => {
      if (event.target !== window) {
        this.reportError({
          type: 'resource',
          url: (event.target as HTMLScriptElement).src ||
               (event.target as HTMLLinkElement).href,
          element: event.target.tagName
        });
      }
    }, true);
  }

  private reportError(error: ErrorReport): void {
    const report: ErrorReport = {
      ...error,
      timestamp: Date.now(),
      userId: this.getUserId(),
      url: window.location.href,
      userAgent: navigator.userAgent,
      viewport: `${window.innerWidth}x${window.innerHeight}`
    };

    this.errorQueue.push(report);

    // 立即上报严重错误
    if (this.isCriticalError(error)) {
      this.flush();
    }
  }

  private isCriticalError(error: ErrorReport): boolean {
    return error.type === 'javascript' ||
           error.type === 'unhandledrejection';
  }

  async flush(): Promise<void> {
    if (this.errorQueue.length === 0) return;

    const reports = [...this.errorQueue];
    this.errorQueue = [];

    try {
      await fetch('/api/metrics/error', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ errors: reports })
      });
    } catch {
      // 上报失败，重新加入队列
      this.errorQueue.push(...reports);
    }
  }

  private getUserId(): string {
    return localStorage.getItem('user_id') || 'anonymous';
  }
}
```

```typescript
        return;

      this.emit('countdown', {
        hours: Math.floor(diff / 3600000),
        minutes: Math.floor((diff % 3600000) / 60000),
        seconds: Math.floor((diff % 60000) / 1000)
      });
```

```typescript
interface WatchRecord {

// 推荐服务
class RecommendationService {
  private userProfile: UserProfile;
  private videoPool: Video[] = [];
  private currentIndex: number = 0;

  constructor(userId: string) {

  private async loadUserProfile(userId: string): Promise<void> {

  }

  // 获取推荐视频列表
  async getRecommendedVideos(count: number = 10): Promise<Video[]> {

      .sort((a, b) => b.score - a.score)
      .slice(0, count);
  }

  // 个性化推荐（协同过滤）
```

```typescript
// 单例模式 - 全局状态管理
class GlobalState {
  private static instance: GlobalState;
  private state: Map<string, any> = new Map();
  private listeners: Map<string, Set<Function>> = new Map();

  private constructor() {}

  static getInstance(): GlobalState {
    if (!GlobalState.instance) {
      GlobalState.instance = new GlobalState();
    }
    return GlobalState.instance;
  }

  get(key: string): any {
    return this.state.get(key);
  }

  set(key: string, value: any): void {
    this.state.set(key, value);
    this.notify(key, value);
  }

  subscribe(key: string, listener: Function): () => void {
    if (!this.listeners.has(key)) {
      this.listeners.set(key, new Set());
    }
    this.listeners.get(key)!.add(listener);

    // 返回取消订阅函数
    return () => {
      this.listeners.get(key)?.delete(listener);
    };
  }

  private notify(key: string, value: any): void {
    this.listeners.get(key)?.forEach(listener => listener(value));
  }
}

// 使用
```

```typescript
interface Command {
  execute(): void;
  undo(): void;
}

class CommandManager {
  private history: Command[] = [];
  private redoStack: Command[] = [];

  execute(command: Command): void {
    command.execute();
    this.history.push(command);
    this.redoStack = []; // 新命令清除重做栈
  }

  undo(): void {
    const command = this.history.pop();
    if (command) {
      command.undo();
      this.redoStack.push(command);
    }
  }

  redo(): void {
    const command = this.redoStack.pop();
    if (command) {
      command.execute();
      this.history.push(command);
    }
  }
}

    this.list.push(this.item);

  }

  undo(): void {
    const index = this.list.indexOf(this.item);
    if (index > -1) {
      this.list.splice(index, 1);
      console.log(`撤销添加: ${this.item}`);
    }
  }

    const index = this.list.indexOf(this.item);
    if (index > -1) {
      this.removedItem = this.list.splice(index, 1)[0];
      console.log(`删除: ${this.removedItem}`);
    }
  }

  undo(): void {
    if (this.removedItem) {
      this.list.push(this.removedItem);
      console.log(`撤销删除: ${this.removedItem}`);
    }
  }
}

// 使用
const list: string[] = [];
const manager = new CommandManager();

manager.execute(new AddItemCommand(list, '项目A'));
manager.execute(new AddItemCommand(list, '项目B'));
manager.execute(new RemoveItemCommand(list, '项目A'));

console.log('当前列表:', list); // ['项目B']

manager.undo();
console.log('撤销后:', list); // ['项目A', '项目B']

manager.redo();
console.log('重做后:', list); // ['项目B']
```

##### 2.6.1 电商订单处理中的命令模式实践

**业务场景**：电商平台的订单创建、支付、发货、退款流程，需要支持订单操作的日志记录、撤销重做、审批流程。

- 需要记录操作日志用于审计
- 需要支持订单操作的撤销（限时取消）

**解决方案**：

```typescript
// 订单命令模式实战

// 1. 订单命令接口
interface OrderCommand {
  execute(): Promise<OrderResult>;
  undo(): Promise<void>;
  getDescription(): string;
}

// 2. 订单上下文
class OrderContext {
  private state: OrderState;
  private history: OrderCommand[] = [];
  private historyIndex: number = -1;

  constructor(private orderId: string) {
    this.state = this.loadOrderState();
  }

  async executeCommand(command: OrderCommand): Promise<OrderResult> {
    try {
      // 记录操作前的状态快照
      const snapshot = this.createSnapshot();

      // 执行命令
      const result = await command.execute();

      // 成功，记录到历史记录
      this.history = this.history.slice(0, this.historyIndex + 1);
      this.history.push({
        command,
        snapshot,
        timestamp: Date.now()
      });
      this.historyIndex++;

      // 更新订单状态
      this.state = this.loadOrderState();

      // 发送状态变更通知
      this.notifyObservers('orderUpdated', this.state);

      return result;
    } catch (error) {
      // 记录失败日志
      this.logError(command.getDescription(), error);
      throw error;
    }
  }

  // 限时取消订单（撤销）
  async cancelOrderWithinTimeout(): Promise<void> {
    const lastEntry = this.history[this.historyIndex];
    if (!lastEntry) return;

    const timeDiff = Date.now() - lastEntry.timestamp;
    if (timeDiff > 30 * 60 * 1000) {
      // 超过30分钟，不能取消
      throw new Error('订单已超过可取消时限');
    }

    await lastEntry.command.undo();
    this.historyIndex--;

    this.state = this.loadOrderState();
    this.notifyObservers('orderCancelled', this.state);
  }

  private createSnapshot(): OrderSnapshot {
    return {
      state: { ...this.state },
      items: [...this.state.items],
      version: this.state.version
    };
  }

  private loadOrderState(): OrderState {
    // 从数据库加载最新状态
    return db.orders.findOne({ id: this.orderId });
  }

  private logError(operation: string, error: Error): void {
    logger.error('OrderOperationFailed', {
      orderId: this.orderId,
      operation,
      error: error.message,
      stack: error.stack,
      timestamp: new Date().toISOString()
    });
  }

  private notifyObservers(event: string, data: any): void {
    eventBus.emit(`order:${event}`, data);
    // 通知相关服务
    notificationService.sendToUser(this.state.userId, event, data);
  }
}

      await orderService.softDelete(this.previousOrderId);
    }

  }
```

```typescript
class BasePermissionChecker implements PermissionChecker {
  check(permission: string, context: PermissionContext): boolean {
    // 从缓存或数据库获取用户权限
    const userPermissions = this.getUserPermissions(context.userId);
    return userPermissions.includes(permission);

  constructor(protected checker: PermissionChecker) {}

  abstract check(permission: string, context: PermissionContext): boolean;
}

// 5. 数据范围装饰器
class DataScopeDecorator extends PermissionDecorator {
  check(permission: string, context: PermissionContext): boolean {
    // 先检查基础权限
    if (!this.checker.check(permission, context)) {
      return false;
    }

    // 检查数据范围权限
    const permissionConfig = PERMISSION_CONFIG[permission];
    if (!permissionConfig?.dataScope) {
      return true;
    }

    // 根据数据范围过滤
    return this.checkDataScope(permissionConfig.dataScope, context);
  }

  private checkDataScope(
    requiredScope: DataScopeType,
    context: PermissionContext
  ): boolean {
    const userScope = context.dataScope;

    const scopeHierarchy: DataScopeType[] = [
      'self', 'department', 'department_and_below', 'all'
    ];

    const userLevel = scopeHierarchy.indexOf(userScope);
    const requiredLevel = scopeHierarchy.indexOf(requiredScope);

    return userLevel >= requiredLevel;
  }
}

// 6. 时间范围装饰器
class TimeRangeDecorator extends PermissionDecorator {
  check(permission: string, context: PermissionContext): boolean {
    if (!this.checker.check(permission, context)) {
      return false;
    }

    const permissionConfig = PERMISSION_CONFIG[permission];
    if (!permissionConfig?.timeRange) {
      return true;
    }

    const now = new Date();
    const { startTime, endTime } = permissionConfig.timeRange;

    return now >= startTime && now <= endTime;
  }
}

// 7. IP 白名单装饰器
class IpWhitelistDecorator extends PermissionDecorator {
  check(permission: string, context: PermissionContext): boolean {
    if (!this.checker.check(permission, context)) {
      return false;
    }

    const permissionConfig = PERMISSION_CONFIG[permission];
    if (!permissionConfig?.ipWhitelist) {
      return true;
    }

    const userIp = context.clientIp;
    return permissionConfig.ipWhitelist.includes(userIp);
  }
}

// 8. 权限验证装饰器工厂
class PermissionFactory {
  private static baseChecker = new BasePermissionChecker();

  static create(permissions: PermissionConfig[]): PermissionChecker {
    let checker: PermissionChecker = this.baseChecker;

    // 按顺序包装装饰器
    for (const perm of permissions) {
      if (perm.dataScope) {
        checker = new DataScopeDecorator(checker);
      }
      if (perm.timeRange) {
        checker = new TimeRangeDecorator(checker);
      }
      if (perm.ipWhitelist) {
        checker = new IpWhitelistDecorator(checker);
      }
    }

    return checker;
  }

  // 快捷方法：创建带所有常用装饰器的检查器
  static createWithDefaults(permission: string): PermissionChecker {
    const config = PERMISSION_CONFIG[permission];
    if (!config) {
      return this.baseChecker;
    }

    return this.create([config]);
  }
}

// 9. 在组件中使用
function withPermission<P extends object>(
  WrappedComponent: React.ComponentType<P>,
  permission: string
): React.FC<P> {
  return (props) => {
    const context = usePermissionContext();
    const hasPermission = PermissionFactory.createWithDefaults(permission)
      .check(permission, context);

    if (!hasPermission) {
      return null; // 或返回占位符
    }

    return <WrappedComponent {...props} />;
  };
}
```

##### 2.6.3 发布订阅模式在状态管理中的实践

**业务场景**：电商平台的购物车、商品列表、用户信息等多模块需要实时同步。

**实际问题**：
- 多组件需要共享状态
- 状态变更需要通知多个订阅者
- 需要支持跨页面、跨组件的状态同步

**解决方案**：

```typescript
// 发布订阅模式实战 - 事件总线实现

// 1. 事件总线
class EventBus {
  private events: Map<string, Set<EventHandler>> = new Map();
  private onceEvents: Map<string, Set<EventHandler>> = new Map();

  // 订阅事件
  on(event: string, handler: EventHandler): () => void {
    if (!this.events.has(event)) {
      this.events.set(event, new Set());
    }
    this.events.get(event)!.add(handler);

    // 返回取消订阅函数
    return () => this.off(event, handler);
  }

  // 一次性订阅
  once(event: string, handler: EventHandler): void {
    if (!this.onceEvents.has(event)) {
      this.onceEvents.set(event, new Set());
    }
    this.onceEvents.get(event)!.add(handler);
  }

  // 取消订阅
  off(event: string, handler: EventHandler): void {
    this.events.get(event)?.delete(handler);
  }

  // 发布事件
  emit(event: string, data?: any): void {
    // 处理一次性事件
    if (this.onceEvents.has(event)) {
      this.onceEvents.get(event)!.forEach(handler => {
        try {
          handler(data);
        } catch (error) {
          console.error(`Event handler error: ${event}`, error);
        }
      });
      this.onceEvents.delete(event);
    }

    // 处理常驻事件
    if (this.events.has(event)) {
      this.events.get(event)!.forEach(handler => {
        try {
          handler(data);
        } catch (error) {
          console.error(`Event handler error: ${event}`, error);
        }
      });
    }
  }

  // 清理所有事件
  clear(): void {
    this.events.clear();
    this.onceEvents.clear();
  }
}

// 4. React Hook 封装
function useEventBus(bus: EventBus) {
  const [state, setState] = useState<any>(null);

  useEffect(() => {
    const handlers: Map<string, Function> = new Map();

    // 自动清理
    return () => {
      handlers.forEach((handler, event) => {
        bus.off(event, handler);
      });
    };
  }, [bus]);

  const subscribe = useCallback((event: string, handler: Function) => {
    const wrappedHandler = (data: any) => {
      setState(data);
      handler(data);
    };

    bus.on(event, wrappedHandler);
    handlers.set(event, wrappedHandler);

    return () => bus.off(event, wrappedHandler);
  }, [bus]);

  return { subscribe, bus };
}
```

##### 2.6.4 状态机在审批流中的应用

**业务场景**：企业审批流程（请假、报销、采购等），需要管理复杂的状态流转和规则。

**实际问题**：

- 需要支持会签、或签、加签等多种审批方式

**解决方案**：

```typescript
// 状态机模式在审批流中的实践

// 1. 审批状态定义
enum ApprovalStatus {
  DRAFT = 'draft',           // 草稿
  PENDING = 'pending',       // 待审批
  APPROVING = 'approving',   // 审批中
  APPROVED = 'approved',     // 已通过
  REJECTED = 'rejected',    // 已拒绝
  CANCELLED = 'cancelled',  // 已取消
  EXPIRED = 'expired'       // 已过期
}

// 2. 审批事件
enum ApprovalEvent {
  SUBMIT = 'submit',           // 提交
  APPROVE = 'approve',         // 批准
  REJECT = 'reject',          // 拒绝
  CANCEL = 'cancel',          // 取消
  EXPIRE = 'expire',          // 过期
  RECALL = 'recall',          // 撤回
  TRANSFER = 'transfer'       // 转交
}

// 3. 状态机配置
const approvalStateMachine: StateMachineConfig<ApprovalStatus, ApprovalEvent> = {
  initial: ApprovalStatus.DRAFT,

  states: {
    [ApprovalStatus.DRAFT]: {
      on: {
        [ApprovalEvent.SUBMIT]: ApprovalStatus.PENDING
      }
    },
    [ApprovalStatus.PENDING]: {
      on: {
        [ApprovalEvent.APPROVE]: ApprovalStatus.APPROVING,
        [ApprovalEvent.REJECT]: ApprovalStatus.REJECTED,
        [ApprovalEvent.CANCEL]: ApprovalStatus.CANCELLED,
        [ApprovalEvent.EXPIRE]: ApprovalStatus.EXPIRED,
        [ApprovalEvent.TRANSFER]: ApprovalStatus.PENDING
      }
    },
    [ApprovalStatus.APPROVING]: {
      on: {
        [ApprovalEvent.APPROVE]: ApprovalStatus.APPROVED,
        [ApprovalEvent.REJECT]: ApprovalStatus.REJECTED,
        [ApprovalEvent.CANCEL]: ApprovalStatus.CANCELLED
      }
    },
    [ApprovalStatus.APPROVED]: {
      // 终态，无法转换
    },
    [ApprovalStatus.REJECTED]: {
      // 终态，无法转换
      on: {
        [ApprovalEvent.SUBMIT]: ApprovalStatus.PENDING  // 重新提交
      }
    },
    [ApprovalStatus.CANCELLED]: {
      // 终态
    },
    [ApprovalStatus.EXPIRED]: {
      on: {
        [ApprovalEvent.SUBMIT]: ApprovalStatus.PENDING  // 重新提交
      }
    }
  }

// 4. 审批状态机类
class ApprovalStateMachine {
  private currentState: ApprovalStatus;
  private history: StateTransition[] = [];
  private context: ApprovalContext;

  constructor(
    private config: StateMachineConfig<ApprovalStatus, ApprovalEvent>,
    initialState: ApprovalStatus,
    context: ApprovalContext
  ) {
    this.currentState = initialState;
    this.context = context;
  }

  // 执行状态转换
  async transition(event: ApprovalEvent, data?: TransitionData): Promise<TransitionResult> {
    const currentStateConfig = this.config.states[this.currentState];
    const nextState = currentStateConfig?.on?.[event];

    if (!nextState) {
      throw new Error(
        `无效的状态转换: ${this.currentState} --${event}--> ${nextState}`
      );
    }

    // 执行转换前的钩子
    await this.beforeTransition(event, nextState, data);

    // 记录转换历史
    const transition: StateTransition = {
      from: this.currentState,
      to: nextState,
      event,
      operator: data?.operatorId,
      timestamp: new Date(),
      comment: data?.comment
    };
    this.history.push(transition);

    // 更新状态
    const previousState = this.currentState;
    this.currentState = nextState;

    // 执行转换后的钩子
    await this.afterTransition(previousState, nextState, data);

    return {
      success: true,
      from: previousState,
      to: nextState,
      transition
    };
  }

  // 转换前钩子
  private async beforeTransition(
    event: ApprovalEvent,
    nextState: ApprovalStatus,
    data?: TransitionData
  ): Promise<void> {
    // 权限检查
    await this.checkPermission(event, data);

    // 业务规则验证
    await this.validateBusinessRules(event, data);

    // 触发审批流程（如果需要多级审批）
    if (nextState === ApprovalStatus.APPROVING) {
      await this.startApprovalProcess(data);
    }
  }

  // 转换后钩子
  private async afterTransition(
    from: ApprovalStatus,
    to: ApprovalStatus,
    data?: TransitionData
  ): Promise<void> {
    // 记录审批日志
    await this.logTransition(from, to, data);

    // 发送通知
    await this.sendNotification(from, to, data);

    // 执行后续操作
    if (to === ApprovalStatus.APPROVED) {
      await this.executePostApprovalActions();
    }
  }

  private async checkPermission(event: ApprovalEvent, data?: TransitionData): Promise<void> {
    const userId = data?.operatorId;
    const permission = this.getRequiredPermission(event);

    if (!permission) return;

    const hasPermission = await permissionService.check(userId, permission);
    if (!hasPermission) {
      throw new Error(`用户 ${userId} 缺少权限: ${permission}`);
    }
  }

  private async validateBusinessRules(event: ApprovalEvent, data?: TransitionData): Promise<void> {
    // 可以配置化的业务规则
    const rules = this.getBusinessRules(event);

    for (const rule of rules) {
      const isValid = await rule.validate(this.context, data);
      if (!isValid) {
        throw new Error(`业务规则验证失败: ${rule.message}`);
      }
    }
  }

  private getRequiredPermission(event: ApprovalEvent): string | null {
    const permissionMap: Record<ApprovalEvent, string> = {
      [ApprovalEvent.SUBMIT]: 'approval:submit',
      [ApprovalEvent.APPROVE]: 'approval:approve',
      [ApprovalEvent.REJECT]: 'approval:reject',
      [ApprovalEvent.CANCEL]: 'approval:cancel',
      [ApprovalEvent.RECALL]: 'approval:recall',
      [ApprovalEvent.TRANSFER]: 'approval:transfer',
      [ApprovalEvent.EXPIRE]: null
    };
    return permissionMap[event];
  }

  private getBusinessRules(event: ApprovalEvent): BusinessRule[] {
    // 根据事件类型返回对应的业务规则
    return [];
  }

  getState(): ApprovalStatus {
    return this.currentState;
  }

  getHistory(): StateTransition[] {
    return [...this.history];
  }
}
```

### 4.1 SOLID 原则

| 原则 | 说明 | 违反后果 |
|------|------|----------|
| **S 单一职责** | 每个类/函数只负责一项功能 | 难以维护、难以测试 |
| **O 开闭原则** | 对扩展开放，对修改关闭 | 频繁修改影响稳定性 |
| **L 里氏替换** | 子类能替换父类 | 破坏继承体系 |
| **I 接口隔离** | 使用多个专用接口 | 依赖冗余 |
| **D 依赖倒置** | 依赖抽象而非具体 | 难以替换实现 |


##### 6.4.2 灰度发布实战

**业务场景**：新功能上线需要灰度发布，逐步放量验证稳定性。

**灰度策略**：
- 按用户 ID 哈希进行流量分配
- 支持百分比灰度和白名单灰度
- 自动回滚异常版本

**实现代码**：

```typescript
// 灰度发布前端实现

interface GrayReleaseConfig {
  // 灰度规则
  rules: GrayReleaseRule[];
  // 回滚配置
  rollback: RollbackConfig;
  // 监控配置
  monitoring: MonitoringConfig;
}

interface GrayReleaseRule {
  name: string;
  // 流量百分比 (0-100)
  percentage: number;
  // 白名单用户
  whitelist?: string[];
  // 用户标签
  tags?: string[];
  // 设备类型
  device?: string[];
  // 地区
  regions?: string[];
  // 版本范围
  versionRange?: {
    min?: string;
    max?: string;
  };
}

class GrayReleaseManager {
  private config: GrayReleaseConfig;
  private userFeatures: Map<string, Set<string>> = new Map();

  constructor(config: GrayReleaseConfig) {
    this.config = config;
    this.initUserFeatures();
  }

  // 初始化用户特征
  private async initUserFeatures(): Promise<void> {
    // 从后端获取用户特征
    const features = await this.fetchUserFeatures();
    features.forEach((feature, userId) => {
      this.userFeatures.set(userId, new Set(feature));
    });
  }

  // 检查用户是否在灰度中
  isInGrayRelease(featureKey: string, userId: string): boolean {
    const rule = this.config.rules.find(r => r.name === featureKey);
    if (!rule) return false;

    // 1. 检查白名单
    if (rule.whitelist?.includes(userId)) {
      return true;
    }

    // 2. 检查用户标签
    if (rule.tags?.length) {
      const userTags = this.userFeatures.get(userId) || new Set();
      if (rule.tags.some(tag => userTags.has(tag))) {
        return true;
      }
    }

    // 3. 按百分比灰度
    if (rule.percentage > 0) {
      const hash = this.hashUserId(userId, featureKey);
      return hash < rule.percentage;
    }

    return false;
  }

  // 用户 ID 哈希（保证同一用户每次路由到同一版本）
  private hashUserId(userId: string, featureKey: string): number {
    const str = `${userId}:${featureKey}`;
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return Math.abs(hash) % 100;
  }

  // 获取用户应该使用的功能版本
  getFeatureVersion(featureKey: string, userId: string): string {
    const isGray = this.isInGrayRelease(featureKey, userId);
    return isGray ? 'gray' : 'stable';
  }

  // 监控灰度指标并自动回滚
  async monitorAndRollback(): Promise<void> {
    const metrics = await this.fetchMetrics();

    for (const rule of this.config.rules) {
      const ruleMetrics = metrics[rule.name];

      // 检查错误率
      if (ruleMetrics.errorRate > this.config.rollback.errorRateThreshold) {
        console.error(`灰度版本错误率过高: ${rule.name}`, {
          errorRate: ruleMetrics.errorRate,
          threshold: this.config.rollback.errorRateThreshold
        });

        await this.triggerRollback(rule.name);
      }

      // 检查性能退化
      if (ruleMetrics.p50Latency > this.config.rollback.latencyThreshold) {
        console.error(`灰度版本性能退化: ${rule.name}`, {
          p50Latency: ruleMetrics.p50Latency,
          threshold: this.config.rollback.latencyThreshold
        });

        await this.triggerRollback(rule.name);
      }
    }
  }

  private async triggerRollback(featureKey: string): Promise<void> {
    // 记录回滚事件
    await this.logRollback(featureKey);

    // 发送告警
    await this.sendAlert({
      type: 'gray_rollback',
      feature: featureKey,
      timestamp: new Date()
    });

    // 通知相关人员
    notificationService.notify(`灰度版本 ${featureKey} 已自动回滚`, 'critical');
  }
}

// React 组件中使用
function GrayReleaseWrapper({
  featureKey,
  children,
  fallback = null
}: {
  featureKey: string;
  children: React.ReactNode;
  fallback?: React.ReactNode;
}) {
  const [isGray, setIsGray] = useState(false);
  const { user } = useUser();

  useEffect(() => {
    const grayManager = new GrayReleaseManager(grayConfig);
    setIsGray(grayManager.isInGrayRelease(featureKey, user.id));
  }, [featureKey, user.id]);

  if (!isGray) {
    return <>{fallback}</>;
  }

  return <>{children}</>;
}

// 使用示例
function NewCheckoutPage() {

      fallback={<OldCheckoutPage />}
```

```typescript
  }

  private setupGlobalHandlers(): void {
    // JavaScript 错误
    window.onerror = (message, source, lineno, colno, error) => {
      this.reportError({
        type: 'js',
        message,
        source,
        lineno,
        colno,
        stack: error?.stack,
      });
    };

    // Promise 拒绝
    window.onunhandledrejection = (event) => {
      this.reportError({
        type: 'promise',
        reason: event.reason,
      });
    };

    // 资源加载错误
    window.addEventListener('error', (event) => {
      if (event.target !== window) {
        this.reportError({
          type: 'resource',
          url: (event.target as HTMLScriptElement).src,
        });
      }
    }, true);
  }

    fetch(this.endpoint, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        ...error,
        timestamp: Date.now(),
        userAgent: navigator.userAgent,
        url: window.location.href,
      }),
    });
  }
```

```typescript
// 降级策略
class DegradationStrategy {
  // 功能降级
  static degradeFeature(
    feature: string,
    primary: () => any,
    fallback: () => any,
    timeout: number = 1000
  ): Promise<any> {
    return new Promise((resolve) => {
      const timer = setTimeout(() => {
        console.warn(`功能 ${feature} 超时，使用降级方案`);
        resolve(fallback());
      }, timeout);

      primary()
        .then((result) => {
          clearTimeout(timer);
          resolve(result);
        })
        .catch((error) => {
          clearTimeout(timer);
          console.error(`功能 ${feature} 失败:`, error);
          resolve(fallback());
        });
    });
  }

  // 渲染降级
  static renderDegrade(
    primary: () => JSX.Element,
    fallback: () => JSX.Element,
    shouldDegrade: () => boolean
  ): JSX.Element {
    if (shouldDegrade()) {
      return fallback();
    }
    return primary();
  }

  // 缓存降级
  static async cacheDegrade<T>(
    key: string,
    fetcher: () => Promise<T>,
    cache: Cache,
    ttl: number = 5 * 60 * 1000
  ): Promise<T> {
    // 尝试从缓存获取
    const cached = await cache.get(key);
    if (cached && Date.now() - cached.timestamp < ttl) {
      return cached.data;
    }

    // 缓存不存在或过期，获取新数据
    try {
      const data = await fetcher();
      await cache.set(key, { data, timestamp: Date.now() });
      return data;
    } catch (error) {
      // 获取失败，返回过期缓存
      if (cached) {
        console.warn('使用过期缓存:', key);
        return cached.data;
      }
      throw error;
    }
  }
}
```

##### 8.2.2 主应用与子应用通信实践

**通信方案选择**：

| 方案 | 适用场景 | 优点 | 缺点 |
|------|----------|------|------|
| Props 传递 | 简单数据 | 简单、直接 | 深度嵌套时不方便 |
| CustomEvent | 跨应用通信 | 解耦、灵活 | 需要手动清理 |
| qiankun initGlobalState | 共享状态 | 官方推荐 | 有版本限制 |

**最佳实践**：

```typescript
  user: null,
  permissions: [],
  theme: 'light'

  console.log('[主应用] 状态变化', state, prev);
  // 通知相关组件
  eventBus.emit('globalStateChanged', state);
});

  theme: 'dark'

// 2. 子应用 - 使用主应用状态
// hooks/useGlobalState.ts
import { useState, useEffect } from 'react';

export function useGlobalState() {
  const [state, setState] = useState<any>({});

  useEffect(() => {
    if (!window.__POWERED_BY_QIANKUN__) {
      return;
    }

    // 订阅主应用状态变化
    // eslint-disable-next-line react-hooks/exhaustive-deps
    const unsubscribe = window.__INJECTED_PUBLIC_PATH_BY_QIANKUN__?.onGlobalStateChange?.(
      (newState: any) => {
        setState(newState);
      },
      true // 立即触发一次
    );

    return () => {
      unsubscribe?.();
    };
  }, []);

  return state;
}

// 3. 事件通信 - 用于复杂场景
// eventBus.ts
class MicroAppEventBus {
  private listeners: Map<string, Function[]> = new Map();

  on(event: string, callback: Function): () => void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, []);
    }
    this.listeners.get(event)!.push(callback);

    return () => {
      const callbacks = this.listeners.get(event);
      const index = callbacks?.indexOf(callback) ?? -1;
      if (index > -1) {
        callbacks?.splice(index, 1);
      }
    };
  }

  emit(event: string, data?: any): void {
    const callbacks = this.listeners.get(event);
    callbacks?.forEach(cb => cb(data));
  }
}

// 跨应用通信
export const microAppBus = new MicroAppEventBus();

// 主应用发送事件
function notifySubApps(event: string, data: any) {
  microAppBus.emit(event, data);
}

// 子应用监听事件
function subscribeToMainApp(event: string, callback: Function) {
  // 通过 props 传递的回调
  return props?.subscribe?.(event, callback);
}
```

```typescript
// 领域模型设计

// 领域实体
class Order {
  private id: string;
  private items: OrderItem[];
  private status: OrderStatus;
  private createdAt: Date;

  constructor(id: string) {
    this.id = id;
    this.items = [];
    this.status = 'pending';
    this.createdAt = new Date();
  }

  // 领域方法
  addItem(product: Product, quantity: number): void {
    if (this.status !== 'pending') {
      throw new Error('只能对pending状态的订单添加商品');
    }
    this.items.push(new OrderItem(product, quantity));
  }

  submit(): void {
    if (this.items.length === 0) {
      throw new Error('订单不能为空');
    }
    this.status = 'submitted';
  }

  // 值对象
  getTotal(): Money {
    return this.items.reduce(
      (sum, item) => sum.add(item.getSubtotal()),
      Money.zero()
    );
  }
}

// 值对象
class Money {
  constructor(
    private amount: number,
    private currency: string = 'CNY'
  ) {}

  static zero(): Money {
    return new Money(0);
  }

  add(other: Money): Money {
    if (this.currency !== other.currency) {
      throw new Error('货币类型不一致');
    }
    return new Money(this.amount + other.amount, this.currency);
  }
}
```
