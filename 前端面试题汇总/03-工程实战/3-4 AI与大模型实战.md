## 1.1 机器学习基础

### 1.1.1 机器学习定义与分类

**什么是机器学习？**

机器学习（Machine Learning，ML）是人工智能的一个分支，它使计算机系统能够从数据中学习并改进性能，而无需被明确编程。机器学习的核心思想是通过算法让计算机从大量数据中发现规律、提取特征，并利用这些规律进行预测或决策。

**机器学习的三大类型：**

1. **监督学习（Supervised Learning）**
   - 定义：在有标签的数据集上进行训练，学习输入到输出的映射关系
   - 典型应用：分类、回归、目标检测
   - 常用算法：线性回归、逻辑回归、决策树、随机森林、支持向量机、神经网络
   - 示例场景：邮件spam分类、房价预测、图像识别

2. **无监督学习（Unsupervised Learning）**
   - 定义：在无标签的数据集上训练，发现数据的内在结构和模式
   - 典型应用：聚类、降维、异常检测
   - 常用算法：K-means聚类、主成分分析（PCA）、自编码器
   - 示例场景：用户分群、推荐系统、异常交易检测

3. **强化学习（Reinforcement Learning）**
   - 定义：通过与环境交互，学习最优策略以最大化奖励
   - 典型应用：游戏AI、机器人控制、自动驾驶
   - 常用算法：Q-learning、Deep Q Network (DQN)、Policy Gradient、Actor-Critic
   - 示例场景：AlphaGo、机器人路径规划、动态定价

### 1.1.2 常用机器学习算法详解

**线性回归（Linear Regression）**

线性回归是最基础的监督学习算法，用于预测连续值输出。

```python
# 线性回归详细实现
import numpy as np

class LinearRegression:
    """线性回归实现"""

    def __init__(self, learning_rate=0.01, n_iterations=1000):
        self.learning_rate = learning_rate
        self.n_iterations = n_iterations
        self.weights = None
        self.bias = None

    def fit(self, X, y):
        """训练模型"""
        n_samples, n_features = X.shape

        # 初始化参数
        self.weights = np.zeros(n_features)
        self.bias = 0

        # 梯度下降优化
        for _ in range(self.n_iterations):
            # 前向传播：计算预测值
            y_pred = np.dot(X, self.weights) + self.bias

            # 计算梯度
            dw = (1/n_samples) * np.dot(X.T, (y_pred - y))
            db = (1/n_samples) * np.sum(y_pred - y)

            # 更新参数
            self.weights -= self.learning_rate * dw
            self.bias -= self.learning_rate * db

    def predict(self, X):
        """预测"""
        return np.dot(X, self.weights) + self.bias

# 使用示例
X = np.array([[1, 1], [1, 2], [2, 2], [2, 3]])
y = np.dot(X, [1, 2]) + 3  # y = 1*x1 + 2*x2 + 3

model = LinearRegression(learning_rate=0.01, n_iterations=1000)
model.fit(X, y)

print(f"学习到的权重: {model.weights}")
print(f"学习到的偏置: {model.bias}")
print(f"预测结果: {model.predict(np.array([[3, 3]]))}")
```

**逻辑回归（Logistic Regression）**

逻辑回归用于二分类问题，输出概率值。

```python
# 逻辑回归实现
import numpy as np

class LogisticRegression:
    """逻辑回归实现"""

    def __init__(self, learning_rate=0.01, n_iterations=1000):
        self.learning_rate = learning_rate
        self.n_iterations = n_iterations
        self.weights = None
        self.bias = None

    def sigmoid(self, z):
        """Sigmoid激活函数"""
        return 1 / (1 + np.exp(-z))

    def fit(self, X, y):
        """训练模型"""
        n_samples, n_features = X.shape

        # 初始化参数
        self.weights = np.zeros(n_features)
        self.bias = 0

        # 梯度下降优化
        for _ in range(self.n_iterations):
            # 前向传播
            linear = np.dot(X, self.weights) + self.bias
            y_pred = self.sigmoid(linear)

            # 计算梯度（使用交叉熵损失）
            dw = (1/n_samples) * np.dot(X.T, (y_pred - y))
            db = (1/n_samples) * np.sum(y_pred - y)

            # 更新参数
            self.weights -= self.learning_rate * dw
            self.bias -= self.learning_rate * db

    def predict_proba(self, X):
        """预测概率"""
        linear = np.dot(X, self.weights) + self.bias
        return self.sigmoid(linear)

    def predict(self, X, threshold=0.5):
        """预测类别"""
        return self.predict_proba(X) >= threshold

# 使用示例：垃圾邮件分类
# 特征：邮件中关键词出现频率
X = np.array([
    [1, 0, 1, 0],  # 邮件1
    [0, 1, 1, 0],  # 邮件2
    [1, 1, 0, 0],  # 邮件3
    [0, 0, 1, 1],  # 邮件4
])
y = np.array([1, 1, 0, 0])  # 标签：1=垃圾邮件，0=正常邮件

model = LogisticRegression()
model.fit(X, y)

# 预测新邮件
new_email = np.array([[1, 0, 0, 1]])
is_spam = model.predict(new_email)
spam_probability = model.predict_proba(new_email)

print(f"是否为垃圾邮件: {is_spam}")
print(f"垃圾邮件概率: {spam_probability:.2%}")
```

**决策树（Decision Tree）**

决策树通过构建树形结构进行决策，易于理解和解释。

```python
# 决策树实现
import numpy as np
from collections import Counter

class DecisionTree:
    """决策树分类器实现"""

    def __init__(self, max_depth=10, min_samples_split=2):
        self.max_depth = max_depth
        self.min_samples_split = min_samples_split
        self.tree = None

    def _entropy(self, y):
        """计算信息熵"""
        counter = Counter(y)
        probabilities = [count / len(y) for count in counter.values()]
        return -sum(p * np.log2(p) for p in probabilities if p > 0)

    def _information_gain(self, X, y, threshold, feature_index):
        """计算信息增益"""
        # 分割数据集
        left_mask = X[:, feature_index] <= threshold
        right_mask = ~left_mask

        if len(y[left_mask]) == 0 or len(y[right_mask]) == 0:
            return 0

        # 计算父节点熵
        parent_entropy = self._entropy(y)

        # 计算子节点加权熵
        n = len(y)
        n_left, n_right = len(y[left_mask]), len(y[right_mask])
        e_left = self._entropy(y[left_mask])
        e_right = self._entropy(y[right_mask])

        child_entropy = (n_left / n) * e_left + (n_right / n) * e_right

        return parent_entropy - child_entropy

    def _find_best_split(self, X, y):
        """找到最佳分割点"""
        best_gain = 0
        best_threshold = None
        best_feature = None

        n_features = X.shape[1]

        for feature_index in range(n_features):
            # 获取该特征的所有唯一值作为候选阈值
            thresholds = np.unique(X[:, feature_index])

            for threshold in thresholds:
                gain = self._information_gain(X, y, threshold, feature_index)

                if gain > best_gain:
                    best_gain = gain
                    best_threshold = threshold
                    best_feature = feature_index

        return best_feature, best_threshold, best_gain

    def _build_tree(self, X, y, depth=0):
        """递归构建决策树"""
        n_samples = len(y)
        n_labels = len(np.unique(y))

        # 停止条件
        if (depth >= self.max_depth or
            n_samples < self.min_samples_split or
            n_labels == 1):
            # 返回叶节点：多数类
            return Counter(y).most_common(1)[0][0]

        # 找到最佳分割
        feature_index, threshold, gain = self._find_best_split(X, y)

        if gain == 0:
            return Counter(y).most_common(1)[0][0]

        # 分割数据集
        left_mask = X[:, feature_index] <= threshold
        right_mask = ~left_mask

        # 递归构建子树
        left_tree = self._build_tree(X[left_mask], y[left_mask], depth + 1)
        right_tree = self._build_tree(X[right_mask], y[right_mask], depth + 1)

        return {
            'feature_index': feature_index,
            'threshold': threshold,
            'left': left_tree,
            'right': right_tree
        }

    def fit(self, X, y):
        """训练模型"""
        self.tree = self._build_tree(X, y)

    def _predict_sample(self, x, tree):
        """预测单个样本"""
        if not isinstance(tree, dict):
            return tree

        feature_index = tree['feature_index']
        threshold = tree['threshold']

        if x[feature_index] <= threshold:
            return self._predict_sample(x, tree['left'])
        else:
            return self._predict_sample(x, tree['right'])

    def predict(self, X):
        """预测"""
        return np.array([self._predict_sample(x, self.tree) for x in X])

# 使用示例
X = np.array([
    [2, 3], [1, 2], [3, 1], [4, 3],
    [3, 4], [1, 5], [5, 2], [4, 4]
])
y = np.array([0, 0, 0, 0, 1, 1, 1, 1])

tree = DecisionTree(max_depth=3)
tree.fit(X, y)

print(f"预测结果: {tree.predict(X)}")
```

**随机森林（Random Forest）**

随机森林是集成学习方法，通过构建多棵决策树并进行投票来提高准确性和稳定性。

```python
# 随机森林实现
import numpy as np
from collections import Counter
from DecisionTree import DecisionTree  # 假设上面的决策树类

class RandomForest:
    """随机森林分类器"""

    def __init__(self, n_trees=10, max_depth=10, min_samples_split=2):
        self.n_trees = n_trees
        self.max_depth = max_depth
        self.min_samples_split = min_samples_split
        self.trees = []

    def _bootstrap(self, X, y):
        """Bootstrap采样：有放回随机采样"""
        n_samples = len(X)
        indices = np.random.choice(n_samples, n_samples, replace=True)
        return X[indices], y[indices]

    def fit(self, X, y):
        """训练随机森林"""
        self.trees = []

        for _ in range(self.n_trees):
            # Bootstrap采样
            X_boot, y_boot = self._bootstrap(X, y)

            # 构建决策树
            tree = DecisionTree(
                max_depth=self.max_depth,
                min_samples_split=self.min_samples_split
            )
            tree.fit(X_boot, y_boot)
            self.trees.append(tree)

    def predict(self, X):
        """预测：多数投票"""
        # 收集所有树的预测结果
        tree_preds = np.array([tree.predict(X) for tree in self.trees])

        # 多数投票
        final_preds = []
        for i in range(len(X)):
            votes = tree_preds[:, i]
            counter = Counter(votes)
            final_preds.append(counter.most_common(1)[0][0])

        return np.array(final_preds)

# 使用示例
from sklearn.datasets import make_classification
X, y = make_classification(n_samples=100, n_features=4, random_state=42)

forest = RandomForest(n_trees=20)
forest.fit(X, y)

print(f"训练集准确率: {(forest.predict(X) == y).mean():.2%}")
```

**常用评估指标：**

```python
# 模型评估指标实现
import numpy as np
from collections import Counter

class ModelEvaluator:
    """模型评估器"""

    @staticmethod
    def confusion_matrix(y_true, y_pred):
        """计算混淆矩阵"""
        classes = np.unique(y_true)
        n_classes = len(classes)
        matrix = np.zeros((n_classes, n_classes), dtype=int)

        for true_class, pred_class in zip(y_true, y_pred):
            true_idx = np.where(classes == true_class)[0][0]
            pred_idx = np.where(classes == pred_class)[0][0]
            matrix[true_idx][pred_idx] += 1

        return matrix

    @staticmethod
    def accuracy(y_true, y_pred):
        """准确率"""
        return np.mean(y_true == y_pred)

    @staticmethod
    def precision(y_true, y_pred, class_label=1):
        """精确率"""
        true_positives = np.sum((y_true == class_label) & (y_pred == class_label))
        predicted_positives = np.sum(y_pred == class_label)

        if predicted_positives == 0:
            return 0
        return true_positives / predicted_positives

    @staticmethod
    def recall(y_true, y_pred, class_label=1):
        """召回率"""
        true_positives = np.sum((y_true == class_label) & (y_pred == class_label))
        actual_positives = np.sum(y_true == class_label)

        if actual_positives == 0:
            return 0
        return true_positives / actual_positives

    @staticmethod
    def f1_score(y_true, y_pred, class_label=1):
        """F1分数"""
        precision = ModelEvaluator.precision(y_true, y_pred, class_label)
        recall = ModelEvaluator.recall(y_true, y_pred, class_label)

        if precision + recall == 0:
            return 0
        return 2 * (precision * recall) / (precision + recall)

    @staticmethod
    def roc_curve(y_true, y_pred_proba):
        """计算ROC曲线"""
        thresholds = np.linspace(0, 1, 101)
        tprs = []
        fprs = []

        for threshold in thresholds:
            y_pred = (y_pred_proba >= threshold).astype(int)

            # 计算TPR和FPR
            tp = np.sum((y_true == 1) & (y_pred == 1))
            fp = np.sum((y_true == 0) & (y_pred == 1))
            tn = np.sum((y_true == 0) & (y_pred == 0))
            fn = np.sum((y_true == 1) & (y_pred == 0))

            tpr = tp / (tp + fn) if (tp + fn) > 0 else 0
            fpr = fp / (fp + tn) if (fp + tn) > 0 else 0

            tprs.append(tpr)
            fprs.append(fpr)

        return fprs, tprs, thresholds

    @staticmethod
    def auc(y_true, y_pred_proba):
        """计算AUC值"""
        fprs, tprs, _ = ModelEvaluator.roc_curve(y_true, y_pred_proba)

        # 梯形法则计算AUC
        auc = 0
        for i in range(len(fprs) - 1):
            auc += (fprs[i+1] - fprs[i]) * (tprs[i] + tprs[i+1]) / 2

        return auc

# 使用示例
y_true = np.array([1, 0, 1, 0, 1, 0, 1, 0])
y_pred = np.array([1, 0, 1, 0, 0, 0, 1, 1])
y_pred_proba = np.array([0.9, 0.3, 0.8, 0.4, 0.6, 0.2, 0.85, 0.45])

evaluator = ModelEvaluator()

print(f"准确率: {evaluator.accuracy(y_true, y_pred):.4f}")
print(f"精确率: {evaluator.precision(y_true, y_pred):.4f}")
print(f"召回率: {evaluator.recall(y_true, y_pred):.4f}")
print(f"F1分数: {evaluator.f1_score(y_true, y_pred):.4f}")
print(f"AUC: {evaluator.auc(y_true, y_pred_proba):.4f}")
```

```python
# 过拟合与欠拟合的识别与处理
import numpy as np
import matplotlib.pyplot as plt
from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression, Ridge
from sklearn.pipeline import Pipeline

def generate_data(n_samples=100):

    np.random.seed(42)
    X = np.linspace(0, 10, n_samples)
    y = np.sin(X) + np.random.normal(0, 0.1, n_samples)
    return X.reshape(-1, 1), y

def plot_model_comparison():
    """可视化不同复杂度的模型"""

    # 准备测试数据

        model = Pipeline([

            ('linear', LinearRegression())
        ])

        # 训练模型
        model.fit(X, y)

        # 预测
        y_pred = model.predict(X_test)

        axes[i].set_title(f'多项式度数={degree}, R²={train_score:.3f}')
        axes[i].legend()
        axes[i].set_ylim(-1.5, 1.5)

    plt.tight_layout()
    plt.savefig('model_complexity.png')

# 使用正则化防止过拟合
class RegularizedLinearRegression:
    """带L2正则化的线性回归（岭回归）"""

    def __init__(self, alpha=1.0):
        self.alpha = alpha
        self.weights = None
        self.bias = None

    def fit(self, X, y):
        n_samples, n_features = X.shape

        # 闭式解：θ = (X^T X + αI)^(-1) X^T y
        XTX = np.dot(X.T, X)
        regularization = self.alpha * np.eye(n_features)
        XTX_reg = XTX + regularization

        try:
            theta = np.linalg.solve(XTX_reg, np.dot(X.T, y))
        except np.linalg.LinAlgError:
            theta = np.linalg.lstsq(XTX_reg, np.dot(X.T, y), rcond=None)[0]

        self.weights = theta[:-1]
        self.bias = theta[-1]

    def predict(self, X):
        return np.dot(X, self.weights) + self.bias

# 使用示例
X_train, y_train = generate_data(50)
X_test, y_test = generate_data(100)

# 比较不同正则化强度的效果
alphas = [0.001, 0.1, 1, 10, 100]
for alpha in alphas:
    # 扩展多项式特征
    poly = PolynomialFeatures(degree=10)
    X_train_poly = poly.fit_transform(X_train)
    X_test_poly = poly.transform(X_test)

    # 训练岭回归
    model = RegularizedLinearRegression(alpha=alpha)
    model.fit(X_train_poly, y_train)

    train_pred = model.predict(X_train_poly)
    test_pred = model.predict(X_test_poly)

    train_mse = np.mean((train_pred - y_train) ** 2)
    test_mse = np.mean((test_pred - y_test) ** 2)

    print(f"alpha={alpha:6.3f}: 训练MSE={train_mse:.4f}, 测试MSE={test_mse:.4f}")
```

### 1.1.4 特征工程

**特征工程常用技术：**

```python
# 特征工程技术实现
import numpy as np

class FeatureEngineer:
    """特征工程工具类"""

    @staticmethod
    def normalize(X):
        """最小-最大归一化"""
        min_val = X.min(axis=0)
        max_val = X.max(axis=0)
        return (X - min_val) / (max_val - min_val + 1e-8)

    @staticmethod
    def standardize(X):
        """标准化（Z-score归一化）"""
        mean = X.mean(axis=0)
        std = X.std(axis=0)
        return (X - mean) / (std + 1e-8)

    @staticmethod
    def log_transform(X):
        """对数变换：处理偏态分布"""
        return np.log1p(X)  # log(1+x) 避免log(0)

    @staticmethod
    def polynomial_features(X, degree=2):
        """多项式特征扩展"""
        n_samples, n_features = X.shape
        poly_features = [X]

        for d in range(2, degree + 1):
            for i in range(n_features):
                poly_features.append(X[:, i:i+1] ** d)

        return np.hstack(poly_features)

    @staticmethod
    def one_hot_encode(X, categories_list):
        """独热编码"""
        n_samples = X.shape[0]
        encoded = []

        for i, categories in enumerate(categories_list):
            one_hot = np.zeros((n_samples, len(categories)))
            for j, cat in enumerate(categories):
                one_hot[:, j] = (X[:, i] == cat).astype(int)
            encoded.append(one_hot)

        return np.hstack(encoded)

    @staticmethod
    def binning(X, n_bins=5, strategy='uniform'):
        """特征分箱"""
        if strategy == 'uniform':
            # 等宽分箱
            bins = np.linspace(X.min(), X.max(), n_bins + 1)
        else:
            # 等频分箱
            bins = np.percentile(X, np.linspace(0, 100, n_bins + 1))

        return np.digitize(X, bins) - 1

    @staticmethod
    def interaction_features(X, feature_indices=None):
        """交互特征：特征相乘"""
        if feature_indices is None:
            feature_indices = range(X.shape[1])

        interactions = []
        for i in feature_indices:
            for j in feature_indices:
                if i < j:
                    interactions.append(X[:, i] * X[:, j])

        return np.column_stack([X] + interactions)
```

```python
# 神经网络基础实现
import numpy as np

class NeuralNetwork:
    """简单神经网络实现"""

    def __init__(self, layer_sizes):
        """
        初始化神经网络
        layer_sizes: 每层神经元数量，如 [784, 256, 128, 10]
        """
        self.layer_sizes = layer_sizes
        self.n_layers = len(layer_sizes)
        self.weights = []
        self.biases = []

        # 初始化权重和偏置
        np.random.seed(42)
        for i in range(self.n_layers - 1):
            # He初始化：适合ReLU激活函数
            w = np.random.randn(layer_sizes[i], layer_sizes[i+1]) * np.sqrt(2.0 / layer_sizes[i])
            b = np.zeros((1, layer_sizes[i+1]))
            self.weights.append(w)
            self.biases.append(b)

    def sigmoid(self, x):
        """Sigmoid激活函数"""
        return 1 / (1 + np.exp(-np.clip(x, -500, 500)))

    def sigmoid_derivative(self, x):
        """Sigmoid导数"""
        return x * (1 - x)

    def relu(self, x):
        """ReLU激活函数"""
        return np.maximum(0, x)

    def relu_derivative(self, x):
        """ReLU导数"""
        return (x > 0).astype(float)

    def softmax(self, x):
        """Softmax激活函数（用于多分类输出层）"""
        exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
        return exp_x / np.sum(exp_x, axis=1, keepdims=True)

    def forward(self, X):
        """前向传播"""
        self.activations = [X]
        self.z_values = []

        current = X
        for i in range(len(self.weights) - 1):
            z = np.dot(current, self.weights[i]) + self.biases[i]
            self.z_values.append(z)
            current = self.relu(z)  # 隐藏层使用ReLU
            self.activations.append(current)

        # 输出层
        z = np.dot(current, self.weights[-1]) + self.biases[-1]
        self.z_values.append(z)
        output = self.softmax(z)  # 输出层使用Softmax
        self.activations.append(output)

        return output

    def backward(self, X, y, learning_rate):
        """反向传播"""
        m = X.shape[0]

        # 输出层误差
        delta = self.activations[-1] - y

        # 反向传播
        for i in range(len(self.weights) - 1, -1, -1):
            # 计算梯度
            dw = np.dot(self.activations[i].T, delta) / m
            db = np.sum(delta, axis=0, keepdims=True) / m

            # 更新参数
            self.weights[i] -= learning_rate * dw
            self.biases[i] -= learning_rate * db

            # 计算上一层的误差
            if i > 0:
                delta = np.dot(delta, self.weights[i].T) * self.relu_derivative(self.z_values[i-1])

    def train(self, X, y, epochs=1000, learning_rate=0.1):
        """训练神经网络"""
        for epoch in range(epochs):
            # 前向传播
            output = self.forward(X)

            # 计算损失（交叉熵损失）
            loss = -np.sum(y * np.log(output + 1e-8)) / X.shape[0]

            # 反向传播
            self.backward(X, y, learning_rate)

            if epoch % 100 == 0:
                print(f"Epoch {epoch}, Loss: {loss:.4f}")

    def predict(self, X):
        """预测"""
        output = self.forward(X)
        return np.argmax(output, axis=1)
```

```python
class LSTM:

    def __init__(self, input_size, hidden_size, output_size):
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size

        # LSTM权重初始化
        # 四个门：输入门、遗忘门、输出门、候选值
        combined_size = input_size + hidden_size

        # Xavier初始化
        scale = np.sqrt(2.0 / combined_size)

        # 门控权重
        self.Wf = np.random.randn(combined_size, hidden_size) * scale  # 遗忘门
        self.Wi = np.random.randn(combined_size, hidden_size) * scale  # 输入门
        self.Wc = np.random.randn(combined_size, hidden_size) * scale  # 候选值
        self.Wo = np.random.randn(combined_size, hidden_size) * scale  # 输出门

        # 偏置
        self.bf = np.zeros((1, hidden_size))
        self.bi = np.zeros((1, hidden_size))
        self.bc = np.zeros((1, hidden_size))
        self.bo = np.zeros((1, hidden_size))

        # 输出层权重
        self.Wy = np.random.randn(hidden_size, output_size) * np.sqrt(2.0 / hidden_size)
        self.by = np.zeros((1, output_size))

    def sigmoid(self, x):
        return 1 / (1 + np.exp(-np.clip(x, -500, 500)))

    def forward(self, X):
        """前向传播"""
        n_timesteps = X.shape[0]

        # 初始化
        h = np.zeros((1, self.hidden_size))
        c = np.zeros((1, self.hidden_size))  # 细胞状态

        self.inputs = []
        self.fs = []  # 遗忘门
        self.is_ = []  # 输入门
        self.cs = []  # 候选值
        self.os = []  # 输出门
        self.hs = [h]  # 隐藏状态
        self.cs_states = [c]  # 细胞状态

        for t in range(n_timesteps):
            xt = X[t].reshape(1, -1)
            self.inputs.append(xt)

            # 拼接隐藏状态和输入
            combined = np.hstack([h, xt])

            # 遗忘门：决定保留多少之前的记忆
            f = self.sigmoid(np.dot(combined, self.Wf) + self.bf)
            self.fs.append(f)

            # 输入门：决定加入多少新信息
            i = self.sigmoid(np.dot(combined, self.Wi) + self.bi)
            self.is_.append(i)

            # 候选值：新的候选记忆
            c_tilde = np.tanh(np.dot(combined, self.Wc) + self.bc)
            self.cs.append(c_tilde)

            # 更新细胞状态
            c = f * c + i * c_tilde
            self.cs_states.append(c)

            # 输出门：决定输出什么
            o = self.sigmoid(np.dot(combined, self.Wo) + self.bo)
            self.os.append(o)

            # 更新隐藏状态
            h = o * np.tanh(c)
            self.hs.append(h)

        # 输出层
        y = np.dot(h, self.Wy) + self.by
        return y

    def predict(self, X):
        output = self.forward(X)
        return np.argmax(output, axis=1)
```

```python
# Transformer注意力机制实现
import numpy as np

class MultiHeadAttention:
    """多头注意力机制"""

    def __init__(self, d_model, n_heads):
        self.d_model = d_model
        self.n_heads = n_heads
        self.d_k = d_model // n_heads

        # 初始化权重矩阵
        self.W_q = np.random.randn(d_model, d_model) * np.sqrt(2.0 / d_model)
        self.W_k = np.random.randn(d_model, d_model) * np.sqrt(2.0 / d_model)
        self.W_v = np.random.randn(d_model, d_model) * np.sqrt(2.0 / d_model)
        self.W_o = np.random.randn(d_model, d_model) * np.sqrt(2.0 / d_model)

    def scaled_dot_product_attention(self, Q, K, V, mask=None):
        """缩放点积注意力"""
        # 计算注意力分数
        d_k = Q.shape[-1]
        scores = np.matmul(Q, K.transpose(-2, -1)) / np.sqrt(d_k)

        # 应用掩码（可选）
        if mask is not None:
            scores = np.where(mask == 0, -1e9, scores)

        # Softmax归一化
        attention_weights = self.softmax(scores)

        # 加权求和
        output = np.matmul(attention_weights, V)

        return output, attention_weights

    def softmax(self, x):
        exp_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
        return exp_x / np.sum(exp_x, axis=-1, keepdims=True)

    def forward(self, X, mask=None):
        """前向传播"""
        n_samples, seq_len, d_model = X.shape

        # 线性变换生成Q, K, V
        Q = np.matmul(X, self.W_q).reshape(n_samples, seq_len, self.n_heads, self.d_k).transpose(0, 2, 1, 3)
        K = np.matmul(X, self.W_k).reshape(n_samples, seq_len, self.n_heads, self.d_k).transpose(0, 2, 1, 3)
        V = np.matmul(X, self.W_v).reshape(n_samples, seq_len, self.n_heads, self.d_k).transpose(0, 2, 1, 3)

        # 多头注意力
        attention_output, _ = self.scaled_dot_product_attention(Q, K, V, mask)

        # 拼接多头输出
        attention_output = attention_output.transpose(0, 2, 1, 3).reshape(n_samples, seq_len, d_model)

        # 输出线性变换
        output = np.matmul(attention_output, self.W_o)

        return output

class PositionalEncoding:
    """位置编码"""

    def __init__(self, d_model, max_len=5000):
        self.d_model = d_model

        # 创建位置编码矩阵
        pe = np.zeros((max_len, d_model))

        for pos in range(max_len):
            for i in range(0, d_model, 2):
                # 偶数位置：sin
                pe[pos, i] = np.sin(pos / np.power(10000, (2 * i) / d_model))
                # 奇数位置：cos
                if i + 1 < d_model:
                    pe[pos, i + 1] = np.cos(pos / np.power(10000, (2 * (i + 1)) / d_model))

        self.pe = pe

    def forward(self, X):
        """添加位置编码"""
        n_samples, seq_len, d_model = X.shape
        return X + self.pe[:seq_len, :]

class FeedForward:
    """前馈神经网络"""

    def __init__(self, d_model, d_ff):
        self.d_model = d_model
        self.d_ff = d_ff

        # 两层全连接网络
        self.W1 = np.random.randn(d_model, d_ff) * np.sqrt(2.0 / d_model)
        self.W2 = np.random.randn(d_ff, d_model) * np.sqrt(2.0 / d_ff)
        self.b1 = np.zeros((1, d_ff))
        self.b2 = np.zeros((1, d_model))

    def forward(self, X):
        """前向传播"""
        # 第一层：扩展维度 + ReLU激活
        hidden = np.maximum(0, np.matmul(X, self.W1) + self.b1)
        # 第二层：压缩回原始维度
        output = np.matmul(hidden, self.W2) + self.b2
        return output

class TransformerBlock:
    """Transformer编码器块"""

    def __init__(self, d_model, n_heads, d_ff, dropout=0.1):
        self.attention = MultiHeadAttention(d_model, n_heads)
        self.feed_forward = FeedForward(d_model, d_ff)

        # Layer Normalization
        self.norm1 = LayerNormalization(d_model)
        self.norm2 = LayerNormalization(d_model)

        self.dropout = dropout

    def forward(self, X, mask=None):
        # 多头注意力 + 残差连接 + LayerNorm
        attention_output = self.attention.forward(X, mask)
        X = self.norm1.forward(X + attention_output)

        # 前馈网络 + 残差连接 + LayerNorm
        ff_output = self.feed_forward.forward(X)
        X = self.norm2.forward(X + ff_output)

        return X

class LayerNormalization:
    """层归一化"""

    def __init__(self, d_model, epsilon=1e-6):
        self.epsilon = epsilon
        self.gamma = np.ones((1, d_model))
        self.beta = np.zeros((1, d_model))

    def forward(self, X):
        # 计算均值和方差
        mean = np.mean(X, axis=-1, keepdims=True)
        variance = np.var(X, axis=-1, keepdims=True)

        # 归一化
        X_normalized = (X - mean) / np.sqrt(variance + self.epsilon)

        # 缩放和偏移
        return self.gamma * X_normalized + self.beta
```

```python
# 简化版GPT模型实现
import numpy as np

class SimplifiedGPT:
    """简化版GPT模型"""

    def __init__(self, vocab_size, d_model=512, n_heads=8, n_layers=6, max_seq_len=1024):
        self.vocab_size = vocab_size
        self.d_model = d_model
        self.n_heads = n_heads
        self.n_layers = n_layers
        self.max_seq_len = max_seq_len

        # 词嵌入
        self.token_embedding = np.random.randn(vocab_size, d_model) * 0.01
        self.position_embedding = np.random.randn(max_seq_len, d_model) * 0.01

        # Transformer层
        self.transformer_layers = []
        for _ in range(n_layers):
            self.transformer_layers.append({
                'attention': np.random.randn(d_model, d_model) * 0.01,
                'ffn': np.random.randn(d_model, d_model * 4) * 0.01,
                'layer_norm_g': np.ones(d_model),
                'layer_norm_b': np.zeros(d_model)
            })

        # 输出层
        self.lm_head = np.random.randn(d_model, vocab_size) * 0.01

    def layer_norm(self, x, g, b, epsilon=1e-6):
        """层归一化"""
        mean = np.mean(x, axis=-1, keepdims=True)
        std = np.std(x, axis=-1, keepdims=True)
        return g * (x - mean) / (std + epsilon) + b

    def forward(self, input_ids):
        """前向传播"""
        batch_size, seq_len = input_ids.shape

        # 词嵌入 + 位置嵌入
        token_embeds = self.token_embedding[input_ids]
        position_ids = np.arange(seq_len)
        position_embeds = self.position_embedding[position_ids]

        hidden_states = token_embeds + position_embeds

        # Transformer层
        for layer in self.transformer_layers:
            # 简化版自注意力
            attn_scores = np.matmul(hidden_states, layer['attention'])
            attn_scores = attn_scores / np.sqrt(self.d_model)
            attn_probs = self.softmax(attn_scores)
            attn_output = np.matmul(attn_probs, hidden_states)

            # 残差连接 + 层归一化
            hidden_states = self.layer_norm(
                hidden_states + attn_output,
                layer['layer_norm_g'],
                layer['layer_norm_b']
            )

            # 前馈网络（简化）
            ffn_output = np.matmul(hidden_states, layer['ffn'])
            ffn_output = np.maximum(0, ffn_output)  # ReLU

            # 残差连接 + 层归一化
            hidden_states = self.layer_norm(
                hidden_states + ffn_output,
                layer['layer_norm_g'],
                layer['layer_norm_b']
            )

        # 语言模型头
        logits = np.matmul(hidden_states, self.lm_head)

        return logits

    def softmax(self, x):
        exp_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
        return exp_x / np.sum(exp_x, axis=-1, keepdims=True)

    def generate(self, input_ids, max_new_tokens=50, temperature=1.0, top_k=None):
        """文本生成"""
        for _ in range(max_new_tokens):
            # 只保留最后max_seq_len个token
            input_ids_cond = input_ids[:, -self.max_seq_len:]

            # 前向传播
            logits = self.forward(input_ids_cond)

            # 取最后一个token的logits
            next_token_logits = logits[:, -1, :] / temperature

            # Top-k采样
            if top_k is not None:
                indices = np.argsort(next_token_logits)[-top_k:]
                next_token_logits[~np.isin(np.arange(len(next_token_logits)), indices)] = -np.inf

            # 转换为概率并采样
            probs = self.softmax(next_token_logits)
            next_token = np.random.choice(len(probs[0]), p=probs[0])

            # 添加到序列
            input_ids = np.concatenate([input_ids, [[next_token]]], axis=1)

        return input_ids
```

```javascript
class LLMClient {
    constructor(config) {

     * 聊天完成 API
     * @param {Array} messages - 消息数组
     * @param {Object} options - 可选参数
     */
    async chat(messages, options = {}) {
        const requestOptions = {
            ...this.defaultOptions,
            ...options,
            model: options.model || this.model,
            messages: messages
        };

        const response = await fetch(`${this.baseURL}/chat/completions`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.apiKey}`
            },
            body: JSON.stringify(requestOptions)
        });

        if (!response.ok) {
            const error = await response.json();
            throw new LLMError(error);
        }

        return response.json();
    }

     * 流式聊天完成
     * @param {Array} messages - 消息数组
     * @param {Function} onChunk - 流式数据回调
     * @param {Object} options - 可选参数
     */
    async streamChat(messages, onChunk, options = {}) {
        const requestOptions = {
            ...this.defaultOptions,
            ...options,
            model: options.model || this.model,
            messages: messages,
            stream: true
        };

        const response = await fetch(`${this.baseURL}/chat/completions`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.apiKey}`
            },
            body: JSON.stringify(requestOptions)
        });

        if (!response.ok) {
            const error = await response.json();
            throw new LLMError(error);
        }

        const reader = response.body.getReader();
        const decoder = new TextDecoder('utf-8');
        let buffer = '';

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;

            buffer += decoder.decode(value, { stream: true });
            const lines = buffer.split('\n');
            buffer = lines.pop() || '';

            for (const line of lines) {
                const trimmed = line.trim();
                if (!trimmed || !trimmed.startsWith('data:')) continue;

                const data = trimmed.slice(6).trim();
                if (data === '[DONE]') return;

                try {
                    const parsed = JSON.parse(data);
                    const content = parsed.choices?.[0]?.delta?.content || '';
                    if (content) {
                        onChunk(content, parsed);
                    }
                } catch (e) {
                    console.warn('Parse chunk error:', e);
                }
            }
        }
    }

    /**
     * 嵌入向量生成
     * @param {string|string[]} text - 文本或文本数组
     */
    async embeddings(text) {
        const texts = Array.isArray(text) ? text : [text];

        const response = await fetch(`${this.baseURL}/embeddings`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.apiKey}`
            },
            body: JSON.stringify({
                model: 'text-embedding-3-small',
                input: texts
            })
        });

        if (!response.ok) {
            const error = await response.json();
            throw new LLMError(error);
        }

        return response.json();
    }
}
```

```javascript
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'x-api-key': this.apiKey,
                'anthropic-version': '2023-06-01'

            body: JSON.stringify({
                model: options.model || this.defaultModel,
                messages: this.formatMessages(messages),
                max_tokens: options.max_tokens || 4096,
                temperature: options.temperature || 1,
                stream: true
            })
        });

        const reader = response.body.getReader();
        const decoder = new TextDecoder('utf-8');
        let buffer = '';

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;

            buffer += decoder.decode(value, { stream: true });
            const lines = buffer.split('\n');
            buffer = lines.pop() || '';

            for (const line of lines) {
                if (!line.startsWith('data:')) continue;

                const data = line.slice(5).trim();
                if (data === '[DONE]') return;

                try {
                    const parsed = JSON.parse(data);
                    if (parsed.type === 'content_block_delta') {
                        const content = parsed.delta?.text || '';
                        if (content) onChunk(content, parsed);
                    }
                } catch (e) {}
            }
        }
    }

    formatMessages(messages) {
        // 过滤掉system消息，单独处理
        return messages.filter(m => m.role !== 'system');
    }

        const systemMsg = messages.find(m => m.role === 'system');
```

```javascript
class StreamProcessor {
    constructor() {
        this.buffer = '';
        this.eventListeners = new Map();
    }

    /**
     * 处理 SSE 流数据
     * @param {ReadableStreamDefaultReader} reader - 可读流读取器
     * @param {Function} onMessage - 消息回调
     */
    async processStream(reader, onMessage) {
        const decoder = new TextDecoder('utf-8');

        while (true) {
            const { done, value } = await reader.read();
            if (done) {
                // 处理剩余缓冲数据
                if (this.buffer.trim()) {
                    this.parseSSEData(this.buffer, onMessage);
                }
                break;
            }

            this.buffer += decoder.decode(value, { stream: true });

            // 按行分割
            const lines = this.buffer.split('\n');
            this.buffer = lines.pop() || '';

            for (const line of lines) {
                if (this.isValidSSELine(line)) {
                    this.parseSSEData(line, onMessage);
                }
            }
        }
    }

    /**
     * 验证SSE行是否有效
     */
    isValidSSELine(line) {
        const trimmed = line.trim();
        return trimmed && !trimmed.startsWith(':');
    }

    /**
     * 解析SSE数据行
     */
    parseSSEData(line, onMessage) {
        // 格式: "data: {\"content\": \"hello\"}"
        if (!line.startsWith('data:')) return;

        const data = line.slice(5).trim();
        if (data === '[DONE]') {
            onMessage({ type: 'done' });
            return;
        }

        try {
            const parsed = JSON.parse(data);
            onMessage({ type: 'chunk', data: parsed });
        } catch (e) {
            console.warn('Failed to parse SSE data:', data);
        }
    }

    /**
     * 解析多种格式的SSE数据
     */
    parseSSERaw(text) {
        const result = [];
        const lines = text.split('\n');

        let currentEvent = null;
        let currentData = [];

        for (const line of lines) {
            if (line.startsWith('event:')) {
                currentEvent = line.slice(6).trim();
                continue;
            }

            if (line.startsWith('data:')) {
                const data = line.slice(5).trim();
                if (data) {
                    currentData.push(data);
                }
                continue;
            }

            // 空行表示事件结束
            if (!line.trim() && currentData.length > 0) {
                result.push({
                    event: currentEvent,
                    data: currentData.join('\n')
                });
                currentData = [];
                currentEvent = null;
            }
        }

        return result;
    }
}

                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({

                })

            if (!response.ok) {
                throw new Error('Request failed');
            }

            // 处理流式响应
            const reader = response.body.getReader();
            const processor = new StreamProcessor();

            await processor.processStream(reader, (event) => {
                if (event.type === 'chunk') {
                    const content = event.data.choices?.[0]?.delta?.content || '';
                    if (content) {
                        this.fullContent += content;
                        this.updateMessageContent(this.fullContent);
                        onChunk?.(content);
                    }
                }
            });

        } catch (error) {
            this.showError(error.message);
        } finally {
            this.hideTypingIndicator();
            this.isStreaming = false;
        }
```

```javascript
class AIWebSocketClient {
    constructor(url, options = {}) {

        this.maxReconnectAttempts = options.maxReconnectAttempts || 5;

        this.messageQueue = [];
        this.listeners = new Map();
    }

    connect() {
        return new Promise((resolve, reject) => {
            this.ws = new WebSocket(this.url);

            this.ws.onopen = () => {
                console.log('WebSocket connected');
                this.reconnectAttempts = 0;
                this.flushMessageQueue();
                resolve();
            };

            this.ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                this.handleMessage(data);
            };

            this.ws.onerror = (error) => {
                console.error('WebSocket error:', error);
                this.emit('error', error);
            };

            this.ws.onclose = () => {
                console.log('WebSocket closed');
                this.emit('close');
                this.attemptReconnect();
            };
        });
    }

    send(type, payload) {
        const message = JSON.stringify({ type, payload, timestamp: Date.now() });

        if (this.ws?.readyState === WebSocket.OPEN) {
            this.ws.send(message);
        } else {
            this.messageQueue.push(message);
        }
    }

    sendMessage(content, conversationId = null) {
        this.send('chat', { content, conversationId });
    }

    sendTyping() {
        this.send('typing', { isTyping: true });
    }

    handleMessage(data) {
        switch (data.type) {
            case 'chunk':
                this.emit('chunk', data.payload);
                break;
            case 'complete':
                this.emit('complete', data.payload);
                break;
            case 'error':
                this.emit('error', data.payload);
                break;
            case 'typing':
                this.emit('typing', data.payload);
                break;
            default:
                this.emit('message', data);
        }
    }

    attemptReconnect() {
        if (this.reconnectAttempts < this.maxReconnectAttempts) {
            this.reconnectAttempts++;
            console.log(`Attempting reconnect (${this.reconnectAttempts}/${this.maxReconnectAttempts})...`);

            setTimeout(() => {
                this.connect().catch(console.error);
            }, this.reconnectDelay * this.reconnectAttempts);
        }
    }

    flushMessageQueue() {
        while (this.messageQueue.length > 0) {
            const message = this.messageQueue.shift();
            this.ws.send(message);
        }
    }

    on(event, callback) {
        if (!this.listeners.has(event)) {
            this.listeners.set(event, []);
        }
        this.listeners.get(event).push(callback);
    }

    off(event, callback) {
        const callbacks = this.listeners.get(event);
        if (callbacks) {
            const index = callbacks.indexOf(callback);
            if (index > -1) callbacks.splice(index, 1);
        }
    }

    emit(event, data) {
        const callbacks = this.listeners.get(event);
        if (callbacks) {
            callbacks.forEach(cb => cb(data));
        }
    }

    close() {
        this.maxReconnectAttempts = 0;  // 防止重连
        this.ws?.close();
    }
}
```

```javascript
class PromptEngineer {
    /**
     * 零样本提示 (Zero-shot Prompting)
     * 直接给出任务，不提供示例
     */
    static zeroShot(task, input) {
        return `任务: ${task}\n输入: ${input}\n输出:`;
    }

    /**
     * 少样本提示 (Few-shot Prompting)
     * 提供少量示例帮助模型理解任务
     */
    static fewShot(task, examples, input) {
        const exampleStr = examples.map(ex =>
            `输入: ${ex.input}\n输出: ${ex.output}`
        ).join('\n\n');

        return `任务: ${task}\n\n示例:\n${exampleStr}\n\n请根据以上示例完成以下任务:\n输入: ${input}\n输出:`;
    }

    /**
     * 思维链提示 (Chain of Thought, CoT)
     * 要求模型展示推理过程
     */
    static chainOfThought(task, input) {
        return `${task}\n\n请一步步思考，并展示你的推理过程。\n\n问题: ${input}\n\n推理过程:`;
    }

    /**
     * 自我一致性提示 (Self-Consistency)
     * 多次采样，综合多个结果
     */
    static selfConsistency(task, input, options = {}) {
        const { nSamples = 5, temperature = 0.7 } = options;

        return {
            prompt: `${task}\n\n问题: ${input}\n\n请给出你的推理过程和答案。`,
            options: { temperature, n: nSamples },
            aggregation: 'majority_vote'  // 多数投票
        };
    }

    /**
     * 角色提示 (Role Prompting)
     * 给模型设定一个角色
     */
    static rolePrompt(role, background, task) {
        return `你是一个${role}。\n\n${background}\n\n现在，请帮我完成以下任务:\n${task}`;
    }

    /**
     * 结构化输出提示
     * 指定输出格式
     */
    static structuredOutput(task, outputFormat) {
        const formatSpec = this.generateFormatSpec(outputFormat);

        return `${task}\n\n请严格按照以下JSON格式输出:\n${formatSpec}`;
    }

    static generateFormatSpec(format) {
        if (format.type === 'json') {
            return `\`\`\`json\n${JSON.stringify(format.schema, null, 2)}\n\`\`\``;
        }

        if (format.type === 'markdown') {
            return format.sections.map(s =>
                `## ${s.title}\n${s.description}`
            ).join('\n\n');
        }

        return '';
    }

    /**
     * 上下文学习 (In-Context Learning)
     * 在prompt中动态提供相关上下文
     */
    static inContextLearning(task, contextDocs, query) {
        const context = contextDocs.map((doc, i) =>
            `[参考文档 ${i + 1}]\n标题: ${doc.title}\n内容: ${doc.content}`
        ).join('\n\n');

        return `请根据以下参考文档回答问题。\n\n${context}\n\n---\n\n问题: ${query}\n\n回答:`;
    }

    /**
     * 思维树提示 (Tree of Thoughts)
     * 探索多种解决思路
     */
    static treeOfThoughts(task, input) {
        return `${task}\n\n问题: ${input}\n\n请从多个角度思考这个问题，探索不同的解决思路，然后给出最佳方案。\n\n思考过程:\n1. 思路一:\n2. 思路二:\n3. 思路三:\n\n综合分析:`;
    }

    /**
     * ReAct提示 (Reasoning + Acting)
     * 结合推理和行动
     */
    static ReAct(task, input, tools) {

            `- ${t.name}: ${t.description} (参数: ${t.parameters})`

    }
```

```javascript
class ConversationContextManager {
    constructor(maxTokens = 6000) {
        this.maxTokens = maxTokens;
        this.messages = [];
        this.systemPrompt = '';
    }

    setSystemPrompt(prompt) {
        this.systemPrompt = prompt;
        this.messages.unshift({ role: 'system', content: prompt });
    }

    addMessage(role, content) {
        this.messages.push({ role, content, timestamp: Date.now() });
        this.pruneIfNeeded();
    }

    pruneIfNeeded() {
        let totalTokens = this.estimateTokens(this.messages);

        while (totalTokens > this.maxTokens && this.messages.length > 1) {
            // 移除最旧的用户消息（保留system）
            const nonSystemMessages = this.messages.filter(m => m.role !== 'system');
            if (nonSystemMessages.length > 0) {
                const oldestIndex = this.messages.findIndex(m => m.role !== 'system');
                this.messages.splice(oldestIndex, 1);
            }
            totalTokens = this.estimateTokens(this.messages);
        }
    }

    estimateTokens(messages) {
        // 粗略估算：每个字符约0.25个token
        return messages.reduce((sum, msg) =>
            sum + msg.content.length * 0.25, 0
        );
    }

    getContext() {
        return this.messages;
    }

    clear() {
        this.messages = this.systemPrompt ?
            [{ role: 'system', content: this.systemPrompt }] : [];
    }
}
```

```javascript
                if (i < maxRetries - 1) {

class RateLimiter {
    constructor() {
        this.limits = {
            openai: { rpm: 500, rpd: 1000000 },      // 每分钟500次，每天100万
            anthropic: { rpm: 50, rpd: 100000 },
            deepseek: { rpm: 120, rpd: 100000 }
        };

        this.requests = {
            openai: { minute: [], day: [] },
            anthropic: { minute: [], day: [] },
            deepseek: { minute: [], day: [] }
        };
    }

    async check(provider) {
        const now = Date.now();
        const limits = this.limits[provider];
        const requests = this.requests[provider];

        // 清理过期请求记录
        const minuteAgo = now - 60000;
        const dayAgo = now - 86400000;

        requests.minute = requests.minute.filter(t => t > minuteAgo);
        requests.day = requests.day.filter(t => t > dayAgo);

        // 检查限制
        if (requests.minute.length >= limits.rpm) {
            throw new Error(`Rate limit exceeded. Please wait ${(requests.minute[0] - minuteAgo) / 1000} seconds`);
        }

        if (requests.day.length >= limits.rpd) {
            throw new Error('Daily rate limit exceeded');
        }

        // 记录请求
        requests.minute.push(now);
        requests.day.push(now);
    }
```

```javascript
        this.maxRetries = options.maxRetries || 3;

        this.maxDelay = options.maxDelay || 30000;
        this.exponentialBase = options.exponentialBase || 2;
        this.jitter = options.jitter !== false;
    }

    async execute(fn, context = '') {
        let lastError;

        for (let attempt = 0; attempt <= this.maxRetries; attempt++) {
            try {
                return await fn();
            } catch (error) {
                lastError = error;

                console.log(`Attempt ${attempt + 1} failed:`, error.message);

                // 不重试的情况
                if (!this.shouldRetry(error, attempt)) {
                    console.log('Not retrying - non-retryable error');
                    throw error;
                }

                // 计算延迟
                const delay = this.calculateDelay(attempt);
                console.log(`Retrying in ${delay}ms...`);

                await this.sleep(delay);
            }
        }

        throw lastError;
    }

    shouldRetry(error, attempt) {
        if (attempt >= this.maxRetries) return false;
        if (!ErrorClassifier.isRetryable(error)) return false;
        if (ErrorClassifier.isAuthError(error)) return false;
        return true;
    }

    calculateDelay(attempt) {
        // 指数退避
        let delay = this.baseDelay * Math.pow(this.exponentialBase, attempt);

        // 添加随机抖动
        if (this.jitter) {
            delay = delay * (0.5 + Math.random() * 0.5);
        }

        // 不超过最大延迟
        return Math.min(delay, this.maxDelay);
    }

    sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}

class APIErrorHandler {
    static handleResponse(response, options = {}) {
        const { onError, onRateLimit } = options;

        if (!response.ok) {
            return response.json().then(errorData => {
                const error = new RetryableError(
                    errorData.error?.message || `HTTP ${response.status}`,
                    response.status,
                    errorData.error?.code,
                    ErrorClassifier.isRetryable({ status: response.status })
                );

                // 特殊处理速率限制
                if (response.status === 429) {
                    const retryAfter = response.headers.get('Retry-After');
                    error.retryAfter = retryAfter ? parseInt(retryAfter) : null;

                    onRateLimit?.(error);
                }

                onError?.(error);
                throw error;
            });
        }

        return response.json();
    }

    static handleStream(response, onChunk, onError) {
        const reader = response.body.getReader();
        const decoder = new TextDecoder('utf-8');
        let buffer = '';

        const processStream = async () => {
            try {
                while (true) {
                    const { done, value } = await reader.read();

                    if (done) {
                        // 处理剩余数据
                        if (buffer) {
                            this.parseSSEChunk(buffer, onChunk);
                        }
                        break;
                    }

                    buffer += decoder.decode(value, { stream: true });
                    const lines = buffer.split('\n');
                    buffer = lines.pop() || '';

                    for (const line of lines) {
                        if (line.startsWith('data: ')) {
                            this.parseSSEChunk(line, onChunk);
                        }
                    }
                }
            } catch (error) {
                onError?.(error);
            }
        };

        return processStream();
    }

    static parseSSEChunk(line, onChunk) {
        const data = line.slice(6).trim();

        if (data === '[DONE]') return;

        try {
            const parsed = JSON.parse(data);
            onChunk(parsed);
        } catch (e) {
            // 忽略解析错误
        }
    }
```

```javascript
class RequestCache {
    constructor(options = {}) {
        this.maxSize = options.maxSize || 100;
        this.ttl = options.ttl || 3600000; // 1小时
        this.cache = new Map();
    }

    get(key) {
        const entry = this.cache.get(key);

        if (!entry) return null;

        // 检查过期
        if (Date.now() - entry.timestamp > this.ttl) {
            this.cache.delete(key);
            return null;
        }

        // 更新访问时间
        entry.lastAccessed = Date.now();
        entry.accessCount++;

        return entry.value;
    }

    set(key, value) {
        // 清理过期项
        this.cleanup();

        // 检查大小限制
        if (this.cache.size >= this.maxSize) {
            this.evict();
        }

        this.cache.set(key, {
            value,
            timestamp: Date.now(),
            lastAccessed: Date.now(),
            accessCount: 0
        });
    }

    cleanup() {
        const now = Date.now();

        for (const [key, entry] of this.cache) {
            if (now - entry.timestamp > this.ttl) {
                this.cache.delete(key);
            }
        }
    }

    evict() {
        // LRU: 删除最少使用的项
        let minAccess = Infinity;
        let minKey = null;

        for (const [key, entry] of this.cache) {
            if (entry.accessCount < minAccess) {
                minAccess = entry.accessCount;
                minKey = key;
            }
        }

        if (minKey) {
            this.cache.delete(minKey);
        }
    }

    clear() {
        this.cache.clear();
    }
}
```

```javascript
            statementCoverage: this.analyzeStatementCoverage(code, tests),

        // 简化实现
        const codeLines = code.split('\n').filter(l => l.trim() && !l.trim().startsWith('//'));
        const testCalls = tests.split('\n').filter(l => l.includes('expect') || l.includes('it('));

        return Math.min((testCalls.length / codeLines.length) * 100, 100).toFixed(1);
    }

    analyzeBranchCoverage(code, tests) {
        // 检测分支语句
        const branches = (code.match(/(?:if|else|switch|case|\?)/g) || []).length;
        const branchTests = (tests.match(/(?:toBeTruthy|toBeFalsy|toEqual|toContain)/g) || []).length;

        return Math.min((branchTests / branches) * 100, 100).toFixed(1);
    }

    analyzeFunctionCoverage(code, tests) {
        const functions = (code.match(/(?:function\s+\w+|\w+\s*=|=>\s*\{)/g) || []).length;
        const functionTests = (tests.match(/(?:describe|it|test)\(/g) || []).length;

        return Math.min((functionTests / functions) * 100, 100).toFixed(1);
    }

    findMissingTests(code, tests) {
        // 简单的缺失测试检测
        const missing = [];

        // 检测异常处理
        if (code.includes('throw') && !tests.includes('toThrow')) {
            missing.push('异常处理测试');

        // 检测异步
```

```javascript
// WebGPU AI推理示例

class WebGPUManager {
    constructor() {
        this.device = null;
        this.commandQueue = null;
        this.computePipeline = null;
    }

    async initialize() {
        if (!navigator.gpu) {
            throw new Error('WebGPU not supported');
        }

        const adapter = await navigator.gpu.requestAdapter();
        this.device = await adapter.requestDevice();

        console.log('WebGPU initialized');
        return this.device;
    }

    /**
     * 运行矩阵乘法计算
     */
    async runMatrixMultiply(A, B) {
        const m = A.length;
        const k = A[0].length;
        const n = B[0].length;

        // 创建缓冲区
        const bufferA = this.createBuffer(A.flat(), GPUBufferUsage.STORAGE);
        const bufferB = this.createBuffer(B.flat(), GPUBufferUsage.STORAGE);
        const bufferC = this.createBuffer(new Float32Array(m * n), GPUBufferUsage.STORAGE | GPUBufferUsage.COPY_SRC);

        // 创建计算管线
        const shaderCode = `
            @group(0) @binding(0) var<storage, read> A: array<f32>;
            @group(0) @binding(1) var<storage, read> B: array<f32>;
            @group(0) @binding(2) var<storage, read_write> C: array<f32>;

            @compute @workgroup_size(8, 8)
            fn main(@builtin(global_invocation_id) global_id: vec3<u32>) {
                let row = global_id.y;
                let col = global_id.x;
                let K = ${k}u;

                if (row >= ${m}u || col >= ${n}u) {
                    return;
                }

                var sum = 0.0;
                for (var i = 0u; i < K; i = i + 1u) {
                    sum = sum + A[row * K + i] * B[i * ${n}u + col];
                }

                C[row * ${n}u + col] = sum;
            }
        `;

        const computePipeline = await this.createComputePipeline(shaderCode);

        // 执行计算
        const commandEncoder = this.device.createCommandEncoder();
        const passEncoder = commandEncoder.beginComputePass();
        passEncoder.setPipeline(computePipeline);
        passEncoder.setBindGroup(0, this.device.createBindGroup({
            layout: computePipeline.getBindGroupLayout(0),
            entries: [
                { binding: 0, resource: { buffer: bufferA } },
                { binding: 1, resource: { buffer: bufferB } },
                { binding: 2, resource: { buffer: bufferC } }
            ]
        }));

        const workgroupCountX = Math.ceil(n / 8);
        const workgroupCountY = Math.ceil(m / 8);
        passEncoder.dispatchWorkgroups(workgroupCountX, workgroupCountY);
        passEncoder.end();

        this.device.queue.submit([commandEncoder.finish()]);

        // 读取结果
        const resultBuffer = this.createBuffer(new Float32Array(m * n), GPUBufferUsage.COPY_DST | GPUBufferUsage.MAP_READ);
        const copyEncoder = this.device.createCommandEncoder();
        copyEncoder.copyBufferToBuffer(bufferC, 0, resultBuffer, 0, m * n * 4);
        this.device.queue.submit([copyEncoder.finish()]);

        await resultBuffer.mapAsync(GPUMapMode.READ);
        const result = new Float32Array(resultBuffer.getMappedRange());

        resultBuffer.unmap();

        // 重塑为矩阵
        return Array.from({ length: m }, (_, i) =>
            Array.from({ length: n }, (_, j) => result[i * n + j])
        );
    }

    createBuffer(data, usage) {
        const buffer = this.device.createBuffer({
            size: data.byteLength,
            usage: usage | GPUBufferUsage.COPY_DST,
            mappedAtCreation: true
        });

        if (data instanceof Float32Array) {
            new Float32Array(buffer.getMappedRange()).set(data);
        } else {
            new Uint8Array(buffer.getMappedRange()).set(data);
        }

        buffer.unmap();
        return buffer;
    }

    async createComputePipeline(shaderCode) {
        const shaderModule = this.device.createShaderModule({ code: shaderCode });

        return this.device.createComputePipeline({
            layout: 'auto',
            compute: {
                module: shaderModule,
                entryPoint: 'main'
            }
        });
    }
}
```

```javascript
     * 本地模型推理
     */

        const model = this.activeModel || 'llama2';

        const response = await fetch('http://localhost:11434/api/generate', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                model,
                prompt,
                stream: options.stream || false,
                options: {
                    temperature: options.temperature || 0.7,
                    top_p: options.top_p || 0.9,
                    max_tokens: options.maxTokens || 512
                }
            })
        });

        if (!response.ok) {
            throw new Error('Generation failed');
        }

        if (options.stream) {
            return this.handleStreamResponse(response);
        }

        const data = await response.json();
        return data.response;
    }

    async handleStreamResponse(response) {
        const reader = response.body.getReader();
        const decoder = new TextDecoder();
        let result = '';

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;

            const chunk = decoder.decode(value);
            const lines = chunk.split('\n').filter(Boolean);

            for (const line of lines) {
                try {
                    const data = JSON.parse(line);
                    if (data.response) {
                        result += data.response;
                        options.onChunk?.(data.response);
                    }
                } catch (e) {}
            }
        }

        return result;
    }

    /**
     * 列出可用模型
     */
    async listModels() {
        const response = await fetch('http://localhost:11434/api/tags');
        const data = await response.json();

        return data.models.map(m => ({
            name: m.name,
            size: m.size,
            modified: m.modified_at
        }));
    }

    /**
     * 拉取模型
     */
    async pullModel(modelName, onProgress) {
        const response = await fetch('http://localhost:11434/api/pull', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ name: modelName })

        const reader = response.body.getReader();

            const { done, value } = await reader.read();
            if (done) break;

            const chunk = decoder.decode(value);
            const lines = chunk.split('\n').filter(Boolean);

            for (const line of lines) {
                try {
                    const data = JSON.parse(line);
                    onProgress?.(data);
                } catch (e) {}
            }
        }
    }
```

```javascript
class ChatUI {
    constructor(options) {
        this.container = options.container;
        this.onMessage = options.onMessage;
        this.onClear = options.onClear;

        `;

        this.inputArea = this.container.querySelector('.chat-input');

        // 绑定事件
        this.bindEvents();
    }

    bindEvents() {
        const sendButton = this.container.querySelector('.send-button');

        // 点击发送
        sendButton.addEventListener('click', () => this.sendMessage());

        // Ctrl+Enter 发送
        this.inputArea.addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'Enter') {
                this.sendMessage();
            }
        });

        // 自动调整高度
        this.inputArea.addEventListener('input', () => {
            this.inputArea.style.height = 'auto';
            this.inputArea.style.height = Math.min(this.inputArea.scrollHeight, 200) + 'px';
        });

        const content = this.inputArea.value.trim();
        if (!content) return;

        this.inputArea.value = '';
        this.onMessage(content);
    }

    addMessage(role, content) {
        const messageElement = document.createElement('div');
        messageElement.className = `message message-${role}`;
        messageElement.innerHTML = this.renderContent(content);

        this.messagesContainer.appendChild(messageElement);
        this.scrollToBottom();
    }

    appendToLastMessage(content) {
        const lastMessage = this.messagesContainer.lastElementChild;
        if (lastMessage) {
            lastMessage.innerHTML += this.renderContent(content);
            this.scrollToBottom();
        }
    }

    renderContent(content) {
        // 简单Markdown渲染
        return content
            .replace(/```([\s\S]*?)```/g, '<pre><code>$1</code></pre>')
            .replace(/`([^`]+)`/g, '<code>$1</code>')
            .replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')
            .replace(/\*([^*]+)\*/g, '<em>$1</em>')
            .replace(/\n/g, '<br>');
    }

    showLoading() {
        const loadingElement = document.createElement('div');
        loadingElement.className = 'message message-assistant loading';
        loadingElement.innerHTML = '<span class="loading-dots">...</span>';

        this.messagesContainer.appendChild(loadingElement);
        this.scrollToBottom();
    }

    hideLoading() {
        const loadingElement = this.messagesContainer.querySelector('.loading');
        if (loadingElement) {
            loadingElement.remove();
        }
    }

    showError(message) {
        const errorElement = document.createElement('div');
        errorElement.className = 'message message-error';
        errorElement.textContent = `错误: ${message}`;

        this.messagesContainer.appendChild(errorElement);
    }

    clearMessages() {
        this.messagesContainer.innerHTML = '';
    }

    scrollToBottom() {
        this.messagesContainer.scrollTop = this.messagesContainer.scrollHeight;
    }
}
```

```javascript
            suffix,
            language: document.languageId,
```

```typescript
  key: keyof T;
  title: string;
  width?: number;
  sortable?: boolean;
  filterable?: boolean;
  editable?: boolean;
  render?: (value: T[keyof T], record: T, index: number) => ReactNode;
  editRender?: (value: T[keyof T], record: T, onSave: (val: T[keyof T]) => void) => ReactNode;
}

export interface TableProps<T> {
  data: T[];
  columns: Column<T>[];
  selectable?: boolean;
  pagination?: PaginationConfig;
  onSave?: (record: T, field: keyof T, value: T[keyof T]) => void;
  onExport?: (data: T[]) => void;

// useTableState.ts - Agent 生成的状态管理 Hook
function useTableState<T>(initialData: T[]) {
  const [data, setData] = useState<T[]>(initialData);
  const [sortConfig, setSortConfig] = useState<SortConfig<T> | null>(null);
  const [filterConfig, setFilterConfig] = useState<FilterConfig<T>>({});
  const [pagination, setPagination] = useState<PaginationState>({
    current: 1,
    pageSize: 20,
  });
  const [selectedRowKeys, setSelectedRowKeys] = useState<Set<string>>([]);

  // 排序逻辑
  const sortedData = useMemo(() => {
    if (!sortConfig) return data;
    return [...data].sort((a, b) => {
      const aVal = a[sortConfig.key];
      const bVal = b[sortConfig.key];
      const modifier = sortConfig.direction === 'asc' ? 1 : -1;
      if (aVal < bVal) return -1 * modifier;
      if (aVal > bVal) return 1 * modifier;
      return 0;
    });
  }, [data, sortConfig]);

  // 筛选逻辑
  const filteredData = useMemo(() => {
    return sortedData.filter(row => {
      return Object.entries(filterConfig).every(([key, filter]) => {
        if (!filter) return true;
        const value = row[key as keyof T];
        return String(value).toLowerCase().includes(filter.toLowerCase());
      });
    });
  }, [sortedData, filterConfig]);

  // 分页逻辑
  const paginatedData = useMemo(() => {
    const start = (pagination.current - 1) * pagination.pageSize;
    return filteredData.slice(start, start + pagination.pageSize);
  }, [filteredData, pagination]);

  return {
    data: paginatedData,
    total: filteredData.length,
    sortConfig,
    filterConfig,
    pagination,
    selectedRowKeys,
    setSortConfig,
    setFilterConfig,
    setPagination,
    setSelectedRowKeys,
  };
}
```

```typescript
// useUserStore.ts
import { create } from 'zustand';
import { devtools } from 'zustand/middleware';

interface UserState {
  users: User[];
  loading: boolean;
  error: string | null;
  currentUser: User | null;
  fetchUsers: () => Promise<void>;
  fetchUserById: (id: string) => Promise<void>;
  updateUser: (id: string, data: Partial<User>) => Promise<void>;
  deleteUser: (id: string) => Promise<void>;
}

export const useUserStore = create<UserState>()(
  devtools(
    (set, get) => ({
      users: [],
      loading: false,
      error: null,
      currentUser: null,

      fetchUsers: async () => {
        set({ loading: true, error: null });
        try {
          const users = await userApi.getUsers();
          set({ users, loading: false });
        } catch (error) {
          set({ error: error.message, loading: false });
        }
      },

      fetchUserById: async (id: string) => {
        set({ loading: true, error: null });
        try {
          const currentUser = await userApi.getUserById(id);
          set({ currentUser, loading: false });
        } catch (error) {
          set({ error: error.message, loading: false });
        }
      },

      updateUser: async (id: string, data: Partial<User>) => {
        set({ loading: true, error: null });
        try {
          const updatedUser = await userApi.updateUser(id, data);
          set((state) => ({
            users: state.users.map((u) => (u.id === id ? updatedUser : u)),
            currentUser: state.currentUser?.id === id ? updatedUser : state.currentUser,
            loading: false,
          }));
        } catch (error) {
          set({ error: error.message, loading: false });
        }
      },

      deleteUser: async (id: string) => {
        set({ loading: true, error: null });
        try {
          await userApi.deleteUser(id);
          set((state) => ({
            users: state.users.filter((u) => u.id !== id),
            loading: false,
          }));
        } catch (error) {
          set({ error: error.message, loading: false });
        }
      },
    }),
    { name: 'user-store' }
  )
);
```

### 7. 总结：合理利用 AI 的核心原则

| 原则 | 说明 |
|------|------|
| **AI 是助手不是替代者** | 核心逻辑必须人工掌控 |
| **质量第一** | AI 代码必须经过审查 |
| **安全优先** | 敏感信息不发送给 AI |
| **持续学习** | AI 工具快速迭代，保持学习 |
| **人机协作** | 发挥各自优势，互补共赢 |
| **效果导向** | 用 AI 解决实际问题，不要为了用而用 |
