## 二、音视频技术基础

### 2.1 WebRTC原理详解

#### 2.1.1 WebRTC核心概念

WebRTC (Web Real-Time Communication) 是一项实时通信技术，它使浏览器和移动应用能够通过简单的API进行点对点(P2P)通信。WebRTC的核心目标是实现无需第三方插件的实时音视频通话和数据传输。

**WebRTC的三大核心API：**

1. **MediaStream (getUserMedia)**
   - 用于获取本地音视频流
   - 访问摄像头和麦克风
   - 支持屏幕共享

2. **RTCPeerConnection**
   - 建立和管理P2P连接
   - 处理音视频传输
   - 负责ICE候选交换

3. **RTCDataChannel**
   - 点对点数据传输
   - 支持可靠和不可靠模式
   - 低延迟数据传输

**WebRTC协议栈：**

```
┌─────────────────────────────────────────┐
│              应用层                       │
│  (JavaScript API: getUserMedia, etc.)    │
├─────────────────────────────────────────┤
│           Session Layer                  │
│    (SIP, WebSocket, HTTP Signaling)      │
├─────────────────────────────────────────┤
│           Transport Layer                │
│      (ICE, STUN, TURN, RTP/SRTP)         │
├─────────────────────────────────────────┤
│            Media Layer                   │
│      (VP8, VP9, H.264, Opus)            │
└─────────────────────────────────────────┘
```

#### 2.1.2 WebRTC连接建立过程

WebRTC连接建立是一个复杂的过程，涉及多个步骤：

**1. 信令服务器 (Signaling Server)**

信令服务器负责协调连接的建立过程，交换元数据。常用的信令协议包括WebSocket和HTTP。

```javascript
// 信令服务器示例 (Node.js + WebSocket)
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

// 房间管理
const rooms = new Map();

wss.on('connection', (ws) => {
    let currentRoom = null;
    let currentUser = null;

    ws.on('message', (message) => {
        const data = JSON.parse(message);

        switch (data.type) {
            case 'join':
                // 用户加入房间
                currentRoom = data.roomId;
                currentUser = data.userId;

                if (!rooms.has(currentRoom)) {
                    rooms.set(currentRoom, new Set());
                }
                rooms.get(currentRoom).add({ ws, userId: currentUser });

                // 通知房间内其他用户
                broadcastToRoom(currentRoom, {
                    type: 'user-joined',
                    userId: currentUser
                }, currentUser);
                break;

            case 'offer':
                // 转发offer给目标用户
                sendToUser(data.targetUserId, {
                    type: 'offer',
                    offer: data.offer,
                    from: currentUser
                });
                break;

            case 'answer':
                // 转发answer给发起者
                sendToUser(data.targetUserId, {
                    type: 'answer',
                    answer: data.answer,
                    from: currentUser
                });
                break;

            case 'ice-candidate':
                // 转发ICE候选
                sendToUser(data.targetUserId, {
                    type: 'ice-candidate',
                    candidate: data.candidate,
                    from: currentUser
                });
                break;

            case 'leave':
                // 用户离开
                handleUserLeave(currentRoom, currentUser);
                break;
        }
    });

    ws.on('close', () => {
        handleUserLeave(currentRoom, currentUser);
    });
});

function broadcastToRoom(roomId, message, excludeUser) {
    if (!rooms.has(roomId)) return;

    rooms.get(roomId).forEach(({ ws, userId }) => {
        if (userId !== excludeUser && ws.readyState === WebSocket.OPEN) {
            ws.send(JSON.stringify(message));
        }
    });
}

function sendToUser(userId, message) {
    rooms.forEach((users) => {
        users.forEach(({ ws, userId: id }) => {
            if (id === userId && ws.readyState === WebSocket.OPEN) {
                ws.send(JSON.stringify(message));
            }
        });
    });
}

function handleUserLeave(roomId, userId) {
    if (!roomId || !rooms.has(roomId)) return;

    const room = rooms.get(roomId);
    room.forEach(({ ws, userId: id }) => {
        if (id === userId) {
            room.delete({ ws, userId: id });
        }
    });

    // 通知房间内其他用户
    broadcastToRoom(roomId, {
        type: 'user-left',
        userId: userId
    }, userId);

    if (room.size === 0) {
        rooms.delete(roomId);
    }
}
```

**3. 连接建立完整流程**

```javascript
// 完整的WebRTC连接建立流程

class WebRTCClient {
    constructor(options = {}) {
        this.options = options;
        this.pc = null;
        this.localStream = null;
        this.remoteStream = null;
        this.signalCallback = options.onSignal || (() => {});

        this.initPeerConnection();
    }

    // 初始化PeerConnection
    initPeerConnection() {
        const config = {
            iceServers: this.options.iceServers || [
                { urls: 'stun:stun.l.google.com:19302' }
            ]
        };

        this.pc = new RTCPeerConnection(config);

        // 监听远程流
        this.pc.ontrack = (event) => {
            console.log('收到远程轨道:', event.streams[0]);
            this.remoteStream = event.streams[0];
            if (this.options.onRemoteStream) {
                this.options.onRemoteStream(event.streams[0]);
            }
        };

        // 监听ICE候选
        this.pc.onicecandidate = (event) => {
            if (event.candidate) {
                this.signalCallback({
                    type: 'ice-candidate',
                    candidate: event.candidate
                });
            }
        };

        // 监听连接状态
        this.pc.oniceconnectionstatechange = () => {
            console.log('ICE状态:', this.pc.iceConnectionState);
            if (this.options.onStateChange) {
                this.options.onStateChange(this.pc.iceConnectionState);
            }
        };

        // 监听ICE候选收集完成
        this.pc.onicegatheringstatechange = () => {
            console.log('ICE收集状态:', this.pc.iceGatheringState);
        };
    }

    // 获取本地媒体流
    async getLocalStream(constraints = { video: true, audio: true }) {
        try {
            this.localStream = await navigator.mediaDevices.getUserMedia(constraints);

            // 将本地流添加到PeerConnection
            this.localStream.getTracks().forEach(track => {
                this.pc.addTrack(track, this.localStream);
            });

            return this.localStream;
        } catch (error) {
            console.error('获取媒体设备失败:', error);
            throw error;
        }
    }

    // 创建offer (发起方)
    async createOffer() {
        const offer = await this.pc.createOffer({
            offerToReceiveAudio: true,
            offerToReceiveVideo: true
        });

        await this.pc.setLocalDescription(offer);

        return offer;
    }

    // 创建answer (接收方)
    async createAnswer(offer) {
        await this.pc.setRemoteDescription(new RTCSessionDescription(offer));

        const answer = await this.pc.createAnswer();
        await this.pc.setLocalDescription(answer);

        return answer;
    }

    // 处理answer (发起方)
    async handleAnswer(answer) {
        await this.pc.setRemoteDescription(new RTCSessionDescription(answer));
    }

    // 处理ICE候选
    async handleIceCandidate(candidate) {
        try {
            await this.pc.addIceCandidate(new RTCIceCandidate(candidate));
        } catch (error) {
            console.error('添加ICE候选失败:', error);
        }
    }

    // 关闭连接
    close() {
        if (this.localStream) {
            this.localStream.getTracks().forEach(track => track.stop());
        }
        if (this.pc) {
            this.pc.close();
        }
    }
}

// 使用示例 - 发起方
async function startCall() {
    const caller = new WebRTCClient({
        onSignal: (signal) => {
            // 发送信号给接收方
            sendToPeer(signal);
        },
        onRemoteStream: (stream) => {
            document.getElementById('remoteVideo').srcObject = stream;
        }
    });

    // 获取本地流并显示
    const localStream = await caller.getLocalStream();
    document.getElementById('localVideo').srcObject = localStream;

    // 创建offer
    const offer = await caller.createOffer();
    // 发送offer给接收方
    sendToPeer({ type: 'offer', sdp: offer });
}

// 接收信号处理
function handleSignal(signal) {
    switch (signal.type) {
        case 'offer':
            // 接收offer并创建answer
            handleOffer(signal.sdp);
            break;
        case 'answer':
            caller.handleAnswer(signal.sdp);
            break;
        case 'ice-candidate':
            caller.handleIceCandidate(signal.candidate);
            break;
    }
}
```

```javascript
// 媒体录制
class MediaRecorder {
    constructor(stream, options = {}) {
        this.stream = stream;
        this.recorder = null;
        this.chunks = [];

        this.options = {
            mimeType: options.mimeType || 'video/webm;codecs=vp9',
            audioBitsPerSecond: options.audioBitsPerSecond || 128000,
            videoBitsPerSecond: options.videoBitsPerSecond || 2500000
        };
    }

    start() {
        // 检查支持的MIME类型
        let mimeType = this.options.mimeType;
        if (!MediaRecorder.isTypeSupported(mimeType)) {
            // 尝试备选类型
            const alternatives = [
                'video/webm;codecs=vp8',
                'video/webm;codecs=h264',
                'video/webm'
            ];
            for (const alt of alternatives) {
                if (MediaRecorder.isTypeSupported(alt)) {
                    mimeType = alt;
                    break;
                }
            }
        }

        this.recorder = new MediaRecorder(this.stream, {
            mimeType,
            audioBitsPerSecond: this.options.audioBitsPerSecond,
            videoBitsPerSecond: this.options.videoBitsPerSecond
        });

        this.chunks = [];

        this.recorder.ondataavailable = (event) => {
            if (event.data && event.data.size > 0) {
                this.chunks.push(event.data);
            }
        };

        this.recorder.start(1000); // 每秒收集一次数据
    }

    stop() {
        return new Promise((resolve) => {
            this.recorder.onstop = () => {
                const blob = new Blob(this.chunks, { type: this.options.mimeType });
                resolve(blob);
            };
            this.recorder.stop();
        });
    }

    pause() {
        this.recorder.pause();
    }

    resume() {
        this.recorder.resume();
```

```javascript
// 完整的WebSocket信令服务器
const WebSocket = require('ws');
const { v4: uuidv4 } = require('uuid');

class SignalingServer {
    constructor(port = 8080) {
        this.wss = new WebSocket.Server({ port });
        this.rooms = new Map(); // roomId -> Set(ws)
        this.userRooms = new Map(); // ws -> roomId
        this.userIds = new Map(); // ws -> userId

        console.log(`信令服务器启动在端口 ${port}`);
        this.setupHandlers();
    }

    setupHandlers() {
        this.wss.on('connection', (ws, req) => {
            const userId = uuidv4();
            console.log(`用户 ${userId} 连接`);

            ws.isAlive = true;
            ws.userId = userId;

            ws.on('pong', () => {
                ws.isAlive = true;
            });

            ws.on('message', (message) => {
                try {
                    const data = JSON.parse(message);
                    this.handleMessage(ws, data);
                } catch (e) {
                    console.error('消息解析失败:', e);
                }
            });

            ws.on('close', () => {
                this.handleDisconnect(ws);
            });

            ws.on('error', (error) => {
                console.error(`用户 ${userId} 连接错误:`, error);
            });
        });

        // 心跳检测
        setInterval(() => {
            this.wss.clients.forEach((ws) => {
                if (!ws.isAlive) {
                    return ws.terminate();
                }
                ws.isAlive = false;
                ws.ping();
            });
        }, 30000);
    }

    handleMessage(ws, data) {
        console.log('收到消息:', data.type);

        switch (data.type) {
            case 'join':
                this.handleJoin(ws, data.roomId);
                break;

            case 'leave':
                this.handleLeave(ws);
                break;

            case 'offer':
                this.handleOffer(ws, data);
                break;

            case 'answer':
                this.handleAnswer(ws, data);
                break;

            case 'ice-candidate':
                this.handleIceCandidate(ws, data);
                break;

            case 'room-users':
                this.handleRoomUsers(ws);
                break;

            case 'video-quality':
                this.handleVideoQuality(ws, data);
                break;

            case 'mute':
                this.handleMute(ws, data);
                break;

            default:
                console.log('未知消息类型:', data.type);
        }
    }

    handleJoin(ws, roomId) {
        // 离开当前房间（如果有）
        if (this.userRooms.has(ws)) {
            this.handleLeave(ws);
        }

        // 加入新房间
        if (!this.rooms.has(roomId)) {
            this.rooms.set(roomId, new Set());
        }

        this.rooms.get(roomId).add(ws);
        this.userRooms.set(ws, roomId);
        this.userIds.set(ws, data.userId);

        console.log(`用户 ${ws.userId} 加入房间 ${roomId}`);

        // 通知用户加入成功
        ws.send(JSON.stringify({
            type: 'joined',
            roomId,
            userId: ws.userId
        }));

        // 通知房间内其他用户
        this.broadcast(roomId, {
            type: 'user-joined',
            userId: ws.userId
        }, ws);

        // 发送房间内用户列表给新用户
        this.sendRoomUsers(ws, roomId);
    }

    handleLeave(ws) {
        const roomId = this.userRooms.get(ws);
        if (!roomId) return;

        const room = this.rooms.get(roomId);
        if (room) {
            room.delete(ws);

            if (room.size === 0) {
                this.rooms.delete(roomId);
            } else {
                // 通知房间内其他用户
                this.broadcast(roomId, {
                    type: 'user-left',
                    userId: ws.userId
                }, ws);
            }
        }

        this.userRooms.delete(ws);
        this.userIds.delete(ws);

        ws.send(JSON.stringify({
            type: 'left',
            roomId
        }));
    }

    handleOffer(ws, data) {
        const { targetUserId, sdp, roomId } = data;

        const targetWs = this.findUserById(targetUserId, roomId);
        if (targetWs) {
            targetWs.send(JSON.stringify({
                type: 'offer',
                sdp,
                from: ws.userId
            }));
        }
    }

    handleAnswer(ws, data) {
        const { targetUserId, sdp, roomId } = data;

        const targetWs = this.findUserById(targetUserId, roomId);
        if (targetWs) {
            targetWs.send(JSON.stringify({
                type: 'answer',
                sdp,
                from: ws.userId
            }));
        }
    }

    handleIceCandidate(ws, data) {
        const { targetUserId, candidate, roomId } = data;

        const targetWs = this.findUserById(targetUserId, roomId);
        if (targetWs) {
            targetWs.send(JSON.stringify({
                type: 'ice-candidate',
                candidate,
                from: ws.userId
            }));
        }
    }

    handleRoomUsers(ws) {
        const roomId = this.userRooms.get(ws);
        if (roomId) {
            this.sendRoomUsers(ws, roomId);
        }
    }

    handleVideoQuality(ws, data) {
        // 转发视频质量请求给其他用户
        const roomId = this.userRooms.get(ws);
        if (roomId) {
            this.broadcast(roomId, {
                type: 'video-quality',
                userId: ws.userId,
                quality: data.quality
            }, ws);
        }
    }

    handleMute(ws, data) {
        const roomId = this.userRooms.get(ws);
        if (roomId) {
            this.broadcast(roomId, {
                type: 'mute',
                userId: ws.userId,
                kind: data.kind,
                muted: data.muted
            }, ws);
        }
    }

    handleDisconnect(ws) {
        this.handleLeave(ws);
        console.log(`用户 ${ws.userId} 断开连接`);
    }

    broadcast(roomId, message, excludeWs = null) {
        const room = this.rooms.get(roomId);
        if (!room) return;

        const messageStr = JSON.stringify(message);
        room.forEach((client) => {
```

```javascript
class QualityMonitor {
    constructor(pc) {
        this.pc = pc;
        this.statsInterval = null;
        this.callbacks = [];

        this.statsInterval = setInterval(() => {

            clearInterval(this.statsInterval);

    async collectStats() {
        const stats = await this.pc.getStats();
        const report = {
            timestamp: Date.now(),
            connection: {},
            local: {},
            remote: {},
            network: {}
        };

        stats.forEach((stat) => {
            switch (stat.type) {
                case 'candidate-pair':
                    report.connection = {
                        state: stat.state,
                        currentRoundTripTime: stat.currentRoundTripTime,
                        availableOutgoingBitrate: stat.availableOutgoingBitrate,
                        availableIncomingBitrate: stat.availableIncomingBitrate,
                        packetsReceived: stat.packetsReceived,
                        packetsSent: stat.packetsSent,
                        bytesReceived: stat.bytesReceived,
                        bytesSent: stat.bytesSent
                    };
                    break;

                case 'inbound-rtp':
                    if (stat.kind === 'video') {
                        report.remote.video = {
                            packetsLost: stat.packetsLost,
                            jitter: stat.jitter,
                            fractionLost: stat.fractionLost,
                            framesDecoded: stat.framesDecoded,
                            bytesReceived: stat.bytesReceived,
                            lastPacketReceivedTimestamp: stat.lastPacketReceivedTimestamp
                        };
                    } else if (stat.kind === 'audio') {
                        report.remote.audio = {
                            packetsLost: stat.packetsLost,
                            jitter: stat.jitter,
                            fractionLost: stat.fractionLost,
                            bytesReceived: stat.bytesReceived
                        };
                    }
                    break;

                case 'outbound-rtp':
                    if (stat.kind === 'video') {
                        report.local.video = {
                            packetsSent: stat.packetsSent,
                            bytesSent: stat.bytesSent,
                            framesEncoded: stat.framesEncoded,
                            keyFramesEncoded: stat.keyFramesEncoded
                        };
                    } else if (stat.kind === 'audio') {
                        report.local.audio = {
                            packetsSent: stat.packetsSent,
                            bytesSent: stat.bytesSent
                        };
                    }
                    break;

                case 'track':
                    if (stat.trackIdentifier === 'audio') {
                        report.network.audio = {
                            audioLevel: stat.audioLevel,
                            totalAudioEnergy: stat.totalAudioEnergy,
                            echoReturnLoss: stat.echoReturnLoss
                        };
                    }
                    break;

                case 'remote-inbound-rtp':
                    if (stat.kind === 'video') {
                        report.network.video = {
                            packetsLost: stat.packetsLost,
                            jitter: stat.jitter,
                            roundTripTime: stat.roundTripTime
                        };
                    }
                    break;
            }
        });

        this.callbacks.forEach(cb => cb(report));
        return report;
    }

    onReport(callback) {
        this.callbacks.push(callback);
    }

    // 计算平均统计
    getAverageStats(duration = 5000) {
        return new Promise((resolve) => {
            const samples = [];

            const collectSample = () => {
                this.collectStats().then(report => {
                    samples.push(report);
                });
            };

            collectSample();
            const interval = setInterval(collectSample, duration);

            setTimeout(() => {
                clearInterval(interval);

                if (samples.length === 0) {
                    resolve(null);
                    return;
                }

                // 计算平均值
                const avg = {
                    timestamp: Date.now(),
                    connection: {},
                    remote: { video: {}, audio: {} },
                    local: { video: {}, audio: {} },
                    network: { video: {}, audio: {} }
                };

                const sum = (arr, key) => arr.reduce((acc, r) => acc + (r[key] || 0), 0);
                const avgValue = (arr, key) => sum(arr, key) / arr.length;

                const lastSample = samples[samples.length - 1];
                avg.connection = lastSample.connection;

                avg.remote.video = {
                    packetsLost: avgValue(samples, 'remote.video.packetsLost'),
                    fractionLost: avgValue(samples, 'remote.video.fractionLost')
                };

                avg.network.video = {
                    roundTripTime: avgValue(samples, 'network.video.roundTripTime')
                };

                resolve(avg);
            }, duration);
        });
    }

class AdaptiveBitrateController {
    constructor(pc, options = {}) {
        this.pc = pc;
        this.qualityMonitor = new QualityMonitor(pc);
        this.minBitrate = options.minBitrate || 500000;
        this.maxBitrate = options.maxBitrate || 5000000;
        this.targetBitrate = options.targetBitrate || 2000000;
        this.adjustmentInterval = options.adjustmentInterval || 3000;

        this.currentQuality = 'medium';
        this.isAdjusting = false;
    }

    start() {
        this.qualityMonitor.start(1000);

        this.qualityMonitor.onReport((report) => {
            this.adjustBitrate(report);
        });
    }

    adjustBitrate(report) {
        if (this.isAdjusting) return;

        const rtt = report.connection.currentRoundTripTime || 0;
        const packetsLost = report.remote.video?.packetsLost || 0;
        const fractionLost = report.remote.video?.fractionLost || 0;

        // 基于丢包率调整
        if (fractionLost > 0.1) {
            // 丢包率过高，降低码率
            this.decreaseBitrate(0.7);
        } else if (fractionLost > 0.05) {
            this.decreaseBitrate(0.85);
        } else if (fractionLost < 0.02 && rtt < 0.1) {
            // 网络条件良好，提高码率
            this.increaseBitrate(1.2);
        }
    }

    async decreaseBitrate(factor) {
        this.isAdjusting = true;
        this.targetBitrate = Math.max(this.minBitrate, this.targetBitrate * factor);

        await this.applyBitrate();
        console.log(`降低码率到: ${this.targetBitrate / 1000}kbps`);

        setTimeout(() => {
            this.isAdjusting = false;
        }, this.adjustmentInterval);
    }

    async increaseBitrate(factor) {
        this.isAdjusting = true;
        this.targetBitrate = Math.min(this.maxBitrate, this.targetBitrate * factor);

        await this.applyBitrate();
        console.log(`提高码率到: ${this.targetBitrate / 1000}kbps`);

        setTimeout(() => {
            this.isAdjusting = false;
        }, this.adjustmentInterval);
    }

    async applyBitrate() {
        const senders = this.pc.getSenders();

        for (const sender of senders) {
            if (sender.track?.kind === 'video') {
                const params = sender.getParameters();
                if (!params.encodings) {
                    params.encodings = [{}];
                }

                params.encodings[0].maxBitrate = this.targetBitrate;
                await sender.setParameters(params);
            }
        }
    }

    stop() {
        this.qualityMonitor.stop();
    }
```

```dart
class ApiService {
  final String baseUrl;

  ApiService({this.baseUrl = 'https://api.example.com'});

  Future<T> get<T>(
    String endpoint, {
    Map<String, String>? queryParams,
    T Function(dynamic)? parser,
  }) async {
    try {
      final uri = Uri.parse('$baseUrl$endpoint').replace(
        queryParameters: queryParams,
      );

      final response = await http.get(uri).timeout(
        const Duration(seconds: 30),
      );

      if (response.statusCode == 200) {
        final data = json.decode(response.body);
        return parser != null ? parser(data) : data as T;
      } else {
        throw ApiException(
          statusCode: response.statusCode,
          message: 'Request failed',
        );
      }
    } catch (e) {
      if (e is ApiException) rethrow;
      throw ApiException(message: e.toString());
    }
  }

  Future<T> post<T>(
    String endpoint, {
    Map<String, dynamic>? body,
    T Function(dynamic)? parser,
  }) async {
    try {
      final uri = Uri.parse('$baseUrl$endpoint');

      final response = await http.post(
        uri,
        headers: {'Content-Type': 'application/json'},
        body: json.encode(body),
      ).timeout(const Duration(seconds: 30));

      if (response.statusCode == 200 || response.statusCode == 201) {
        final data = json.decode(response.body);
        return parser != null ? parser(data) : data as T;
      } else {
        throw ApiException(
          statusCode: response.statusCode,
          message: 'Request failed',
        );
      }
    } catch (e) {
      if (e is ApiException) rethrow;
      throw ApiException(message: e.toString());
    }
  }
}
```

```javascript
// Service Worker - sw.js

const CACHE_NAME = 'my-pwa-cache-v1';
const urlsToCache = [
  '/',
  '/index.html',
  '/styles/main.css',
  '/scripts/main.js',
  '/images/logo.png',
  '/offline.html',
];

// 安装事件
self.addEventListener('install', (event) => {
  console.log('Service Worker installing...');

  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      console.log('Opened cache');
      return cache.addAll(urlsToCache);
    })
  );

  // 立即激活
  self.skipWaiting();
});

// 激活事件
self.addEventListener('activate', (event) => {
  console.log('Service Worker activating...');

  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames.map((cacheName) => {
          // 删除旧缓存
          if (cacheName !== CACHE_NAME) {
            console.log('Deleting old cache:', cacheName);
            return caches.delete(cacheName);
          }
        })
      );
    })
  );

  // 立即控制所有页面
  event.waitUntil(self.clients.claim());
});

// 请求拦截
self.addEventListener('fetch', (event) => {
  const { request } = event;
  const url = new URL(request.url);

  // 只处理同源请求
  if (url.origin !== location.origin) {
    return;
  }

  // 网络优先策略 - API请求
  if (request.url.includes('/api/')) {
    event.respondWith(networkFirst(request));
    return;
  }

  // 缓存优先策略 - 静态资源
  if (
    request.destination === 'style' ||
    request.destination === 'script' ||
    request.destination === 'image'
  ) {
    event.respondWith(cacheFirst(request));
    return;
  }

  // 后备策略 - 其他请求
  event.respondWith(staleWhileRevalidate(request));
});

// 缓存优先策略
async function cacheFirst(request) {
  const cachedResponse = await caches.match(request);

  if (cachedResponse) {
    return cachedResponse;
  }

  try {
    const networkResponse = await fetch(request);

    if (networkResponse.ok) {
      const cache = await caches.open(CACHE_NAME);
      cache.put(request, networkResponse.clone());
    }

    return networkResponse;
  } catch (error) {
    return new Response('Offline', { status: 503 });
  }
}

// 网络优先策略
async function networkFirst(request) {
  try {
    const networkResponse = await fetch(request);

    if (networkResponse.ok) {
      const cache = await caches.open(CACHE_NAME);
      cache.put(request, networkResponse.clone());
    }

    return networkResponse;
  } catch (error) {
    const cachedResponse = await caches.match(request);

    if (cachedResponse) {
      return cachedResponse;
    }

    return new Response(JSON.stringify({ error: 'Offline' }), {
      headers: { 'Content-Type': 'application/json' },
    });
  }
}

// Stale-While-Revalidate策略
async function staleWhileRevalidate(request) {
  const cachedResponse = await caches.match(request);

  const fetchPromise = fetch(request)
    .then((networkResponse) => {
      if (networkResponse.ok) {
        const cache = caches.open(CACHE_NAME);
        cache.then((c) => c.put(request, networkResponse.clone()));
      }
      return networkResponse;
    })
    .catch(() => cachedResponse);

  return cachedResponse || fetchPromise;
}

// 后台同步
self.addEventListener('sync', (event) => {
  console.log('Background sync:', event.tag);

  if (event.tag === 'sync-data') {
    event.waitUntil(syncData());
  }
});

async function syncData() {
  // 同步数据到服务器
  const cache = await caches.open('api-cache');
  const requests = await cache.keys();

  for (const request of requests) {
    if (request.url.includes('/api/')) {
      const cachedResponse = await cache.match(request);
      const data = await cachedResponse.json();

      try {
        await fetch(request.url, {
          method: 'POST',
          body: JSON.stringify(data),
          headers: { 'Content-Type': 'application/json' },
        });

        // 同步成功后删除
        await cache.delete(request);
      } catch (error) {
        console.error('Sync failed:', error);
      }
    }
  }
}

// 推送通知
self.addEventListener('push', (event) => {
  const data = event.data ? event.data.json() : {};

  const options = {
    body: data.body || '您有新的消息',
    icon: '/images/notification-icon.png',
    badge: '/images/badge.png',
    vibrate: [100, 50, 100],

      url: data.url || '/',
    },
    actions: [
      { action: 'view', title: '查看' },
      { action: 'close', title: '关闭' },
    ],
  };
```

```javascript
// 完整的WebRTC音视频通话应用

// 1. 客户端实现
class VideoCallApp {
  constructor() {
    this.localStream = null;
    this.remoteStream = null;
    this.peerConnection = null;
    this.roomId = null;
    this.userId = this.generateUserId();
    this.signalServer = null;
    this.qualityMonitor = null;

    this.config = {
      iceServers: [
        { urls: 'stun:stun.l.google.com:19302' },
        { urls: 'stun:stun1.l.google.com:19302' },
      ],
    };
  }

  generateUserId() {
    return 'user_' + Math.random().toString(36).substr(2, 9);
  }

  // 初始化
  async init() {
    // 获取本地媒体流
    try {
      this.localStream = await navigator.mediaDevices.getUserMedia({
        video: {
          width: { ideal: 1280 },
          height: { ideal: 720 },
          frameRate: { ideal: 30 },
        },
        audio: {
          echoCancellation: true,
          noiseSuppression: true,
          autoGainControl: true,
        },
      });

      // 显示本地视频
      this.showLocalVideo(this.localStream);
      return true;
    } catch (error) {
      console.error('获取媒体设备失败:', error);
      return false;
    }
  }

  // 连接房间
  async joinRoom(roomId) {
    this.roomId = roomId;

    // 连接信令服务器
    await this.connectToSignalingServer();

    // 发送加入房间请求
    this.sendSignal({
      type: 'join',
      roomId: roomId,
      userId: this.userId,
    });
  }

  // 创建PeerConnection
  createPeerConnection(peerId) {
    const pc = new RTCPeerConnection(this.config);

    // 添加本地流
    this.localStream.getTracks().forEach((track) => {
      pc.addTrack(track, this.localStream);
    });

    // 监听远程流
    pc.ontrack = (event) => {
      this.remoteStream = event.streams[0];
      this.showRemoteVideo(event.streams[0]);
    };

    // 监听ICE候选
    pc.onicecandidate = (event) => {
      if (event.candidate) {
        this.sendSignal({
          type: 'ice-candidate',
          candidate: event.candidate,
          targetUserId: peerId,
          roomId: this.roomId,
        });
      }
    };

    // 监听连接状态
    pc.oniceconnectionstatechange = () => {
      console.log('ICE连接状态:', pc.iceConnectionState);
      this.handleConnectionStateChange(pc.iceConnectionState);
    };

    // 监听ICE收集状态
    pc.onicegatheringstatechange = () => {
      console.log('ICE收集状态:', pc.iceGatheringState);
    };

    this.peerConnection = pc;

    // 启动质量监控
    this.startQualityMonitor();

    return pc;
  }

  // 创建offer
  async createOffer(peerId) {
    const pc = this.createPeerConnection(peerId);

    const offer = await pc.createOffer({
      offerToReceiveAudio: true,
      offerToReceiveVideo: true,
    });

    await pc.setLocalDescription(offer);

    this.sendSignal({
      type: 'offer',
      sdp: offer,
      targetUserId: peerId,
      roomId: this.roomId,
    });
  }

    await pc.setRemoteDescription(new RTCSessionDescription(offer));

    const answer = await pc.createAnswer();
    await pc.setLocalDescription(answer);

    this.sendSignal({
      type: 'answer',
      sdp: answer,
      targetUserId: peerId,
      roomId: this.roomId,
    });
  }

  // 处理answer
  async handleAnswer(peerId, answer) {
    if (!this.peerConnection) return;

    await this.peerConnection.setRemoteDescription(
      new RTCSessionDescription(answer)
    );
  }

  // 处理ICE候选
  async handleIceCandidate(peerId, candidate) {
    if (!this.peerConnection) return;

    try {
      await this.peerConnection.addIceCandidate(new RTCIceCandidate(candidate));
    } catch (error) {
      console.error('添加ICE候选失败:', error);
    }
  }

  // 切换音频
  toggleAudio() {
    const audioTrack = this.localStream.getAudioTracks()[0];
    if (audioTrack) {
      audioTrack.enabled = !audioTrack.enabled;
      return audioTrack.enabled;
    }
    return false;
  }

    if (videoTrack) {
      videoTrack.enabled = !videoTrack.enabled;
      return videoTrack.enabled;
    }
    return false;

    const videoTrack = this.localStream.getVideoTracks()[0];

    const sender = this.peerConnection
      .getSenders()
      .find((s) => s.track?.kind === 'video');

    if (sender && videoTrack) {
      await sender.replaceTrack(videoTrack);
    }

    this.currentScreenTrack.stop();
    this.currentScreenTrack = null;
  }

    const videoElement = document.getElementById('localVideo');
    if (videoElement) {
      videoElement.srcObject = stream;
    }
  }

  // 显示远程视频
  showRemoteVideo(stream) {
    const videoElement = document.getElementById('remoteVideo');
    if (videoElement) {
      videoElement.srcObject = stream;
    }
  }

  // 连接信令服务器
  async connectToSignalingServer() {
    return new Promise((resolve, reject) => {
      this.signalServer = new WebSocket('wss://your-signaling-server.com');

      this.signalServer.onopen = () => {
        console.log('信令服务器连接成功');
        resolve();
      };

      this.signalServer.onerror = (error) => {
        console.error('信令服务器错误:', error);
        reject(error);
      };

      this.signalServer.onmessage = (event) => {
        const data = JSON.parse(event.data);
        this.handleSignalMessage(data);
      };

      this.signalServer.onclose = () => {
        console.log('信令服务器断开');
      };
    });
  }

  // 发送信号
  sendSignal(data) {
    if (this.signalServer && this.signalServer.readyState === WebSocket.OPEN) {
      this.signalServer.send(JSON.stringify(data));
    }
  }

  // 处理信号消息
  async handleSignalMessage(data) {
    const { type, from, sdp, candidate, roomId } = data;

    switch (type) {
      case 'user-joined':
        console.log('用户加入:', from);
        await this.createOffer(from);
        break;

      case 'user-left':
        console.log('用户离开:', from);
        this.handleRemoteUserLeft();
        break;

      case 'offer':
        console.log('收到offer');
        await this.handleOffer(from, sdp);
        break;

      case 'answer':
        console.log('收到answer');
        await this.handleAnswer(from, sdp);
        break;

      case 'ice-candidate':
        console.log('收到ICE候选');
        await this.handleIceCandidate(from, candidate);
        break;
    }
  }

  // 处理远程用户离开
  handleRemoteUserLeft() {
    if (this.remoteStream) {
      this.remoteStream.getTracks().forEach((track) => track.stop());
      this.remoteStream = null;
    }

    if (this.peerConnection) {
      this.peerConnection.close();
      this.peerConnection = null;
    }

    // 隐藏远程视频
    const videoElement = document.getElementById('remoteVideo');
    if (videoElement) {
      videoElement.srcObject = null;
    }
  }

  // 处理连接状态变化
  handleConnectionStateChange(state) {
    switch (state) {
      case 'connected':
        console.log('通话已连接');
        break;
      case 'disconnected':
        console.log('通话断开');
        break;
      case 'failed':
        console.error('通话失败');
        break;
      case 'closed':
        console.log('通话已关闭');
        break;
    }
  }

  // 启动质量监控
  startQualityMonitor() {
    this.qualityMonitor = new QualityMonitor(this.peerConnection);
    this.qualityMonitor.start(2000);

    this.qualityMonitor.onReport((report) => {
      console.log('通话质量:', {
        rtt: report.connection.currentRoundTripTime,
        packetsLost: report.remote.video?.packetsLost,
        fractionLost: report.remote.video?.fractionLost,
      });
    });
  }

  // 结束通话
  endCall() {
    // 停止本地流
    if (this.localStream) {
      this.localStream.getTracks().forEach((track) => track.stop());
      this.localStream = null;
    }

    // 停止远程流
    if (this.remoteStream) {
      this.remoteStream.getTracks().forEach((track) => track.stop());
      this.remoteStream = null;
    }

    // 关闭PeerConnection
    if (this.peerConnection) {
      this.peerConnection.close();
      this.peerConnection = null;
    }

    // 停止质量监控
    if (this.qualityMonitor) {
      this.qualityMonitor.stop();
      this.qualityMonitor = null;
    }

    // 发送离开房间信号
    this.sendSignal({
      type: 'leave',
      roomId: this.roomId,
      userId: this.userId,
    });

    // 关闭信令服务器连接
    if (this.signalServer) {
      this.signalServer.close();
      this.signalServer = null;
    }
  }
}
```

```javascript
export function useStorage(key, initialValue) {
  const [value, setValue] = useState(() => {
    const stored = adapter.storage.get(key);
    return stored !== null && stored !== undefined ? stored : initialValue;
  });

  const setStoredValue = useCallback(
    (newValue) => {
      const valueToStore = typeof newValue === 'function' ? newValue(value) : newValue;
      setValue(valueToStore);
      adapter.storage.set(key, valueToStore);
    },
    [key, value]
  );

  const removeValue = useCallback(() => {
    setValue(initialValue);
    adapter.storage.remove(key);
  }, [key, initialValue]);

  return [value, setStoredValue, removeValue];
}
```

```javascript
// 1. 码率自适应
class AdaptiveBitrate {
  constructor(pc) {
    this.pc = pc;
    this.senders = new Map();
    this.targetBitrate = 2000000; // 2Mbps
    this.minBitrate = 500000; // 500kbps
    this.maxBitrate = 5000000; // 5Mbps
    this.lastQualityCheck = 0;
  }

  // 根据网络状况调整码率
  async adjustBitrate(stats) {
    const now = Date.now();
    if (now - this.lastQualityCheck < 2000) return; // 每2秒检查一次

    const { rtt, packetsLost, fractionLost, availableBandwidth } = stats;

    // 丢包率高，降低码率
    if (fractionLost > 0.1) {
      this.targetBitrate = Math.max(
        this.minBitrate,
        this.targetBitrate * 0.5
      );
    } else if (fractionLost > 0.05) {
      this.targetBitrate = Math.max(
        this.minBitrate,
        this.targetBitrate * 0.75
      );
    } else if (fractionLost < 0.01 && rtt < 0.1) {
      // 网络条件好，提高码率
      this.targetBitrate = Math.min(
        this.maxBitrate,
        this.targetBitrate * 1.25
      );
    }

    // 如果有可用带宽信息，使用它
    if (availableBandwidth) {
      this.targetBitrate = Math.min(
        this.targetBitrate,
        availableBandwidth * 0.8
      );
    }

    await this.applyBitrate();
    this.lastQualityCheck = now;
  }

  // 应用码率设置
  async applyBitrate() {
    const senders = this.pc.getSenders();

    for (const sender of senders) {
      if (sender.track?.kind === 'video') {
        const params = sender.getParameters();

        if (!params.encodings) {
          params.encodings = [{}];
        }

        params.encodings[0].maxBitrate = this.targetBitrate;

        // 调整分辨率
        if (this.targetBitrate < 1000000) {
          params.encodings[0].scaleResolutionDownBy = 2;
        } else if (this.targetBitrate < 2000000) {
          params.encodings[0].scaleResolutionDownBy = 1.5;
        } else {
          params.encodings[0].scaleResolutionDownBy = 1;
        }

        await sender.setParameters(params);
      }
    }
  }
}

class StatsMonitor {
  constructor(pc) {
    this.pc = pc;
    this.statsHistory = [];
    this.maxHistoryLength = 100;
  }

  async getStats() {
    const stats = await this.pc.getStats();
    const report = {};

    stats.forEach((stat) => {
      if (stat.type === 'candidate-pair' && stat.state === 'succeeded') {
        report.rtt = stat.currentRoundTripTime;
        report.availableBandwidth = stat.availableOutgoingBitrate;
        report.packetsLost = stat.packetsLost;
        report.packetsSent = stat.packetsSent;
        report.bytesSent = stat.bytesSent;
      }

      if (stat.type === 'inbound-rtp' && stat.kind === 'video') {
        report.videoPacketsLost = stat.packetsLost;
        report.videoJitter = stat.jitter;
        report.fractionLost = stat.fractionLost;
        report.framesDecoded = stat.framesDecoded;
      }

      if (stat.type === 'outbound-rtp' && stat.kind === 'video') {
        report.framesEncoded = stat.framesEncoded;
        report.keyFramesEncoded = stat.keyFramesEncoded;
      }
    });

    this.addToHistory(report);
    return report;
  }

  addToHistory(stats) {
    this.statsHistory.push({
      timestamp: Date.now(),
      ...stats,
    });

    if (this.statsHistory.length > this.maxHistoryLength) {
      this.statsHistory.shift();
    }
  }

  getAverageStats() {
    if (this.statsHistory.length === 0) return null;

    const sum = this.statsHistory.reduce((acc, stats) => ({
      rtt: (acc.rtt || 0) + (stats.rtt || 0),
      packetsLost: (acc.packetsLost || 0) + (stats.packetsLost || 0),
      fractionLost: (acc.fractionLost || 0) + (stats.fractionLost || 0),
    }), {});

    const count = this.statsHistory.length;
    return {
      avgRtt: sum.rtt / count,
      avgPacketsLost: sum.packetsLost / count,
      avgFractionLost: sum.fractionLost / count,
    };
  }
}
```

```javascript
class CacheManager {
  constructor() {
    this.cache = new Map();
    this.expireTime = 5 * 60 * 1000; // 5分钟
  }

  set(key, value, ttl = this.expireTime) {
    this.cache.set(key, {
      value,
      expireTime: Date.now() + ttl,
    });
  }

  get(key) {
    const item = this.cache.get(key);

    if (!item) return null;

    if (Date.now() > item.expireTime) {
      this.cache.delete(key);
      return null;
    }

    return item.value;
  }

  has(key) {
    return this.get(key) !== null;
  }

  delete(key) {
    this.cache.delete(key);
  }

  clear() {
    this.cache.clear();
  }

  // LRU缓存
  setWithLRU(key, value, ttl = this.expireTime) {
    if (this.cache.size >= 100) {
      // 删除最老的项
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    this.set(key, value, ttl);
  }
}

class RequestOptimizer {
  constructor() {
    this.pendingRequests = new Map();
    this.cache = new CacheManager();
  }

  // 请求防抖
  debounceRequest(key, request, delay = 300) {
    return new Promise((resolve, reject) => {
      const existingRequest = this.pendingRequests.get(key);

      if (existingRequest) {
        resolve(existingRequest);
        return;
      }

      const timeoutId = setTimeout(() => {
        this.pendingRequests.delete(key);
      }, delay);

      request()
        .then((result) => {
          clearTimeout(timeoutId);
          this.pendingRequests.delete(key);
          resolve(result);
        })
        .catch((error) => {
          clearTimeout(timeoutId);
          this.pendingRequests.delete(key);
          reject(error);
        });

      this.pendingRequests.set(key, request());
    });
  }

  // 请求缓存
  cachedRequest(key, request, ttl = 60000) {
    const cached = this.cache.get(key);

    if (cached) {
      return Promise.resolve(cached);
    }

    return request().then((result) => {
      this.cache.set(key, result, ttl);
      return result;
    });
  }

  // 请求合并
  batchRequest(keys, requestFn) {
    return Promise.all(keys.map((key) => requestFn(key)));
  }
}
```
