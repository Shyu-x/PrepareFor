### 2.2 认证模块详解

#### 2.2.1 JWT 策略

```typescript
// backend/src/auth/strategies/jwt.strategy.ts

import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';
import { ConfigService } from '@nestjs/config';
import { UsersService } from '@/users/users.service';
import { User } from '@/users/users.entity';

/**
 * JWT 认证策略
 *
 * 功能:
 * - 从请求中提取 JWT token
 * - 验证 token 有效性
 * - 获取并返回用户信息
 *
 * 装饰器: @UseGuards(JwtAuthGuard)
 */
@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(
    private configService: ConfigService,
    private usersService: UsersService,
  ) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      ignoreExpiration: false,
      secretOrKeyProvider: configService.get<string>('JWT_SECRET'),
    });
  }

  /**
   * 验证 token 并返回用户
   *
   * @param payload JWT payload 包含用户 ID
   * @returns 用户对象
   * @throws UnauthorizedException token 无效或用户不存在
   */
  async validate(payload: JwtPayload): Promise<User> {
    const { sub: userId } = payload;

    // 从数据库获取用户
    const user = await this.usersService.findById(userId);

    if (!user) {
      throw new UnauthorizedException('用户不存在');
    }

    if (!user.isActive) {
      throw new UnauthorizedException('用户已被禁用');
    }

    return user;
  }
}

/**
 * JWT Payload 接口
 */
export interface JwtPayload {
  sub: string;        // 用户 ID
  username: string;   // 用户名
  iat: number;        // 签发时间
  exp: number;        // 过期时间
}
```

#### 2.2.2 认证服务

```typescript
// backend/src/auth/auth.service.ts

import {
  Injectable,
  UnauthorizedException,
  BadRequestException,
  ConflictException,
} from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { ConfigService } from '@nestjs/config';
import * as bcrypt from 'bcrypt';
import { UsersService } from '@/users/users.service';
import { User } from '@/users/users.entity';

interface LoginDto {
  username: string;
  password: string;
}

interface RegisterDto {
  username: string;
  password: string;
  email?: string;
  displayName?: string;
}

interface AuthResponse {
  user: Omit<User, 'password'>;
  accessToken: string;
  refreshToken: string;
}

/**
 * 认证服务
 *
 * 功能:
 * - 用户注册
 * - 用户登录
 * - 刷新 token
 * - 验证用户
 * - 修改密码
 */
@Injectable()
export class AuthService {
  constructor(
    private usersService: UsersService,
    private jwtService: JwtService,
    private configService: ConfigService,
  ) {}

  /**
   * 用户注册
   *
   * @param dto 注册信息
   * @returns 认证响应 (用户 + tokens)
   * @throws ConflictException 用户名已存在
   * @throws BadRequestException 密码强度不足
   */
  async register(dto: RegisterDto): Promise<AuthResponse> {
    // 验证密码强度
    this.validatePassword(dto.password);

    // 检查用户名是否已存在
    const existingUser = await this.usersService.findByUsername(dto.username);
    if (existingUser) {
      throw new ConflictException('用户名已存在');
    }

    // 检查邮箱是否已存在
    if (dto.email) {
      const existingEmail = await this.usersService.findByEmail(dto.email);
      if (existingEmail) {
        throw new ConflictException('邮箱已被注册');
      }
    }

    // 加密密码 (10 轮盐)
    const hashedPassword = await bcrypt.hash(dto.password, 10);

    // 创建用户
    const user = await this.usersService.create({
      ...dto,
      password: hashedPassword,
      displayName: dto.displayName || dto.username,
      isActive: true,
      status: 'offline',
    });

    // 生成 tokens
    const tokens = await this.generateTokens(user);

    return {
      user,
      ...tokens,
    };
  }

  /**
   * 用户登录
   *
   * @param dto 登录信息
   * @returns 认证响应
   * @throws UnauthorizedException 用户名或密码错误
   */
  async login(dto: LoginDto): Promise<AuthResponse> {
    // 查找用户
    const user = await this.usersService.findByUsername(dto.username);

    if (!user) {
      throw new UnauthorizedException('用户名或密码错误');
    }

    // 验证密码
    const isPasswordValid = await bcrypt.compare(dto.password, user.password);

    if (!isPasswordValid) {
      throw new UnauthorizedException('用户名或密码错误');
    }

    // 检查用户是否被禁用
    if (!user.isActive) {
      throw new UnauthorizedException('账号已被禁用');
    }

    // 更新用户状态为在线
    await this.usersService.updateStatus(user.id, 'online');

    // 生成 tokens
    const tokens = await this.generateTokens(user);

    return {
      user,
      ...tokens,
    };
  }

  /**
   * 刷新访问令牌
   *
   * @param user 当前用户
   * @returns 新的访问令牌
   */
  async refreshToken(user: User): Promise<{ accessToken: string }> {
    const payload: JwtPayload = {
      sub: user.id,
      username: user.username,
      iat: Math.floor(Date.now() / 1000),
      exp: Math.floor(Date.now() / 1000) + this.configService.get('JWT_EXPIRES_IN'),
    };

    const accessToken = this.jwtService.sign(payload);

    return { accessToken };
  }

  /**
   * 验证用户
   *
   * @param userId 用户 ID
   * @returns 用户对象
   */
  async validateUser(userId: string): Promise<User> {
    return this.usersService.findById(userId);
  }

  /**
   * 修改密码
   *
   * @param userId 用户 ID
   * @param oldPassword 旧密码
   * @param newPassword 新密码
   * @throws UnauthorizedException 旧密码错误
   * @throws BadRequestException 新密码强度不足
   */
  async changePassword(
    userId: string,
    oldPassword: string,
    newPassword: string,
  ): Promise<void> {
    const user = await this.usersService.findById(userId);

    // 验证旧密码
    const isOldPasswordValid = await bcrypt.compare(oldPassword, user.password);
    if (!isOldPasswordValid) {
      throw new UnauthorizedException('旧密码错误');
    }

    // 验证新密码强度
    this.validatePassword(newPassword);

    // 加密新密码
    const hashedNewPassword = await bcrypt.hash(newPassword, 10);

    // 更新密码
    await this.usersService.updatePassword(userId, hashedNewPassword);
  }

  /**
   * 生成 access token 和 refresh token
   *
   * @param user 用户对象
   * @returns tokens
   */
  private async generateTokens(user: User): Promise<{
    accessToken: string;
    refreshToken: string;
  }> {
    const payload: JwtPayload = {
      sub: user.id,
      username: user.username,
      iat: Math.floor(Date.now() / 1000),
      exp: Math.floor(Date.now() / 1000) + this.configService.get('JWT_EXPIRES_IN'),
    };

    const accessToken = this.jwtService.sign(payload);

    // Refresh token 有效期更长 (7天)
    const refreshTokenPayload = {
      sub: user.id,
      iat: Math.floor(Date.now() / 1000),
      exp: Math.floor(Date.now() / 1000) + this.configService.get('JWT_REFRESH_EXPIRES_IN'),
    };

    const refreshToken = this.jwtService.sign(refreshTokenPayload);

    return { accessToken, refreshToken };
  }

  /**
   * 验证密码强度
   *
   * @param password 密码
   * @throws BadRequestException 密码强度不足
   */
  private validatePassword(password: string): void {
    if (password.length < 8) {
      throw new BadRequestException('密码长度至少8位');
    }

    if (!/[A-Z]/.test(password)) {
      throw new BadRequestException('密码必须包含大写字母');
    }

    if (!/[a-z]/.test(password)) {
      throw new BadRequestException('密码必须包含小写字母');
    }

    if (!/[0-9]/.test(password)) {
      throw new BadRequestException('密码必须包含数字');
    }
  }
}
```

```typescript
import {
  Entity,
  PrimaryGeneratedColumn,
  Column,
  OneToMany,
  ManyToOne,
  Tree,
  TreeChildren,
  TreeParent,
  CreateDateColumn,
  UpdateDateColumn,
} from 'typeorm';
import { Block } from './block.entity';
import { User } from '@/users/users.entity';

/**
 * 文档实体
 *
 * 使用 TypeORM 的树形结构装饰器实现文档的层级关系
 *
 * @Tree("closure-table") - 使用闭包表实现树形结构
 * - 优点: 查询子树、祖先路径效率高
 * - 缺点: 插入/删除需要更新闭包表
 */
@Entity('documents')
@Tree('closure-table')
export class DocumentEntity {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column({ type: 'varchar', length: 255 })
  title: string;

  @Column({ type: 'varchar', default: 'document' })
  type: 'document' | 'folder' | 'workspace';

  @Column({ type: 'text', nullable: true })
  summary: string; // AI 生成的智能摘要

  @Column({ type: 'boolean', default: false })
  isStarred: boolean;

  @Column({ type: 'boolean', default: false })
  isDeleted: boolean; // 软删除标记

  @Column({ type: 'varchar', length: 500, nullable: true })
  cover: string;

  @Column({ type: 'varchar', length: 100, nullable: true })
  icon: string;

  @Column({ type: 'varchar', length: 50, nullable: true })
  color: string;

  @Column({ type: 'int', default: 0 })
  sort: number;

  // 所有者关系
  @Column({ type: 'uuid', nullable: true })
  ownerId: string;

  @ManyToOne(() => User, { onDelete: 'SET NULL' })
  owner?: User;

  @Column({ type: 'jsonb', nullable: true })
  members: {
    id: string;
    name: string;
    role: 'owner' | 'admin' | 'editor' | 'reader';
  }[];

  @Column({ type: 'varchar', length: 50, default: 'private' })
  permission: 'private' | 'workspace' | 'public';

  @Column({ type: 'boolean', default: true })
  allowComment: boolean;

  @Column({ type: 'boolean', default: true })
  allowCopy: boolean;

  @Column({ type: 'boolean', default: true })
  allowDownload: boolean;

  @Column({ type: 'varchar', length: 255, nullable: true })
  password: string;

  @Column({ type: 'timestamp', nullable: true })
  expiresAt: Date;

  @Column({ type: 'int', default: 0 })
  viewCount: number;

  @Column({ type: 'int', default: 0 })
  likeCount: number;

  @Column({ type: 'int', default: 0 })
  currentVersion: number;

  @Column({ type: 'int', default: 0 })
  editCount: number;

  // 块关系 (级联删除)
  @OneToMany(() => Block, (block) => block.document, { cascade: true })
  blocks: Block[];

  // 树形关系
  @TreeChildren()
  children: DocumentEntity[];

  @TreeParent()
  parent: DocumentEntity;

  @CreateDateColumn({ type: 'timestamp' })
  createdAt: Date;

  @UpdateDateColumn({ type: 'timestamp' })
  updatedAt: Date;
}

/**
 * 块实体
 */
@Entity('blocks')
export class BlockEntity {

  order: number;

  @CreateDateColumn({ type: 'timestamp' })
  createdAt: Date;
```

#### 2.3.2 文档服务实现

```typescript
// backend/src/documents/documents.service.ts

import { Injectable, NotFoundException, ForbiddenException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, In, IsNull, MoreThan } from 'typeorm';
import { DocumentEntity } from './document.entity';
import { BlockEntity } from './block.entity';
import { BlockLockService } from './block-lock.service';
import { EditOperationService } from './edit-operation.service';
import { Cache } from 'cache-manager';

/**
 * 文档服务
 *
 * 功能:
 * - 文档 CRUD 操作
 * - 文档树管理
 * - 块操作
 * - 收藏/回收站/永久删除
 * - 智能摘要
 * - 分块加载
 * - 增量同步
 */
@Injectable()
export class DocumentsService {
  constructor(
    @InjectRepository(DocumentEntity)
    private documentsRepository: Repository<DocumentEntity>,
    @InjectRepository(BlockEntity)
    private blocksRepository: Repository<BlockEntity>,
    private blockLockService: BlockLockService,
    private editOperationService: EditOperationService,
    @Inject(CACHE_MANAGER) private cacheManager: Cache,
  ) {}

  /**
   * 获取文档树
   *
   * @param userId 用户 ID
   @returns 文档树结构
   */
  async findTree(userId: string): Promise<DocumentEntity[]> {
    // 尝试从缓存获取
    const cacheKey = `documents:tree:${userId}`;
    const cached = await this.cacheManager.get<DocumentEntity[]>(cacheKey);
    if (cached) {
      return cached;
    }

    // 查询根节点 (没有父节点的文档)
    const rootDocuments = await this.documentsRepository.find({
      where: {
        parent: IsNull(),
        isDeleted: false,
      },
      relations: ['owner', 'children'],
    });

    // 过滤用户有权限访问的文档
    const filtered = rootDocuments.filter(doc => this.hasPermission(doc, userId));

    // 缓存 5 分钟
    await this.cacheManager.set(cacheKey, filtered, 300);

    return filtered;
  }

  /**
   * 获取单个文档
   *
   * @param id 文档 ID
   * @param userId 用户 ID
   * @returns 文档详情
   * @throws NotFoundException 文档不存在
   * @throws ForbiddenException 无访问权限
   */
  async findOne(id: string, userId?: string): Promise<DocumentEntity> {
    const document = await this.documentsRepository.findOne({
      where: { id },
      relations: ['owner', 'parent', 'children', 'blocks'],
    });

    if (!document) {
      throw new NotFoundException('文档不存在');
    }

    if (userId && !this.hasPermission(document, userId)) {
      throw new ForbiddenException('无访问权限');
    }

    // 更新查看次数
    await this.documentsRepository.increment('viewCount', { id });

    return document;
  }

  /**
   * 创建文档
   *
   * @param createDto 创建 DTO
   * @param ownerId 所有者 ID
   * @returns 创建的文档
   */
  async create(
    createDto: {
      title: string;
      type?: string;
      parentId?: string;
      blocks?: any[];
    },
    ownerId: string,
  ): Promise<DocumentEntity> {
    // 验证父文档权限
    if (createDto.parentId) {
      const parent = await this.documentsRepository.findOne({
        where: { id: createDto.parentId },
      });

      if (!parent) {
        throw new NotFoundException('父文档不存在');
      }

      if (parent.isDeleted) {
        throw new ForbiddenException('不能在回收站中创建文档');
      }

      // 检查父文档类型
      if (parent.type !== 'folder') {
        throw new BadRequestException('只能在文件夹中创建文档');
      }

      if (!this.hasPermission(parent, ownerId)) {
        throw new ForbiddenException('无权限在该位置创建文档');
      }
    }

    // 创建文档
    const document = this.documentsRepository.create({
      ...createDto,
      ownerId,
      members: [{ id: ownerId, name: '', role: 'owner' }],
    });

    // 创建初始块
    if (createDto.blocks && createDto.blocks.length > 0) {
      const blocks = createDto.blocks.map((block, index) =>
        this.blocksRepository.create({
          ...block,
          documentId: document.id,
          order: index,
        }),
      );
      document.blocks = blocks;
    }

    // 如果是文件夹且没有子节点，创建空数组
    if (createDto.type === 'folder') {
      document.children = [];
    }

    const savedDocument = await this.documentsRepository.save(document);

    // 清除文档树缓存
    await this.clearDocumentTreeCache();

    return savedDocument;
  }

  /**
   * 更新文档
   *
   * @param id 文档 ID
   * @param updateDto 更新 DTO
   * @param userId 用户 ID
   * @returns 更新后的文档
   */
  async update(
    id: string,
    updateDto: Partial<{
      title: string;
      type: string;
      summary: string;
      cover: string;
      icon: string;
      color: string;
      permission: string;
      password?: string;
      expiresAt?: Date;
    }>,
    userId: string,
  ): Promise<DocumentEntity> {
    const document = await this.findOne(id, userId);

    // 检查编辑权限
    if (!this.canEdit(document, userId)) {
      throw new ForbiddenException('无编辑权限');
    }

    // 更新字段
    Object.assign(document, updateDto);

    const updated = await this.documentsRepository.save(document);

    // 清除文档树缓存
    await this.clearDocumentTreeCache();

    return updated;
  }

  /**
   * 删除文档（软删除）
   *
   * @param id 文档 ID
   * @param userId 用户 ID
   */
  async moveToTrash(id: string, userId: string): Promise<void> {
    const document = await this.findOne(id, userId);

    // 检查删除权限
    if (!this.canDelete(document, userId)) {
      throw new ForbiddenException('无删除权限');
    }

    // 软删除
    document.isDeleted = true;
    await this.documentsRepository.save(document);

    // 清除文档树缓存
    await this.clearDocumentTreeCache();
  }

  /**
   * 恢复文档
   *
   * @param id 文档 ID
   * @param userId 用户 ID
   */
  async restoreFromTrash(id: string, userId: string): Promise<void> {
    const document = await this.documentsRepository.findOne({
      where: { id, isDeleted: true },
      relations: ['owner', 'parent'],
    });

    if (!document) {
      throw new NotFoundException('文档不存在或不在回收站中');
    }

    // 检查权限
    if (!this.canDelete(document, userId)) {
      throw new ForbiddenException('无恢复权限');
    }

    // 如果父文档也在回收站中，需要先恢复父文档
    if (document.parent && document.parent.isDeleted) {
      throw new BadRequestException('父文档也在回收站中，请先恢复父文档');
    }

    document.isDeleted = false;
    await this.documentsRepository.save(document);

    // 清除文档树缓存
    await this.clearDocumentTreeCache();
  }

  /**
   * 永久删除文档
   *
   * @param id 文档 ID
   * @param userId 用户 ID
   */
  async permanentlyDelete(id: string, userId: string): Promise<void> {
    const document = await this.documentsRepository.findOne({
      where: { id, isDeleted: true },
    });

    if (!document) {
      throw new NotFoundException('文档不存在或不在回收站中');
    }

    // 检查权限
    if (!this.canDelete(document, userId)) {
      throw new ForbiddenException('无删除权限');
    }

    // 永久删除 (级联删除所有块)
    await this.documentsRepository.remove(document);

    // 清除缓存
    await this.cacheManager.del(`documents:${id}`);
    await this.clearDocumentTreeCache();
  }

  /**
   * 分块加载文档块
   *
   * @param documentId 文档 ID
   * @param offset 偏移量
   * @param limit 限制数量
   * @param includeTotal 是否包含总数
   * @returns 块数据
   */
  async findBlocksChunked(
    documentId: string,
    { offset = 0, limit = 50, includeTotal = false } = {},
  ): Promise<{
    blocks: BlockEntity[];
    total?: number;
  }> {
    // 尝试从缓存获取
    const cacheKey = `documents:${documentId}:blocks:${offset}:${limit}`;
    const cached = await this.cacheManager.get<{ blocks: BlockEntity[]; total?: number }>(
      cacheKey,
    );
    if (cached) {
      return cached;
    }

    // 查询块
    const [blocks, totalResult] = await Promise.all([
      this.blocksRepository.find({
        where: { documentId },
        order: { order: 'ASC' },
        skip: offset,
        take: limit,
      }),
      includeTotal ? this.blocksRepository.count({ where: { documentId } }) : undefined,
    ]);

    const result = {
      blocks,
      total: totalResult,
    };

    // 缓存 30 秒
    await this.cacheManager.set(cacheKey, result, 30);

    return result;
  }

  /**
   * 增量同步块
   *
   * @param documentId 文档 ID
   * @param since 时间戳（毫秒）
   * @returns 增量变更
   */
  async findBlocksSince(
    documentId: string,
    since: number,
  ): Promise<{
    updated: BlockEntity[];
    created: BlockEntity[];
    deleted: string[];
  }> {
    const timestamp = new Date(since);

    const [updated, created, deletedBlocks] = await Promise.all([
      this.blocksRepository.find({
        where: {
          documentId,
          updatedAt: MoreThan(timestamp),
        },
        order: { updatedAt: 'ASC' },
      }),
      this.blocksRepository.find({
        where: {
          documentId,
          createdAt: MoreThan(timestamp),
        },
        order: { createdAt: 'ASC' },
      }),
      // 从编辑操作记录获取删除的块 ID
      this.editOperationService.getDeletedBlockIdsSince(documentId, since),
    ]);

    return {
      updated,
      created,
      deleted: deletedBlocks,
    };
  }

  /**
   * 更新块
   *
   * @param documentId 文档 ID
   * @param blockId 块 ID
   * @param changes 变更内容
   * @param userId 用户 ID
   */
  async updateBlock(
    documentId: string,
    blockId: string,
    changes: {
      content?: string;
      type?: string;
      properties?: any;
    },
    userId: string,
  ): Promise<BlockEntity> {
    // 验证文档权限
    const document = await this.findOne(documentId, userId);
    if (!this.canEdit(document, userId)) {
      throw new ForbiddenException('无编辑权限');
    }

    // 获取块
    const block = await this.blocksRepository.findOne({
      where: { id: blockId, documentId },
    });

    if (!block) {
      throw new NotFoundException('块不存在');
    }

    // 检查块锁
    const lock = await this.blockLockService.getActiveLock(documentId, blockId);
    if (lock && lock.userId !== userId) {
      throw new ForbiddenException(
        `该块正被 ${lock.userName} 编辑，无法修改`,
      );
    }

    // 更新块
    Object.assign(block, changes);
    const updatedBlock = await this.blocksRepository.save(block);

    // 记录编辑操作
    await this.editOperationService.recordOperation({
      documentId,
      blockId,
      operationType: 'update',
      operationData: {
        oldValue: block,
        newValue: changes,
      },
      userId,
      userName: '', // TODO: 从用户服务获取
    });

    // 清除块缓存
    await this.cacheManager.del(`documents:${documentId}:blocks:*`);

    return updatedBlock;
  }

  /**
   * 添加块
   *
   * @param documentId 文档 ID
   * @param blockData 块数据
   * @param afterBlockId 在哪个块后添加
   * @param userId 用户 ID
   */
  async addBlock(
    documentId: string,
    blockData: {
      type: string;
      content?: string;
      properties?: any;
    },
    afterBlockId?: string,
    userId: string,
  ): Promise<BlockEntity> {
    // 验证文档权限
    const document = await this.findOne(documentId, userId);
    if (!this.canEdit(document, userId)) {
      throw new ForbiddenException('无编辑权限');
    }

    // 计算顺序
    let order = 0;
    if (afterBlockId) {
      const afterBlock = await this.blocksRepository.findOne({
        where: { id: afterBlockId, documentId },
      });
      if (afterBlock) {
        order = afterBlock.order + 1;
      }
    } else {
      const maxOrder = await this.blocksRepository.maximum('order', {
        where: { documentId },
      });
      order = (maxOrder || 0) + 1;
    }

    // 更新后续块的顺序
    if (order > 0) {
      await this.blocksRepository
        .createQueryBuilder()
        .update(BlockEntity)
        .set({ order: () => '"order" + 1' })
        .where({ documentId, order: MoreThanOrEqual(order) })
        .andWhere({ id: Not(afterBlockId) })
        .execute();
    }

    // 创建块
    const block = this.blocksRepository.create({
      ...blockData,
      documentId,
      order,
    });

    const savedBlock = await this.blocksRepository.save(block);

    // 记录编辑操作
    await this.editOperationService.recordOperation({
      documentId,
      blockId: savedBlock.id,
      operationType: 'add',
      operationData: { position: order },
      userId,
      userName: '',
    });

    // 清除块缓存
    await this.cacheManager.del(`documents:${documentId}:blocks:*`);

    return savedBlock;
  }

  /**
   * 删除块
   *
   * @param documentId 文档 ID
   * @param blockId 块 ID
   * @param userId 用户 ID
   */
  async deleteBlock(
    documentId: string,
    blockId: string,
    userId: string,
  ): Promise<void> {
    // 验证文档权限
    const document = await this.findOne(documentId, userId);
    if (!this.canEdit(document, userId)) {
      throw new ForbiddenException('无编辑权限');
    }

    // 检查块锁
    const lock = await this.blockLockService.getActiveLock(documentId, blockId);
    if (lock && lock.userId !== userId) {
      throw new ForbiddenException(
        `该块正被 ${lock.userName} 编辑，无法删除`,
      );
    }

    // 删除块
    await this.blocksRepository.delete({ id: blockId, documentId });

    // 记录编辑操作
    await this.editOperationService.recordOperation({
      documentId,
      blockId,
      operationType: 'delete',
      operationData: {},
      userId,
      userName: '',
    });

    // 清除块缓存
    await this.cacheManager.del(`documents:${documentId}:blocks:*`);
  }

  /**
   * 切换收藏状态
   *
   * @param id 文档 ID
   * @param userId 用户 ID
   * @returns 是否已收藏
   */
  async toggleStar(id: string, userId: string): Promise<boolean> {
    const document = await this.findOne(id, userId);

    document.isStarred = !document.isStarred;
    await this.documentsRepository.save(document);

    return document.isStarred;
  }

  /**
   * 智能分析和摘要
   *
   * @param id 文档 ID
   * @param userId 用户 ID
   * @returns 生成的摘要
   */
  async analyzeAndSummary(id: string, userId: string): Promise<string> {
    const document = await this.findOne(id, userId);

    // TODO: 集成 AI 服务生成智能摘要
    const summary = `本文档包含 ${document.blocks.length} 个块，主要内容包括...`;

    document.summary = summary;
    await this.documentsRepository.save(document);

    return summary;
  }

  /**
   * 检查用户是否有访问权限
   */
  private hasPermission(document: DocumentEntity, userId: string): boolean {
    // 所有者
    if (document.ownerId === userId) {
      return true;
    }

    // 私有文档
    if (document.permission === 'private') {
      return false;
    }

    // 工作空间文档
    if (document.permission === 'workspace') {
      const member = document.members?.find((m) => m.id === userId);
      return !!member;
    }

    // 公开文档
    if (document.permission === 'public') {
      return true;
    }

    return false;
  }

  /**
   * 检查用户是否可以编辑
   */
  private canEdit(document: DocumentEntity, userId: string): boolean {
    if (document.ownerId === userId) {
      return true;
    }

    const member = document.members?.find((m) => m.id === userId);
    return ['owner', 'admin', 'editor'].includes(member?.role);
  }

  /**
   * 检查用户是否可以删除
   */
  private canDelete(document: DocumentEntity, userId: string): boolean {
    if (document.ownerId === userId) {
      return true;
    }

    const member = document.members?.find((m) => m.id === userId);
    return ['owner', 'admin'].includes(member?.role);
  }

  /**
   * 清除文档树缓存
   */
  private async clearDocumentTreeCache(): Promise<void> {
    // TODO: 需要获取所有用户 ID 并清除其文档树缓存
    // 这里简化为清除所有文档树缓存

  }
```

### 2.4 块锁服务

```typescript
// backend/src/documents/block-lock.service.ts

import { Injectable, NotFoundException, BadRequestException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { BlockLockEntity } from './block-lock.entity';

/**
 * 块锁服务
 *
 * 功能:
 * - 获取块锁
 * - 释放块锁
 * - 刷新块锁
 * - 清理过期锁
 */
@Injectable()
export class BlockLockService {
  private readonly LOCK_DURATION = 5 * 60 * 1000; // 5分钟

  constructor(
    @InjectRepository(BlockLockEntity)
    private blockLockRepository: Repository<BlockLockEntity>,
  ) {}

  /**
   * 获取块锁
   *
   * @param dto 锁请求 DTO
   * @returns 锁信息
   * @throws BadRequestException 块已被锁定
   */
  async acquireLock(
    dto: {
      documentId: string;
      blockId: string;
      userId: string;
      userName: string;
      lockType?: 'edit' | 'read';
    },
  ): Promise<BlockLockEntity> {
    // 检查是否已有活跃锁
    const existingLock = await this.blockLockRepository.findOne({
      where: {
        blockId: dto.blockId,
        isLocked: true,
        expiresAt: MoreThan(new Date()),
      },
    });

    if (existingLock) {
      // 如果是同一个用户，刷新锁
      if (existingLock.userId === dto.userId) {
        return this.refreshLock(existingLock.id, dto.userId);
      }

      throw new BadRequestException(
        `该块正被 ${existingLock.userName} 锁定，无法编辑`,
      );
    }

    // 创建锁
    const lock = this.blockLockRepository.create({
      documentId: dto.documentId,
      blockId: dto.blockId,
      userId: dto.userId,
      userName: dto.userName,
      lockType: dto.lockType || 'edit',
      lockedAt: Date.now(),
      expiresAt: Date.now() + this.LOCK_DURATION,
      isLocked: true,
    });

    return this.blockLockRepository.save(lock);
  }

  /**
   * 释放块锁
   *
   * @param lockId 锁 ID
   * @param userId 用户 ID
   * @throws BadRequestException 无权限释放
   */
  async releaseLock(lockId: string, userId: string): Promise<void> {
    const lock = await this.blockLockRepository.findOne({
      where: { id: lockId },
    });

    if (!lock) {
      throw new NotFoundException('锁不存在');
    }

    if (lock.userId !== userId) {
      throw new BadRequestException('无权限释放该锁');
    }

    lock.isLocked = false;
    await this.blockLockRepository.save(lock);
  }

  /**
   * 刷新块锁
   *
   * @param lockId 锁 ID
   * @param userId 用户 ID
   * @returns 刷新后的锁
   */
  async refreshLock(lockId: string, userId: string): Promise<BlockLockEntity> {
    const lock = await this.blockLockRepository.findOne({
      where: { id: lockId },
    });

    if (!lock) {
      throw new NotFoundException('锁不存在');
    }

    if (lock.userId !== userId) {
      throw new BadRequestException('无权限刷新该锁');
    }

    lock.lockedAt = Date.now();
    lock.expiresAt = Date.now() + this.LOCK_DURATION;

    return this.blockLockRepository.save(lock);
  }

  /**
   * 获取文档的所有活跃锁
   *
   * @param documentId 文档 ID
   * @returns 活跃锁列表
   */
  async getActiveLocks(documentId: string): Promise<BlockLockEntity[]> {
    return this.blockLockRepository.find({
      where: {
        documentId,
        isLocked: true,
        expiresAt: MoreThan(new Date()),
      },
      order: { lockedAt: 'DESC' },
    });
  }

  /**
   * 获取块的活跃锁
   *
   * @param documentId 文档 ID
   * @param blockId 块 ID
   * @returns 锁信息
   */
  async getActiveLock(
    documentId: string,
    blockId: string,
  ): Promise<BlockLockEntity | null> {
    return this.blockLockRepository.findOne({
      where: {
        documentId,
        blockId,
        isLocked: true,
        expiresAt: MoreThan(new Date()),
      },
    });
  }

  /**
   * 强制释放锁 (管理员操作)
   *
   * @param lockId 锁 ID
   * @param adminUserId 管理员用户 ID
   */
  async forceReleaseLock(lockId: string, adminUserId: string): Promise<void> {
    const lock = await this.blockLockRepository.findOne({
      where: { id: lockId },
    });

    if (!lock) {
      throw new NotFoundException('锁不存在');
    }

    lock.isLocked = false;
    await this.blockLockRepository.save(lock);
  }

  /**
   * 清理过期锁
   *
   * @param documentId 文档 ID
   * @returns 清理的锁数量
   */
  async cleanupExpiredLocks(documentId: string): Promise<number> {
    const result = await this.blockLockRepository.update(
      { isLocked: false },
      {
        where: {
          documentId,
          expiresAt: LessThan(new Date()),
        },
      },
    );

    return result.affected || 0;
  }

  /**
   * 检查块是否被指定用户锁定
   *
   * @param documentId 文档 ID
   * @param blockId 块 ID
   * @param userId 用户 ID
   * @returns 是否被锁定
   */
  async isBlockLockedByUser(
    documentId: string,
    blockId: string,
    userId: string,
  ): Promise<boolean> {
    const lock = await this.getActiveLock(documentId, blockId);

    return lock?.userId === userId;
  }
}
```

```typescript
export class TerminalService {

    socketPath: process.env.DOCKER_SOCKET_PATH || '/var/run/docker.sock',
  });

  private readonly logger = new Logger(TerminalService.name);
  private readonly sessions = new Map<string, any>();

  /**
   * 创建终端会话
   *
   * @param dto 会话参数
   * @returns 会话信息
   */
  async createSession(dto: {
    sessionId: string;
    workspaceId: string;
    mode: 'webenvos' | 'debian';
    cols?: number;
    rows?: number;
  }): Promise<any> {
    const container = await this.docker.getContainer(dto.workspaceId);

    // 创建 Exec 实例
    const exec = await container.exec({
      Cmd: ['/bin/bash', '-l'],
      AttachStdin: true,
      AttachStdout: true,
      AttachStderr: true,
      Tty: true,
    });

    // 启动 Exec
    await exec.start({ Detach: false });

    // 创建会话
    const session = {
      id: dto.sessionId,
      containerId: dto.workspaceId,
      execId: exec.id,
      pty: null,
      resizeCols: dto.cols || 80,
      resizeRows: dto.rows || 24,
    };

    this.sessions.set(dto.sessionId, session);

    this.logger.log(`Session created: ${dto.sessionId}, exec: ${exec.id}`);

    // 启动输出流
    this.startOutputStream(dto.sessionId, exec);

    return session;
  }

  /**
   * 启动输出流
   *
   * @param sessionId 会话 ID
   * @param exec Exec 实例
   */
  private async startOutputStream(sessionId: string, exec: any) {
    // 监听输出
    exec.start({ hijack: false, stdin: false }, (err, stream) => {
      if (err) {
        this.logger.error(`Failed to start stream: ${err.message}`);
        return;
      }

      stream.on('data', (chunk: Buffer) => {
        // 发送输出到 WebSocket
        // TODO: 通过 TerminalGateway 发送
        this.logger.log(`Session ${sessionId} output: ${chunk.length} bytes`);
      });

      stream.on('error', (err) => {
        this.logger.error(`Stream error: ${err.message}`);
      });

      stream.on('end', () => {
        this.logger.log(`Session ${sessionId} stream ended`);
      });
    });
  }

  /**
   * 写入 PTY
   *
   * @param sessionId 会话 ID
   * @param data 输入数据
   */
  async writeToPty(sessionId: string, data: string): Promise<void> {
    const session = this.sessions.get(sessionId);

    if (!session) {
      throw new NotFoundException(`Session not found: ${sessionId}`);
    }

    const container = await this.docker.getContainer(session.containerId);

    // 获取 Exec 实例
    const exec = await container.exec.inspect(session.execId);

    // 写入输入
    await exec.resize({ h: session.resizeRows, w: session.resizeCols });
    // TODO: 实际写入输入
  }

  /**
   * 调整 PTY 大小
   *
   * @param sessionId 会话 ID
   * @param cols 列数
   * @param rows 行数
   */
  async resizePty(sessionId: string, cols: number, rows: number): Promise<void> {
    const session = this.sessions.get(sessionId);

    if (!session) {
      throw new NotFoundException(`Session not found: ${sessionId}`);
    }

    const container = await this.docker.getContainer(session.containerId);

    const exec = await container.exec.inspect(session.execId);

    await exec.resize({ h: rows, w: cols });

    session.resizeCols = cols;
    session.resizeRows = rows;
  }

  /**
   * 销毁会话
   *
   * @param sessionId 会话 ID
   */
  async destroySession(sessionId: string): Promise<void> {
    const session = this.sessions.get(sessionId);

    if (!session) {
      throw new NotFoundException(`Session not found: ${sessionId}`);
    }

    this.sessions.delete(sessionId);

    this.logger.log(`Session destroyed: ${sessionId}`);
  }

  /**
   * 获取容器统计信息
   *
   * @param containerId 容器 ID
   * @returns 统计信息
   */
  async getContainerStats(containerId: string): Promise<any> {
    const container = await this.docker.getContainer(containerId);
    return container.stats({ stream: false });
  }

  /**
   * 获取容器列表
   *
   * @returns 容器列表
   */
  async getContainers(): Promise<any[]> {
    const containers = await this.docker.listContainers({ all: false });

    return containers.map((container) => ({
      id: container.Id,
      name: container.Names[0]?.replace(/^\//, '') || '',
      state: container.State,
      image: container.Image,
      created: container.Created,
    }));
  }
}
```

```sql
-- 闭包表触发器
CREATE OR REPLACE FUNCTION document_closure_trigger()
RETURNS TRIGGER AS $$
BEGIN
    -- 插入新的子节点时
    IF TG_OP = 'INSERT' AND NEW.id IS NOT NULL THEN
        -- 插入自己
        INSERT INTO document_closure (ancestor_id, descendant_id, depth)
        VALUES (NEW.id, NEW.id, 0);

        -- 如果有父节点，复制父节点的所有祖先
        IF NEW.parent_id IS NOT NULL THEN
            INSERT INTO document_closure (ancestor_id, descendant_id, depth)
            SELECT ancestor_id, NEW.id, depth + 1
            FROM document_closure
            WHERE descendant_id = NEW.parent_id;
        END IF;
    END IF;

    -- 更新父节点时
    IF TG_OP = 'UPDATE' AND NEW.parent_id != OLD.parent_id THEN
        -- 删除旧的层级关系
        DELETE FROM document_closure
        WHERE descendant_id = OLD.id;

        -- 插入新的层级关系
        INSERT INTO document_closure (ancestor_id, descendant_id, depth)
        VALUES (NEW.id, NEW.id, 0);

        IF NEW.parent_id IS NOT NULL THEN
            INSERT INTO document_closure (ancestor_id, descendant_id, depth)
            SELECT ancestor_id, NEW.id, depth + 1
            FROM document_closure
            WHERE descendant_id = NEW.parent_id;
        END IF;
    END IF;

    -- 删除节点时
    IF TG_OP = 'DELETE' THEN
        DELETE FROM document_closure
        WHERE descendant_id = OLD.id
        OR ancestor_id = OLD.id;
    END IF;

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER document_closure_trigger
    BEFORE INSERT OR UPDATE OR DELETE ON documents
    FOR EACH ROW EXECUTE FUNCTION document_closure_trigger();
```

```sql
DECLARE
    version_count INT;
BEGIN
    -- 统计版本数量
    SELECT COUNT(*) INTO version_count
    FROM document_versions
    WHERE document_id = NEW.document_id;

    -- 如果超过100个，删除最旧的版本
    IF version_count > 100 THEN
        DELETE FROM document_versions
        WHERE document_id = NEW.document_id
        AND id != NEW.id
        ORDER BY created_at ASC
        LIMIT version_count - 100;

        RAISE NOTICE 'Version count exceeded 100, oldest versions deleted';
    END IF;

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

    AFTER INSERT ON document_versions
    FOR EACH ROW
    WHEN (NEW.version_number > 100)  -- 只在新版本号超过100时触发
```

```typescript
/**

 * 功能:

@Injectable()

    await this.cacheManager.set(key, tree, ttl);
  }

   * 获取缓存的文档树

  async getCachedDocumentTree(userId: string): Promise<any[] | undefined> {

  async cacheBlocks(

    offset: number,
    limit: number,
    blocks: any[],
    ttl: number = 30, // 30秒
  ): Promise<void> {
    const key = `documents:${documentId}:blocks:${offset}:${limit}`;
    await this.cacheManager.set(key, { blocks }, ttl);

   * 获取缓存的块
   */
  async getCachedBlocks(
    documentId: string,
    offset: number,
    limit: number,
  ): Promise<any | undefined> {
    const key = `documents:${documentId}:blocks:${offset}:${limit}`;
    return await this.cacheManager.get(key);
  }

   * 清除文档相关缓存
   */
  async clearDocumentCache(documentId: string): Promise<void> {
    // 清除块缓存
    const keys = await this.cacheManager.store.keys(`documents:${documentId}:*`);
    for (const key of keys) {
      await this.cacheManager.del(key);
    }
  }

  /**
   * 缓存用户会话
   *
   * @param userId 用户 ID
   * @param session 会话数据
   * @param ttl 生存时间
   */
  async cacheUserSession(
    userId: string,
    session: any,
    ttl: number = 3600, // 1小时
  ): Promise<void> {
    const key = `users:session:${userId}`;
    await this.cacheManager.set(key, session, ttl);

  async getCachedUserSession(

    const key = `users:session:${userId}`;
    return await this.cacheManager.get(key);

   * 限流计数
   *
   * @param key 键
   * @param ttl 生存时间
   * @returns 当前计数
   */
  async incrementRateLimit(
    key: string,
    ttl: number = 60,
  ): Promise<number> {
    const keyWithTTL = `ratelimit:${key}`;

    // 尝试获取现有计数
    let count = await this.cacheManager.get(keyWithTTL);

    if (count === undefined) {
      // 不存在，初始化为1
      count = 1;
    } else {
      // 存在，递增
      count = parseInt(count, 10) + 1;
    }

    await this.cacheManager.set(keyWithTTL, count, ttl);

    return count;
  }

  /**
   * 清除限流计数
   */

    await this.cacheManager.del(keyWithTTL);
  }

   */
  async getState(key: string): Promise<any | undefined> {
    const keyWithTTL = `state:${key}`;
    return await this.cacheManager.get(keyWithTTL);
  }
```

## 五、面试题与答案

### 5.1 数据库设计

**Q: 为什么 FastDocument 使用闭包表实现文档树？**

A: 闭包表的优势：
1. **查询效率**: 查询子树和祖先路径都是 O(1) 操作
2. **层级遍历**: 支持任意深度的树遍历
3. **无需递归**: 不需要递归查询，避免性能问题

查询示例：
```sql
-- 查找某个节点的所有子节点
SELECT descendant_id FROM document_closure
WHERE ancestor_id = 'doc-123';

-- 查找某个节点的所有祖先节点
SELECT ancestor_id FROM document_closure
WHERE descendant_id = 'doc-456';
```

**Q: 如何实现版本历史管理？**

A: 使用独立的版本表存储完整快照：
```typescript
interface DocumentVersion {
  id: string;
  documentId: string;
  versionNumber: number;
  title: string;
  blocks: Block[];  // JSON 存储完整块数据
  createdBy: string;
  createdByName: string;
  metadata: {
    autoSave: boolean;
    wordCount: number;
    blockCount: number;
  };
}
```

优点：
1. **完整快照**: 包含完整的文档状态，可以精确恢复
2. **独立查询**: 版本列表查询不影响主表性能
3. **差异化存储**: 可以优化为只存储差异（TODO）

### 5.2 缓存策略

**Q: FastDocument 的缓存策略是什么？**

A: 分层缓存策略：
1. **文档树缓存**: Redis，5分钟 TTL
2. **块缓存**: Redis，30秒 TTL（按 offset + limit 作为键）
3. **用户会话缓存**: Redis，1小时 TTL

缓存键命名规范：
```
documents:tree:{userId}
documents:{documentId}:blocks:{offset}:{limit}
users:session:{userId}
ratelimit:{action}
state:{key}
```

失效策略：
- 文档更新 → 清除文档树缓存
- 块更新 → 清除块缓存
- 用户登出 → 清除会话缓存

### 5.3 并发控制

**Q: 如何处理多人同时编辑同一块的冲突？**

A: 使用块锁机制：
1. **编辑前获取锁**: `acquireLock(documentId, blockId, userId)`
2. **锁自动过期**: 5分钟后自动释放
3. **同一用户刷新锁**: 如果是同一个用户可以刷新锁

锁实体：
```typescript
interface BlockLock {
  id: string;
  documentId: string;
  blockId: string;
  userId: string;
  userName: string;
  lockType: 'edit' | 'read';
  lockedAt: number;
  expiresAt: number;
  isLocked: boolean;
}
```

防冲突措施：
- 后端验证锁的所有权
- 前端显示"正在编辑"状态
- 支持管理员强制释放锁
