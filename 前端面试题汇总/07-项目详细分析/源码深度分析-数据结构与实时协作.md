```typescript
/**
 * 原子化文档状态中心
 * 支持本地持久化缓存、多端同步逻辑及分类过滤
 */
export const useDocumentStore = create<DocumentState>()(
  persist(
    (set, get) => ({
      // 初始状态
      id: generateId(),
      title: "加载中...",
      blocks: [],
      isSaving: false,
      isOnline: false,
      onlineUsers: [],
      typingUsers: {},
      selectedBlockId: null,
      focusedBlockId: null,
      pendingConflicts: [],
      showConflictDialog: false,
      history: [],
      historyIndex: -1,
      canUndo: false,
      canRedo: false,
      versions: [],
      isLoadingVersions: false,

      // 分块加载状态
      chunkedLoading: {
        loadedChunks: new Map(),
        totalBlocks: 0,
        chunkSize: 50, // 每次加载 50 个块
        lastSyncTime: null,
Date: new Date(),
      },
      isLoadingChunk: false,

      // ========== 方法实现 ==========
      fetchDocument: async (id: string) => {
        try {
          const res = await api.get(`${API_URL}/${id}`);
          set({
            id: res.data.id,
            title: res.data.title,
            blocks: res.data.blocks?.length > 0 ? res.data.blocks : [
              { id: "init-1", type: "h1", content: res.data.title || "未命名文档" }
            ],
            isOnline: true,
            chunkedLoading: {
              loadedChunks: new Map([[0, res.data.blocks || []]]),
              totalBlocks: res.data.blocks?.length || 0,
              chunkSize: 50,
              lastSyncTime: new Date(),
            },
          });
        } catch (error) {
          console.warn("无法从云端获取文档，正在回退至本地缓存");
          set({ isOnline: false });
        }
      },

      // 分块加载文档块（大文件优化）
      loadBlocksChunked: async (offset: number, limit: number) => {
        const { id, chunkedLoading } = get();
        set({ isLoadingChunk: true });

        try {
          const res = await api.get(`${API_URL}/${id}/blocks`, {
            params: { offset, limit, includeTotal: true },
          });

          const newChunks = new Map(chunkedLoading.loadedChunks);
          newChunks.set(offset, res.data.blocks);

          set({
            chunkedLoading: {
              ...chunkedLoading,
              loadedChunks: newChunks,
              totalBlocks: res.data.total || chunkedLoading.totalBlocks,
            },
            // 合并所有已加载的块到主 blocks 数组
            blocks: Array.from(newChunks.values()).flat(),
            isLoadingChunk: false,
          });
        } catch (error) {
          console.error("分块加载失败:", error);
          set({ isLoadingChunk: false });
        }
      },

      // 增量同步块
      syncBlocks: async () => {
        const { id, chunkedLoading } = get();
        if (!chunkedLoading.lastSyncTime) return;

        try {
          const res = await api.get(`${API_URL}/${id}/sync`, {
            params: { since: chunkedLoading.lastSyncTime.toISOString() },
          });

          if (res.data.updated?.length > 0 || res.data.created?.length > 0) {
            // 更新本地块
            const currentBlocks = get().blocks;
            const updatedMap = new Map(currentBlocks.map(b => [b.id, b]));

            // 更新已存在的块
            res.data.updated?.forEach((block: Block) => {
              updatedMap.set(block.id, block);
            });

            // 添加新块
            res.data.created?.forEach((block: Block) => {
              if (!updatedMap.has(block.id)) {
                updatedMap.set(block.id, block);
              }
            });

            const newBlocks = Array.from(updatedMap.values()).sort((a, b) => (a.order || 0) - (b.order || 0));

            set({
              blocks: newBlocks,
              chunkedLoading: {
                ...chunkedLoading,
                totalBlocks: newBlocks.length,
                lastSyncTime: new Date(),
              },
            });
          }
        } catch (error) {
          console.error("增量同步失败:", error);
        }
      },

      // 检查是否还有更多块可以加载
      hasMoreBlocks: () => {
        const { chunkedLoading } = get();
        const currentLoaded = Array.from(chunkedLoading.loadedChunks.values()).flat().length;
        return currentLoaded < chunkedLoading.totalBlocks;
      },

      // 添加块
      addBlock: (type: Block["type"], afterId?: string) => {
        get().saveHistory();
        set((state) => {
          const newBlock: Block = { id: Math.random().toString(36).substr(2, 9), type, content: "" };
          if (!afterId) return { blocks: [...state.blocks, newBlock] };

          const index = state.blocks.findIndex((b) => b.id === afterId);
          const newBlocks = [...state.blocks];
          newBlocks.splice(index + 1, 0, newBlock);
          return { blocks: newBlocks };
        });
      },

      // 更新块（本地优化：不直接发送请求）
      updateBlock: (id: string, content: string, properties?: Block["properties"], type?: Block["type"]) => {
        get().saveHistory();
        set((state) => ({
          blocks: state.blocks.map((b) =>
            b.id === id
              ? { ...b, content, type: type || b.type, properties: { ...b.properties, ...properties } }
              : b
          ),
        }));
      },

      // 远程更新块（触发 API 请求）
      remoteUpdateBlock: (id: string, content: string, properties?: Block["properties"], type?: Block["type"]) =>
        set((state) => ({
          blocks: state.blocks.map((b) =>
            b.id === id
              ? { ...b, content, type: type || b.type, properties: { ...b.properties, ...properties } }
              : b
          ),
        })),

      // 批量更新块（优化大量块更新场景）
      updateBlocksBatch: async (blocks: { id: string; content: string; type?: string; properties?: any }[]) => {
        const { id } = get();
        try {
          // 先更新本地状态
          set((state) => ({
            blocks: state.blocks.map((b) => {
              const update = blocks.find(u => u.id === b.id);
              if (update) {
                return {
                  ...b,
                  content: update.content,
                  type: (update.type as BlockType) || b.type,
                  properties: { ...b.properties, ...update.properties },
                };
              }
              return b;
            }),
          }));

          // 然后发送批量更新请求
          await api.post(`${API_URL}/${id}/blocks/batch`, { blocks });
        } catch (error) {
          console.error("批量更新失败:", error);
        }
      },

      // 块转化：将块转换为另一种类型
      transformBlock: (id: string, newType: Block["type"]) => {
        get().saveHistory();
        set((state) => ({
          blocks: state.blocks.map((b) =>
            b.id === id ? { ...b, type: newType } : b
          ),
        }));
      },

      // 块缩进：增加或减少缩进（通过调整父级或层级）
      indentBlock: (id: string, direction: 'increase' | 'decrease') => {
        get().saveHistory();
        set((state) => {
          const index = state.blocks.findIndex((b) => b.id === id);
          if (index === -1) return state;

          const newBlocks = [...state.blocks];
          // 简单实现：缩进只是调整位置
          if (direction === 'increase' && index > 0) {
            // 将当前块移到前一个块的位置之后（作为子级）
            const [current] = newBlocks.splice(index, 1);
            newBlocks.splice(index, 0, current);
          } else if (direction === 'decrease' && index > 0) {
            // 将当前块移到前一个块之前（提升级别）
            const [current] = newBlocks.splice(index, 1);
            newBlocks.splice(index - 1, 0, current);
          }
          return { blocks: newBlocks };
        });
      },

      // 保存历史记录（限制 50 条）
      saveHistory: () => {
        const { blocks, title, history, historyIndex } = get();
        const newHistory = history.slice(0, historyIndex + 1);
        newHistory.push({ blocks: JSON.parse(JSON.stringify(blocks)), title });

        // 限制历史记录长度
        if (newHistory.length > 50) {
          newHistory.shift();
        }

        set({
          history: newHistory,
          historyIndex: newHistory.length - 1,
          canUndo: newHistory.length > 1,
          canRedo: false
        });
      },

      // 撤销
      undo: () => {
        const { history, historyIndex } = get();
        if (historyIndex > 0) {
          const prevState = history[historyIndex - 1];
          set({
            blocks: prevState.blocks,
            title: prevState.title,
            historyIndex: historyIndex - 1,
            canUndo: historyIndex - 1 > 0,
            canRedo: true
          });
        }
      },

      // 重做
      redo: () => {
        const { history, historyIndex } = get();
        if (historyIndex < history.length - 1) {
          const nextState = history[historyIndex + 1];
          set({
            blocks: nextState.blocks,
            title: nextState.title,
            historyIndex: historyIndex + 1,
            canUndo: true,
            canRedo: historyIndex + 1 < history.length - 1
          });
        }
      },
    }),
    {
      name: "fastdoc-main-storage",
      // 使用 localStorage 持久化
      storage: createJSONStorage(() => localStorage),
      // 只持久化部分状态（优化性能）
      partialize: (state) => ({
        title: state.title,
        blocks: state.blocks,
        id: state.id
      }),
    }
  )
);
```

**Zustand 特点总结**：
1. **轻量级**: 无 Provider 嵌套，直接调用 hooks
2. **TypeScript 友好**: 完整的类型推导
3. **中间件支持**:
   - `persist`: 持久化到 localStorage
   - `createJSONStorage`: JSON 序列化
   - `partialize`: 只持久化必要字段
4. **无依赖状态**: 响应式更新，不触发重渲染

### 分块加载优化

#### 问题背景

当文档包含大量块（如 1000+ 个块）时，一次性加载会导致：
- **网络延迟**: 大数据传输时间长
- **内存占用**: 前端需要解析大量 JSON
- **渲染阻塞**: 需要渲染所有块，造成卡顿

#### 解决方案

**前端分块加载**（`documentStore.ts`）：

```typescript
interface ChunkedLoadingState {
  loadedChunks: Map<number, Block[]>; // 已加载的块分块
  totalBlocks: number;               // 总块数
  chunkSize: number;                // 块大小（默认 50）
  lastSyncTime: Date | null;       // 上次同步时间
}

// 初始化加载
fetchDocument: async (id: string) => {
  set({
    chunkedLoading: {
      loadedChunks: new Map([[0, blocks]]), // 第一个分块包含所有块
      totalBlocks: blocks.length,
      chunkSize: 50,
    },
  });
};

// 滚动加载更多块
loadBlocksChunked: async (offset: number, limit: number) => {
  const res = await api.get(`${API_URL}/${id}/blocks`, {
    params: { offset, limit },
  });

  const newChunks = new Map(chunkedLoading.loadedChunks);
  newChunks.set(offset, res.data.blocks);

  set({
    chunkedLoading: {
      ...chunkedLoading,
      loadedChunks: newChunks,
      totalBlocks: res.data.total,
    },
    // 合并所有已加载的块到主 blocks 数组
    blocks: Array.from(newChunks.values()).flat(),
  });
};

// 检查是否还有更多块
hasMoreBlocks: () => {
  const currentLoaded = Array.from(chunkedLoading.loadedChunks.values()).flat().length;
  return currentLoaded < chunkedLoading.totalBlocks;
};
```

**后端分块加载**（`documents.service.ts`）：

```typescript
@Injectable()
export class DocumentsService {
  /**
   * 分块加载文档块（大文件优化）
   * @param docId 文档ID
   * @param options 分页选项
   */
  async findBlocksChunked(docId: string, options: ChunkLoadOptions) {
    const { offset, limit, includeTotal = true } = options;
    const cacheKey = getBlocksCacheKey(docId, offset, limit);

    // 尝试从缓存获取
    const cached = await this.cacheService.get(cacheKey, { ttl: 60 });
    if (cached) {
      return cached;
    }

    // 查询数据库
    const [blocks, total] = await this.blockRepo.findAndCount({
      where: { document: { id: docId } },
      order: { order: 'ASC' },
      skip: offset,
      take: limit,
    });

    // 缓存结果（TTL: 30s，较短因为块会频繁更新）
    await this.cacheService.set(cacheKey, { blocks, total }, { ttl: 30 });

    return includeTotal ? { blocks, total } : { blocks };
  }
}
```

```typescript
 * @param docId 文档ID

  });

  return { updated: blocks, created: newBlocks };
}
```

```typescript
// 保存历史记录
saveHistory: () => {
  const { blocks, title, history, historyIndex } = get();
  const newHistory = history.slice(0, historyIndex +1);
  newHistory.push({ blocks: JSON.parse(JSON.stringify(blocks)), title });

  // 限制历史记录长度（避免内存溢出）
  if (newHistory.length > 50) {
    newHistory.shift();
  }

  set({
    history: newHistory,
    historyIndex: newHistory.length - 1,
    canUndo: newHistory.length > 1,
    canRedo: false
  });
};

// 撤销
undo: () => {
  const { history, historyIndex } = get();
  if (historyIndex > 0) {
    const prevState = history[historyIndex - 1];
    set({
      blocks: prevState.blocks,
      title: prevState.title,
      historyIndex: historyIndex - 1,
      canUndo: historyIndex - 1 > 0,
      canRedo: true
    });
  }
};

// 重做
redo: () => {
  const { history, historyIndex } = get();
  if (historyIndex < history.length - 1) {
    const nextState = history[historyIndex + 1];
    set({
      blocks: nextState.blocks,
      title: nextState.title,
      historyIndex: historyIndex + 1,
      canUndo: true,
      canRedo: historyIndex + 1 < history.length - 1
    });
  }
};
```

**撤销重做优化点**：
1. **状态快照**: 使用 `JSON.parse(JSON.stringify())` 深拷贝 blocks
2. **历史限制**: 最多保存 50 条历史，避免内存问题
3. **索引管理**: 使用 `historyIndex` 追踪当前位置
4. **按钮状态**: 根据 `canUndo`/`canRedo` 启用/禁用按钮

#### Redis 缓存实现

```typescript
@Injectable()
export class CacheService {
  constructor(
    private readonly redis: Redis,
  ) {}

  /**
   * 获取缓存
   * @param key 缓存键
   * @param options 缓存选项
   */
  async get(key: string, options: { prefix?: string; ttl?: number } = {}): Promise<T | null> {
    const fullKey = options.prefix ? `${options.prefix}${key}` : key;
    const cached = await this.redis.get(fullKey);
    if (!cached) return null;

    return JSON.parse(cached);
  }

  /**
   * 设置缓存
   * @param key 缓存键
   * @param value 缓存值
   * @param options 缓存选项
   */
  async set(key: string, value: T, options: { prefix?: string; ttl?: number } = {}): Promise<void> {
    const fullKey = options.prefix ? `${options.prefix}${key}` : key;
    const serialized = JSON.stringify(value);

    if (options.ttl) {
      await this.redis.setex(fullKey, serialized, options.ttl);
    } else {
      await this.redis.set(fullKey, serialized);
    }
  }

  /**
   * 删除缓存
   */
  async delete(key: string, options: { prefix?: string } = {}): Promise<void> {
    const fullKey = options.prefix ? `${options.prefix}${key}` : key;
    await this.redis.del(fullKey);
  }

  /**
   * 按模式删除缓存
   */
  async deleteByPattern(pattern: string, options: { prefix?:?: string } = {}): Promise<void> {
    const fullPattern = options.prefix ? `${options.prefix}${pattern}` : pattern;
    const keys = await this.redis.keys(fullPattern);
    if (keys.length > 0) {
      await this.redis.del(keys);
    }
  }
}
```

```typescript
export const useGameStore = create<GameState>((set) => ({
  // 初始状态
  players: [],
  localPlayerId: '',
  currentTurn: 0,
  direction: 1,
  deck: [],
  drawPile: [],
  discardPile: [],
  playerHands: [[], [], [], [], [], []], // 最多 4 玩家
  isReversed: false,
  isColorMatch: false,
  currentPlayerMustDraw: false,
  hasCalledUno: false,

  initGame: (players: Player[]) => {
    set({
      players,
      localPlayerId: players[0].id,
      deck: [], // 在游戏开始时初始化
      drawPile: [],
      discardPile: [],
      playerHands: players.map(() => []), // 清空手牌
      currentTurn: 0,
      hasCalledUno: false,
    });
  },

  startGame: () => {
    const { players } = get();

    // 创建 UNO 卡组（每种颜色 0-9 两张）
    const deck = createDeck();

    // 洗牌给每个玩家
    const playerHands: Card[][] = [];
    players.forEach((player, index) => {
      const hand = deck.splice(0, 7); // 每人发 7 张牌
      playerHands.push(hand);
    });

      deck,
      playerHands,
      currentTurn: 0,
      currentPlayerMustDraw: players[0].id === players[0].id, // 第一局开始时，玩家 0 摸牌

    socket.emit('game-start', { gameId, deck, playerHands });

    // 验证是否轮到该玩家
    if (currentTurn !== playerIndex) return;

    // 从手牌中移除卡牌
    const hand = [...playerHands[playerIndex]];
    const card = hand.splice(cardIndex, 1)[0];

    // 添加到弃牌堆
    const discardPile = [...get().discardPile, card];

    set({
      playerHands: playerHands.map((h, i) => i === playerIndex ? hand : h),

    if (!card) return;

      deck,
      playerHands: playerHands.map((h, i) => i === playerIndex ? hand : h),

    socket.emit('card-drawn', { gameId, playerIndex, card });

    if (hand.length !== 1) return;

    audioManager.play(SoundEffect.SHOUT_UNO);

    // 同步到服务器
    socket.emit('uno-called', { gameId, playerIndex });
  },
}));
```

```typescript
  isHand: boolean,
  isReverse: boolean

  const angleStep = 0.1; // 弧度

  if (isHand) {
    // 手牌呈扇形展开
    const angle = cardIndex * angleStep;
    const radius = 3;

      Math.cos(angle) * radius,
      Math.sin(angle) * radius,
      0,
    ];
  } else {
    // 弃牌堆呈线形排列
    const x = cardIndex * spacing;
    return [x, 0, 0];
  }
```

**文件位置**: `webenv-os/src/lib/filesystem/zenfs.ts`

```typescript
import { configure, FS, configureFs } from '@zenfs/core';
import IndexedDB from 'indexedfs';
import { MemoryFS } from 'memoryfs';

class VirtualFileSystem {
  private fs: FS | null = null;
  private isInitialized = false;

  /**
   * 初始化虚拟文件系统
   * @param useIndexedDB 是否使用 IndexedDB（持久化）或 MemoryFS（临时）
   */
  async initialize(useIndexedDB = true) {
    if (this.isInitialized) return;

    if (useIndexedDB) {
      // 使用 IndexedDB 后端
      const idbfs = new IndexedDB('webenv-fs', {
        storeName: 'files',
      });

      await configure({
        mounts: {
          '/': idbfs,
        },
      });
    } else {
      // 使用内存后端（临时使用）
      await configure({
        mounts: {
          '/': new MemoryFS(),
        },
      });
    }

    this.fs = await configureFs();
    this.isInitialized = true;
  }

  /**
   * 写入文件
   */
  async writeFile(path: string, content: string): Promise<void> {
    if (!this.fs) await this.initialize();

    // 确保目录存在
    const dir = path.dirname(path);
    await this.fs.promises.mkdir(dir, { recursive: true });

    // 写入文件
    await this.fs.promises.writeFile(path, content);
  }

  /**
   * 读取文件
   */
  async readFile(path: string): Promise<string> {
    if (!this.fs) await this.initialize();

    const content = await this.fs.promises.readFile(path, 'utf8');
    return content as string;
  }

  /**
   * 创建目录
   */
  async mkdir(path: string): Promise<void> {
    if (!this.fs) await this.initialize();
    await this.fs.promises.mkdir(path, { recursive: true });
  }

  /**
   * 列出目录内容
   */
  async readdir(path: string): Promise<string[]> {
    if (!this.fs) await this.initialize();

    const entries = await this.fs.promises.readdir(path);
    return entries as string[];
  }

  /**
   * 删除文件
   */
  async unlink(path: string): Promise<void> {
    if (!this.fs) await this.initialize();
    await this.fs.promises.unlink(path);
  }

  /**
   * 获取文件信息
   */
  async stat(path: string): Promise<{
    isFile: boolean;
    isDirectory: boolean;
    size: number;
  modifiedAt: Date;
  }> {
    if (!this.fs) await this.initialize();

    const stats = await this.fs.promises.stat(path);
    return {
      isFile: stats.isFile(),
      isDirectory: stats.isDirectory(),
      size: stats.size,
      modifiedAt: stats.mtime,
    };
  }
}
```
