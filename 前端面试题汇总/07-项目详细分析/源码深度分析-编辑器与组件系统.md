#### 2.1.3 documentStore 状态管理

```typescript
// frontend/src/store/documentStore.ts

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface DocumentState {
  // 文档数据
  documents: Document[];
  currentDocumentId: string | null;
  blocks: Block[];

  // 分块加载
  blockChunks: Map<number, Block[]>;
  chunkSize: number;
  loadedChunks: Set<number>;
  totalBlockCount: number;

  // 历史记录
  history: Block[][];
  historyIndex: number;
  maxHistorySize: number;

  // UI 状态
  isLoading: boolean;
  isSaving: boolean;
  slashMenuVisible: boolean;
  slashMenuPosition: { x: number; y: number };
  activeBlockId: string | null;
  selectedBlockIds: string[];

  // 协作状态
  collaborators: User[];
  myUserId: string;

  // 文档操作
  createDocument: (title: string, parentId?: string) => Promise<string>;
  loadDocument: (id: string) => Promise<void>;
  saveDocument: () => Promise<void>;

  // 块操作
  addBlock: (type: BlockType, afterBlockId?: string) => void;
  updateBlock: (blockId: string, changes: Partial<Block>) => void;
  deleteBlock: (blockId: string) => void;
  moveBlock: (blockId: string, afterBlockId: string) => void;
  transformBlock: (blockId: string, newType: BlockType) => void;

  // 分块加载
  loadBlocksChunked: (offset: number, limit?: number) => Promise<void>;
  syncBlocksSince: (timestamp: number) => Promise<void>;

  // 历史操作
  undo: () => void;
  redo: () => void;
  saveHistory: () => void;

  // UI 操作
  setSlashMenuVisible: (visible: boolean, position?: { x: number; y: number }) => void;
  setActiveBlockId: (id: string | null) => void;
  setSelectedBlockIds: (ids: string[]) => void;
}

export const useDocumentStore = create<DocumentState>()(
  persist(
    (set, get) => ({
      // 初始状态
      documents: [],
      currentDocumentId: null,
      blocks: [],
      blockChunks: new Map(),
      chunkSize: 50,
      loadedChunks: new Set(),
      totalBlockCount: 0,
      history: [],
      historyIndex: -1,
      maxHistorySize: 50,
      isLoading: false,
      isSaving: false,
      slashMenuVisible: false,
      slashMenuPosition: { x: 0, y: 0 },
      activeBlockId: null,
      selectedBlockIds: [],
      collaborators: [],
      myUserId: '',

      // 创建文档
      createDocument: async (title, parentId) => {
        set({ isLoading: true });
        try {
          const response = await api.post(API_URL, { title, parentId, type: 'document', blocks: [] });
          const docId = response.data.id;

          // 创建初始块
          const initialBlock: Block = {
            id: generateId(),
            type: 'text',
            content: '',
            order: 0
          };

          set({
            currentDocumentId: docId,
            blocks: [initialBlock],
            blockChunks: new Map([[0, [initialBlock]]]),
            loadedChunks: new Set([0]),
            totalBlockCount: 1,
            history: [[initialBlock]],
            historyIndex: 0
          });

          return docId;
        } finally {
          set({ isLoading: false });
        }
      },

      // 加载文档 (分块加载)
      loadDocument: async (id) => {
        set({ isLoading: true, currentDocumentId: id });
        try {
          // 先加载元数据
          const docResponse = await api.get(`${API_URL}/${id}`);
          const document = docResponse.data;

          // 加载第一块数据
          await get().loadBlocksChunked(0, get().chunkSize);

          set({
            totalBlockCount: document.totalBlocks || get().blockChunks.get(0)?.length || 0
          });
        } finally {
          set({ isLoading: false });
        }
      },

      // 分块加载
      loadBlocksChunked: async (offset, limit = 50) => {
        const { currentDocumentId, chunkSize, blockChunks, loadedChunks } = get();
        if (!currentDocumentId) return;

        const chunkIndex = Math.floor(offset / chunkSize);

        // 检查是否已加载
        if (loadedChunks.has(chunkIndex)) {
          return;
        }

        set({ isLoading: true });
        try {
          const response = await api.get(`${API_URL}/${currentDocumentId}/blocks`, {
            params: { offset, limit, includeTotal: chunkIndex === 0 }
          });

          const { blocks, total } = response.data;

          set({
            blockChunks: new Map(blockChunks).set(chunkIndex, blocks),
            loadedChunks: new Set(loadedChunks).add(chunkIndex),
            totalBlockCount: chunkIndex === 0 ? total : get().totalBlockCount
          });
        } finally {
          set({ isLoading: false });
        }
      },

      // 增量同步
      syncBlocksSince: async (timestamp) => {
        const { currentDocumentId, blocks } = get();
        if (!currentDocumentId) return;

        const response = await api.get(`${API_URL}/${currentDocumentId}/sync`, {
          params: { since: timestamp }
        });

        const { updated, created, deleted } = response.data;

        set((state) => {
          let newBlocks = [...blocks];

          // 应用更新
          updated.forEach((block: Block) => {
            const index = newBlocks.findIndex(b => b.id === block.id);
            if (index !== -1) {
              newBlocks[index] = block;
            }
          });

          // 添加新块
          created.forEach((block: Block) => {
            newBlocks.push(block);
          });

          // 删除块
          deleted.forEach((blockId: string) => {
            newBlocks = newBlocks.filter(b => b.id !== blockId);
          });

          return { blocks: newBlocks };
        });
      },

      // 添加块
      addBlock: (type, afterBlockId) => {
        const { blocks } = get();
        const insertIndex = afterBlockId
          ? blocks.findIndex(b => b.id === afterBlockId) + 1
          : blocks.length;

        const newBlock: Block = {
          id: generateId(),
          type,
          content: '',
          order: insertIndex
        };

        const newBlocks = [...blocks];
        newBlocks.splice(insertIndex, 0, newBlock);

        set({ blocks: newBlocks, activeBlockId: newBlock.id });

        // 保存历史
        get().saveHistory();

        // 通过 Socket 广播
        if (get().currentDocumentId) {
          socketClient.emit('block:add', {
            documentId: get().currentDocumentId,
            block: newBlock,
            afterBlockId
          });
        }
      },

      // 更新块
      updateBlock: (blockId, changes) => {
        const { blocks, activeBlockId } = get();

        const newBlocks = blocks.map(block =>
          block.id === blockId ? { ...block, ...changes } : block
        );

        set({ blocks: newBlocks });

        // 如果更新的是活跃块，更新 activeBlockId 引用的块
        if (activeBlockId === blockId) {
          set({ activeBlockId: blockId });
        }

        // 防抖保存
        debouncedSave(get().saveDocument);

        // 通过 Socket 广播
        if (get().currentDocumentId) {
          socketClient.emit('block:update', {
            documentId: get().currentDocumentId,
            blockId,
            changes
          });
        }
      },

      // 删除块
      deleteBlock: (blockId) => {
        const { blocks } = get();
        const newBlocks = blocks.filter(b => b.id !== blockId);

        set({ blocks: newBlocks });

        // 保存历史
        get().saveHistory();

        // 通过 Socket 广播
        if (get().currentDocumentId) {
          socketClient.emit('block:delete', {
            documentId: get().currentDocumentId,
            blockId
          });
        }
      },

      // 移动块
      moveBlock: (blockId, afterBlockId) => {
        const { blocks } = get();
        const blockIndex = blocks.findIndex(b => b.id === blockId);
        const afterIndex = afterBlockId
          ? blocks.findIndex(b => b.id === afterBlockId)
          : blocks.length - 1;

        if (blockIndex === -1 || blockIndex === afterIndex) return;

        const [block] = blocks.splice(blockIndex, 1);
        blocks.splice(afterIndex + 1, 0, block);

        set({ blocks });

        // 保存历史
        get().saveHistory();
      },

      // 转换块类型
      transformBlock: (blockId, newType) => {
        get().updateBlock(blockId, { type: newType });
      },

      // 保存历史
      saveHistory: () => {
        const { blocks, history, historyIndex, maxHistorySize } = get();

        // 如果当前不在历史栈顶端，删除后续历史
        const newHistory = historyIndex < history.length - 1
          ? history.slice(0, historyIndex + 1)
          : history;

        // 深拷贝当前状态
        const blocksCopy = JSON.parse(JSON.stringify(blocks));

        // 添加新历史
        newHistory.push(blocksCopy);

        // 限制历史大小
        if (newHistory.length > maxHistorySize) {
          newHistory.shift();
        }

        set({
          history: newHistory,
          historyIndex: newHistory.length - 1
        });
      },

      // 撤销
      undo: () => {
        const { history, historyIndex } = get();

        if (historyIndex <= 0) return;

        const newIndex = historyIndex - 1;
        const blocks = JSON.parse(JSON.stringify(history[newIndex]));

        set({
          blocks,
          historyIndex: newIndex
        });
      },

      // 重做
      redo: () => {
        const { history, historyIndex } = get();

        if (historyIndex >= history.length - 1) return;

        const newIndex = historyIndex + 1;
        const blocks = JSON.parse(JSON.stringify(history[newIndex]));

        set({
          blocks,
          historyIndex: newIndex
        });
      },

      // 保存文档
      saveDocument: async () => {
        const { currentDocumentId, blocks, isSaving } = get();

        if (!currentDocumentId || isSaving) return;

        set({ isSaving: true });
        try {
          await api.put(`${API_URL}/${currentDocumentId}`, { blocks });
        } finally {
          set({ isSaving: false });
        }
      },

      // 设置快捷菜单
      setSlashMenuVisible: (visible, position) => {
        set({
          slashMenuVisible: visible,
          slashMenuPosition: position || { x: 0, y: 0 }
        });
      },

      // 设置活跃块
      setActiveBlockId: (id) => {
        set({ activeBlockId: id });
      },

      // 设置选中的块
      setSelectedBlockIds: (ids) => {
        set({ selectedBlockIds: ids });
      }
    }),
    {
      name: 'document-storage',
      partialize: (state) => ({
        documents: state.documents,
        currentDocumentId: state.currentDocumentId,
        myUserId: state.myUserId
      })
    }
  )
);

// 防抖保存 (1秒)
const debouncedSave = debounce((fn: () => Promise<void>) => fn(), 1000);
```

```typescript
import Prism from 'prismjs';
import 'prismjs/themes/prism-tomorrow.css';
// 导入语言支持
import 'prismjs/components/prism-javascript';

import 'prismjs/components/prism-php';
import 'prismjs/components/prism-swift';
import 'prismjs/components/prism-kotlin';
import 'prismjs/components/prism-scala';
import 'prismjs/components/prism-haskell';
import 'prismjs/components/prism-erlang';
import 'prismjs/components/prism-elixir';
import 'prismjs/components/prism-sql';
import 'prismjs/components/prism-bash';
import 'prismjs/components/prism-powershell';
import 'prismjs/components/prism-batch';
import 'prismjs/components/prism-json';
import 'prismjs/components/prism-yaml';
import 'prismjs/components/prism-markdown';
import 'prismjs/components/prism-docker';
import 'prismjs/components/prism-graphql';
import 'prismjs/components/prism-latex';
import 'prismjs/components/prism-nginx';
import 'prismjs/components/prism-apache';

// 语法高亮缓存
const highlightCache = new WeakMap<CodeBlockProps, string>();

const CodeBlock: React.FC<CodeBlockProps> = ({
  code,
  language = 'javascript',
  theme = 'dark',
  showLineNumbers = true,
  showCopyButton = true,
  showLanguageSelector = true,
  maxHeight,
  editable = false,
  onChange,
  className,
  style
}) => {
  const codeRef = useRef<HTMLPreElement>(null);
  const [isEditing, setIsEditing] = useState(false);
  const [copied, setCopied] = useState(false);
  const [selectedLang, setSelectedLang] = useState(language);

  // 语法高亮
  const highlightedCode = useMemo(() => {
    const key = { code, language: selectedLang };
    const cached = highlightCache.get(key);
    if (cached) return cached;

    // 获取语言语法
    const grammar = Prism.languages[selectedLang] || Prism.languages.plaintext;

    // 执行高亮
    const result = Prism.highlight(code, grammar, selectedLang);

    // 缓存结果
    highlightCache.set(key, result);

    return result;
  }, [code, selectedLang]);

  // 复制到剪贴板
  const handleCopy = useCallback(async () => {
    await navigator.clipboard.writeText(code);
    setCopied(true);

    // 2秒后重置状态
    setTimeout(() => setCopied(false), 2000);
  }, [code]);

  // 处理代码变更
  const handleChange = useCallback((newCode: string) => {
    onChange?.(newCode);
  }, [onChange]);

  // 分割代码为行
  const lines = code.split('\n');

  return (
    <div
      className={clsx(
        'code-block',
        `theme-${theme}`,
        editable && 'editable',
        className
      )}
      style={style}
    >
      {/* 语言选择器和复制按钮 */}
      {(showLanguageSelector || showCopyButton) && (
        <div className="code-block-toolbar">
          {showLanguageSelector && (
            <select
              value={selectedLang}
              onChange={(e) => setSelectedLang(e.target.value)}
              className="language-selector"
            >
              {SUPPORTED_LANGUAGES.map(lang => (
                <option key={lang.id} value={lang.id}>
                  {lang.name}
                </option>
              ))}
            </select>
          )}

          {showCopyButton && (
            <button
              onClick={handleCopy}
              className="copy-button"
            >
              {copied ? '已复制' : '复制'}
            </button>
          )}
        </div>
      )}

      {/* 代码内容 */}
      {isEditing && editable ? (
        <textarea
          value={code}
          onChange={(e) => handleChange(e.target.value)}
          onBlur={() => setIsEditing(false)}
          autoFocus
          className="code-textarea"
          spellCheck={false}
        />
      ) : (
        <pre
          ref={codeRef}
          className="code-pre"
          style={{ maxHeight }}
        >
          {showLineNumbers ? (
            <div className="code-with-line-numbers">
              <div className="line-numbers">
                {lines.map((_, index) => (
                  <span key={index} className="line-number">
                    {index + 1}
                  </span>
                ))}
              </div>
              <code
                dangerouslySetInnerHTML={{ __html: highlightedCode }}
              />
            </div>
          ) : (
            <code dangerouslySetInnerHTML={{ __html: highlightedCode }} />
          )}
        </pre>
      )}

      {editable && (
        <button
          onClick={() => setIsEditing(true)}
          className="edit-button"
        >
          编辑
        </button>
      )}
    </div>
  );
};
```

#### 2.3.2 表格组件实现

```typescript
// frontend/src/components/editor/TableBlock.tsx

const TableBlock: React.FC<TableBlockProps> = ({
  data,
  caption,
  editable = true,
  onChange,
  showHeaders = true,
  showRowNumbers = false,
  fixedHeader = true,
  minColumnWidth = 50,
  maxColumnWidth = 400,
  className
}) => {
  const [selectedCell, setSelectedCell] = useState<{ row: number; col: number } | null>(null);
  const [editingCell, setEditingCell] = useState<{ row: number; col: number } | null>(null);
  const [resizingColumn, setResizingColumn] = useState<string | null>(null);
  const [dragStart, setDragStart] = useState<{ x: number; colId: string } | null>(null);

  const tableRef = useRef<HTMLTableElement>(null);
  const { columns, rows } = data;

  // 添加行
  const addRow = useCallback((afterRowIndex?: number) => {
    const newCells = columns.map(col => ({
      id: generateId(),
      content: ''
    }));

    const newRow: TableRow = {
      id: generateId(),
      cells: newCells
    };

    const newRows = afterRowIndex !== undefined
      ? [...rows.slice(0, afterRowIndex + 1), newRow, ...rows.slice(afterRowIndex + 1)]
      : [...rows, newRow];

    onChange?.({ ...data, rows: newRows });
  }, [columns, rows, data, onChange]);

  // 添加列
  const addColumn = useCallback((afterColIndex?: number) => {
    const newColumn: TableColumn = {
      id: generateId(),
      width: 150
    };

    const newColumns = afterColIndex !== undefined
      ? [...columns.slice(0, afterColIndex + 1), newColumn, ...columns.slice(afterColIndex + 1)]
      : [...columns, newColumn];

    const newRows = rows.map(row => ({
      ...row,
      cells: [
        ...row.cells.slice(0, (afterColIndex ?? -1) + 1),
        { id: generateId(), content: '' },
        ...row.cells.slice((afterColIndex ?? -1) + 1)
      ]
    }));

    onChange?.({ columns: newColumns, rows: newRows });
  }, [columns, rows, data, onChange]);

  // 删除行
  const deleteRow = useCallback((rowIndex: number) => {
    if (rows.length <= 1) return;

    const newRows = rows.filter((_, index) => index !== rowIndex);
    onChange?.({ ...data, rows: newRows });
  }, [rows, data, onChange]);

  // 删除列
  const deleteColumn = useCallback((colIndex: number) => {
    if (columns.length <= 1) return;

    const newColumns = columns.filter((_, index) => index !== colIndex);
    const newRows = rows.map(row => ({
      ...row,
      cells: row.cells.filter((_, index) => index !== colIndex)
    }));

    onChange?.({ columns: newColumns, rows: newRows });
  }, [columns, rows, data, onChange]);

  // 更新单元格内容
  const updateCell = useCallback((rowIndex: number, colIndex: number, content: string) => {
    const newRows = rows.map((row, rIdx) => {
      if (rIdx === rowIndex) {
        return {
          ...row,
          cells: row.cells.map((cell, cIdx) =>
            cIdx === colIndex ? { ...cell, content } : cell
          )
        };
      }
      return row;
    });

    onChange?.({ ...data, rows: newRows });
  }, [rows, data, onChange]);

  // 更新列宽
  const updateColumnWidth = useCallback((colId: string, width: number) => {
    const clampedWidth = Math.max(minColumnWidth, Math.min(maxColumnWidth, width));

    const newColumns = columns.map(col =>
      col.id === colId ? { ...col, width: clampedWidth } : col
    );

    onChange?.({ ...data, columns: newColumns });
  }, [columns, data, onChange, minColumnWidth, maxColumnWidth]);

  // 处理列宽拖拽开始
  const handleResizeStart = useCallback((e: React.MouseEvent, colId: string) => {
    setResizingColumn(colId);
    setDragStart({ x: e.clientX, colId });
  }, []);

  // 处理列宽拖拽
  useEffect(() => {
    if (!resizingColumn || !dragStart) return;

    const handleMouseMove = (e: MouseEvent) => {
      const deltaX = e.clientX - dragStart.x;
      const column = columns.find(col => col.id === dragStart.colId);
      if (column) {
        updateColumnWidth(dragStart.colId, (column.width || 150) + deltaX);
        setDragStart({ x: e.clientX, colId: dragStart.colId });
      }
    };

    const handleMouseUp = () => {
      setResizingColumn(null);
      setDragStart(null);
    };

    window.addEventListener('mousemove', handleMouseMove);
    window.addEventListener('mouseup', handleMouseUp);

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [resizingColumn, dragStart, columns, updateColumnWidth]);

  // 渲染表头
  const renderHeader = () => {
    if (!showHeaders) return null;

    return (
      <thead>
        <tr>
          {showRowNumbers && <th className="row-number-header">#</th>}
          {columns.map((col, colIndex) => (
            <th
              key={col.id}
              style={{ width: col.width }}
              className={clsx(
                'table-header',
                selectedCell?.col === colIndex && 'selected'
              )}
            >
              <CellEditor
                content={rows[0]?.cells[colIndex]?.content || ''}
                isEditing={editingCell?.row === 0 && editingCell?.col === colIndex}
                onChange={(content) => updateCell(0, colIndex, content)}
                onEditStart={() => setEditingCell({ row: 0, col: colIndex })}
                onEditEnd={() => setEditingCell(null)}
              />
              <div
                className="column-resizer"
                onMouseDown={(e) => handleResizeStart(e, col.id)}
              />
            </th>
          ))}
        </tr>
      </thead>
    );
  };

  // 渲染表格主体
  const renderBody = () => {
    const startIndex = showHeaders ? 1 : 0;
    const displayRows = rows.slice(startIndex);

    return (
      <tbody>
        {displayRows.map((row, rowIndex) => {
          const actualRowIndex = startIndex + rowIndex;

          return (
            <tr
              key={row.id}
              className={clsx(
                'table-row',
                selectedCell?.row === actualRowIndex && 'selected'
              )}
            >
              {showRowNumbers && (
                <td className="row-number-cell">{actualRowIndex}</td>
              )}
              {row.cells.map((cell, colIndex) => (
                <td
                  key={cell.id}
                  colSpan={cell.colspan}
                  rowSpan={cell.rowspan}
                  style={{
                    textAlign: columns[colIndex]?.align || 'left'
                  }}
                  onClick={() => setSelectedCell({ row: actualRowIndex, col: colIndex })}
                  onDoubleClick={() => {
                    if (editable) {
                      setEditingCell({ row: actualRowIndex, col: colIndex });
                    }
                  }}
                  className={clsx(
                    'table-cell',
                    selectedCell?.row === actualRowIndex &&
                    selectedCell?.col === colIndex && 'selected'
                  )}
                >
                  {editingCell?.row === actualRowIndex && editingCell?.col === colIndex ? (
                    <input
                      value={cell.content}
                      onChange={(e) => updateCell(actualRowIndex, colIndex, e.target.value)}
                      onBlur={() => setEditingCell(null)}
                      onKeyDown={(e) => {
                        if (e.key === 'Enter') {
                          e.preventDefault();
                          setEditingCell(null);
                        }
                      }}
                      autoFocus
                    />
                  ) : (
                    cell.content
                  )}
                </td>
              ))}
            </tr>
          );
        })}
      </tbody>
    );
  };

  return (
    <div className={clsx('table-block', className)} ref={tableRef}>
      {caption && <caption>{caption}</caption>}

      <table>
        {renderHeader()}
        {renderBody()}
      </table>

      {editable && (
        <div className="table-actions">
          <button onClick={() => addRow()}>添加行</button>
          <button onClick={() => addColumn()}>添加列</button>
          {selectedCell && (
            <>
              <button onClick={() => deleteRow(selectedCell.row)}>删除行</button>
              <button onClick={() => deleteColumn(selectedCell.col)}>删除列</button>
            </>
          )}
        </div>
      )}
    </div>
  );
};

// 单元格编辑器组件
const CellEditor: React.FC<{
  content: string;
  isEditing: boolean;
  onChange: (content: string) => void;
  onEditStart: () => void;
  onEditEnd: () => void;
}> = ({ content, isEditing, onChange, onEditStart, onEditEnd }) => {
  const [value, setValue] = useState(content);

  useEffect(() => {
    setValue(content);
  }, [content]);

  if (isEditing) {
    return (
      <input
        value={value}
        onChange={(e) => {
          setValue(e.target.value);
          onChange(e.target.value);
        }}
        onBlur={onEditEnd}
        onKeyDown={(e) => {
          if (e.key === 'Enter') {
            onEditEnd();
          }
        }}
        autoFocus
      />
    );
  }

  return (
    <div onClick={onEditStart} className="cell-editor">
      {content}
    </div>
  );
};
```

```typescript
// frontend/src/lib/socket.ts

import { io, Socket } from 'socket.io-client';

class SocketClient {
  private static instance: SocketClient;
  private socket: Socket | null = null;

  private constructor() {}

  public static getInstance(): SocketClient {
    if (!SocketClient.instance) {
      SocketClient.instance = new SocketClient();
    }
    return SocketClient.instance;
  }

  public connect(url: string, token?: string): Socket {
    if (this.socket?.connected) {
      return this.socket;
    }

    this.socket = io(url, {
      auth: { token },
      transports: ['websocket', 'polling'],
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionAttempts: 10
    });

    // 连接事件
    this.socket.on('connect', () => {
      console.log('Socket connected');
    });

    // 断开连接
    this.socket.on('disconnect', (reason) => {
      console.log('Socket disconnected:', reason);
    });

    // 错误处理
    this.socket.on('error', (error) => {
      console.error('Socket error:', error);
    });

    return this.socket;
  }

  public disconnect(): void {
    if (this.socket) {
      this.socket.disconnect();
      this.socket = null;
    }
  }

  public emit(event: string, data?: any): void {
    if (this.socket?.connected) {
      this.socket.emit(event, data);
    }
  }

  public on(event: string, callback: (...args: any[]) => void): void {
    if (this.socket) {
      this.socket.on(event, callback);
    }
  }

  public off(event: string, callback?: (...args: any[]) => void): void {
    if (this.socket) {
      this.socket.off(event, callback);
    }
  }

  public get socket(): Socket | null {
    return this.socket;
  }
}

export const socketClient = SocketClient.getInstance();
```

```typescript
  // 过滤掉自己的光标

  // 为每个用户分配颜色

      '#ff5252', '#e040fb', '#7c4dff', '#536dfe',
      '#448aff', '#40c4ff', '#18ffff', '#64ffda',
      '#69f0ae', '#b2ff59', '#eeff41', '#ffff00',
      '#ffd740', '#ffab40', '#ff6e40'

    let hash = 0;
    for (let i = 0; i < userId.length; i++) {
      hash = userId.charCodeAt(i) + ((hash << 5) - hash);
    }

    return colors[Math.abs(hash) % colors.length];

        <CollabCursor
          key={cursor.userId}
          cursor={cursor}
          color={getUserColor(cursor.userId)}
          ref={(el) => {
            if (el) {
              cursorRefs.current.set(cursor.userId, el);
            }
          }}
        />
      ))}
    </>
  );
};
```

```typescript
// frontend/src/lib/fs/VirtualFileSystemService.ts

import { promises as fs } from '@zenfs/core';

/**
 * 文件统计信息
 */
interface VirtualStats {
  size: number;
  isFile: boolean;
  isDirectory: boolean;
  mtimeMs: number;
  birthtimeMs: number;
}

/**
 * 目录项
 */
interface DirEntry {
  name: string;
  isFile: boolean;
  isDirectory: boolean;
}

/**
 * 虚拟文件系统服务
 */
export class VirtualFileSystemService {
  private static instance: VirtualFileSystemService;

  private constructor() {}

  /**
   * 获取单例实例
   */
  public static getInstance(): VirtualFileSystemService {
    if (!VirtualFileSystemService.instance) {
      VirtualFileSystemService.instance = new VirtualFileSystemService();
    }
    return VirtualFileSystemService.instance;
  }

  /**
   * 读取文件内容
   */
  async readFile(path: string, encoding: BufferEncoding = 'utf-8'): Promise<string> {
    try {
      const buffer = await fs.readFile(path);
      return buffer.toString(encoding);
    } catch (error) {
      console.error(`Failed to read file ${path}:`, error);
      throw new Error(`Failed to read file: ${path}`);
    }
  }

  /**
   * 写入文件内容
   */
  async writeFile(path: string, content: string, encoding: BufferEncoding = 'utf-8'): Promise<void> {
    try {
      // 确保父目录存在
      const dir = path.substring(0, path.lastIndexOf('/'));
      if (dir && !(await this.exists(dir))) {
        await fs.mkdir(dir, { recursive: true });
      }

      await fs.writeFile(path, content, encoding);

      // 同步到localStorage (元数据)
      await this.syncMetadata();
    } catch (error) {
      console.error(`Failed to write file ${path}:`, error);
      throw new Error(`Failed to write file: ${path}`);
    }
  }

  /**
   * 读取目录内容
   */
  async readdir(path: string): Promise<DirEntry[]> {
    try {
      const entries = await fs.readdir(path, { withFileTypes: true });

      return entries.map(entry => ({
        name: entry.name,
        isFile: entry.isFile(),
        isDirectory: entry.isDirectory()
      }));
    } catch (error) {
      console.error(`Failed to read directory ${path}:`, error);
      throw new Error(`Failed to read directory: ${path}`);
    }
  }

  /**
   * 获取文件统计信息
   */
  async stat(path: string): Promise<VirtualStats> {
    try {
      const stats = await fs.stat(path);

      return {
        size: stats.size,
        isFile: stats.isFile(),
        isDirectory: stats.isDirectory(),
        mtimeMs: stats.mtimeMs,
        birthtimeMs: stats.birthtimeMs || Date.now()
      };
    } catch (error) {
      console.error(`Failed to stat ${path}:`, error);
      throw new Error(`Failed to stat: ${path}`);
    }
  }

  /**
   * 检查路径是否存在
   */
  async exists(path: string): Promise<boolean> {
    try {
      await fs.access(path);
      return true;
    } catch {
      return false;
    }
  }

  /**
   * 创建目录
   */
  async mkdir(path: string, recursive: boolean = true): Promise<void> {
    try {
      await fs.mkdir(path, { recursive });
    } catch (error) {
      console.error(`Failed to create directory ${path}:`, error);
      throw new Error(`Failed to create directory: ${path}`);
    }
  }

  /**
   * 删除文件或目录
   */
  async rm(path: string, recursive: boolean = false): Promise<void> {
    try {
      await fs.rm(path, { recursive });
    } catch (error) {
      console.error(`Failed to remove ${path}:`, error);
      throw new Error(`Failed to remove: ${path}`);
    }
  }

  /**
   * 重命名文件或目录
   */
  async rename(oldPath: string, newPath: string): Promise<void> {
    try {
      await fs.rename(oldPath, newPath);

      // 同步到localStorage (元数据)
      await this.syncMetadata();
    } catch (error) {
      console.error(`Failed to rename ${oldPath} to ${newPath}:`, error);
      throw new Error(`Failed to rename`);
    }
  }

  /**
   * 复制文件
   */
  async copy(src: string, dest: string): Promise<void> {
    try {
      await fs.copyFile(src, dest);

      // 同步到localStorage (元数据)
      await this.syncMetadata();
    } catch (error) {
      console.error(`Failed to copy ${src} to ${dest}:`, error);
      throw new Error(`Failed to copy`);
    }
  }

  /**
   * 创建读取流
   */
  async createReadStream(path: string): Promise<ReadableStream<Uint8Array>> {
    try {
      const buffer = await fs.readFile(path);

      return new ReadableStream<Uint8Array>({
        start(controller) {
          controller.enqueue(buffer);
          controller.close();
        }
      });
    } catch (error) {
      console.error(`Failed to create read stream for ${path}:`, error);
      throw new Error(`Failed to create read stream: ${path}`);
    }
  }

  /**
   * 创建写入流
   */
  async createWriteStream(path: string): Promise<WritableStream<Uint8Array>> {
    try {
      const chunks: Uint8Array[] = [];

      return new WritableStream<Uint8Array>({
        write(chunk) {
          chunks.push(chunk);
        },
        async close() {
          // 合并所有chunks
          const totalLength = chunks.reduce((sum, chunk) => sum + chunk.length, 0);
          const merged = new Uint8Array(totalLength);
          let offset = 0;

          for (const chunk of chunks) {
            merged.set(chunk, offset);
            offset += chunk.length;
          }

          await fs.writeFile(path, merged);

          // 同步到localStorage (元数据)
          await this.syncMetadata();
        }
      });
    } catch (error) {
      console.error(`Failed to create write stream for ${path}:`, error);
      throw new Error(`Failed to create write stream: ${path}`);
    }
  }

  /**
   * 同步元数据到localStorage
   */
  private async syncMetadata(): Promise<void> {
    try {
      // 获取所有文件和目录的元数据
      const metadata: Record<string, VirtualStats> = {};

      // 递归遍历文件系统
      const traverse = async (path: string) => {
        const entries = await this.readdir(path);

        for (const entry of entries) {
          const fullPath = `${path}/${entry.name}`.replace(/\/+/g, '/');

          const stats = await this.stat(fullPath);
          metadata[fullPath] = stats;

          if (entry.isDirectory) {
            await traverse(fullPath);
          }
        }
      };

      await traverse('/');

      // 保存到localStorage
      localStorage.setItem('webenv-fs-metadata', JSON.stringify(metadata));
    } catch (error) {
      console.error('Failed to sync metadata:', error);
    }
  }
}

/**
```
```typescript
 * 创建IDE Store
 */
export const useIDEStore = create<IDEState>((set, get) => ({
  // 初始状态
  sidebarVisible: true,
  sidebarWidth: 240,
  openFiles: [],
  currentFile: null,
  panels: [
    { id: 'terminal', visible: false, height: 200 },
    { id: 'output', visible: false, height: 200 },
    { id: 'problems', visible: false, height: 200 },
    { id: 'debug-console', visible: false, height: 200 }
  ],
  activePanel: null,
  dirtyFiles: new Set(),
  searchVisible: false,
  searchQuery: '',
  searchReplace: false,

  // 切换侧边栏
  toggleSidebar: () => set((state) => ({
    sidebarVisible: !state.sidebarVisible
  })),

  // 设置侧边栏宽度
  setSidebarWidth: (width) => set({ sidebarWidth: width }),

  // 打开文件
  openFile: (path) => set((state) => {
    if (!state.openFiles.includes(path)) {
      return {
        openFiles: [...state.openFiles, path],
        currentFile: path
      };
    }
    return { currentFile: path };
  }),

  // 关闭文件

    const newOpenFiles = state.openFiles.filter(f => f !== path);

    // 如果关闭的是当前文件，切换到下一个或上一个
    let newCurrentFile = state.currentFile;
    if (state.currentFile === path) {
      const currentIndex = state.openFiles.indexOf(path);
      if (newOpenFiles.length > 0) {
        newCurrentFile = newOpenFiles[Math.min(currentIndex, newOpenFiles.length - 1)];
      } else {
        newCurrentFile = null;
      }
    }

    // 清除dirty标记
    const newDirtyFiles = new Set(state.dirtyFiles);
    newDirtyFiles.delete(path);

    return {
      openFiles: newOpenFiles,
      currentFile: newCurrentFile,
      dirtyFiles: newDirtyFiles
    };
  }),

  // 设置当前文件
  setCurrentFile: (path) => set({ currentFile: path }),

  // 切换面板
  togglePanel: (panel) => set((state) => ({
    panels: state.panels.map(p =>
      p.id === panel ? { ...p, visible: !p.visible } : p
    ),
    activePanel: state.panels.find(p => p.id === panel)?.visible ? panel : state.activePanel
  })),

  // 设置面板高度
  setPanelHeight: (panel, height) => set((state) => ({
    panels: state.panels.map(p =>
      p.id === panel ? { ...p, height } : p
    )
  })),

  // 设置活动面板
  setActivePanel: (panel) => set({ activePanel: panel }),

  // 标记文件为脏
  markDirty: (path) => set((state) => {
    const newDirtyFiles = new Set(state.dirtyFiles);
    newDirtyFiles.add(path);
    return { dirtyFiles: newDirtyFiles };
  }),

  // 标记文件为干净
  markClean: (path) => set((state) => {
    const newDirtyFiles = new Set(state.dirtyFiles);
    newDirtyFiles.delete(path);
    return { dirtyFiles: newDirtyFiles };
  }),

  // 切换搜索
  toggleSearch: () => set((state) => ({
    searchVisible: !state.searchVisible
  })),

  // 设置搜索查询
  setSearchQuery: (query) => set({ searchQuery: query }),

  // 设置搜索替换模式
  setSearchReplace: (replace) => set({ searchReplace: replace })
}));
```

## 五、面试题与答案

### 5.1 FastDocument 相关

**Q: FastDocument 的 Block 数据结构是如何设计的？为什么选择这种设计？**

A: FastDocument 使用原子化的 Block 数据结构，每个块都有唯一 ID、类型、内容和属性：

```typescript
interface Block {
  id: string;
  type: BlockType;
  content: string;
  properties?: BlockProperties;
  order?: number;
}
```

这种设计的优势：
1. **灵活扩展**: 每种块类型可以有自己的 properties，互不干扰
2. **易于协作**: 块级别的版本控制和同步，减少冲突
3. **性能优化**: 虚拟滚动只渲染可见块
4. **可组合性**: 支持块的嵌套和组合（如表格、列表）

**Q: 如何实现大文档的分块加载？**

A: 使用 `Map<number, Block[]>` 存储分块数据，`chunkSize` 默认为 50：

```typescript
const loadBlocksChunked = async (offset: number, limit: number = 50) => {
  const chunkIndex = Math.floor(offset / chunkSize);

  // 检查是否已加载
  if (loadedChunks.has(chunkIndex)) return;

  // 加载数据
  const response = await api.get(`/documents/${id}/blocks`, {
    params: { offset, limit }
  });

  // 存储到 Map
  blockChunks.set(chunkIndex, response.data.blocks);
  loadedChunks.add(chunkIndex);
};
```

```typescript
const socket = io(url, {

// 存储凭证
localStorage.setItem('uno_credential', JSON.stringify({

socket.on('reconnect', () => {
```

```typescript
// 只订阅 gameState，其他状态变化不会触发重渲染
```

### 5.3 WebEnv-OS 相关

**Q: WebEnv-OS 的虚拟文件系统是如何实现的？**

A: 使用 ZenFS（BrowserFS 的现代继任者）创建虚拟文件系统：

```typescript
// 挂载 IndexedDB 后端
await configure({
  mounts: [{
    path: '/',
    backend: IndexedDB,
    options: {
      dbName: 'webenv-os-fs',
      storeName: 'files'
    }
  }]
});

// 使用标准 Node.js fs API
await fs.writeFile('/test.txt', 'Hello World');
const content = await fs.readFile('/test.txt', 'utf-8');
```

这种设计的优势：
1. **标准接口**: 使用 Node.js fs API，兼容现有代码
2. **持久化**: 基于 IndexedDB，数据持久化存储
3. **多后端**: 支持挂载不同类型的存储后端

**Q: 如何实现文件自动保存？**

A: 使用防抖函数在内容变更后自动保存：

```typescript
const debouncedSave = useMemo(
  () => debounce(async (path: string, content: string) => {
    // 写入虚拟文件系统
    await vfs.writeFile(path, content);

    // 同步到后端
    await api.put('/api/files', { path, content });
  }, 1000), // 1秒防抖
  []
);

// 编辑器变更时调用
onChange={(content) => {
  debouncedSave(path, content);
}}
```
