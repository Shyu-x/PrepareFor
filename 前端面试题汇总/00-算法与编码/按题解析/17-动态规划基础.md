## 17.1 打家劫舍（LeetCode 198）

```javascript
// 记忆化搜索
function robMemo(nums) {
    const memo = {};
    function dfs(i) {
        if (i >= nums.length) return 0;
        if (memo[i] !== undefined) return memo[i];
        memo[i] = Math.max(dfs(i + 1), dfs(i + 2) + nums[i]);
        return memo[i];
    }
    return dfs(0);
}

// 递推
function rob(nums) {
    if (!nums.length) return 0;
    if (nums.length === 1) return nums[0];

    const dp = new Array(nums.length);
    dp[0] = nums[0];
    dp[1] = Math.max(nums[0], nums[1]);

    for (let i = 2; i < nums.length; i++) {
        dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
    }
    return dp[nums.length - 1];
}

// 空间优化
function robOptimized(nums) {
    if (!nums.length) return 0;
    let prev2 = 0, prev1 = nums[0];
    for (let i = 1; i < nums.length; i++) {
        const curr = Math.max(prev1, prev2 + nums[i]);
        prev2 = prev1;
        prev1 = curr;
    }
    return prev1;
}
```

## 17.2 爬楼梯（LeetCode 70）

```javascript
function climbStairs(n) {
    if (n <= 2) return n;
    let prev2 = 1, prev1 = 2;
    for (let i = 3; i <= n; i++) {
        const curr = prev1 + prev2;
        prev2 = prev1;
        prev1 = curr;
    }
    return prev1;
}
```

## 17.3 使用最小花费爬楼梯（LeetCode 746）

```javascript
function minCostClimbingStairs(cost) {
    const n = cost.length;
    let prev2 = cost[0], prev1 = cost[1];

    for (let i = 2; i < n; i++) {
        const curr = Math.min(prev1, prev2) + cost[i];
        prev2 = prev1;
        prev1 = curr;
    }
    return Math.min(prev1, prev2);
}
```
