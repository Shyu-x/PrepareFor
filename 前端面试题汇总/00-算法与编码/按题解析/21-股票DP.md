# 21 股票买卖系列

## 21.1 买卖股票的最佳时机 II（LeetCode 122）

```javascript
function maxProfit(prices) {
    let profit = 0;
    for (let i = 1; i < prices.length; i++) {
        profit += Math.max(0, prices[i] - prices[i - 1]);
    }
    return profit;
}
```

## 21.2 买卖股票的最佳时机（LeetCode 121）

```javascript
function maxProfit(prices) {
    let minPrice = Infinity;
    let maxProfit = 0;

    for (const price of prices) {
        minPrice = Math.min(minPrice, price);
        maxProfit = Math.max(maxProfit, price - minPrice);
    }
    return maxProfit;
}
```

## 21.3 买卖股票的最佳时机 III（LeetCode 123）

```javascript
function maxProfit(prices) {
    const n = prices.length;
    const k = 2;
    const dp = Array(n).fill(null).map(() =>
        Array(k + 1).fill(null).map(() => [0, 0])
    );

    // dp[i][j][0] 不持有，dp[i][j][1] 持有
    dp[0][0][1] = -prices[0];
    dp[0][1][1] = -prices[0];

    for (let i = 1; i < n; i++) {
        for (let j = 0; j <= k; j++) {
            dp[i][j][0] = Math.max(
                dp[i - 1][j][0],
                dp[i - 1][j][1] + prices[i]
            );
            if (j > 0) {
                dp[i][j][1] = Math.max(
                    dp[i - 1][j][1],
                    dp[i - 1][j - 1][0] - prices[i]
                );
            }
        }
    }
    return dp[n - 1][k][0];
}
```

## 21.4 最佳买卖股票时机含冷冻期（LeetCode 309）

```javascript
function maxProfit(prices) {
    const n = prices.length;
    // dp[i][0] 持有, dp[i][1] 不持有且非冷冻, dp[i][2] 冷冻
    const dp = Array(n).fill(null).map(() => [0, 0, 0]);

    dp[0][0] = -prices[0];

    for (let i = 1; i < n; i++) {
        dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][1] - prices[i]);
        dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][2]);
        dp[i][2] = dp[i - 1][0] + prices[i];
    }
    return Math.max(dp[n - 1][1], dp[n - 1][2]);
}
```

## 21.5 买卖股票的最佳时机含手续费（LeetCode 714）

```javascript
function maxProfit(prices, fee) {
    let cash = 0;
    let hold = -prices[0];

    for (let i = 1; i < prices.length; i++) {
        cash = Math.max(cash, hold + prices[i] - fee);
        hold = Math.max(hold, cash - prices[i]);
    }
    return cash;
}
```
