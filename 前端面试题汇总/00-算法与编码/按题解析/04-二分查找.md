## 4.1 红蓝染色法模板

```javascript
/**
 * 红蓝染色法 - 二分查找通用模板
 * 红色：不满足条件
 * 蓝色：满足条件
 * 找蓝色区域左边界
 */
function binarySearch(nums, target) {
    let left = -1;       // 红色边界
    let right = nums.length;  // 蓝色边界

    while (left + 1 < right) {
        const mid = left + Math.floor((right - left) / 2);
        if (isBlue(mid, nums, target)) {
            right = mid;
        } else {
            left = mid;
        }
    }
    return right;
}
```

## 4.2 在排序数组中查找元素的第一个和最后一个位置（LeetCode 34）

```javascript
function searchRange(nums, target) {
    const leftBorder = findLeft(nums, target);
    const rightBorder = findRight(nums, target);
    return [leftBorder, rightBorder];
}

function findLeft(nums, target) {
    let left = 0, right = nums.length - 1;
    let border = -1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] >= target) {
            border = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    return border;
}

function findRight(nums, target) {
    let left = 0, right = nums.length - 1;
    let border = -1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] <= target) {
            border = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return border;
}
```
