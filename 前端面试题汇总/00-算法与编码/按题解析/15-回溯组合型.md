# 15 回溯-组合型

组合型回溯：找满足条件的组合，需要剪枝

---

## 15.1 组合（LeetCode 77）

```javascript
function combine(n, k) {
    const result = [];
    const path = [];

    function backtrack(start) {
        if (path.length === k) {
            result.push([...path]);
            return;
        }

        // 剪枝：剩余元素不够
        const need = k - path.length;
        for (let i = start; i <= n - need + 1; i++) {
            path.push(i);
            backtrack(i + 1);
            path.pop();
        }
    }
    backtrack(1);
    return result;
}
```

## 15.2 组合总和 III（LeetCode 216）

```javascript
function combinationSum3(k, n) {
    const result = [];
    const path = [];

    function backtrack(start, sum) {
        if (path.length === k) {
            if (sum === n) result.push([...path]);
            return;
        }

        const need = k - path.length;
        for (let i = start; i <= 9; i++) {
            if (sum + i > n) break;  // 剪枝
            path.push(i);
            backtrack(i + 1, sum + i);
            path.pop();
        }
    }
    backtrack(1, 0);
    return result;
}
```

## 15.3 括号生成（LeetCode 22）

```javascript
function generateParenthesis(n) {
    const result = [];
    const path = [];

    function backtrack(open, close) {
        if (open === n && close === n) {
            result.push(path.join(''));
            return;
        }

        if (open < n) {
            path.push('(');
            backtrack(open + 1, close);
            path.pop();
        }
        if (close < open) {
            path.push(')');
            backtrack(open, close + 1);
            path.pop();
        }
    }
    backtrack(0, 0);
    return result;
}
```
