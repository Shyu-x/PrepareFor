## 1. React 性能优化技巧

### 1.1 组件重渲染优化

**问题分析：** 不必要的组件重渲染会导致性能问题

**解决方案：**

```tsx
// 1. 使用 React.memo 缓存组件
const MemoizedButton = React.memo(function Button({
  onClick,
  children
}: {
  onClick: () => void;
  children: React.ReactNode;
}) {
  console.log('Button rendered');
  return <button onClick={onClick}>{children}</button>;
});

// 2. 使用 useMemo 缓存计算结果
function ExpensiveComponent({ data }: { data: number[] }) {
  const sortedData = useMemo(() => {
    console.log('Sorting data...');
    return [...data].sort((a, b) => a - b);
  }, [data]);

  const average = useMemo(() => {
    return sortedData.reduce((a, b) => a + b, 0) / sortedData.length;
  }, [sortedData]);

  return <div>Average: {average}</div>;
}

// 3. 使用 useCallback 缓存回调函数
function ParentComponent() {
  const [count, setCount] = useState(0);

  const handleClick = useCallback(() => {
    console.log('Clicked');
    setCount(c => c + 1);
  }, []);

  // 只有当 handleClick 变化时才会重新渲染
  return <MemoizedButton onClick={handleClick}>Click me</MemoizedButton>;
}
```

### 1.2 虚拟化长列表

**问题：** 渲染大量 DOM 元素导致卡顿

**解决方案：**

```tsx
import { useVirtualizer } from '@tanstack/react-virtual';

function VirtualizedList({ items }: { items: string[] }) {
  const parentRef = useRef<HTMLDivElement>(null);

  const virtualizer = useVirtualizer({
    count: items.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 35,
    overscan: 5,
  });

  return (
    <div
      ref={parentRef}
      style={{
        height: '400px',
        overflow: 'auto',
      }}
    >
      <div
        style={{
          height: `${virtualizer.getTotalSize()}px`,
          width: '100%',
          position: 'relative',
        }}
      >
        {virtualizer.getVirtualItems().map((virtualItem) => (
          <div
            key={virtualItem.key}
            style={{
              position: 'absolute',
              top: 0,
              left: 0,
              width: '100%',
              height: `${virtualItem.size}px`,
              transform: `translateY(${virtualItem.start}px)`,
            }}
          >
            {items[virtualItem.index]}
          </div>
        ))}
      </div>
    </div>
  );
}
```

### 5.2 编辑器内容缓存

```tsx
function useEditorState(initialValue: string) {
  const [value, setValue] = useState(initialValue);
  const [history, setHistory] = useState<string[]>([initialValue]);
  const [historyIndex, setHistoryIndex] = useState(0);

  const pushChange = useCallback((newValue: string) => {
    setHistory((prev) => {
      const newHistory = prev.slice(0, historyIndex + 1);
      return [...newHistory, newValue];
    });
    setHistoryIndex((prev) => prev + 1);
    setValue(newValue);
  }, [historyIndex]);

  const undo = useCallback(() => {
    if (historyIndex > 0) {
      setHistoryIndex((prev) => prev - 1);
      setValue(history[historyIndex - 1]);
    }
  }, [historyIndex, history]);

  const redo = useCallback(() => {
    if (historyIndex < history.length - 1) {
      setHistoryIndex((prev) => prev + 1);
      setValue(history[historyIndex + 1]);
    }
  }, [historyIndex, history]);

  return { value, pushChange, undo, redo, canUndo: historyIndex > 0, canRedo: historyIndex < history.length - 1 };
}
```

```tsx
import { useRef, useMemo } from 'react';
import * as THREE from 'three';

function OptimizedMesh() {
  const meshRef = useRef<THREE.Mesh>(null);

  // 使用 useMemo 缓存几何体和材质
  const geometry = useMemo(() => {
    return new THREE.BoxGeometry(1, 1, 1);
  }, []);

  const material = useMemo(() => {
    return new THREE.MeshStandardMaterial({
      color: 'blue',
      metalness: 0.5,
      roughness: 0.5,
    });

  // 使用 useFrame 进行优化更新
  useFrame((state, delta) => {
    if (meshRef.current) {
      meshRef.current.rotation.x += delta * 0.5;
    }
  });

  return (
    <mesh ref={meshRef} geometry={geometry} material={material} />
  );
}
```

```tsx
function LargeFileEditor() {
  const handleEditorMount = (editor) => {
    // 大文件优化配置
    editor.updateOptions({
      minimap: { enabled: false }, // 禁用 minimap
      wordWrap: 'off', // 禁用自动换行
      smoothScrolling: false, // 禁用平滑滚动
      cursorBlinking: 'solid', // 禁用光标动画
      renderWhitespace: 'selection', // 只在选择时显示空白符
      folding: false, // 禁用代码折叠
      glyphMargin: false, // 禁用字形边距
    });
  };

  return (
    <Editor
      defaultLanguage="javascript"
      defaultValue="// Large file content here"
      height="500px"
      onMount={handleEditorMount}
      options={{
        // 初始配置
        maxTokenizationLineLength: 5000,
        automaticLayout: true,
        scrollBeyondLastLine: false,
      }}
    />
  );
}
```
