```tsx
// 创建 IDE 状态 Store，带持久化中间件
export const useIDEStore = create<IDEState>()(
  persist(
    // set: 更新状态的方法
    // get: 获取当前状态的方法
    (set, get) => ({
      // ===== 初始状态 =====

      workspaceRoot: null,      // 工作区根目录，初始为空
      fileTree: [],              // 文件树，初始为空数组
      currentFile: null,         // 当前编辑的文件
      openFiles: [],             // 打开的文件列表
      activeEditor: null,        // 当前激活的编辑器实例 ID
      sidebarVisible: true,      // 侧边栏默认可见
      activeSidebarView: "files", // 默认显示文件管理器
      bottomPanelVisible: false, // 底部面板默认隐藏
      activeBottomTab: "terminal", // 默认显示终端标签
      // ... 更多初始状态

      // ===== 状态设置方法 =====

      // 设置工作区根目录，同时清空相关状态
      setWorkspaceRoot: (path) => set({
        workspaceRoot: path,
        fileTree: [],            // 清空文件树
        currentFile: null,       // 清空当前文件
        openFiles: [],           // 清空打开的文件
      }),

      // 设置文件树
      setFileTree: (tree) => set({ fileTree: tree }),

      // 设置当前文件
      setCurrentFile: (path) => set({ currentFile: path }),

      // 设置打开的文件列表
      setOpenFiles: (files) => set({ openFiles: files }),

      // 更新单个文件属性（不可变更新）
      updateFile: (path, updates) => set((state) => ({
        fileTree: state.fileTree.map((item) =>
          item.path === path ? { ...item, ...updates } : item
        ),
      })),

      // 删除文件（从文件树和打开列表中移除）
      deleteFile: (path) => set((state) => ({
        fileTree: state.fileTree.filter((item) => item.path !== path),
        openFiles: state.openFiles.filter((p) => p !== path),
        currentFile: state.currentFile === path ? null : state.currentFile,
      })),

      // 重命名文件（更新所有相关引用）
      renameFile: (oldPath, newPath) => set((state) => ({
        fileTree: state.fileTree.map((item) =>
          item.path === oldPath ? { ...item, path: newPath } : item
        ),
        openFiles: state.openFiles.map((p) =>
          p === oldPath ? newPath : p
        ),
        currentFile: state.currentFile === oldPath ? newPath : state.currentFile,
      })),

      // 切换侧边栏视图（智能切换逻辑）
      setActiveSidebarView: (view) => {
        const state = get();
        // 如果点击当前视图且侧边栏可见，则关闭侧边栏
        if (state.activeSidebarView === view && state.sidebarVisible) {
          set({ sidebarVisible: false });
        } else {
          // 否则切换到新视图并显示侧边栏
          set({ activeSidebarView: view, sidebarVisible: true });
        }
      },

      // 切换侧边栏可见性
      toggleSidebar: () => set((state) => ({
        sidebarVisible: !state.sidebarVisible
      })),

      // 设置底部面板标签（同时显示面板）
      setActiveBottomTab: (tab) => set({
        activeBottomTab: tab,
        bottomPanelVisible: true
      }),

      // 切换底部面板可见性
      toggleBottomPanel: () => set((state) => ({
        bottomPanelVisible: !state.bottomPanelVisible
      })),

      // 显示指定面板（快捷方法）
      showBottomPanel: (tab) => set({
        activeBottomTab: tab,
        bottomPanelVisible: true
      }),

      // 追加运行输出（不可变数组更新）
      appendRunOutput: (line) => set((state) => ({
        runOutput: [...state.runOutput, line]
      })),

      // 清空运行输出
      clearRunOutput: () => set({ runOutput: [] }),

      // 设置预览（打开预览面板）
      setPreview: ({ url, title, type, sourcePath }) => set({
        previewUrl: url,
        previewTitle: title,
        previewType: type,
        previewSourcePath: sourcePath || null,
        bottomPanelVisible: true,  // 显示底部面板
        activeBottomTab: "preview", // 切换到预览标签
      }),

      // 清除预览
      clearPreview: () => set({
        previewUrl: null,
        previewTitle: "",
        previewType: "none",
        previewSourcePath: null,
        isPreviewOpen: false
      }),

      // 切换断点
      toggleBreakpoint: (file, line) => set((state) => {
        // 查找断点是否已存在
        const exists = state.breakpoints.find(
          (b) => b.file === file && b.line === line
        );
        // 如果存在则移除，否则添加
        return {
          breakpoints: exists
            ? state.breakpoints.filter(
                (b) => !(b.file === file && b.line === line)
              )
            : [...state.breakpoints, { file, line }],
        };
      }),

      // 更新设置（浅合并）
      updateSettings: (newSettings) => set((state) => ({
        settings: { ...state.settings, ...newSettings }
      })),

      // 重置 Store 到初始状态
      reset: () => set({
        workspaceRoot: null,
        fileTree: [],
        currentFile: null,
        openFiles: [],
        // ... 更多需要重置的字段
      }),
    }),
    // ===== 5. 持久化配置 =====
    {
      name: "ide-storage", // localStorage 存储键名
    }
  )
);
```

```tsx
/**
 * IDE 主布局
 * 使用 position:fixed + inset:0 确保占满整个视口，不依赖父元素高度链
 */

// 标记为客户端组件（Next.js App Router 要求）
"use client";

// ===== 1. 导入依赖 =====

import { useCallback, useEffect, useMemo, useState } from "react"; // React Hooks
import ActivityBar from "./ide/ActivityBar";           // 左侧活动栏
import Sidebar from "./ide/Sidebar";                   // 侧边栏
import EditorPane from "./editor/EditorPane";         // 编辑器窗格
import PanelArea from "./ide/PanelArea";              // 底部面板
import StatusBar from "./StatusBar";                  // 状态栏
import CommandPalette, { Command } from "./CommandPalette"; // 命令面板
import { useIDEStore } from "@/store/useIDEStore";    // IDE 状态 Store
import QuickOpen from "./ide/QuickOpen";              // 快速打开
import GoToSymbol from "./ide/GoToSymbol";            // 跳转符号
import FindReplace from "./ide/FindReplace";          // 查找替换
import GlobalSymbolSearch from "./ide/GlobalSymbolSearch"; // 全局符号搜索
import GlobalReplace from "./ide/GlobalReplace";      // 全局替换
import MenuBar from "./ide/MenuBar";                  // 菜单栏
import { formatContent, toggleLineComment } from "@/lib/editor/textActions"; // 文本操作工具
import { Modal, Input, message } from "antd";        // Ant Design 组件
import { mutate } from "swr";                         // 数据突变
import WorkspaceSelector from "./ide/WorkspaceSelector"; // 工作区选择器
import WorkspaceManager from "./WorkspaceManager";     // 工作区管理器

// ===== 2. 常量定义 =====

const STATUS_BAR_HEIGHT = 22; // 状态栏高度（像素）

// ===== 3. 内部组件 - 包含所有 Hooks =====

function IDELayoutContent() {
  // ===== 3.1 Modal 状态管理 =====

  // 命令面板可见性
  const [commandPaletteVisible, setCommandPaletteVisible] = useState(false);
  // 快速打开可见性
  const [quickOpenVisible, setQuickOpenVisible] = useState(false);
  // 跳转符号对话框可见性
  const [goToSymbolVisible, setGoToSymbolVisible] = useState(false);
  // 查找替换对话框可见性
  const [findVisible, setFindVisible] = useState(false);
  // 替换对话框可见性
  const [replaceVisible, setReplaceVisible] = useState(false);
  // 全局替换对话框可见性
  const [globalReplaceVisible, setGlobalReplaceVisible] = useState(false);
  // 全局符号搜索对话框可见性
  const [globalSymbolVisible, setGlobalSymbolVisible] = useState(false);
  // 重命名对话框可见性
  const [renameVisible, setRenameVisible] = useState(false);
  // 重命名输入值
  const [renameValue, setRenameValue] = useState("");
  // 工作区管理器对话框可见性
  const [workspaceManagerVisible, setWorkspaceManagerVisible] = useState(false);

  // ===== 3.2 从 Store 获取状态和方法 =====

  const {
    toggleSidebar,           // 切换侧边栏
    showBottomPanel,         // 显示底部面板
    setActiveSidebarView,    // 设置侧边栏视图
    keybindings,             // 快捷键配置
    currentFile,             // 当前文件
    openFiles,               // 打开的文件列表
    setOpenFiles,           // 设置打开的文件
    setCurrentFile,          // 设置当前文件
    appendRunOutput,         // 追加运行输出
    clearRunOutput,          // 清空运行输出
    renameFile,              // 重命名文件
    settings,                // 用户设置
    workspaceRoot,           // 工作区根目录
    setWorkspaceRoot,        // 设置工作区根目录
    togglePreview,           // 切换预览
  } = useIDEStore();

  // ===== 3.3 获取文件系统操作函数 =====

  const { readFile, writeFile, renameNode } = useFileSystem(workspaceRoot || "default");

  // ===== 3.4 快捷键映射 =====
  // 使用 useMemo 缓存，避免每次渲染重新计算
  const keyMap = useMemo(() => keybindings, [keybindings]);

  // ===== 3.5 辅助函数 =====

  // 规范化按键组合（将事件转换为标准格式如 "Ctrl+Shift+P"）
  const normalizeCombo = (e: KeyboardEvent) => {
    const parts: string[] = [];
    if (e.ctrlKey || e.metaKey) parts.push("Ctrl");  // 添加 Ctrl
    if (e.shiftKey) parts.push("Shift");             // 添加 Shift
    if (e.altKey) parts.push("Alt");                  // 添加 Alt
    const key = e.key.length === 1 ? e.key.toUpperCase() : e.key; // 处理按键名
    parts.push(key === "`" ? "`" : key);              // 添加最终按键
    return parts.join("+");                           // 组合为字符串
  };

  // ===== 3.6 命令定义 =====

  // 可在命令面板中执行的命令列表
  const commands: Command[] = [
    {
      id: "toggle-sidebar",
      name: "Toggle Sidebar",
      description: "Show/Hide the side bar",
      category: "view",
      shortcut: keybindings.toggleSidebar,
      action: () => toggleSidebar(),
    },
    {
      id: "toggle-terminal",
      name: "Toggle Terminal",
      description: "Show/Hide terminal panel",
      category: "view",
      shortcut: keybindings.toggleTerminal,
      action: () => showBottomPanel("terminal"),
    },
  ];

  // ===== 3.7 获取文件注释符号 =====

  // 根据文件扩展名返回对应的注释符号
  const getCommentToken = (path: string) => {
    const ext = path.split(".").pop()?.toLowerCase(); // 获取扩展名
    if (!ext) return "//";        // 无扩展名默认 //
    if (["py", "rb", "sh", "yml"].includes(ext)) return "#";   // Python/Shell
    if (["sql"].includes(ext)) return "--";  // SQL
    return "//";                   // 默认
  };

  // ===== 3.8 更新文件内容 =====

  const updateFileContent = async (path: string, next: string) => {
    try {
      await writeFile(path, next);                    // 写入文件
      mutate(["filesystem-tree", workspaceRoot]);     // 刷新文件树缓存
    } catch (e) {
      console.error("Failed to update file:", e);    // 错误处理
    }
  };

  // ===== 3.9 格式化文档处理 =====

  const handleFormatDocument = useCallback(async () => {
    if (!currentFile) return;                         // 无当前文件则返回
    try {
      const content = await readFile(currentFile);   // 读取文件内容
      // 格式化内容
      const formatted = formatContent(content, {
        trimTrailingWhitespace: true,
        insertFinalNewline: true,
        tabSize: settings.tabSize,
        insertSpaces: true,
      });
      await updateFileContent(currentFile, formatted); // 写入格式化后的内容
      appendRunOutput("[Info] 已格式化当前文件");      // 输出提示
      showBottomPanel("output");                      // 显示输出面板
    } catch (e) {
      console.error("Format failed:", e);
    }
  }, [appendRunOutput, currentFile, settings.tabSize, showBottomPanel, updateFileContent, readFile]);

  // ===== 3.10 切换注释处理 =====

  const handleToggleComment = useCallback(async () => {
    if (!currentFile) return;
    try {
      const content = await readFile(currentFile);
      const token = getCommentToken(currentFile);      // 获取注释符号
      const next = toggleLineComment(content, token); // 切换注释
      await updateFileContent(currentFile, next);
      appendRunOutput("[Info] 已切换注释");
      showBottomPanel("output");
    } catch (e) {
      console.error("Toggle comment failed:", e);
    }
  }, [appendRunOutput, currentFile, showBottomPanel, updateFileContent, readFile]);

  // ===== 3.11 打开重命名对话框 =====

  const openRenameDialog = useCallback(() => {
    if (!currentFile) return;
    const name = currentFile.split("/").pop() || "";   // 提取文件名
    setRenameValue(name);                              // 设置初始值
    setRenameVisible(true);                             // 显示对话框
  }, [currentFile]);

  // ===== 3.12 处理重命名 =====

  const handleRename = useCallback(async () => {
    if (!currentFile) return;
    const trimmed = renameValue.trim();                // 去除空格
    if (!trimmed) return;                             // 空值返回
    // 计算新路径
    const baseDir = currentFile.split("/").slice(0, -1).join("/") || "/";
    const newPath = trimmed.includes("/")
      ? trimmed                                          // 完整路径
      : baseDir === "/"
        ? `/${trimmed}`                                 // 根目录下的文件
        : `${baseDir}/${trimmed}`;                      // 相对路径

    if (newPath === currentFile) {                      // 路径未变
      setRenameVisible(false);
      return;
    }

    try {
      await renameNode(currentFile, newPath);          // 调用后端重命名
      renameFile(currentFile, newPath);                 // 更新 Store 状态
      setOpenFiles(openFiles.map((p) =>
        p === currentFile ? newPath : p
      ));                                                // 更新打开的文件列表
      setCurrentFile(newPath);                          // 更新当前文件
      setRenameVisible(false);                          // 关闭对话框
      mutate(["filesystem-tree", workspaceRoot]);       // 刷新文件树
      appendRunOutput("[Info] 已重命名文件");
      showBottomPanel("output");
    } catch (e) {
      console.error("Rename failed:", e);
      message.error("重命名异常");
    }
  }, [appendRunOutput, currentFile, openFiles, renameFile, renameValue, showBottomPanel, setCurrentFile, setOpenFiles, workspaceRoot, renameNode]);

  // ===== 3.13 键盘事件处理 =====

  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      const combo = normalizeCombo(e);                  // 规范化按键
      const match = Object.entries(keyMap).find(      // 查找匹配
        ([, value]) => value.toLowerCase() === combo.toLowerCase()
      );
      if (!match) return;                              // 无匹配返回
      const [action] = match;                          // 获取动作名

      e.preventDefault();                               // 阻止默认行为

      // 根据动作执行相应处理
      switch (action) {
        case "commandPalette":
          setCommandPaletteVisible(true);
          break;
        case "quickOpen":
          setQuickOpenVisible(true);
          break;
        // ... 更多 case
        case "toggleSidebar":
          toggleSidebar();
          break;
        case "formatDocument":
          handleFormatDocument();
          break;
        // ...
      }
    };

    // 绑定/解绑键盘事件
    window.addEventListener("keydown", handleKeyDown);
    return () => window.removeEventListener("keydown", handleKeyDown);
  }, [keyMap, toggleSidebar, handleFormatDocument, ...]);

  // ===== 3.14 渲染布局 =====

  return (
    <>
      {/* 主布局容器 - 使用固定定位占满全屏 */}
      <div
        style={{
          position: "fixed",
          top: 0, left: 0, right: 0, bottom: 0,
          display: "flex",
          flexDirection: "column",   // 垂直排列
          overflow: "hidden",        // 隐藏溢出
          background: "var(--color-background)",
          color: "var(--color-text)",
        }}
      >
        {/* 菜单栏 */}
        <MenuBar
          onCommandPalette={() => setCommandPaletteVisible(true)}
          onQuickOpen={() => setQuickOpenVisible(true)}
          // ... 更多回调
        />

        {/* 主内容区 */}
        <div style={{ flex: 1, display: "flex", flexDirection: "row" }}>
          <ActivityBar />    {/* 左侧活动栏 */}
          <Sidebar />         {/* 侧边栏 */}
          {/* 编辑器区域 */}
          <div style={{ flex: 1, display: "flex", flexDirection: "column" }}>
            <EditorPane />    {/* 编辑器窗格 */}
            <PanelArea />     {/* 底部面板 */}
          </div>
        </div>

        {/* 状态栏 */}
        <div style={{ height: STATUS_BAR_HEIGHT }}>
          <StatusBar />
        </div>
      </div>

      {/* Modal 组件 - 命令面板 */}
      <CommandPalette
        visible={commandPaletteVisible}
        onClose={() => setCommandPaletteVisible(false)}
        commands={commands}
      />

      {/* 更多 Modal 组件... */}
    </>
  );
}
```

```tsx
// ===== 1. 窗口创建方法 =====

const createWindow = useCallback((
  id: string,           // 窗口唯一 ID
  title: string,        // 窗口标题
  content: ReactNode,   // 窗口内容
  options?: Partial<WindowItem>, // 可选配置

      id,
      appId: options?.appId,         // 应用 ID
      title,                          // 标题
      content,                        // 内容
      visible: true,                  // 初始可见
      minimized: false,               // 初始未最小化
      maximized: false,               // 初始未最大化
      // 位置：级联偏移避免完全重叠
      position: options?.position || {
        x: 100 + prev.length * 20,
        y: 100 + prev.length * 20,
      },
      // 尺寸：默认 600x400
      size: options?.size || { width: 600, height: 400 },
      // 计算 zIndex：新窗口在最上层
      zIndex: Math.max(...prev.map((w) => w.zIndex), 100) + 1,
      windowControls: options?.windowControls || windowControls,
      ...options,
    };

    return [...prev, newWindow];     // 添加到窗口列表
  });

  setActiveWindowId(id);               // 设置为活动窗口
}, [windowControls]);

// ===== 2. 窗口激活（置顶）=====

const activateWindow = useCallback((id: string) => {
  setActiveWindowId(id);               // 更新活动窗口 ID

  setWindows((prev) => {
    // 获取当前最大 zIndex
    const maxZIndex = Math.max(...prev.map((w) => w.zIndex), 100);

    // 已在最上层，无需更新
    const current = prev.find(w => w.id === id);
    if (current && current.zIndex === maxZIndex) return prev;

    // 更新 zIndex：新窗口置顶
    return prev.map((w) =>
      w.id === id ? { ...w, zIndex: maxZIndex + 1 } : w
    );
  });
}, []);
```

## 6. 事件处理流程

### 6.1 文件打开流程

```
1. 用户双击文件树中的文件
   ↓
2. FileTree 组件调用 setCurrentFile(path)
   ↓
3. useIDEStore 更新 currentFile 状态
   ↓
4. EditorPane 检测到 currentFile 变化
   ↓
5. EditorPane 调用 readFile(path) 读取内容
   ↓
6. CodeEditor 组件渲染文件内容
```

### 6.2 快捷键触发流程

```
1. 用户按下键盘（如 Ctrl+P）
   ↓
2. IDELayout 的 handleKeyDown 捕获事件
   ↓
3. normalizeCombo(e) 转换为 "Ctrl+P"
   ↓
4. 查找 keyMap 中匹配的 action
   ↓
5. 执行对应的状态更新或 UI 操作
   ↓
6. 组件重渲染反映变化
```

### 6.3 窗口激活流程

```
1. 用户点击窗口
   ↓
2. Window 组件触发 onMouseDown
   ↓
3. 调用 activateWindow(id)
   ↓
4. useState 更新 activeWindowId
   ↓
5. 计算新的 zIndex 层级
   ↓
6. 窗口重新渲染到最上层
```
