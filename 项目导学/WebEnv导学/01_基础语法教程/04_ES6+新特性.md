## 1. let 和 const

### 1.1 let 与 var 的区别

```javascript
// 1. 块级作用域
{
    let blockLet = '块级变量';
    var blockVar = '函数级变量';
}
// console.log(blockLet); // 报错：blockLet is not defined
console.log(blockVar);    // "函数级变量"

// 2. 不存在变量提升
// console.log(hoistedLet); // 报错：Cannot access before initialization
let hoistedLet = 'let 变量';

// var 存在变量提升（值为 undefined）
console.log(hoistedVar); // undefined（不会报错）
var hoistedVar = 'var 变量';

// 3. 不能重复声明
let count = 1;
// let count = 2; // 报错：Identifier 'count' has already been declared

// var 可以重复声明
var count2 = 1;
var count2 = 2; // 不报错
```

### 1.2 const 特性

```javascript
// 1. 声明时必须赋值
// const num; // 报错：Missing initializer

// 2. 不能重新赋值
const PI = 3.14159;
// PI = 3.14; // 报错：Assignment to constant variable

// 3. 对象和数组可以修改内容
const user = { name: '张三' };
user.name = '李四'; // 可以
// user = {}; // 报错

const arr = [1, 2, 3];
arr.push(4); // 可以
// arr = []; // 报错

// 4. 冻结对象（真正不可变）
const frozen = Object.freeze({ name: '张三' });
// frozen.name = '李四'; // 严格模式报错
```

### 1.3 使用场景

```javascript
// 优先使用 const
const API_URL = 'https://api.example.com';
const MAX_RETRIES = 3;
const EMPTY_ARRAY = [];

// 需要重新赋值时使用 let
let currentIndex = 0;
currentIndex++;
for (let i = 0; i < 10; i++) { }

// 避免使用 var
// 原因：变量提升导致难以理解的 bug，缺乏块级作用域
```

## 2. 箭头函数

### 2.1 基本语法

```javascript
// 传统函数
function add(a, b) {
    return a + b;
}

// 箭头函数
const add = (a, b) => {
    return a + b;
};

// 简写：单行函数体可省略 return 和大括号
const add = (a, b) => a + b;

// 单参数可省略括号
const double = x => x * 2;

// 无参数
const getRandom = () => Math.random();

// 多行函数体需要 return
const calculate = (a, b) => {
    const sum = a + b;
    const product = a * b;
    return { sum, product };
};
```

### 2.2 this 绑定

```javascript
// 普通函数中的 this
function Person() {
    this.age = 0;

    // 定时器中的 this 指向 window
    setInterval(function() {
        this.age++; // this 指向 window
        console.log(this.age); // NaN
    }, 1000);

    // 使用 that 保存 this
    const that = this;
    setInterval(function() {
        that.age++;
        console.log(that.age);
    }, 1000);

    // 箭头函数中的 this 继承外部作用域
    setInterval(() => {
        this.age++; // this 指向 Person 实例
        console.log(this.age);
    }, 1000);
}
```

### 2.3 其他区别

```javascript
// 1. 箭头函数没有 arguments
// const fn = () => console.log(arguments); // 报错

// 使用剩余参数代替
const fn = (...args) => console.log(args);
fn(1, 2, 3); // [1, 2, 3]

// 2. 箭头函数不能作为构造函数
// const Person = () => {};
// new Person(); // 报错：Person is not a constructor

// 3. 箭头函数没有 prototype
const fn = () => {};
console.log(fn.prototype); // undefined
```

### 2.4 实际应用

```javascript
// 数组方法中使用
const numbers = [1, 2, 3, 4, 5];

// map
const doubled = numbers.map(n => n * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// filter
const evens = numbers.filter(n => n % 2 === 0);
console.log(evens); // [2, 4]

// find
const found = numbers.find(n => n > 3);
console.log(found); // 4

// reduce
const sum = numbers.reduce((acc, n) => acc + n, 0);
console.log(sum); // 15

// 事件处理
button.addEventListener('click', (event) => {
    console.log(event.target);
});

// 回调地狱解决方案
fetchData()
    .then(data => processData(data))
    .then(processed => saveData(processed))
    .catch(error => handleError(error));
```

## 3. 模板字符串

### 3.1 基本用法

```javascript
// 使用反引号（`）定义
const name = '张三';
const age = 25;

// 字符串拼接
const message = '我叫' + name + '，今年' + age + '岁。';

// 模板字符串
const template = `我叫${name}，今年${age}岁。`;
console.log(template); // 我叫张三，今年25岁。

// 表达式
const a = 10, b = 20;
console.log(`${a} + ${b} = ${a + b}`); // 10 + 20 = 30

// 函数调用
function greet(name) {
    return `你好，${name}！`;
}
console.log(`${greet('张三')}`); // 你好，张三！
```

### 3.2 多行字符串

```javascript
// 传统方式：需要使用 \n 或字符串拼接
const html1 = '<div>\n' +
    '<h1>标题</h1>\n' +
    '<p>段落</p>\n' +
    '</div>';

// 模板字符串：直接换行
const html2 = `
<div>
    <h1>标题</h1>
    <p>段落</p>
</div>
`;

console.log(html2);
```

### 3.3 标签模板

```javascript
// 定义标签函数
function highlight(strings, ...values) {
    let result = '';
    strings.forEach((str, i) => {
        result += str;
        if (i < values.length) {
            result += `<mark>${values[i]}</mark>`;
        }
    });
    return result;
}

const name = 'JavaScript';
const level = '重要';

// 使用标签模板
const output = highlight`${name}是一门${level}的编程语言`;
console.log(output);
// <mark>JavaScript</mark>是一门<mark>重要</mark>的编程语言
```

## 4. 解构赋值

### 4.1 数组解构

```javascript
// 基本解构
const [a, b, c] = [1, 2, 3];
console.log(a, b, c); // 1 2 3

// 跳过元素
const [first, , third] = [1, 2, 3];
console.log(first, third); // 1 3

// 剩余模式
const [head, ...tail] = [1, 2, 3, 4, 5];
console.log(head); // 1
console.log(tail); // [2, 3, 4, 5]

// 默认值
const [x, y, z = 10] = [1, 2];
console.log(x, y, z); // 1 2 10

// 交换变量
let m = 1, n = 2;
[m, n] = [n, m];
console.log(m, n); // 2 1

// 嵌套解构
const [a, [b, c]] = [1, [2, 3]];
console.log(a, b, c); // 1 2 3
```

### 4.2 对象解构

```javascript

// 重命名

console.log(userName); // 张三

// 默认值
const { name, gender = '未知' } = { name: '张三', gender: 'male' };
console.log(gender); // male

// 剩余模式
const { name, ...rest } = { name: '张三', age: 25, city: 'Beijing' };
console.log(rest); // { age: 25, city: 'Beijing' }

// 嵌套解构
const {
    name,
    address: { city, country }

console.log(city, country); // Beijing China
```

    config: { debug: true }

## 6. Promise 基础

### 6.1 Promise 概念

```javascript
// 创建 Promise
const promise = new Promise((resolve, reject) => {
    // 异步操作
    const success = true;

    if (success) {
        resolve('操作成功');
    } else {
        reject('操作失败');
    }
});

// Promise 状态
// pending（进行中）
// fulfilled（已成功）
// rejected（已失败）
```

### 6.2 then, catch, finally

```javascript
const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        const random = Math.random();
        if (random > 0.5) {
            resolve(random);
        } else {
            reject(new Error('随机数太小'));
        }
    }, 1000);
});

// then：处理成功
promise
    .then(result => {
        console.log('成功：', result);
        return result * 2;
    })
    .then(double => {
        console.log('翻倍：', double);
    });

// catch：处理错误
promise
    .then(result => console.log(result))
    .catch(error => console.error('错误：', error.message));

// finally：无论成功失败都执行
promise
    .then(result => console.log(result))
    .catch(error => console.error(error))
    .finally(() => console.log('完成'));
```

### 6.3 Promise 静态方法

```javascript
// Promise.resolve() - 立即 resolved 的 Promise
Promise.resolve(42).then(console.log);

// Promise.reject() - 立即 rejected 的 Promise
Promise.reject(new Error('失败')).catch(console.error);

// Promise.all() - 全部 resolved 后返回
const p1 = Promise.resolve(1);
const p2 = Promise.resolve(2);
const p3 = Promise.resolve(3);

Promise.all([p1, p2, p3])
    .then(results => console.log(results)); // [1, 2, 3]

// Promise.race() - 返回最先 settled 的结果
Promise.race([
    new Promise(r => setTimeout(() => r(1), 1000)),
    new Promise(r => setTimeout(() => r(2), 500))
]).then(console.log); // 2

// Promise.allSettled() - 全部 settled 后返回
Promise.allSettled([
    Promise.resolve(1),
    Promise.reject(new Error('失败'))
]).then(results => {
    results.forEach((r, i) => {
        if (r.status === 'fulfilled') {
            console.log(`P${i}: ${r.value}`);
        } else {
            console.log(`P${i}: ${r.reason.message}`);
        }
    });
});

// Promise.any() - 返回第一个 fulfilled 的结果
Promise.any([
    Promise.reject(new Error('失败1')),
    Promise.resolve('成功'),
    Promise.reject(new Error('失败2'))
]).then(console.log); // 成功
```

### 6.4 链式调用

```javascript
// 模拟异步操作
function delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}

// 链式调用
delay(1000)
    .then(() => {
        console.log('第1步完成');
        return delay(1000);
    })
    .then(() => {
        console.log('第2步完成');
        return delay(1000);
    })
    .then(() => {
        console.log('第3步完成');
    })
    .catch(error => console.error('出错：', error));
```

## 7. async/await

### 7.1 基本用法

```javascript
// async 函数
async function fetchData() {
    return '数据';
}

// 等价于
function fetchData() {
    return Promise.resolve('数据');
}

// await 等待 Promise
async function main() {
    const result = await Promise.resolve('Hello');
    console.log(result); // Hello
}
main();

// 错误处理
async function fetchWithError() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('请求失败：', error);
        throw error;
    }
}
```

### 7.2 并行执行

```javascript
// 串行执行（等待每个）
async function serial() {
    const result1 = await fetch('/api1');
    const result2 = await fetch('/api2');
    console.log(result1, result2);
}

// 并行执行（同时发起）
async function parallel() {
    const [response1, response2] = await Promise.all([
        fetch('/api1'),
        fetch('/api2')
    ]);
    console.log(response1, response2);
}

// Promise.all 更优雅
async function parallel2() {
    const results = await Promise.all([
        fetch('/api1').then(r => r.json()),
        fetch('/api2').then(r => r.json())
    ]);
    console.log(results);
}
```

### 7.3 实际应用

```javascript
// 模拟 API 调用
function fetchUser(id) {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve({ id, name: `用户${id}`, age: 20 + id });
        }, 500);
    });
}

function fetchPosts(userId) {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve([
                { id: 1, title: '文章1' },
                { id: 2, title: '文章2' }
            ]);
        }, 500);
    });
}

// 获取用户和文章
async function getUserData(userId) {
    try {
        const user = await fetchUser(userId);
        const posts = await fetchPosts(userId);
        return { user, posts };
    } catch (error) {
        console.error('获取数据失败：', error);
        return null;
    }
}

getUserData(1).then(console.log);
```

## 9. 类

### 9.1 基本语法

```javascript
// ES5 构造函数
function Person(name, age) {
    this.name = name;
    this.age = age;
}
Person.prototype.sayHello = function() {
    return '你好，我是' + this.name;
};
Person.prototype.staticProp = '静态属性';

// ES6 类
class Person {
    // 构造函数
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    // 方法（自动添加到 prototype）
    sayHello() {
        return `你好，我是${this.name}`;
    }

    // getter/setter
    get info() {
        return `${this.name}, ${this.age}岁`;
    }

    set info(value) {
        [this.name, this.age] = value.split(',');
    }

    // 静态方法
    static create(name, age) {
        return new Person(name, age);
    }

    // 静态属性
    static species = '人类';
}

// 使用
const person = new Person('张三', 25);
console.log(person.sayHello());
console.log(person.info);
person.info = '李四,30';
console.log(person.name); // 李四
console.log(Person.species); // 人类
```

### 9.2 继承

```javascript
class Animal {
    constructor(name) {
        this.name = name;
    }

    speak() {
        return `${this.name} 发出了声音`;
    }

    static isAnimal(obj) {
        return obj instanceof Animal;
    }
}

class Dog extends Animal {
    constructor(name, breed) {
        super(name); // 调用父类构造函数
        this.breed = breed;
    }

    // 重写方法
    speak() {
        return `${this.name} 汪汪叫`;
    }

    // 新增方法
    fetch() {
        return `${this.name} 捡回了球`;
    }
}

const dog = new Dog('旺财', '金毛');
console.log(dog.speak()); // 旺财 汪汪叫
console.log(dog.fetch()); // 旺财 捡回了球
console.log(Animal.isAnimal(dog)); // true
```

### 9.3 私有字段

```javascript
class Counter {
    // 公有字段
    count = 0;

    // 私有字段（#开头）
    #history = [];

    // 静态私有字段
    static #instances = 0;

    constructor() {
        Counter.#instances++;
    }

    increment() {
        this.count++;
        this.#history.push(this.count);
        return this.count;
    }

    getHistory() {
        return [...this.#history];
    }

    static getInstances() {
        return Counter.#instances;
    }
}

const counter = new Counter();
counter.increment();
counter.increment();
console.log(counter.getHistory()); // [1, 2]
// console.log(counter.#history); // 报错
console.log(Counter.getInstances()); // 1
```

## 10. Symbol

### 10.1 基本用法

```javascript
// 创建 Symbol
const sym1 = Symbol('描述');
const sym2 = Symbol('描述');

console.log(sym1 === sym2); // false（每次都是新的）

// 作为对象属性
const obj = {
    [Symbol('id')]: 1,
    name: '张三'
};

// 获取 Symbol 属性
const keys = Object.getOwnPropertySymbols(obj);
console.log(keys);
```

### 10.2 常用场景

```javascript
// 1. 定义类的私有属性
class MyClass {
    #privateMethod() {
        return '私有方法';
    }

    publicMethod() {
        return this.#privateMethod();
    }
}

// 2. Symbol.for() - 全局 Symbol 注册表
const s1 = Symbol.for('myKey');
const s2 = Symbol.for('myKey');
console.log(s1 === s2); // true
console.log(Symbol.keyFor(s1)); // 'myKey'

// 3. 内置 Symbol
const arr = [1, 2, 3];
console.log(arr[Symbol.iterator]); // 函数

// Symbol.hasInstance
class EvenNumber {
    static [Symbol.hasInstance](obj) {
        return Number(obj) % 2 === 0;
    }
}
console.log(2 instanceof EvenNumber); // true
console.log(3 instanceof EvenNumber); // false

// Symbol.toStringTag
class Person {
    get [Symbol.toStringTag]() {
        return 'Person';
    }
}
console.log(Object.prototype.toString.call(new Person())); // [object Person]
```

## 11. Set 和 Map

### 11.1 Set

```javascript
// 创建 Set
const set = new Set([1, 2, 3, 3, 3]);
console.log(set); // Set(3) { 1, 2, 3 }

// 添加元素
set.add(4);
set.add(4); // 重复添加无效
console.log(set.size); // 4

// 删除元素
set.delete(3);
console.log(set.has(3)); // false

// 清空
set.clear();

// 遍历
const fruits = new Set(['apple', 'banana', 'orange']);
fruits.forEach(value => console.log(value));

for (let item of fruits) {
    console.log(item);
}

// 数组去重
const arr = [1, 2, 2, 3, 3, 3];
const unique = [...new Set(arr)];
console.log(unique); // [1, 2, 3]
```

### 11.2 Map

```javascript
// 创建 Map
const map = new Map([
    ['name', '张三'],
    ['age', 25]
]);

// 设置
map.set('city', 'Beijing');
map.set('age', 26); // 修改已有键

// 获取
console.log(map.get('name')); // 张三
console.log(map.has('age'));  // true

// 删除
map.delete('city');
map.clear();

// 遍历
map.forEach((value, key) => {
    console.log(`${key}: ${value}`);

for (let [key, value] of map) {
    console.log(key, value);
}

// Map 与对象转换
const obj = { a: 1, b: 2 };
const mapFromObj = new Map(Object.entries(obj));
const objFromMap = Object.fromEntries(mapFromObj);
```

## 12. 可选链和空值合并

### 12.1 可选链（Optional Chaining）

```javascript
// 访问深层属性
const user = {
    profile: {
        name: '张三',
        address: {
            city: 'Beijing'
        }
    }
};

// 传统写法
const city = user && user.profile && user.profile.address && user.profile.address.city;

// 可选链（ES2020）
const city2 = user?.profile?.address?.city;

// 方法调用
const result = user?.getName?.();

// 数组访问
const arr = null;
const first = arr?.[0];
```

### 12.2 空值合并（Nullish Coalescing）

```javascript
// 传统默认值
const value = null;
const result = value || '默认值';
// 问题：0、'' 也会被当作假值

// 空值合并（ES2020）
const value1 = null;
const value2 = 0;
const value3 = '';

console.log(value1 ?? '默认值'); // 默认值
console.log(value2 ?? '默认值'); // 0（因为 null/undefined 才是默认值）
console.log(value3 ?? '默认值'); // ''

// 组合使用
const user = {
    name: '',
    age: 0
};

const displayName = user.name ?? '匿名';
const displayAge = user.age ?? '未知';

console.log(displayName); // ''（空字符串不是 null/undefined）
console.log(displayAge);  // 0
```

// API 响应处理
const response = {

    return {

    };
}

## 13. 装饰器基础

### 13.1 什么是装饰器？

装饰器是一种在不修改原函数的情况下，为其添加额外功能的语法。

```javascript
// 简单装饰器
function log(fn) {
    return function(...args) {
        console.log('调用函数：', fn.name);
        console.log('参数：', args);
        const result = fn.apply(this, args);
        console.log('返回值：', result);
        return result;
    };
}

@log
function add(a, b) {
    return a + b;
}

add(1, 2);
// 输出：
// 调用函数： add
// 参数： [1, 2]
// 返回值： 3
```

### 13.2 类装饰器

```javascript
// 类装饰器
function sealed(constructor) {
    Object.seal(constructor);
    Object.seal(constructor.prototype);
}

@sealed
class Person {
    constructor(name) {
        this.name = name;
    }
}

// 装饰器工厂
function color(color) {
    return function(target) {
        target.prototype.color = color;
    };
}

@color('blue')
class Car {
    drive() {
        console.log('Driving...');
    }
}

const car = new Car();
console.log(car.color); // blue
```

### 13.3 方法装饰器

```javascript
// 方法装饰器
function readonly(target, property, descriptor) {
    descriptor.writable = false;
    return descriptor;
}

class Calculator {
    @readonly
    add(a, b) {
        return a + b;
    }
}

const calc = new Calculator();
// calc.add = () => {}; // 报错：Cannot assign to read only property
```

### 13.4 多个装饰器

```javascript
// 多个装饰器从上到下执行
function first() {
    console.log('first: 初始化');
    return function(target, property, descriptor) {
        console.log('first: 修改');
    };
}

function second() {
    console.log('second: 初始化');
    return function(target, property, descriptor) {
        console.log('second: 修改');
    };
}

class MyClass {
    @first()
    @second()
    method() {}
}

// 输出：
// second: 初始化
// first: 初始化
// second: 修改
// first: 修改
```
