# 优化挑战

本章节将提供一系列优化挑战，帮助你提升代码性能和重构能力。

## 1. 性能优化挑战

### 挑战 1：优化大型列表渲染

**问题描述：**
项目需要展示一个包含 10000+ 个文件的项目浏览器，目前每次渲染都会导致页面卡顿。

**当前代码：**

```tsx
// 问题代码
function FileList({ files }: { files: File[] }) {
  return (
    <ul>
      {files.map((file) => (
        <FileItem key={file.id} file={file} />
      ))}
    </ul>
  );
}

function FileItem({ file }: { file: File }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <li>
      <div onClick={() => setExpanded(!expanded)}>
        {file.name}
      </div>
      {expanded && file.children && (
        <ul>
          {file.children.map((child) => (
            <FileItem key={child.id} file={child} />
          ))}
        </ul>
      )}
    </li>
  );
}
```

**优化目标：**
1. 使用虚拟列表渲染
2. 减少不必要的重渲染
3. 添加展开/收起动画

**答案：**

```tsx
import { useVirtualizer } from '@tanstack/react-virtual';
import { memo, useState, useCallback, useMemo } from 'react';

// 使用 memo 缓存 FileItem
const FileItem = memo(function FileItem({
  file,
  depth = 0,
  onToggle,
}: {
  file: File;
  depth?: number;
  onToggle?: (id: string) => void;
}) {
  return (
    <div
      className="flex items-center py-1 px-2 hover:bg-gray-100 cursor-pointer"
      style={{ paddingLeft: `${depth * 16 + 8}px` }}
      onClick={() => onToggle?.(file.id)}
    >
      {file.children && (
        <span className="mr-1 text-xs">
          {file.isExpanded ? '▼' : '▶'}
        </span>
      )}
      <span className="mr-2">{file.isFolder ? '📁' : '📄'}</span>
      <span>{file.name}</span>
    </div>
  );
});

// 展平文件树
function flattenTree(
  files: File[],
  expandedIds: Set<string>
): { file: File; depth: number }[] {
  const result: { file: File; depth: number }[] = [];

  function traverse(items: File[], depth: number) {
    for (const item of items) {
      result.push({ file: item, depth });
      if (item.children && expandedIds.has(item.id)) {
        traverse(item.children, depth + 1);
      }
    }
  }

  traverse(files, 0);
  return result;
}

// 虚拟列表组件
function VirtualFileList({ files }: { files: File[] }) {
  const [expandedIds, setExpandedIds] = useState<Set<string>>(new Set());

  // 展平树结构
  const flatItems = useMemo(
    () => flattenTree(files, expandedIds),
    [files, expandedIds]
  );

  // 虚拟化
  const parentRef = useRef<HTMLDivElement>(null);

  const virtualizer = useVirtualizer({
    count: flatItems.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 32,
    overscan: 10,
  });

  const handleToggle = useCallback((id: string) => {
    setExpandedIds((prev) => {
      const next = new Set(prev);
      if (next.has(id)) {
        next.delete(id);
      } else {
        next.add(id);
      }
      return next;
    });
  }, []);

  return (
    <div
      ref={parentRef}
      className="h-full overflow-auto"
      style={{ height: '500px' }}
    >
      <div
        style={{
          height: `${virtualizer.getTotalSize()}px`,
          width: '100%',
          position: 'relative',
        }}
      >
        {virtualizer.getVirtualItems().map((virtualItem) => {
          const { file, depth } = flatItems[virtualItem.index];
          return (
            <div
              key={file.id}
              style={{
                position: 'absolute',
                top: 0,
                left: 0,
                width: '100%',
                height: `${virtualItem.size}px`,
                transform: `translateY(${virtualItem.start}px)`,
              }}
            >
              <FileItem
                file={{ ...file, isExpanded: expandedIds.has(file.id) }}
                depth={depth}
                onToggle={handleToggle}
              />
            </div>
          );
        })}
      </div>
    </div>
  );
}

// 类型定义
interface File {
  id: string;
  name: string;
  isFolder: boolean;
  children?: File[];
  isExpanded?: boolean;
}
```

### 挑战 2：优化 Monaco Editor 性能

**问题描述：**
当编辑大文件（10000+ 行）时，编辑器响应变慢，需要优化性能。

**优化方案：**

```tsx
import Editor from '@monaco-editor/react';
import { useRef, useCallback, useMemo } from 'react';
import * as monaco from 'monaco-editor';

function OptimizedLargeFileEditor({
  content,
  language = 'javascript',
  onChange,
}: {
  content: string;
  language?: string;
  onChange?: (value: string) => void;
}) {
  const editorRef = useRef<monaco.editor.IStandaloneCodeEditor | null>(null);
  const monacoRef = useRef<typeof monaco | null>(null);

  // 编辑器挂载完成
  const handleEditorDidMount = useCallback(
    (editor: monaco.editor.IStandaloneCodeEditor, monaco: typeof Monaco) => {
      editorRef.current = editor;
      monacoRef.current = monaco;

      // 性能优化配置
      editor.updateOptions({
        // 禁用不必要的功能
        minimap: { enabled: false },
        folding: false,
        glyphMargin: false,
        lineNumbers: 'on',
        renderWhitespace: 'selection',
        smoothScrolling: false,
        cursorBlinking: 'solid',
        cursorSmoothCaretAnimation: 'off',

        // 优化渲染
        maxTokenizationLineLength: 5000,
        automaticLayout: true,

        // 滚动优化
        scrollBeyondLastLine: false,
        fastScrollSensitivity: 5,
      });

      // 大文件警告
      const lineCount = editor.getModel()?.getLineCount() || 0;
      if (lineCount > 5000) {
        console.warn(`Large file detected: ${lineCount} lines`);
      }

      // 限制撤销历史
      editor.setModel(editor.getModel());
    },
    []
  );

  // 变化事件优化
  const handleChange = useCallback(
    (value: string | undefined) => {
      // 使用防抖处理大量输入
      onChange?.(value || '');
    },
    [onChange]
  );

  // 只在语言变化时更新
  const languageValue = useMemo(() => language, [language]);

  return (
    <Editor
      height="100%"
      language={languageValue}
      value={content}
      onChange={handleChange}
      onMount={handleEditorDidMount}
      theme="vs-dark"
      loading={<div className="p-4">Loading editor...</div>}
      options={{
        // 基础配置
        fontSize: 14,
        fontFamily: "'Fira Code', monospace",
        tabSize: 2,

        // 性能相关
        renderLineHighlight: 'none',
        overviewRulerBorder: false,
        hideCursorInOverviewRuler: true,
        scrollbar: {
          vertical: 'auto',
          horizontal: 'auto',
          verticalScrollbarSize: 10,
          horizontalScrollbarSize: 10,
        },
      }}
    />
  );
}
```

### 挑战 3：优化 Zustand 状态更新

**问题描述：**
状态更新时导致整个应用重新渲染，需要优化选择器使用。

**当前代码：**

```tsx
// 问题代码
function App() {
  const user = useStore((state) => state.user);
  const theme = useStore((state) => state.theme);
  const files = useStore((state) => state.files);
  const settings = useStore((state) => state.settings);

  return (
    <div>
      <Header user={user} theme={theme} />
      <FileList files={files} />
      <SettingsPanel settings={settings} />
    </div>
  );
}
```

**优化方案：**

```tsx
import { create } from 'zustand';
import { shallow } from 'zustand/shallow';

// 创建分离的 store
interface AppStore {
  user: User | null;
  theme: 'light' | 'dark';
  files: File[];
  settings: Settings;
  actions: {
    setUser: (user: User | null) => void;
    setTheme: (theme: 'light' | 'dark') => void;
    setFiles: (files: File[]) => void;
    updateSettings: (settings: Partial<Settings>) => void;
  };
}

const useAppStore = create<AppStore>((set) => ({
  user: null,
  theme: 'light',
  files: [],
  settings: {},
  actions: {
    setUser: (user) => set({ user }),
    setTheme: (theme) => set({ theme }),
    setFiles: (files) => set({ files }),
    updateSettings: (newSettings) =>
      set((state) => ({ settings: { ...state.settings, ...newSettings } })),
  },
}));

// 优化后的组件
function App() {
  return (
    <div>
      <Header />
      <FileList />
      <SettingsPanel />
    </div>
  );
}

// 每个组件只订阅需要的状态
function Header() {
  // 使用 shallow 比较对象
  const user = useAppStore((state) => state.user, shallow);
  const theme = useAppStore((state) => state.theme);

  return (
    <header className={theme}>
      {user?.name}
    </header>
  );
}

function FileList() {
  // 使用 shallow 比较数组
  const files = useAppStore((state) => state.files, shallow);

  return (
    <ul>
      {files.map((file) => (
        <FileItem key={file.id} file={file} />
      ))}
    </ul>
  );
}

function SettingsPanel() {
  const settings = useAppStore((state) => state.settings, shallow);

  return <div>{JSON.stringify(settings)}</div>;
}

// 进一步优化：创建自定义 hook
function useUser() {
  return useAppStore((state) => state.user);
}

function useTheme() {
  return useAppStore((state) => state.theme);
}

function useFiles() {
  return useAppStore((state) => state.files, shallow);
}

function useSettings() {
  return useAppStore((state) => state.settings, shallow);
}
```

### 挑战 4：重构类组件为函数组件

**原始代码（类组件）：**

```tsx
// 原始类组件
class UserProfile extends React.Component<
  { userId: string },
  { user: User | null; loading: boolean; error: string | null }
> {
  state = {
    user: null,
    loading: true,
    error: null,
  };

  componentDidMount() {
    this.fetchUser();
  }

  componentDidUpdate(prevProps: { userId: string }) {
    if (prevProps.userId !== this.props.userId) {
      this.fetchUser();
    }
  }

  async fetchUser() {
    this.setState({ loading: true, error: null });
    try {
      const response = await fetch(`/api/users/${this.props.userId}`);
      if (!response.ok) throw new Error('Failed to fetch');
      const user = await response.json();
      this.setState({ user, loading: false });
    } catch (error) {
      this.setState({
        error: error instanceof Error ? error.message : 'Unknown error',
        loading: false,
      });
    }
  }

  render() {
    const { user, loading, error } = this.state;

    if (loading) return <div>Loading...</div>;
    if (error) return <div>Error: {error}</div>;
    if (!user) return <div>No user found</div>;

    return (
      <div>
        <h1>{user.name}</h1>
        <p>{user.email}</p>
      </div>
    );
  }
}
```

**重构为函数组件：**

```tsx
// 重构后的函数组件
import { useState, useEffect, useCallback } from 'react';

interface User {
  id: string;
  name: string;
  email: string;
}

interface UserProfileProps {
  userId: string;
}

function UserProfile({ userId }: UserProfileProps) {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchUser = useCallback(async () => {
    setLoading(true);
    setError(null);

    try {
      const response = await fetch(`/api/users/${userId}`);
      if (!response.ok) throw new Error('Failed to fetch');
      const data = await response.json();
      setUser(data);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
    } finally {
      setLoading(false);
    }
  }, [userId]);

  useEffect(() => {
    fetchUser();
  }, [fetchUser]);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  if (!user) return <div>No user found</div>;

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}

// 进一步优化：提取自定义 Hook
function useUser(userId: string) {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;

    async function fetchUser() {
      setLoading(true);
      setError(null);

      try {
        const response = await fetch(`/api/users/${userId}`);
        if (!response.ok) throw new Error('Failed to fetch');
        const data = await response.json();
        if (!cancelled) {
          setUser(data);
        }
      } catch (err) {
        if (!cancelled) {
          setError(err instanceof Error ? err.message : 'Unknown error');
        }
      } finally {
        if (!cancelled) {
          setLoading(false);
        }
      }
    }

    fetchUser();

    return () => {
      cancelled = true;
    };
  }, [userId]);

  return { user, loading, error };
}

function UserProfileRefactored({ userId }: UserProfileProps) {
  const { user, loading, error } = useUser(userId);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  if (!user) return <div>No user found</div>;

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}
```

```tsx
const useUser = () => useAppStore((state) => state.user);
const useTheme = () => useAppStore((state) => state.theme);

  const theme = useTheme();
  const setTheme = useSetTheme();

  return (

      </button>
    </div>
  );
}
```

### 挑战 6：实现实时协作编辑

**需求：**
1. 使用 Yjs 实现多个用户同时编辑同一个文档
2. 显示其他用户的光标位置
3. 同步用户列表

**实现方案：**

```tsx
import { useEffect, useState, useRef, useCallback } from 'react';
import * as Y from 'yjs';
import { WebsocketProvider } from 'y-websocket';
import Editor from '@monaco-editor/react';
import * as monaco from 'monaco-editor';

interface Collaborator {
  id: string;
  name: string;
  color: string;
  cursor?: { lineNumber: number; column: number };
}

interface CollaborativeEditorProps {
  documentId: string;
  userName: string;
}

export function CollaborativeEditor({
  documentId,
  userName,
}: CollaborativeEditorProps) {
  const [collaborators, setCollaborators] = useState<Collaborator[]>([]);
  const editorRef = useRef<monaco.editor.IStandaloneCodeEditor | null>(null);
  const monacoRef = useRef<typeof monaco | null>(null);
  const ydocRef = useRef<Y.Doc | null>(null);

  useEffect(() => {
    // 创建 Yjs 文档
    const ydoc = new Y.Doc();
    ydocRef.current = ydoc;

    // 创建 WebSocket Provider
    const provider = new WebsocketProvider(
      'wss://demos.yjs.dev',
      documentId,
      ydoc
    );

    // 获取共享文本
    const ytext = ydoc.getText('monaco');

    // 监听其他用户的更新
    provider.awareness.setLocalStateField('user', {
      name: userName,
      color: '#' + Math.floor(Math.random() * 16777215).toString(16),
    });

    // 监听用户变化
    const handleAwarenessChange = () => {
      const states = Array.from(provider.awareness.getStates().entries());
      const users: Collaborator[] = states
        .filter(([clientId]) => clientId !== ydoc.clientID)
        .map(([clientId, state]: [number, any]) => ({
          id: clientId.toString(),
          name: state.user?.name || 'Anonymous',
          color: state.user?.color || '#888',
          cursor: state.cursor,
        }));
      setCollaborators(users);
    };

    provider.awareness.on('change', handleAwarenessChange);

    // 同步编辑器内容
    if (editorRef.current && monacoRef.current) {
      const editor = editorRef.current;

      // 监听本地变化并同步到 Yjs
      editor.onDidChangeModelContent((event) => {
        ydoc.transact(() => {
          event.changes.forEach((change) => {
            if (change.text) {
              ytext.insert(change.rangeOffset, change.text);
            }
            if (change.rangeLength > 0) {
              ytext.delete(change.rangeOffset, change.rangeLength);
            }
          });
        });
      });

      // 监听 Yjs 变化并更新编辑器
      ytext.observe((event) => {
        const model = editor.getModel();
        if (!model) return;

        // 简单实现：重新设置整个内容
        // 生产环境应使用 operation transform
        const content = ytext.toString();
        if (editor.getValue() !== content) {
          const position = editor.getPosition();
          editor.setValue(content);
          if (position) {
            editor.setPosition(position);
          }
        }
      });

      // 监听光标位置变化
      editor.onDidChangeCursorPosition((event) => {
        provider.awareness.setLocalStateField('cursor', {
          lineNumber: event.position.lineNumber,
          column: event.position.column,
        });
      });
    }

    // 清理
    return () => {
      provider.disconnect();
      ydoc.destroy();
    };
  }, [documentId, userName]);

  const handleEditorDidMount = (
    editor: monaco.editor.IStandaloneCodeEditor,
    monaco: typeof Monaco
  ) => {
    editorRef.current = editor;
    monacoRef.current = monaco;
  };

  return (
    <div className="relative">
      {/* 协作者列表 */}
      <div className="absolute top-2 right-2 z-10 flex gap-2">
        {collaborators.map((user) => (
          <div
            key={user.id}
            className="px-2 py-1 rounded text-xs text-white"
            style={{ backgroundColor: user.color }}
          >
            {user.name}
          </div>
        ))}
      </div>

      <Editor
        height="500px"
        defaultLanguage="javascript"
        onMount={handleEditorDidMount}
        theme="vs-dark"
      />
    </div>
  );
}
```

### 挑战 7：实现撤销/重做功能

**需求：**
1. 为编辑器实现本地撤销/重做
2. 支持键盘快捷键
3. 添加 UI 按钮

**实现方案：**

```tsx
import { useState, useCallback, useRef, useEffect } from 'react';
import Editor from '@monaco-editor/react';
import * as monaco from 'monaco-editor';

interface HistoryEntry {
  content: string;
  cursorPosition: monaco.Position;
}

interface UseUndoRedoOptions {
  initialValue?: string;
}

function useUndoRedo({ initialValue = '' }: UseUndoRedoOptions = {}) {
  const [value, setValue] = useState(initialValue);
  const [history, setHistory] = useState<HistoryEntry[]>([{ content: initialValue, cursorPosition: new monaco.Position(1, 1) }]);
  const [historyIndex, setHistoryIndex] = useState(0);

  const pushHistory = useCallback((newEntry: HistoryEntry) => {
    setHistory((prev) => {
      // 移除当前索引之后的所有历史
      const newHistory = prev.slice(0, historyIndex + 1);
      // 添加新条目
      return [...newHistory, newEntry];
    });
    setHistoryIndex((prev) => prev + 1);
  }, [historyIndex]);

  const undo = useCallback(() => {
    if (historyIndex > 0) {
      setHistoryIndex((prev) => prev - 1);
      return history[historyIndex - 1];
    }
    return null;
  }, [history, historyIndex]);

  const redo = useCallback(() => {
    if (historyIndex < history.length - 1) {
      setHistoryIndex((prev) => prev + 1);
      return history[historyIndex + 1];
    }
    return null;
  }, [history, historyIndex]);

  const updateValue = useCallback((newValue: string, cursorPosition?: monaco.Position) => {
    setValue(newValue);
    pushHistory({
      content: newValue,
      cursorPosition: cursorPosition || new monaco.Position(1, 1),
    });
  }, [pushHistory]);

  return {
    value,
    setValue: updateValue,
    undo,
    redo,
    canUndo: historyIndex > 0,
    canRedo: historyIndex < history.length - 1,
  };
}

function EditorWithUndoRedo() {
  const editorRef = useRef<monaco.editor.IStandaloneCodeEditor | null>(null);
  const {
    value,
    setValue,
    undo,
    redo,
    canUndo,
    canRedo,
  } = useUndoRedo({ initialValue: '// Start coding here\n' });

  const handleEditorMount = (editor: monaco.editor.IStandaloneCodeEditor) => {
    editorRef.current = editor;

    // 添加键盘快捷键
    editor.addCommand(monaco.KeyMod.CtrlCmd | monaco.KeyCode.KeyZ, () => {
      const entry = undo();
      if (entry) {
        editor.setValue(entry.content);
        editor.setPosition(entry.cursorPosition);
      }
    });

    editor.addCommand(monaco.KeyMod.CtrlCmd | monaco.KeyCode.KeyY, () => {
      const entry = redo();
      if (entry) {
        editor.setValue(entry.content);
        editor.setPosition(entry.cursorPosition);
      }
    });

    // Mac 的 Redo 是 Cmd+Shift+Z
    editor.addCommand(
      monaco.KeyMod.CtrlCmd | monaco.KeyMod.Shift | monaco.KeyCode.KeyZ,
      () => {
        const entry = redo();
        if (entry) {
          editor.setValue(entry.content);
          editor.setPosition(entry.cursorPosition);
        }
      }
    );
  };

  const handleChange = (newValue: string | undefined) => {
    if (newValue !== undefined && editorRef.current) {
      setValue(newValue, editorRef.current.getPosition());
    }
  };

  return (
    <div>
      <div className="flex gap-2 mb-2">
        <button
          onClick={() => {
            const entry = undo();
            if (entry && editorRef.current) {
              editorRef.current.setValue(entry.content);
              editorRef.current.setPosition(entry.cursorPosition);
            }
          }}
          disabled={!canUndo}
          className="px-3 py-1 bg-gray-200 rounded disabled:opacity-50"
        >
          Undo (Ctrl+Z)
        </button>
        <button
          onClick={() => {
            const entry = redo();
            if (entry && editorRef.current) {
              editorRef.current.setValue(entry.content);
              editorRef.current.setPosition(entry.cursorPosition);
            }
          }}
          disabled={!canRedo}
          className="px-3 py-1 bg-gray-200 rounded disabled:opacity-50"
        >
          Redo (Ctrl+Y)
        </button>
      </div>

      <Editor
        height="400px"
        defaultLanguage="javascript"
        value={value}
        onChange={handleChange}
        onMount={handleEditorMount}
        theme="vs-dark"
      />
    </div>
  );
}
```

### 挑战 8：实现主题切换功能

**需求：**
1. 支持亮色/暗色主题切换
2. Monaco Editor 主题同步
3. 持久化主题设置

**实现方案：**

```tsx
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { useEffect, useState } from 'react';

type Theme = 'light' | 'dark';

interface ThemeStore {
  theme: Theme;
  setTheme: (theme: Theme) => void;
  toggleTheme: () => void;
}

const useThemeStore = create<ThemeStore>()(
  persist(
    (set, get) => ({
      theme: 'dark',
      setTheme: (theme) => set({ theme }),
      toggleTheme: () =>
        set({ theme: get().theme === 'dark' ? 'light' : 'dark' }),
    }),
    {
      name: 'theme-storage',
    }
  )
);

// 主题应用 Hook
function useTheme() {
  const theme = useThemeStore((state) => state.theme);
  const toggleTheme = useThemeStore((state) => state.toggleTheme);

  useEffect(() => {
    // 应用到 HTML 根元素
    document.documentElement.classList.remove('light', 'dark');
    document.documentElement.classList.add(theme);
  }, [theme]);

  return { theme, toggleTheme };
}

// 主题切换组件
function ThemeToggle() {
  const { theme, toggleTheme } = useTheme();

  return (
    <button
      onClick={toggleTheme}
      className="p-2 rounded-lg bg-gray-200 dark:bg-gray-700"
    >
      {theme === 'dark' ? '🌙' : '☀️'}
    </button>
  );
}

// 带主题的 Monaco Editor
function ThemedEditor() {
  const { theme } = useTheme();

  return (
    <Editor
      height="500px"
      theme={theme === 'dark' ? 'vs-dark' : 'vs'}
      defaultLanguage="javascript"
      defaultValue="// Write your code here"
    />
  );
}

// 完整的应用组件
function App() {
  const { theme } = useTheme();

  return (
    <div className={`min-h-screen ${theme === 'dark' ? 'bg-gray-900 text-white' : 'bg-white text-gray-900'}`}>
      <header className="flex justify-between items-center p-4 border-b">
        <h1>WebEnv IDE</h1>
        <ThemeToggle />
      </header>
      <main className="p-4">
        <ThemedEditor />
      </main>
    </div>
  );
}
```

### 挑战 9：实现文件搜索功能

**需求：**
1. 快速搜索文件
2. 支持模糊匹配
3. 键盘导航支持

```tsx
import { useState, useMemo, useCallback, useRef, useEffect } from 'react';
import Editor from '@monaco-editor/react';

interface FileItem {
  id: string;
  name: string;
  path: string;
  language?: string;
}

interface FileSearchProps {
  files: FileItem[];
  onFileSelect: (file: FileItem) => void;
}

function FileSearchModal({ files, onFileSelect }: FileSearchProps) {
  const [query, setQuery] = useState('');
  const [selectedIndex, setSelectedIndex] = useState(0);
  const inputRef = useRef<HTMLInputElement>(null);

  // 模糊搜索
  const filteredFiles = useMemo(() => {
    if (!query) return files.slice(0, 10);

    const lowerQuery = query.toLowerCase();
    return files
      .filter((file) => file.name.toLowerCase().includes(lowerQuery))
      .slice(0, 10);
  }, [files, query]);

  // 键盘导航
  const handleKeyDown = useCallback(
    (e: React.KeyboardEvent) => {
      switch (e.key) {
        case 'ArrowDown':
          e.preventDefault();
          setSelectedIndex((prev) =>
            prev < filteredFiles.length - 1 ? prev + 1 : prev
          );
          break;
        case 'ArrowUp':
          e.preventDefault();
          setSelectedIndex((prev) => (prev > 0 ? prev - 1 : prev));
          break;
        case 'Enter':
          e.preventDefault();
          if (filteredFiles[selectedIndex]) {
            onFileSelect(filteredFiles[selectedIndex]);
          }
          break;
        case 'Escape':
          setQuery('');
          break;
      }
    },
    [filteredFiles, selectedIndex, onFileSelect]
  );

  // 全局快捷键打开搜索
  useEffect(() => {
    const handleGlobalKeyDown = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'p') {
        e.preventDefault();
        inputRef.current?.focus();
      }
    };

    window.addEventListener('keydown', handleGlobalKeyDown);
    return () => window.removeEventListener('keydown', handleGlobalKeyDown);
  }, []);

  return (
    <div className="w-full max-w-md mx-auto">
      <input
        ref={inputRef}
        type="text"
        value={query}
        onChange={(e) => {
          setQuery(e.target.value);
          setSelectedIndex(0);
        }}
        onKeyDown={handleKeyDown}
        placeholder="Search files... (Ctrl+P)"
        className="w-full px-4 py-2 border rounded-lg"
        autoFocus
      />

      <ul className="mt-2 border rounded-lg overflow-hidden">
        {filteredFiles.map((file, index) => (
          <li
            key={file.id}
            className={`px-4 py-2 cursor-pointer flex justify-between ${
              index === selectedIndex
                ? 'bg-blue-500 text-white'
                : 'hover:bg-gray-100'
            }`}
            onClick={() => onFileSelect(file)}
            onMouseEnter={() => setSelectedIndex(index)}
          >
            <span>{file.name}</span>
            <span className="text-sm opacity-60">{file.path}</span>
          </li>
        ))}
      </ul>

      {filteredFiles.length === 0 && (
        <p className="text-center text-gray-500 mt-4">No files found</p>
      )}
    </div>
  );
}

// 使用示例
function App() {
  const files: FileItem[] = [
    { id: '1', name: 'index.js', path: 'src/index.js', language: 'javascript' },
    { id: '2', name: 'App.tsx', path: 'src/App.tsx', language: 'typescript' },
    { id: '3', name: 'styles.css', path: 'src/styles.css', language: 'css' },
  ];

  const handleFileSelect = (file: FileItem) => {
    console.log('Selected:', file);
  };

  return (
    <div className="p-8">
      <h1 className="text-2xl mb-4">File Search</h1>
      <FileSearchModal files={files} onFileSelect={handleFileSelect} />
    </div>
  );
}
```
