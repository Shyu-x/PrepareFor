# 项目功能实现

本章节将帮助你实现 WebEnv 项目中的核心功能，包括 IDE 布局、窗口管理系统、文件树组件和编辑器标签页。

## 1. 实现一个简单的 IDE 布局

### 1.1 项目结构分析

**目标：** 创建一个类似 VSCode 的 IDE 布局

**布局结构：**
```
+------------------+------------------------+
|     Sidebar      |       Editor          |
|  (文件资源管理器)  |    (代码编辑器)        |
|                  |                       |
|  - 文件树        |  - 标签页             |
|                  |  - 代码编辑区         |
+------------------+------------------------+
|              Terminal (终端)              |
+------------------------------------------+
```

```tsx
// components/Terminal.tsx
'use client';

import { useEffect, useRef } from 'react';
import { Terminal as XTerm } from 'xterm';
import { FitAddon } from 'xterm-addon-fit';
import 'xterm/css/xterm.css';

export function Terminal() {
  const terminalRef = useRef<HTMLDivElement>(null);
  const xtermRef = useRef<XTerm | null>(null);

  useEffect(() => {
    if (!terminalRef.current) return;

    // 创建终端实例
    const terminal = new XTerm({
      fontSize: 14,
      fontFamily: 'Menlo, Monaco, "Courier New", monospace',
      theme: {
        background: '#1e1e1e',
        foreground: '#cccccc',
      },
      cursorBlink: true,
    });

    // 添加 fit addon
    const fitAddon = new FitAddon();
    terminal.loadAddon(fitAddon);

    // 打开终端
    terminal.open(terminalRef.current);
    fitAddon.fit();

    // 写入欢迎信息
    terminal.writeln('\x1b[32mWelcome to WebEnv Terminal\x1b[0m');
    terminal.writeln('Type "help" for available commands.');
    terminal.write('\r\n$ ');

    // 处理输入
    terminal.onData((data) => {
      // 简单的回显
      terminal.write(data);

      // 按下回车时处理命令
      if (data === '\r') {
        terminal.write('\r\n$ ');
      }
    });

    xtermRef.current = terminal;

    // 处理窗口大小变化
    const handleResize = () => {
      fitAddon.fit();
    };

    window.addEventListener('resize', handleResize);

    return () => {
      window.removeEventListener('resize', handleResize);
      terminal.dispose();
    };
  }, []);

  return <div ref={terminalRef} className="h-full w-full bg-[#1e1e1e]" />;
}
```

```tsx
      },
    ],
  },
  {

    language: 'json',
    content: '{\n  "name": "webenv",\n  "version": "1.0.0"\n}',
  },
];

  const [activeFileId, setActiveFileId] = useState<string>('2');

  // 查找文件内容
  const findFile = (
    files: FileNode[],
    id: string
  ): FileNode | undefined => {
    for (const file of files) {
      if (file.id === id) return file;
      if (file.children) {
        const found = findFile(file.children, id);
        if (found) return found;
      }
    }
    return undefined;
  };

  const activeFile = findFile(initialFiles, activeFileId);

  return (
    <IDELayout
      sidebar={
        <Sidebar
          files={initialFiles}
          activeFile={activeFileId}
          onFileSelect={setActiveFileId}
        />
      }
      editor={
        <EditorArea
          language={activeFile?.language}
          value={activeFile?.content}
        />
      }
      terminal={<Terminal />}
    />
  );
}
```

## 2. 实现窗口管理系统

### 2.1 窗口状态管理

```tsx
// stores/windowStore.ts
import { create } from 'zustand';

interface Window {
  id: string;
  title: string;
  content: React.ReactNode;
  position: { x: number; y: number };
  size: { width: number; height: number };
  zIndex: number;
  isMinimized: boolean;
  isMaximized: boolean;
}

interface WindowStore {
  windows: Window[];
  activeWindowId: string | null;

  addWindow: (window: Omit<Window, 'zIndex'>) => void;
  removeWindow: (id: string) => void;
  focusWindow: (id: string) => void;
  minimizeWindow: (id: string) => void;
  maximizeWindow: (id: string) => void;
  restoreWindow: (id: string) => void;
  moveWindow: (id: string, position: { x: number; y: number }) => void;
  resizeWindow: (id: string, size: { width: number; height: number }) => void;
}

let windowCounter = 1;

export const useWindowStore = create<WindowStore>((set, get) => ({
  windows: [],
  activeWindowId: null,

  addWindow: (window) =>
    set((state) => {
      const maxZIndex = Math.max(0, ...state.windows.map((w) => w.zIndex));
      const newWindow: Window = {
        ...window,
        zIndex: maxZIndex + 1,
      };
      return {
        windows: [...state.windows, newWindow],
        activeWindowId: newWindow.id,
      };
    }),

  removeWindow: (id) =>
    set((state) => ({
      windows: state.windows.filter((w) => w.id !== id),
      activeWindowId:
        state.activeWindowId === id ? null : state.activeWindowId,
    })),

  focusWindow: (id) =>
    set((state) => {
      const maxZIndex = Math.max(
        0,
        ...state.windows.map((w) => w.zIndex)
      );
      return {
        windows: state.windows.map((w) =>
          w.id === id ? { ...w, zIndex: maxZIndex + 1 } : w
        ),
        activeWindowId: id,
      };
    }),

  minimizeWindow: (id) =>
    set((state) => ({
      windows: state.windows.map((w) =>
        w.id === id ? { ...w, isMinimized: true } : w
      ),
    })),

  maximizeWindow: (id) =>
    set((state) => ({
      windows: state.windows.map((w) =>
        w.id === id
          ? { ...w, isMaximized: true, position: { x: 0, y: 0 } }
          : w
      ),
    })),

  restoreWindow: (id) =>
    set((state) => ({
      windows: state.windows.map((w) =>
        w.id === id ? { ...w, isMinimized: false, isMaximized: false } : w
      ),
    })),

  moveWindow: (id, position) =>
    set((state) => ({
      windows: state.windows.map((w) =>
        w.id === id ? { ...w, position } : w
      ),
    })),

  resizeWindow: (id, size) =>
    set((state) => ({
      windows: state.windows.map((w) =>
        w.id === id ? { ...w, size } : w
      ),
    })),
}));
```

### 2.2 窗口组件

```tsx
// components/Window.tsx
'use client';

import { useRef, useState, useEffect } from 'react';
import { useWindowStore } from '@/stores/windowStore';

interface WindowProps {
  id: string;
  title: string;
  children: React.ReactNode;
  initialPosition?: { x: number; y: number };
  initialSize?: { width: number; height: number };
}

export function Window({
  id,
  title,
  children,
  initialPosition = { x: 100, y: 100 },
  initialSize = { width: 600, height: 400 },
}: WindowProps) {
  const {
    focusWindow,
    minimizeWindow,
    maximizeWindow,
    restoreWindow,
    moveWindow,
    resizeWindow,
    removeWindow,
  } = useWindowStore();

  const windowRef = useRef<HTMLDivElement>(null);
  const [isDragging, setIsDragging] = useState(false);
  const [isResizing, setIsResizing] = useState(false);
  const [dragStart, setDragStart] = useState({ x: 0, y: 0 });

  const window = useWindowStore((state) =>
    state.windows.find((w) => w.id === id)
  );

  if (!window) return null;
  if (window.isMinimized) return null;

  const handleMouseDown = (e: React.MouseEvent) => {
    focusWindow(id);
    setIsDragging(true);
    setDragStart({
      x: e.clientX - window.position.x,
      y: e.clientY - window.position.y,
    });
  };

  const handleMouseMove = (e: React.MouseEvent) => {
    if (isDragging && !window.isMaximized) {
      moveWindow(id, {
        x: e.clientX - dragStart.x,
        y: e.clientY - dragStart.y,
      });
    }
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  return (
    <div
      ref={windowRef}
      className="absolute bg-gray-900 border border-gray-700 rounded-lg shadow-2xl overflow-hidden flex flex-col"
      style={{
        left: window.position.x,
        top: window.position.y,
        width: window.isMaximized ? '100%' : window.size.width,
        height: window.isMaximized ? 'calc(100% - 40px)' : window.size.height,
        zIndex: window.zIndex,
      }}
      onClick={() => focusWindow(id)}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
      onMouseLeave={handleMouseUp}
    >
      {/* 标题栏 */}
      <div
        className="h-8 bg-gray-800 flex items-center px-2 cursor-move select-none"
        onMouseDown={handleMouseDown}
      >
        {/* 窗口按钮 */}
        <div className="flex gap-2 mr-4">
          <button
            className="w-3 h-3 rounded-full bg-red-500 hover:bg-red-600"
            onClick={(e) => {
              e.stopPropagation();
              removeWindow(id);
            }}
          />
          <button
            className="w-3 h-3 rounded-full bg-yellow-500 hover:bg-yellow-600"
            onClick={(e) => {
              e.stopPropagation();
              minimizeWindow(id);
            }}
          />
          <button
            className="w-3 h-3 rounded-full bg-green-500 hover:bg-green-600"
            onClick={(e) => {
              e.stopPropagation();
              if (window.isMaximized) {
                restoreWindow(id);
              } else {
                maximizeWindow(id);
              }
            }}
          />
        </div>

        {/* 标题 */}
        <span className="flex-1 text-center text-sm text-gray-300">
          {title}
        </span>
      </div>

      {/* 内容区 */}
      <div className="flex-1 overflow-auto">{children}</div>
    </div>
  );
}
```

## 3. 实现文件树组件

### 3.1 完整的文件树组件

```tsx
// components/FileTree.tsx
'use client';

import { useState, useCallback } from 'react';

export interface FileNode {
  id: string;
  name: string;
  type: 'file' | 'folder';
  language?: string;
  children?: FileNode[];
}

interface FileTreeProps {
  files: FileNode[];
  activeFileId?: string;
  onFileSelect: (file: FileNode) => void;
  onFileCreate?: (parentId: string | null, name: string, type: 'file' | 'folder') => void;
  onFileDelete?: (fileId: string) => void;
  onFileRename?: (fileId: string, newName: string) => void;
}

export function FileTree({
  files,
  activeFileId,
  onFileSelect,
  onFileCreate,
  onFileDelete,
  onFileRename,
}: FileTreeProps) {
  return (
    <div className="text-sm">
      {files.map((file) => (
        <FileTreeNode
          key={file.id}
          node={file}
          activeFileId={activeFileId}
          onFileSelect={onFileSelect}
          onFileCreate={onFileCreate}
          onFileDelete={onFileDelete}
          onFileRename={onFileRename}
          level={0}
        />
      ))}
    </div>
  );
}

interface FileTreeNodeProps {
  node: FileNode;
  activeFileId?: string;
  onFileSelect: (file: FileNode) => void;
  onFileCreate?: (parentId: string | null, name: string, type: 'file' | 'folder') => void;
  onFileDelete?: (fileId: string) => void;
  onFileRename?: (fileId: string, newName: string) => void;
  level: number;
}

function FileTreeNode({
  node,
  activeFileId,
  onFileSelect,
  onFileCreate,
  onFileDelete,
  onFileRename,
  level,
}: FileTreeNodeProps) {
  const [isExpanded, setIsExpanded] = useState(true);
  const [isEditing, setIsEditing] = useState(false);
  const [editName, setEditName] = useState(node.name);

  const handleClick = () => {
    if (node.type === 'folder') {
      setIsExpanded(!isExpanded);
    } else {
      onFileSelect(node);
    }
  };

  const handleRename = () => {
    if (editName.trim() && editName !== node.name) {
      onFileRename?.(node.id, editName.trim());
    }
    setIsEditing(false);
  };

  const getFileIcon = (node: FileNode) => {
    if (node.type === 'folder') {
      return isExpanded ? '📂' : '📁';
    }

    const iconMap: Record<string, string> = {
      js: '🟨',
      jsx: '⚛️',
      ts: '🔷',
      tsx: '⚛️',
      json: '📋',
      css: '🎨',
      html: '🌐',
      md: '📝',
      png: '🖼️',
      jpg: '🖼️',
      svg: '🖼️',
    };

    const ext = node.name.split('.').pop() || '';
    return iconMap[ext] || '📄';
  };

  const isActive = node.id === activeFileId;

  return (
    <div>
      <div
        className={`flex items-center px-2 py-1 cursor-pointer hover:bg-gray-700 ${
          isActive ? 'bg-blue-600 text-white' : 'text-gray-300'
        }`}
        style={{ paddingLeft: `${level * 12 + 8}px` }}
        onClick={handleClick}
      >
        {/* 展开/折叠箭头 */}
        {node.type === 'folder' && (
          <span className="w-4 text-xs mr-1">
            {isExpanded ? '▼' : '▶'}
          </span>
        )}

        {/* 文件夹图标 */}
        {node.type === 'folder' && <span className="mr-2">{getFileIcon(node)}</span>}

        {/* 文件名 */}
        {isEditing ? (
          <input
            type="text"
            value={editName}
            onChange={(e) => setEditName(e.target.value)}
            onBlur={handleRename}
            onKeyDown={(e) => {
              if (e.key === 'Enter') handleRename();
              if (e.key === 'Escape') setIsEditing(false);
            }}
            className="bg-gray-700 text-white px-1 text-sm outline-none"
            autoFocus
            onClick={(e) => e.stopPropagation()}
          />
        ) : (
          <span
            className="truncate flex-1"
            onDoubleClick={(e) => {
              e.stopPropagation();
              setIsEditing(true);
            }}
          >
            {node.name}
          </span>
        )}

        {/* 文件操作按钮 */}
        {onFileDelete && node.type === 'file' && (
          <button
            className="opacity-0 hover:opacity-100 text-xs px-1 hover:text-red-400"
            onClick={(e) => {
              e.stopPropagation();
              onFileDelete(node.id);
            }}
          >
            ✕
          </button>
        )}
      </div>

      {/* 子节点 */}
      {node.type === 'folder' && isExpanded && node.children && (
        <div>
          {node.children.map((child) => (
            <FileTreeNode
              key={child.id}
              node={child}
              activeFileId={activeFileId}
              onFileSelect={onFileSelect}
              onFileCreate={onFileCreate}
              onFileDelete={onFileDelete}
              onFileRename={onFileRename}
              level={level + 1}
            />
          ))}

          {/* 新建文件/文件夹按钮 */}
          {onFileCreate && (
            <div
              className="flex items-center px-2 py-1 cursor-pointer text-gray-500 hover:text-gray-300"
              style={{ paddingLeft: `${(level + 1) * 12 + 8}px` }}
            >
              <button
                className="mr-2 text-xs hover:text-blue-400"
                onClick={(e) => {
                  e.stopPropagation();
                  onFileCreate(node.id, 'untitled', 'file');
                }}
              >
                + 📄
              </button>
              <button
                className="text-xs hover:text-blue-400"
                onClick={(e) => {
                  e.stopPropagation();
                  onFileCreate(node.id, 'new-folder', 'folder');
                }}
              >
                + 📁
              </button>
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```

## 4. 实现编辑器标签页

### 4.1 标签页管理系统

```tsx
// components/EditorTabs.tsx
'use client';

import { useRef } from 'react';
import { FileNode } from './FileTree';

interface EditorTabsProps {
  files: FileNode[];
  activeFileId?: string;
  onTabSelect: (fileId: string) => void;
  onTabClose: (fileId: string) => void;
}

export function EditorTabs({
  files,
  activeFileId,
  onTabSelect,
  onTabClose,
}: EditorTabsProps) {
  const tabsRef = useRef<HTMLDivElement>(null);

  // 展平文件树，获取所有文件
  const flattenFiles = (nodes: FileNode[]): FileNode[] => {
    const result: FileNode[] = [];
    for (const node of nodes) {
      if (node.type === 'file') {
        result.push(node);
      }
      if (node.children) {
        result.push(...flattenFiles(node.children));
      }
    }
    return result;
  };

  const openFiles = flattenFiles(files).filter((f) =>
    files.some((file) => file.id === f.id)
  );

  // 简化版本：直接使用传入的 files
  const tabs = files.filter((f) => f.type === 'file');

  return (
    <div
      ref={tabsRef}
      className="h-9 bg-gray-800 flex items-end overflow-x-auto"
    >
      {tabs.map((file) => {
        const isActive = file.id === activeFileId;

        return (
          <div
            key={file.id}
            className={`group flex items-center h-8 px-3 cursor-pointer border-r border-gray-700 min-w-[120px] max-w-[200px] ${
              isActive
                ? 'bg-gray-900 text-white'
                : 'bg-gray-800 text-gray-400 hover:bg-gray-750'
            }`}
            onClick={() => onTabSelect(file.id)}
          >
            {/* 文件图标 */}
            <span className="mr-2 text-xs">{getFileIcon(file)}</span>

            {/* 文件名 */}
            <span className="truncate flex-1 text-sm">{file.name}</span>

            {/* 关闭按钮 */}
            <button
              className={`ml-2 w-4 h-4 flex items-center justify-center rounded hover:bg-gray-600 ${
                isActive ? 'opacity-100' : 'opacity-0 group-hover:opacity-100'
              }`}
              onClick={(e) => {
                e.stopPropagation();
                onTabClose(file.id);
              }}
            >
              <span className="text-xs">✕</span>
            </button>

            {/* 修改指示器 */}
            {file.language && (
              <div className="ml-1 w-2 h-2 rounded-full bg-blue-400" />
            )}
          </div>
        );
      })}
    </div>
  );
}

function getFileIcon(file: FileNode): string {
  const iconMap: Record<string, string> = {
    js: '🟨',
    jsx: '⚛️',
    ts: '🔷',
    tsx: '⚛️',
    json: '📋',
    css: '🎨',
    html: '🌐',
    md: '📝',
  };

  const ext = file.name.split('.').pop() || '';
  return iconMap[ext] || '📄';
}
```

```tsx
// hooks/useEditorTabs.ts
import { create } from 'zustand';

interface Tab {
  id: string;
  name: string;
  language?: string;
  content?: string;
  isDirty: boolean;
}

interface EditorTabsStore {
  tabs: Tab[];
  activeTabId: string | null;

  openTab: (tab: Omit<Tab, 'isDirty'>) => void;
  closeTab: (id: string) => void;
  setActiveTab: (id: string) => void;
  updateTabContent: (id: string, content: string) => void;
  getActiveTab: () => Tab | undefined;
}

export const useEditorTabsStore = create<EditorTabsStore>((set, get) => ({
  tabs: [],
  activeTabId: null,

  openTab: (tab) =>
    set((state) => {
      // 如果标签已存在，切换到该标签
      const existingTab = state.tabs.find((t) => t.id === tab.id);
      if (existingTab) {
        return { activeTabId: tab.id };
      }

      // 否则添加新标签
      return {
        tabs: [...state.tabs, { ...tab, isDirty: false }],
        activeTabId: tab.id,
      };
    }),

  closeTab: (id) =>
    set((state) => {
      const newTabs = state.tabs.filter((t) => t.id !== id);
      let newActiveId = state.activeTabId;

      // 如果关闭的是当前活动标签，切换到上一个
      if (state.activeTabId === id) {
        const index = state.tabs.findIndex((t) => t.id === id);
        newActiveId = newTabs[index - 1]?.id || newTabs[0]?.id || null;
      }

      return {
        tabs: newTabs,
        activeTabId: newActiveId,
      };
    }),

  setActiveTab: (id) => set({ activeTabId: id }),

  updateTabContent: (id, content) =>
    set((state) => ({
      tabs: state.tabs.map((tab) =>
        tab.id === id ? { ...tab, content, isDirty: true } : tab
      ),
    })),

  getActiveTab: () => {
    const state = get();
    return state.tabs.find((t) => t.id === state.activeTabId);
  },
}));

// 使用示例
function EditorWithTabs() {
  const { tabs, activeTabId, openTab, closeTab, setActiveTab, updateTabContent } =
    useEditorTabsStore();

  const handleFileSelect = (file: FileNode) => {
    openTab({
      id: file.id,
      name: file.name,
      language: file.language,
      content: file.content,
    });
  };

  const activeTab = tabs.find((t) => t.id === activeTabId);

  return (
    <div className="h-full flex flex-col">
      <div className="flex">
        {tabs.map((tab) => (
          <div
            key={tab.id}
            className={`px-3 py-2 cursor-pointer border-r ${
              tab.id === activeTabId
                ? 'bg-gray-900 text-white'
                : 'bg-gray-800 text-gray-400'
            }`}
            onClick={() => setActiveTab(tab.id)}
          >
            {tab.name}
            {tab.isDirty && <span className="ml-1 text-blue-400">●</span>}
            <button
              className="ml-2"
              onClick={(e) => {
                e.stopPropagation();
                closeTab(tab.id);
              }}
            >
              ✕
            </button>
          </div>
        ))}
      </div>

      {activeTab && (
        <Editor
          language={activeTab.language}
          value={activeTab.content}
          onChange={(value) => updateTabContent(activeTab.id, value || '')}
        />
      )}
    </div>
  );
}
```
