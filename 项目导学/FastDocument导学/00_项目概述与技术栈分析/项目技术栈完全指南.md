# FastDocument 项目技术栈完全指南

## 一、项目概述

FastDocument 是一个功能完整的现代化文档协作平台，采用原子化设计理念，支持：

- **文档管理**: 树状结构、原子化块编辑器
- **实时协作**: WebSocket 毫秒级同步
- **知识库**: 空间管理、双向链接、知识图谱
- **项目管理**: 看板、甘特图、日历
- **视频会议**: LiveKit WebRTC 集成
- **评论批注**: 文档评论、文本标注
- **权限系统**: 角色管理、分享控制

```typescript
// store/documentStore.ts
import { create } from 'zustand'
import { persist, createJSONStorage } from 'zustand/middleware'

interface DocumentState {
  documents: Document[]
  currentDoc: Document | null
  blocks: Block[]
  setCurrentDoc: (doc: Document) => void
  addBlock: (block: Block) => void
  updateBlock: (id: string, content: string) => void
}

export const useDocumentStore = create<DocumentState>()(
  persist(
    (set, get) => ({
      documents: [],
      currentDoc: null,
      blocks: [],

      setCurrentDoc: (doc) => set({ currentDoc: doc }),

      addBlock: (block) => set((state) => ({
        blocks: [...state.blocks, block]
      })),

      updateBlock: (id, content) => set((state) => ({
        blocks: state.blocks.map(b =>
          b.id === id ? { ...b, content } : b
        )
      })),
    }),
    {
      name: 'fastdoc-document-storage',
      storage: createJSONStorage(() => localStorage),
    }
  )
)

// 使用组件
function DocumentEditor() {
  const { blocks, updateBlock } = useDocumentStore()

  return (
    <div>
      {blocks.map(block => (
        <EditableBlock
          key={block.id}
          content={block.content}
          onChange={(content) => updateBlock(block.id, content)}
        />
      ))}
    </div>
  )
}
```

**客户端封装（实际源码）**:
```typescript
// lib/socket.ts
import { io, Socket } from "socket.io-client";

/**
 * 后端服务地址，优先使用环境变量
 */
const SOCKET_URL = process.env.NEXT_PUBLIC_BACKEND_URL || "http://localhost:5555";

/**
 * Socket.io 客户端包装类
 * 用于管理连接、房间加入及消息收发
 */
class SocketClient {
  private socket: Socket | null = null;
  private messageListeners: Function[] = [];
  private usersListeners: Function[] = [];

  /**
   * 建立连接并绑定基础状态监听
   */
  connect(onStatusChange?: (online: boolean) => void) {
    if (!this.socket) {
      this.socket = io(SOCKET_URL, {
        reconnectionAttempts: 5,
        timeout: 10000,
      });

      this.socket.on("connect", () => {
        console.log("成功连接至 WebSocket 服务器");
        if (onStatusChange) onStatusChange(true);
      });

      this.socket.on("disconnect", () => {
        console.warn("与 WebSocket 服务器断开连接");
        if (onStatusChange) onStatusChange(false);
      });

      this.socket.on("chatMessage", (data) => {
        this.messageListeners.forEach(listener => listener(data));
      });

      this.socket.on("onlineUsersUpdate", (data) => {
        this.usersListeners.forEach(listener => listener(data));
      });
    }
    return this.socket;
  }

  /**
   * 加入特定的文档协作房间
   */
  joinDocument(docId: string, userName: string) {
    if (this.socket) {
      this.socket.emit("joinDocument", { docId, userName });
    }
  }

  /**
   * 监听在线用户列表更新
   */
  onOnlineUsersUpdate(callback: (users: { id: string; name: string }[]) => void) {
    if (this.socket) {
      this.socket.on("onlineUsersUpdate", callback);
    }
  }

  /**
   * 发送块更新消息
   */
  updateBlock(docId: string, blockId: string, content: string, type?: string, properties?: any) {
    if (this.socket) {
      this.socket.emit("updateBlock", { docId, blockId, content, type, properties });
    }
  }

  /**
   * 批量更新块（增量同步优化）
   */
  updateBlocksBatch(docId: string, blocks: { id: string; content: string; type?: string; properties?: any }[]) {
    if (this.socket) {
      this.socket.emit("updateBlocksBatch", { docId, blocks });
    }
  }

  /**
   * 监听批量块更新
   */
  onBlocksUpdated(callback: (data: { docId: string; blocks: any[] }) => void) {
    if (this.socket) {
      this.socket.on("blocksUpdated", callback);
    }
    return () => {
      if (this.socket) {
        this.socket.off("blocksUpdated", callback);
      }
    };
  }

  /**
   * 请求分块加载（大型文档优化）
   */
  loadBlocksChunked(docId: string, offset: number, limit: number) {
    if (this.socket) {
      this.socket.emit("loadBlocksChunked", { docId, offset, limit });
    }
  }

  /**
   * 增量同步请求
   */
  syncBlocks(docId: string, since: string) {
    if (this.socket) {
      this.socket.emit("syncBlocks", { docId, since });
    }
  }

  /**
   * 发送"正在输入"状态
   */
  sendTyping(docId: string, userId: string, userName: string) {
    if (this.socket) {
      this.socket.emit("typing", { docId, userId, userName });
    }
  }

  /**
   * 发送聊天消息
   */
  sendChat(docId: string, userId: string, userName: string, content: string, mentions?: string[]) {
    if (this.socket) {
      this.socket.emit("chatMessage", {
        docId,
        userId,
        userName,
        content,
        timestamp: Date.now(),
        mentions: mentions || []
      });
    }
  }

  /**
   * 监听远程块更新消息
   */
  onBlockUpdated(callback: (data: any) => void) {
    if (this.socket) {
      this.socket.on("blockUpdated", callback);
    }
  }

  /**
   * 监听聊天消息
   */
  onChatMessage(callback: (data: any) => void) {
    this.messageListeners.push(callback);
    return () => {
      this.messageListeners = this.messageListeners.filter(l => l !== callback);
    };
  }

  /**
   * 断开连接
   */
  disconnect() {
    if (this.socket) {
      this.socket.disconnect();
      this.socket = null;
    }
  }
}

/**
 * 导出单例 Socket 客户端
 */
export const socketClient = new SocketClient();
```

#### LiveKit Client 2.17.2

**视频会议集成（实际源码）**:
```typescript
// lib/livekit.ts
/**
 * LiveKit 客户端封装类
 * 视频会议功能（简化版）
 */

type Room = any;
type RemoteParticipant = any;
type LocalParticipant = any;

/**
 * LiveKit 房间连接选项
 */
interface RoomConnectOptions {
  audio?: boolean;
  video?: boolean;
  screenShare?: boolean;
}

/**
 * LiveKit 客户端封装类
 * 用于管理视频会议房间连接及事件处理
 */
export class LiveKitClient {
  private room: Room | null = null;
  private localParticipant: LocalParticipant | null = null;

  // 事件回调
  onParticipantJoined?: (participant: RemoteParticipant) => void;
  onParticipantLeft?: (participant: RemoteParticipant) => void;
  onTrackSubscribed?: (track: any, participant: RemoteParticipant) => void;
  onTrackUnsubscribed?: (track: any, participant: RemoteParticipant) => void;
  onConnectionStateChange?: (state: string) => void;
  onSpeakingChanged?: (participant: RemoteParticipant, isSpeaking: boolean) => void;

  /**
   * 连接到 LiveKit 房间
   */
  async connect(url: string, token: string, _options?: RoomConnectOptions): Promise<Room | null> {
    try {
      // 动态导入 livekit-client
      const livekit = await import("livekit-client");

      // 创建房间实例
      this.room = new livekit.Room();

      // 连接房间
      await this.room.connect(url, token);

      // 设置房间事件
      this.setupRoomEvents();

      // 获取本地参与者
      this.localParticipant = this.room?.localParticipant;

      return this.room;
    } catch (error) {
      console.error("连接 LiveKit 房间失败:", error);
      return null;
    }
  }

  /**
   * 设置房间事件
   */
  private setupRoomEvents() {
    if (!this.room) return;

    // 动态获取 RoomEvent
    import("livekit-client").then(livekit => {
      const RoomEvent = livekit.RoomEvent;

      // 参与者加入
      this.room?.on(RoomEvent.ParticipantConnected, (participant: RemoteParticipant) => {
        this.onParticipantJoined?.(participant);
      });

      // 参与者离开
      this.room?.on(RoomEvent.ParticipantDisconnected, (participant: RemoteParticipant) => {
        this.onParticipantLeft?.(participant);
      });

      // 轨道订阅
      this.room?.on(RoomEvent.TrackSubscribed, (track: any, _publication: any, participant: RemoteParticipant) => {
        this.onTrackSubscribed?.(track, participant);
      });

      // 轨道取消订阅
      this.room?.on(RoomEvent.TrackUnsubscribed, (track: any, _publication: any, participant: RemoteParticipant) => {
        this.onTrackUnsubscribed?.(track, participant);
      });

      // 连接状态变化
      this.room?.on(RoomEvent.ConnectionStateChanged, (state: any) => {
        this.onConnectionStateChange?.(state);
      });
    });
  }

  /**
   * 切换麦克风
   */
  async toggleMicrophone(enabled: boolean) {
    if (!this.localParticipant) return;

    try {
      const audioTracks = (this.localParticipant as any).audioTracks;
      if (audioTracks) {
        for (const [, publication] of audioTracks) {
          if (publication.track) {
            if (enabled) {
              await publication.track.enable();
            } else {
              await publication.track.disable();
            }
          }
        }
      }
    } catch (error) {
      console.error("切换麦克风失败:", error);
    }
  }

  /**
   * 切换摄像头
   */
  async toggleCamera(enabled: boolean) {
    if (!this.localParticipant) return;

    try {
      const videoTracks = (this.localParticipant as any).videoTracks;
      if (videoTracks) {
        for (const [, publication] of videoTracks) {
          if (publication.track) {
            if (enabled) {
              await publication.track.enable();
            } else {
              await publication.track.disable();
            }
          }
        }
      }
    } catch (error) {
      console.error("切换摄像头失败:", error);
    }
  }

  /**
   * 屏幕共享
   */
  async startScreenShare() {
    if (!this.localParticipant) return;

    try {
      await (this.localParticipant as any).setScreenShareEnabled(true);
    } catch (error) {
      console.error("开始屏幕共享失败:", error);
    }
  }

  /**
   * 停止屏幕共享
   */
  async stopScreenShare() {
    if (!this.localParticipant) return;

    try {
      await (this.localParticipant as any).setScreenShareEnabled(false);
    } catch (error) {
      console.error("停止屏幕共享失败:", error);
    }
  }

  /**
   * 静音指定参与者
   */
  async muteParticipant(participantSid: string) {
    if (!this.room) return;

    const participants = (this.room as any).participants;
    if (!participants) return;

    const participant = participants.get(participantSid);
    if (!participant) return;

    try {
      const audioTracks = (participant as any).audioTracks;
      if (audioTracks) {
        for (const [, publication] of audioTracks) {
          if (publication.track) {
            await publication.track.disable();
          }
        }
      }
    } catch (error) {
      console.error("静音参与者失败:", error);
    }
  }

  /**
   * 获取房间参与者
   */
  getParticipants(): Map<string, RemoteParticipant> {
    const participants = (this.room as any)?.participants;
    return participants || new Map();
  }

  /**
   * 获取房间连接状态
   */
  getConnectionState(): string {
    return (this.room as any)?.state || "disconnected";
  }

  /**
   * 获取房间实例
   */
  getRoom(): Room | null {
    return this.room;
  }

  /**
   * 断开连接
   */
  async disconnect() {
    if (this.room) {
      await this.room.disconnect();
      this.room = null;
      this.localParticipant = null;
    }
  }
}
```

#### Yjs CRDT 协作编辑

**Yjs 协作管理器（实际源码）**:
```typescript
// lib/yjs.ts
/**
 * Yjs CRDT 协作服务
 * 提供基于 CRDT (Conflict-free Replicated Data Type) 的实时协作功能
 */

import * as Y from "yjs";
import { WebsocketProvider } from "y-websocket";
import { IndexeddbPersistence } from "y-indexeddb";

// 文档类型定义
export interface CRDTDocument {
  id: string;
  doc: Y.Doc;
  provider: WebsocketProvider | null;
  persistence: IndexeddbPersistence | null;
}

// 协作状态
export interface CollaborationState {
  connected: boolean;
  users: AwarenessUser[];
  syncing: boolean;
}

// 在线用户信息
export interface AwarenessUser {
  id: string;
  name: string;
  color: string;
  cursor?: {
    anchor: number;
    head: number;
  };
}

// 块操作类型
export type BlockOperationType = "add" | "update" | "delete" | "move";

/**
 * Yjs 协作管理器
 * 管理多个文档的 CRDT 协作连接
 */
class YjsCollaborationManager {
  private documents: Map<string, CRDTDocument> = new Map();
  private currentDocId: string | null = null;
  private serverUrl: string;
  private userId: string;
  private userName: string;
  private userColor: string;

  constructor() {
    // 从环境变量或默认值获取 WebSocket 服务器地址
    this.serverUrl = process.env.NEXT_PUBLIC_COLLABORATION_WS_URL || "ws://localhost:1234";
    this.userId = "";
    this.userName = "";
    this.userColor = this.generateRandomColor();
  }

  /**
   * 初始化用户信息
   */
  initUser(userId: string, userName: string) {
    this.userId = userId;
    this.userName = userName;
  }

  /**
   * 生成随机颜色
   */
  private generateRandomColor(): string {
    const colors = [
      "#0D9488", // Teal
      "#0EA5E9", // Sky Blue
      "#8B5CF6", // Violet
      "#EC4899", // Pink
      "#F59E0B", // Amber
      "#10B981", // Emerald
      "#6366F1", // Indigo
      "#F43F5E", // Rose
    ];
    return colors[Math.floor(Math.random() * colors.length)];
  }

  /**
   * 连接文档协作
   */
  async connectDocument(
    docId: string,
    onSync: () => void,
    onAwarenessChange: (users: AwarenessUser[]) => void
  ): Promise<CRDTDocument> {
    // 如果已存在，直接返回
    const existing = this.documents.get(docId);
    if (existing) {
      return existing;
    }

    // 创建新的 Yjs 文档
    const doc = new Y.Doc();

    // 创建 IndexedDB 持久化 (离线支持)
    const persistence = new IndexeddbPersistence(docId, doc);

    // 等待 IndexedDB 加载完成
    await new Promise<void>((resolve) => {
      persistence.once("synced", () => {
        console.log("IndexedDB 内容已加载");
        resolve();
      });
    });

    // 创建 WebSocket 提供者 (实时协作)
    let provider: WebsocketProvider | null = null;

    try {
      provider = new WebsocketProvider(this.serverUrl, docId, doc, {
        connect: true,
      });

      // 设置用户信息
      provider.awareness.setLocalState({
        user: {
          id: this.userId,
          name: this.userName,
          color: this.userColor,
        },
      });

      // 监听同步事件
      provider.on("sync", (isSynced: boolean) => {
        if (isSynced) {
          console.log("文档已与服务器同步");
          onSync();
        }
      });

      // 监听在线用户变化
      provider.awareness.on("change", () => {
        const states = provider?.awareness.getStates();
        const users: AwarenessUser[] = [];

        states?.forEach((state, clientId) => {
          if (state.user && clientId !== doc.clientID) {
            users.push({
              id: state.user.id || String(clientId),
              name: state.user.name || "未知用户",
              color: state.user.color || "#666",
              cursor: state.cursor,
            });
          }
        });

        onAwarenessChange(users);
      });

      // 监听连接状态
      provider.on("status", (event: { status: string }) => {
        console.log("协作连接状态:", event.status);
      });
    } catch (error) {
      console.warn("WebSocket 连接失败，将仅使用本地离线模式:", error);
    }

    const crdtDoc: CRDTDocument = {
      id: docId,
      doc,
      provider,
      persistence,
    };

    this.documents.set(docId, crdtDoc);
    this.currentDocId = docId;

    return crdtDoc;
  }

  /**
   * 断开文档协作
   */
  disconnectDocument(docId: string) {
    const crdtDoc = this.documents.get(docId);
    if (!crdtDoc) return;

    // 断开 WebSocket 连接
    if (crdtDoc.provider) {
      crdtDoc.provider.disconnect();
      crdtDoc.provider.destroy();
    }

    // 销毁持久化
    if (crdtDoc.persistence) {
      crdtDoc.persistence.destroy();
    }

    // 销毁文档
    crdtDoc.doc.destroy();

    this.documents.delete(docId);

    if (this.currentDocId === docId) {
      this.currentDocId = null;
    }
  }

  /**
   * 获取当前文档
   */
  getCurrentDoc(): Y.Doc | null {
    if (!this.currentDocId) return null;
    const crdtDoc = this.documents.get(this.currentDocId);
    return crdtDoc?.doc || null;
  }

  /**
   * 获取当前文档的 Y.Array (用于块列表)
   */
  getBlocksArray(): Y.Array<any> | null {
    const doc = this.getCurrentDoc();
    if (!doc) return null;
    return doc.getArray("blocks");
  }

  /**
   * 添加块
   */
  addBlock(index: number, block: any) {
    const blocks = this.getBlocksArray();
    if (!blocks) return;
    blocks.insert(index, [block]);
  }

  /**
   * 更新块
   */
  updateBlock(blockId: string, updates: Partial<any>) {
    const blocks = this.getBlocksArray();
    if (!blocks) return;

    blocks.forEach((block, index) => {
      if (block.id === blockId) {
        const newBlock = { ...block, ...updates };
        blocks.delete(index, 1);
        blocks.insert(index, [newBlock]);
      }
    });
  }

  /**
   * 删除块
   */
  deleteBlock(blockId: string) {
    const blocks = this.getBlocksArray();
    if (!blocks) return;

    blocks.forEach((block, index) => {
      if (block.id === blockId) {
        blocks.delete(index, 1);
      }
    });
  }

  /**
   * 移动块
   */
  moveBlock(fromIndex: number, toIndex: number) {
    const blocks = this.getBlocksArray();
    if (!blocks) return;

    const block = blocks.get(fromIndex);
    if (block) {
      blocks.delete(fromIndex, 1);
      blocks.insert(toIndex > fromIndex ? toIndex - 1 : toIndex, [block]);
    }
  }

  /**
   * 观察块变化
   */
  observeBlocks(callback: (event: Y.YArrayEvent<any>) => void) {
    const blocks = this.getBlocksArray();
    if (!blocks) return;
    blocks.observe(callback);
  }

  /**
   * 获取所有在线用户
   */
  getOnlineUsers(): AwarenessUser[] {
    if (!this.currentDocId) return [];

    const crdtDoc = this.documents.get(this.currentDocId);
    if (!crdtDoc?.provider) return [];

    const states = crdtDoc.provider.awareness.getStates();
    const users: AwarenessUser[] = [];

    states.forEach((state, clientId) => {
      if (state.user) {
        users.push({
          id: state.user.id || String(clientId),
          name: state.user.name || "未知用户",
          color: state.user.color || "#666",
        });
      }
    });

    return users;
  }

  /**
   * 更新光标位置
   */
  updateCursor(anchor: number, head: number) {
    if (!this.currentDocId) return;

    const crdtDoc = this.documents.get(this.currentDocId);
    if (!crdtDoc?.provider) return;

    crdtDoc.provider.awareness.setLocalStateField("cursor", { anchor, head });
  }

  /**
   * 检查是否已连接
   */
  isConnected(): boolean {
    if (!this.currentDocId) return false;

    const crdtDoc = this.documents.get(this.currentDocId);
    if (!crdtDoc?.provider) return false;

    return crdtDoc.provider.wsconnected;
  }

  /**
   * 导出文档为 JSON
   */
  exportAsJSON(): any {
    const doc = this.getCurrentDoc();
    if (!doc) return null;

    const blocks = this.getBlocksArray();
    const meta = doc.getMap("meta");

    return {
      blocks: blocks?.toJSON(),
      meta: meta?.toJSON(),
    };
  }

  /**
   * 销毁所有协作连接
   */
  destroy() {
    this.documents.forEach((_, docId) => {
      this.disconnectDocument(docId);
    });
  }
}
```

### 3.3 JWT 认证

**认证流程**:
```typescript
// auth/auth.service.ts
import { Injectable, UnauthorizedException } from '@nestjs/common'
import { JwtService } from '@nestjs/jwt'
import * as bcrypt from 'bcrypt'

@Injectable()
export class AuthService {
  constructor(private jwtService: JwtService) {}

  async loginWithPassword(username: string, password: string) {
    const user = await this.validateUser(username, password)
    if (!user) {
      throw new UnauthorizedException('用户名或密码错误')
    }

    const payload = { sub: user.id, username: user.username, role: user.role }
    return {
      access_token: this.jwtService.sign(payload),
      user: {
        id: user.id,
        username: user.username,
        role: user.role,
      },
    }
  }

  private async validateUser(username: string, password: string) {
    // 查询用户
    const user = await this.userRepository.findOne({ where: { username } })
    if (!user) return null

    // 验证密码
    const isPasswordValid = await bcrypt.compare(password, user.password)
    return isPasswordValid ? user : null
  }
}
```

```typescript
import {
  WebSocketGateway,
  WebSocketServer,
  SubscribeMessage,
  OnGatewayConnection,
  OnGatewayDisconnect,
} from '@nestjs/websockets'
import { Server, Socket } from 'socket.io'

@WebSocketGateway({
  cors: { origin: '*' },
  namespace: '/documents',
})
export class DocumentsGateway implements OnGatewayConnection, OnGatewayDisconnect {
  @WebSocketServer()
  server: Server

  handleConnection(client: Socket) {
    console.log(`Client connected: ${client.id}`)
  }

  handleDisconnect(client: Socket) {
    console.log(`Client disconnected: ${client.id}`)
  }

  @SubscribeMessage('joinDocument')
  handleJoinDocument(client: Socket, payload: { docId: string; userId: string; userName: string }) {
    client.join(payload.docId)
    // 广播新用户加入
    client.to(payload.docId).emit('userJoined', {
      userId: payload.userId,
      userName: payload.userName,
    })
  }

  @SubscribeMessage('updateBlock')
  handleBlockUpdate(client: Socket, payload: { docId: string; blockId: string; content: string }) {
    // 广播给房间内其他用户
    client.to(payload.docId).emit('blockUpdated', payload)
    // 异步持久化到数据库
    this.documentsService.updateBlock(payload.blockId, payload.content)
  }
}
```

### 6.2 状态管理原则

1. **单一职责**: 每个 store 只管理一种类型的状态
2. **不可变性**: 使用 immutable 方式更新状态
3. **持久化**: 重要数据存储到 localStorage
4. **派生状态**: 使用 selector 计算派生数据

### 6.3 实时协作原理

1. 用户加入文档房间（Socket join）
2. 编辑操作发送到服务器
3. 服务器广播给房间内其他用户
4. 异步持久化到数据库
5. 使用操作转换（OT）解决冲突
