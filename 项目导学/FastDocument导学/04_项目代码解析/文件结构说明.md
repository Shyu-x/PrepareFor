### 3.2 主页面文件

```typescript
// app/page.tsx
'use client'

import { useEffect } from 'react'
import { useSearchParams } from 'next/navigation'
import { useUserStore } from '@/store/userStore'
import { useDocumentStore } from '@/store/documentStore'
import Sidebar from '@/components/Sidebar'
import Header from '@/components/Header'
import Dashboard from '@/components/Dashboard'
import Editor from '@/components/Editor'
import VideoConference from '@/components/VideoConference'
import KnowledgeBaseView from '@/components/KnowledgeBaseView'
// ... 更多视图组件

export default function HomePage() {
  const searchParams = useSearchParams()

  // 从 URL 获取当前视图和文档 ID
  const currentPage = searchParams.get('page') || 'dashboard'
  const docId = searchParams.get('doc')

  // 初始化用户状态
  const { fetchCurrentUser, user } = useUserStore()

  // 初始化文档状态
  const { fetchDocuments } = useDocumentStore()

  useEffect(() => {
    // 启动时获取当前用户
    fetchCurrentUser()
    // 获取文档列表
    fetchDocuments()
  }, [])

  // 根据当前页面渲染不同视图
  const renderContent = () => {
    switch (currentPage) {
      case 'dashboard':
      case 'all':
      case 'starred':
      case 'trash':
        return <Dashboard currentView={currentPage} />
      case 'editor':
        return docId ? <Editor documentId={docId} /> : <Dashboard />
      case 'meeting':
        return <VideoConference />
      case 'knowledge':
        return <KnowledgeBaseView />
      // ... 更多视图
      default:
        return <Dashboard />
    }
  }

  return (
    <div className="flex h-screen">
      <Sidebar />
      <div className="flex-1 flex flex-col overflow-hidden">
        <Header />
        <main className="flex-1 overflow-auto bg-gray-50">
          {renderContent()}
        </main>
      </div>
    </div>
  )
}
```

### 3.3 API 客户端配置

```typescript
// lib/api.ts
import axios, { AxiosError } from 'axios'

const API_URL = process.env.NEXT_PUBLIC_BACKEND_URL || 'http://localhost:5555'

// 创建 axios 实例
const api = axios.create({
  baseURL: API_URL,
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
})

// 请求拦截器
api.interceptors.request.use(
  (config) => {
    // 从 localStorage 获取 token
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
api.interceptors.response.use(
  (response) => response,
  (error: AxiosError) => {
    if (error.response?.status === 401) {
      // 清除登录信息
      localStorage.removeItem('token')
      localStorage.removeItem('user')
      // 跳转到登录页
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

// 文档 API
export const documentsApi = {
  list: () => api.get('/documents'),
  get: (id: string) => api.get(`/documents/${id}`),
  create: (data: { title: string; type: string; parentId?: string }) =>
    api.post('/documents', data),
  update: (id: string, data: any) => api.patch(`/documents/${id}`, data),
  delete: (id: string) => api.delete(`/documents/${id}`),
  getTree: () => api.get('/documents/tree'),
  toggleStar: (id: string) => api.post(`/documents/${id}/toggle-star`),
  trash: (id: string) => api.post(`/documents/${id}/trash`),
  restore: (id: string) => api.post(`/documents/${id}/restore`),
}

// 块 API
export const blocksApi = {
  getBlocks: (docId: string) => api.get(`/documents/${docId}/blocks`),
  createBlock: (docId: string, data: any) =>
    api.post(`/documents/${docId}/blocks`, data),
  updateBlock: (blockId: string, data: any) =>
    api.patch(`/blocks/${blockId}`, data),
  deleteBlock: (blockId: string) => api.delete(`/blocks/${blockId}`),
  reorder: (docId: string, blockIds: string[]) =>
    api.post(`/documents/${docId}/blocks/reorder`, { blockIds }),
}

// 导出
export default api
```

### 3.4 Socket.io 客户端封装

```typescript
// lib/socket.ts
import { io, Socket } from 'socket.io-client'

type ConnectionCallback = (online: boolean) => void

class SocketClient {
  private socket: Socket | null = null
  private connectionCallbacks: ConnectionCallback[] = []

  connect(onStatusChange?: ConnectionCallback) {
    if (this.socket?.connected) return this.socket

    const backendUrl =
      process.env.NEXT_PUBLIC_BACKEND_URL || 'http://localhost:5555'

    this.socket = io(backendUrl, {
      transports: ['websocket', 'polling'],
      reconnection: true,
      reconnectionAttempts: 5,
      reconnectionDelay: 1000,
    })

    this.socket.on('connect', () => {
      console.log('Socket connected:', this.socket?.id)
      this.connectionCallbacks.forEach((cb) => cb(true))
    })

    this.socket.on('disconnect', () => {
      console.log('Socket disconnected')
      this.connectionCallbacks.forEach((cb) => cb(false))
    })

    if (onStatusChange) {
      this.connectionCallbacks.push(onStatusChange)
    }

    return this.socket
  }

  disconnect() {
    if (this.socket) {
      this.socket.disconnect()
      this.socket = null
    }
  }

  // 加入文档房间
  joinDocumentRoom(docId: string, userId: string, userName: string) {
    this.socket?.emit('joinDocument', { docId, userId, userName })
  }

  // 退出文档房间
  leaveDocumentRoom(docId: string) {
    this.socket?.emit('leaveDocument', { docId })
  }

  // 发送块更新
  emitBlockUpdate(data: {
    docId: string
    blockId: string
    content: string
  }) {
    this.socket?.emit('updateBlock', data)
  }

  // 监听块更新
  onBlockUpdate(callback: (data: any) => void) {
    this.socket?.on('updateBlock', callback)
  }

  // 移除块更新监听
  offBlockUpdate(callback?: (data: any) => void) {
    if (callback) {
      this.socket?.off('updateBlock', callback)
    } else {
      this.socket?.off('updateBlock')
    }
  }

  // 监听在线用户
  onOnlineUsersUpdate(callback: (users: any[]) => void) {
    this.socket?.on('onlineUsers', callback)
  }

  // 发送聊天消息
  sendChatMessage(data: { docId: string; message: string }) {
    this.socket?.emit('chatMessage', data)
  }

  // 监听聊天消息
  onChatMessage(callback: (data: any) => void) {
    this.socket?.on('newChatMessage', callback)
  }

  // 光标移动
  emitCursorMove(data: { docId: string; userId: string; position: any }) {
    this.socket?.emit('cursorMove', data)
  }

  onCursorMove(callback: (data: any) => void) {
    this.socket?.on('cursorMoved', callback)
  }
}

export const socketClient = new SocketClient()
```

## 总结

FastDocument 项目采用了现代化的前端架构：

- **Next.js App Router**：文件系统路由
- **Zustand**：轻量级状态管理
- **组件化**：按功能模块组织
- **工具库分离**：API、Socket、工具函数分离
- **类型安全**：TypeScript 全面支持

后端采用 NestJS 模块化设计，每个功能模块都有清晰的职责划分。
