  content: string;
  isEditing?: boolean;
  onChange?: (content: string) => void;
}

export function TextBlock({ content, isEditing = false, onChange }: TextBlockProps) {
  const [editing, setEditing] = useState(isEditing);
  const [value, setValue] = useState(content);

  const handleBlur = () => {
    setEditing(false);
    if (onChange && value !== content) {
      onChange(value);
    }
  };

  if (editing) {
    return (
      <input
        type="text"
        value={value}
        onChange={(e) => setValue(e.target.value)}
        onBlur={handleBlur}
        autoFocus
      />
    );
  }

  return (
    <div onClick={() => setEditing(true)}>
      {content || '点击编辑...'}
    </div>
  );
}

**参考答案**

```typescript
import { useState, useEffect, useRef } from 'react';

interface AutoSaveEditorProps {
  documentId: string;
  initialContent: string;
  onSave: (content: string) => Promise<void>;
}

export function AutoSaveEditor({ documentId, initialContent, onSave }: AutoSaveEditorProps) {
  const [content, setContent] = useState(initialContent);
  const [saving, setSaving] = useState(false);
  const [lastSaved, setLastSaved] = useState<Date | null>(null);
  const timerRef = useRef<NodeJS.Timeout>();

  // 自动保存 Effect
  useEffect(() => {
    if (timerRef.current) {
      clearTimeout(timerRef.current);
    }

    timerRef.current = setTimeout(async () => {
      if (content !== initialContent) {
        setSaving(true);
        try {
          await onSave(content);
          setLastSaved(new Date());
        } catch (error) {
          console.error('Save failed:', error);
        } finally {
          setSaving(false);
        }
      }
    }, 2000);

    return () => {
      if (timerRef.current) {
        clearTimeout(timerRef.current);
      }
    };
  }, [content, initialContent, onSave]);

  return (
    <div>
      <textarea
        value={content}
        onChange={(e) => setContent(e.target.value)}
        rows={10}
        style={{ width: '100%' }}
      />
      <div className="status">
        {saving ? '保存中...' : lastSaved ? `已保存: ${lastSaved.toLocaleTimeString()}` : '未保存'}
      </div>
    </div>
  );
}
```

**参考答案**

```typescript
import { useState, useEffect, useCallback } from 'react';

export function useLocalStorage<T>(key: string, initialValue: T) {
  // 读取初始值
  const [storedValue, setStoredValue] = useState<T>(() => {
    if (typeof window === 'undefined') {
      return initialValue;
    }
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error('Error reading localStorage:', error);
      return initialValue;
    }
  });

  // 更新值的函数
  const setValue = useCallback((value: T | ((val: T) => T)) => {
    try {
      const valueToStore = value instanceof Function
        ? value(storedValue)
        : value;
      setStoredValue(valueToStore);
      if (typeof window !== 'undefined') {
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
      }
    } catch (error) {
      console.error('Error setting localStorage:', error);
    }
  }, [key, storedValue]);

  // 监听其他标签页的更改
  useEffect(() => {
    const handleStorageChange = (e: StorageEvent) => {
      if (e.key === key && e.newValue) {
        setStoredValue(JSON.parse(e.newValue));
      }
    };
    window.addEventListener('storage', handleStorageChange);
    return () => window.removeEventListener('storage', handleStorageChange);
  }, [key]);

  return [storedValue, setValue] as const;
}
```

import { useEffect, useRef, useState, useCallback } from 'react';
import { io, Socket } from 'socket.io-client';

interface User {
  id: string;
  name: string;
  color: string;
}

export function useCollaboration(documentId: string) {
  const [connected, setConnected] = useState(false);
  const [users, setUsers] = useState<User[]>([]);
  const socketRef = useRef<Socket>();

  useEffect(() => {
    // 初始化连接
    socketRef.current = io('/collaboration', {
      query: { documentId }
    });

    socketRef.current.on('connect', () => {
      setConnected(true);
    });

    socketRef.current.on('disconnect', () => {
      setConnected(false);
    });

    socketRef.current.on('users:update', (updatedUsers: User[]) => {
      setUsers(updatedUsers);
    });

    return () => {
      socketRef.current?.disconnect();
    };
  }, [documentId]);

  const sendUpdate = useCallback((blockId: string, content: string) => {
    socketRef.current?.emit('block:update', { blockId, content });
  }, []);

  const subscribeToBlock = useCallback((blockId: string) => {
    socketRef.current?.emit('block:subscribe', { blockId });
  }, []);

  return {
    connected,
    users,
    sendUpdate,
    subscribeToBlock
  };
}
