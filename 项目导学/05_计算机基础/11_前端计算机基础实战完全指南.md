#### 1.1.1 DOM树与树形结构

```javascript
// DOM（文档对象模型）本质上就是一棵树形结构
// 每个HTML元素都是树上的一个节点

/*
 * HTML结构：
 * <html>
 *   <head>
 *     <title>标题</title>
 *   </head>
 *   <body>
 *     <div>
 *       <h1>标题</h1>
 *       <p>段落</p>
 *     </div>
 *   </body>
 * </html>
 *
 * DOM树结构：
 *           html
 *          /    \
 *        head   body
 *         |      |
 *       title    div
 *                /  \
 *               h1   p
 */

// DOM遍历的复杂度
// - 获取子元素：O(1)
// - 获取父元素：O(1)
// - 获取兄弟元素：O(1)（因为有previousSibling/nextSibling）
// - 查找特定元素：O(n)（需要遍历）

// DOM查询操作的时间复杂度
document.querySelector('.class');  // O(n) - 需要遍历DOM树
document.getElementById('id');      // O(1) - 浏览器内部维护ID索引

// 面试点：为什么getElementById更快？
// 因为浏览器在解析HTML时会建立ID到元素的哈希表
// 这是浏览器做的一种优化，与我们学习哈希表的原理一致
```

#### 1.1.2 CSS选择器与匹配算法

```javascript
// 浏览器解析CSS选择器是从右向左匹配的

// 选择器：div.container .item.active
/*
 * 匹配过程：
 * 1. 首先找到所有 .active 的元素
 * 2. 然后在这些元素的祖先中找 .item

 * 为什么从右向左？

// 不好的写法：body div div div span { ... }
// 好的写法：.specific-class { ... }

// 2. 避免通配符选择器
// 不好的写法：.container * { ... }
// 好的写法：直接选择目标元素

// 3. 使用class选择器而非属性选择器
// 不好的写法：[type="text"] { ... }
// 好的写法：.text-input { ... }

// 4. ID选择器是O(1)，尽量使用
```

### 1.2 浏览器缓存机制

#### 1.2.1 缓存层级与数据结构

```javascript
// 浏览器缓存策略 - 本质上是多种数据结构的综合应用

/*
 * 浏览器缓存位置（速度从快到慢）：
 * 1. Service Worker 缓存 - 可编程的缓存
 * 2. Memory Cache（内存缓存）- 存储在内存中，关闭标签页消失
 * 3. Disk Cache（磁盘缓存）- 存储在硬盘上，持久化
 * 4. Push Cache（推送缓存）- HTTP/2的服务器推送
 */

// Memory Cache 使用的数据结构
// - 实际上是LRU（最近最少使用）缓存
// - 使用双向链表 + 哈希表实现
// - O(1) 的时间复杂度完成get和put操作

// LRU Cache原理（面试高频）
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity; // 缓存容量
    this.cache = new Map();    // 哈希表：O(1)查找
  }

  get(key) {
    if (!this.cache.has(key)) {
      return -1;
    }

    // 将访问的元素移到末尾（最近使用）
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);

    return value;
  }

  put(key, value) {
    if (this.cache.has(key)) {
      // 已存在，更新并移到末尾
      this.cache.delete(key);
      this.cache.set(key, value);
    } else {
      // 不存在，检查容量
      if (this.cache.size >= this.capacity) {
        // 删除最久未使用的（链表头部）
        const firstKey = this.cache.keys().next().value;
        this.cache.delete(firstKey);
      }

      this.cache.set(key, value);
    }
  }
}

// 实际应用：浏览器使用LRU缓存策略管理内存缓存
// 当缓存满时，删除最近最少使用的项
```

#### 1.2.2 HTTP缓存策略

```javascript
// HTTP缓存控制头部

// 强缓存（不需要发送请求）
Response Headers:
  Cache-Control: max-age=3600        // 缓存有效期（秒）
  Expires: Thu, 03 Apr 2025 12:00:00 GMT  // 过期时间点

// 协商缓存（需要发送请求验证）
Request Headers:
  If-None-Match: "abc123"             // 上次请求的ETag值
  If-Modified-Since: Thu, 03 Apr 2025 10:00:00 GMT

Response Headers:
  ETag: "def456"                      // 资源标识
  Last-Modified: Thu, 03 Apr 2025 10:00:00 GMT

// 缓存优先级
// 1. Cache-Control > Expires（因为Expires是HTTP/1.0的）
// 2. ETag > Last-Modified（更精确）

// Service Worker缓存（可编程缓存）
const CACHE_NAME = 'my-cache-v1';

self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((cachedResponse) => {
      if (cachedResponse) {
        // 返回缓存
        return cachedResponse;
      }

      // 缓存未命中，发送网络请求
      return fetch(event.request).then((response) => {
        // 检查响应是否有效
        if (!response || response.status !== 200) {
          return response;
        }

        // 缓存新的响应
        const responseClone = response.clone();
        caches.open(CACHE_NAME).then((cache) => {
          cache.put(event.request, responseClone);
        });

        return response;
      });
    })
  );
});
```

### 1.3 浏览器安全机制

#### 1.3.1 XSS与数据结构（字符串处理）

```javascript
// XSS（跨站脚本攻击）本质是对用户输入的"不信任"
// 与数据结构的字符串处理密切相关

// 攻击示例：用户输入了<script>标签
const userInput = '<script>alert("xss")</script>';

// 恶意代码会被浏览器执行

// 防御策略一：转义（Escape）
function escapeHtml(str) {
  const escapeMap = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#39;'
  };

  // 使用正则替换
  return str.replace(/[&<>"']/g, (char) => escapeMap[char]);
}

console.log(escapeHtml(userInput));
// 输出：&lt;script&gt;alert(&quot;xss&quot;)&lt;/script&gt;

// 防御策略二：白名单过滤
function sanitizeHtml(dangerousHtml) {
  // 只允许部分标签
  const allowedTags = ['p', 'br', 'b', 'i', 'em', 'strong'];
  const allowedAttrs = ['href', 'title'];

  // 这里可以使用DOMParser解析HTML
  // 然后遍历节点树，只保留允许的标签和属性
  const parser = new DOMParser();
  const doc = parser.parseFromString(dangerousHtml, 'text/html');

  // 递归遍历并过滤
  function sanitizeNode(node) {
    if (node.nodeType === Node.TEXT_NODE) {
      return node;
    }

    if (node.nodeType === Node.ELEMENT_NODE) {
      if (!allowedTags.includes(node.tagName.toLowerCase())) {
        // 不允许的标签，用文本替换
        return document.createTextNode(node.textContent);
      }

      // 创建新的干净节点
      const cleanNode = document.createElement(node.tagName);

      // 复制允许的属性
      for (const attr of node.attributes) {
        if (allowedAttrs.includes(attr.name)) {
          cleanNode.setAttribute(attr.name, attr.value);
        }
      }

      // 递归处理子节点
      for (const child of node.childNodes) {
        cleanNode.appendChild(sanitizeNode(child));
      }

      return cleanNode;
    }

    return null;
  }

  return doc.body.innerHTML;
}
```

#### 1.3.2 CSRF与Token机制

```javascript
// CSRF（跨站请求伪造）
// 原理：利用用户已登录的身份，自动发送恶意请求

// 防御策略：CSRF Token

// 服务端生成Token
function generateCSRFToken() {
  // 使用随机字符串作为Token
  const randomBytes = crypto.randomBytes(32);
  return randomBytes.toString('hex');
}

// 页面中嵌入Token
// <input type="hidden" name="csrf_token" value="abc123...">

// 验证Token（服务端）
function validateCSRFToken(requestToken, sessionToken) {
  if (!requestToken || !sessionToken) {
    return false;
  }

  // 使用定时安全比较（防止时序攻击）
  return timingSafeEqual(
    Buffer.from(requestToken),
    Buffer.from(sessionToken)
  );
}

// 实际应用：Vue/React等框架都有CSRF Token的支持
// Angular的HTTP客户端自动处理CSRF Token
// axios等库也支持配置CSRF Token
```

### 2.1 作用域与作用域链（链表结构）

```javascript
// 作用域链的本质是一条链表
// 从内到外逐级查找变量

/*
 * 作用域链示意：
 *
 * globalContext (作用域链头)
 *   - name: "global"
 *   - outer: null
 *
 *   functionContext (第二层)
 *   - name: "function"
 *   - outer: globalContext
 *
 *     functionContext2 (第三层)
 *     - name: "inner"
 *     - outer: functionContext
 *
 * 作用域链查找：从当前作用域开始，逐级向上查找
 * functionContext2查找变量：
 * 1. 先在自己的scope中查找
 * 2. 没找到，通过outer找到functionContext
 * 3. 还没找到，通过outer找到globalContext
 * 4. 最终没找到，抛出ReferenceError
 */

// 闭包与作用域链
function createCounter() {
  let count = 0; // 这个变量被闭包引用，不会被垃圾回收

  return function() {
    count++;
    return count;
  };
}

const counter = createCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

// count变量仍然存在，因为内部函数形成了闭包
// 作用域链：内部函数 → createCounter → global
```

### 2.2 原型链与链表

```javascript

// 本质上是一种链表的变体

 * 原型链结构：
 *
 * object (隐式原型，指向null)

// Object.create的原理
function myCreate(obj) {

  F.prototype = obj; // 将其原型指向传入的对象
  return new F();
}

console.log(Object.getPrototypeOf(child) === parent); // true

// 链表的实际应用：表单验证链
class ValidationChain {
  constructor(value) {
    this.value = value;
    this.rules = [];
  }

  addRule(validator, message) {
    this.rules.push({ validator, message });
    return this; // 支持链式调用
  }

  validate() {
    for (const rule of this.rules) {
      if (!rule.validator(this.value)) {
        return { valid: false, message: rule.message };
      }
    }
    return { valid: true };
  }
}

const validator = new ValidationChain('')
  .addRule(v => v.length > 0, '不能为空')
  .addRule(v => v.length >= 6, '长度至少6位')
  .addRule(v => /^[a-zA-Z]/.test(v), '必须以字母开头');

console.log(validator.validate());
```

### 2.3 Promise与状态机

```javascript
// Promise本质上是状态机
// 三种状态：pending（待定）、fulfilled（已兑现）、rejected（已拒绝）

/*
 * Promise状态转换：
 *
 *    pending
 *    /      \
 *   ↓        ↓
 * fulfilled  rejected
 *   ↓        ↓
 * (不可逆)  (不可逆)
 */

// 手写Promise（简化版）
class MyPromise {
  constructor(executor) {
    this.state = 'pending';
    this.value = undefined;
    this.handlers = [];

    // 成功处理
    const resolve = (value) => {
      if (this.state !== 'pending') return;
      this.state = 'fulfilled';
      this.value = value;
      this.handlers.forEach(this.executeHandlers);
    };

    // 失败处理
    const reject = (reason) => {
      if (this.state !== 'pending') return;
      this.state = 'rejected';
      this.value = reason;
      this.handlers.forEach(this.executeHandlers);
    };

    try {
      executor(resolve, reject);
    } catch (error) {
      reject(error);
    }
  }

  executeHandlers = ({ onFulfilled, onRejected, resolve, reject }) => {
    // 微任务处理
    queueMicrotask(() => {
      if (this.state === 'fulfilled') {
        const result = onFulfilled ? onFulfilled(this.value) : this.value;
        resolve(result);
      } else if (this.state === 'rejected') {
        const result = onRejected ? onRejected(this.value) : this.value;
        reject(result);
      }
    });
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      this.handlers.push({
        onFulfilled,
        onRejected,
        resolve,
        reject
      });

      if (this.state !== 'pending') {
        this.executeHandlers({ onFulfilled, onRejected, resolve, reject });
      }
    });
  }

  catch(onRejected) {
    return this.then(null, onRejected);
  }

  finally(onFinally) {
    return this.then(
      value => {
        onFinally();
        return value;
      },
      reason => {
        onFinally();
        throw reason;
      }
    );
  }
}

// Promise.all - 所有Promise都成功才成功
async function promiseAll(promises) {
  const results = [];
  let completedCount = 0;

  return new Promise((resolve, reject) => {
    promises.forEach((promise, index) => {
      Promise.resolve(promise)
        .then(result => {
          results[index] = result;
          completedCount++;

          if (completedCount === promises.length) {
            resolve(results);
          }
        })
        .catch(reject);
    });
  });
}

// Promise.race - 任意一个Promise先完成就完成
function promiseRace(promises) {
  return new Promise((resolve, reject) => {
    promises.forEach(promise => {
      Promise.resolve(promise)
        .then(resolve)
        .catch(reject);
    });
  });
}

// Promise.allSettled - 所有Promise都完成后才完成（无论成功失败）
function promiseAllSettled(promises) {
  return Promise.all(
    promises.map(promise =>
      Promise.resolve(promise)
        .then(
          value => ({ status: 'fulfilled', value }),
          reason => ({ status: 'rejected', reason })
        )
    )
  );
}
```

### 2.4 事件委托与数组操作

```javascript
// 事件委托利用了DOM树的层级结构
// 本质上是对树形结构的遍历和操作

// 场景：为一个列表中的每个元素添加点击事件
// 不使用事件委托（性能差）
document.querySelectorAll('.list-item').forEach((item, index) => {
  item.addEventListener('click', () => {
    console.log(`点击了第${index}个元素`);
  });
});

// 使用事件委托（性能好）
document.querySelector('.list').addEventListener('click', (event) => {
  // event.target是被点击的实际元素
  // matches()方法检查元素是否符合选择器
  if (event.target.matches('.list-item')) {
    const index = [...event.target.parentNode.children].indexOf(event.target);
    console.log(`点击了第${index}个元素`);
  }
});

// 事件委托的原理
// 1. 将事件监听器绑定到父元素
// 2. 事件冒泡时，检查event.target
// 3. 根据target决定是否处理事件

// 手写事件委托工具函数
function delegate(parentSelector, childSelector, eventType, handler) {
  const parent = document.querySelector(parentSelector);

  parent.addEventListener(eventType, (event) => {
    const target = event.target;

    // 遍历target及其祖先，找到匹配选择器的元素
    let current = target;
    while (current && current !== parent) {
      if (current.matches(childSelector)) {
        // 绑定正确的this和event对象
        handler.call(current, event);
        break;
      }
      current = current.parentElement;
    }
  });
}

// 使用示例
delegate('.container', '.btn', 'click', function(event) {
  console.log('点击了按钮', this.textContent);
});

// 数组的高效操作
// 前端开发中经常需要对数据进行筛选、映射、排序等操作

// 模拟Vue的响应式数组方法
function createReactiveArray(arr, callback) {
  const reactiveArr = [...arr];

  const methods = ['push', 'pop', 'shift', 'unshift', 'splice', 'sort', 'reverse'];

  methods.forEach(method => {
    reactiveArr[method] = function(...args) {
      const result = Array.prototype[method].apply(this, args);
      callback(); // 触发响应式更新
      return result;
    };
  });

  return reactiveArr;
}

const arr = createReactiveArray([1, 2, 3], () => {
  console.log('数组发生变化');
});

arr.push(4); // 触发回调，输出"数组发生变化"
```

## 三、框架底层原理与计算机基础

### 3.1 Virtual DOM与树结构

```javascript
// Virtual DOM是Facebook提出的概念
// 核心思想：用JavaScript对象模拟DOM树，减少真实DOM操作

// Virtual DOM节点定义
class VNode {
  constructor(tag, props, children) {
    this.tag = tag;           // 标签名
    this.props = props;       // 属性
    this.children = children; // 子节点
    this.key = props?.key;   // 用于diff优化
  }

  // 渲染成真实DOM
  render() {
    const element = document.createElement(this.tag);

    // 设置属性
    for (const [key, value] of Object.entries(this.props || {})) {
      if (key === 'className') {
        element.className = value;
      } else if (key.startsWith('on')) {
        const eventName = key.slice(2).toLowerCase();
        element.addEventListener(eventName, value);
      } else {
        element.setAttribute(key, value);
      }
    }

    // 创建子节点
    for (const child of this.children || []) {
      if (typeof child === 'string') {
        element.appendChild(document.createTextNode(child));
      } else {
        element.appendChild(child.render());
      }
    }

    return element;
  }
}

// 创建Virtual DOM
function h(tag, props, ...children) {
  return new VNode(tag, props, children);
}

// 示例
const vdom = h('div', { className: 'container' },
  h('h1', null, '标题'),
  h('p', null, '这是一段文字'),
  h('button', { onClick: () => console.log('点击') }, '点击我')
);

// Virtual DOM的diff算法
function diff(oldVNode, newVNode) {
  const patches = [];

  // 1. 类型不同，直接替换
  if (typeof oldVNode !== typeof newVNode ||
      oldVNode.tag !== newVNode.tag) {
    patches.push({ type: 'REPLACE', node: newVNode });
    return patches;
  }

  // 2. 类型相同，比较属性
  if (oldVNode.tag) {
    const propsPatch = diffProps(oldVNode.props, newVNode.props);
    if (Object.keys(propsPatch).length > 0) {
      patches.push({ type: 'PROPS', props: propsPatch });
    }

    // 3. 比较子节点
    diffChildren(oldVNode.children, newVNode.children, patches);
  } else {
    // 文本节点，内容不同则替换
    if (oldVNode !== newVNode) {
      patches.push({ type: 'TEXT', text: newVNode });
    }
  }

  return patches;
}

// diffProps：比较属性差异
function diffProps(oldProps, newProps) {
  const patches = {};
  const allKeys = new Set([...Object.keys(oldProps), ...Object.keys(newProps)]);

  for (const key of allKeys) {
    if (oldProps[key] !== newProps[key]) {
      patches[key] = newProps[key];
    }
  }

  return patches;
}

// diffChildren：比较子节点
// 使用key来优化列表diff（类似React的key）
function diffChildren(oldChildren, newChildren, patches) {
  const oldKeys = new Map(oldChildren.map((c, i) => [c.key || i, { node: c, index: i }])) ;

  let newKeys = new Map(newChildren.map((c, i) => [c.key || i, { node: c, index: i }]));

  // 遍历新children，处理新增和变化
  for (const [key, { node: newNode, index: newIndex }] of newKeys) {
    if (oldKeys.has(key)) {
      // 存在，递归比较
      const { node: oldNode } = oldKeys.get(key);
      diff(oldNode, newNode);
    } else {
      // 新增
      patches.push({ type: 'INSERT', node: newNode, index: newIndex });
    }
  }

  // 遍历旧children，处理删除
  for (const [key, { node: oldNode, index: oldIndex }] of oldKeys) {
```

```javascript
// Vue 3的响应式系统使用ES6 Proxy实现
// 本质上是发布订阅模式的应用

// 手写简化版响应式系统
function reactive(obj) {
  // 收集依赖
  const dependences = new Map();

  // 存储所有订阅者
  const subscribers = new Set();

  // 收集当前活跃的订阅者
  let activeSubscriber = null;

  function subscribe(callback) {
    activeSubscriber = callback;
    callback(); // 立即执行，触发getter进行依赖收集
    activeSubscriber = null;
  }

  const proxy = new Proxy(obj, {
    get(target, key, receiver) {
      const value = Reflect.get(target, key, receiver);

      // 如果有活跃订阅者，收集依赖
      if (activeSubscriber) {
        if (!dependences.has(key)) {
          dependences.set(key, new Set());
        }
        dependences.get(key).add(activeSubscriber);
      }

      // 如果是对象，递归处理
      if (value !== null && typeof value === 'object') {
        return reactive(value);
      }

      return value;
    },

    set(target, key, value, receiver) {
      const oldValue = target[key];

      // 设置新值
      Reflect.set(target, key, value, receiver);

      // 只在值确实变化时通知
      if (oldValue !== value) {
        // 通知该key的所有订阅者
        const subs = dependences.get(key);
        if (subs) {
          subs.forEach(callback => callback());
        }

        // 通知所有订阅者（全局更新）
        subscribers.forEach(callback => callback());
      }

      return true;
    }
  });

  // effect：创建副作用/订阅
  proxy.$effect = function(callback) {
    subscribers.add(callback);

    // 初始化时收集依赖
    subscribe(callback);

    // 返回取消订阅函数
    return () => {
      subscribers.delete(callback);
    };
  };

  return proxy;
}

// 使用示例
const state = reactive({
  count: 0,
  name: '张三'
});

// 订阅count变化
state.$effect(() => {
  console.log(`count变化了：${state.count}`);
});

// 订阅所有变化
state.$effect(() => {
  console.log(`任意属性变化：${state.count}, ${state.name}`);
});

state.count++; // 触发订阅
state.name = '李四'; // 触发订阅

// 计算属性
function computed(getter) {
  let value;
  let isDirty = true;

  const effect = () => {
    isDirty = true;
    value = getter();
  };

  return {
    get value() {
      if (isDirty) {
        subscribe(effect);
        isDirty = false;
      }
      return value;
    }
  };
}

// 使用计算属性
```

```javascript
function matchRoute(path) {
  // 精确匹配
  if (routeTable[path]) {
    return { path, component: routeTable[path], params: {} };
  }

  // 通配符匹配
  for (const [pattern, component] of Object.entries(routeTable)) {
    const paramNames = [];
    const regexPattern = pattern.replace(/:([^/]+)/g, (_, name) => {
      paramNames.push(name);
      return '([^/]+)';
    });

    const regex = new RegExp(`^${regexPattern}$`);
    const match = path.match(regex);

    if (match) {
      const params = {};
      paramNames.forEach((name, index) => {
        params[name] = match[index + 1];
      });

      return { path, component, params };
    }
  }
```

```javascript
// 场景：限制同时只能有N个请求在发送
class RequestQueue {
  constructor(concurrency = 6) {
    this.concurrency = concurrency; // 并发数限制
    this.running = 0;             // 当前运行数
    this.queue = [];             // 请求队列
  }

  // 添加请求
  add(request) {
    return new Promise((resolve, reject) => {
      this.queue.push({ request, resolve, reject });
      this.process();
    });
  }

  // 处理队列
  process() {
    // 如果队列为空或已达并发上限，等待
    while (this.running < this.concurrency && this.queue.length > 0) {
      const { request, resolve, reject } = this.queue.shift();

      this.running++;

      // 执行请求
      request()
        .then(resolve)
        .catch(reject)
        .finally(() => {
          this.running--;
          this.process(); // 继续处理队列
        });
    }
  }
}

class RequestCache {
  constructor() {
    this.cache = new Map();
    this.pending = new Map(); // 存储正在进行的请求
  }

  get(url) {
    const cached = this.cache.get(url);
    if (cached && Date.now() - cached.timestamp < cached.ttl) {
      return Promise.resolve(cached.data);
    }
    return null;
  }

  set(url, data, ttl = 60000) {
    this.cache.set(url, {
      data,
      timestamp: Date.now(),
      ttl
    });
  }

  // 请求复用：相同URL的请求只发送一次
  async fetch(url, options = {}) {
    // 检查缓存
    const cached = this.get(url);
    if (cached) return cached;

    // 检查是否有正在进行的请求
    if (this.pending.has(url)) {
      return this.pending.get(url);
    }

    // 创建新请求
    const promise = fetch(url, options)
      .then(response => response.json())
      .then(data => {
        this.set(url, data, options.ttl);
        this.pending.delete(url);
        return data;
      })
      .catch(error => {
        this.pending.delete(url);
        throw error;
      });

    this.pending.set(url, promise);
    return promise;
  }
}
```

```javascript
// 轮询实现：使用队列管理待处理的消息

class PollingManager {
  constructor(fetchFn, interval = 5000) {
    this.fetchFn = fetchFn;
    this.interval = interval;
    this.isPolling = false;
    this.queue = [];     // 消息队列
    this.handlers = [];   // 消息处理器
    this.lastData = null; // 上一次获取的数据
  }

  // 启动轮询
  start() {
    if (this.isPolling) return;

    this.isPolling = true;
    this.poll();

    return this;
  }

  // 停止轮询
  stop() {
    this.isPolling = false;
    return this;
  }

  // 轮询逻辑
  async poll() {
    if (!this.isPolling) return;

    try {
      const data = await this.fetchFn();

      // 比较新旧数据，找出差异
      const newItems = this.diff(this.lastData, data);

      if (newItems.length > 0) {
        // 新数据加入队列
        this.queue.push(...newItems);
        // 通知所有处理器
        this.notify();
      }

      this.lastData = data;
    } catch (error) {
      console.error('轮询失败:', error);
    }

    // 定时器实现轮询
    setTimeout(() => this.poll(), this.interval);
  }

  // 订阅消息
  subscribe(handler) {
    this.handlers.push(handler);

    // 返回取消订阅函数
    return () => {
      const index = this.handlers.indexOf(handler);
      if (index > -1) {
        this.handlers.splice(index, 1);
      }
    };
  }

  // 通知所有处理器
  notify() {
    this.handlers.forEach(handler => handler(this.queue));

    // 清空队列
    this.queue = [];
  }

  // 计算差异（简化版，使用ID比较）
  diff(oldData, newData) {
    if (!oldData) return newData || [];

    const oldIds = new Set((oldData || []).map(item => item.id));
    return (newData || []).filter(item => !oldIds.has(item.id));
  }
}

  5000 // 5秒轮询一次
);
```

```javascript
// Node.js的EventEmitter是发布订阅模式的实现
// 核心是一个事件表，存储事件名到处理器数组的映射

class EventEmitter {
  constructor() {
    this.events = new Map(); // 事件表：事件名 → 处理器数组
  }

  // 绑定事件
  on(eventName, handler) {
    if (!this.events.has(eventName)) {
      this.events.set(eventName, []);
    }
    this.events.get(eventName).push(handler);

    // 返回取消订阅函数
    return () => this.off(eventName, handler);
  }

  // 绑定一次性事件
  once(eventName, handler) {
    const wrapper = (...args) => {
      handler(...args);
      this.off(eventName, wrapper);
    };
    return this.on(eventName, wrapper);
  }

  // 解绑事件
  off(eventName, handler) {
    if (!this.events.has(eventName)) return;

    const handlers = this.events.get(eventName);
    const index = handlers.indexOf(handler);
    if (index > -1) {
      handlers.splice(index, 1);
    }
  }

  // 触发事件
  emit(eventName, ...args) {
    if (!this.events.has(eventName)) return;

    const handlers = [...this.events.get(eventName)];
    handlers.forEach(handler => {
      try {
        handler(...args);
      } catch (error) {
        console.error(`事件 ${eventName} 处理出错:`, error);
      }
    });
  }

  // 返回已订阅的事件列表
  eventNames() {
    return [...this.events.keys()];
  }

  // 返回某个事件的处理器数量
  listenerCount(eventName) {
    return this.events.get(eventName)?.length || 0;
  }
}

// 使用示例
const emitter = new EventEmitter();

// 订阅
const unsubscribe = emitter.on('message', (data) => {
  console.log('收到消息:', data);
});

  console.log('另一个处理器:', data);
});

// 一次性事件
emitter.once('connect', () => {
  console.log('连接成功（只触发一次）');
});

// 触发事件

emitter.emit('connect');

// 取消订阅

class InterruptablEventEmitter extends EventEmitter {
  emit(eventName, ...args) {
    if (!this.events.has(eventName)) return;

    const handlers = [...this.events.get(eventName)];
    for (const handler of handlers) {
      const result = handler(...args);
      // 如果返回false，中断传播
```

```javascript
function encode(str) {
  const buf = Buffer.alloc(str.length * 3);

  let offset = 0;
  for (const char of str) {
    const code = char.codePointAt(0);

    if (code <= 0x7F) {
      // ASCII字符，单字节
      buf[offset++] = code;
    } else if (code <= 0x7FF) {
      // 110xxxxx 10xxxxxx
      buf[offset++] = 0xC0 | (code >> 6);
      buf[offset++] = 0x80 | (code & 0x3F);
    } else {
      // 1110xxxx 10xxxxxx 10xxxxxx
      buf[offset++] = 0xE0 | (code >> 12);
      buf[offset++] = 0x80 | ((code >> 6) & 0x3F);
      buf[offset++] = 0x80 | (code & 0x3F);
    }
  }

  return buf.slice(0, offset);
}

function decode(buf) {
  let result = '';

  for (let i = 0; i < buf.length;) {
    const byte = buf[i];

    let code;
    if ((byte & 0x80) === 0) {
      // 0xxxxxxx - 单字节
      code = byte;
      i += 1;
    } else if ((byte & 0xE0) === 0xC0) {
      // 110xxxxx - 双字节
      code = ((byte & 0x1F) << 6) |
             (buf[i + 1] & 0x3F);
      i += 2;
    } else {
      // 1110xxxx - 三字节
      code = ((byte & 0x0F) << 12) |
             ((buf[i + 1] & 0x3F) << 6) |
             (buf[i + 2] & 0x3F);
      i += 3;
    }

    result += String.fromCodePoint(code);
  }

  return result;
}
```

### 6.1 防抖节流与队列

```javascript
// 防抖（Debounce）：事件触发n秒后执行，n秒内再次触发则重新计时
// 核心思想：使用定时器，每次触发都清空之前的定时器

function debounce(fn, wait) {
  let timeoutId = null;

  return function (...args) {
    // 每次调用都清除之前的定时器
    if (timeoutId) {
      clearTimeout(timeoutId);
    }

    // 设置新的定时器
    timeoutId = setTimeout(() => {
      fn.apply(this, args);
      timeoutId = null;
    }, wait);
  };
}

// 防抖的高级版本：支持立即执行和取消
function debounceAdvanced(fn, wait, options = {}) {
  let timeoutId = null;
  let lastArgs = null;

  const { immediate = false, maxWait = null } = options;
  let maxTimeoutId = null;

  function invoke() {
    if (lastArgs) {
      fn.apply(this, lastArgs);
      lastArgs = null;
    }
  }

  function debounced(...args) {
    lastArgs = args;

    if (timeoutId) {
      clearTimeout(timeoutId);
    }

    if (immediate && !timeoutId) {
      fn.apply(this, args);
    }

    timeoutId = setTimeout(() => {
      invoke();
      timeoutId = null;
    }, wait);

    // 最大等待时间
    if (maxWait !== null && !maxTimeoutId) {
      maxTimeoutId = setTimeout(() => {
        invoke();
        maxTimeoutId = null;
      }, maxWait);
    }
  }

  // 取消函数
  debounced.cancel = () => {
    if (timeoutId) {
      clearTimeout(timeoutId);
      timeoutId = null;
    }
    if (maxTimeoutId) {
      clearTimeout(maxTimeoutId);
      maxTimeoutId = null;
    }
    lastArgs = null;
  };

  return debounced;
}

// 节流（Throttle）：规定时间内只执行一次
// 核心思想：使用时间戳或定时器

// 时间戳版本（第一次触发立即执行）
function throttle(fn, wait) {
  let previous = 0;

  return function (...args) {
    const now = Date.now();

    if (now - previous >= wait) {
      fn.apply(this, args);
      previous = now;
    }
  };
}

// 定时器版本（最后一次触发也会执行）
function throttleTrailing(fn, wait) {
  let timeoutId = null;

  return function (...args) {
    if (!timeoutId) {
      timeoutId = setTimeout(() => {
        fn.apply(this, args);
        timeoutId = null;
      }, wait);
    }
  };
}

// 双版本结合（第一次和最后一次都会执行）
function throttle(fn, wait) {
  let timeoutId = null;
  let previous = 0;

  return function (...args) {
    const now = Date.now();

    if (!previous) {
      previous = now;
    }

    const remaining = wait - (now - previous);

    if (remaining <= 0 || remaining > wait) {
      if (timeoutId) {
        clearTimeout(timeoutId);
        timeoutId = null;
      }
      fn.apply(this, args);
      previous = now;
    } else if (!timeoutId) {
      timeoutId = setTimeout(() => {
        fn.apply(this, args);
        previous = Date.now();
        timeoutId = null;
      }, remaining);
    }
  };
}

// 应用场景
// 防抖：搜索框输入（等用户停止输入后才搜索）、窗口调整大小
const handleSearch = debounce((keyword) => {
  console.log('搜索:', keyword);
}, 300);

// 节流：滚动事件、按钮防重复点击
const handleScroll = throttle(() => {
  console.log('滚动位置:', window.scrollY);
}, 100);
```

### 6.2 虚拟列表与哈希表

```javascript
// 虚拟列表：只渲染可见区域的列表项
// 解决大数据量列表的渲染性能问题

// 核心思想：用哈希表存储已渲染的列表项

class VirtualList {
  constructor(container, items, itemHeight, renderItem) {
    this.container = container;     // 容器元素
    this.items = items;             // 完整数据
    this.itemHeight = itemHeight;    // 每项高度
    this.renderItem = renderItem;   // 渲染函数
    this.renderedItems = new Map(); // 哈希表：index → 元素

    this.container.style.overflow = 'auto';
    this.container.style.position = 'relative';

    // 计算可见项数量
    this.visibleCount = Math.ceil(container.clientHeight / itemHeight);

    // 创建内容区域（撑起滚动条）
    this.content = document.createElement('div');
    this.content.style.height = `${items.length * itemHeight}px`;
    this.content.style.position = 'absolute';
    this.container.appendChild(this.content);

    // 绑定滚动事件
    this.container.addEventListener('scroll', () => this.onScroll());

    // 初始渲染
    this.render();
  }

  onScroll() {
    // 使用requestAnimationFrame优化性能
    requestAnimationFrame(() => this.render());
  }

  render() {
    const scrollTop = this.container.scrollTop;

    // 计算可见范围的起始和结束索引
    const startIndex = Math.floor(scrollTop / this.itemHeight);
    const endIndex = startIndex + this.visibleCount + 1;

    // 清理不在可见范围的项
    for (const [index, element] of this.renderedItems) {
      if (index < startIndex || index >= endIndex) {
        element.remove();
        this.renderedItems.delete(index);
      }
    }

    // 渲染可见范围的项
    for (let i = startIndex; i < endIndex; i++) {
      if (!this.renderedItems.has(i) && i >= 0 && i < this.items.length) {
        const element = this.renderItem(this.items[i], i);
        element.style.position = 'absolute';
        element.style.top = `${i * this.itemHeight}px`;
        element.style.left = '0';
        element.style.right = '0';
        element.style.height = `${this.itemHeight}px`;

        this.content.appendChild(element);
        this.renderedItems.set(i, element);
      }
    }
  }

  // 更新数据
  setItems(items) {
    this.items = items;
    this.content.style.height = `${items.length * this.itemHeight}px`;

    // 清空已渲染
    for (const element of this.renderedItems.values()) {
      element.remove();
    }
    this.renderedItems.clear();

    // 重新渲染
    this.render();
  }

  // 滚动到指定位置
  scrollToIndex(index) {
    this.container.scrollTop = index * this.itemHeight;
  }
}

const virtualList = new VirtualList(

  data,
  50, // 每项高度50px
  (item) => {
    const div = document.createElement('div');
    div.textContent = item.name;

  }

class PullToRefresh {
  constructor(container, onRefresh) {
    this.container = container;
    this.onRefresh = onRefresh;
    this.isRefreshing = false;

    this.currentY = 0;

    container.addEventListener('touchstart', (e) => this.onTouchStart(e));
    container.addEventListener('touchmove', (e) => this.onTouchMove(e));
    container.addEventListener('touchend', () => this.onTouchEnd());
  }

  onTouchStart(e) {
    if (this.isRefreshing) return;
    this.startY = e.touches[0].clientY;

    if (this.isRefreshing) return;
    this.currentY = e.touches[0].clientY;

    if (this.currentY > this.startY && this.container.scrollTop === 0) {
      const pullDistance = this.currentY - this.startY;
      // 显示loading指示器
      this.showLoading(pullDistance);
    }
  }

  onTouchEnd() {
    const pullDistance = this.currentY - this.startY;

    if (pullDistance > 60) {
      this.isRefreshing = true;
      this.onRefresh().finally(() => {
        this.isRefreshing = false;
        this.hideLoading();
      });
    } else {
      this.hideLoading();
    }
  }
```
