```javascript
// 理解数组的内存模型：连续内存空间
// 这就是为什么中间操作是O(n)的本质原因
```

**链表（Linked List）**

```javascript
// 单向链表节点
class ListNode {
  constructor(val) {
    this.val = val;
    this.next = null;
  }
}

// 链表反转 - 前端面试高频题
function reverseLinkList(head) {
  let prev = null;
  let curr = head;

  while (curr !== null) {
    const next = curr.next; // 先保存下一个节点
    curr.next = prev;       // 反转当前节点的指向
    prev = curr;            // prev前移
    curr = next;            // curr前移
  }

  return prev; // 新的头节点
}

// 链表 vs 数组
// - 链表：插入删除O(1)，查找O(n)，不需要连续内存
// - 数组：插入删除O(n)，查找O(1)，需要连续内存
```

**栈（Stack）和队列（Queue）**

```javascript
// 栈：后进先出（LIFO）
// 实际应用：函数调用栈、表达式求值、括号匹配
const stack = [];
stack.push(1);    // 入栈
stack.pop();      // 出栈
stack.peek();     // 查看栈顶
stack.isEmpty();  // 是否为空

// 队列：先进先出（FIFO）
// 实际应用：任务队列、事件循环、打印队列
class Queue {
  constructor() {
    this.items = [];
  }

  enqueue(item) {
    this.items.push(item);
  }

  dequeue() {
    return this.items.shift(); // Array的shift是O(n)操作
  }
}

// 优化：使用对象实现更高效的队列
class EfficientQueue {
  constructor() {
    this.items = {};
    this.head = 0;
    this.tail = 0;
  }

  enqueue(item) {
    this.items[this.tail++] = item;
  }

  dequeue() {
    if (this.isEmpty()) return null;
    const item = this.items[this.head];
    delete this.items[this.head++];
    return item;
  }

  isEmpty() {
    return this.head === this.tail;
  }
}
```

#### 2.1.2 树形结构

**二叉树**

```javascript
// 二叉树节点定义
class TreeNode {
  constructor(val) {
    this.val = val;
    this.left = null;
    this.right = null;
  }
}

// 二叉树的遍历 - 前端理解Virtual DOM的基础
// 前序遍历：根-左-右
function preOrderTraversal(root) {
  const result = [];

  function traverse(node) {
    if (node === null) return;
    result.push(node.val);    // 访问根
    traverse(node.left);      // 遍历左子树
    traverse(node.right);     // 遍历右子树
  }

  traverse(root);
  return result;
}

// 中序遍历：左-根-右（ BST的中序遍历是有序的）
function inOrderTraversal(root) {
  const result = [];

  function traverse(node) {
    if (node === null) return;
    traverse(node.left);
    result.push(node.val);
    traverse(node.right);
  }

  traverse(root);
  return result;
}

// 后序遍历：左-右-根
function postOrderTraversal(root) {
  const result = [];

  function traverse(node) {
    if (node === null) return;
    traverse(node.left);
    traverse(node.right);
    result.push(node.val);
  }

  traverse(root);
  return result;
}

// 层序遍历（广度优先）
function levelOrderTraversal(root) {
  if (root === null) return [];

  const result = [];
  const queue = [root];

  while (queue.length > 0) {
    const levelSize = queue.length;
    const currentLevel = [];

    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift();
      currentLevel.push(node.val);

      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }

    result.push(currentLevel);
  }

  return result;
}
```

**二叉搜索树（BST）**

```javascript
// BST的性质：左子树所有节点小于根，右子树所有节点大于根
// 查找、插入、删除的平均时间复杂度是O(log n)

class BST {
  constructor() {
    this.root = null;
  }

  // 插入
  insert(val) {
    const newNode = new TreeNode(val);

    if (this.root === null) {
      this.root = newNode;
      return;
    }

    this.insertNode(this.root, newNode);
  }

  insertNode(node, newNode) {
    if (newNode.val < node.val) {
      if (node.left === null) {
        node.left = newNode;
      } else {
        this.insertNode(node.left, newNode);
      }
    } else {
      if (node.right === null) {
        node.right = newNode;
      } else {
        this.insertNode(node.right, newNode);
      }
    }
  }

  // 查找
  search(val) {
    return this.searchNode(this.root, val);
  }

  searchNode(node, val) {
    if (node === null) return false;

    if (val < node.val) {
      return this.searchNode(node.left, val);
    } else if (val > node.val) {
      return this.searchNode(node.right, val);
    } else {
      return true;
    }
  }
}
```

**堆（Heap）**

```javascript
// 堆是一种完全二叉树，常用于优先级队列
// 最大堆：父节点大于等于子节点
// 最小堆：父节点小于等于子节点

class MinHeap {
  constructor() {
    this.heap = [];
  }

  // 获取父节点、左子节点、右子节点的索引
  getParentIndex(i) {
    return Math.floor((i - 1) / 2);
  }

  getLeftChildIndex(i) {
    return 2 * i + 1;
  }

  getRightChildIndex(i) {
    return 2 * i + 2;
  }

  // 交换元素
  swap(i, j) {
    [this.heap[i], this.heap[j]] = [this.heap[j], this.heap[i]];
  }

  // 上浮操作 - 插入后调整
  shiftUp(index) {
    while (index > 0) {
      const parentIndex = this.getParentIndex(index);

      if (this.heap[parentIndex] <= this.heap[index]) {
        break;
      }

      this.swap(parentIndex, index);
      index = parentIndex;
    }
  }

  // 下沉操作 - 删除后调整
  shiftDown(index) {
    while (this.getLeftChildIndex(index) < this.heap.length) {
      let smallestChildIndex = this.getLeftChildIndex(index);
      const rightIndex = this.getRightChildIndex(index);

      if (rightIndex < this.heap.length &&
          this.heap[rightIndex] < this.heap[smallestChildIndex]) {
        smallestChildIndex = rightIndex;
      }

      if (this.heap[index] <= this.heap[smallestChildIndex]) {
        break;
      }

      this.swap(index, smallestChildIndex);
      index = smallestChildIndex;
    }
  }

  // 插入
  insert(val) {
    this.heap.push(val);
    this.shiftUp(this.heap.length - 1);
  }

  // 弹出最小值
  extractMin() {
    if (this.heap.length === 0) return null;
    if (this.heap.length === 1) return this.heap.pop();

    const min = this.heap[0];
    this.heap[0] = this.heap.pop();
    this.shiftDown(0);

    return min;
  }

  // 获取最小值（不删除）
  peek() {
    return this.heap[0];
  }
}

// 堆的应用场景：
// 1. 优先级队列（任务调度）
// 2. Top K 问题
// 3. 合并有序文件
// 4. 实时流数据的中位数
```

#### 2.1.3 哈希表

```javascript
// 哈希表：key-value映射，通过哈希函数直接计算出存储位置
// 平均时间复杂度：插入、删除、查找都是O(1)

class HashTable {
  constructor(size = 100) {
    this.size = size;
    this.buckets = new Array(size).fill(null).map(() => []);
  }

  // 哈希函数
  hash(key) {
    let hash = 0;
    for (let i = 0; i < key.length; i++) {
      hash = (hash * 31 + key.charCodeAt(i)) % this.size;
    }
    return hash;
  }

  // 插入
  set(key, value) {
    const index = this.hash(key);
    const bucket = this.buckets[index];

    // 如果key已存在，则更新
    for (let i = 0; i < bucket.length; i++) {
      if (bucket[i].key === key) {
        bucket[i].value = value;
        return;
      }
    }

    bucket.push({ key, value });
  }

  // 获取
  get(key) {
    const index = this.hash(key);
    const bucket = this.buckets[index];

    for (const item of bucket) {
      if (item.key === key) {
        return item.value;
      }
    }

    return undefined;
  }

  // 删除
  delete(key) {
    const index = this.hash(key);
    const bucket = this.buckets[index];

    for (let i = 0; i < bucket.length; i++) {
      if (bucket[i].key === key) {
        bucket.splice(i, 1);
        return true;
      }
    }

    return false;
  }
}

// JavaScript中Map的实现原理就是哈希表
// 但Map使用的是更高效的哈希函数，并且可以存储任意类型的key

const map = new Map();
map.set('name', '张三');
map.set({ id: 1 }, '对象作为key'); // 可以用对象作为key
map.get('name'); // '张三'

// Set也是基于哈希表实现的
const set = new Set([1, 2, 3, 2, 1]);
console.log([...set]); // [1, 2, 3] 自动去重
```

#### 2.1.4 图

```javascript
// 图：顶点集合 + 边集合
// 应用：社交网络、路由算法、依赖关系

class Graph {
  constructor() {
    this.adjacencyList = new Map(); // 邻接表
  }

  // 添加顶点
  addVertex(vertex) {
    if (!this.adjacencyList.has(vertex)) {
      this.adjacencyList.set(vertex, []);
    }
  }

  // 添加边
  addEdge(vertex1, vertex2) {
    if (!this.adjacencyList.has(vertex1)) {
      this.addVertex(vertex1);
    }
    if (!this.adjacencyList.has(vertex2)) {
      this.addVertex(vertex2);
    }

    this.adjacencyList.get(vertex1).push(vertex2);
    this.adjacencyList.get(vertex2).push(vertex1); // 无向图
  }

  // 深度优先遍历（DFS）
  dfs(start) {
    const visited = new Set();
    const result = [];

    const dfsHelper = (vertex) => {
      visited.add(vertex);
      result.push(vertex);

      const neighbors = this.adjacencyList.get(vertex) || [];
      for (const neighbor of neighbors) {
        if (!visited.has(neighbor)) {
          dfsHelper(neighbor);
        }
      }
    };

    dfsHelper(start);
    return result;
  }

  // 广度优先遍历（BFS）
  bfs(start) {
    const visited = new Set();
    const result = [];
    const queue = [start];
    visited.add(start);

    while (queue.length > 0) {
      const vertex = queue.shift();
      result.push(vertex);

      const neighbors = this.adjacencyList.get(vertex) || [];
      for (const neighbor of neighbors) {
        if (!visited.has(neighbor)) {
          visited.add(neighbor);
          queue.push(neighbor);
        }
      }
    }

    return result;
  }
}

// 图的应用场景：
// 1. 前端路由：页面跳转关系
// 2. 依赖分析：npm包的依赖图
// 3. 状态管理：Redux/Vuex的状态流转图
// 4. Virtual DOM：组件树结构
```

```javascript
// Virtual DOM本质上就是一棵树形结构
// diff算法就是两棵树的比较过程

// 简化版diff算法思路：
function diff(oldTree, newTree) {
  const patches = [];

  // 1. 比较根节点
  if (oldTree.tag !== newTree.tag) {
    // 标签不同，直接替换
    patches.push({ type: 'REPLACE', node: newTree });
  } else {
    // 标签相同，比较属性
    const attrsPatch = diffAttrs(oldTree.attrs, newTree.attrs);
    if (Object.keys(attrsPatch).length > 0) {
      patches.push({ type: 'ATTRS', attrs: attrsPatch });
    }

    // 2. 递归比较子节点
    diffChildren(oldTree.children, newTree.children, patches);
  }

  return patches;
}

// 理解diff的优化策略：
// 1. 同层比较：只比较同一层级的节点，不跨层级比较
// 2. key优化：使用key标识节点，帮助diff识别哪些节点可以复用
// 3. 列表对比：使用最长递增子序列算法（LIS）最小化移动
```

#### 2.2.3 防抖节流与栈

```javascript
// 防抖（Debounce）：事件触发n秒后执行，如果n秒内再次触发，则重新计时
// 原理：使用定时器，每次触发都清除之前的定时器

function debounce(func, wait) {
  let timeout = null;

  return function (...args) {
    // 每次调用都清除之前的定时器
    if (timeout) clearTimeout(timeout);

    timeout = setTimeout(() => {
      func.apply(this, args);
    }, wait);
  };
}

// 防抖的应用：搜索框输入（输入完成后才搜索）、窗口 resize

// 节流（Throttle）：规定时间内只执行一次
// 原理：使用时间戳或定时器

function throttle(func, wait) {
  let previous = 0;

  return function (...args) {
    const now = Date.now();

    if (now - previous > wait) {
      func.apply(this, args);
      previous = now;
    }
  };
}

// 节流的应用：滚动事件、按钮防重复点击
```

#### 3.2.4 HTTP/1.1与HTTP/2的区别

```javascript
// HTTP/1.1的问题：

// 1. 队头阻塞（Head of Line Blocking）
//    一个请求耗时很长，后续请求必须等待
// 解决：多个TCP连接、域名分片

// 2. 头部冗余
//    每次请求都要携带大量相同的头部信息
// 解决：HTTP/2的头部压缩

// 3. 未压缩的Body
//    请求体和响应体没有压缩
// 解决：HTTP/2默认压缩

// HTTP/2的改进：

// 1. 多路复用（Multiplexing）
//    一个TCP连接可以同时发送多个请求和响应
//    解决：二进制分帧层

// 2. 头部压缩（HPACK）
//    使用静态表和动态表压缩头部
//    减少30-40%的头部大小

// 3. 服务器推送（Server Push）
//    服务器可以主动推送资源
//    减少客户端请求

// 4. 请求优先级
//    可以为请求设置优先级
//    重要资源优先传输

// HTTP/3（QUIC）
// - 基于UDP协议
// - 0-RTT连接建立
// - 彻底解决队头阻塞问题
```

### 3.3 TCP与UDP

#### 3.3.1 TCP三次握手

```javascript
// TCP是面向连接的协议，通过三次握手建立可靠连接

/*
 *                    Client                      Server
 *                      |                          |
 *                      |----- SYN (seq=x) ------>|   第一次握手
 *                      |                          |
 *                      |<---- SYN-ACK -----------|   第二次握手
 *                      |    (seq=y, ack=x+1)      |
 *                      |                          |
 *                      |----- ACK (ack=y+1) ---->|   第三次握手
 *                      |                          |
 *                      |     建立连接完成         |
 *                      |                          |
 */

// 为什么是三次？
// - 第一次：服务器确认"客户端的发送能力正常"
// - 第二次：客户端确认"服务端的接收和发送能力正常"
// - 第三次：服务器确认"客户端的接收能力正常"

// 如果是两次握手：
// 客户端发送SYN后超时，重新发送SYN2
// 但SYN1后来到达服务器，服务器会认为是一个新连接
// 导致资源浪费和状态混乱

// 四次挥手（断开连接）
/*
 *                    Client                      Server
 *                      |                          |
 *                      |----- FIN ----瓜--------->|   第一次挥手
 *                      |<---- ACK ----------------|   第二次挥手
 *                      |    (服务器可能还有数据)    |
 *                      |<---- FIN ----------------|   第三次挥手
 *                      |----- ACK ---------------->|   چهار挥手
 *                      |                          |
 */
```

#### 3.3.2 TCP可靠传输原理

```javascript
// 1. 序列号（Sequence Number）
//    每个字节都有一个序列号
//    用于标识数据的顺序和完整性

// 2. 确认应答（ACK）
//    接收方收到数据后，发送ACK确认
//    超时重传：如果发送方在超时时间内没收到ACK，就重发数据

// 3. 流量控制
//    滑动窗口机制
//    接收方告诉发送方自己还能接收多少数据
//    避免接收方缓存溢出

// 4. 拥塞控制
//    慢启动：从小的拥塞窗口开始
//    拥塞避免：窗口达到阈值后，缓慢增长
//    快重传：收到3个重复ACK时立即重传
//    快恢复：跳过慢启动，从较大窗口开始

// TCP vs UDP
// TCP：面向连接、可靠、有序、有流量控制、有拥塞控制
// UDP：无连接、不可靠、无序、无流量控制、无拥塞控制
// 选择依据：是否需要可靠传输，对实时性要求如何
```

### 3.5 HTTPS加密原理

```javascript
// HTTPS = HTTP + TLS/SSL

// TLS握手过程（简化版）
/*
 *                    Client                      Server
 *                      |                          |
 * 1. 客户端问候        |----- ClientHello -------->|   支持的TLS版本、加密套件、随机数
 *                      |                          |
 * 2. 服务器问候        |<---- ServerHello ---------|   选定的加密套件、随机数
 *                      |<---- 证书 ----------------|   服务器证书（公钥+身份信息）
 *                      |<---- ServerHelloDone -----|   服务器问候结束
 *                      |                          |
 * 3. 密钥交换          |----- ClientKeyExchange --->  加密的预备主密钥
 *                      |----- ChangeCipherSpec --->  密钥材料已准备好
 *                      |----- Finished ------------->  握手消息摘要（验证）
 *                      |                          |
 * 4. 加密通信开始      |<---- ChangeCipherSpec ----|
 *                      |<---- Finished -------------|
 *                      |                          |
 *                      |   加密的HTTP数据          |
 *                      |<==========================>|
 */

// 对称加密 vs 非对称加密

// 对称加密：加密和解密使用同一个密钥
// 优点：速度快
// 缺点：密钥传输不安全
const key = 'secret_key_123'; // 双方需要共享这个密钥
const encrypted = symmetricEncrypt(data, key);

// 非对称加密：加密用公钥，解密用私钥
// 优点：公钥可以公开，安全性高
// 缺点：速度慢
const { publicKey, privateKey } = generateKeyPair();
const encrypted = asymmetricEncrypt(data, publicKey); // 用公钥加密
const decrypted = asymmetricDecrypt(encrypted, privateKey); // 用私钥解密

// HTTPS的实际做法：
// 1. 使用非对称加密交换密钥（解决对称加密的密钥传输问题）
// 2. 然后使用对称加密进行数据传输（利用对称加密的速度优势）

// 数字证书与CA
// 证书内容：公钥、域名、有效期、签名等
// CA（证书颁发机构）：验证服务器身份，颁发证书
// 浏览器内置根证书：信任链的起点
```

```javascript
// 重连机制实现
class ReconnectingWebSocket {
  constructor(url, options = {}) {
    this.url = url;
    this.reconnectInterval = options.reconnectInterval || 1000;
    this.maxReconnectAttempts = options.maxReconnectAttempts || 5;
    this.reconnectAttempts = 0;
    this.connect();
  }

  connect() {
    this.ws = new WebSocket(this.url);

    this.ws.onopen = (event) => {
      this.reconnectAttempts = 0;
      this.onopen?.(event);
    };

    this.ws.onmessage = (event) => {
      this.onmessage?.(event);
    };

    this.ws.onclose = (event) => {
      this.onclose?.(event);
      this.attemptReconnect();
    };

    this.ws.onerror = (error) => {
      this.onerror?.(error);
    };
  }

  attemptReconnect() {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      console.log('达到最大重连次数');
      return;
    }

    this.reconnectAttempts++;
    console.log(`尝试第${this.reconnectAttempts}次重连`);

    setTimeout(() => {
      this.connect();
    }, this.reconnectInterval * this.reconnectAttempts);
  }

  send(data) {
    if (this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(data);
    }
  }

  close(code, reason) {
    this.ws.close(code, reason);
  }
}
```

```javascript
// - 存放生存时间短的对象
// - 分为From和To两个空间

//    - 清除：清除边界外的对象
```

```javascript
  console.log(`主进程 ${process.pid} 正在运行`);

  // 衍生工作进程
  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }

  cluster.on('exit', (worker, code, signal) => {
    console.log(`工作进程 ${worker.process.pid} 已退出`);
    // 重新启动工作进程
    cluster.fork();
  });
} else {
  // 工作进程可以运行任何Node.js代码
  const http = require('http');
  http.createServer((req, res) => {
    res.end(`响应来自进程 ${process.pid}`);
  }).listen(8000);
}
```

## 五、编译原理

### 5.1 编译过程概述

```javascript
// 编译原理的核心概念

/*
 * 编译过程分为几个阶段：
 *
 * 源代码 → [词法分析] → 词法单元序列 → [语法分析] → 抽象语法树(AST)
 *   AST → [语义分析] → 语义注解的AST → [优化] → 中间表示
 *   中间表示 → [代码生成] → 目标代码
 */

// 词法分析（Lexical Analysis）
// 将源代码字符串转换成词法单元（Token）序列

// 例如：let x = 5 + 3;
// 转换成：
// [
//   { type: 'KEYWORD', value: 'let' },
//   { type: 'IDENTIFIER', value: 'x' },
//   { type: 'OPERATOR', value: '=' },
//   { type: 'NUMBER', value: '5' },
//   { type: 'OPERATOR', value: '+' },
//   { type: 'NUMBER', value: '3' },
//   { type: 'SEMICOLON', value: ';' }
// ]

// 语法分析（Syntactic Analysis）
// 将词法单元序列转换成抽象语法树（AST）

// 例如：5 + 3 * 2
// 生成AST：
//       (+)
//      /   \
//    5      (*)
//        /    \
//      3       2

// 语义分析（Semantic Analysis）
// - 类型检查
// - 作用域分析
// - 含义检查
```

### 5.3 手写简易编译器

```javascript
// 手写一个简易的四则运算编译器

// 第一步：词法分析（Lexer）
function lexer(input) {
  const tokens = [];
  let pos = 0;

  while (pos < input.length) {
    const char = input[pos];

    // 跳过空白
    if (/\s/.test(char)) {
      pos++;
      continue;
    }

    // 数字
    if (/\d/.test(char)) {
      let num = '';
      while (pos < input.length && /\d/.test(input[pos])) {
        num += input[pos++];
      }
      tokens.push({ type: 'NUMBER', value: parseInt(num) });
      continue;
    }

    // 操作符
    if (['+', '-', '*', '/', '(', ')'].includes(char)) {
      tokens.push({ type: 'OPERATOR', value: char });
      pos++;
      continue;
    }

    throw new Error(`未知字符: ${char}`);
  }

  return tokens;
}

// 第二步：语法分析（Parser）- 递归下降法
function parser(tokens) {
  let pos = 0;

  function peek() {
    return tokens[pos];
  }

  function consume() {
    return tokens[pos++];
  }

  // 表达式：项 + 项 | 项 - 项
  function parseExpression() {
    let left = parseTerm();

    while (peek() && (peek().value === '+' || peek().value === '-')) {
      const op = consume().value;
      const right = parseTerm();
      left = { type: 'BinaryExpression', operator: op, left, right };
    }

    return left;
  }

  // 项：因子 * 因子 | 因子 / 因子
  function parseTerm() {
    let left = parseFactor();

    while (peek() && (peek().value === '*' || peek().value === '/')) {
      const op = consume().value;
      const right = parseFactor();
      left = { type: 'BinaryExpression', operator: op, left, right };
    }

    return left;
  }

  // 因子：数字 | (表达式)
  function parseFactor() {
    const token = peek();

    if (token.type === 'NUMBER') {
      consume();
      return { type: 'NumberLiteral', value: token.value };
    }

    if (token.value === '(') {
      consume();
      const expr = parseExpression();
      if (peek().value !== ')') {
        throw new Error('缺少右括号');
      }
      consume();
      return expr;
    }

    throw new Error(`意外的token: ${JSON.stringify(token)}`);
  }

  return parseExpression();
}

// 第三步：求值（Evaluator）
function evaluate(node) {
  switch (node.type) {
    case 'NumberLiteral':
      return node.value;

    case 'BinaryExpression':
      const left = evaluate(node.left);
      const right = evaluate(node.right);

      switch (node.operator) {
        case '+': return left + right;
        case '-': return left - right;
        case '*': return left * right;
        case '/': return left / right;
      }

    default:
      throw new Error(`未知节点类型: ${node.type}`);
  }
}

// 使用示例
const code = '2 + 3 * 4';
const tokens = lexer(code);
console.log('词法单元:', tokens);

const ast = parser(tokens);
console.log('抽象语法树:', JSON.stringify(ast, null, 2));

const result = evaluate(ast);
console.log('计算结果:', result); // 14

// 注意：这个简单编译器不支持运算符优先级
// 2 + 3 * 4 应该解析为 2 + (3 * 4) = 14
// 但我们当前的parser实际上是左结合的
```
