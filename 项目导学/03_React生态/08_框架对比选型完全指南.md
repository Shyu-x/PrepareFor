#### Vue渲染架构

```

   ├── 执行render函数

4. 挂载/更新阶段
   ├── Diff算法比较新旧VNode
   ├── 批量更新DOM
   └── 完成视图更新

关键特点：
- 模板编译可离线进行
- 支持预编译，优化体积
```

**Vue渲染实现代码示例：**

```typescript
// Vue 3 渲染器核心实现
/**
 * 创建渲染器
 * 渲染器是Vue跨平台的核心，通过注入平台特定操作实现不同平台渲染
 */
function createRenderer(options: RendererOptions) {
  // 平台特定操作
  const {
    createElement,
    insert,
    remove,
    patchProp,
    createText,
    createComment
  } = options;

  /**
   * 渲染函数
   * @param vnode 虚拟节点
   * @param container 容器元素
   */
  function render(vnode: VNode, container: Element) {
    if (vnode === null) {
      // 如果vnode为null，卸载组件
      if (container._vnode) {
        unmount(container._vnode);
      }
    } else {
      // 创建或更新组件
      patch(
        container._vnode || null,  // 旧vnode
        vnode,                       // 新vnode
        container
      );
    }
    // 缓存vnode
    container._vnode = vnode;
  }

  /**
   * 打补丁（更新/挂载）
   */
  function patch(
    oldVNode: VNode | null,
    newVNode: VNode,
    container: Element,
    anchor?: Element
  ) {
    // 如果新旧vnode类型不同，直接替换
    if (oldVNode && oldVNode.type !== newVNode.type) {
      unmount(oldVNode);
      oldVNode = null;
    }

    const { type } = newVNode;

    // 根据类型执行不同的patch策略
    if (typeof type === 'string') {
      // 普通HTML元素
      processElement(oldVNode, newVNode, container, anchor);
    } else if (typeof type === 'object' || typeof type === 'function') {
      // 组件
      processComponent(oldVNode, newVNode, container);
    } else if (type === Text) {
      // 文本节点
      processText(oldVNode, newVNode, container);
    } else if (type === Comment) {
      // 注释节点
      processComment(oldVNode, newVNode, container);
    }
  }

  /**
   * 处理普通元素
   */
  function processElement(
    oldVNode: VNode | null,
    newVNode: VNode,
    container: Element,
    anchor?: Element
  ) {
    if (!oldVNode) {
      // 挂载新元素
      mountElement(newVNode, container, anchor);
    } else {
      // 更新已有元素
      updateElement(oldVNode, newVNode);
    }
  }

  /**
   * 挂载元素
   */
  function mountElement(vnode: VNode, container: Element, anchor?: Element) {
    const el = createElement(vnode.type as string);

    // 设置props
    for (const key in vnode.props) {
      if (key.startsWith('on')) {
        // 事件处理
        el.addEventListener(
          key.slice(2).toLowerCase(),
          vnode.props[key]
        );
      } else {
        // 普通属性
        patchProp(el, key, null, vnode.props[key]);
      }
    }

    // 递归挂载子节点
    if (Array.isArray(vnode.children)) {
      vnode.children.forEach(child => {
        patch(null, child, el);
      });
    } else if (typeof vnode.children === 'string') {
      // 文本子节点
      el.textContent = vnode.children;
    }

    // 插入到容器
    insert(el, container, anchor);

    return el;
  }

  /**
   * 更新元素
   */
  function updateElement(oldVNode: VNode, newVNode: VNode) {
    const el = oldVNode.el as Element;

    // 更新props
    for (const key in newVNode.props) {
      const oldValue = oldVNode.props[key];
      const newValue = newVNode.props[key];

      if (oldValue !== newValue) {
        if (key.startsWith('on')) {
          // 事件处理更新
          el.removeEventListener(
            key.slice(2).toLowerCase(),
            oldValue
          );
          el.addEventListener(
            key.slice(2).toLowerCase(),
            newValue
          );
        } else {
          // 普通属性更新
          patchProp(el, key, oldValue, newValue);
        }
      }
    }

    // 更新子节点（简化实现）
    // 实际Vue会进行完整的Diff算法比较
  }

  return { render };
}
```

```typescript
/**
 * React Fiber架构核心概念
 * React 16引入的新渲染架构，解决了旧版本同步阻塞的问题
 */

// Fiber节点结构
interface Fiber {
  // 节点类型
  type: string | Function | null;

  // DOM节点或组件实例
  stateNode: any;

  // 指向子Fiber
  child: Fiber | null;

  // 指向兄弟Fiber
  sibling: Fiber | null;

  // 指向父Fiber
  return: Fiber | null;

  // 更新相关
  pendingProps: any;
  memoizedProps: any;
  memoizedState: any;

  // 副作用标记
  flags: Flags;
  subtreeFlags: Flags;

  // 替代方案（用于快速切换）
  alternate: Fiber | null;

  // 优先级调度
  lanes: Lanes;
  childLanes: Lanes;

  // 引用
  ref: Ref | null;
}

// 副作用类型
const Flags = {
  Placement: 0b000001,        // 插入节点
  Update: 0b000010,           // 更新节点
  Delete: 0b000100,            // 删除节点
  Callback: 0b001000,          // 回调副作用
  DidCapture: 0b010000,       // 错误捕获
  Ref: 0b100000                 // Ref操作
};

/**
 * React工作循环
 * 实现了可中断的渲染机制
 */
function workLoop(continuationRoot: Fiber) {
  // 模拟requestIdleCallback的行为
  let shouldYield = false;

  while (currentUnitOfWork !== null && !shouldYield) {
    // 执行一个fiber的工作
    currentUnitOfWork = performUnitOfWork(currentUnitOfWork);

    // 检查是否需要让出主线程
    // 在真实的React中，这里会检查时间片
    shouldYield = shouldYieldTo主线程();
  }
}

/**
 * 执行一个fiber的工作
 * 返回下一个待处理的fiber
 */
function performUnitOfWork(fiber: Fiber): Fiber | null {
  // 1. 开始处理fiber
  beginWork(fiber);

  // 2. 如果有子fiber，返回子fiber
  if (fiber.child !== null) {
    return fiber.child;
  }

  // 3. 没有子fiber，完成当前fiber
  let nextFiber = fiber;
  while (nextFiber !== null) {
    // 完成当前fiber
    completeUnitOfWork(nextFiber);

    // 如果有兄弟fiber，返回兄弟fiber
    if (nextFiber.sibling !== null) {
      return nextFiber.sibling;
    }

    // 否则返回父fiber
    nextFiber = nextFiber.return;
  }

  return null;
}

/**
 * React调度器
 * 管理任务优先级和执行顺序
 */
class Scheduler {
  // 任务队列
  private taskQueue: PriorityQueue<Task> = new PriorityQueue();

  // 等待中的任务
  private callbackId: number | null = null;

  /**
   * 调度任务
   */
  scheduleCallback(task: Task) {
    this.taskQueue.enqueue(task);

    // 请求调度
    if (this.callbackId === null) {
      this.callbackId = requestCallback(flushWork);
    }
  }

  /**
   * 刷新所有待处理的任务
   */
  function flushWork() {
    // 获取当前时间
    const now = getCurrentTime();

    // 处理所有已过期的任务
    while (true) {
      const task = taskQueue.peek();
      if (task === null) {
        break;
      }

      if (task.expirationTime <= now) {
        taskQueue.dequeue();
        runTask(task);
      } else {
        break;
      }
    }
  }
}
```

### 4.2 响应式实现代码对比

#### Vue 3 Proxy响应式

```typescript
/**
 * Vue 3 Proxy 响应式系统
 * 核心思想：代理对象，拦截操作，自动追踪
 */

// 基础响应式实现
function reactive<T extends object>(target: T): T {
  return new Proxy(target, {
    get(target, key, receiver) {
      // 收集依赖
      track(target, key);

      const result = Reflect.get(target, key, receiver);

      // 如果是对象，递归创建响应式
      if (result !== null && typeof result === 'object') {
        return reactive(result);
      }

      return result;
    },

    set(target, key, value, receiver) {
      const oldValue = target[key as keyof T];

      const result = Reflect.set(target, key, value, receiver);

      // 只有值变化时才触发更新
      if (oldValue !== value) {
        trigger(target, key, oldValue, value);
      }

      return result;
    },

    deleteProperty(target, key) {
      const hadKey = Object.prototype.hasOwnProperty.call(target, key);
      const result = Reflect.deleteProperty(target, key);

      if (hadKey && result) {
        trigger(target, key);
      }

      return result;
    }
  });
}

// 依赖追踪
function track(target: object, key: string | symbol) {
  if (activeEffect) {
    // 将当前effect添加到key的依赖中
    const depsMap = targetMap.get(target);
    if (!depsMap) {
      targetMap.set(target, depsMap = new Map());
    }

    let deps = depsMap.get(key);
    if (!deps) {
      depsMap.set(key, deps = new Set());
    }

    deps.add(activeEffect);
    activeEffect.deps.push(deps);
  }
}

// 派发更新
function trigger(target: object, key: string | symbol, oldValue?: any, newValue?: any) {
  const depsMap = targetMap.get(target);
  if (!depsMap) return;

  const deps = depsMap.get(key);
  if (!deps) return;

  deps.forEach(effect => {
    if (effect.options.scheduler) {
      effect.options.scheduler(effect);
    } else {
      effect();
    }
  });
}
```

```typescript
function useState<T>(initialValue: T): [T, (value: T | ((prev: T) => T)) => void] {
  let state = initialValue;

  function setState(newValue: T | ((prev: T) => T)) {
    // 函数式更新
    if (typeof newValue === 'function') {
      newValue = (newValue as Function)(state);
    }

    // 不可变性检查
    if (!Object.is(newValue, state)) {
      state = newValue;

      // 触发更新

  }
```

```
    └── 建议使用Pinia管理状态

中型团队（5-15人）
├── 推荐：React 或 Vue
├── 理由：
│   ├── 组件化开发利于分工
│   ├── 生态丰富满足各种需求
│   ├── 可建立统一的开发规范
│   └── 技术储备相对充足
└── 注意事项：
    ├── 需要Code Review机制
    ├── 状态管理需统一
    └── 定期技术分享

大型团队（15人以上）
├── 推荐：Angular 或 React + 严格规范
├── 理由：
│   ├── Angular约定俗成，减少分歧
│   ├── TypeScript强类型利于协作
│   ├── 依赖注入利于大型项目
│   ├── 完善的测试体系
│   └── 长期维护性好
└── 注意事项：
    ├── 投入培训成本
    ├── 建立组件库
    └── 完善CI/CD流程
```

## 十五、未来发展趋势

### 15.1 框架发展方向

```
三大框架的演进方向：

Vue 未来方向：
├── Vapor Mode（无虚拟DOM）
│   ├── 编译时优化
│   ├── 接近原生性能
│   └── 预计2025+发布
├── 更强的TypeScript支持
├── 服务端组件完善
└── 移动端增强（Uni-app整合）

React 未来方向：
├── React Server Components完善
├── 异步渲染稳定化
├── Offscreen Canvas集成
├── 更好的开发者工具
└── 持续的性能优化

Angular 未来方向：
├── Signal-based响应式
├── 部分水合（Partial Hydration）
├── 独立组件（Standalone Components）
├── 更小的捆绑体积
└── 更好的SSR/SSG支持
```

### 15.2 新兴框架挑战

```
新兴框架对三大的冲击：

SolidJS
├── 特点：类React语法，编译时优化
├── 性能：比React快1-3倍
├── 现状：社区增长中，生产可用
└── 定位：轻量级React替代

Svelte / SvelteKit
├── 特点：编译时框架，无运行时
├── 性能：包体积最小
├── 现状：社区快速发展
└── 定位：下一代Web框架

Qwik
├── 特点： resumability（恢复执行）
├── 性能：首屏性能极佳
├── 现状：早期阶段
└── 定位：极致首屏性能

Astro
├── 特点：多框架并存，内容优先
├── 性能：默认零JS
├── 现状：静态站点首选
└── 定位：内容驱动网站
```

### 15.3 选型建议总结

```
选型建议总结：

选择 Vue 当：
├── 团队以JavaScript为主
├── 需要快速迭代产品
├── 项目规模中小型
├── 团队成员水平参差
└── 偏好简洁的API设计

选择 React 当：
├── 团队有TypeScript经验
├── 需要最大生态支持
├── 移动端RN是必选项
├── 需要灵活的架构选择
└── 团队偏好函数式编程

选择 Angular 当：
├── 团队有强类型经验
├── 项目是企业级大型应用
├── 需要严格的代码规范
├── 长期维护是主要场景
└── 偏好约定俗成的开发模式
```
