```typescript
// 上报指标
function reportMetric(name: string, metric: any) {
  const data = {
    name,
    value: metric.value,
    id: metric.id,
    delta: metric.delta,
    entries: metric.entries,
    timestamp: Date.now(),
    url: window.location.href,
    userAgent: navigator.userAgent,
    connection: (navigator as any).connection?.effectiveType,
  };

  // 使用sendBeacon确保发送
  if (navigator.sendBeacon) {
    const blob = new Blob([JSON.stringify(data)], { type: 'application/json' });
    navigator.sendBeacon('/api/metrics', blob);
  } else {
    fetch('/api/metrics', {
      method: 'POST',
      body: JSON.stringify(data),
      keepalive: true,
    });
  }
}
```

```typescript
// Performance API详解

// 1. 导航计时
function getNavigationTiming() {
  const timing = performance.getEntriesByType('navigation')[0] as PerformanceNavigationTiming;

  return {
    // DNS查询
    dns: timing.domainLookupEnd - timing.domainLookupStart,

    // TCP连接
    tcp: timing.connectEnd - timing.connectStart,

    // SSL握手
    ssl: timing.secureConnectionStart > 0
      ? timing.connectEnd - timing.secureConnectionStart
      : 0,

    // TTFB
    ttfb: timing.responseStart - timing.requestStart,

    // 内容下载
    download: timing.responseEnd - timing.responseStart,

    // DOM解析
    domParse: timing.domInteractive - timing.responseEnd,

    // DOM内容加载
    domContentLoaded: timing.domContentLoadedEventEnd - timing.fetchStart,

    // 页面完全加载
    load: timing.loadEventEnd - timing.fetchStart,
  };
}

// 2. 资源计时
function getResourceTiming() {
  const resources = performance.getEntriesByType('resource') as PerformanceResourceTiming[];

  return resources.map((resource) => ({
    name: resource.name,
    type: getResourceType(resource.initiatorType),
    duration: resource.duration,
    size: resource.transferSize,
    // 各阶段时间
    dns: resource.domainLookupEnd - resource.domainLookupStart,
    tcp: resource.connectEnd - resource.connectStart,
    ttfb: resource.responseStart - resource.requestStart,
    download: resource.responseEnd - resource.responseStart,
  }));
}

function getResourceType(initiatorType: string): string {
  const typeMap: Record<string, string> = {
    script: 'js',
    link: 'css',
    img: 'image',
    fetch: 'api',
    xmlhttprequest: 'api',
    other: 'other',
  };
  return typeMap[initiatorType] || 'other';
}

// 3. 长任务检测
function observeLongTasks() {
  const observer = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
      const task = entry as PerformanceLongTaskTiming;

      console.log('长任务:', {
        duration: task.duration,
        startTime: task.startTime,
        name: task.name,
        attribution: (task as any).attribution?.map((a: any) => ({
          name: a.name,
          containerType: a.containerType,
          containerName: a.containerName,
        })),
      });
    }
  });

  observer.observe({ entryTypes: ['longtask'] });
}

// 4. 内存使用
function getMemoryUsage() {
  const memory = (performance as any).memory;

  if (!memory) return null;

  return {
    usedJSHeapSize: memory.usedJSHeapSize,
    totalJSHeapSize: memory.totalJSHeapSize,
    jsHeapSizeLimit: memory.jsHeapSizeLimit,
    usage: (memory.usedJSHeapSize / memory.jsHeapSizeLimit * 100).toFixed(2) + '%',
  };
}

// 5. 布局偏移
function observeLayoutShifts() {
  let clsValue = 0;
  let clsEntries: any[] = [];

  const observer = new PerformanceObserver((list) => {
    for (const entry of list.getEntries()) {
      if (!(entry as any).hadRecentInput) {
        clsValue += (entry as any).value;
        clsEntries.push(entry);
      }
    }
  });

  observer.observe({ entryTypes: ['layout-shift'] });

  // 页面卸载时上报
  window.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'hidden') {
      console.log('CLS:', clsValue);
    }
  });
}
```

```typescript
// 3. 自定义性能追踪类
class PerformanceTracker {
  private marks: Map<string, number> = new Map();

  start(name: string) {
    this.marks.set(name, performance.now());
  }

  end(name: string): number {
    const startTime = this.marks.get(name);
    if (!startTime) {
      console.warn(`未找到标记: ${name}`);
      return 0;
    }

    const duration = performance.now() - startTime;
    this.marks.delete(name);

    return duration;
  }

  measure(name: string, callback: () => void): number {
    this.start(name);
    callback();
    return this.end(name);
  }

  async measureAsync(name: string, callback: () => Promise<void>): Promise<number> {
    this.start(name);
    await callback();
    return this.end(name);
  }
}
```

## 4. 性能监控SDK实现

### 4.1 完整SDK实现

```typescript
// 性能监控SDK

interface PerformanceConfig {
  reportUrl: string;
  sampleRate: number;
  whiteScreenTime: number;
  reportInterval: number;
}

class PerformanceMonitor {
  private config: PerformanceConfig;
  private metrics: Map<string, any> = new Map();
  private queue: any[] = [];
  private isInitialized = false;

  constructor(config: Partial<PerformanceConfig> = {}) {
    this.config = {
      reportUrl: '/api/performance',
      sampleRate: 1,
      whiteScreenTime: 3000,
      reportInterval: 5000,
      ...config,
    };
  }

  // 初始化
  init() {
    if (this.isInitialized) return;

    // 采集Web Vitals
    this.collectWebVitals();

    // 采集资源性能
    this.collectResourceTiming();

    // 采集长任务
    this.collectLongTasks();

    // 采集错误
    this.collectErrors();

    // 采集白屏
    this.detectWhiteScreen();

    // 定时上报
    this.startReportTimer();

    // 页面卸载时上报
    this.setupUnloadHandler();

    this.isInitialized = true;
  }

  // 采集Web Vitals
  private collectWebVitals() {
    import('web-vitals').then(({ onLCP, onFID, onCLS, onFCP, onTTFB, onINP }) => {
      onLCP((metric) => this.addMetric('LCP', metric));
      onFID((metric) => this.addMetric('FID', metric));
      onCLS((metric) => this.addMetric('CLS', metric));
      onFCP((metric) => this.addMetric('FCP', metric));
      onTTFB((metric) => this.addMetric('TTFB', metric));
      onINP((metric) => this.addMetric('INP', metric));
    });
  }

  // 采集资源性能
  private collectResourceTiming() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.entryType === 'resource') {
          this.addMetric('resource', {
            name: entry.name,
            type: (entry as any).initiatorType,
            duration: entry.duration,
            size: (entry as any).transferSize,
          });
        }
      }
    });

    observer.observe({ entryTypes: ['resource'] });
  }

  // 采集长任务
  private collectLongTasks() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        this.addMetric('longTask', {
          duration: entry.duration,
          startTime: entry.startTime,
        });
      }
    });

    observer.observe({ entryTypes: ['longtask'] });
  }

  // 采集错误
  private collectErrors() {
    // JS错误
    window.addEventListener('error', (event) => {
      this.addMetric('error', {
        type: 'js',
        message: event.message,
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno,
      });
    });

    // Promise错误
    window.addEventListener('unhandledrejection', (event) => {
      this.addMetric('error', {
        type: 'promise',
        message: event.reason?.message || String(event.reason),
      });
    });

    // 资源错误
    window.addEventListener('error', (event) => {
      if (event.target !== window) {
        this.addMetric('error', {
          type: 'resource',
          tagName: (event.target as any).tagName,
          src: (event.target as any).src || (event.target as any).href,
        });
      }
    }, true);
  }

  // 白屏检测
  private detectWhiteScreen() {
    const startTime = Date.now();

    const checkWhiteScreen = () => {
      const elements = document.elementsFromPoint(
        window.innerWidth / 2,
        window.innerHeight / 2
      );

      // 检查中心点是否有内容
      const hasContent = elements.some((el) => {
        const style = window.getComputedStyle(el);
        return style.backgroundColor !== 'rgba(0, 0, 0, 0)' &&
               style.backgroundColor !== 'transparent';
      });

      if (!hasContent && Date.now() - startTime > this.config.whiteScreenTime) {
        this.addMetric('whiteScreen', {
          duration: Date.now() - startTime,
        });
      }
    };

    // 页面加载后检查
    window.addEventListener('load', () => {
      setTimeout(checkWhiteScreen, this.config.whiteScreenTime);
    });
  }

  // 添加指标
  addMetric(name: string, data: any) {
    // 采样
    if (Math.random() > this.config.sampleRate) return;

    this.queue.push({
      name,
      data,
      timestamp: Date.now(),
      url: window.location.href,
      userAgent: navigator.userAgent,
    });
  }

  // 上报
  private report() {
    if (this.queue.length === 0) return;

    const data = [...this.queue];
    this.queue = [];

    const blob = new Blob([JSON.stringify(data)], { type: 'application/json' });

    if (navigator.sendBeacon) {
      navigator.sendBeacon(this.config.reportUrl, blob);
    } else {
      fetch(this.config.reportUrl, {
        method: 'POST',
        body: blob,
        keepalive: true,
      });
    }
  }

  // 定时上报
  private startReportTimer() {
    setInterval(() => this.report(), this.config.reportInterval);
  }

  // 页面卸载处理
  private setupUnloadHandler() {
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') {
        this.report();
      }
    });
  }
}

// 使用
const monitor = new PerformanceMonitor({
  reportUrl: '/api/performance',
  sampleRate: 0.1, // 10%采样
});

monitor.init();
```

### 4.2 数据上报优化

```typescript
// 数据上报优化

class ReportManager {
  private queue: any[] = [];
  private maxQueueSize = 10;
  private maxRetryCount = 3;
  private retryDelay = 1000;
  private reportUrl: string;

  constructor(reportUrl: string) {
    this.reportUrl = reportUrl;
  }

  // 添加数据
  add(data: any) {
    this.queue.push(data);

    if (this.queue.length >= this.maxQueueSize) {
      this.flush();
    }
  }

  // 立即上报
  async flush() {
    if (this.queue.length === 0) return;

    const data = [...this.queue];
    this.queue = [];

    await this.sendWithRetry(data);
  }

  // 带重试的发送
  private async sendWithRetry(data: any[], retryCount = 0): Promise<boolean> {
    try {
      const response = await fetch(this.reportUrl, {
        method: 'POST',
        body: JSON.stringify(data),
        headers: { 'Content-Type': 'application/json' },
        keepalive: true,
      });

      if (response.ok) {
        return true;
      }

      throw new Error(`HTTP ${response.status}`);
    } catch (error) {
      if (retryCount < this.maxRetryCount) {
        await this.delay(this.retryDelay * Math.pow(2, retryCount));
        return this.sendWithRetry(data, retryCount + 1);
      }

      // 重试失败，存入本地
      this.saveToLocal(data);
      return false;
    }
  }

  // 延迟
  private delay(ms: number): Promise<void> {
    return new Promise((resolve) => setTimeout(resolve, ms));
  }

  // 本地存储
  private saveToLocal(data: any) {
    try {
      const stored = JSON.parse(localStorage.getItem('perf_queue') || '[]');
      stored.push(...data);
      localStorage.setItem('perf_queue', JSON.stringify(stored.slice(-100)));
    } catch (e) {
      // 存储失败
    }
  }

  // 恢复本地数据
  restoreFromLocal() {
    try {
      const stored = JSON.parse(localStorage.getItem('perf_queue') || '[]');
      if (stored.length > 0) {
        this.queue.push(...stored);
        localStorage.removeItem('perf_queue');
      }
    } catch (e) {
      // 恢复失败
    }
  }
}
```

## 5. 性能分析与优化

### 5.1 性能分析工具

```typescript
// 性能分析工具

// 1. 性能评分
function calculatePerformanceScore(metrics: {
  LCP: number;
  FID: number;
  CLS: number;
}): number {
  const scores = {
    LCP: calculateLCPScore(metrics.LCP),
    FID: calculateFIDScore(metrics.FID),
    CLS: calculateCLSScore(metrics.CLS),
  };

  // 加权平均
  return scores.LCP * 0.3 + scores.FID * 0.3 + scores.CLS * 0.4;
}

function calculateLCPScore(lcp: number): number {
  if (lcp <= 2500) return 100;
  if (lcp <= 4000) return 100 - ((lcp - 2500) / 1500) * 50;
  return 50 - Math.min(50, ((lcp - 4000) / 2000) * 50);
}

function calculateFIDScore(fid: number): number {
  if (fid <= 100) return 100;
  if (fid <= 300) return 100 - ((fid - 100) / 200) * 50;
  return 50 - Math.min(50, ((fid - 300) / 200) * 50);
}

function calculateCLSScore(cls: number): number {
  if (cls <= 0.1) return 100;
  if (cls <= 0.25) return 100 - ((cls - 0.1) / 0.15) * 50;
  return 50 - Math.min(50, ((cls - 0.25) / 0.15) * 50);
}

// 2. 性能瓶颈识别
function identifyBottlenecks(timing: PerformanceNavigationTiming): string[] {
  const bottlenecks: string[] = [];

  // DNS查询慢
  const dns = timing.domainLookupEnd - timing.domainLookupStart;
  if (dns > 100) {
    bottlenecks.push(`DNS查询慢: ${dns}ms`);
  }

  // TCP连接慢
  const tcp = timing.connectEnd - timing.connectStart;
  if (tcp > 200) {
    bottlenecks.push(`TCP连接慢: ${tcp}ms`);
  }

  // TTFB慢
  const ttfb = timing.responseStart - timing.requestStart;
  if (ttfb > 600) {
    bottlenecks.push(`TTFB慢: ${ttfb}ms`);
  }

  // 下载慢
  const download = timing.responseEnd - timing.responseStart;
  if (download > 1000) {
    bottlenecks.push(`资源下载慢: ${download}ms`);
  }

  // DOM解析慢
  const domParse = timing.domInteractive - timing.responseEnd;
  if (domParse > 1000) {
    bottlenecks.push(`DOM解析慢: ${domParse}ms`);
  }

  return bottlenecks;
}

// 3. 优化建议生成
function generateOptimizationSuggestions(bottlenecks: string[]): string[] {
  const suggestions: string[] = [];

  bottlenecks.forEach((bottleneck) => {
    if (bottleneck.includes('DNS')) {
      suggestions.push('考虑使用DNS预解析: <link rel="dns-prefetch">');
    }
    if (bottleneck.includes('TCP')) {
      suggestions.push('考虑使用预连接: <link rel="preconnect">');
    }
    if (bottleneck.includes('TTFB')) {
      suggestions.push('优化服务器响应时间，考虑使用CDN');
    }
    if (bottleneck.includes('下载')) {
      suggestions.push('压缩资源、使用CDN、启用Gzip/Brotli');
    }
    if (bottleneck.includes('DOM')) {
      suggestions.push('减少DOM节点数量、延迟加载非关键内容');
    }
  });

  return suggestions;
}
```

## 6. 面试高频问题

### 问题1：什么是Core Web Vitals？

**答案：** Core Web Vitals是Google定义的核心Web性能指标：
1. **LCP**：最大内容绘制，衡量加载性能
2. **FID**：首次输入延迟，衡量交互性
3. **CLS**：累积布局偏移，衡量视觉稳定性

### 问题2：如何采集性能数据？

**答案：**
1. 使用Web Vitals库采集核心指标
2. 使用Performance API采集详细数据
3. 使用PerformanceObserver监听性能事件
4. 自定义性能标记

### 问题3：如何优化LCP？

**答案：**
1. 优化服务器响应时间
2. 使用CDN
3. 优化关键资源加载
4. 预加载关键图片
5. 使用SSR/SSG

### 问题4：如何减少CLS？

**答案：**
1. 为图片设置宽高
2. 预留广告位空间
3. 避免在现有内容上方插入内容
4. 使用transform动画

### 问题5：性能监控如何实现？

**答案：**
1. 数据采集：Web Vitals、Performance API
2. 数据上报：sendBeacon、fetch
3. 数据存储：时序数据库
4. 数据分析：聚合、可视化
5. 告警机制：阈值告警
