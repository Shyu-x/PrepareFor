# 云原生架构：服务网格、链路追踪与可观测性 (2026版)

## 1. 概述：从微服务到服务网格

在 2026 年的云原生架构中，传统的微服务已经进化到了"服务网格"时代。开发者不再需要关心服务间的网络通信、限流、熔断——这些横切关注点被下放到了基础设施层。

### 2.2 Sidecar 代理工作原理

```yaml
# 典型的 Istio 注入配置
apiVersion: v1
kind: Namespace
metadata:
  name: production
  labels:
    istio-injection: enabled  # 自动注入 Sidecar
```

**Sidecar 的职责**：
- **流量拦截**：所有进出 Pod 的流量都经过 Envoy
- **服务发现**：从 Pilot 获取服务注册信息
- **负载均衡**：支持多种均衡算法
- **熔断器**：防止级联故障
- **重试/超时**：声明式配置

### 2.3 流量管理配置

```yaml
# VirtualService：路由规则
apiVersion: networking.istio.io/v1beta1
kind: VirtualService
metadata:
  name: my-service
spec:
  hosts:
    - my-service
  http:
    - match:
        - headers:
            x-user-type:
              exact: premium
      route:
        - destination:
            host: my-service
            subset: v2  # 高级用户走 v2 版本
          weight: 100

    - route:
        - destination:
            host: my-service
            subset: v1  # 普通用户走 v1 版本
          weight: 100

---
# DestinationRule：目的地策略
apiVersion: networking.istio.io/v1beta1
kind: DestinationRule
metadata:
  name: my-service
spec:
  host: my-service
  trafficPolicy:
    connectionPool:
      tcp:
        maxConnections: 100
      http:
        h2UpgradePolicy: UPGRADE
        http1MaxPendingRequests: 100
        http2MaxRequests: 1000
    outlierDetection:
      consecutive5xxErrors: 5
      interval: 30s
      baseEjectionTime: 30s
```

```yaml

  name: backend-service

  host: backend-service
```

### 3.2 OpenTelemetry Python SDK 实战

```python
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.resources import Resource
from opentelemetry.semconv.resource import ResourceAttributes

# 1. 创建资源（服务信息）
resource = Resource.create({
    ResourceAttributes.SERVICE_NAME: "user-service",
    ResourceAttributes.SERVICE_VERSION: "1.0.0",
    ResourceAttributes.DEPLOYMENT_ENVIRONMENT: "production",
})

# 2. 创建追踪器提供者
provider = TracerProvider(resource=resource)

# 3. 配置导出器（发送到 Collector）
processor = BatchSpanProcessor(
    OTLPSpanExporter(
        endpoint="http://otel-collector:4317",
        insecure=True
    )
)
provider.add_span_processor(processor)

# 4. 设置为全局追踪器
trace.set_tracer_provider(provider)

# 5. 获取追踪器
tracer = trace.get_tracer(__name__)

# 6. 创建 Span
def get_user(user_id: int):
    with tracer.start_as_current_span(
        "get_user",
        attributes={
            "user.id": user_id,
            "db.system": "postgresql",
        }
    ) as span:
        # 设置Span 状态
        span.set_status(trace.Status(trace.StatusCode.OK))

        # 添加事件
        span.add_event("Fetching from database")

        try:
            user = db.query(f"SELECT * FROM users WHERE id = {user_id}")
            span.set_attribute("user.found", True)
            return user
        except Exception as e:
            # 记录异常
            span.record_exception(e)
            span.set_status(trace.Status(trace.StatusCode.ERROR, str(e)))
            raise
```

### 3.3 NestJS + OpenTelemetry 集成

```typescript
// trace.service.ts
import { NodeSDK } from '@opentelemetry/sdk-node';
import { getNodeAutoInstrumentations } from '@opentelemetry/auto-instrumentations-node';
import { OTLPTraceExporter } from '@opentelemetry/exporter-trace-otlp-grpc';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes } from '@opentelemetry/semantic-conventions';

const sdk = new NodeSDK({
  resource: new Resource({
    [SemanticResourceAttributes.SERVICE_NAME]: 'api-gateway',
    [SemanticResourceAttributes.SERVICE_VERSION]: '1.0.0',
  }),
  traceExporter: new OTLPTraceExporter({
    url: 'http://otel-collector:4317',
  }),
  instrumentations: [
    getNodeAutoInstrumentations({
      // 忽略某些自动插桩
      '@opentelemetry/instrumentation-fs': { enabled: false },
    }),
  ],
});

sdk.start();
```

### 3.4 链路关联：在微服务间传递 Trace ID

```typescript
// NestJS 拦截器：自动传播 trace context
@Injectable()
export class TracingInterceptor implements NestInterceptor {
  intercept(context: ExecutionContext, next: CallHandler) {
    const request = context.switchToHttp().getRequest();
    const response = context.switchToHttp().getResponse();

    // 从请求头获取 trace context
    const span = trace.getActiveSpan();

    if (span) {
      // 添加 HTTP 属性
      span.setAttribute('http.method', request.method);
      span.setAttribute('http.url', request.url);
      span.setAttribute('http.host', request.headers.host);

      // 将 trace ID 注入响应头，方便前端排查
      response.setHeader('X-Trace-ID', span.spanContext().traceId);
    }

    return next.handle();
  }
}

// 异步传播：使用 W3C Trace Context 标准
// 请求头: traceparent: 00-{traceId}-{spanId}-{flags}
// 这样 Jaeger/Zipkin 可以正确关联跨服务的请求
```

### 4.3 Grafana 日志查询 (LogQL)

```logql
# 1. 基础查询：查看所有错误日志
{service="api-gateway", env="production"} |= "ERROR"

# 2. 复合查询：包含特定字段的错误
{service="user-service"} | json | level="error" | user_id="12345"

# 3. 正则匹配
{service="order-service"} |~ "timeout after \\d+ms"

# 4. 统计每分钟错误数量
sum by (service) (
  rate({env="production"} |= "ERROR" [1m])
)

# 5. 关联 Trace：查找同一 trace 的所有日志
{service=~".+"} | json | trace_id="abc123def456"

# 6. 查找慢请求（响应时间 > 1s）
{service="api-gateway"} | json | duration_ms > 1000

# 7. 告警：连续 5 分钟错误率 > 1%
sum by (service) (
  rate({env="production"} |= "ERROR" [1m])
)
/
sum by (service) (
  rate({env="production"}[1m])
) > 0.01
```

## 7. 2026 年最佳实践

### 7.1 采样策略

```
高流量服务：保留 1% 的请求
中流量服务：保留 10% 的请求
低流量服务：保留 100% 的请求
错误请求：100% 保留（便于排查）
```

```yaml
# OpenTelemetry 采样配置
apiVersion: opentelemetry.io/v1alpha1
kind: Instrumentation
metadata:
  name: my-app
spec:
  sampling:
    type: parentbased_traceidratio
    ratio: 0.1  # 10% 采样率
  propagators:
    - tracecontext  # W3C Trace Context
    - baggage
```
