# 大模型微调与训练实战完全指南

## 一、大模型微调核心概念：站在巨人肩膀上的学习

### 1.1 什么是模型微调？

想象一下：

**预训练模型**就像一个读完医学院的学生——他学习了大量的医学知识，但从来没有实际诊治过病人。

**微调**就像是实习医生的培训——在导师的指导下，把学到的知识应用到具体病症上。

预训练模型已经学会了：
- 语言理解
- 常识推理
- 基础知识

微调则是让模型学会：
- 特定任务的操作方式
- 特定的输出格式
- 领域专业知识

### 1.2 为什么要微调？

| 对比项 | 预训练模型 | 微调模型 |
|--------|-----------|----------|
| **通用能力** | 强 | 保持 |
| **任务专用能力** | 弱 | 强 |
| **部署成本** | 高（全参数） | 低（轻量化） |
| **训练成本** | 极高 | 较低 |
| **依赖Prompt** | 强依赖 | 弱依赖 |

微调的优势：
1. **任务专用性**：微调后的模型更擅长特定任务
2. **降低成本**：比从头训练便宜100-1000倍
3. **减少幻觉**：特定领域的回答更准确
4. **可控输出**：输出格式更稳定

## 二、LoRA：最流行的微调技术

### 2.1 LoRA核心原理

**LoRA**（Low-Rank Adaptation）是一种高效的参数微调技术。

核心思想：
1. 冻结预训练模型的原始权重（不更新）
2. 在旁边添加一对"小型适配器"（低秩矩阵）
3. 训练时只更新适配器的参数

类比：
```
原始模型 = 一本厚厚的百科全书（知识都在这里）

LoRA = 在百科全书旁边放一个小笔记本
- 百科全书不动（冻结）
- 只在小笔记本上做笔记
- 小笔记本记录了"如何把通用知识应用到特定任务"
```

数学原理：
```
原始权重: W (d × d矩阵)
LoRA更新: ΔW = BA (d × r × r × d → d × d)
其中 r << d (低秩)

最终权重: W' = W + α·ΔW
其中 α 是缩放因子
```

### 2.2 LoRA代码实现

```python
# Python示例：LoRA核心原理实现
import torch
import torch.nn as nn
import math

class LoRALinear(nn.Module):
    """
    LoRA线性层：在原始Linear层旁边添加低秩适配器

    冻结原始权重W，只训练A和B两个低秩矩阵

    参数:
        - in_features: 输入维度
        - out_features: 输出维度
        - rank: 低秩维度（通常4-64）
        - alpha: 缩放因子（通常等于rank）
        - dropout: Dropout概率
    """

    def __init__(
        self,
        in_features: int,
        out_features: int,
        rank: int = 4,
        alpha: int = 4,
        dropout: float = 0.1
    ):
        super().__init__()

        # 保存原始权重结构信息
        self.in_features = in_features
        self.out_features = out_features
        self.rank = rank
        self.alpha = alpha
        self.scaling = alpha / rank  # 缩放因子

        # 1. 冻结原始权重 - 这个参数不更新
        self.weight = nn.Parameter(
            torch.randn(out_features, in_features),
            requires_grad=False
        )

        # 2. LoRA的可学习参数
        # A: 随机初始化（使用随机矩阵）
        self.lora_A = nn.Parameter(
            torch.randn(rank, in_features) / math.sqrt(rank)
        )

        # B: 零初始化（使用零矩阵）
        # 零初始化的原因是：训练开始时 ΔW = BA = 0，不影响原始输出
        self.lora_B = nn.Parameter(
            torch.zeros(out_features, rank)
        )

        # Dropout
        self.lora_dropout = nn.Dropout(p=dropout)

        # 标记为可学习参数
        self.lora_A.requires_grad = True
        self.lora_B.requires_grad = True

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        前向传播

        原始路径: x → W → output
        LoRA路径: x → lora_dropout → lora_A → lora_B → scale → 加到原始输出
        """
        # 原始输出
        original_output = nn.functional.linear(x, self.weight, bias=None)

        # LoRA路径
        # x → dropout → A → B → scaled
        lora_output = self.lora_dropout(x) @ self.lora_A.T @ self.lora_B.T
        lora_output = lora_output * self.scaling

        # 两者相加
        return original_output + lora_output

# 演示：对比参数量
def compare_parameters():
    """对比原始层和LoRA层的参数量"""

    # 假设一个768 -> 768的注意力层
    in_features = 768
    out_features = 768
    rank = 16  # LoRA的低秩维度

    # 原始参数数量
    original_params = in_features * out_features  # 589,824

    # LoRA参数数量
    lora_params = (rank * in_features) + (out_features * rank)  # 24,576

    print(f"原始参数量: {original_params:,}")
    print(f"LoRA参数量: {lora_params:,}")
    print(f"参数减少比例: {lora_params/original_params:.2%}")
    print(f"压缩比: {original_params/lora_params:.1f}x")

compare_parameters()
```

```python
# Python示例：使用HuggingFace PEFT库实现LoRA
from transformers import AutoModelForCausalLM, AutoTokenizer
from peft import LoraConfig, get_peft_model, TaskType
import torch

# 1. 加载预训练模型
model_name = "gpt2"  # 以GPT-2为例
model = AutoModelForCausalLM.from_pretrained(model_name)

# 2. 配置LoRA参数
lora_config = LoraConfig(
    # 任务类型：CAUSAL_LM是因果语言模型（GPT类）
    # 还有 SEQ_CLS（文本分类）、SEQ_2_SEQ_LM（翻译类）等
    task_type=TaskType.CAUSAL_LM,

    # 低秩维度 rank：越大效果越好，但参数量也越多
    # 常用值：8, 16, 32, 64
    r=16,

    # 缩放因子 alpha：通常设为rank的2倍
    lora_alpha=32,

    # Dropout概率
    lora_dropout=0.05,

    # 指定要应用LoRA的层
    # "q_proj", "v_proj"是最常用的，因为Attention的Q和V最关键
    target_modules=["q_proj", "v_proj", "k_proj", "o_proj"],

    # 权重矩阵的偏置处理：一般"all"表示所有偏置都不更新
    bias="none",

    # LoRA-A和LoRA-B的初始化方式
    # "gaussian"表示A用高斯，B用零
    inference_mode=False
)

# 3. 将LoRA应用到模型
model = get_peft_model(model, lora_config)

# 打印可训练参数
model.print_trainable_parameters()

# 输出类似：
# trainable params: 1,234,568 || all params: 125,234,568 || trainable%: 0.985%
```

### 2.4 LoRA训练实战

```python
# Python示例：完整的LoRA训练流程
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    TrainingArguments,
    Trainer,
    DataCollatorForLanguageModeling
)
from peft import LoraConfig, get_peft_model
from datasets import load_dataset
import torch

def prepare_training_data():
    """
    准备训练数据

    训练数据格式（每行一条样本）：
    {"text": "用户: 你好\n助手: 你好，有什么可以帮助你的吗？"}
    """
    # 加载数据集
    dataset = load_dataset("json", data_files="training_data.jsonl")

    # 划分训练集和验证集
    dataset = dataset["train"].train_test_split(test_size=0.1)

    # -tokenize
    tokenizer = AutoTokenizer.from_pretrained("gpt2")
    tokenizer.pad_token = tokenizer.eos_token

    def tokenize(examples):
        result = tokenizer(
            examples["text"],
            truncation=True,
            max_length=512,
            padding="max_length"
        )
        # labels用于语言模型训练
        result["labels"] = result["input_ids"].copy()
        return result

    # 应用tokenize
    tokenized_dataset = dataset.map(
        tokenize,
        batched=True,
        remove_columns=dataset["train"].column_names
    )

    return tokenized_dataset, tokenizer

def train_with_lora():
    """
    使用LoRA微调模型
    """
    # 1. 加载模型
    model = AutoModelForCausalLM.from_pretrained(
        "gpt2",
        device_map="auto",
        torch_dtype=torch.float16
    )

    # 2. 配置LoRA
    lora_config = LoraConfig(
        task_type=TaskType.CAUSAL_LM,
        r=16,
        lora_alpha=32,
        lora_dropout=0.1,
        target_modules=["q_proj", "v_proj"]
    )

    # 3. 应用LoRA
    model = get_peft_model(model, lora_config)

    # 4. 准备数据
    tokenized_dataset, tokenizer = prepare_training_data()

    # 5. 配置训练参数
    training_args = TrainingArguments(
        # 输出目录
        output_dir="./lora_output",

        # 训练轮数
        num_train_epochs=3,

        # Batch size
        per_device_train_batch_size=4,
        per_device_eval_batch_size=4,

        # 优化器
        learning_rate=3e-4,
        warmup_steps=100,

        # 日志和保存
        logging_steps=10,
        save_steps=100,
        save_total_limit=3,

        # 评估
        eval_strategy="steps",
        eval_steps=100,

        # 其他
        fp16=True,  # 混合精度
        gradient_accumulation_steps=2,  # 梯度累积
        max_grad_norm=1.0,  # 梯度裁剪
    )

    # 6. 创建Trainer
    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=tokenized_dataset["train"],
        eval_dataset=tokenized_dataset["test"],
        data_collator=DataCollatorForLanguageModeling(
            tokenizer=tokenizer,
            mlm=False  # 因果语言模型不用MLM
        )
    )

    # 7. 开始训练
    trainer.train()

    # 8. 保存模型
    model.save_pretrained("./lora_finetuned_model")
```

### 2.5 LoRA模型推理

```python
# Python示例：加载和使用LoRA微调的模型
from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig
from peft import PeftModel
import torch

def load_lora_model():
    """
    加载LoRA微调后的模型进行推理
    """
    # 1. 加载基础模型（量化版本，节省显存）
    bnb_config = BitsAndBytesConfig(
        load_in_4bit=True,           # 4-bit量化
        bnb_4bit_quant_type="nf4",
        bnb_4bit_compute_dtype=torch.float16
    )

    base_model = AutoModelForCausalLM.from_pretrained(
        "gpt2",
        quantization_config=bnb_config,
        device_map="auto",
        trust_remote_code=True
    )

    # 2. 加载LoRA权重
    model = PeftModel.from_pretrained(
        base_model,
        "./lora_finetuned_model"  # LoRA权重目录
    )

    # 3. 加载Tokenizer
    tokenizer = AutoTokenizer.from_pretrained("gpt2")
    tokenizer.pad_token = tokenizer.eos_token

    return model, tokenizer

def generate_with_lora(model, tokenizer, prompt: str, max_length: int = 100):
    """
    使用LoRA模型生成文本

    参数:
        prompt: 输入提示词
        max_length: 最大生成长度
    """
    # 构建输入
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)

    # 生成
    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_new_tokens=max_length,
            temperature=0.7,       # 控制随机性
            top_p=0.9,            # Nucleus采样
            do_sample=True,       # 采样生成
            repetition_penalty=1.2  # 重复惩罚
        )

    # 解码
    generated_text = tokenizer.decode(outputs[0], skip_special_tokens=True)

    return generated_text

# 使用示例
if __name__ == "__main__":
    model, tokenizer = load_lora_model()

    prompt = "用户: 你好\n助手:"
    response = generate_with_lora(model, tokenizer, prompt)
    print(response)
```

## 三、QLoRA：消费级GPU也能微调大模型

### 3.1 QLoRA原理

**QLoRA**（Quantized LoRA）是LoRA的进一步优化，结合了4-bit量化和LoRA。

核心创新：
1. **NF4量化**：一种更适合神经网络权重分布的4-bit数据类型
2. **双重量化**：对量化常数也进行量化，进一步节省显存
3. **分页优化器**：处理梯度更新时GPU显存溢出

显存对比（以65B参数模型为例）：
| 方法 | 显存需求 |
|------|----------|
| 全量FP16 | ~130GB |
| LoRA (16bit) | ~80GB |
| QLoRA (4bit) | ~48GB |

**结论**：QLoRA可以在48GB显存的显卡上微调65B参数的模型！

### 3.2 QLoRA代码实现

```python
# Python示例：QLoRA微调
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    BitsAndBytesConfig,
    TrainingArguments,
    Trainer
)
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training
import torch

def load_model_for_qlora(model_name: str):
    """
    加载模型用于QLoRA训练

    参数:
        model_name: 模型名称或路径
    """
    # 1. 4-bit量化配置
    bnb_config = BitsAndBytesConfig(
        # 加载为4-bit
        load_in_4bit=True,

        # 4-bit量化类型：NF4（Normal Float 4）
        # 这是QLoRA论文推荐的量化类型
        bnb_4bit_quant_type="nf4",

        # 计算时使用的数据类型
        bnb_4bit_compute_dtype=torch.float16,

        # 是否对量化常数进行量化（进一步省显存）
        bnb_4bit_use_double_quant=True
    )

    # 2. 加载模型
    model = AutoModelForCausalLM.from_pretrained(
        model_name,
        quantization_config=bnb_config,
        # 使用GPU
        device_map="auto",
        # 信任远程代码（有些模型需要）
        trust_remote_code=True
    )

    # 3. 准备模型用于k-bit训练
    # 这个函数会：
    # - 将模型转换为训练模式
    # - 配置梯度计算
    # - 设置一些必要的标志
    model = prepare_model_for_kbit_training(model)

    return model

def create_qlora_config():
    """
    创建QLoRA配置
    """

    # 主要区别在于底座模型是量化的
    lora_config = LoraConfig(
        r=64,                    # QLoRA推荐使用更大的r
        lora_alpha=16,           # alpha是缩放因子
        target_modules=[
            "q_proj", "v_proj",
            "k_proj", "o_proj",
            "gate_proj", "up_proj", "down_proj"  # LLaMA需要这些
        ],
        bias="none",
        task_type=TaskType.CAUSAL_LM,
        # QLoRA推荐使用更高的dropout
        lora_dropout=0.1
    )

    return lora_config

def train_qlora():
    """
    QLoRA训练流程
    """
    # 1. 加载量化模型
    model = load_model_for_qlora("meta-llama/Llama-2-7b-hf")

    # 2. 应用LoRA
    lora_config = create_qlora_config()
    model = get_peft_model(model, lora_config)

    # 3. 打印可训练参数
    model.print_trainable_parameters()
    # 输出类似：trainable params: 167,772,160 || all params: 6,738,415,616 || trainable%: 2.490%

    # 4. 训练配置
    training_args = TrainingArguments(
        output_dir="./qlora_output",
        num_train_epochs=3,
        per_device_train_batch_size=8,
        gradient_accumulation_steps=2,  # 梯度累积
        learning_rate=2e-4,
        warmup_steps=100,
        max_steps=1000,
        logging_steps=10,
        save_steps=100,
        fp16=True,
        optim="paged_adamw_8bit",  # QLoRA推荐的分页优化器
```

## 四、RLHF：让AI更符合人类偏好

### 4.1 RLHF是什么？

**RLHF**（Reinforcement Learning from Human Feedback，人类反馈强化学习）是一种让AI符合人类偏好的技术。

核心问题：
- 如何让AI的回答"好"而不是"对"？
- "好"是一个主观判断，AI如何学习？

RLHF的解决方案：
1. 让人类对AI的不同回答进行排序
2. 训练一个"奖励模型"（Reward Model）学习人类的偏好
3. 用强化学习让AI的输出获得更高的奖励

### 4.3 奖励模型训练

```python
# Python示例：奖励模型训练
import torch
import torch.nn as nn
from transformers import AutoModel, AutoTokenizer

class RewardModel(nn.Module):
    """
    奖励模型：学习人类偏好

    架构：基于语言模型，输出一个标量分数
    分数越高，表示人类越偏好这个回答
    """

    def __init__(self, base_model_name: str):
        super().__init__()

        # 使用预训练模型作为基础
        self.base_model = AutoModel.from_pretrained(base_model_name)
        self.base_model.config.pad_token_id = 0  # 假设pad token id为0

        # 奖励头：输出一个标量分数
        # 取最后一层隐藏状态的均值作为输入
        self.reward_head = nn.Linear(
            self.base_model.config.hidden_size,
            1,
            bias=False
        )

        # 初始化：奖励头应该初始为接近0
        # 这样训练初期，所有回答的分数都差不多
        nn.init.zeros_(self.reward_head.weight)

    def forward(
        self,
        input_ids: torch.Tensor,
        attention_mask: torch.Tensor
    ) -> torch.Tensor:
        """
        计算奖励分数

        参数:
            input_ids: 输入token的ID
            attention_mask: 注意力掩码

        返回:
            奖励分数 (batch_size,)
        """
        # 获取模型输出
        outputs = self.base_model(
            input_ids=input_ids,
            attention_mask=attention_mask
        )

        # 取最后一层隐藏状态
        last_hidden_state = outputs.last_hidden_state

        # 平均池化 + 奖励头
        # masked_hidden_state：对非pad位置加权平均
        mask_expanded = attention_mask.unsqueeze(-1).expand(last_hidden_state.size()).float()
        sum_hidden = torch.sum(last_hidden_state * mask_expanded, dim=1)
        sum_mask = torch.clamp(mask_expanded.sum(dim=1), min=1e-9)
        pooled_output = sum_hidden / sum_mask

        # 计算奖励
        rewards = self.reward_head(pooled_output)

        return rewards.squeeze(-1)


def train_reward_model():
    """
    训练奖励模型
    """
    # 1. 准备数据
    # 数据格式：
    # [
    #   {
    #     "prompt": "问题：如何学习Python？",
    #     "chosen": "回答A（人类偏好）",
    #     "rejected": "回答B（人类不偏好）"
    #   },
    #   ...
    # ]

    # 2. 创建模型
    reward_model = RewardModel("gpt2")

    # 3. 定义损失函数
    # 奖励模型的损失函数：对比排序损失
    # 目标：让chosen回答的分数高于rejected回答

    def reward_model_loss(chosen_rewards, rejected_rewards):
        """
        奖励模型损失

        使用对比损失：chosen的分数应该比rejected高
        """
        # 计算差值
        # 差值越大，损失越小（正确分类）
        # 差值越小，损失越大（错误分类）
        margin = 1.0  # 期望的安全边际
        diff = chosen_rewards - rejected_rewards

        # 使用hinge loss
        loss = torch.clamp(margin - diff, min=0).mean()

        return loss

    # 4. 训练循环
    optimizer = torch.optim.AdamW(reward_model.parameters(), lr=1e-5)

    for epoch in range(num_epochs):
        for batch in dataloader:
            chosen_ids = batch["chosen_ids"]
            chosen_mask = batch["chosen_mask"]
            rejected_ids = batch["rejected_ids"]
            rejected_mask = batch["rejected_mask"]

            # 前向传播
            chosen_rewards = reward_model(chosen_ids, chosen_mask)
            rejected_rewards = reward_model(rejected_ids, rejected_mask)

            # 计算损失
            loss = reward_model_loss(chosen_rewards, rejected_rewards)

            # 反向传播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
```

### 4.4 PPO强化学习微调

```python
# Python示例：PPO强化学习更新
from trl import PPOTrainer, PPOConfig
from transformers import AutoModelForCausalLM

class RLHFOptimizer:
    """
    使用PPO进行强化学习微调
    """

    def __init__(self, sft_model_path: str, reward_model: RewardModel):
        """
        初始化RLHF优化器

        参数:
            sft_model_path: SFT模型路径
            reward_model: 训练好的奖励模型
        """
        # 加载SFT模型（作为策略模型）
        self.policy_model = AutoModelForCausalLM.from_pretrained(sft_model_path)
        self.ref_model = AutoModelForCausalLM.from_pretrained(sft_model_path)

        self.reward_model = reward_model

        # PPO配置
        ppo_config = PPOConfig(
            learning_rate=1.4e-5,
            batch_size=1,
            steps=20,
            ppo_epochs=4,
            target_kl=0.1  # KL散度约束，防止策略变化太大
        )

        self.ppo_trainer = PPOTrainer(
            config=ppo_config,
            model=self.policy_model,
            ref_model=self.ref_model,
            tokenizer=tokenizer
        )

    def generate_response(self, prompts: list) -> list:
        """
        使用当前策略模型生成回答
        """
        inputs = tokenizer(
            prompts,
            return_tensors="pt",
            padding=True,
            truncation=True
        ).to(self.policy_model.device)

        with torch.no_grad():
            outputs = self.policy_model.generate(
                **inputs,
                max_new_tokens=100,
                do_sample=True,
                temperature=0.7
            )

        responses = tokenizer.batch_decode(outputs, skip_special_tokens=True)
        return responses

    def compute_rewards(self, prompts: list, responses: list) -> list:
        """
        使用奖励模型计算奖励
        """
        rewards = []

        for prompt, response in zip(prompts, responses):
            # 拼接prompt和response
            full_text = prompt + response

            # Tokenize
            inputs = tokenizer(
                full_text,
                return_tensors="pt",
                truncation=True,
                max_length=512
            ).to(self.policy_model.device)

            # 获取奖励分数
            with torch.no_grad():
                reward = self.reward_model(
                    inputs["input_ids"],
                    inputs["attention_mask"]
                )

            rewards.append(reward.cpu().item())

        return rewards

    def train_step(self, prompts: list):
        """
        执行一步PPO训练
        """
        # 1. 生成回答
        responses = self.generate_response(prompts)

        # 2. 计算奖励
        rewards = self.compute_rewards(prompts, responses)

        # 3. PPO更新
        # trl库的PPOTrainer会自动处理：
        # - 计算优势估计
        # - 裁剪策略更新
        # - KL散度惩罚
        stats = self.ppo_trainer.step(prompts, responses, rewards)

        return stats
```

```python
# Python示例：计算困惑度
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
from datasets import load_dataset

def calculate_perplexity(model, tokenizer, dataset, max_length: int = 512):
    """
    计算模型的困惑度（Perplexity）

    困惑度是语言模型的核心评估指标：
    - 衡量模型对文本的"预测能力"
    - PPL = exp(loss)，越低越好
    - 理想情况下，PPL = 1（完美预测）

    参数:
        model: 要评估的模型
        tokenizer: 分词器
        dataset: 评估数据集
        max_length: 最大输入长度

    返回:
        平均困惑度
    """
    model.eval()
    total_loss = 0
    total_tokens = 0

    device = next(model.parameters()).device

    with torch.no_grad():
        for text in dataset:
            # Tokenize
            encodings = tokenizer(
                text,
                return_tensors="pt",
                truncation=True,
                max_length=max_length
            )

            input_ids = encodings["input_ids"].to(device)

            # 计算Loss
            # shift_labels用于因果语言模型：预测下一个token
            outputs = model(input_ids, labels=input_ids)
            loss = outputs.loss

            # 统计
            num_tokens = (input_ids != tokenizer.pad_token_id).sum().item()
            total_loss += loss.item() * num_tokens
            total_tokens += num_tokens

    # 计算平均Loss和困惑度
    avg_loss = total_loss / total_tokens
    perplexity = math.exp(avg_loss)

    return perplexity
```

### 5.3 任务特定评估

```python
# Python示例：问答任务评估
from typing import List, Dict
import re

class QAEvaluator:
    """
    问答系统评估器
    """

    def exact_match_score(self, prediction: str, ground_truth: str) -> float:
        """
        精确匹配（EM）

        预测和标准答案完全一致得1分，否则0分
        """
        return 1.0 if prediction.strip() == ground_truth.strip() else 0.0

    def f1_score(self, prediction: str, ground_truth: str) -> float:
        """
        F1分数（按词计算）

        计算预测和标准答案之间的F1分数
        """
        pred_tokens = prediction.lower().split()
        truth_tokens = ground_truth.lower().split()

        common = set(pred_tokens) & set(truth_tokens)
        num_same = len(common)

        if num_same == 0:
            return 0.0

        precision = num_same / len(pred_tokens)
        recall = num_same / len(truth_tokens)

        f1 = 2 * precision * recall / (precision + recall)
        return f1

    def normalize_answer(self, text: str) -> str:
        """
        标准化答案：去除文章、停顿词等
        """
        # 转小写
        text = text.lower()
        # 去除标点
        text = re.sub(r'[^\w\s]', '', text)
        # 去除冠词
        text = ' '.join([w for w in text.split() if w not in ['a', 'an', 'the']])
        return text.strip()

    def evaluate(
        self,
        predictions: List[str],
        ground_truths: List[str]
    ) -> Dict[str, float]:
        """
        批量评估

        参数:
            predictions: 模型预测列表
            ground_truths: 标准答案列表

        返回:
            评估指标
        """
        em_scores = []
        f1_scores = []

        for pred, truth in zip(predictions, ground_truths):
            # 标准化
            norm_pred = self.normalize_answer(pred)
            norm_truth = self.normalize_answer(truth)

            # 计算分数
            em = self.exact_match_score(norm_pred, norm_truth)
            f1 = self.f1_score(norm_pred, norm_truth)

            em_scores.append(em)
            f1_scores.append(f1)

        return {
            "exact_match": sum(em_scores) / len(em_scores) * 100,
            "f1": sum(f1_scores) / len(f1_scores) * 100
        }


def demo_qa_evaluation():
    """演示问答评估"""

        "深度学习是机器学习的一个分支"
    ]

    ground_truths = [
        "北京",
        "Python编程语言",
        "机器学习"
    ]

    results = evaluator.evaluate(predictions, ground_truths)

    print(f"Exact Match: {results['exact_match']:.1f}%")
    print(f"F1 Score: {results['f1']:.1f}%")
```

### 7.3 完整训练脚本

```python
# Python示例：完整的QLoRA微调脚本
import os
import torch
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    TrainingArguments,
    Trainer,
    DataCollatorForLanguageModeling,
    BitsAndBytesConfig
)
from peft import LoraConfig, get_peft_model, TaskType
from datasets import load_dataset

# 1. 量化配置
def get_bnb_config():
    """获取4-bit量化配置"""
    return BitsAndBytesConfig(
        load_in_4bit=True,
        bnb_4bit_quant_type="nf4",
        bnb_4bit_compute_dtype=torch.float16,
        bnb_4bit_use_double_quant=True
    )

# 2. LoRA配置
def get_lora_config():
    """获取LoRA配置"""
    return LoraConfig(
        r=64,                           # 低秩维度
        lora_alpha=16,                  # 缩放因子
        target_modules=[                # 目标模块
            "q_proj", "k_proj", "v_proj", "o_proj",
            "gate_proj", "up_proj", "down_proj"
        ],
        lora_dropout=0.05,
        bias="none",
        task_type=TaskType.CAUSAL_LM
    )

# 3. 加载模型
def load_model_and_tokenizer(model_path: str):
    """加载模型和分词器"""
    # 量化配置
    bnb_config = get_bnb_config()

    # 加载模型
    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        quantization_config=bnb_config,
        device_map="auto",
        trust_remote_code=True
    )

    # 加载分词器
    tokenizer = AutoTokenizer.from_pretrained(
        model_path,
        trust_remote_code=True
    )

    # ChatGLM3需要特殊配置
    tokenizer.pad_token = tokenizer.eos_token

    return model, tokenizer

# 4. 数据处理
def tokenize_function(examples, tokenizer, max_length: int = 512):
    """Tokenize数据"""
    # 构造输入输出
    texts = []
    for instruction, input_text, output in zip(
        examples["instruction"],
        examples["input"],
        examples["output"]
    ):
        text = f"{instruction}\n\n输入：{input_text}\n\n输出：{output}"
        texts.append(text)

    # Tokenize
    result = tokenizer(
        texts,
        truncation=True,
        max_length=max_length,
        padding=False
    )

    # 设置labels
    result["labels"] = result["input_ids"].copy()

    return result

# 5. 主训练函数
def train():
    """主训练流程"""

    # 配置
    model_path = "THUDM/chatglm3-6b"  # 或本地路径
    output_dir = "./medical_chatbot"

    # 加载模型
    print("加载模型...")
    model, tokenizer = load_model_and_tokenizer(model_path)

    # 应用LoRA
    print("应用LoRA...")
    lora_config = get_lora_config()
    model = get_peft_model(model, lora_config)

    # 打印可训练参数
    model.print_trainable_parameters()

    # 准备数据
    print("准备数据...")
    dataset = prepare_medical_qa_data()

    # Tokenize
    tokenized_dataset = dataset.map(
        lambda x: tokenize_function(x, tokenizer),
        batched=True,
        remove_columns=dataset.column_names
    )

    # 划分训练集
    tokenized_dataset = tokenized_dataset.train_test_split(test_size=0.1)

    # 数据整理器
    data_collator = DataCollatorForLanguageModeling(
        tokenizer=tokenizer,
        mlm=False  # 因果LM不用MLM
    )

    # 训练参数
    training_args = TrainingArguments(
        output_dir=output_dir,
        num_train_epochs=3,
        per_device_train_batch_size=2,
        gradient_accumulation_steps=4,  # 有效batch_size=8
        learning_rate=2e-4,
        warmup_ratio=0.1,
        lr_scheduler_type="cosine",
        logging_steps=10,
        save_steps=100,
        save_total_limit=3,
        fp16=True,
        optim="paged_adamw_8bit",
        max_grad_norm=1.0,
        report_to="tensorboard"
    )

    # 创建Trainer
    trainer = Trainer(
        model=model,
        args=training_args,
        train_dataset=tokenized_dataset["train"],
        eval_dataset=tokenized_dataset["test"],
        data_collator=data_collator
    )

    # 开始训练
    print("开始训练...")
    trainer.train()

    # 保存模型
    print("保存模型...")
    trainer.save_model(output_dir)
    tokenizer.save_pretrained(output_dir)

# 6. 推理函数
def inference():
    """使用微调后的模型推理"""
    from peft import PeftModel
    from transformers import AutoModelForCausalLM, AutoTokenizer

    # 加载基础模型
    base_model = AutoModelForCausalLM.from_pretrained(
        "THUDM/chatglm3-6b",
        device_map="auto",
        trust_remote_code=True
    )

    # 加载LoRA权重
    model = PeftModel.from_pretrained(
        base_model,
        "./medical_chatbot"
    )

    tokenizer = AutoTokenizer.from_pretrained(
        "THUDM/chatglm3-6b",
        trust_remote_code=True
    )

    # 推理
    prompt = """指示：你是一个专业的医疗健康助手，请回答用户的问题。

输入：最近总是头痛，是什么原因？

输出："""

    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)

    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_new_tokens=200,
            temperature=0.7,
            do_sample=True
        )

    response = tokenizer.decode(outputs[0], skip_special_tokens=True)
    print(response)

if __name__ == "__main__":
```

```
│  ├── Adapter: 插入适配层                                       │

│  模型评估                                                       │
```
