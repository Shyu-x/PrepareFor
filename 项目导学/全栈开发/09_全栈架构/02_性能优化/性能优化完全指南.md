## 1. 性能优化基础

### 1.1 性能指标

```
Web性能指标（Web Vitals）

核心指标（Core Web Vitals）
├── LCP（Largest Contentful Paint）- 最大内容绘制
│   └── 目标: < 2.5秒
├── FID（First Input Delay）- 首次输入延迟
│   └── 目标: < 100毫秒
└── CLS（Cumulative Layout Shift）- 累积布局偏移

其他重要指标
├── FCP（First Contentful Paint）- 首次内容绘制
├── TTI（Time to Interactive）- 可交互时间
├── TBT（Total Blocking Time）- 总阻塞时间
└── SI（Speed Index）- 速度指数
```

```javascript
// 防抖（Debounce）
function debounce(func, wait) {
  let timeout;
  return function (...args) {
    clearTimeout(timeout);
    timeout = setTimeout(() => func.apply(this, args), wait);
  };
}

// 节流（Throttle）
function throttle(func, limit) {
  let inThrottle;
  return function (...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => (inThrottle = false), limit);
    }
  };
}

// 使用

  // 处理滚动
}, 100);

  // 处理调整大小
```

```javascript
// 2. 使用对象池复用对象
class ObjectPool {
  constructor(createFn, resetFn, initialSize = 10) {
    this.createFn = createFn;
    this.resetFn = resetFn;
    this.pool = [];

    for (let i = 0; i < initialSize; i++) {
      this.pool.push(this.createFn());
    }
  }

  acquire() {
    return this.pool.pop() || this.createFn();
  }

  release(obj) {
    this.resetFn(obj);
    this.pool.push(obj);
  }
}
```

## 5. 数据库性能优化

### 5.1 查询优化

```sql
-- 1. 使用索引
-- 创建索引
CREATE INDEX idx_user_email ON users(email);

-- 复合索引
CREATE INDEX idx_user_name_email ON users(name, email);

-- 唯一索引
CREATE UNIQUE INDEX idx_user_email_unique ON users(email);

-- 2. 避免SELECT *
-- ❌ 不好的做法
SELECT * FROM users WHERE id = 1;

-- ✅ 好的做法
SELECT id, name, email FROM users WHERE id = 1;

-- 3. 使用LIMIT分页
-- ❌ 不好的做法：性能差
SELECT * FROM posts ORDER BY created_at OFFSET 100000 LIMIT 10;

-- ✅ 好的做法：使用游标分页
SELECT * FROM posts
WHERE id > last_seen_id
ORDER BY id
LIMIT 10;

-- 4. 避免子查询
-- ❌ 不好的做法
SELECT * FROM users
WHERE id IN (SELECT user_id FROM posts WHERE created_at > '2024-01-01');

-- ✅ 好的做法：使用JOIN
SELECT DISTINCT users.*
FROM users
JOIN posts ON users.id = posts.user_id
WHERE posts.created_at > '2024-01-01';

-- 5. 使用EXISTS代替IN
-- ❌ 不好的做法
SELECT * FROM orders
WHERE user_id IN (SELECT id FROM users WHERE status = 'active');

-- ✅ 好的做法
SELECT o.*
FROM orders o
WHERE EXISTS (
  SELECT 1 FROM users u
  WHERE u.id = o.user_id AND u.status = 'active'
);

-- 6. 使用UNION ALL代替UNION
-- ❌ 不好的做法：UNION会去重，性能差
SELECT name FROM users
UNION
SELECT name FROM admins;

-- ✅ 好的做法：UNION ALL不去重，性能好
SELECT name FROM users
UNION ALL
SELECT name FROM admins;
```

```javascript
// 1. Redis缓存
const redis = require('redis');
const client = redis.createClient();

async function getCachedData(key) {
  // 先从缓存获取
  const cached = await client.get(key);
  if (cached) {
    return JSON.parse(cached);
  }

  // 缓存未命中，从数据库获取
  const data = await fetchDataFromDB(key);

  // 写入缓存
  await client.setex(key, 3600, JSON.stringify(data));  // 1小时过期

  return data;
}

// 2. 内存缓存
const NodeCache = require('node-cache');
const cache = new NodeCache({ stdTTL: 3600 });

function getFromCache(key) {
  const cached = cache.get(key);
  if (cached) {
    return cached;
  }

  const data = fetchData(key);
  cache.set(key, data);

  return data;
}

// 3. 缓存穿透
// 使用布隆过滤器
const BloomFilter = require('bloom-filters');
const filter = BloomFilter.bloomFilter(1000, 0.01);

async function getDataWithFilter(key) {
  // 检查布隆过滤器
  if (!filter.has(key)) {
    return null;  // 肯定不存在
  }

  // 从缓存获取
  const cached = await cache.get(key);
  if (cached) {
    return cached;
  }

  // 从数据库获取
  const data = await db.get(key);
  if (data) {
    filter.add(key);  // 添加到布隆过滤器
    await cache.set(key, data);
  }

  return data;
}

// 4. 缓存击穿
// 使用互斥锁
async function getDataWithLock(key) {
  const cached = await cache.get(key);
  if (cached) {
    return cached;
  }

  // 尝试获取锁
  const lockKey = `lock:${key}`;
  const lockAcquired = await cache.set(lockKey, '1', 'PX', 10000, 'NX');

  if (lockAcquired) {
    try {
      // 获取数据
      const data = await db.get(key);
      await cache.set(key, data, 'EX', 3600);
      return data;
    } finally {
      // 释放锁
      await cache.del(lockKey);
    }
  } else {
    // 等待并重试
    await sleep(100);
    return getDataWithLock(key);
  }
}

// 5. 缓存雪崩
// 使用随机过期时间
async function setDataWithRandomTTL(key, data) {
  const baseTTL = 3600;
  const randomTTL = baseTTL + Math.random() * 600;  // 3600-4200秒
  await cache.set(key, data, 'EX', randomTTL);
}
```

```javascript
const wss = new WebSocket.Server({ port: 8080 });

wss.on('connection', (ws) => {
  let heartbeatInterval;

  const heartbeat = () => {
    ws.isAlive = true;
  };

  ws.on('pong', heartbeat);

  ws.on('open', () => {
    ws.isAlive = true;
    heartbeatInterval = setInterval(() => {
      if (ws.isAlive === false) {
        return ws.terminate();
      }
      ws.isAlive = false;
      ws.ping();
    }, 30000);
  });

  ws.on('close', () => {
    clearInterval(heartbeatInterval);
  });
});
```
