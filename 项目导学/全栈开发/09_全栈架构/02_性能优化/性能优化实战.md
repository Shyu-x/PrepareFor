```typescript
// Service Worker 缓存策略

// 1. 缓存优先（Cache First）
// 适用于：静态资源（JS、CSS、字体、图片）
self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => {
      return response || fetch(event.request).then((fetchResponse) => {
        return caches.open('static-v1').then((cache) => {
          cache.put(event.request, fetchResponse.clone());
          return fetchResponse;
        });
      });
    })
  );
});

// 2. 网络优先（Network First）
// 适用于：API 请求、动态内容
self.addEventListener('fetch', (event) => {
  event.respondWith(
    fetch(event.request)
      .then((response) => {
        const responseClone = response.clone();
        caches.open('api-v1').then((cache) => {
          cache.put(event.request, responseClone);
        });
        return response;
      })
      .catch(() => caches.match(event.request))
  );
});

// 3. 网络优先，缓存回退，后台更新（Stale While Revalidate）
// 适用于：需要更新但可接受旧数据的内容
self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((cachedResponse) => {
      const fetchPromise = fetch(event.request).then((networkResponse) => {
        caches.open('dynamic-v1').then((cache) => {
          cache.put(event.request, networkResponse.clone());
        });
        return networkResponse;
      });
      
      return cachedResponse || fetchPromise;
    })
  );
});
```

### 4.3 数据请求优化

```typescript
// 1. 请求去重
class RequestDeduper {
  private pending = new Map<string, Promise<any>>();
  
  async fetch<T>(key: string, fetcher: () => Promise<T>): Promise<T> {
    // 如果已有相同请求在进行中，返回该 Promise
    if (this.pending.has(key)) {
      return this.pending.get(key)!;
    }
    
    // 发起新请求
    const promise = fetcher().finally(() => {
      this.pending.delete(key);
    });
    
    this.pending.set(key, promise);
    return promise;
  }
}

const deduper = new RequestDeduper();

// 使用
const user = await deduper.fetch(`user:${id}`, () => fetchUser(id));

// 2. 请求取消
function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null);
  const [error, setError] = useState<Error | null>(null);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    const controller = new AbortController();
    
    async function fetchData() {
      try {
        setLoading(true);
        const response = await fetch(url, {
          signal: controller.signal,
        });
        const json = await response.json();
        setData(json);
      } catch (err) {
        if (err instanceof Error && err.name !== 'AbortError') {
          setError(err);
        }
      } finally {
        setLoading(false);
      }
    }
    
    fetchData();
    
    return () => controller.abort();
  }, [url]);
  
  return { data, error, loading };
}

// 3. 数据预取
function usePrefetch() {

function useInfiniteUsers() {
  const {
    data,
    fetchNextPage,
    hasNextPage,
    isFetchingNextPage,
  } = useInfiniteQuery({
    queryKey: ['users'],
    queryFn: ({ pageParam = 1 }) => fetchUsers(pageParam),

      lastPage.hasMore ? pages.length + 1 : undefined,
  });
  
  const users = data?.pages.flatMap(page => page.users) ?? [];
  
  return { users, fetchNextPage, hasNextPage, isFetchingNextPage };
}

// 无限滚动组件
function UserList() {
  const { users, fetchNextPage, hasNextPage, isFetchingNextPage } = useInfiniteUsers();
  const loaderRef = useRef<HTMLDivElement>(null);
  
  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && hasNextPage && !isFetchingNextPage) {
          fetchNextPage();
        }
      },
      { threshold: 0.1 }
    );
    
    if (loaderRef.current) {
      observer.observe(loaderRef.current);
    }
    
    return () => observer.disconnect();
  }, [hasNextPage, isFetchingNextPage, fetchNextPage]);
  
  return (
    <div>
      {users.map(user => (
        <UserCard key={user.id} user={user} />
      ))}
      <div ref={loaderRef}>
        {isFetchingNextPage && <Spinner />}

  );
```

```typescript
// 性能监控 SDK
class PerformanceMonitor {
  private metrics: Map<string, number[]> = new Map();
  
  // 记录指标
  record(name: string, value: number) {
    if (!this.metrics.has(name)) {
      this.metrics.set(name, []);
    }
    this.metrics.get(name)!.push(value);
  }
  
  // 获取统计信息
  getStats(name: string) {
    const values = this.metrics.get(name) || [];
    if (values.length === 0) return null;
    
    const sum = values.reduce((a, b) => a + b, 0);
    const avg = sum / values.length;
    const sorted = [...values].sort((a, b) => a - b);
    const p50 = sorted[Math.floor(values.length * 0.5)];
    const p95 = sorted[Math.floor(values.length * 0.95)];
    const p99 = sorted[Math.floor(values.length * 0.99)];
    
    return { count: values.length, avg, p50, p95, p99 };
  }
  
  // 上报数据
  report() {
    const data: Record<string, any> = {};
    
    this.metrics.forEach((_, name) => {
      data[name] = this.getStats(name);
    });
    
    navigator.sendBeacon('/api/performance', JSON.stringify(data));
  }
}

// 使用
const monitor = new PerformanceMonitor();

// 监控 API 请求时间
async function fetchWithMonitor(url: string) {
  const start = performance.now();
  const response = await fetch(url);
  const duration = performance.now() - start;
  
  monitor.record('api_latency', duration);
  
  return response;
}

// 监控组件渲染时间
function useRenderTime(componentName: string) {
  useEffect(() => {
    const start = performance.now();
    
    return () => {
      const duration = performance.now() - start;
      monitor.record(`render_${componentName}`, duration);
    };
  }, [componentName]);
}
```

```typescript
          tagName: (event.target as HTMLElement).tagName,

        });
```

```typescript
// src/monitor/performance.ts - 完整性能监控

import { getCLS, getFID, getLCP, getFCP, getTTFB, Metric } from 'web-vitals';

// 性能数据类型
interface PerformanceData {
  // 页面信息
  url: string;
  path: string;
  referrer: string;

  // Web Vitals
  metrics: {
    cls: number;
    fid: number;
    lcp: number;
    fcp: number;
    ttfb: number;
  };

  // 资源信息
  resources: ResourceInfo[];

  // 自定义指标
  custom: Record<string, number>;

  // 环境信息
  userAgent: string;
  connection: ConnectionInfo;
  memory: MemoryInfo;

  // 时间戳
  timestamp: number;
}

interface ResourceInfo {
  name: string;
  type: string;
  size: number;
  duration: number;
}

interface ConnectionInfo {
  effectiveType: string;
  downlink: number;
  rtt: number;
}

interface MemoryInfo {
  usedJSHeapSize: number;
  totalJSHeapSize: number;
  jsHeapSizeLimit: number;
}

// 性能监控类
class PerformanceMonitor {
  private config: {
    dsn: string;
    sampleRate: number;
    debug: boolean;
  };

  private metrics: Partial<Record<string, number>> = {};
  private customMetrics: Record<string, number> = {};
  private queue: PerformanceData[] = [];
  private isSending = false;

  constructor(config: { dsn: string; sampleRate?: number; debug?: boolean }) {
    this.config = {
      sampleRate: 0.1,
      debug: false,
      ...config,
    };

    this.init();
  }

  private init() {
    // 采集 Web Vitals
    this.collectWebVitals();

    // 页面加载完成后采集
    if (document.readyState === 'complete') {
      this.collectPageLoad();
    } else {
      window.addEventListener('load', () => this.collectPageLoad());
    }

    // 页面卸载时上报
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') {
        this.sendBeacon();
      }
    });

    // 定时上报
    setInterval(() => this.send(), 10000);
  }

  // 采集 Web Vitals
  private collectWebVitals() {
    const sendMetric = (metric: Metric) => {
      this.metrics[metric.name.toLowerCase()] = metric.value;

      if (this.config.debug) {
        console.log(`[Performance] ${metric.name}: ${metric.value}`);
      }
    };

    getCLS(sendMetric);
    getFID(sendMetric);
    getLCP(sendMetric);
    getFCP(sendMetric);
    getTTFB(sendMetric);
  }

  // 采集页面加载性能
  private collectPageLoad() {
    const timing = performance.getEntriesByType('navigation')[0] as PerformanceNavigationTiming;
    if (!timing) return;

    // 页面加载时间
    this.customMetrics['page_load'] = timing.loadEventEnd - timing.fetchStart;
    this.customMetrics['dom_ready'] = timing.domContentLoadedEventEnd - timing.fetchStart;
    this.customMetrics['first_paint'] = timing.responseEnd - timing.fetchStart;

    // DNS、TCP、TTFB
    this.customMetrics['dns_time'] = timing.domainLookupEnd - timing.domainLookupStart;
    this.customMetrics['tcp_time'] = timing.connectEnd - timing.connectStart;
    this.customMetrics['ttfb'] = timing.responseStart - timing.requestStart;

    // DOM 处理时间
    this.customMetrics['dom_parse'] = timing.domInteractive - timing.responseEnd;
    this.customMetrics['dom_complete'] = timing.domComplete - timing.domInteractive;

    // 采集资源信息
    this.collectResources();
  }

  // 采集资源信息
  private collectResources() {
    const resources = performance.getEntriesByType('resource') as PerformanceResourceTiming[];

    const resourceInfo: ResourceInfo[] = resources.map((resource) => ({
      name: resource.name,
      type: this.getResourceType(resource),
      size: resource.transferSize || resource.encodedBodySize || 0,
      duration: resource.duration,
    }));

    // 按类型统计
    const stats = resourceInfo.reduce((acc, r) => {
      acc[r.type] = (acc[r.type] || 0) + r.size;
      return acc;
    }, {} as Record<string, number>);

    Object.entries(stats).forEach(([type, size]) => {
      this.customMetrics[`resource_${type}`] = size;
    });
  }

  // 获取资源类型
  private getResourceType(resource: PerformanceResourceTiming): string {
    const name = resource.name.toLowerCase();

    if (name.endsWith('.js') || name.includes('.js?')) return 'script';
    if (name.endsWith('.css') || name.includes('.css?')) return 'stylesheet';
    if (/\.(png|jpg|jpeg|gif|webp|svg|ico)/.test(name)) return 'image';
    if (/\.(woff|woff2|ttf|eot)/.test(name)) return 'font';
    if (name.includes('/api/') || name.includes('/graphql')) return 'api';

    return 'other';
  }

  // 获取连接信息
  private getConnection(): ConnectionInfo {
    const connection = (navigator as any).connection || {};
    return {
      effectiveType: connection.effectiveType || 'unknown',
      downlink: connection.downlink || 0,
      rtt: connection.rtt || 0,
    };
  }

  // 获取内存信息
  private getMemory(): MemoryInfo {
    const memory = (performance as any).memory || {};
    return {
      usedJSHeapSize: memory.usedJSHeapSize || 0,
      totalJSHeapSize: memory.totalJSHeapSize || 0,
      jsHeapSizeLimit: memory.jsHeapSizeLimit || 0,
    };
  }

  // 记录自定义指标
  public recordMetric(name: string, value: number) {
    this.customMetrics[name] = value;
  }

  // 记录耗时
  public recordTiming(name: string, startTime: number, endTime?: number) {
    const duration = (endTime || performance.now()) - startTime;
    this.customMetrics[name] = duration;
  }

  // 上报数据
  private report() {
    // 采样率检查
    if (Math.random() > this.config.sampleRate) return;

    const data: PerformanceData = {
      url: window.location.href,
      path: window.location.pathname,
      referrer: document.referrer,
      metrics: {
        cls: this.metrics['cls'] || 0,
        fid: this.metrics['fid'] || 0,
        lcp: this.metrics['lcp'] || 0,
        fcp: this.metrics['fcp'] || 0,
        ttfb: this.metrics['ttfb'] || 0,
      },
      resources: [],
      custom: { ...this.customMetrics },
      userAgent: navigator.userAgent,
      connection: this.getConnection(),
      memory: this.getMemory(),
      timestamp: Date.now(),
    };

    this.queue.push(data);
  }

  // 发送数据
  private async send() {
    if (this.isSending || this.queue.length === 0) return;

    this.isSending = true;
    const data = this.queue.splice(0, this.queue.length);

    try {
      await fetch(this.config.dsn, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data),
        keepalive: true,
      });
    } catch (error) {
      // 发送失败，放回队列
      this.queue.unshift(...data);
    } finally {
      this.isSending = false;
    }
  }

  // Beacon 发送
  private sendBeacon() {
    this.report();

    if (this.queue.length === 0) return;

    const data = this.queue.splice(0, this.queue.length);
    navigator.sendBeacon(this.config.dsn, JSON.stringify(data));
  }
}

// 初始化监控
export const performanceMonitor = new PerformanceMonitor({
  dsn: import.meta.env.VITE_MONITOR_DSN,
  sampleRate: 0.1,
  debug: import.meta.env.DEV,
});

// 性能测量工具
export const measure = {
  // 开始测量
  start(name: string) {
    performance.mark(`${name}-start`);
  },

  // 结束测量
  end(name: string) {
    performance.mark(`${name}-end`);
    performance.measure(name, `${name}-start`, `${name}-end`);

    const measure = performance.getEntriesByName(name)[0];
    performanceMonitor.recordMetric(name, measure.duration);

    // 清理
    performance.clearMarks(`${name}-start`);
    performance.clearMarks(`${name}-end`);
    performance.clearMeasures(name);

    return measure.duration;
  },

  // 测量异步函数
  async async<T>(name: string, fn: () => Promise<T>): Promise<T> {
    const start = performance.now();
    try {
      return await fn();
    } finally {
      performanceMonitor.recordTiming(name, start);
    }
  },
};
```

### Q3: 什么是 Web Vitals？如何优化？

**答案：**

Web Vitals 是 Google 定义的核心性能指标，用于衡量用户体验。

**核心指标：**

| 指标 | 全称 | 含义 | 目标值 |
|------|------|------|--------|
| **LCP** | Largest Contentful Paint | 最大内容绘制时间 | < 2.5s |
| **FID** | First Input Delay | 首次输入延迟 | < 100ms |
| **CLS** | Cumulative Layout Shift | 累积布局偏移 | < 0.1 |
| **FCP** | First Contentful Paint | 首次内容绘制 | < 1.8s |
| **TTFB** | Time to First Byte | 首字节时间 | < 600ms |
| **INP** | Interaction to Next Paint | 交互到下一次绘制 | < 200ms |

```typescript
  // 2. 采集页面加载性能
  collectPageLoad() {
    const timing = performance.getEntriesByType('navigation')[0];
    this.report('page_load', timing.loadEventEnd - timing.fetchStart);
    this.report('dom_ready', timing.domContentLoadedEventEnd - timing.fetchStart);
  }

  // 3. 采集资源性能
  collectResources() {
    const resources = performance.getEntriesByType('resource');
    resources.forEach(resource => {
      this.report('resource', {
        name: resource.name,
        type: this.getResourceType(resource),
        size: resource.transferSize,
        duration: resource.duration,
      });
    });
  }

  // 4. 采集自定义指标
  measureCustom(name: string, fn: () => void) {
    const start = performance.now();
    fn();
    const duration = performance.now() - start;
    this.report(name, duration);
  }

  // 5. 上报数据
  report(name: string, value: any) {
    const data = {
      name,
      value,
      url: window.location.href,
      userAgent: navigator.userAgent,
      timestamp: Date.now(),
    };

    // 使用 sendBeacon 保证页面关闭时也能上报
    navigator.sendBeacon('/api/performance', JSON.stringify(data));
  }
}
```

```typescript
function usePagination(items: any[], pageSize: number) {
  const [page, setPage] = useState(1);
  const visibleItems = items.slice(0, page * pageSize);
  const hasMore = visibleItems.length < items.length;

  return { visibleItems, hasMore, loadMore: () => setPage(p => p + 1) };
}

// 3. 无限滚动
function InfiniteList({ items }) {
  const loaderRef = useRef();

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          loadMore();
        }
      },
      { threshold: 0.1 }
    );

    if (loaderRef.current) {
      observer.observe(loaderRef.current);
    }

    return () => observer.disconnect();
  }, []);

  return (
    <>
      {items.map(item => <Item key={item.id} item={item} />)}
      <div ref={loaderRef}>Loading...</div>
    </>
  );
}

// 4. 时间分片渲染
function useTimeSlicedRender(items: any[], chunkSize: number = 50) {
  const [visibleItems, setVisibleItems] = useState<any[]>([]);
  const indexRef = useRef(0);

  useEffect(() => {
    function renderChunk() {
      const nextChunk = items.slice(indexRef.current, indexRef.current + chunkSize);
      setVisibleItems(prev => [...prev, ...nextChunk]);
      indexRef.current += chunkSize;

      if (indexRef.current < items.length) {
        requestAnimationFrame(renderChunk);
      }
    }

    renderChunk();
  }, [items, chunkSize]);

  return visibleItems;
}
```
