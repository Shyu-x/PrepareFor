## 核心架构模式对比

### MVC、MVP、MVVM 架构模式

| 模式 | Model（模型） | View（视图） | Controller/Presenter/ViewModel | 特点 |
|------|--------------|--------------|-------------------------------|------|
| **MVC** | 数据和业务逻辑 | 用户界面 | 协调者，处理输入 | View 直接访问 Model，Controller 臃肿 |
| **MVP** | 数据和业务逻辑 | 用户界面 | 中间人，接口隔离 | View 不直接访问 Model，完全解耦 |
| **MVVM** | 数据和业务逻辑 | 用户界面 | 双向数据绑定 | 自动同步，代码简洁 |

```tsx
// 展示层组件示例：数据展示表格
// 文件路径：src/components/展示层/DataTable.tsx

import { Table, Button, Space } from 'antd';
import { motion } from 'framer-motion';
import type { ColumnsType } from 'antd/es/table';

// 数据类型定义
interface DataRecord {
  id: string;
  name: string;
  status: 'active' | 'inactive';
  createdAt: string;
}

/**
 * 数据展示表格组件 - 展示层职责
 * 1. 纯展示逻辑，不包含业务处理
 * 2. 通过 props 接收数据和回调
 * 3. 使用原子化样式确保一致性
 */
export const DataTable = ({
  dataSource,
  loading,
  onEdit,
  onDelete,
  pagination,
  onTableChange,
}: {
  dataSource: DataRecord[];
  loading: boolean;
  onEdit: (record: DataRecord) => void;
  onDelete: (id: string) => void;
  pagination?: object;
  onTableChange?: () => void;
}) => {
  // 列定义
  const columns: ColumnsType<DataRecord> = [
    {
      title: '名称',
      dataIndex: 'name',
      key: 'name',
      // 纯展示逻辑：状态映射
      render: (name: string) => (
        <span className="font-medium text-gray-900">{name}</span>
      ),
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'status',
      // 纯展示逻辑：状态颜色映射
      render: (status: string) => (
        <span
          className={`px-2 py-1 rounded-full text-xs ${
            status === 'active'
              ? 'bg-green-100 text-green-800'
              : 'bg-gray-100 text-gray-800'
          }`}
        >
          {status === 'active' ? '启用' : '禁用'}
        </span>
      ),
    },
    {
      title: '操作',
      key: 'actions',
      // 纯展示逻辑：操作按钮渲染
      render: (_, record) => (
        <Space>
          <Button
            type="link"
            onClick={() => onEdit(record)}
            className="text-blue-600 hover:text-blue-800"
          >
            编辑
          </Button>
          <Button
            type="link"
            danger
            onClick={() => onDelete(record.id)}
          >
            删除
          </Button>
        </Space>
      ),
    },
  ];

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.3 }}
    >
      <Table
        columns={columns}
        dataSource={dataSource}
        loading={loading}
        rowKey="id"
        pagination={pagination}
        onChange={onTableChange}
        className="shadow-sm rounded-lg overflow-hidden"
      />
    </motion.div>
  );
};
```

```tsx
import Image from 'next/image';
import { Badge } from 'antd';
import { ShoppingCartOutlined } from '@ant-design/icons';
import { motion } from 'framer-motion';

/**
 * 商品卡片组件 - 纯展示层

interface ProductCardProps {

    name: string;
    price: number;
    image: string;
    tag?: string;
    stock: number;
  };
  onAddToCart: (productId: string) => void;
  onClick: (productId: string) => void;

export const ProductCard = ({
  product,
  onAddToCart,
  onClick,
}: ProductCardProps) => {

  return (
    <motion.div
      whileHover={{ scale: 1.02 }}
      whileTap={{ scale: 0.98 }}
      className={`
        relative bg-white rounded-xl shadow-md overflow-hidden
        transition-shadow duration-300 hover:shadow-lg
        ${isOutOfStock ? 'opacity-60' : ''}
      `}
      onClick={() => !isOutOfStock && onClick(product.id)}
    >
      {/* 商品图片 */}
      <div className="relative h-48 w-full">
        <Image
          src={product.image}
          alt={product.name}
          fill
          className="object-cover"
        />

        {product.tag && (
          <Badge
            count={product.tag}
            className="absolute top-2 left-2"
            style={{ backgroundColor: '#f5222d' }}
          />
        )}
        {/* 缺货标记 */}
        {isOutOfStock && (
          <div className="absolute inset-0 flex items-center justify-center bg-black/50">
            <span className="text-white font-bold text-lg">缺货</span>
          </div>
        )}
      </div>

      {/* 商品信息 */}
      <div className="p-4">
        <h3 className="text-lg font-semibold text-gray-900 truncate">
          {product.name}
        </h3>
        <div className="mt-2 flex items-center justify-between">
          <span className="text-2xl font-bold text-red-500">
            ¥{product.price.toFixed(2)}
          </span>
          <motion.button
            whileHover={{ scale: 1.1 }}
            whileTap={{ scale: 0.9 }}

              e.stopPropagation();

              }
            `}
```

### 2.2 MVC/MVP/MVVM 在视图逻辑层的应用

#### MVC 模式 - 传统事件驱动

```tsx
// MVC 模式下的视图逻辑层
// 文件路径：src/components/视图逻辑层/LoginFormMVC.tsx

import { useState } from 'react';
import { Input, Button, message } from 'antd';

/**
 * MVC 模式 - 视图逻辑层
 * Controller 处理事件，业务逻辑在 Controller 中
 */
export const LoginFormMVC = ({
  onLogin, // Controller 传入的登录处理函数
}: {
  onLogin: (username: string, password: string) => Promise<void>;
}) => {
  // View 状态
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);

  // View 事件处理
  const handleSubmit = async () => {
    // View 层不处理业务逻辑，只负责收集用户输入
    if (!username || !password) {
      message.warning('请输入用户名和密码');
      return;
    }

    setLoading(true);
    try {
      // 调用 Controller 处理业务逻辑
      await onLogin(username, password);
      message.success('登录成功');
    } catch (error) {
      message.error('登录失败');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="space-y-4">
      <Input
        placeholder="用户名"
        value={username}
        onChange={(e) => setUsername(e.target.value)}
      />
      <Input.Password
        placeholder="密码"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
      />
      <Button
        type="primary"
        loading={loading}
        onClick={handleSubmit}
        block
      >
        登录
      </Button>
    </div>
  );
};
```

#### MVVM 模式 - 双向数据绑定

```tsx
// MVVM 模式下的视图逻辑层
// 文件路径：src/components/视图逻辑层/LoginFormMVVM.tsx

import { useState, useCallback } from 'react';
import { Input, Button, message } from 'antd';

/**
 * MVVM 模式 - ViewModel 层
 * 封装视图状态和业务逻辑的双向绑定
 */
interface LoginViewModel {
  username: string;
  password: string;
  setUsername: (value: string) => void;
  setPassword: (value: string) => void;
  submit: () => Promise<void>;
  isLoading: boolean;
  errors: {
    username?: string;
    password?: string;
  };
}

/**
 * 使用自定义 Hook 实现 ViewModel
 * 将视图逻辑封装为可复用的 ViewModel
 */
const useLoginViewModel = (): LoginViewModel => {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [errors, setErrors] = useState<{ username?: string; password?: string }>({});

  // 验证逻辑
  const validate = useCallback(() => {
    const newErrors: typeof errors = {};

    if (!username) {
      newErrors.username = '请输入用户名';
    } else if (username.length < 3) {
      newErrors.username = '用户名至少3个字符';
    }

    if (!password) {
      newErrors.password = '请输入密码';
    } else if (password.length < 6) {
      newErrors.password = '密码至少6个字符';
    }

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  }, [username, password]);

  // 提交逻辑
  const submit = useCallback(async () => {
    if (!validate()) return;

    setIsLoading(true);
    try {
      // API 调用
      await fetch('/api/login', {
        method: 'POST',
        body: JSON.stringify({ username, password }),
      });
      message.success('登录成功');
    } catch (error) {
      message.error('登录失败');
    } finally {
      setIsLoading(false);
    }
  }, [username, password, validate]);

  return {
    username,
    password,
    setUsername,
    setPassword,
    submit,
    isLoading,
    errors,
```

```tsx
/**
 * 拖拽排序列表主组件
 */
export const DraggableList = ({
  initialItems,
  onOrderChange,
}: {
  initialItems: { id: string; content: string }[];
  onOrderChange: (items: { id: string; content: string }[]) => void;
}) => {
  const [items, setItems] = useState(initialItems);

  // 处理拖拽结束
  const handleDragEnd = useCallback(
    (event: DragEndEvent) => {
      const { active, over } = event;

      if (over && active.id !== over.id) {
        setItems((currentItems) => {
          const oldIndex = currentItems.findIndex((item) => item.id === active.id);
          const newIndex = currentItems.findIndex((item) => item.id === over.id);

          const newItems = [...currentItems];
          const [removed] = newItems.splice(oldIndex, 1);
          newItems.splice(newIndex, 0, removed);

          // 通知父组件顺序变化
          onOrderChange(newItems);

          return newItems;
        });
      }
    },
    [onOrderChange]
  );

  return (
    <DndContext
      collisionDetection={closestCenter}
      onDragEnd={handleDragEnd}
    >
      <SortableContext
        items={items.map((item) => item.id)}
        strategy={verticalListSortingStrategy}
      >
        <div className="space-y-2">
          {items.map((item) => (
            <SortableItem key={item.id} id={item.id} content={item.content} />
          ))}
        </div>
      </SortableContext>
    </DndContext>
  );
};
```

```typescript
// 业务逻辑层 - 领域模型
// 文件路径：src/domain/entities/Order.ts

/**
 * 订单实体 - 领域模型核心
 * 封装订单相关的业务规则和状态流转
 */

// 值对象：订单状态
enum OrderStatus {
  PENDING = 'PENDING',           // 待支付
  PAID = 'PAID',                 // 已支付
  SHIPPING = 'SHIPPING',         // 发货中
  DELIVERED = 'DELIVERED',       // 已送达
  CANCELLED = 'CANCELLED',       // 已取消
  REFUNDING = 'REFUNDING',       // 退款中
  REFUNDED = 'REFUNDED',         // 已退款
}

// 值对象：订单商品
interface OrderItem {
  productId: string;
  productName: string;
  quantity: number;
  unitPrice: number;
  discount: number; // 折扣金额
}

/**
 * 订单实体 - 聚合根
 * 封装订单完整的业务逻辑
 */
export class Order {
  // 私有属性，确保数据一致性
  private _id: string;
  private _status: OrderStatus;
  private _items: OrderItem[];
  private _totalAmount: number;
  private _paidAmount: number;
  private _createdAt: Date;
  private _updatedAt: Date;
  private _paidAt?: Date;
  private _shippedAt?: Date;
  private _deliveredAt?: Date;

  constructor(params: {
    id: string;
    items: OrderItem[];
    userId: string;
  }) {
    this._id = params.id;
    this._items = params.items;
    this._status = OrderStatus.PENDING;
    this._totalAmount = this.calculateTotalAmount();
    this._paidAmount = 0;
    this._createdAt = new Date();
    this._updatedAt = new Date();
  }

  // 计算订单总金额
  private calculateTotalAmount(): number {
    return this._items.reduce((sum, item) => {
      return sum + (item.unitPrice * item.quantity - item.discount);
    }, 0);
  }

  // 领域方法：支付订单
  pay(paymentAmount: number): void {
    // 业务规则：只有待支付状态才能支付
    if (this._status !== OrderStatus.PENDING) {
      throw new Error('只有待支付的订单才能支付');
    }

    // 业务规则：支付金额必须等于订单金额
    if (paymentAmount < this._totalAmount) {
      throw new Error('支付金额不足');
    }

    this._status = OrderStatus.PAID;
    this._paidAmount = paymentAmount;
    this._paidAt = new Date();
    this._updatedAt = new Date();
  }

  // 领域方法：取消订单
  cancel(reason: string): void {
    // 业务规则：已支付或已发货的订单不能直接取消
    if (
      this._status === OrderStatus.SHIPPING ||
      this._status === OrderStatus.DELIVERED
    ) {
      throw new Error('订单已发货或已送达，无法直接取消');
    }

    // 业务规则：已支付的订单取消需要退款
    if (this._status === OrderStatus.PAID) {
      this.initiateRefund();
    } else {
      this._status = OrderStatus.CANCELLED;
    }

    this._updatedAt = new Date();
  }

  // 领域方法：发货
  ship(trackingNumber: string): void {
    // 业务规则：只有已支付的订单才能发货
    if (this._status !== OrderStatus.PAID) {
      throw new Error('只有已支付的订单才能发货');
    }

    this._status = OrderStatus.SHIPPING;
    this._shippedAt = new Date();
    this._updatedAt = new Date();
  }

  // 领域方法：确认收货
  deliver(): void {
    // 业务规则：只有发货中的订单才能确认收货
    if (this._status !== OrderStatus.SHIPPING) {
      throw new Error('只有发货中的订单才能确认收货');
    }

    this._status = OrderStatus.DELIVERED;
    this._deliveredAt = new Date();
    this._updatedAt = new Date();
  }

  // 领域方法：申请退款
  private initiateRefund(): void {
    this._status = OrderStatus.REFUNDING;
    this._updatedAt = new Date();
  }

  // 领域方法：完成退款
  completeRefund(): void {
    // 业务规则：只有退款中的订单才能完成退款
    if (this._status !== OrderStatus.REFUNDING) {
      throw new Error('只有退款中的订单才能完成退款');
    }

    this._status = OrderStatus.REFUNDED;
    this._updatedAt = new Date();
  }

  // 获取方法（只读访问）
  get id(): string {
    return this._id;
  }

  get status(): OrderStatus {
    return this._status;
  }

  get totalAmount(): number {
    return this._totalAmount;
  }

  get items(): OrderItem[] {
    return [...this._items]; // 返回副本，防止外部修改
  }

  // 状态判断方法
  canCancel(): boolean {
    return (
      this._status === OrderStatus.PENDING ||
      this._status === OrderStatus.PAID
    );
  }

  canRefund(): boolean {
    return (
      this._status === OrderStatus.PAID ||
      this._status === OrderStatus.SHIPPING
    );
  }

  canShip(): boolean {
    return this._status === OrderStatus.PAID;
  }

  canDeliver(): boolean {
    return this._status === OrderStatus.SHIPPING;
  }
}
```

```typescript
// 业务逻辑层 - 订单服务编排
// 文件路径：src/domain/services/OrderService.ts

import { Order, OrderStatus, OrderItem } from '../entities/Order';
import { PaymentService } from './PaymentService';
import { InventoryService } from './InventoryService';
import { NotificationService } from './NotificationService';
import { AuditLogService } from './AuditLogService';

/**
 * 订单业务服务 - 服务编排层
 * 协调多个领域服务完成复杂业务场景
 */
export class OrderService {
  constructor(
    private paymentService: PaymentService,
    private inventoryService: InventoryService,
    private notificationService: NotificationService,
    private auditLogService: AuditLogService
  ) {}

  /**
   * 创建订单 - 完整业务流程
   */
  async createOrder(params: {
    userId: string;
    items: OrderItem[];
    shippingAddress: string;
    paymentMethod: string;
  }): Promise<Order> {
    // 1. 验证库存
    await this.inventoryService.validateStock(params.items);

    // 2. 创建订单
    const order = new Order({
      id: this.generateOrderId(),
      items: params.items,
      userId: params.userId,
    });

    // 3. 记录审计日志
    await this.auditLogService.log('ORDER_CREATED', {
      orderId: order.id,
      userId: params.userId,
      totalAmount: order.totalAmount,
    });

    // 4. 发送通知
    await this.notificationService.sendOrderCreated(order);

    return order;
  }

  /**
   * 支付订单 - 完整业务流程
   */
  async payOrder(params: {
    orderId: string;
    paymentMethod: string;
    paymentAmount: number;
  }): Promise<Order> {
    // 1. 获取订单
    const order = await this.getOrder(params.orderId);

    // 2. 执行业务操作
    order.pay(params.paymentAmount);

    // 3. 处理支付
    const paymentResult = await this.paymentService.processPayment({
      orderId: order.id,
      amount: params.paymentAmount,
      method: params.paymentMethod,
    });

    if (!paymentResult.success) {
      throw new Error(`支付失败: ${paymentResult.errorMessage}`);
    }

    // 4. 更新库存
    await this.inventoryService.reserveStock(order.items);

    // 5. 记录审计日志
    await this.auditLogService.log('ORDER_PAID', {
      orderId: order.id,
      paymentId: paymentResult.paymentId,
      amount: params.paymentAmount,
    });

    // 6. 发送通知
    await this.notificationService.sendOrderPaid(order);

    return order;
  }

  /**
   * 取消订单 - 完整业务流程
   */
  async cancelOrder(params: {
    orderId: string;
    userId: string;
    reason: string;
  }): Promise<Order> {
    // 1. 获取订单
    const order = await this.getOrder(params.orderId);

    // 2. 验证权限
    if (order.id !== params.userId) {
      throw new Error('无权取消此订单');
    }

    // 3. 执行业务操作
    order.cancel(params.reason);

    // 4. 如果已支付，启动退款流程
    if (order.status === OrderStatus.REFUNDING) {
      await this.paymentService.initiateRefund({
        orderId: order.id,
        amount: order.totalAmount,
      });
    }
```

```typescript
// 业务逻辑层 - 购物车结算服务
// 文件路径：src/domain/services/CartCheckoutService.ts

/**
 * 购物车结算业务服务
 * 职责：购物车验证、优惠计算、订单创建
 */

interface CartItem {
  productId: string;
  quantity: number;
  price: number;
}

interface DiscountRule {
  type: 'percentage' | 'fixed' | 'shipping';
  value: number;
  minPurchase?: number;
  applicableProducts?: string[];
}

interface CheckoutResult {
  orderId: string;
  subtotal: number;
  discount: number;
  shippingFee: number;
  total: number;
  appliedPromotions: string[];
}

/**
 * 购物车结算服务
 */
export class CartCheckoutService {
  constructor(
    private productService: ProductService,
    private promotionService: PromotionService,
    private inventoryService: InventoryService
  ) {}

  /**
   * 结算流程 - 核心业务逻辑
   */
  async checkout(params: {
    userId: string;
    items: CartItem[];
    promotionCodes?: string[];
    shippingAddress: string;
  }): Promise<CheckoutResult> {
    // 步骤1：验证购物车商品
    const validItems = await this.validateCartItems(params.items);

    // 步骤2：计算小计
    const subtotal = this.calculateSubtotal(validItems);

    // 步骤3：获取并应用促销
    const { discount, appliedPromotions } = await this.calculateDiscount(
      subtotal,
      validItems,
      params.promotionCodes
    );

    // 步骤4：计算运费
    const shippingFee = this.calculateShippingFee(subtotal, params.shippingAddress);

    // 步骤5：计算总额
    const total = subtotal - discount + shippingFee;

    // 步骤6：创建订单
    const orderId = await this.createOrder({
      userId: params.userId,
      items: validItems,
      total,
    });

    // 步骤7：预留库存
    await this.reserveInventory(validItems);

    return {
      orderId,
      subtotal,
      discount,
      shippingFee,
      total,
      appliedPromotions,
    };
  }

  /**
   * 验证购物车商品
   */
  private async validateCartItems(items: CartItem[]): Promise<CartItem[]> {
    const validItems: CartItem[] = [];

    for (const item of items) {
      // 检查商品是否存在
      const product = await this.productService.getProduct(item.productId);
      if (!product) {
        console.warn(`商品 ${item.productId} 已下架`);
        continue;
      }

      // 检查库存
      const stock = await this.inventoryService.getStock(item.productId);
      if (stock < item.quantity) {
        console.warn(`商品 ${product.name} 库存不足，当前库存: ${stock}`);
        // 可以选择减少数量或跳过
        continue;
      }

      // 检查价格是否变化
      if (product.price !== item.price) {
        console.warn(`商品 ${product.name} 价格已更新`);
        // 更新为最新价格
        validItems.push({
          ...item,
          price: product.price,
        });
      } else {
        validItems.push(item);
      }
    }

    if (validItems.length === 0) {
      throw new Error('购物车中没有有效的商品');
    }

    return validItems;
  }

  /**
   * 计算小计
   */
  private calculateSubtotal(items: CartItem[]): number {
    return items.reduce((sum, item) => sum + item.price * item.quantity, 0);
  }

  /**
   * 计算优惠
   */
  private async calculateDiscount(
    subtotal: number,
    items: CartItem[],
    promotionCodes?: string[]
  ): Promise<{ discount: number; appliedPromotions: string[] }> {
    let totalDiscount = 0;
    const appliedPromotions: string[] = [];

    // 获取适用的促销活动
    const promotions = await this.promotionService.getApplicablePromotions({
      subtotal,
      items,
      codes: promotionCodes,
    });

    // 应用每个促销
    for (const promotion of promotions) {
      const discount = this.calculatePromotionDiscount(promotion, subtotal, items);
      if (discount > 0) {
        totalDiscount += discount;
        appliedPromotions.push(promotion.name);
      }
    }

    // 确保折扣不超过小计
    totalDiscount = Math.min(totalDiscount, subtotal);

    return { discount: totalDiscount, appliedPromotions };
  }

  /**
   * 计算单个促销的折扣
   */
  private calculatePromotionDiscount(
    rule: DiscountRule,
    subtotal: number,
    items: CartItem[]
  ): number {
    // 检查最低消费门槛
    if (rule.minPurchase && subtotal < rule.minPurchase) {
      return 0;
    }

    // 检查适用商品范围
    if (rule.applicableProducts) {
      const applicableItems = items.filter((item) =>
        rule.applicableProducts!.includes(item.productId)
      );
      if (applicableItems.length === 0) {
        return 0;
      }
    }

    switch (rule.type) {
      case 'percentage':
        // 百分比折扣
        return subtotal * (rule.value / 100);

      case 'fixed':
        // 固定金额折扣
        return rule.value;

      case 'shipping':
        // 免运费（返回0，实际运费计算时会考虑）
        return 0;

      default:
        return 0;
    }
  }

  /**
   * 计算运费
   */
  private calculateShippingFee(subtotal: number, address: string): number {
    // 满99元包邮
    if (subtotal >= 99) {
      return 0;
    }

    // 根据地址计算运费（简化示例）
    if (address.includes('偏远地区')) {
      return 20;
    }

    return 10;
  }

  private async createOrder(params: {
    userId: string;
    items: CartItem[];
    total: number;
  }): Promise<string> {
    // 创建订单逻辑
    return `ORD${Date.now()}`;
  }

  private async reserveInventory(items: CartItem[]): Promise<void> {
    // 预留库存逻辑
  }
}
```

```typescript
// 数据服务层 - 统一 API 客户端
// 文件路径：src/services/api/client.ts

import axios, { AxiosInstance, AxiosError, AxiosRequestConfig } from 'axios';
import { API_CONFIG } from '@/config';
import { TokenService } from './tokenService';
import { ErrorHandler } from './errorHandler';

/**
 * API 客户端配置
 */
interface ApiClientConfig {
  baseURL: string;
  timeout: number;
  retryAttempts: number;
}

/**
 * 统一 API 响应格式
 */
interface ApiResponse<T> {
  success: boolean;
  data?: T;
  error?: {
    code: string;
    message: string;
    details?: unknown;
  };
  meta?: {
    page?: number;
    pageSize?: number;
    total?: number;
  };
}

/**
 * API 客户端类
 * 封装所有 HTTP 请求逻辑
 */
class ApiClient {
  private instance: AxiosInstance;
  private retryAttempts: number;

  constructor(config: ApiClientConfig) {
    this.retryAttempts = config.retryAttempts;

    // 创建 axios 实例
    this.instance = axios.create({
      baseURL: config.baseURL,
      timeout: config.timeout,
      headers: {
        'Content-Type': 'application/json',
      },
    });

    // 添加请求拦截器
    this.setupRequestInterceptor();

    // 添加响应拦截器
    this.setupResponseInterceptor();
  }

  /**
   * 请求拦截器 - 添加认证信息
   */
  private setupRequestInterceptor() {
    this.instance.interceptors.request.use(
      async (config) => {
        // 添加 Token
        const token = await TokenService.getAccessToken();
        if (token) {
          config.headers.Authorization = `Bearer ${token}`;
        }

        // 添加请求 ID（用于链路追踪）
        config.headers['X-Request-ID'] = this.generateRequestId();

        return config;
      },
      (error) => {
        return Promise.reject(error);
      }
    );
  }

  /**
   * 响应拦截器 - 统一处理响应和错误
   */
  private setupResponseInterceptor() {
    this.instance.interceptors.response.use(
      (response) => {
        // 检查业务错误
        const apiResponse = response.data as ApiResponse<unknown>;
        if (apiResponse.success === false) {
          return Promise.reject(
            new Error(apiResponse.error?.message || '请求失败')
          );
        }

        return response;
      },
      async (error: AxiosError) => {
        // 处理不同类型的错误
        if (error.response) {
          // 服务器返回错误状态码
          return this.handleServerError(error);
        } else if (error.request) {
          // 请求已发送但没有收到响应
          return this.handleNetworkError(error);
        } else {
          // 请求配置出错
          return this.handleRequestError(error);
        }
      }
    );
  }

  /**
   * 处理服务器错误
   */
  private async handleServerError(error: AxiosError): Promise<never> {
    const status = error.response?.status;

    switch (status) {
      case 400:
        ErrorHandler.handleBadRequest(error);
        break;
      case 401:
        // 尝试刷新 Token
        const refreshed = await TokenService.refreshToken();
        if (refreshed) {
          // 重试原请求
          return this.instance.request(error.config!);
        }
        ErrorHandler.handleUnauthorized();
        break;
      case 403:
        ErrorHandler.handleForbidden();
        break;
      case 404:
        ErrorHandler.handleNotFound(error.config?.url);
        break;
      case 500:
      case 502:
      case 503:
        ErrorHandler.handleServerError(status);
        break;
    }

    return Promise.reject(error);
  }

  /**
   * 处理网络错误
   */
  private handleNetworkError(error: AxiosError): Promise<never> {
    if (axios.isCancel(error)) {
      // 请求被取消，不处理
      return Promise.reject(error);
    }

    ErrorHandler.handleNetworkError();
    return Promise.reject(new Error('网络连接失败，请检查网络设置'));
  }

  /**
   * 处理请求错误
   */
  private handleRequestError(error: AxiosError): Promise<never> {
    ErrorHandler.handleRequestError(error.message);
    return Promise.reject(error);
  }

  /**
   * GET 请求
   */
  async get<T>(
    url: string,
    params?: Record<string, unknown>,
    config?: AxiosRequestConfig
  ): Promise<T> {
    const response = await this.instance.get<ApiResponse<T>>(url, {
      params,
      ...config,
    });
    return response.data.data as T;
  }

  /**
   * POST 请求
   */
  async post<T>(
    url: string,
    data?: unknown,
    config?: AxiosRequestConfig
  ): Promise<T> {
    const response = await this.instance.post<ApiResponse<T>>(url, data, config);
    return response.data.data as T;
  }

  /**
   * PUT 请求
   */
  async put<T>(
    url: string,
    data?: unknown,
    config?: AxiosRequestConfig
  ): Promise<T> {
    const response = await this.instance.put<ApiResponse<T>>(url, data, config);
    return response.data.data as T;
  }

  /**
   * DELETE 请求
   */
  async delete<T>(
    url: string,
    config?: AxiosRequestConfig
  ): Promise<T> {
    const response = await this.instance.delete<ApiResponse<T>>(url, config);
    return response.data.data as T;
  }

  /**
   * 生成请求 ID
   */
  private generateRequestId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  /**
   * 取消所有请求
   */
  cancelAllRequests() {
    // 实现请求取消逻辑
  }
}

// 导出单例
export const apiClient = new ApiClient({
  baseURL: API_CONFIG.baseURL,
  timeout: API_CONFIG.timeout,
  retryAttempts: API_CONFIG.retryAttempts,
});
```

```typescript
        ...state,

        updatedAt: new Date().toISOString(),
      };
    }
  );

  // 乐观更新标题
  const setTitle = (title: string) => {
    updateOptimisticDocument({ title });
    store.updateTitle(title);
  };

  // 乐观更新内容
  const setContent = (content: string) => {
    updateOptimisticDocument({ content });
    store.updateContent(content);
  };

  // 保存
  const save = async () => {
    try {
      await store.saveDocument();
    } catch {
      // 保存失败，可能需要回滚
    }
  };

  return {
    document: optimisticDocument,
    isLoading,
    error,
    setTitle,
    setContent,
    save,
  };
```

```typescript
// 数据访问层 - 缓存服务
// 文件路径：src/services/cache/CacheService.ts

/**
 * 缓存配置
 */
interface CacheConfig {
  ttl: number; // 生存时间（毫秒）
  maxSize: number; // 最大缓存数量
}

/**
 * 缓存项
 */
interface CacheItem<T> {
  value: T;
  expiry: number;
  createdAt: number;
}

/**
 * 通用缓存服务
 * 支持内存缓存和本地存储
 */
export class CacheService {
  private memoryCache: Map<string, CacheItem<unknown>> = new Map();
  private config: CacheConfig;

  constructor(config: CacheConfig = { ttl: 5 * 60 * 1000, maxSize: 100 }) {
    this.config = config;
  }

  /**
   * 设置缓存
   */
  set<T>(key: string, value: T, ttl?: number): void {
    const item: CacheItem<T> = {
      value,
      expiry: Date.now() + (ttl || this.config.ttl),
      createdAt: Date.now(),
    };

    // 内存缓存
    this.memoryCache.set(key, item as CacheItem<unknown>);

    // 检查大小限制
    if (this.memoryCache.size > this.config.maxSize) {
      this.evictOldest();
    }
  }

  /**
   * 获取缓存
   */
  get<T>(key: string): T | null {
    const item = this.memoryCache.get(key) as CacheItem<T> | undefined;

    if (!item) {
      // 尝试从本地存储获取
      const localItem = localStorage.getItem(`cache:${key}`);
      if (localItem) {
        const parsed = JSON.parse(localItem) as CacheItem<T>;
        if (parsed.expiry > Date.now()) {
          // 恢复到内存
          this.memoryCache.set(key, parsed as CacheItem<unknown>);
          return parsed.value;
        } else {
          // 已过期，删除
          localStorage.removeItem(`cache:${key}`);
        }
      }
      return null;
    }

    // 检查过期
    if (item.expiry < Date.now()) {
      this.memoryCache.delete(key);
      return null;
    }

    return item.value;
  }

  /**
   * 删除缓存
   */
  delete(key: string): void {
    this.memoryCache.delete(key);
    localStorage.removeItem(`cache:${key}`);
  }

  /**
   * 清空所有缓存
   */
  clear(): void {
    this.memoryCache.clear();
  }

  /**
   * 驱逐最老的缓存项
   */
  private evictOldest(): void {
    let oldestKey: string | null = null;
    let oldestTime = Infinity;

    for (const [key, item] of this.memoryCache) {
      if (item.createdAt < oldestTime) {
        oldestTime = item.createdAt;
        oldestKey = key;
      }
    }

    if (oldestKey) {
      this.delete(oldestKey);
    }
  }

  /**
   * 持久化到本地存储
   */
  persist(key: string): void {
    const item = this.memoryCache.get(key);
    if (item) {
      localStorage.setItem(`cache:${key}`, JSON.stringify(item));
    }
  }

  /**
   * 获取缓存统计信息
   */
  getStats(): {
    size: number;
    keys: string[];
  } {
    return {
      size: this.memoryCache.size,
      keys: Array.from(this.memoryCache.keys()),
    };
  }
}

// 导出单例
export const cacheService = new CacheService();
```

```typescript
// 基础设施层 - 日志服务
// 文件路径：src/services/infrastructure/Logger.ts

/**
 * 日志级别
 */
enum LogLevel {
  DEBUG = 0,
  INFO = 1,
  WARN = 2,
  ERROR = 3,
  CRITICAL = 4,
}

/**
 * 日志条目
 */
interface LogEntry {
  level: LogLevel;
  message: string;
  context?: Record<string, unknown>;
  timestamp: string;
  userId?: string;
  sessionId?: string;
}

/**
 * 日志服务
 * 统一管理前端日志
 */
class Logger {
  private static instance: Logger;
  private logs: LogEntry[] = [];
  private maxLogs = 1000;
  private level: LogLevel = LogLevel.INFO;
  private userId?: string;
  private sessionId: string;

  private constructor() {
    this.sessionId = this.generateSessionId();
    this.setupGlobalHandlers();
  }

  static getInstance(): Logger {
    if (!Logger.instance) {
      Logger.instance = new Logger();
    }
    return Logger.instance;
  }

  /**
   * 设置用户 ID
   */
  setUser(userId: string): void {
    this.userId = userId;
  }

  /**
   * 设置日志级别
   */
  setLevel(level: LogLevel): void {
    this.level = level;
  }

  /**
   * Debug 日志
   */
  debug(message: string, context?: Record<string, unknown>): void {
    this.log(LogLevel.DEBUG, message, context);
  }

  /**
   * Info 日志
   */
  info(message: string, context?: Record<string, unknown>): void {
    this.log(LogLevel.INFO, message, context);
  }

  /**
   * Warn 日志
   */
  warn(message: string, context?: Record<string, unknown>): void {
    this.log(LogLevel.WARN, message, context);
  }

  /**
   * Error 日志
   */
  error(message: string, context?: Record<string, unknown>): void {
    this.log(LogLevel.ERROR, message, context);
  }

  /**
   * Critical 日志
   */
  critical(message: string, context?: Record<string, unknown>): void {
    this.log(LogLevel.CRITICAL, message, context);
  }

  /**
   * 记录日志
   */
  private log(
    level: LogLevel,
    message: string,
    context?: Record<string, unknown>
  ): void {
    if (level < this.level) return;

    const entry: LogEntry = {
      level,
      message,
      context,
      timestamp: new Date().toISOString(),
      userId: this.userId,
      sessionId: this.sessionId,
    };

    // 添加到内存
    this.logs.push(entry);

    // 检查大小限制
    if (this.logs.length > this.maxLogs) {
      this.logs.shift();
    }

    // 输出到控制台
    this.outputToConsole(entry);

    // 发送到服务器（仅 Error 及以上）
    if (level >= LogLevel.ERROR) {
      this.sendToServer(entry);
    }
  }

  /**
   * 输出到控制台
   */
  private outputToConsole(entry: LogEntry): void {
    const levelNames = ['DEBUG', 'INFO', 'WARN', 'ERROR', 'CRITICAL'];
    const style = this.getConsoleStyle(entry.level);

    console.log(
      `%c[${levelNames[entry.level]}] ${entry.timestamp}`,
      style,
      entry.message,
      entry.context || ''
    );
  }

  /**
   * 获取控制台样式
   */
  private getConsoleStyle(level: LogLevel): string {
    const styles: Record<LogLevel, string> = {
      [LogLevel.DEBUG]: 'color: gray',
      [LogLevel.INFO]: 'color: blue',
      [LogLevel.WARN]: 'color: orange',
      [LogLevel.ERROR]: 'color: red',
      [LogLevel.CRITICAL]: 'color: white; background: red; font-weight: bold',
    };
    return styles[level];
  }

  /**
   * 发送到服务器
   */
  private async sendToServer(entry: LogEntry): Promise<void> {
    try {
      // 节流：批量发送
      await fetch('/api/logs', {
        method: 'POST',
        body: JSON.stringify(entry),
        headers: { 'Content-Type': 'application/json' },
        // 不等待响应
      }).catch(() => {});
    } catch {
      // 静默失败
    }
  }

  /**
   * 设置全局错误处理器
   */
  private setupGlobalHandlers(): void {
    // 未捕获的 Promise 错误
    window.addEventListener('unhandledrejection', (event) => {
      this.error('Unhandled Promise Rejection', {
        reason: event.reason?.message || event.reason,
        stack: event.reason?.stack,
      });
    });

    // 全局错误
    window.addEventListener('error', (event) => {
      this.error('Uncaught Error', {
        message: event.message,
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno,
      });
    });
  }

  /**
   * 生成会话 ID
   */
  private generateSessionId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  /**
   * 获取所有日志
   */
  getLogs(): LogEntry[] {
    return [...this.logs];
  }
}

export const logger = Logger.getInstance();
```

```typescript
/**
 * JWT Token 服务
 */
export class TokenService {
  private static readonly ACCESS_TOKEN_KEY = 'access_token';
  private static readonly REFRESH_TOKEN_KEY = 'refresh_token';
  private static readonly TOKEN_EXPIRY_BUFFER = 5 * 60 * 1000; // 5分钟

  /**
   * 获取 Access Token
   */
  static async getAccessToken(): Promise<string | null> {
    const token = localStorage.getItem(this.ACCESS_TOKEN_KEY);

    if (!token) {
      return null;
    }

    // 检查是否即将过期
    if (this.isTokenExpiringSoon(token)) {
      // 尝试刷新
      const refreshed = await this.refreshToken();
      return refreshed ? localStorage.getItem(this.ACCESS_TOKEN_KEY) : null;
    }

    return token;
  }

  /**
   * 设置 Token
   */
  static setTokens(accessToken: string, refreshToken: string): void {
    localStorage.setItem(this.ACCESS_TOKEN_KEY, accessToken);
    localStorage.setItem(this.ACCESS_TOKEN_KEY, refreshToken);
  }

  /**
   * 清除 Token
   */
  static clearTokens(): void {
    localStorage.removeItem(this.ACCESS_TOKEN_KEY);
    localStorage.removeItem(this.REFRESH_TOKEN_KEY);
  }

  /**
   * 刷新 Token
   */
  static async refreshToken(): Promise<boolean> {
    const refreshToken = localStorage.getItem(this.REFRESH_TOKEN_KEY);

    if (!refreshToken) {
      return false;
    }

    try {
      const response = await fetch('/api/auth/refresh', {
        method: 'POST',
        body: JSON.stringify({ refreshToken }),
      });

      if (!response.ok) {
        this.clearTokens();
        return false;
      }

      const { accessToken, refreshToken: newRefreshToken } = await response.json();
      this.setTokens(accessToken, newRefreshToken);
      return true;
    } catch {
      return false;
    }
  }

  /**
   * 检查 Token 是否即将过期
   */
  private static isTokenExpiringSoon(token: string): boolean {
    try {
      const payload = JSON.parse(atob(token.split('.')[1]));
      const expiry = payload.exp * 1000;
      return Date.now() + this.TOKEN_EXPIRY_BUFFER > expiry;
    } catch {
      return true;
    }
  }
}
```

```tsx
interface RemoteModuleConfig {
  scope: string;
  module: string;
}

/**
 * 动态加载远程模块
 * 支持运行时决定加载哪个远程应用
 */
export function loadRemoteModule<T = any>(config: RemoteModuleConfig): Promise<T> {
  return new Promise((resolve, reject) => {
    // 初始化共享作用域
    const container = window[config.scope];

    if (!container) {
      reject(new Error(`Container ${config.scope} not found`));
      return;
    }

    // 初始化共享依赖
    container.init(__webpack_share_scopes__.default);

    // 加载模块
    const factory = container.get(config.module);

    factory.then((module) => {
      const Module = module();
      const Component = Module.default;
      resolve(Component);
    }).catch(reject);
  });
}
```

```tsx
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { documentApi } from '@/services/api';

/**
 * 文档查询 Hook
 * 自动处理缓存、重新验证、后台更新
 */
export function useDocuments() {
  return useQuery({
    queryKey: ['documents'],
    queryFn: () => documentApi.getAll(),

    // 缓存配置
    staleTime: 5 * 60 * 1000,    // 5分钟内数据视为新鲜
    cacheTime: 30 * 60 * 1000,   // 30分钟后清除缓存

    // 自动重新验证
    refetchOnWindowFocus: true,   // 窗口聚焦时重新验证
    refetchOnReconnect: true,     // 网络重连时重新验证

    // 重试配置
    retry: 3,
    retryDelay: (attemptIndex) => Math.min(1000 * 2 ** attemptIndex, 30000),
  });
}

/**
 * 创建文档 Mutation
 */
export function useCreateDocument() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: documentApi.create,

    // 乐观更新
    onMutate: async (newDocument) => {
      // 取消正在进行的查询
      await queryClient.cancelQueries({ queryKey: ['documents'] });

      // 保存旧数据
      const previousDocuments = queryClient.getQueryData(['documents']);

      // 乐观更新
      queryClient.setQueryData(['documents'], (old: any) => [
        ...old,
        { id: Date.now(), ...newDocument, status: 'creating' },
      ]);

      return { previousDocuments };
    },

    // 错误时回滚
    onError: (err, newDocument, context) => {
      queryClient.setQueryData(['documents'], context.previousDocuments);
    },

    // 成功时重新获取
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['documents'] });
    },
  });
}
```

#### 领域层：订单实体完整实现

```typescript
// 领域层 - 订单实体
// 文件路径：src/domain/entities/Order.ts

import { DomainEvents } from '../events/DomainEvents';
import { OrderPaidEvent } from '../events/OrderPaidEvent';

// 值对象：订单状态
export enum OrderStatus {
  PENDING = 'PENDING',
  PAID = 'PAID',
  SHIPPING = 'SHIPPING',
  DELIVERED = 'DELIVERED',
  CANCELLED = 'CANCELLED',
  REFUNDING = 'REFUNDING',
  REFUNDED = 'REFUNDED',
}

// 值对象：订单商品
export class OrderItem {
  private constructor(
    public readonly productId: string,
    public readonly productName: string,
    public readonly quantity: number,
    public readonly unitPrice: number,
    public readonly discount: number
  ) {
    this.validate();
  }

  // 验证逻辑
  private validate(): void {
    if (this.quantity <= 0) {
      throw new Error('商品数量必须大于0');
    }
    if (this.unitPrice < 0) {
      throw new Error('商品单价不能为负数');
    }
    if (this.discount < 0) {
      throw new Error('折扣金额不能为负数');
    }
  }

  // 计算小计
  get subtotal(): number {
    return this.unitPrice * this.quantity - this.discount;
  }

  // 工厂方法
  static create(data: {
    productId: string;
    productName: string;
    quantity: number;
    unitPrice: number;
    discount?: number;
  }): OrderItem {
    return new OrderItem(
      data.productId,
      data.productName,
      data.quantity,
      data.unitPrice,
      data.discount || 0
    );
  }
}

/**
 * 订单实体 - 聚合根
 * 封装订单完整的业务逻辑和状态流转
 */
export class Order {
  // 私有属性，确保数据一致性
  private _id: string;
  private _userId: string;
  private _status: OrderStatus;
  private _items: OrderItem[];
  private _totalAmount: number;
  private _paidAmount: number;
  private _createdAt: Date;
  private _updatedAt: Date;
  private _paidAt?: Date;
  private _shippedAt?: Date;
  private _deliveredAt?: Date;
  private _cancelledAt?: Date;
  private _cancelReason?: string;

  private constructor(params: {
    id: string;
    userId: string;
    items: OrderItem[];
  }) {
    this._id = params.id;
    this._userId = params.userId;
    this._items = params.items;
    this._status = OrderStatus.PENDING;
    this._totalAmount = this.calculateTotalAmount();
    this._paidAmount = 0;
    this._createdAt = new Date();
    this._updatedAt = new Date();
  }

  // 工厂方法：创建订单
  static create(params: {
    id: string;
    userId: string;
    items: OrderItem[];
  }): Order {
    if (params.items.length === 0) {
      throw new Error('订单必须包含至少一个商品');
    }

    const order = new Order(params);
    return order;
  }

  // 计算订单总金额
  private calculateTotalAmount(): number {
    return this._items.reduce((sum, item) => sum + item.subtotal, 0);
  }

  // 领域方法：支付订单
  pay(paymentAmount: number, paymentMethod: string): void {
    // 业务规则：只有待支付状态才能支付
    if (this._status !== OrderStatus.PENDING) {
      throw new Error('只有待支付的订单才能支付');
    }

    // 业务规则：支付金额必须等于订单金额
    if (paymentAmount < this._totalAmount) {
      throw new Error('支付金额不足');
    }

    // 更新状态
    this._status = OrderStatus.PAID;
    this._paidAmount = paymentAmount;
    this._paidAt = new Date();
    this._updatedAt = new Date();

    // 发布领域事件
    DomainEvents.publish(
      new OrderPaidEvent({
        orderId: this._id,
        userId: this._userId,
        amount: paymentAmount,
        paymentMethod,
        paidAt: this._paidAt,
      })
    );
  }

  // 领域方法：取消订单
  cancel(reason: string): void {
    // 业务规则：已发货或已送达的订单不能直接取消
    if (
      this._status === OrderStatus.SHIPPING ||
      this._status === OrderStatus.DELIVERED
    ) {
      throw new Error('订单已发货或已送达，无法直接取消');
    }

    // 业务规则：已支付的订单取消需要退款
    if (this._status === OrderStatus.PAID) {
      this._status = OrderStatus.REFUNDING;
    } else {
      this._status = OrderStatus.CANCELLED;
    }

    this._cancelReason = reason;
    this._cancelledAt = new Date();
    this._updatedAt = new Date();
  }

  // 领域方法：发货
  ship(trackingNumber: string): void {
    // 业务规则：只有已支付的订单才能发货
    if (this._status !== OrderStatus.PAID) {
      throw new Error('只有已支付的订单才能发货');
    }

    this._status = OrderStatus.SHIPPING;
    this._shippedAt = new Date();
    this._updatedAt = new Date();
  }

  // 领域方法：确认收货
  deliver(): void {
    // 业务规则：只有发货中的订单才能确认收货
    if (this._status !== OrderStatus.SHIPPING) {
      throw new Error('只有发货中的订单才能确认收货');
    }

    this._status = OrderStatus.DELIVERED;
    this._deliveredAt = new Date();
    this._updatedAt = new Date();
  }

  // 领域方法：完成退款
  completeRefund(): void {
    // 业务规则：只有退款中的订单才能完成退款
    if (this._status !== OrderStatus.REFUNDING) {
      throw new Error('只有退款中的订单才能完成退款');
    }

    this._status = OrderStatus.REFUNDED;
    this._updatedAt = new Date();
  }

  // 获取方法（只读访问）
  get id(): string {
    return this._id;
  }

  get userId(): string {
    return this._userId;
  }

  get status(): OrderStatus {
    return this._status;
  }

  get totalAmount(): number {
    return this._totalAmount;
  }

  get items(): OrderItem[] {
    return [...this._items]; // 返回副本，防止外部修改
  }

  // 状态判断方法
  canCancel(): boolean {
    return (
      this._status === OrderStatus.PENDING ||
      this._status === OrderStatus.PAID
    );
  }

  canRefund(): boolean {
    return (
      this._status === OrderStatus.PAID ||
      this._status === OrderStatus.SHIPPING
    );
  }

  canShip(): boolean {
    return this._status === OrderStatus.PAID;
  }

  canDeliver(): boolean {
    return this._status === OrderStatus.SHIPPING;
  }
}
```

```typescript
   * 支付订单 - 完整业务流程
   */
  async payOrder(dto: PayOrderDTO): Promise<OrderDTO> {
    // 1. 获取订单
    const order = await this.orderRepository.findById(dto.orderId);

    if (!order) {
      throw new Error('订单不存在');
    }

    // 2. 执行支付
    const paymentResult = await this.paymentService.processPayment({
      orderId: order.id,
      amount: order.totalAmount,
      method: dto.paymentMethod,
    });

    if (!paymentResult.success) {
      throw new Error(`支付失败: ${paymentResult.errorMessage}`);
    }

    // 3. 更新订单状态
    order.pay(paymentResult.amount, dto.paymentMethod);

    // 4. 预留库存
    await this.inventoryService.reserveStock(order.items);

    // 5. 保存订单
    await this.orderRepository.save(order);

    await this.auditLogService.log('ORDER_PAID', {
      orderId: order.id,
      paymentId: paymentResult.paymentId,
      amount: paymentResult.amount,
    });

    await this.notificationService.sendOrderPaid(order);

    return this.toDTO(order);

        quantity: item.quantity,

        subtotal: item.subtotal,
```

# 第十三章：边缘计算架构

## 13.1 边缘计算概述

### 什么是边缘计算？

边缘计算是一种分布式计算范式，将计算、存储和网络服务推向网络的"边缘"，即靠近数据源和用户的位置。对于前端架构而言，边缘计算意味着：

- **降低延迟**：请求无需回源站，全球 < 50ms 响应
- **减轻源站负载**：静态资源和可缓存内容在边缘处理
- **增强安全性**：DDoS防护、WAF等安全能力前置
- **AI 推理加速**：在边缘运行机器学习模型

### 边缘计算 vs 传统架构

```
┌─────────────────────────────────────────────────────────┐
│                    传统架构                               │
│                                                         │
│  User → Internet → CDN → Origin → Database            │
│          (慢)      (缓存)  (瓶颈)                       │
└─────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────┐
│                    边缘架构                               │
│                                                         │
│  User → Edge Node → Edge Node → Regional → Origin       │
│          (快)      (智能路由)   (缓存)    (按需)         │
└─────────────────────────────────────────────────────────┘
```

```typescript
export function middleware(request: NextRequest) {

  const bucket = getABBucket(request);

  // 基于 Cookie 或 IP 分配

  return Math.random() > 0.5 ? 'B' : 'A';
```

```rust
// src/lib.rs
use wasm_bindgen::prelude::*;
use image::{DynamicImage, ImageFormat, ImageBuffer, Rgba};
use std::io::Cursor;

/// 图像处理 WASM 模块
#[wasm_bindgen]
pub struct ImageProcessor {
    width: u32,
    height: u32,
    data: Vec<u8>,
}

#[wasm_bindgen]
impl ImageProcessor {
    /// 创建处理器
    #[wasm_bindgen(constructor)]
    pub fn new() -> Self {
        Self {
            width: 0,
            height: 0,
            data: Vec::new(),
        }
    }

    /// 从 Base64 加载图片
    #[wasm_bindgen]
    pub fn load_from_base64(&mut self, base64_data: &str) -> Result<(), JsValue> {
        // 移除 data URI 前缀
        let data = if base64_data.starts_with("data:image") {
            &base64_data[base64_data.find(',').unwrap_or(0) + 1..]
        } else {
            base64_data
        };

        // 解码 Base64
        let decoded = base64_decode(data)
            .map_err(|e| JsValue::from_str(&format!("解码失败: {}", e)))?;

        // 解码图片
        let img = image::load_from_memory(&decoded)
            .map_err(|e| JsValue::from_str(&format!("图片格式错误: {}", e)))?;

        self.width = img.width();
        self.height = img.height();
        self.data = img.to_rgba8().into_raw();

        Ok(())
    }

    /// 灰度转换
    #[wasm_bindgen]
    pub fn to_grayscale(&mut self) {
        let img = ImageBuffer::<Rgba<u8>, Vec<u8>>::from_raw(
            self.width, self.height, self.data.clone()
        ).unwrap();

        for pixel in img.pixels() {
            let r = pixel[0] as f32;
            let g = pixel[1] as f32;
            let b = pixel[2] as f32;

            // 使用 ITU-R BT.601 加权
            let gray = (0.299 * r + 0.587 * g + 0.114 * b) as u8;

            self.data[(pixel[0] as usize) * 4] = gray;
            self.data[(pixel[0] as usize) * 4 + 1] = gray;
            self.data[(pixel[0] as usize) * 4 + 2] = gray;
        }
    }

    /// 模糊处理
    #[wasm_bindgen]
    pub fn blur(&mut self, radius: u32) {
        // Box blur 算法
        let kernel_size = (radius * 2 + 1) as usize;
        let mut temp = self.data.clone();

        for y in 0..self.height {
            for x in 0..self.width {
                let mut r_sum = 0u64;
                let mut g_sum = 0u64;
                let mut b_sum = 0u64;
                let mut count = 0u64;

                for ky in 0..kernel_size {
                    for kx in 0..kernel_size {
                        let px = (x + kx as u32).saturating_sub(radius);
                        let py = (y + ky as u32).saturating_sub(radius);

                        if px < self.width && py < self.height {
                            let idx = ((py * self.width + px) * 4) as usize;
                            r_sum += self.data[idx] as u64;
                            g_sum += self.data[idx + 1] as u64;
                            b_sum += self.data[idx + 2] as u64;
                            count += 1;
                        }
                    }
                }

                let idx = ((y * self.width + x) * 4) as usize;
                temp[idx] = (r_sum / count) as u8;
                temp[idx + 1] = (g_sum / count) as u8;
                temp[idx + 2] = (b_sum / count) as u8;
            }
        }

        self.data = temp;
    }

    /// 导出为 PNG Base64
    #[wasm_bindgen]
    pub fn export_as_base64(&self) -> String {
        let img = ImageBuffer::<Rgba<u8>, Vec<u8>>::from_raw(
            self.width, self.height, self.data.clone()
        ).unwrap();

        let mut buffer = Cursor::new(Vec::new());
        img.write_to(&mut buffer, ImageFormat::Png).unwrap();

        let base64 = base64_encode(buffer.into_inner());
        format!("data:image/png;base64,{}", base64)
    }
}

// Base64 辅助函数
fn base64_decode(input: &str) -> Result<Vec<u8>, &'static str> {
    // 简化的 Base64 解码
    // 生产环境使用 base64 crate
    Ok(input.as_bytes().to_vec())
}

fn base64_encode(data: Vec<u8>) -> String {
    // 简化的 Base64 编码
    // 生产环境使用 base64 crate
    use std::io::Write;
    const CHARS: &[u8] = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";

    let mut result = String::new();
    for chunk in data.chunks(3) {
        let b = match chunk.len() {
            1 => [chunk[0], 0, 0],
            2 => [chunk[0], chunk[1], 0],
            _ => [chunk[0], chunk[1], chunk[2]],
        };

        result.push(CHARS[(b[0] >> 2) as usize] as char);
        result.push(CHARS[((b[0] & 0x03) << 4 | b[1] >> 4) as usize] as char);
        if chunk.len() > 1 {
            result.push(CHARS[((b[1] & 0x0f) << 2 | b[2] >> 6) as usize] as char);
        }
        if chunk.len() > 2 {
            result.push(CHARS[(b[2] & 0x3f) as usize] as char);
        }
    }

    let padding = (3 - data.len() % 3) % 3;
    result.push_str(&"=".repeat(padding));

    result
}
```

```typescript
// 前端调用 WASM 模块
import init, { ImageProcessor } from './pkg/image_processor.js';

// 初始化 WASM 模块
await init();

class ImageProcessorService {
  private processor: ImageProcessor;

  constructor() {
    this.processor = new ImageProcessor();
  }

  // 加载图片
  async loadImage(file: File): Promise<void> {
    const base64 = await this.fileToBase64(file);
    this.processor.load_from_base64(base64);
  }

  // 灰度转换
  grayscale(): void {
    this.processor.to_grayscale();
  }

  // 模糊处理
  blur(radius: number = 5): void {
    this.processor.blur(radius);
  }

  // 获取结果
  getResult(): string {
    return this.processor.export_as_base64();
  }

  // 文件转 Base64
  private fileToBase64(file: File): Promise<string> {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = () => resolve(reader.result as string);
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }
}

// React 组件集成
import { useState, useRef } from 'react';

export function ImageEditor() {
  const [imageUrl, setImageUrl] = useState<string>('');
  const [processing, setProcessing] = useState(false);
  const serviceRef = useRef<ImageProcessorService | null>(null);

  const initService = async () => {
    if (!serviceRef.current) {
      serviceRef.current = new ImageProcessorService();
    }
  };

    setImageUrl(serviceRef.current!.getResult());
    setProcessing(false);
  };

  return (

      <input type="file" onChange={handleFileChange} accept="image/*" />
```

```typescript
// GraphQL BFF 服务
import { ApolloServer } from '@apollo/server';
import { startStandaloneServer } from '@apollo/server/standalone';
import { RESTDataSource } from '@apollo/datasource-rest';

// 数据源定义
class UserService extends RESTDataSource {
  constructor() {
    super();
    this.baseURL = process.env.USER_SERVICE_URL;
  }

  async getUser(id: string) {
    return this.get(`/users/${id}`);
  }

  async getUserPreferences(userId: string) {
    return this.get(`/users/${userId}/preferences`);
  }
}

class ProductService extends RESTDataSource {
  constructor() {
    super();
    this.baseURL = process.env.PRODUCT_SERVICE_URL;
  }

  async getProduct(id: string) {
    return this.get(`/products/${id}`);
  }

  async getRecommendedProducts(category: string) {
    return this.get(`/products/recommendations?category=${category}`);
  }
}

// GraphQL 类型定义
const typeDefs = `#graphql
  type User {
    id: ID!
    name: String!
    email: String!
    avatar: String
    preferences: UserPreferences
  }

  type UserPreferences {
    theme: String!
    language: String!
    notifications: Boolean!
  }

  type Product {
    id: ID!
    name: String!
    description: String!
    price: Float!
    image: String
  }

  type DashboardData {
    user: User!
    recommendations: [Product!]!
    recentlyViewed: [Product!]!
  }

  type Query {
    # 获取用户仪表盘数据（聚合查询）
    dashboard(userId: ID!): DashboardData!

    # 获取单个产品
    product(id: ID!): Product

    # 搜索产品
    searchProducts(query: String!, limit: Int = 10): [Product!]!
  }
`;

// 解析器实现
const resolvers = {
  Query: {
    dashboard: async (_, { userId }, { dataSources }) => {
      // 并行获取数据
      const [user, preferences] = await Promise.all([
        dataSources.userService.getUser(userId),
        dataSources.userService.getUserPreferences(userId),
      ]);

      // 基于用户偏好获取推荐
      const recommendations = await dataSources.productService.getRecommendedProducts(
        preferences.favoriteCategory
      );

      return {
        user: { ...user, preferences },
        recommendations,
        recentlyViewed: [], // 简化
      };
    },

    product: (_, { id }, { dataSources }) => {
      return dataSources.productService.getProduct(id);
    },

    searchProducts: async (_, { query, limit }, { dataSources }) => {
      const products = await dataSources.productService.searchProducts(query);
      return products.slice(0, limit);
    },
  },
};

// 启动服务器
const server = new ApolloServer({
  typeDefs,
  resolvers,
});

const { url } = await startStandaloneServer(server, {
  context: async () => ({
    dataSources: {
      userService: new UserService(),
      productService: new ProductService(),
    },
  }),
});

console.log(`GraphQL BFF 运行于 ${url}`);
```

```typescript
class BFFError extends Error {
  constructor(
    message: string,
    public statusCode: number,
    public code: string,
    public details?: Record<string, unknown>
  ) {
    super(message);
    this.name = 'BFFError';
  }

async function bffHandler(event: APIGatewayEvent) {
  try {
    const data = await fetchData(event);
    return formatResponse(data);
  } catch (error) {
    if (error instanceof BFFError) {
      return formatError(error);
    }

    if (error instanceof TimeoutError) {
      return formatError(new BFFError(
        '请求超时',
        504,
        'TIMEOUT',
        { originalError: error.message }
      ));
    }

    if (error instanceof ValidationError) {
      return formatError(new BFFError(
        '数据验证失败',
        400,
        'VALIDATION_ERROR',
        { errors: error.errors }
      ));
    }

    // 未知错误
    console.error('未预期的 BFF 错误:', error);
    return formatError(new BFFError(
      '服务器内部错误',
      500,
      'INTERNAL_ERROR'
    ));
```

```typescript
      if (!response.ok) {

          contexts: {
            api: {
              url: response.url,
              status: response.status,
              statusText: response.statusText,

          },
```

## 18.1 React 18 并发特性深度分析

### 18.1.1 并发渲染架构

React 18 引入的并发渲染是一个根本性的架构变革，其核心是通过「可中断」的渲染机制，解决长期困扰 React 开发者的「阻塞渲染」问题。

**传统渲染 vs 并发渲染对比：**

```
传统渲染（React 17 及之前）：
用户交互 → 开始渲染（不可中断）→ 完成渲染 → 用户响应
         ↑ 100ms的渲染任务会阻塞所有交互

并发渲染（React 18）：
用户交互 → 开始渲染（可中断）→ [更高优先级交互] → 恢复渲染 → 完成
         ↑ 渲染可以被更高优先级的任务打断
```

**核心原理：**

```typescript
// React 18 并发模式的本质是「时间切片」
// 将渲染工作拆分成多个小任务，每个任务执行5ms
// 通过MessageChannel在浏览器空闲时执行

// scheduler包中的优先级调度
const MAX_INTERVAL = 5; // 每片最大执行时间

export const ImmediatePriority = 1;  // 同步优先级（如用户输入）

export const NormalPriority = 3;       // 正常（如数据获取）
export const LowPriority = 4;          // 低（如预加载）
export const IdlePriority = 5;         // 空闲（如后台计算）
```

```
├── 单一数据源
├── State 只读
├── 使用纯函数 Reducer 修改
└── 强调不可变性

Zustand 理念：「简洁性」
├── 极简 API 设计
├── 灵活的状态管理
├── 支持不可变或可变更新
└── 强调开发体验
```

```typescript
export const fetchUser = createAsyncThunk(

        state.loading = false;
        state.data = action.payload;
```

### 18.4.3 重构迁移指南

**从 Redux 迁移到 Zustand：**

```typescript
// 第一步：安装 Zustand

// 原 Redux Slice
const counterSlice = createSlice({
  name: 'counter',
  initialState: { count: 0 },
  reducers: {
    increment: (state) => { state.count += 1; },
    decrement: (state) => { state.count -= 1; },
    incrementBy: (state, action) => { state.count += action.payload; },
  },
});

// Zustand Store
const useCounterStore = create<CounterState>()(
  immer((set) => ({
    count: 0,
    increment: () => set((s) => { s.count += 1; }),
    decrement: () => set((s) => { s.count -= 1; }),
    incrementBy: (amount: number) => set((s) => { s.count += amount; }),
  }))
);

// 第三步：移除 Provider，修改组件
// Redux
import { Provider } from 'react-redux';

function App() {
  return (
    <Provider store={store}>
      <Counter />
    </Provider>
  );
}

function App() {
  return <Counter />; // 直接使用

  // Redux
```

## 18.5 DDD 六边形架构（端口和适配器）

### 18.5.1 架构核心概念

**六边形架构（Ports & Adapters）原理：**

```
                    ┌─────────────────────────────────┐
                    │         应用内核 (Core)           │
                    │  ┌─────────────────────────────┐  │
                    │  │      领域层 (Domain)         │  │
                    │  │  ┌───────────────────────┐  │  │
                    │  │  │   实体 (Entities)     │  │  │
                    │  │  │   值对象 (VO)         │  │  │
                    │  │  │   领域服务            │  │  │
                    │  │  │   聚合根 (Aggregate)  │  │  │
                    │  │  └───────────────────────┘  │  │
                    │  └─────────────────────────────┘  │
                    │  ┌─────────────────────────────┐  │
                    │  │    应用层 (Application)      │  │
                    │  │  ┌───────────────────────┐  │  │
                    │  │  │    用例 (Use Cases)   │  │  │
                    │  │  │    DTO / Mapper       │  │  │
                    │  │  └───────────────────────┘  │  │
                    │  └─────────────────────────────┘  │
                    └─────────────────────────────────┘
                                    ↑
                                    │ 端口 (Ports)
                    ┌───────────────┴───────────────┐
                    │                               │
              ┌─────▼─────┐                 ┌──────▼─────┐
              │  主端口    │                 │   次端口    │
              │ (Driving) │                 │ (Driven)   │
              └─────┬─────┘                 └──────┬─────┘
                    │                               │
              ┌─────▼─────┐                 ┌──────▼─────┐
              │  适配器    │                 │   适配器   │
              │ (HTTP/CLI)│                 │(DB/API/File)│
              └───────────┘                 └────────────┘
```

### 18.5.3 领域层实现

```typescript
// domain/entities/Order.ts
// 领域实体：包含核心业务规则

export enum OrderStatus {
  DRAFT = 'DRAFT',           // 草稿
  PENDING = 'PENDING',       // 待支付
  PAID = 'PAID',            // 已支付
  SHIPPING = 'SHIPPING',     // 配送中
  DELIVERED = 'DELIVERED',   // 已送达
  CANCELLED = 'CANCELLED',   // 已取消
}

export interface OrderItemProps {
  productId: string;
  productName: string;
  unitPrice: number;
  quantity: number;
}

export interface OrderProps {
  id?: string;
  customerId: string;
  items: OrderItemProps[];
  shippingAddress: Address;
  status?: OrderStatus;
  createdAt?: Date;
  updatedAt?: Date;
}

/**
 * 订单聚合根
 * 封装订单的所有业务规则
 */
export class Order {
  // 私有构造函数，通过工厂方法创建
  private constructor(private props: OrderProps) {}

  // 工厂方法：创建新订单
  static create(props: Omit<OrderProps, 'id' | 'status' | 'createdAt' | 'updatedAt'>): Order {
    // 业务规则验证
    if (!props.items || props.items.length === 0) {
      throw new Error('订单必须包含至少一个商品');
    }

    // 验证商品数量
    for (const item of props.items) {
      if (item.quantity <= 0) {
        throw new Error(`商品 ${item.productName} 数量必须大于0`);
      }
    }

    return new Order({
      ...props,
      id: generateId(),
      status: OrderStatus.DRAFT,
      createdAt: new Date(),
      updatedAt: new Date(),
    });
  }

  // 领域方法：提交订单
  submit(): void {
    if (this.props.status !== OrderStatus.DRAFT) {
      throw new Error('只能提交草稿状态的订单');
    }
    this.props.status = OrderStatus.PENDING;
    this.props.updatedAt = new Date();
  }

  // 领域方法：取消订单
  cancel(): void {
    // 只有特定状态可以取消
    const cancellable = [OrderStatus.DRAFT, OrderStatus.PENDING];
    if (!cancellable.includes(this.props.status)) {
      throw new Error('当前状态不允许取消订单');
    }
    this.props.status = OrderStatus.CANCELLED;
    this.props.updatedAt = new Date();
  }

  // 计算订单总价（领域逻辑）
  calculateTotal(): number {
    return this.props.items.reduce(
      (sum, item) => sum + item.unitPrice * item.quantity,
      0
    );
  }

  // Getter
  get id() { return this.props.id; }
  get status() { return this.props.status; }
  get total() { return this.calculateTotal(); }
  get items() { return [...this.props.items]; }
}
```

### 18.5.4 应用层实现

```typescript
// application/use-cases/CreateOrderUseCase.ts
// 用例：编排领域对象，实现业务流程

export class CreateOrderUseCase implements OrderInputPort {
  constructor(
    // 依赖倒置：通过接口注入
    private orderRepository: OrderRepositoryPort,
    private paymentPort: PaymentPort,
    private notificationPort: NotificationPort,
  ) {}

  async execute(input: CreateOrderInput): Promise<OrderOutput> {
    // 1. 创建订单实体（领域逻辑）
    const order = Order.create({
      customerId: input.customerId,
      items: input.items,
      shippingAddress: input.shippingAddress,
    });

    // 2. 处理支付（防腐层）
    const paymentResult = await this.paymentPort.processPayment(order);

    if (!paymentResult.success) {
      throw new Error('支付失败：' + paymentResult.errorMessage);
    }

    // 3. 持久化
    await this.orderRepository.save(order);

    // 4. 发送通知
    await this.notificationPort.sendOrderConfirmation(order);

    // 5. 返回 DTO
    return OrderMapper.toDTO(order);
  }
}

// DTO 映射器
class OrderMapper {
  static toDTO(order: Order): OrderOutput {
    return {
      id: order.id,
      customerId: order.props.customerId,
      items: order.props.items,
      total: order.total,
      status: order.status,
      createdAt: order.props.createdAt,
    };
  }
}
```
