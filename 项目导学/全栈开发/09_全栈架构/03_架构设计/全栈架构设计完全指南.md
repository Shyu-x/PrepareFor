## 1. 架构设计基础

### 1.1 架构设计原则

```
架构设计核心原则

1. 高内聚（High Cohesion）
   └── 相关功能聚集在一起

2. 低耦合（Low Coupling）
   └── 模块间依赖最小化

3. 单一职责（Single Responsibility）
   └── 每个模块只做一件事

4. 开闭原则（Open-Closed）
   └── 对扩展开放，对修改关闭

5. 依赖倒置（Dependency Inversion）
   └── 依赖抽象，不依赖具体实现

6. 接口隔离（Interface Segregation）
   └── 客户端不应该依赖不需要的接口

7. 最小知识原则（Law of Demeter）
   └── 只与直接朋友通信
```

### 1.2 架构层次

```
典型全栈应用架构

┌─────────────────────────────────────┐
│      表现层（Presentation）          │

              ↓

└─────────────────────────────────────┘
              ↓

└─────────────────────────────────────┘
```

### 1.3 架构模式对比

| 架构模式 | 优点 | 缺点 | 适用场景 |
|---------|------|------|----------|
| **单体架构** | 开发简单、部署简单 | 扩展性差、维护困难 | 小型项目 |
| **微服务架构** | 高扩展性、技术栈灵活 | 复杂度高、运维成本大 | 大型复杂项目 |
| **Serverless架构** | 零运维、按需付费 | 冷启动、调试困难 | 事件驱动应用 |
| **前后端分离** | 前后端独立开发、技术栈灵活 | 首屏加载慢 | 现代Web应用 |

```javascript
// src/services/UserService.js
const UserRepository = require('../repositories/UserRepository');
const { validateUser } = require('../utils/validator');

class UserService {
  async getUserById(id) {
    return await UserRepository.findById(id);
  }

  async createUser(userData) {
    // 验证数据
    const validatedData = validateUser(userData);

    // 检查用户是否已存在
    const existingUser = await UserRepository.findByEmail(validatedData.email);
    if (existingUser) {
      throw new Error('用户已存在');
    }

    // 创建用户
    return await UserRepository.create(validatedData);
  }

  async updateUser(id, userData) {
    const user = await UserRepository.findById(id);
    if (!user) {
      throw new Error('用户不存在');
    }

    return await UserRepository.update(id, userData);
  }
}

module.exports = new UserService();

// src/repositories/UserRepository.js
const User = require('../models/User');

class UserRepository {
  async findById(id) {
    return await User.findByPk(id);
  }

  async findByEmail(email) {
    return await User.findOne({ where: { email } });
  }

  async create(userData) {
    return await User.create(userData);
  }

  async update(id, userData) {
    const user = await User.findByPk(id);
    if (!user) {
      throw new Error('用户不存在');
    }
    return await user.update(userData);
  }

  async delete(id) {
    const user = await User.findByPk(id);
    if (!user) {
      throw new Error('用户不存在');
    }
    await user.destroy();
  }
}
```

```javascript
// shared/circuit-breaker/CircuitBreaker.js
class CircuitBreaker {
  constructor(options = {}) {
    this.threshold = options.threshold || 5;  // 失败阈值
    this.timeout = options.timeout || 60000;  // 超时时间（毫秒）
    this.failures = 0;
    this.state = 'CLOSED';  // CLOSED, OPEN, HALF_OPEN
    this.lastFailureTime = null;
  }

  async execute(fn) {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.timeout) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('熔断器打开中，拒绝请求');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  onSuccess() {
    this.failures = 0;
    if (this.state === 'HALF_OPEN') {
      this.state = 'CLOSED';
    }
  }

  onFailure() {
    this.failures++;
    this.lastFailureTime = Date.now();

    if (this.failures >= this.threshold) {
      this.state = 'OPEN';
    }
  }
}

// 使用熔断器
const circuitBreaker = new CircuitBreaker({
  threshold: 5,
  timeout: 60000,
});

async function callExternalService() {
  return await circuitBreaker.execute(async () => {
    const response = await axios.get('http://external-service/api/data');
    return response.data;
  });
}
```

```javascript
// services/order-service/src/services/OrderService.js
class OrderService {
  async createOrder(orderData) {
    const order = await OrderRepository.create(orderData);

    // Saga步骤1：创建订单
    try {
      // 发布订单创建事件
      await rabbitmq.publish('order.created', {
        orderId: order.id,
        userId: order.userId,
        total: order.total,
      });

      // 等待支付服务确认
      const paymentResult = await this.waitForPaymentConfirmation(order.id);

      if (!paymentResult.success) {
        // 回滚：删除订单
        await OrderRepository.delete(order.id);
        throw new Error('支付失败，订单已取消');
      }

      // 发布订单完成事件
      await rabbitmq.publish('order.completed', {
        orderId: order.id,
        userId: order.userId,
      });

      return order;
    } catch (error) {
      // 回滚：删除订单
      await OrderRepository.delete(order.id);
      throw error;
    }
  }

  async waitForPaymentConfirmation(orderId) {
    return new Promise((resolve, reject) => {
      const timeout = setTimeout(() => {
        reject(new Error('支付超时'));
      }, 30000);

      rabbitmq.consume(`payment.order.${orderId}`, (message) => {
        clearTimeout(timeout);
        const result = JSON.parse(message.content.toString());
        resolve(result);
      });
    });
  }
}

// 3. 事件溯源（Event Sourcing）
// shared/events/OrderEventStore.js
const { EventStoreDBClient } = require('@eventstoredb/eventstoredb');

const client = new EventStoreDBClient({
  endpoint: 'http://eventstoredb:2113',
});

class OrderEventStore {
  static async appendEvents(orderId, events) {
    const streamName = `order-${orderId}`;

    for (const event of events) {
      await client.appendToStream(streamName, {
        type: event.type,
        data: JSON.stringify(event.data),
      });
    }
  }

  static async getEvents(orderId) {
    const streamName = `order-${orderId}`;
    const events = [];

    for await (const event of client.readStream(streamName)) {
      events.push({
        type: event.event.type,
        data: JSON.parse(event.event.data.toString()),
      });
    }

    return events;
  }

  static async rebuildOrder(orderId) {
    const events = await this.getEvents(orderId);
    let order = null;

    for (const event of events) {
      switch (event.type) {
        case 'OrderCreated':
          order = event.data;
          break;
        case 'OrderPaid':
          order.status = 'PAID';
          break;
        case 'OrderShipped':
          order.status = 'SHIPPED';
          break;
        case 'OrderDelivered':
          order.status = 'DELIVERED';
          break;
      }
    }

    return order;
  }
}

module.exports = OrderEventStore;
```

```javascript
import { getToken, setToken, removeToken } from './storage';

export const authService = {
  // 保存token
  saveToken: (token) => {
    setToken(token);
  },

  // 获取token
  getToken: () => {
    return getToken();
  },

  // 移除token
  removeToken: () => {
    removeToken();
  },

  // 检查是否已认证
  isAuthenticated: () => {
    return !!getToken();
  },

  // 解码token
  decodeToken: (token) => {
    const base64Url = token.split('.')[1];
    const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
    const jsonPayload = decodeURIComponent(
      atob(base64)
        .split('')
        .map((c) => '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2))
        .join('')
    );
    return JSON.parse(jsonPayload);
  },

  // 检查token是否过期
  isTokenExpired: (token) => {
    try {
      const decoded = authService.decodeToken(token);
      const now = Date.now() / 1000;
      return decoded.exp < now;
    } catch (error) {
      return true;
    }
  },

  // 刷新token
  refreshToken: async () => {
    try {
      const token = getToken();
      const response = await fetch('/api/auth/refresh', {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${token}`,
        },
      });
      const data = await response.json();
      setToken(data.token);
      return data.token;
    } catch (error) {
      removeToken();
      throw error;
    }
  },
};
```

```javascript
// 1. 重试机制
class Retry {
  constructor(options = {}) {
    this.maxAttempts = options.maxAttempts || 3;
    this.delay = options.delay || 1000;
  }

  async execute(fn) {
    let lastError;

    for (let attempt = 1; attempt <= this.maxAttempts; attempt++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;

        if (attempt < this.maxAttempts) {
          await this.sleep(this.delay * attempt);
        }
      }
    }

    throw lastError;
  }

  sleep(ms) {
    return new Promise((resolve) => setTimeout(resolve, ms));
  }
}

// 3. 限流
class RateLimiter {
  constructor(options = {}) {
    this.limit = options.limit || 100;  // 每秒请求数
    this.tokens = this.limit;
    this.interval = setInterval(() => {
      this.tokens = this.limit;
    }, 1000);
  }

  async acquire() {
    while (this.tokens <= 0) {
      await this.sleep(10);
    }

    this.tokens--;
  }

  sleep(ms) {
    return new Promise((resolve) => setTimeout(resolve, ms));
  }
}

// 使用限流
const rateLimiter = new RateLimiter({ limit: 100 });

await rateLimiter.acquire();
await makeRequest();

// 4. 降级策略
class Fallback {
  static async execute(primaryFn, fallbackFn) {
    try {
      return await primaryFn();
    } catch (error) {
      console.error('主逻辑失败，使用降级策略:', error);
      return await fallbackFn();
    }
  }
}

// 使用降级
const result = await Fallback.execute(
  () => fetchFromAPI(),  // 主逻辑
  () => fetchFromCache()  // 降级逻辑
);
```
