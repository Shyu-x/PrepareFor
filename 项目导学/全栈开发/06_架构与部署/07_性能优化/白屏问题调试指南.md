## 1. 白屏问题概述

### 1.1 什么是白屏问题？

白屏问题是指用户访问网页时，页面长时间显示空白，无法看到任何内容的现象。这通常发生在：
- 页面加载初期
- JavaScript执行失败
- CSS加载阻塞
- 资源加载超时

### 2.1 关键渲染路径分析

```javascript
// 分析关键渲染路径

function analyzeCriticalPath() {
  const timing = performance.getEntriesByType('navigation')[0];

  const analysis = {
    // DNS 查询时间
    dns: timing.domainLookupEnd - timing.domainLookupStart,

    // TCP 连接时间
    tcp: timing.connectEnd - timing.connectStart,

    // SSL 握手时间
    ssl: timing.secureConnectionStart > 0
      ? timing.connectEnd - timing.secureConnectionStart
      : 0,

    // TTFB（首字节时间）
    ttfb: timing.responseStart - timing.requestStart,

    // 内容下载时间
    download: timing.responseEnd - timing.responseStart,

    // DOM 解析时间
    domParse: timing.domInteractive - timing.responseEnd,

    // 资源加载时间
    resourceLoad: timing.loadEventStart - timing.domContentLoadedEventEnd,

    // 总加载时间
    total: timing.loadEventEnd - timing.fetchStart,
  };

  // 找出瓶颈
  const maxTime = Math.max(...Object.values(analysis));
  const bottleneck = Object.keys(analysis).find(
    key => analysis[key] === maxTime
  );

  console.log('性能分析:', analysis);
  console.log('瓶颈:', bottleneck, maxTime, 'ms');

  return analysis;
}

// 分析资源加载
function analyzeResources() {
  const resources = performance.getEntriesByType('resource');

  // 按类型分组
  const byType = resources.reduce((acc, resource) => {
    const type = getResourceType(resource.initiatorType);
    if (!acc[type]) {
      acc[type] = [];
    }
    acc[type].push({
      name: resource.name,
      duration: resource.duration,
      size: resource.transferSize,
    });
    return acc;
  }, {});

  // 找出慢资源
  const slowResources = resources
    .filter(r => r.duration > 1000)
    .sort((a, b) => b.duration - a.duration)
    .slice(0, 10);

  console.log('慢资源:', slowResources);

  return { byType, slowResources };
}

function getResourceType(initiatorType) {
  const typeMap = {
    script: 'js',
    link: 'css',
    img: 'image',
    fetch: 'api',
    xmlhttprequest: 'api',
  };
  return typeMap[initiatorType] || 'other';
}
```

```javascript
    task.attribution.forEach(attr => {

        acc[source] = 0;
      }
      acc[source] += task.duration;
    });
```

```javascript
  });

  console.log('阻塞资源:', blockingResources);

  return blockingResources;
}
```

```javascript
// 1. 未捕获的同步错误

window.onerror = function(message, source, lineno, colno, error) {

  // 上报错误
  reportError({ message, source, lineno, colno, error });
  // 显示错误页面

  return true; // 阻止默认错误处理

// 2. 未捕获的Promise错误
fetch('/api/data')
  .then(response => response.json())

  .then(data => console.log(data));

window.addEventListener('unhandledrejection', (event) => {
  console.error('未处理的Promise拒绝:', event.reason);
  event.preventDefault();
  // 上报错误
  reportError(event.reason);
  // 显示错误页面

});

// 3. React错误边界
class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }

  componentDidCatch(error, errorInfo) {
    console.error('React错误:', error, errorInfo);
    reportError({ error, errorInfo });
  }

    if (this.state.hasError) {
      return (
        <div className="error-page">
          <h1>出错了</h1>
          <p>{this.state.error.message}</p>
          <button onClick={() => window.location.reload()}>
            刷新页面
          </button>
        </div>
      );
    }

    return this.props.children;
  }
}

    <ErrorBoundary>
      <MainContent />
    </ErrorBoundary>
  );
```

## 5. 白屏监控方案

### 5.1 前端监控SDK

```javascript
// 白屏监控SDK实现

class WhiteScreenMonitor {
  constructor(options = {}) {
    this.options = {
      threshold: 3000, // 白屏阈值（毫秒）
      checkInterval: 500, // 检查间隔
      maxChecks: 10, // 最大检查次数
      ...options,
    };

    this.startTime = Date.now();
    this.checkCount = 0;
    this.isWhiteScreen = false;
  }

  // 开始监控
  start() {
    // 监听FCP
    this.observeFCP();

    // 定期检查
    this.checkTimer = setInterval(() => {
      this.check();
    }, this.options.checkInterval);

    // 页面加载完成后停止检查
    window.addEventListener('load', () => {
      setTimeout(() => this.stop(), 1000);
    });
  }

  // 观察FCP
  observeFCP() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.name === 'first-contentful-paint') {
          const fcpTime = entry.startTime;
          if (fcpTime > this.options.threshold) {
            this.report('fcp_slow', { fcpTime });
          }
        }
      }
    });

    observer.observe({ entryTypes: ['paint'] });
  }

  // 检查白屏
  check() {
    this.checkCount++;

    // 检查是否有内容
    const hasContent = this.checkContent();

    if (!hasContent && this.checkCount >= this.options.maxChecks) {
      this.isWhiteScreen = true;
      this.report('white_screen', {
        duration: Date.now() - this.startTime,
        checkCount: this.checkCount,
      });
      this.stop();
    }
  }

  // 检查页面内容
  checkContent() {
    // 方法1：检查关键元素
    const criticalElements = document.querySelectorAll(
      'main, #root, .app, [data-content]'
    );
    if (criticalElements.length > 0) {
      for (const el of criticalElements) {
        if (el.offsetHeight > 0 && el.offsetWidth > 0) {
          return true;
        }
      }
    }

    // 方法2：检查body内容
    const body = document.body;
    if (body && body.innerHTML.length > 100) {
      return true;
    }

    // 方法3：像素采样
    return this.pixelSampling();
  }

  // 像素采样检测
  pixelSampling() {
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;

    try {
      // 截图当前页面
      ctx.drawWindow(window, 0, 0, canvas.width, canvas.height, 'white');

      // 采样检测
      const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
      const data = imageData.data;

      let nonWhitePixels = 0;
      const sampleRate = 100; // 每100像素采样一次

      for (let i = 0; i < data.length; i += 4 * sampleRate) {
        const r = data[i];
        const g = data[i + 1];
        const b = data[i + 2];

        // 检查是否为白色
        if (r < 250 || g < 250 || b < 250) {
          nonWhitePixels++;
        }
      }

      return nonWhitePixels > 10; // 至少有10个非白色像素
    } catch (e) {
      // 跨域限制，无法截图
      return false;
    }
  }

  // 上报
  report(type, data) {
    const reportData = {
      type,
      url: window.location.href,
      userAgent: navigator.userAgent,
      timestamp: Date.now(),
      ...data,
    };

    console.log('白屏监控上报:', reportData);

    // 发送到监控服务
    if (navigator.sendBeacon) {
      navigator.sendBeacon('/api/monitor', JSON.stringify(reportData));
    } else {
      fetch('/api/monitor', {
        method: 'POST',
        body: JSON.stringify(reportData),
      });
    }
  }

  // 停止监控
  stop() {
    if (this.checkTimer) {
      clearInterval(this.checkTimer);
      this.checkTimer = null;
    }
  }
}

// 使用
const monitor = new WhiteScreenMonitor({
  threshold: 3000,
  checkInterval: 500,
});

monitor.start();
```

```javascript
    getFID(this.onMetric.bind(this));
    getFCP(this.onMetric.bind(this));
    getLCP(this.onMetric.bind(this));
    getTTFB(this.onMetric.bind(this));

    // 页面卸载时上报
    window.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') {
        this.sendReport();
      }
    });
  }

  onMetric(metric) {
    this.metrics[metric.name] = {
      value: metric.value,
      id: metric.id,
      delta: metric.delta,
    };

    // 检查是否超标
    this.checkThreshold(metric);
  }

  checkThreshold(metric) {
    const thresholds = {
      LCP: 2500,
      FID: 100,
      CLS: 0.1,
      FCP: 1800,
      TTFB: 600,
    };

    const threshold = thresholds[metric.name];
    if (threshold && metric.value > threshold) {
      console.warn(`${metric.name} 超标: ${metric.value} > ${threshold}`);
    }
  }

  sendReport() {
    const data = {
      url: window.location.href,
      metrics: this.metrics,
      timestamp: Date.now(),
      userAgent: navigator.userAgent,
      connection: navigator.connection?.effectiveType,
      memory: performance.memory?.usedJSHeapSize,
    };

    // 使用 sendBeacon 确保发送
    navigator.sendBeacon(
      '/api/performance',
      JSON.stringify(data)
    );
  }
}

// 初始化
new PerformanceReporter();
```

## 6. 面试高频问题

### 问题1：如何定位白屏问题？

**答案：**
1. 使用Chrome DevTools Performance面板分析
2. 检查Network面板资源加载情况
3. 查看Console是否有JavaScript错误
4. 分析关键渲染路径
5. 检查FCP/LCP等性能指标

### 问题2：白屏时间如何计算？

**答案：**
- 白屏时间 = domLoading - fetchStart
- 或使用FCP（首次内容绘制）时间
- 使用Performance API获取精确数据

### 问题3：如何优化白屏时间？

**答案：**
1. SSR服务端渲染
2. 骨架屏/Loading占位
3. 代码分割和懒加载
4. 关键CSS内联
5. 资源预加载
6. CDN加速

### 问题4：什么是FCP和LCP？

**答案：**
- **FCP（First Contentful Paint）**：首次内容绘制，页面首次显示任何内容的时间
- **LCP（Largest Contentful Paint）**：最大内容绘制，页面主要内容渲染完成的时间

### 问题5：如何监控线上白屏问题？

**答案：**
1. 使用PerformanceObserver监听性能指标
2. 实现像素采样检测白屏
3. 全局错误捕获
4. 资源加载失败监控
5. 上报到监控系统
