# 安全运维与等保合规

## 前言：安全运维不是选修课

想象你开了一家便利店。刚开始只有你一个人看店，防盗措施可能很简单——有监控、有锁就行了。但随着店越开越大、越来越值钱，你就需要更专业的安保：保安、保险柜、报警系统、防盗门，甚至需要和警方联网。

软件系统也一样。系统小的时候，黑客可能懒得攻击你。但一旦系统有价值、有用户数据，你就成了目标。而且，在中国做企业，安全合规不是"想做就做，不想做就算了"的事——有些行业、有些数据，是法律强制要求保护的。

这一章，我们从安全运维讲起，涵盖安全扫描、渗透测试、等保2.0合规、以及数据安全保护。内容会比较严肃，因为安全无小事。

// 要求：使用多因素认证，强密码策略

// 实现1：多因素认证（MFA）
class MultiFactorAuth {
  async authenticate(userId, credentials) {
    // 第一因素：密码
    const passwordValid = await this.verifyPassword(
      userId,
      credentials.password
    );
    if (!passwordValid) {
      throw new AuthError('密码错误');
    }

    // 第二因素：短信验证码
    const smsCode = credentials.smsCode;
    const smsValid = await this.verifySmsCode(userId, smsCode);
    if (!smsValid) {
      throw new AuthError('验证码错误或已过期');
    }

    // 第三因素：TOTP（可选，高安全场景）
    if (credentials.totp) {
      const totpValid = await this.verifyTotp(userId, credentials.totp);
      if (!totpValid) {
        throw new AuthError('TOTP验证失败');
      }
    }

    return this.createSession(userId);
  }
}

    const iv = crypto.randomBytes(12);
    const cipher = crypto.createCipheriv('aes-256-gcm', key, iv);

    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');

      iv: iv.toString('hex'),

// 手机号脱敏
function maskPhoneNumber(phone) {
  if (!phone || phone.length < 7) return phone;
  return phone.slice(0, 3) + '****' + phone.slice(-4);
}

// 邮箱脱敏
function maskEmail(email) {
  if (!email) return email;
  const [name, domain] = email.split('@');
  if (!name || !domain) return email;
  const maskedName = name.slice(0, 2) + '***';
  return maskedName + '@' + domain;
}

// 身份证号脱敏
function maskIdCard(idCard) {
  if (!idCard || idCard.length < 10) return idCard;
  return idCard.slice(0, 6) + '********' + idCard.slice(-4);
}

// 银行卡号脱敏
function maskBankCard(cardNumber) {
  if (!cardNumber || cardNumber.length < 8) return cardNumber;
  return cardNumber.slice(0, 6) + '********' + cardNumber.slice(-4);
}

// 姓名脱敏
function maskName(name, type = 'middle') {
  if (!name) return name;

  switch (type) {
    case 'full':
      // 全脱敏：张三 → *
      return '*'.repeat(name.length);

    case 'middle':
      // 中间脱敏（默认）：张三 → 张*
      return name.slice(0, 1) + '*';

    case 'first':
      // 只保留姓氏：张三丰 → 张*丰
      if (name.length === 2) {
        return name.slice(0, 1) + '*';
      }
      return name.slice(0, 1) + '*' + name.slice(-1);

    default:
      return name;
  }
}

// 批量脱敏对象
function maskPersonalData(data) {
  return {
    // 一般信息
    name: maskName(data.name),
    phone: maskPhoneNumber(data.phone),
    email: maskEmail(data.email),

    idCard: maskIdCard(data.idCard),
    bankCard: maskBankCard(data.bankCard),

}

// API响应脱敏中间件
function maskResponseMiddleware(req, res, next) {
  // 需要脱敏的字段映射
  const fieldMasking = {
    'phone': (v) => maskPhoneNumber(v),
    'email': (v) => maskEmail(v),
    'idCard': (v) => maskIdCard(v),
    'bankCard': (v) => maskBankCard(v),
    'name': (v) => maskName(v),
  };

  // 保存原始json方法
  const originalJson = res.json.bind(res);

  // 重写json方法
  res.json = function (data) {
    if (typeof data === 'object' && data !== null) {
      // 递归脱敏
      const maskedData = maskObject(data, fieldMasking);
      return originalJson.call(this, maskedData);
    }
    return originalJson.call(this, data);
  };

  next();
}

// 递归脱敏对象
function maskObject(obj, fieldMasking, depth = 0) {
  // 防止无限递归
  if (depth > 10) return obj;

  if (Array.isArray(obj)) {
    return obj.map(item => maskObject(item, fieldMasking, depth + 1));
  }

  if (typeof obj === 'object' && obj !== null) {
    const result = {};
    for (const [key, value] of Object.entries(obj)) {
      if (fieldMasking[key]) {
        result[key] = fieldMasking[key](value);
      } else if (typeof value === 'object') {
        result[key] = maskObject(value, fieldMasking, depth + 1);
      } else {
        result[key] = value;
      }
    }
    return result;
  }

  return obj;
}

## 附录：常见安全合规问题

### Q1: 等保必须做吗？

A：在中国，如果你的系统涉及重要信息基础设施、或者达到一定规模，答案是必须的。具体来说：
- 政府机关：必须做
- 金融、医疗、教育等关键行业：必须做
- 其他行业：建议做，不做可能面临行政处罚

### Q2: GDPR适用于中国公司吗？

A：GDPR的适用范围是"处理欧盟居民个人数据"。如果你的产品有欧盟用户，就必须遵守GDPR。无论公司在哪里注册。

### Q3: 如何发现数据泄露？

A：
- 数据库访问监控
- 异常数据导出告警
- DLP（数据防泄漏）系统
- 员工举报
- 第三方通报

### Q4: 密码应该如何存储？

A：
- 绝对不能明文存储
- 不能使用MD5/SHA1（可被彩虹表破解）
- 推荐：bcrypt、Argon2、PBKDF2
- 加盐哈希，防止彩虹表攻击

### Q5: 如何防止SQL注入？

A：
- 使用参数化查询（最有效）
- 输入验证（辅助手段）
- 使用ORM框架
- 最小权限原则（数据库账户权限）
- Web应用防火墙（WAF）

### Q6: 如何防止XSS？

A：
- 输出编码（HTML转义）
- HTTP Only Cookie
- Content Security Policy (CSP)
- 输入验证（辅助手段）
