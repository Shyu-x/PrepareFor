# 代码执行工具设计实战

## 一、概述：工具接口设计的核心原则

在 AI Agent 架构中，工具（Tool）是 Agent 与外部世界交互的桥梁。代码执行工具（Code Execution Tool）更是 Agent 实现自主推理、验证假设、动态计算的核心能力。一个设计良好的代码执行工具需要同时满足以下目标：

| 目标 | 描述 | 质量标准 |
|------|------|----------|
| **功能正确性** | 正确解析参数、正确执行代码、正确返回结果 | 零运行时崩溃 |
| **安全隔离性** | 在沙箱中执行代码，防止恶意操作 | 多层防护、零逃逸 |
| **资源可控性** | 限制 CPU、内存、时间，防止资源耗尽 | 硬性限制 + 软性监控 |
| **可观测性** | 记录执行过程，便于调试和审计 | 完整日志 + 指标 |
| **容错恢复性** | 优雅处理错误，提供有意义的错误信息 | 错误分类 + 重试 |
| **性能高效性** | 快速响应，低延迟执行 | P99 延迟 < 500ms |

**设计哲学**：代码执行工具是 Agent 系统的边界节点，既要足够强大以支持复杂计算，又要足够安全以应对不可信输入。

### 2.1 统一工具接口规范

```typescript
/**
 * 统一工具接口定义（Tool Interface）
 * 所有代码执行工具必须实现此接口
 */
interface CodeExecutionTool {
  // 工具唯一标识符
  readonly name: string;
  // 工具版本号（用于兼容性追踪）
  readonly version: string;
  // 工具描述（Agent 用于理解工具用途）
  readonly description: string;
  // 参数模式定义（JSON Schema）
  readonly parameterSchema: ParameterSchema;
  // 工具执行的主入口
  execute(params: ToolParams): Promise<ToolResult>;
  // 工具健康检查
  healthCheck(): Promise<HealthStatus>;
  // 获取工具元信息
  getMetadata(): ToolMetadata;
}

/**
 * 参数模式定义（使用 JSON Schema 标准）
 */
interface ParameterSchema {
  type: 'object';
  properties: {
    [paramName: string]: {
      type: 'string' | 'number' | 'boolean' | 'array' | 'object';
      description: string;
      required?: boolean;
      default?: unknown;
      minimum?: number;
      maximum?: number;
      minLength?: number;
      maxLength?: number;
      enum?: unknown[];
      pattern?: string;
      items?: {
        type: string;
        minItems?: number;
        maxItems?: number;
      };
      properties?: Record<string, unknown>;
    };
  };
  required: string[];
  additionalProperties: boolean;
}

/**
 * 工具参数输入
 */
interface ToolParams {
  action: string;
  args: Record<string, unknown>;
  options?: {
    timeout?: number;
    enableRetry?: boolean;
    maxRetries?: number;
    environment?: string;
    callbackUrl?: string;
  };
  traceId?: string;
  caller?: string;
}

/**
 * 工具执行结果
 */
interface ToolResult {
  success: boolean;
  resultId: string;
  toolName: string;
  timestamp: number;
  duration: number;
  data?: unknown;
  error?: ToolError;
  metadata: {
    timeoutUsed: number;
    memoryPeak: number;
    cpuTime: number;
    retryCount: number;
    sandboxId?: string;
    codeLines?: number;
  };
}

/**
 * 工具错误定义
 */
interface ToolError {
  code: string;
  message: string;
  details: string;
  suggestion?: string;
  stackTrace?: string;
}
```

### 2.2 工具基类实现

```typescript
/**
 * 代码执行工具基类
 * 提供通用功能，所有具体工具继承此基类
 */
abstract class BaseCodeExecutionTool implements CodeExecutionTool {
  abstract readonly name: string;
  abstract readonly version: string;
  abstract readonly description: string;
  abstract readonly parameterSchema: ParameterSchema;

  protected isInitialized: boolean = false;
  protected executionCount: number = 0;
  protected errorCount: number = 0;
  protected lastHealthCheck: HealthStatus | null = null;

  /**
   * 工具执行主入口
   * 包含通用前置处理和后置处理逻辑
   */
  async execute(params: ToolParams): Promise<ToolResult> {
    const startTime = Date.now();
    const resultId = this.generateResultId();

    try {
      // 前置处理：参数校验
      const validationResult = this.validateParams(params);
      if (!validationResult.valid) {
        return this.createErrorResult(
          resultId,
          'INVALID_PARAMS',
          `参数校验失败：${validationResult.errors.join('; ')}`,
          `请检查参数格式是否符合 schema 要求`,
          startTime
        );
      }

      // 前置处理：资源预估（防止资源超限）
      const resourceEstimation = this.estimateResourceUsage(params.args);
      if (resourceEstimation.exceedsLimit) {
        return this.createErrorResult(
          resultId,
          'RESOURCE_EXCEEDED',
          `预估资源使用量超过限制：${resourceEstimation.message}`,
          `请减少代码量或简化算法`,
          startTime
        );
      }

      // 核心执行（在子类中实现）
      const result = await this.doExecute(params);
      const verifiedResult = this.verifyResult(result, params);
      this.executionCount++;

      return {
        success: true,
        resultId,
        toolName: this.name,
        timestamp: Date.now(),
        duration: Date.now() - startTime,
        data: verifiedResult,
        metadata: {
          timeoutUsed: params.options?.timeout ?? this.getDefaultTimeout(),
          memoryPeak: verifiedResult.memoryPeak ?? 0,
          cpuTime: verifiedResult.cpuTime ?? 0,
          retryCount: params.options?.enableRetry ? (params.options.maxRetries ?? 0) : 0,
          sandboxId: verifiedResult.sandboxId,
          codeLines: verifiedResult.codeLines,
        },
      };
    } catch (error) {
      this.errorCount++;
      return this.handleExecutionError(error, resultId, startTime);
    }
  }

  /** 子类必须实现的执行逻辑 */
  protected abstract doExecute(params: ToolParams): Promise<unknown>;

  /** 参数校验（子类可重写） */
  protected validateParams(params: ToolParams): { valid: boolean; errors: string[] } {
    const errors: string[] = [];

    // 检查必需参数
    for (const requiredParam of this.parameterSchema.required) {
      if (
        params.args[requiredParam] === undefined ||
        params.args[requiredParam] === null
      ) {
        errors.push(`缺少必需参数：${requiredParam}`);
      }
    }

    // 类型校验和范围校验
    for (const [key, value] of Object.entries(params.args)) {
      const schema = this.parameterSchema.properties[key];
      if (!schema) continue;

      const typeError = this.checkType(key, value, schema);
      if (typeError) errors.push(typeError);

      const rangeError = this.checkRange(key, value, schema);
      if (rangeError) errors.push(rangeError);
    }

    return { valid: errors.length === 0, errors };
  }

  /** 类型检查 */
  private checkType(
    key: string,
    value: unknown,
    schema: ParameterSchema['properties'][string]
  ): string | null {
    const actualType = Array.isArray(value) ? 'array' : typeof value;
    if (schema.type === 'number' && actualType !== 'number')
      return `参数 ${key} 应为 number 类型，实际为 ${actualType}`;
    if (schema.type === 'string' && actualType !== 'string')
      return `参数 ${key} 应为 string 类型`;
    if (schema.type === 'boolean' && actualType !== 'boolean')
      return `参数 ${key} 应为 boolean 类型`;
    if (schema.type === 'array' && !Array.isArray(value))
      return `参数 ${key} 应为 array 类型`;
    if (schema.type === 'object' && (actualType !== 'object' || Array.isArray(value)))
      return `参数 ${key} 应为 object 类型`;
    return null;
  }

  /** 范围检查 */
  private checkRange(
    key: string,
    value: unknown,
    schema: ParameterSchema['properties'][string]
  ): string | null {
    if (schema.type === 'string' && typeof value === 'string') {
      if (schema.minLength !== undefined && value.length < schema.minLength)
        return `参数 ${key} 长度 ${value.length} 小于最小值 ${schema.minLength}`;
      if (schema.maxLength !== undefined && value.length > schema.maxLength)
        return `参数 ${key} 长度 ${value.length} 大于最大值 ${schema.maxLength}`;
      if (schema.pattern) {
        const regex = new RegExp(schema.pattern);
        if (!regex.test(value))
          return `参数 ${key} 不符合正则表达式要求`;
      }
    }
    if (schema.type === 'number' && typeof value === 'number') {
      if (schema.minimum !== undefined && value < schema.minimum)
        return `参数 ${key} 值 ${value} 小于最小值 ${schema.minimum}`;
      if (schema.maximum !== undefined && value > schema.maximum)
        return `参数 ${key} 值 ${value} 大于最大值 ${schema.maximum}`;
    }
    if (schema.type === 'array' && Array.isArray(value)) {
      if (schema.items?.maxItems !== undefined && value.length > schema.items.maxItems)
        return `参数 ${key} 数组长度 ${value.length} 大于最大值 ${schema.items.maxItems}`;
    }
    return null;
  }

  /** 资源使用预估（子类可重写） */
  protected estimateResourceUsage(_args: Record<string, unknown>): {
    exceedsLimit: boolean;
    message?: string;
  } {
    return { exceedsLimit: false };
  }

  /** 结果验证（子类可重写） */
  protected verifyResult(
    result: unknown,
    _params: ToolParams
  ): unknown & { memoryPeak?: number; cpuTime?: number; sandboxId?: string; codeLines?: number } {
    return result as any;
  }

  protected getDefaultTimeout(): number { return 30000; }

  protected createErrorResult(
    resultId: string,
    code: string,
    message: string,
    suggestion: string,
    startTime: number
  ): ToolResult {
    return {
      success: false,
      resultId,
      toolName: this.name,
      timestamp: Date.now(),
      duration: Date.now() - startTime,
      error: { code, message, details: '', suggestion },
      metadata: { timeoutUsed: 0, memoryPeak: 0, cpuTime: 0, retryCount: 0 },
    };
  }

  protected handleExecutionError(error: unknown, resultId: string, startTime: number): ToolResult {
    const message = error instanceof Error ? error.message : String(error);
    return this.createErrorResult(
      resultId, 'EXECUTION_ERROR',
      `执行时发生错误：${message}`,
      '请检查代码逻辑是否正确', startTime
    );
  }

  async healthCheck(): Promise<HealthStatus> {
    const healthy = this.isInitialized && this.errorCount / Math.max(this.executionCount, 1) < 0.1;
    const status: HealthStatus = {
      healthy,
      toolName: this.name,
      timestamp: Date.now(),
      metrics: {
        totalExecutions: this.executionCount,
        errorRate: this.executionCount > 0 ? this.errorCount / this.executionCount : 0,
        avgLatency: 0,
      },
      lastError: this.lastHealthCheck?.lastError ?? null,
    };
    this.lastHealthCheck = status;
    return status;
  }

  getMetadata(): ToolMetadata {
    return {
      name: this.name,
      version: this.version,
      description: this.description,
      parameterSchema: this.parameterSchema,
      capabilities: this.getCapabilities(),
      rateLimit: this.getRateLimit(),
    };
  }

  protected getCapabilities(): string[] { return ['execute', 'healthCheck']; }
  protected getRateLimit(): { requestsPerMinute: number; burstSize: number } {
    return { requestsPerMinute: 60, burstSize: 10 };
  }

  private generateResultId(): string {
    return `${this.name}-${Date.now()}-${Math.random().toString(36).substring(2, 8)}`;
  }
}

interface HealthStatus {
  healthy: boolean;
  toolName: string;
  timestamp: number;
  metrics: { totalExecutions: number; errorRate: number; avgLatency: number };
  lastError: string | null;
}

interface ToolMetadata {
  name: string;
  version: string;
  description: string;
  parameterSchema: ParameterSchema;
  capabilities: string[];
  rateLimit: { requestsPerMinute: number; burstSize: number };
}
```

## 三、参数校验的深度实现

### 3.1 多层参数校验架构

```typescript
/**
 * 多层参数校验器
 * 从语法层、语义层、安全层三个维度校验参数
 */
class MultiLayerParameterValidator {
  private schemaValidator: SchemaValidator;
  private semanticRules: SemanticRule[];
  private securityScanner: SecurityScanner;

  constructor() {
    this.schemaValidator = new SchemaValidator();
    this.semanticRules = this.initializeSemanticRules();
    this.securityScanner = new SecurityScanner();
  }

  /**
   * 三层校验流程：语法 → 语义 → 安全
   * 任何一层失败都应拒绝执行
   */
  validate(params: ToolParams, schema: ParameterSchema): ValidationResult {
    const allErrors: ValidationError[] = [];

    // 第一层：语法校验（JSON Schema）
    const schemaErrors = this.schemaValidator.validate(params.args, schema);
    allErrors.push(...schemaErrors.map(e => ({ ...e, layer: 'SYNTAX' as const })));

    // 第二层：语义校验（仅在语法通过后执行）
    if (schemaErrors.length === 0) {
      const semanticErrors = this.validateSemantics(params.args);
      allErrors.push(...semanticErrors.map(e => ({ ...e, layer: 'SEMANTIC' as const })));
    }

    // 第三层：安全校验（始终执行）
    const securityErrors = this.securityScanner.scan(params.args);
    allErrors.push(...securityErrors.map(e => ({ ...e, layer: 'SECURITY' as const })));

    return {
      valid: allErrors.length === 0,
      errors: allErrors,
      passedLayers: allErrors.filter(e => e.layer === 'SYNTAX').length === 0
        ? (allErrors.filter(e => e.layer === 'SEMANTIC').length === 0 ? 3 : 2)
        : 1,
    };
  }

  /** 语义校验规则初始化 */
  private initializeSemanticRules(): SemanticRule[] {
    return [
      {
        field: 'code',
        validate: (value) => {
          if (typeof value !== 'string') return null;
          const lines = value.split('\n').length;
          if (lines > 1000) {
            return { valid: false, message: `代码行数 ${lines} 超过限制（1000行），请分段执行` };
          }
          return null;
        },
      },
      {
        field: 'filename',
        validate: (value) => {
          if (typeof value !== 'string') return null;
          if (value.includes('..') || value.includes('~') || value.startsWith('/')) {
            return { valid: false, message: `禁止使用绝对路径或路径遍历：${value}` };
          }
          const allowedExtensions = ['.py', '.js', '.ts', '.sh', '.sql', '.json'];
          const ext = value.substring(value.lastIndexOf('.'));
          if (!allowedExtensions.includes(ext)) {
            return { valid: false, message: `禁止的文件类型：${ext}` };
          }
          return null;
        },
      },
      {
        field: 'memoryLimit',
        validate: (value) => {
          if (typeof value !== 'number') return null;
          if (value < 64 * 1024 * 1024) return { valid: false, message: '内存限制太小，最小 64MB' };
          if (value > 2 * 1024 * 1024 * 1024) return { valid: false, message: '内存限制超过 2GB' };
          return null;
        },
      },
      {
        field: 'timeout',
        validate: (value) => {
          if (typeof value !== 'number') return null;
          if (value < 1000) return { valid: false, message: '超时时间太短，最小 1000ms' };
          if (value > 300000) return { valid: false, message: '超时时间超过最大允许值 5 分钟' };
          return null;
        },
      },
    ];
  }

  private validateSemantics(args: Record<string, unknown>): ValidationError[] {
    const errors: ValidationError[] = [];
    for (const rule of this.semanticRules) {
      if (args[rule.field] !== undefined) {
        const result = rule.validate(args[rule.field]);
        if (result && !result.valid) {
          errors.push({ field: rule.field, message: result.message, code: 'SEMANTIC_ERROR' });
        }
      }
    }
    return errors;
  }
}

/**
 * 安全扫描器：检测潜在恶意输入
 */
class SecurityScanner {
  // 危险模式列表（按严重程度分级）
  private dangerousPatterns = [
    // CRITICAL：命令注入和动态代码执行
    { pattern: /os\.system\s*\(\s*["'][^"']+/g, name: 'os.system 调用', severity: 'HIGH' },
    { pattern: /subprocess\.run\s*\([^)]*shell\s*=\s*True/g, name: 'shell=True 风险', severity: 'HIGH' },
    { pattern: /\b(eval|exec|compile)\s*\(/g, name: 'eval/exec/compile 动态执行', severity: 'HIGH' },
    { pattern: /__import__\s*\(/g, name: '__import__ 动态导入', severity: 'MEDIUM' },
    // 文件系统危险操作
    { pattern: /open\s*\([^)]*['"]\/etc['"]/, name: '访问 /etc 敏感目录', severity: 'HIGH' },
    { pattern: /open\s*\([^)]*['"]\/root['"]/, name: '访问 /root 目录', severity: 'HIGH' },
    { pattern: /open\s*\([^)]*['"]\/proc['"]/, name: '访问 /proc 目录', severity: 'HIGH' },
    { pattern: /chmod\s*\(\s*0o?7/, name: '设置最高权限', severity: 'HIGH' },
    // 网络请求
    { pattern: /requests\.get\s*\(/g, name: 'HTTP GET 请求', severity: 'LOW' },
    { pattern: /fetch\s*\(/g, name: 'Fetch API 请求', severity: 'LOW' },
    { pattern: /http\.createServer\s*\(/g, name: '创建 HTTP 服务器', severity: 'MEDIUM' },
    // 序列化攻击
    { pattern: /pickle\.loads?\s*\(/g, name: 'Python pickle 反序列化', severity: 'HIGH' },
    { pattern: /yaml\.load\s*\([^)]*(?!Loader\s*=\s*yaml\.SafeLoader\)/g, name: '不安全的 YAML 加载', severity: 'HIGH' },
    // 进程操作
    { pattern: /child_process/g, name: '子进程操作', severity: 'MEDIUM' },
    { pattern: /os\.kill\s*\(/g, name: '杀死进程', severity: 'HIGH' },
    // 无限循环
    { pattern: /while\s+True\s*:/g, name: '无限循环 while True', severity: 'MEDIUM' },
  ];

  scan(args: Record<string, unknown>): ValidationError[] {
    const errors: ValidationError[] = [];
    const code = args['code'] as string;
    if (!code || typeof code !== 'string') return errors;

    for (const { pattern, name, severity } of this.dangerousPatterns) {
      if (pattern.test(code)) {
        errors.push({
          field: 'code',
          message: `检测到潜在危险操作：${name}（${severity}）`,
          code: `SECURITY_${severity}`,
        });
      }
    }
    return errors;
  }
}
```

## 四、超时控制

### 4.1 多层次超时控制机制

```typescript
/**
 * 多层次超时控制管理器
 * 从请求层、执行层、系统层三个层次控制超时
 */
class TimeoutController {
  /**
   * 请求层超时：整个工具执行的超时控制
   */
  static createRequestTimeoutController(
    timeoutMs: number,
    onTimeout: () => void
  ): AbortController {
    const controller = new AbortController();
    const timer = setTimeout(() => {
      controller.abort();
      onTimeout();
    }, timeoutMs);
    controller.signal.addEventListener('abort', () => clearTimeout(timer));
    return controller;
  }

  /**
   * 执行层超时：使用 Promise.race 实现精确超时控制
   */
  static async withExecutionTimeout<T>(
    promise: Promise<T>,
    timeoutMs: number,
    timeoutError: string = `执行超时（${timeoutMs}ms）`
  ): Promise<T> {
    const timeoutPromise = new Promise<never>((_, reject) => {
      setTimeout(() => {
        reject(new ExecutionTimeoutException(timeoutError, timeoutMs));
      }, timeoutMs);
    });
    return Promise.race([promise, timeoutPromise]);
  }

  /**
   * 带冷却时间的超时控制（防止短时超时后立即重试）
   */
  static async withCooldownTimeout<T>(
    promise: Promise<T>,
    timeoutMs: number,
    cooldownMs: number = 100
  ): Promise<T> {
    try {
      return await this.withExecutionTimeout(promise, timeoutMs);
    } catch (error) {
      if (error instanceof ExecutionTimeoutException) {
        // 超时后等待冷却时间，再抛出错误
        // 防止错误处理代码来不及清理就继续执行
        await new Promise(resolve => setTimeout(resolve, cooldownMs));
      }
      throw error;
    }
  }
}

/** 执行超时异常 */
class ExecutionTimeoutException extends Error {
  constructor(message: string, public readonly timeoutMs: number) {
    super(message);
    this.name = 'ExecutionTimeoutException';
  }
}

/**
 * 渐进式超时策略
 * 对于不确定能否在固定时间内完成的操作，使用渐进超时
 */
class ProgressiveTimeoutController {
  private readonly initialTimeout: number;
  private readonly maxTimeout: number;
  private readonly multiplier: number;
  private currentTimeout: number;

  constructor(options: {
    initialTimeout?: number;
    maxTimeout?: number;
    multiplier?: number;
  } = {}) {
    this.initialTimeout = options.initialTimeout ?? 5000;
    this.maxTimeout = options.maxTimeout ?? 60000;
    this.multiplier = options.multiplier ?? 1.5;
    this.currentTimeout = this.initialTimeout;
  }

  increase(): void {
    this.currentTimeout = Math.min(this.currentTimeout * this.multiplier, this.maxTimeout);
  }

  reset(): void { this.currentTimeout = this.initialTimeout; }

  getCurrentTimeout(): number { return this.currentTimeout; }

  async executeWithProgressiveTimeout<T>(
    operation: (timeout: number) => Promise<T>,
    maxAttempts: number = 5
  ): Promise<T> {
    let lastError: unknown;

    for (let attempt = 0; attempt < maxAttempts; attempt++) {
      try {
        const result = await TimeoutController.withExecutionTimeout(
          operation(this.currentTimeout), this.currentTimeout
        );
        this.reset();
        return result;
      } catch (error) {
        lastError = error;
        if (error instanceof ExecutionTimeoutException) {
          this.increase();
          console.log(`尝试 #${attempt + 1} 超时，下次超时时间：${this.currentTimeout}ms`);
        } else {
          throw error;
        }
      }
    }
    throw lastError;
  }
}
```

```typescript
/**
 * 结果截断策略配置
 */
interface TruncationConfig {
  maxOutputBytes: number;      // 默认 1MB
  maxOutputLines: number;      // 默认 10000 行
  maxArrayLength: number;      // 默认 1000
  maxObjectKeys: number;       // 默认 100
  maxDepth: number;           // 默认 5
  preserveErrors: boolean;     // 默认 true（错误信息优先保留）
  preserveLastNLines: number; // 默认 50（保留结尾 N 行对调试很重要）
  computeSummary: boolean;     // 默认 true
}

/**
 * 结果截断器
 * 处理代码执行后的超大规模输出
 */
class ResultTruncator {
  private config: TruncationConfig;

  constructor(config: Partial<TruncationConfig> = {}) {
    this.config = {
      maxOutputBytes: config.maxOutputBytes ?? 1024 * 1024,
      maxOutputLines: config.maxOutputLines ?? 10000,
      maxArrayLength: config.maxArrayLength ?? 1000,
      maxObjectKeys: config.maxObjectKeys ?? 100,
      maxDepth: config.maxDepth ?? 5,
      preserveErrors: config.preserveErrors ?? true,
      preserveLastNLines: config.preserveLastNLines ?? 50,
      computeSummary: config.computeSummary ?? true,
    };
  }

  truncate(result: unknown, isError: boolean = false): TruncatedResult {
    // 错误输出优先保留
    if (isError && this.config.preserveErrors && typeof result === 'string') {
      return this.truncateString(result, true);
    }
    if (typeof result === 'string') return this.truncateString(result, isError);
    if (typeof result === 'object' && result !== null) return this.truncateObject(result);
    return { data: result, truncated: false, summary: null };
  }

  /** 字符串截断（优先保留开头和结尾） */
  private truncateString(str: string, preserveLast: boolean): TruncatedResult {
    const encoder = new TextEncoder();
    const byteLength = encoder.encode(str).length;
    const lines = str.split('\n');

    if (byteLength <= this.config.maxOutputBytes && lines.length <= this.config.maxOutputLines) {
      return { data: str, truncated: false, summary: null };
    }

    if (preserveLast && this.config.preserveLastNLines > 0) {
      const headCount = Math.min(100, this.config.maxOutputLines - this.config.preserveLastNLines);
      const headLines = lines.slice(0, headCount);
      const tailLines = lines.slice(-this.config.preserveLastNLines);
      const combined = [
        ...headLines,
        `... [省略中间 ${lines.length - headCount - tailLines.length} 行] ...`,
        ...tailLines,
      ].join('\n');
      const truncated = this.truncateToBytes(combined, this.config.maxOutputBytes);
      return {
        data: truncated,
        truncated: true,
        summary: {
          originalSize: byteLength,
          originalLines: lines.length,
          truncatedSize: new TextEncoder().encode(truncated).length,
          truncatedLines: truncated.split('\n').length,
          truncationReason: 'OUTPUT_SIZE_LIMIT',
        },
      };
    }

    const truncated = this.truncateToBytes(str, this.config.maxOutputBytes);
    return {
      data: truncated,
      truncated: true,
      summary: {
        originalSize: byteLength,
        originalLines: lines.length,
        truncatedSize: new TextEncoder().encode(truncated).length,
        truncatedLines: truncated.split('\n').length,
        truncationReason: 'OUTPUT_SIZE_LIMIT',
      },
    };
  }

  /** 对象截断（用于结构化结果） */
  private truncateObject(obj: unknown): TruncatedResult {
    const serialized = JSON.stringify(obj);
    if (serialized.length <= this.config.maxOutputBytes) {
      return { data: obj, truncated: false, summary: null };
    }
    const truncated = this.truncateDeep(obj, 0);
    const truncatedJson = JSON.stringify(truncated);
    return {
      data: truncated,
      truncated: true,
      summary: {
        originalSize: new TextEncoder().encode(serialized).length,
        originalLines: serialized.split('\n').length,
        truncatedSize: new TextEncoder().encode(truncatedJson).length,
        truncatedLines: truncatedJson.split('\n').length,
        truncationReason: 'OBJECT_SIZE_LIMIT',
      },
    };
  }

  /**
   * 深度截断（递归处理嵌套结构）
   * 保留开头和结尾的数组/对象元素，中间用省略号代替
   */
  private truncateDeep(obj: unknown, depth: number): unknown {
    if (depth > this.config.maxDepth) return '[MAX_DEPTH_EXCEEDED]';

    if (Array.isArray(obj)) {
      if (obj.length <= this.config.maxArrayLength) {
        return obj.map(item => this.truncateDeep(item, depth + 1));
      }
      const headCount = Math.floor(this.config.maxArrayLength / 2);
      const tailCount = this.config.maxArrayLength - headCount;
      return [
        ...obj.slice(0, headCount).map(item => this.truncateDeep(item, depth + 1)),
        `[... 省略中间 ${obj.length - this.config.maxArrayLength} 项 ...]`,
        ...obj.slice(-tailCount).map(item => this.truncateDeep(item, depth + 1)),
      ];
    }

    if (typeof obj === 'object' && obj !== null) {
      const keys = Object.keys(obj as Record<string, unknown>);
      if (keys.length <= this.config.maxObjectKeys) {
        const result: Record<string, unknown> = {};
        for (const key of keys) {
          result[key] = this.truncateDeep((obj as Record<string, unknown>)[key], depth + 1);
        }
        return result;
      }
      const headCount = Math.floor(this.config.maxObjectKeys / 2);
      const tailCount = this.config.maxObjectKeys - headCount;
      const result: Record<string, unknown> = {};
      keys.slice(0, headCount).forEach(key => {
        result[key] = this.truncateDeep((obj as Record<string, unknown>)[key], depth + 1);
      });
      result[`...[省略 ${keys.length - this.config.maxObjectKeys} 个键]...`] = null;
      keys.slice(-tailCount).forEach(key => {
        result[key] = this.truncateDeep((obj as Record<string, unknown>)[key], depth + 1);
      });
      return result;
    }

    return obj;
  }

  /** 按字节数截断字符串（二分查找优化） */
  private truncateToBytes(str: string, maxBytes: number): string {
    const encoder = new TextEncoder();
    const encoded = encoder.encode(str);
    if (encoded.length <= maxBytes) return str;

    let left = 0, right = str.length;
    while (left < right) {
      const mid = Math.floor((left + right + 1) / 2);
      if (encoder.encode(str.substring(0, mid)).length <= maxBytes) {
        left = mid;
      } else {
        right = mid - 1;
      }
    }
    return str.substring(0, left) + '\n... [输出被截断] ...';
  }
}

interface TruncatedResult {
  data: unknown;
  truncated: boolean;
  summary: {
    originalSize: number;
    originalLines: number;
    truncatedSize: number;
    truncatedLines: number;
    truncationReason: string;
  } | null;
}
```

```typescript
class StaticAnalysisLayer extends SecurityLayer {
  readonly name = 'StaticAnalysis';

  private readonly rules: SecurityRule[] = [
    { id: 'CMD_001', pattern: /os\.system\s*\(\s*["'][^"']+/g, severity: 'HIGH', message: '检测到命令注入风险：os.system', suggestion: '使用 subprocess.run() 并传入参数列表' },
    { id: 'CMD_002', pattern: /subprocess\.run\s*\([^)]*shell\s*=\s*True/g, severity: 'HIGH', message: 'subprocess shell=True 风险', suggestion: '避免使用 shell=True' },
    { id: 'DYNEVAL_001', pattern: /\b(eval|exec|compile)\s*\(/g, severity: 'HIGH', message: '检测到动态代码执行：eval/exec/compile', suggestion: '避免使用动态代码执行' },
    { id: 'FILE_001', pattern: /\.\.\/|\.\.\\\/|~\/|~\//g, severity: 'HIGH', message: '检测到路径遍历模式', suggestion: '使用 pathlib 或 os.path 处理文件路径' },
    { id: 'SFILE_001', pattern: /\/(etc\/passwd|etc\/shadow|etc\/group|root\/|home\/.ssh)/g, severity: 'HIGH', message: '检测到访问系统敏感文件', suggestion: '禁止访问系统敏感目录' },
    { id: 'SERIAL_001', pattern: /pickle\.load\s*\(|marshal\.loads\s*\(/g, severity: 'HIGH', message: '不安全的反序列化：pickle/marshal', suggestion: '使用 json 或安全序列化库' },
    { id: 'LOOP_001', pattern: /while\s+True\s*:/g, severity: 'MEDIUM', message: '检测到无限循环 while True', suggestion: '确保循环有终止条件或添加超时' },
  ];

  async check(code: string, params: ToolParams): Promise<LayerCheckResult> {
    const violations: Violation[] = [];

    for (const rule of this.rules) {
      const matches = code.match(rule.pattern);
      if (matches) {
        violations.push({
          ruleId: rule.id, pattern: rule.pattern.source,
          matches: matches.length, severity: rule.severity,
          message: rule.message, suggestion: rule.suggestion,
        });
      }
    }

    if (violations.length === 0) return { allowed: true };

    // 按严重程度排序
    violations.sort((a, b) => {
      const order = { CRITICAL: 0, HIGH: 1, MEDIUM: 2, LOW: 3 };
      return order[a.severity as keyof typeof order] - order[b.severity as keyof typeof order];
    });

    const topViolation = violations[0];

    // HIGH 及以上：有超时控制则允许（有警告），无超时则拒绝
    if (topViolation.severity === 'HIGH') {
      if (params.options?.timeout && params.options.timeout <= 30000) {
        return {
          allowed: true,
          reason: `检测到 ${violations.length} 个警告（最高级别：HIGH）`,
          severity: 'HIGH',
          suggestion: topViolation.suggestion,
        };
      }
      return {
        allowed: false,
        reason: '检测到 HIGH 级别安全警告且未设置超时控制',
        severity: 'HIGH',
        suggestion: '请设置合理的超时时间（不超过 30 秒）后重试',
      };
    }

    return { allowed: true };
  }
}
```

```typescript
    const code = args.code as string;
    const timeout = (args.timeout as number) ?? this.getDefaultTimeout();
    const memoryLimit = (args.memoryLimit as number) ?? 512 * 1024 * 1024;
    const packages = (args.packages as string[]) ?? [];
    const captureOutput = (args.captureOutput as boolean) ?? true;

    // 第一步：创建容器（带资源限制）
    const container = await this.sandbox.createContainer({
      image: 'python:3.11-slim',
      resources: { memoryLimit, timeout },
      timeout,
      onKill: () => { console.error(`Python 进程超过 ${timeout}ms 限制被终止`); },
    });

    try {
      // 第二步：安装依赖
      if (packages.length > 0) {
        await container.run(['pip', 'install', ...packages]);
      }

      // 第三步：写入代码文件（安全处理引号）
      const safeCode = code.replace(/'/g, "'\"'\"'");
      await container.run(['bash', '-c', `echo '${safeCode}' > /tmp/exec_code.py`]);

      // 第四步：执行代码（带超时控制）
      const startTime = Date.now();
      const executionResult = await TimeoutController.withExecutionTimeout(
        container.run(['python', '/tmp/exec_code.py']),
        timeout,
        `Python 代码执行超时（${timeout}ms）`
      );
      const duration = Date.now() - startTime;

      // 第五步：截断输出
      const stdoutTruncated = captureOutput
        ? this.truncator.truncate(executionResult.stdout, false)
        : null;
      const stderrTruncated = this.truncator.truncate(executionResult.stderr, true);

      return {
        stdout: stdoutTruncated?.data ?? null,
        stderr: stderrTruncated?.data ?? null,
        exitCode: executionResult.exitCode,
        duration,
        memoryPeak: executionResult.memoryPeak,

        stderrTruncated: stderrTruncated?.truncated ?? false,

        stderrSummary: stderrTruncated?.summary ?? null,
      };
    } finally {
      await container.cleanup();
    }
  }

    }
    return { exceedsLimit: false };
  }
```

```typescript
    }
    this.tools.set(tool.name, tool);
    console.log(`工具注册成功：${tool.name} v${tool.version}`);

    this.toolAliases.set(alias, toolName);
  }

  get(name: string): CodeExecutionTool | undefined {
    const actualName = this.toolAliases.get(name) ?? name;
    return this.tools.get(actualName);
  }

  /** 获取所有工具的元信息（供 Agent 发现） */
  listTools(): ToolMetadata[] {
    return Array.from(this.tools.values()).map(tool => tool.getMetadata());
  }

  searchTools(query: string): ToolMetadata[] {
    const lowerQuery = query.toLowerCase();
    return this.listTools().filter(tool =>
      tool.name.toLowerCase().includes(lowerQuery) ||
      tool.description.toLowerCase().includes(lowerQuery)
    );
  }

  /** 健康检查所有工具 */
  async checkAllHealth(): Promise<Map<string, HealthStatus>> {
    const results = new Map<string, HealthStatus>();
    await Promise.all(
      Array.from(this.tools.entries()).map(async ([name, tool]) => {
        const status = await tool.healthCheck();
        results.set(name, status);
      })
    );
    return results;
  }
```

2. **限制资源**：硬性限制 + 软性监控，防止资源耗尽
3. **渐进增强**：从简单检查到复杂分析，先快后慢
4. **可追溯**：完整的执行日志和审计记录
5. **优雅降级**：部分失败不影响整体系统稳定性

**核心设计公式**：

```
优秀代码执行工具 = 标准化接口 × 多层校验 × 智能超时 × 安全截断 × 纵深防御
```
