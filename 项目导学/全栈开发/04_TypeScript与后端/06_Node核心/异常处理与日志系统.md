## 1. 异常处理

### 1.1 全局异常捕获

```javascript
// Express 全局错误处理
const express = require('express');
const app = express();

// 1. 同步错误处理
app.get('/error', (req, res) => {
  throw new Error('同步错误');
});

// 2. 异步错误处理
app.get('/async-error', async (req, res, next) => {
  try {
    const result = await asyncOperation();
    res.json(result);
  } catch (error) {
    next(error);  // 传递给错误处理中间件
  }
});

// 3. 错误处理中间件
app.use((err, req, res, next) => {
  console.error('错误堆栈:', err.stack);

  const status = err.status || err.statusCode || 500;
  const message = process.env.NODE_ENV === 'production'
    ? '服务器内部错误'
    : err.message;

  res.status(status).json({
    success: false,
    error: {
      message,
      code: err.code || 'INTERNAL_ERROR',
      ...(process.env.NODE_ENV === 'development' && { stack: err.stack })
    }
  });
});
```

### 1.2 自定义错误类

```javascript
// 1. 基础错误类
class AppError extends Error {
  constructor(message, statusCode = 500, code = 'INTERNAL_ERROR') {
    super(message);
    this.statusCode = statusCode;
    this.code = code;
    this.status = `${statusCode}`.startsWith('4') ? 'fail' : 'error';
    Error.captureStackTrace(this, this.constructor);
  }
}

// 2. 客户端错误
class ClientError extends AppError {
  constructor(message = '请求错误', statusCode = 400) {
    super(message, statusCode, 'CLIENT_ERROR');
  }
}

// 3. 具体错误类型
class NotFoundError extends AppError {
  constructor(message = '资源不存在') {
    super(message, 404, 'NOT_FOUND');
  }
}

class UnauthorizedError extends AppError {
  constructor(message = '未授权') {
    super(message, 401, 'UNAUTHORIZED');
  }
}

class ForbiddenError extends AppError {
  constructor(message = '禁止访问') {
    super(message, 403, 'FORBIDDEN');
  }
}

class ValidationError extends AppError {
  constructor(message = '验证失败', errors = []) {
    super(message, 400, 'VALIDATION_ERROR');
    this.errors = errors;
  }
}

class ConflictError extends AppError {
  constructor(message = '资源冲突') {
    super(message, 409, 'CONFLICT');
  }
}

// 使用
app.get('/user/:id', async (req, res, next) => {
  const user = await findUser(req.params.id);

  if (!user) {
    throw new NotFoundError('用户不存在');
  }

  if (!user.isActive) {
    throw new ForbiddenError('用户已被禁用');
  }

  res.json(user);
});
```

### 1.3 错误日志

```javascript
// 1. 基础错误日志
function logError(error) {
  console.error(`[${new Date().toISOString()}]`, error.message);
  console.error(error.stack);
}

// 2. 结构化日志
const logger = {
  error: (message, meta = {}) => {
    console.error(JSON.stringify({
      level: 'error',
      message,
      timestamp: new Date().toISOString(),
      ...meta
    }));
  },

  warn: (message, meta = {}) => {
    console.warn(JSON.stringify({
      level: 'warn',
      message,
      timestamp: new Date().toISOString(),
      ...meta
    }));
  },

  info: (message, meta = {}) => {
    console.log(JSON.stringify({
      level: 'info',
      message,
      timestamp: new Date().toISOString(),
      ...meta
    }));
  }
};

// 3. 高级日志库 - Winston
const winston = require('winston');

const logger = winston.createLogger({
  level: 'info',
  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.json()
  ),
  transports: [
    new winston.transports.File({ filename: 'error.log', level: 'error' }),
    new winston.transports.File({ filename: 'combined.log' }),
    new winston.transports.Console()
  ]
});

// 使用
app.use((err, req, res, next) => {
  logger.error(err.message, {
    url: req.url,
    method: req.method,
    stack: err.stack,

  });
  next(err);
```

## 2. 日志系统

### 2.1 日志级别

```javascript
// 日志级别
const levels = {
  error: 0,
  warn: 1,
  info: 2,
  http: 3,
  debug: 4
};

// 自定义日志格式
const logFormat = winston.format.combine(
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
  winston.format.errors({ stack: true }),
  winston.format.printf(({ timestamp, level, message, stack }) => {
    return `${timestamp} [${level.toUpperCase()}]: ${message}${stack ? '\n' + stack : ''}`;
  })
);

// 配置文件
const logger = winston.createLogger({
  level: process.env.LOG_LEVEL || 'info',
  levels,
  format: logFormat,
  transports: [
    // 控制台输出
    new winston.transports.Console({
      format: winston.format.combine(
        winston.format.colorize(),
        logFormat
      )
    }),

    // 文件输出
    new winston.transports.File({ filename: 'logs/error.log', level: 'error' }),
    new winston.transports.File({ filename: 'logs/combined.log' })
  ]
});

// 业务日志
logger.info('用户登录', { userId: 123, ip: '127.0.0.1' });
logger.warn('请求频率过高', { ip: '127.0.0.1', count: 100 });
logger.error('数据库连接失败', { error: err.message });
```

## 5. CI/CD

### 5.1 GitHub Actions

```yaml
# .github/workflows/ci.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run linter
        run: npm run lint

      - name: Run tests
        run: npm test

      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info

  build:
    needs: test
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'

    steps:
      - uses: actions/checkout@v4

      - name: Build Docker image
        run: docker build -t myapp:${{ github.sha }} .

      - name: Push to registry
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker push myapp:${{ github.sha }}

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'

    steps:
      - name: Deploy to server
        uses: appleboy/ssh-action@v1
        with:
          host: ${{ secrets.HOST }}
          username: ${{ secrets.USERNAME }}
          key: ${{ secrets.SSH_KEY }}
          script: |
            docker pull myapp:${{ github.sha }}
            docker-compose up -d
```

### 5.2 自动化部署脚本

```bash
#!/bin/bash

# deploy.sh - 部署脚本

# 配置
APP_NAME="myapp"
IMAGE="myregistry.io/${APP_NAME}"
CONTAINER="${APP_NAME}-app"
ENV_FILE=".env.production"

# 颜色输出
RED='\033[0;31m'
GREEN='\033[0;32m'
NC='\033[0m'

echo_step() {
  echo -e "${GREEN}[DEPLOY]${NC} $1"
}

echo_error() {
  echo -e "${RED}[ERROR]${NC} $1"
}

# 检查环境
if [ ! -f "$ENV_FILE" ]; then
  echo_error "Environment file not found: $ENV_FILE"
  exit 1
fi

# 拉取最新代码
echo_step "Pulling latest code..."
git pull origin main

# 构建镜像
echo_step "Building Docker image..."
docker build -t "${IMAGE}:${GITHUB_SHA:-local}" .

# 登录镜像仓库
echo_step "Logging in to registry..."
echo "$DOCKER_PASSWORD" | docker login -u "$DOCKER_USERNAME" --password-stdin

# 推送镜像
echo_step "Pushing image..."
docker push "${IMAGE}:${GITHUB_SHA:-local}"

# 登录服务器
echo_step "Logging into server..."
ssh $SSH_USER@$SSH_HOST "docker login -u $DOCKER_USERNAME -p $DOCKER_PASSWORD"

# 部署
echo_step "Deploying to server..."
ssh $SSH_USER@$SSH_HOST << 'EOF'
  cd /opt/myapp
  docker-compose pull
  docker-compose up -d --no-deps
EOF

echo_step "Deployment complete!"
```

```typescript
export function metricsMiddleware(req: any, res: any, next: any) {
  const start = Date.now();

  // 响应完成后记录指标
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;

        route,
        status_code: res.statusCode,
      },
      duration
    );
  });

  next();
}
```

```typescript
// src/logger/index.ts - 结构化日志系统

import winston from 'winston';
import { ElasticsearchTransport } from 'winston-elasticsearch';

// 日志字段定义
interface LogContext {
  // 请求追踪
  traceId?: string;
  spanId?: string;
  requestId?: string;

  // 用户信息
  userId?: string;
  userRole?: string;

  // 请求信息
  method?: string;
  path?: string;
  statusCode?: number;
  responseTime?: number;

  // 业务信息
  action?: string;
  resource?: string;
  resourceId?: string;

  // 错误信息
  errorName?: string;
  errorMessage?: string;
  errorStack?: string;

  // 其他
  [key: string]: any;
}

// 日志格式
const logFormat = winston.format.combine(
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss.SSS' }),
  winston.format.errors({ stack: true }),
  winston.format.json()
);

// 开发环境格式
const devFormat = winston.format.combine(
  winston.format.colorize(),
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
  winston.format.printf(({ timestamp, level, message, ...meta }) => {
    const metaStr = Object.keys(meta).length ? JSON.stringify(meta, null, 2) : '';
    return `${timestamp} [${level}]: ${message} ${metaStr}`;
  })
);

// 创建 Logger
function createLogger(service: string): winston.Logger {
  const transports: winston.transport[] = [
    // 控制台输出
    new winston.transports.Console({
      format: process.env.NODE_ENV === 'production' ? logFormat : devFormat,
    }),

    // 文件输出
    new winston.transports.File({
      filename: 'logs/error.log',
      level: 'error',
      format: logFormat,
    }),
    new winston.transports.File({
      filename: 'logs/combined.log',
      format: logFormat,
    }),
  ];

  // Elasticsearch 输出（生产环境）
  if (process.env.NODE_ENV === 'production' && process.env.ELASTICSEARCH_URL) {
    transports.push(
      new ElasticsearchTransport({
        level: 'info',
        clientOpts: {
          node: process.env.ELASTICSEARCH_URL,
        },
        indexPrefix: `logs-${service}`,
      })
    );
  }

  return winston.createLogger({
    level: process.env.LOG_LEVEL || 'info',
    defaultMeta: {
      service,
      environment: process.env.NODE_ENV,
      hostname: process.env.HOSTNAME,
    },
    format: logFormat,
    transports,
  });
}

// 全局 Logger 实例
const logger = createLogger('backend');

// 请求日志中间件
export function requestLogger(req: any, res: any, next: any) {
  const start = Date.now();
  const traceId = req.headers['x-trace-id'] || generateTraceId();

  // 设置追踪 ID
  req.traceId = traceId;
  res.setHeader('x-trace-id', traceId);

  // 记录请求开始
  logger.info('请求开始', {
    traceId,
    method: req.method,
    path: req.path,
    query: req.query,
    userAgent: req.headers['user-agent'],
    ip: req.ip,
  });

  // 记录请求结束
  res.on('finish', () => {
    const duration = Date.now() - start;

    logger.info('请求完成', {
      traceId,
      method: req.method,
      path: req.path,
      statusCode: res.statusCode,
      responseTime: duration,
      userId: req.user?.id,
    });
  });

  next();
}

// 生成追踪 ID
function generateTraceId(): string {
  return `${Date.now().toString(36)}-${Math.random().toString(36).slice(2)}`;
}

// 业务日志封装
export const businessLogger = {
  // 用户操作
  userAction(action: string, userId: string, details: any) {
    logger.info('用户操作', {
      category: 'user_action',
      action,
      userId,
      ...details,
    });
  },

  // 订单事件
  orderEvent(event: string, orderId: string, details: any) {
    logger.info('订单事件', {
      category: 'order_event',
      event,
      orderId,
      ...details,
    });
  },

  // 支付事件
  paymentEvent(event: string, paymentId: string, details: any) {
    logger.info('支付事件', {
      category: 'payment_event',
      event,
      paymentId,
      ...details,
    });
  },

  // 安全事件
  securityEvent(event: string, details: any) {
    logger.warn('安全事件', {
      category: 'security_event',
      event,
      ...details,
    });
  },
};

export { logger, LogContext };
```

### Q2: 如何实现分布式追踪？

**答案：**

```typescript
// 分布式追踪实现

// 1. Trace Context 传播
interface TraceContext {
  traceId: string;    // 全局唯一追踪 ID
  spanId: string;     // 当前 span ID
  parentSpanId?: string;  // 父 span ID
  sampled: boolean;   // 是否采样
}

// 2. 生成 Trace Context
function generateTraceContext(): TraceContext {
  return {
    traceId: generateTraceId(),
    spanId: generateSpanId(),
    sampled: Math.random() < 0.1, // 10% 采样
  };
}

// 3. HTTP 中间件
function tracingMiddleware(req: any, res: any, next: any) {
  // 从请求头提取或生成 trace context
  const traceContext: TraceContext = {
    traceId: req.headers['x-trace-id'] || generateTraceId(),
    spanId: generateSpanId(),
    parentSpanId: req.headers['x-span-id'],
    sampled: req.headers['x-sampled'] === 'true',
  };

  // 存储到请求上下文
  req.traceContext = traceContext;

  // 设置响应头
  res.setHeader('x-trace-id', traceContext.traceId);

  // 创建 span
  const span = tracer.startSpan(`${req.method} ${req.path}`, {
    attributes: {
      'http.method': req.method,
      'http.url': req.url,
      'http.route': req.route?.path,
    },
  });

  // 请求结束
  res.on('finish', () => {
    span.setAttribute('http.status_code', res.statusCode);
    span.end();
  });

  next();
}

// 4. 服务间调用传播
async function callExternalService(url: string, traceContext: TraceContext) {
  const response = await fetch(url, {
    headers: {
      'x-trace-id': traceContext.traceId,
      'x-span-id': generateSpanId(),
      'x-parent-span-id': traceContext.spanId,
    },
  });
  return response;
}

// 5. 数据库调用追踪
async function tracedQuery(query: string, traceContext: TraceContext) {
  const span = tracer.startSpan('db.query', {
    attributes: {
      'db.system': 'postgresql',
      'db.statement': query,
    },
  });

  try {
    const result = await db.query(query);
    span.setStatus({ code: SpanStatusCode.OK });
    return result;
  } catch (error) {
    span.setStatus({ code: SpanStatusCode.ERROR, message: error.message });
    throw error;
  } finally {
    span.end();
  }
}
```

```typescript
// 内存缓冲 + 批量发送
class BufferedTransport extends winston.transports.Stream {
  private buffer: any[] = [];
  private timer: NodeJS.Timer;

  constructor() {
    super();
    this.timer = setInterval(() => this.flush(), 5000);
  }

  log(info: any, callback: () => void) {
    this.buffer.push(info);
    if (this.buffer.length >= 100) {
      this.flush();
    }
    callback();

    if (this.buffer.length === 0) return;

    try {
      await sendToLogServer(logs);
    } catch (error) {
      // 发送失败，写回本地文件
      this.writeToLocalFile(logs);
    }
  }
}
```

### 12.1 完整错误处理系统

```typescript
// src/errors/index.ts - 完整错误处理系统

import { Request, Response, NextFunction } from 'express';
import { logger } from '../logger';

// 错误类型定义
enum ErrorCode {
  // 客户端错误 4xx
  BAD_REQUEST = 'BAD_REQUEST',
  UNAUTHORIZED = 'UNAUTHORIZED',
  FORBIDDEN = 'FORBIDDEN',
  NOT_FOUND = 'NOT_FOUND',
  CONFLICT = 'CONFLICT',
  VALIDATION_ERROR = 'VALIDATION_ERROR',

  // 服务端错误 5xx
  INTERNAL_ERROR = 'INTERNAL_ERROR',
  SERVICE_UNAVAILABLE = 'SERVICE_UNAVAILABLE',
  DATABASE_ERROR = 'DATABASE_ERROR',
  EXTERNAL_SERVICE_ERROR = 'EXTERNAL_SERVICE_ERROR',
}

// 基础错误类
class AppError extends Error {
  public readonly statusCode: number;
  public readonly code: ErrorCode;
  public readonly isOperational: boolean;
  public readonly details?: any;

  constructor(
    message: string,
    statusCode: number = 500,
    code: ErrorCode = ErrorCode.INTERNAL_ERROR,
    isOperational: boolean = true,
    details?: any
  ) {
    super(message);
    this.statusCode = statusCode;
    this.code = code;
    this.isOperational = isOperational;
    this.details = details;

    Error.captureStackTrace(this, this.constructor);
  }
}

// 具体错误类
class BadRequestError extends AppError {
  constructor(message: string = '请求参数错误', details?: any) {
    super(message, 400, ErrorCode.BAD_REQUEST, true, details);
  }
}

class UnauthorizedError extends AppError {
  constructor(message: string = '未授权访问') {
    super(message, 401, ErrorCode.UNAUTHORIZED);
  }
}

class ForbiddenError extends AppError {
  constructor(message: string = '禁止访问') {
    super(message, 403, ErrorCode.FORBIDDEN);
  }
}

class NotFoundError extends AppError {
  constructor(message: string = '资源不存在') {
    super(message, 404, ErrorCode.NOT_FOUND);
  }
}

class ConflictError extends AppError {
  constructor(message: string = '资源冲突') {
    super(message, 409, ErrorCode.CONFLICT);
  }
}

class ValidationError extends AppError {
  constructor(message: string = '数据验证失败', details?: any) {
    super(message, 400, ErrorCode.VALIDATION_ERROR, true, details);
  }
}

class DatabaseError extends AppError {
  constructor(message: string = '数据库错误', details?: any) {
    super(message, 500, ErrorCode.DATABASE_ERROR, false, details);
  }
}

class ExternalServiceError extends AppError {
  constructor(service: string, details?: any) {
    super(`外部服务错误: ${service}`, 503, ErrorCode.EXTERNAL_SERVICE_ERROR, false, details);
  }
}

// 错误响应格式
interface ErrorResponse {
  success: false;
  error: {
    code: ErrorCode;
    message: string;
    details?: any;
    traceId?: string;
    stack?: string;
  };
  timestamp: string;
}

// 全局错误处理中间件
function errorHandler(
  err: Error | AppError,
  req: Request,
  res: Response,
  next: NextFunction
) {
  const traceId = (req as any).traceId;

  // 如果是 AppError
  if (err instanceof AppError) {
    const response: ErrorResponse = {
      success: false,
      error: {
        code: err.code,
        message: err.message,
        details: err.details,
        traceId,
      },
      timestamp: new Date().toISOString(),
    };

    // 开发环境返回堆栈
    if (process.env.NODE_ENV === 'development') {
      response.error.stack = err.stack;
    }

    // 记录日志
    if (err.isOperational) {
      logger.warn('业务错误', {
        traceId,
        code: err.code,
        message: err.message,
        details: err.details,
        path: req.path,
        method: req.method,
      });
    } else {
      logger.error('系统错误', {
        traceId,
        code: err.code,
        message: err.message,
        stack: err.stack,
        path: req.path,
        method: req.method,
      });
    }

    return res.status(err.statusCode).json(response);
  }

  // 未知错误
  logger.error('未捕获错误', {
    traceId,
    message: err.message,
    stack: err.stack,
    path: req.path,
    method: req.method,
  });

  const response: ErrorResponse = {
    success: false,
    error: {
      code: ErrorCode.INTERNAL_ERROR,
      message: process.env.NODE_ENV === 'production'
        ? '服务器内部错误'
        : err.message,
      traceId,
      ...(process.env.NODE_ENV === 'development' && { stack: err.stack }),
    },
    timestamp: new Date().toISOString(),
  };

  return res.status(500).json(response);
}

// 404 处理
function notFoundHandler(req: Request, res: Response, next: NextFunction) {
  const error = new NotFoundError(`路由不存在: ${req.method} ${req.path}`);
  next(error);
}

// 异步错误包装器
function asyncHandler(fn: Function) {
  return (req: Request, res: Response, next: NextFunction) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
}

// 未捕获异常处理
function setupErrorHandlers() {
  process.on('uncaughtException', (error: Error) => {
    logger.error('未捕获异常', {
      message: error.message,
      stack: error.stack,
    });

    // 优雅关闭
    process.exit(1);
  });

  process.on('unhandledRejection', (reason: any, promise: Promise<any>) => {
    logger.error('未处理的 Promise 拒绝', {
      reason: reason instanceof Error ? reason.message : String(reason),
      stack: reason instanceof Error ? reason.stack : undefined,
    });
  });
}

export {
  AppError,
  BadRequestError,
  UnauthorizedError,
  ForbiddenError,
  NotFoundError,
  ConflictError,
  ValidationError,
  DatabaseError,
  ExternalServiceError,
  ErrorCode,
  errorHandler,
  notFoundHandler,
  asyncHandler,
  setupErrorHandlers,
};
```

## 7. 日志系统与集中式日志管理

### 7.1 分层日志规范

```javascript
// 分层日志系统实现
/**
 * 日志级别分层：
 * - error: 错误级别 - 系统无法正常运行的错误
 * - warn: 警告级别 - 需要关注但不影响功能的问题
 * - info: 信息级别 - 重要的业务事件
 * - debug: 调试级别 - 开发环境详细调试信息
 * - trace: 追踪级别 - 最详细的执行追踪
 */

class Logger {
  constructor(options = {}) {
    this.level = options.level || 'info';
    this.context = options.context || {};
    this.transports = options.transports || [new ConsoleTransport()];
    this.levels = {
      error: 0,
      warn: 1,
      info: 2,
      debug: 3,
      trace: 4
    };
  }

  // 判断是否应该记录该级别
  shouldLog(level) {
    return this.levels[level] <= this.levels[this.level];
  }

  // 格式化日志条目
  formatEntry(level, message, meta = {}) {
    return {
      timestamp: new Date().toISOString(),
      level,
      message,
      context: { ...this.context, ...meta.context },
      requestId: meta.requestId,
      userId: meta.userId,
      duration: meta.duration,
      error: meta.error ? {
        name: meta.error.name,
        message: meta.error.message,
        stack: meta.error.stack
      } : undefined
    };
  }

  // 异步写入日志
  async log(level, message, meta = {}) {
    if (!this.shouldLog(level)) return;

    const entry = this.formatEntry(level, message, meta);

    for (const transport of this.transports) {
      try {
        await transport.write(entry);
      } catch (error) {
        console.error('日志写入失败:', error);
      }
    }
  }

  error(message, meta) { return this.log('error', message, meta); }
  warn(message, meta) { return this.log('warn', message, meta); }
  info(message, meta) { return this.log('info', message, meta); }
  debug(message, meta) { return this.log('debug', message, meta); }
  trace(message, meta) { return this.log('trace', message, meta); }
}

// 控制台传输器
class ConsoleTransport {
  async write(entry) {
    const color = {
      error: '\x1b[31m',
      warn: '\x1b[33m',
      info: '\x1b[36m',
      debug: '\x1b[90m',
      trace: '\x1b[90m'
    }[entry.level] || '';

    const reset = '\x1b[0m';
    const timestamp = entry.timestamp.split('T')[1].split('.')[0];

    console.log(
      `${color}[${timestamp}] [${entry.level.toUpperCase()}]${reset} ${entry.message}`,
      entry.context ? JSON.stringify(entry.context) : '',
      entry.error ? `\n${entry.error.stack}` : ''
    );
  }
}

// HTTP传输器（发送到ELK等日志系统）
class HttpTransport {
  constructor(options) {
    this.url = options.url;
    this.batchSize = options.batchSize || 100;
    this.flushInterval = options.flushInterval || 5000;
    this.buffer = [];

    setInterval(() => this.flush(), this.flushInterval);
  }

  async write(entry) {
    this.buffer.push(entry);

    if (this.buffer.length >= this.batchSize) {
      await this.flush();
    }
  }

  async flush() {
    if (this.buffer.length === 0) return;

    const entries = this.buffer.splice(0, this.buffer.length);

    try {
      await fetch(this.url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ logs: entries })
      });
    } catch (error) {
      console.error('日志发送失败:', error);
      this.buffer.unshift(...entries);
    }
  }
}
```

### 7.2 请求日志中间件

```javascript
// Express请求日志中间件
function requestLogger(logger) {
  return (req, res, next) => {
    const requestId = crypto.randomUUID();
    const startTime = Date.now();

    req.requestId = requestId;
    res.setHeader('X-Request-Id', requestId);

    logger.info('请求开始', {
      requestId,
      method: req.method,
      path: req.path,
      query: req.query,
      ip: req.ip,
      userAgent: req.get('user-agent')
    });

    res.on('finish', () => {
      const duration = Date.now() - startTime;
      const level = res.statusCode >= 400 ? 'warn' : 'info';

      logger.log(level, '请求完成', {
        requestId,
        method: req.method,
        path: req.path,
        statusCode: res.statusCode,
        duration,
        userId: req.user?.id
      });

      if (duration > 1000) {
        logger.warn('慢请求警告', {
          requestId,
          duration,
          threshold: 1000
        });
      }
    });

    next();
  };
}
```

```javascript
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.001, 0.005, 0.015, 0.05, 0.1, 0.2, 0.5, 1, 2, 5]
});

  name: 'http_requests_total',
  help: 'HTTP请求总数',
  labelNames: ['method', 'route', 'status_code']
});

// 指标中间件
function metricsMiddleware(req, res, next) {
  const startTime = process.hrtime();

  res.on('finish', () => {
    const [seconds, nanoseconds] = process.hrtime(startTime);
    const duration = seconds + nanoseconds / 1e9;

    const route = req.route?.path || req.path;
    const labels = {
      method: req.method,
      route: route,
      status_code: res.statusCode
    };

    httpRequestDuration.observe(labels, duration);
    httpRequestTotal.inc(labels);
  });

  next();
}

  try {
    res.set('Content-Type', register.contentType);
    res.end(await register.metrics());
  } catch (error) {
    res.status(500).end(error.message);
  }
});
```
