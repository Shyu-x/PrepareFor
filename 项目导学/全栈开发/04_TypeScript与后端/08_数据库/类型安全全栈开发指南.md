## 1. 类型安全ORM概述

### 1.1 为什么需要类型安全

```
传统开发模式：
  前端 ←→ API ←→ 后端 ←→ 数据库
           ↓
        类型断裂

类型安全模式：
  前端 ←→ 类型共享 ←→ 后端 ←→ 数据库
           ↓
        端到端类型安全
```

### 1.2 主流类型安全ORM对比

| ORM | 特点 | 适用场景 | 类型支持 |
|-----|------|----------|----------|
| **Prisma** | 功能完善、易用性好 | 中大型项目 | 完整 |
| **Drizzle** | 轻量、性能高 | 追求性能 | 完整 |
| **TypeORM** | 功能全面 | 企业级 | 部分 |
| **Sequelize** | 传统方案 | 遗留项目 | 基础 |

```typescript
  return prisma.post.delete({
    where: { id },
  });

// 事务
async function createPostWithTags(postData: {
  title: string;
  content: string;
  authorId: string;
  tags: string[];
}) {
  return prisma.$transaction(async (tx) => {
    const post = await tx.post.create({
      data: {
        title: postData.title,
        content: postData.content,
        authorId: postData.authorId,
        tags: {
          connectOrCreate: postData.tags.map((tag) => ({
            where: { name: tag },
            create: { name: tag },
          })),
        },
      },
      include: { tags: true },
    });
    return post;
  });
}
```

## 3. Drizzle ORM实战

### 3.1 Schema定义

```typescript
// db/schema.ts
import { pgTable, serial, text, timestamp, boolean, integer, pgEnum } from 'drizzle-orm/pg-core';
import { relations } from 'drizzle-orm';

// 枚举类型
export const roleEnum = pgEnum('role', ['admin', 'user', 'guest']);

// 用户表
export const users = pgTable('users', {
  id: serial('id').primaryKey(),
  email: text('email').notNull().unique(),
  name: text('name'),
  role: roleEnum('role').default('user'),
  createdAt: timestamp('created_at').defaultNow(),
});

// 文章表
export const posts = pgTable('posts', {
  id: serial('id').primaryKey(),
  title: text('title').notNull(),
  content: text('content'),
  published: boolean('published').default(false),
  authorId: integer('author_id').references(() => users.id),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
});

// 关系定义
export const usersRelations = relations(users, ({ many }) => ({
  posts: many(posts),
}));

export const postsRelations = relations(posts, ({ one }) => ({
  author: one(users, {
    fields: [posts.authorId],
    references: [users.id],
  }),
}));

// 类型导出
export type User = typeof users.$inferSelect;
export type NewUser = typeof users.$inferInsert;
export type Post = typeof posts.$inferSelect;
export type NewPost = typeof posts.$inferInsert;
```

### 3.2 查询操作

```typescript
import { db } from './db';
import { users, posts } from './schema';
import { eq, desc, like, and, or } from 'drizzle-orm';

// 插入
async function createUser(data: NewUser) {
  return db.insert(users).values(data).returning();
}

// 查询
async function getUserByEmail(email: string) {
  return db.select().from(users).where(eq(users.email, email)).limit(1);
}

// 条件查询
async function getPublishedPosts() {
  return db
    .select({
      id: posts.id,
      title: posts.title,
      authorName: users.name,
    })
    .from(posts)
    .leftJoin(users, eq(posts.authorId, users.id))
    .where(eq(posts.published, true))
    .orderBy(desc(posts.createdAt));
}

// 更新
async function updatePost(id: number, data: Partial<NewPost>) {
  return db.update(posts).set(data).where(eq(posts.id, id)).returning();
}

// 删除
async function deletePost(id: number) {
  return db.delete(posts).where(eq(posts.id, id));
}

// 批量操作
async function bulkCreatePosts(postsData: NewPost[]) {
  return db.insert(posts).values(postsData).returning();
}
```

## 4. tRPC类型安全API

### 4.1 服务端配置

```typescript
// server/trpc.ts
import { initTRPC } from '@trpc/server';
import { z } from 'zod';
import { prisma } from './prisma';

const t = initTRPC.create();

// 导出路由器
export const router = t.router;
export const publicProcedure = t.procedure;

// 私有过程（带认证）
export const protectedProcedure = t.procedure.use(async ({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({ code: 'UNAUTHORIZED' });
  }
  return next({ ctx: { ...ctx, user: ctx.user } });
});

// 路由器
export const appRouter = router({
  // 用户相关
  getUser: publicProcedure
    .input(z.object({ id: z.string() }))
    .query(async ({ input }) => {
      return prisma.user.findUnique({
        where: { id: input.id },
        include: { posts: true },
      });
    }),

  createUser: publicProcedure
    .input(
      z.object({
        email: z.string().email(),
        name: z.string().min(1),
        password: z.string().min(6),
      })
    )
    .mutation(async ({ input }) => {
      const hashedPassword = await hashPassword(input.password);
      return prisma.user.create({
        data: {
          email: input.email,
          name: input.name,
          password: hashedPassword,
        },
      });
    }),

  // 文章相关（需要认证）
  createPost: protectedProcedure
    .input(
      z.object({
        title: z.string().min(1),
        content: z.string(),
        tags: z.array(z.string()).optional(),
      })
    )
    .mutation(async ({ ctx, input }) => {
      return prisma.post.create({
        data: {
          title: input.title,
          content: input.content,
          authorId: ctx.user.id,
          tags: input.tags
            ? {
                connectOrCreate: input.tags.map((tag) => ({
                  where: { name: tag },
                  create: { name: tag },
                })),
              }
            : undefined,
        },
      });
    }),

  getMyPosts: protectedProcedure.query(async ({ ctx }) => {
    return prisma.post.findMany({
      where: { authorId: ctx.user.id },
      orderBy: { createdAt: 'desc' },
    });
  }),
});

// 导出类型
export type AppRouter = typeof appRouter;
```

```typescript
// 1. 定义Schema（Prisma/Drizzle）
model Post {
  id       String @id @default(cuid())
  title    String
  content  String
  author   User   @relation(fields: [authorId], references: [id])
  authorId String
}

// 2. 创建API（tRPC）
const postRouter = router({
  list: publicProcedure
    .input(z.object({
      limit: z.number().min(1).max(100).default(10),
      cursor: z.string().optional(),
    }))
    .query(async ({ input }) => {
      const posts = await prisma.post.findMany({
        take: input.limit + 1,
        cursor: input.cursor ? { id: input.cursor } : undefined,
        orderBy: { createdAt: 'desc' },
        include: { author: true },
      });

      let nextCursor: string | undefined;

        const nextItem = posts.pop();
        nextCursor = nextItem!.id;

      return { posts, nextCursor };
    }),
});

        page.posts.map((post) => (

        ))
      )}
      {hasNextPage && <button onClick={() => fetchNextPage()}>加载更多</button>}
    </div>
```

### 6.1 中间件与上下文

```typescript
// server/context.ts - 创建请求上下文
import type { FetchCreateContextFnOptions } from '@trpc/server/adapters/fetch';

export async function createContext(opts: FetchCreateContextFnOptions) {
  // 从请求头获取用户信息（JWT验证）
  const authHeader = opts.req.headers.get('authorization');
  const token = authHeader?.replace('Bearer ', '');

  let user = null;
  if (token) {
    try {
      user = await verifyJWT(token);
    } catch (e) {
      // Token无效，用户未登录
    }
  }

  return {
    user,        // 当前用户
    prisma,      // 数据库客户端
    startTime: Date.now(),  // 请求开始时间
  };
}

// 类型定义
export type Context = Awaited<ReturnType<typeof createContext>>;
```

```typescript
// server/trpc.ts - 高级中间件
import { initTRPC, TRPCError } from '@trpc/server';
import { z } from 'zod';

const t = initTRPC.context<Context>().create();

// 公共过程（无需认证）
export const publicProcedure = t.procedure;

// 速率限制中间件
const rateLimitMiddleware = t.middleware(async ({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({
      code: 'UNAUTHORIZED',
      message: '需要登录'
    });
  }

  // 检查速率限制
  const key = `ratelimit:${ctx.user.id}`;
  const current = await redis.incr(key);

  if (current === 1) {
    await redis.expire(key, 60); // 60秒过期
  }

  if (current > 100) { // 每分钟最多100次
    throw new TRPCError({
      code: 'TOO_MANY_REQUESTS',
      message: '请求过于频繁'
    });
  }

  return next({
    ctx: {
      ...ctx,
      rateLimit: current,
    },
  });
});

// 日志中间件
const loggingMiddleware = t.middleware(async ({ ctx, next, path, type, input }) => {
  const start = Date.now();

  const result = await next({
    ctx: {
      ...ctx,
      requestId: generateRequestId(),
    },
  });

  const duration = Date.now() - start;

  console.log({
    requestId: ctx.requestId,
    path,
    type,
    duration,
    userId: ctx.user?.id,
    success: !result.error,
  });

  return result;
});

// 组合中间件
export const protectedProcedure = t.procedure
  .use(rateLimitMiddleware)
  .use(loggingMiddleware);

// 带角色检查的中间件
const requireRole = (role: 'admin' | 'moderator') =>
  t.middleware(({ ctx, next }) => {
    if (!ctx.user) {
      throw new TRPCError({ code: 'UNAUTHORIZED' });
    }
    if (ctx.user.role !== role && ctx.user.role !== 'admin') {
      throw new TRPCError({ code: 'FORBIDDEN' });
    }
    return next({ ctx });
  });

export const adminProcedure = t.procedure.use(requireRole('admin'));
```

```typescript
import { router, publicProcedure, protectedProcedure } from '../trpc';
import { z } from 'zod';
import { hash, compare } from 'bcrypt';

export const userRouter = router({
  // 获取用户信息
  getById: publicProcedure
    .input(z.object({ id: z.string() }))
    .query(async ({ ctx, input }) => {
      const user = await ctx.prisma.user.findUnique({
        where: { id: input.id },
        select: {
          id: true,
          name: true,
          email: true,
          avatar: true,
          createdAt: true,
        },
      });

      if (!user) {
        throw new TRPCError({
          code: 'NOT_FOUND',
          message: '用户不存在',
        });
      }

      return user;
    }),

  // 搜索用户
  search: publicProcedure
    .input(z.object({
      query: z.string().min(2),
      limit: z.number().min(1).max(50).default(10),
    }))
    .query(async ({ ctx, input }) => {
      return ctx.prisma.user.findMany({
        where: {
          OR: [
            { name: { contains: input.query } },
            { email: { contains: input.query } },
          ],
        },
        take: input.limit,
        select: {
          id: true,
          name: true,
          avatar: true,
        },
      });
    }),

  // 更新个人资料
  updateProfile: protectedProcedure
    .input(z.object({
      name: z.string().min(2).optional(),
      avatar: z.string().url().optional(),
      bio: z.string().max(500).optional(),
    }))
    .mutation(async ({ ctx, input }) => {
      return ctx.prisma.user.update({
        where: { id: ctx.user.id },
        data: input,
      });
    }),

  // 修改密码
  changePassword: protectedProcedure
    .input(z.object({
      oldPassword: z.string(),
      newPassword: z.string().min(8),
    }))
    .mutation(async ({ ctx, input }) => {
      // 验证旧密码
      const user = await ctx.prisma.user.findUnique({
        where: { id: ctx.user.id },
      });

      const isValid = await compare(input.oldPassword, user!.password);
      if (!isValid) {
        throw new TRPCError({
          code: 'BAD_REQUEST',
          message: '原密码错误',
        });
      }

      // 更新密码
      const hashedPassword = await hash(input.newPassword, 12);
      await ctx.prisma.user.update({
        where: { id: ctx.user.id },
        data: { password: hashedPassword },
      });

      return { success: true };
    }),
});
```

### 6.3 订阅与实时更新

```typescript
// server/routers/realtime.ts - 实时订阅
import { router, publicProcedure, protectedProcedure } from '../trpc';
import { z } from 'zod';
import { observable } from '@trpc/server/observable';
import { EventEmitter } from 'events';

const ee = new EventEmitter();

export const realtimeRouter = router({
  // 订阅文档变化
  subscribeToDocument: protectedProcedure
    .input(z.object({ documentId: z.string() }))
    .subscription(({ ctx, input }) => {
      return observable<DocumentEvent>((emit) => {
        // 加入房间
        const roomId = `document:${input.documentId}`;

        const onUpdate = (data: DocumentEvent) => {
          emit.next(data);
        };

        // 监听事件
        ee.on(roomId, onUpdate);

        // 清理函数
        return () => {
          ee.off(roomId, onUpdate);
        };
      });
    }),

  // 发布更新
  publishUpdate: protectedProcedure
    .input(z.object({
      documentId: z.string(),
      content: z.string(),
      version: z.number(),
    }))
    .mutation(({ ctx, input }) => {
      const roomId = `document:${input.documentId}`;

      // 广播更新
      ee.emit(roomId, {
        type: 'update',
        documentId: input.documentId,
        content: input.content,
        version: input.version,
        userId: ctx.user.id,
        timestamp: new Date(),
      });

      return { success: true };
    }),
});

// 类型定义
interface DocumentEvent {
  type: 'update' | 'delete' | 'join' | 'leave';
  documentId: string;
  content?: string;
  version?: number;
  userId: string;
  timestamp: Date;
}
```

```typescript
// client/realtime.ts - 客户端订阅
import { useState, useEffect } from 'react';
import { trpc } from './trpc';

function DocumentEditor({ documentId }: { documentId: string }) {
  const [content, setContent] = useState('');

  // 使用useQuery订阅实时更新
  const { data: document } = trpc.document.getById.useQuery(
    { id: documentId },
    { refetchInterval: false } // 禁用自动轮询
  );

  // 使用useSubscription监听实时事件
  trpc.realtime.subscribeToDocument.useSubscription(
    { documentId },
    {
      onData(event) {
        if (event.type === 'update') {
          setContent(event.content!);
        }
      },
    }
  );

  // 发送更新
  const updateDocument = trpc.realtime.publishUpdate.useMutation();

  const handleChange = async (newContent: string) => {
    setContent(newContent);

    await updateDocument.mutateAsync({
      documentId,
      content: newContent,
      version: (document?.version || 0) + 1,
    });
  };

  return (
    <div>
      <textarea
        value={content}
        onChange={(e) => handleChange(e.target.value)}
      />
    </div>
  );
}
```

```typescript
        take: limit + 1, // 多取一条判断是否有更多
        cursor: cursor ? { id: cursor } : undefined,
        orderBy: { createdAt: 'desc' },
        include: {
          author: {
            select: { id: true, name: true, avatar: true },
          },
          _count: { select: { comments: true, likes: true } },
        },
      });

      let nextCursor: string | undefined = undefined;
      if (items.length > limit) {
        const nextItem = items.pop();
        nextCursor = nextItem!.id;
      }

      return {
        items,
        nextCursor,
      };
    }),

  // 搜索分页
  searchPosts: publicProcedure
    .input(z.object({
      query: z.string(),
      page: z.number().min(1).default(1),
      pageSize: z.number().min(1).max(50).default(10),
      sortBy: z.enum(['relevance', 'date', 'popularity']).default('date'),
    }))
    .query(async ({ ctx, input }) => {
      const { query, page, pageSize, sortBy } = input;
      const skip = (page - 1) * pageSize;

      const where = {
        OR: [
          { title: { contains: query, mode: 'insensitive' as const } },
          { content: { contains: query, mode: 'insensitive' as const } },
        ],
        published: true,
      };

      const orderBy =
        sortBy === 'date'
          ? { createdAt: 'desc' as const }
          : sortBy === 'popularity'
          ? { likes: { _count: 'desc' as const } }
          : undefined;

      const [items, total] = await Promise.all([
        ctx.prisma.post.findMany({
          where,
          skip,
          take: pageSize,
          orderBy,
          include: {
            author: { select: { name: true, avatar: true } },
          },
        }),
        ctx.prisma.post.count({ where }),
      ]);

      return {
        items,
        pagination: {
          page,
          pageSize,
          total,
          totalPages: Math.ceil(total / pageSize),
          hasMore: page * pageSize < total,
        },
      };
    }),
});
```

```typescript
// 聚合查询示例
async function getAnalytics() {
  // 基础聚合
  const totalUsers = await prisma.user.count();
  const totalPosts = await prisma.post.count();

  // 聚合统计
  const stats = await prisma.post.aggregate({
    _count: true,
    _avg: { viewCount: true },
    _sum: { likeCount: true },
    _min: { createdAt: true },
    _max: { updatedAt: true },
  });

  // 分组查询
  const postsByMonth = await prisma.$queryRaw`
    SELECT
      DATE_TRUNC('month', "createdAt") as month,
      COUNT(*) as count,
      SUM("viewCount") as totalViews
    FROM "Post"
    WHERE "createdAt" > NOW() - INTERVAL '1 year'
    GROUP BY DATE_TRUNC('month', "createdAt")
    ORDER BY month DESC
  `;

  // 用户活跃度统计
  const userActivity = await prisma.user.findMany({
    select: {
      name: true,
      _count: {
        select: {
          posts: true,
          comments: true,
        },
      },
    },
    orderBy: {
      posts: { _count: 'desc' },
    },
    take: 10,
  });

  return {
    totalUsers,
    totalPosts,
    stats,
    postsByMonth,
    topUsers: userActivity,
  };
}
```

```prisma
      async findManySafe(args) {
        return this.findMany({
          ...args,
          where: {
            ...args.where,
            deletedAt: null,
          },
        });
      },

      async softDelete(id: string, deletedBy: string) {

          where: { id },

            deletedBy,

    },
  },
```
