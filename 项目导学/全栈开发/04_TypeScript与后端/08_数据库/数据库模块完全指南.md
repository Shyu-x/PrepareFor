## 1. MongoDB 数据库

### 1.1 MongoDB 基础概念

MongoDB 是一个基于文档的 NoSQL 数据库，使用 JSON 格式存储数据，灵活且可扩展。

```
┌─────────────────────────────────────────────────────────────────┐
│                    MongoDB 数据模型                                │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  关系型数据库                 MongoDB                              │
│  ─────────────               ──────                              │
│  数据库 (Database)            数据库 (Database)                    │
│  表 (Table)                  集合 (Collection)                   │
│  行 (Row)                    文档 (Document)                      │
│  列 (Column)                 字段 (Field)                         │
│  主键 (Primary Key)           _id (ObjectId)                     │
│  索引 (Index)                索引 (Index)                        │
│                                                                  │
│  示例：                                                       │
│  ─────────────                                            │
│  {                                                        │
│    "_id": ObjectId("..."),                                 │
│    "username": "zhangsan",                                  │
│    "email": "zhangsan@example.com",                         │
│    "age": 25,                                               │
│    "address": {                                              │
│      "city": "Beijing",                                     │
│      "country": "China"                                      │
│    },                                                        │
│    "tags": ["developer", "coder"],                          │
│    "createdAt": ISODate("2024-01-01")                        │
│  }                                                           │

└─────────────────────────────────────────────────────────────────┘
```

### 1.3 CRUD 操作

```javascript
// 使用 MongoDB Node.js 驱动程序
const { MongoClient } = require('mongodb');

const client = new MongoClient('mongodb://localhost:27017', {
  useNewUrlParser: true,
  useUnifiedTopology: true
});

async function main() {
  await client.connect();
  const db = client.db('mydb');
  const collection = db.collection('users');

  // ============================================
  // 1. 插入文档
  // ============================================

  // 插入单个文档
  const result = await collection.insertOne({
    username: 'zhangsan',
    email: 'zhangsan@example.com',
    age: 25,
    createdAt: new Date()
  });
  console.log('插入的文档ID:', result.insertedId);

  // 插入多个文档
  const results = await collection.insertMany([
    { username: 'lisi', email: 'lisi@example.com', age: 30 },
    { username: 'wangwu', email: 'wangwu@example.com', age: 28 },
    { username: 'zhaoliu', email: 'zhaoliu@example.com', age: 35 }
  ]);
  console.log('插入的文档数量:', results.insertedCount);

  // ============================================
  // 2. 查询文档
  // ============================================

  // 查询所有文档
  const allUsers = await collection.find({}).toArray();
  console.log('所有用户:', allUsers);

  // 条件查询
  const user = await collection.findOne({ username: 'zhangsan' });
  console.log('查找用户:', user);

  // 比较运算符
  const youngUsers = await collection.find({ age: { $gt: 25 } }).toArray(); // age > 25
  const adultUsers = await collection.find({ age: { $gte: 18, $lte: 30 } }).toArray(); // 18 <= age <= 30
  const specificUsers = await collection.find({ age: { $in: [25, 30, 35] } }).toArray(); // age in [25, 30, 35]

  // 逻辑运算符
  const complexQuery = await collection.find({
    $or: [
      { age: { $gt: 30 } },
      { username: 'zhangsan' }
    ]
  }).toArray();

  // 数组查询
  await collection.insertOne({
    username: 'test',
    tags: ['developer', 'admin', 'user']
  });

  const devUsers = await collection.find({
    tags: { $all: ['developer', 'admin'] } // 包含所有指定元素
  }).toArray();

  const specificTagUsers = await collection.find({
    tags: 'developer' // 数组中包含该元素
  }).toArray();

  // 嵌套文档查询
  await collection.insertOne({
    username: 'nested',
    address: { city: 'Beijing', country: 'China' }
  });

  const beijingUsers = await collection.find({
    'address.city': 'Beijing' // 点号访问嵌套字段
  }).toArray();

  // ============================================
  // 3. 更新文档
  // ============================================

  // 更新单个文档
  const updateResult = await collection.updateOne(
    { username: 'zhangsan' },
    {
      $set: { age: 26 }, // 设置字段值
      $currentDate: { updatedAt: true } // 设置当前日期
    }
  );
  console.log('更新数量:', updateResult.modifiedCount);

  // 更新多个文档
  await collection.updateMany(
    { age: { $lt: 30 } },
    { $inc: { age: 1 } } // 字段值增加
  );

  // 原子操作符
  await collection.updateOne(
    { username: 'zhangsan' },
    {
      $set: { 'address.city': 'Shanghai' },
      $push: { tags: 'newTag' }, // 数组添加元素
      $pull: { tags: 'oldTag' } // 数组移除元素
    }
  );

  // 替换文档
  await collection.replaceOne(
    { username: 'zhangsan' },
    { username: 'zhangsan', email: 'new@example.com', age: 27 }
  );

  // ============================================
  // 4. 删除文档
  // ============================================

  // 删除单个文档
  const deleteResult = await collection.deleteOne({ username: 'zhangsan' });
  console.log('删除数量:', deleteResult.deletedCount);

  // 删除多个文档
  await collection.deleteMany({ age: { $gt: 40 } });

  // ============================================
  // 5. 投影查询（只返回指定字段）
  // ============================================

  const projectedUsers = await collection.find(
    {},
    { projection: { username: 1, email: 1, _id: 0 } } // 只返回 username 和 email
  ).toArray();

  // ============================================
  // 6. 排序、分页、限制
  // ============================================

  const sortedUsers = await collection
    .find({})
    .sort({ age: -1, username: 1 }) // -1 降序，1 升序
    .skip(0) // 跳过前 N 条
    .limit(10) // 限制返回数量
    .toArray();
```

### 1.4 聚合管道

聚合管道是 MongoDB 最强大的功能之一，用于数据处理和转换。

```javascript
// ============================================
// 聚合管道基础
// ============================================

// $match - 筛选数据
// $group - 分组
// $project - 投影/重塑
// $sort - 排序
// $limit - 限制数量
// $skip - 跳过
// $unwind - 展开数组
// $lookup - 连接其他集合

// 示例：用户统计分析
const pipeline = [
  // 1. 筛选有效用户
  {
    $match: {
      age: { $gte: 18 },
      isActive: true
    }
  },

  // 2. 按年龄段分组统计
  {
    $group: {
      _id: {
        $switch: {
          branches: [
            { case: { $lt: ['$age', 20] }, then: '18-19' },
            { case: { $lt: ['$age', 30] }, then: '20-29' },
            { case: { $lt: ['$age', 40] }, then: '30-39' }
          ],
          default: '40+'
        }
      },
      count: { $sum: 1 }, // 统计数量
      avgAge: { $avg: '$age' }, // 平均年龄
      totalAge: { $sum: '$age' }, // 年龄总和
      users: { $push: '$username' } // 收集用户名
    }
  },

  // 3. 排序
  {
    $sort: { count: -1 }
  },

  // 4. 重塑输出
  {
    $project: {
      _id: 0,
      ageGroup: '$_id',
      count: 1,
      avgAge: { $round: ['$avgAge', 1] },
      userCount: { $size: '$users' }
    }
  }
];

const result = await collection.aggregate(pipeline).toArray();

// ============================================
// 更多聚合示例
// ============================================

// 示例1：数组展开与统计
const orders = db.collection('orders');
const orderPipeline = [
  { $unwind: '$items' }, // 展开订单中的商品数组
  {
    $group: {
      _id: '$items.productId',
      totalQuantity: { $sum: '$items.quantity' },
      orderCount: { $sum: 1 }
    }
  },
  { $sort: { totalQuantity: -1 } },
  { $limit: 10 }
];
const topProducts = await orders.aggregate(orderPipeline).toArray();

// 示例2：日期处理
const datePipeline = [
  {
    $project: {
      year: { $year: '$createdAt' },
      month: { $month: '$createdAt' },
      dayOfWeek: { $dayOfWeek: '$createdAt' },
      day: { $dayOfMonth: '$createdAt' },
      username: 1
    }
  },
  {
    $group: {
      _id: { year: '$year', month: '$month' },
      count: { $sum: 1 }
    }
  }
];
const monthlyStats = await collection.aggregate(datePipeline).toArray();

// 示例3：条件聚合
const conditionalPipeline = [
  {
    $project: {
      username: 1,
      age: 1,
      category: {
        $cond: {
          if: { $gte: ['$age', 30] },
          then: 'senior',
          else: 'junior'
        }
      }
    }
  }
];
```

### 1.5 索引优化

```javascript
// ============================================
// 创建索引
// ============================================

// 单字段索引
await collection.createIndex({ username: 1 }); // 升序
await collection.createIndex({ email: -1 }); // 降序

// 复合索引（注意字段顺序）
await collection.createIndex({ age: 1, username: -1 });

// 唯一索引
await collection.createIndex({ email: 1 }, { unique: true });

// 稀疏索引（只索引非空值）
await collection.createIndex({ phone: 1 }, { sparse: true });

// 文本索引（全文搜索）
await collection.createIndex({ description: 'text', content: 'text' });

// 多键索引（数组字段自动创建）
await collection.createIndex({ tags: 1 });

// 哈希索引（分片集群）
await collection.createIndex({ _id: 'hashed' });

// ============================================
// 查看索引
// ============================================

const indexes = await collection.indexes();
console.log('索引列表:', indexes);

// ============================================
// 删除索引
// ============================================

await collection.dropIndex('username_1');
await collection.dropIndexes(); // 删除所有索引（保留 _id）

// ============================================
// 索引实战优化
// ============================================

// 查询覆盖（索引包含所有查询字段）
// 索引: { userId: 1, name: 1, email: 1 }
// 查询: db.users.find({ userId: 123 }, { name: 1, email: 1, _id: 0 })

// 复合索引设计原则
// 1. 等值查询字段放前面
// 2. 排序字段放中间
// 3. 范围查询字段放后面
// 例如：经常按 status 查询，按 createdAt 排序
// 索引: { status: 1, createdAt: 1 }

// 使用 explain 分析查询
const explainResult = await collection
  .find({ age: { $gt: 25 }, username: 'test' })
  .explain('executionStats');

console.log('执行统计:', explainResult.executionStats);
```

### 1.6 Mongoose ODM

Mongoose 为 MongoDB 提供了对象文档映射（ODM），使代码更加优雅。

```javascript
const mongoose = require('mongoose');

// 连接 MongoDB
mongoose.connect('mongodb://localhost:27017/mydb', {
  useNewUrlParser: true,
  useUnifiedTopology: true
});

// ============================================
// 定义 Schema
// ============================================

const userSchema = new mongoose.Schema({
  username: {
    type: String,
    required: true,
    unique: true,
    minlength: 3,
    maxlength: 20,
    trim: true
  },
  email: {
    type: String,
    required: true,
    unique: true,
    lowercase: true,
    match: /^\S+@\S+\.\S+$/
  },
  age: {
    type: Number,
    min: 0,
    max: 150,
    default: 18
  },
  // 枚举类型
  role: {
    type: String,
    enum: ['user', 'admin', 'moderator'],
    default: 'user'
  },
  // 嵌套文档
  address: {
    city: String,
    country: { type: String, default: 'China' },
    zipCode: String
  },
  // 数组
  tags: [String],
  // 引用其他集合
  friends: [{
    type: mongoose.Schema.Types.ObjectId,
    ref: 'User'
  }],
  // 动态属性
  meta: {
    type: Map,
    of: String
  }
}, {
  timestamps: true, // 自动添加 createdAt 和 updatedAt
  toJSON: { virtuals: true }, // 虚拟属性转为 JSON
  toObject: { virtuals: true }
});

// ============================================
// 虚拟属性
// ============================================

userSchema.virtual('fullName').get(function() {
  return `${this.firstName} ${this.lastName}`;
});

// 虚拟populate
userSchema.virtual('friendDetails', {
  ref: 'User',
  localField: 'friends',
  foreignField: '_id',
  justOne: false
});

// ============================================
// 实例方法
// ============================================

userSchema.methods.comparePassword = async function(candidatePassword) {
  return bcrypt.compare(candidatePassword, this.password);
};

userSchema.methods.toPublicJSON = function() {
  return {
    id: this._id,
    username: this.username,
    email: this.email,
    role: this.role

};

// ============================================
// 静态方法
// ============================================

userSchema.statics.findByUsername = function(username) {
  return this.findOne({ username });
};

userSchema.statics.findActiveAdmins = function() {
  return this.find({ role: 'admin', isActive: true });
};

// ============================================
// 中间件（钩子）
// ============================================

// 保存前
userSchema.pre('save', async function(next) {
  if (this.isModified('password')) {
    this.password = await bcrypt.hash(this.password, 10);
  }
  next();
});

// 保存后
userSchema.post('save', function(doc) {
  console.log('用户已保存:', doc.username);
});

// 验证前
userSchema.pre('validate', function(next) {
  if (this.email && !this.email.endsWith('@company.com')) {
    // 自定义验证
  }
  next();
});

// ============================================
// 创建 Model
// ============================================

const User = mongoose.model('User', userSchema);

// ============================================
// 使用 Model
// ============================================

// 创建文档
const user = new User({
  username: 'zhangsan',
  email: 'zhangsan@example.com',
  age: 25,
  address: { city: 'Beijing' },
  tags: ['developer']
});

await user.save();

// 查询
const users = await User.find({ age: { $gte: 18 } })
  .sort({ createdAt: -1 })
  .limit(10)
  .select('username email')
  .exec();

// 链式调用
const user = await User.findOne({ username: 'zhangsan' })
  .populate('friends')
  .exec();

// 更新
await User.findByIdAndUpdate(userId, {

// 删除
await User.findByIdAndDelete(userId);

// ============================================
// 高级查询
// ============================================

// OR 查询
const users = await User.find({
  $or: [{ age: { $lt: 20 } }, { role: 'admin' }]

// 正则查询
```

### 2.3 SQL 基础

```sql
-- ============================================
-- 创建表
-- ============================================

CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  username VARCHAR(50) NOT NULL UNIQUE,
  email VARCHAR(100) NOT NULL UNIQUE,
  password VARCHAR(255) NOT NULL,
  age INTEGER DEFAULT 18,
  role VARCHAR(20) DEFAULT 'user',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 带约束的表
CREATE TABLE orders (
  id SERIAL PRIMARY KEY,
  user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  total_amount DECIMAL(10, 2) NOT NULL,
  status VARCHAR(20) DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT positive_amount CHECK (total_amount > 0)
);

-- 临时表
CREATE TEMPORARY TABLE temp_orders AS
SELECT * FROM orders WHERE created_at > '2024-01-01';

-- ============================================
-- CRUD 操作
-- ============================================

-- 插入数据
INSERT INTO users (username, email, password, age, role)
VALUES ('zhangsan', 'zhangsan@example.com', 'hashed_pwd', 25, 'admin');

INSERT INTO users (username, email, password)
VALUES
  ('lisi', 'lisi@example.com', 'pwd1'),
  ('wangwu', 'wangwu@example.com', 'pwd2');

-- 查询数据
SELECT * FROM users;
SELECT username, email FROM users WHERE age > 20;
SELECT * FROM users ORDER BY created_at DESC LIMIT 10;
SELECT COUNT(*), role FROM users GROUP BY role;

-- 更新数据
UPDATE users
SET age = 26, updated_at = CURRENT_TIMESTAMP
WHERE username = 'zhangsan';

-- 删除数据
DELETE FROM users WHERE id = 1;

-- ============================================
-- 关联查询
-- ============================================

-- 内连接
SELECT o.id, u.username, o.total_amount
FROM orders o
INNER JOIN users u ON o.user_id = u.id;

-- 左连接
SELECT u.username, o.id, o.total_amount
FROM users u
LEFT JOIN orders o ON u.id = o.user_id;

-- 右连接
SELECT u.username, o.id, o.total_amount
FROM users u
RIGHT JOIN orders o ON u.id = o.user_id;

-- 多表连接
SELECT u.username, o.id, p.product_name, oi.quantity
FROM users u
JOIN orders o ON u.id = o.user_id
JOIN order_items oi ON o.id = oi.order_id
JOIN products p ON oi.product_id = p.id;

-- ============================================
-- 子查询
-- ============================================

-- 找出订单超过平均金额的用户
SELECT u.username, o.total_amount
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE o.total_amount > (
  SELECT AVG(total_amount) FROM orders
);

-- 使用 EXISTS
SELECT u.username
FROM users u
WHERE EXISTS (
  SELECT 1 FROM orders o WHERE o.user_id = u.id
);

-- ============================================
-- 高级查询
-- ============================================

-- 窗口函数
SELECT
  username,
  age,
  AVG(age) OVER (PARTITION BY role) as avg_age_by_role,
  ROW_NUMBER() OVER (ORDER BY age DESC) as row_num,
  RANK() OVER (ORDER BY age DESC) as rank,
  DENSE_RANK() OVER (ORDER BY age DESC) as dense_rank
FROM users;

-- 公共表表达式 (CTE)
WITH recent_orders AS (
  SELECT * FROM orders
  WHERE created_at > CURRENT_DATE - INTERVAL '30 days'
),
user_order_counts AS (
  SELECT user_id, COUNT(*) as order_count
  FROM recent_orders
  GROUP BY user_id
)
SELECT u.username, uoc.order_count
FROM users u
JOIN user_order_counts uoc ON u.id = uoc.user_id;

-- 递归查询（组织架构）
WITH RECURSIVE org_chart AS (
  -- 基础查询
  SELECT id, name, manager_id, 1 as level
  FROM employees
  WHERE manager_id IS NULL

  UNION ALL

  -- 递归部分
  SELECT e.id, e.name, e.manager_id, oc.level + 1
  FROM employees e
  JOIN org_chart oc ON e.manager_id = oc.id
)
SELECT * FROM org_chart;
```

### 2.4 事务与锁

```sql
-- ============================================
-- 事务控制
-- ============================================

-- 开始事务
BEGIN;

-- 或
START TRANSACTION;

-- 提交事务
COMMIT;

-- 回滚事务
ROLLBACK;

-- 保存点
SAVEPOINT savepoint_name;
ROLLBACK TO SAVEPOINT savepoint_name;
RELEASE SAVEPOINT savepoint_name;

-- ============================================
-- 事务示例
-- ============================================

BEGIN;

-- 扣款
UPDATE accounts
SET balance = balance - 100
WHERE user_id = 1;

-- 充值
UPDATE accounts
SET balance = balance + 100
WHERE user_id = 2;

-- 检查并提交/回滚
IF @@ERROR = 0 THEN
  COMMIT;
ELSE
  ROLLBACK;
END;

-- ============================================
-- 锁机制
-- ============================================

-- 行级锁
SELECT * FROM users
WHERE id = 1
FOR UPDATE;  -- 排他锁，阻止其他事务修改

SELECT * FROM users
WHERE id = 1
FOR SHARE;   -- 共享锁，允许其他事务读取

-- 表级锁
LOCK TABLE users IN ACCESS EXCLUSIVE MODE;

-- 锁模式
-- ACCESS SHARE: SELECT
-- ROW SHARE: SELECT FOR UPDATE
-- ROW EXCLUSIVE: UPDATE, DELETE, INSERT
-- SHARE UPDATE EXCLUSIVE: ANALYZE, VACUUM
-- SHARE: CREATE INDEX
-- SHARE ROW EXCLUSIVE: ALTER TABLE
-- ACCESS EXCLUSIVE: DROP TABLE, TRUNCATE

-- 查看锁
SELECT * FROM pg_locks
WHERE relation = 'users'::regclass;

-- 查看阻塞查询
SELECT
  blocked_locks.pid AS blocked_pid,
  blocking_locks.pid AS blocking_pid,
  blocked_activity.query AS blocked_query,
  blocking_activity.query AS blocking_query
FROM pg_catalog.pg_locks blocked_locks
JOIN pg_catalog.pg_locks blocking_locks
  ON blocked_locks.relation = blocking_locks.relation
JOIN pg_catalog.pg_stat_activity blocked_activity
  ON blocked_locks.pid = blocked_activity.pid
JOIN pg_catalog.pg_stat_activity blocking_activity
  ON blocking_locks.pid = blocking_activity.pid
WHERE NOT blocked_locks.granted;

-- ============================================
-- 隔离级别
-- ============================================

-- 设置隔离级别
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
SET SESSION CHARACTERISTICS AS TRANSACTION ISOLATION LEVEL SERIALIZABLE;

-- 隔离级别说明
-- READ UNCOMMITTED: 可能发生脏读（PostgreSQL 不支持）
-- READ COMMITTED: 只能读取已提交的数据（默认）
-- REPEATABLE READ: 保证同一事务中多次读取一致
-- SERIALIZABLE: 最高隔离级别，事务串行执行
```

### 3.4 缓存策略

```javascript
// ============================================
// 1. Cache-Aside（旁路缓存）- 最常用
// ============================================

/*
  读流程:
  1. 查询缓存
  2. 缓存命中 → 返回数据
  3. 缓存未命中 → 查询数据库
  4. 存入缓存 → 返回数据

  写流程:
  1. 更新数据库
  2. 删除缓存（或更新缓存）
*/

async function getUserById(userId) {
  // 1. 查询缓存
  const cacheKey = `user:${userId}`;
  const cached = await client.get(cacheKey);

  if (cached) {
    return JSON.parse(cached);
  }

  // 2. 缓存未命中，查询数据库
  const user = await db.users.findById(userId);

  if (user) {
    // 3. 存入缓存，设置过期时间
    await client.setEx(cacheKey, 3600, JSON.stringify(user));
  }

  return user;
}

async function updateUser(userId, data) {
  // 1. 更新数据库
  const user = await db.users.update(userId, data);

  // 2. 删除缓存
  await client.del(`user:${userId}`);

  return user;
}

// ============================================
// 2. Write-Through（写穿透）
// ============================================

async function updateUserWriteThrough(userId, data) {
  // 1. 同时更新缓存和数据库
  await Promise.all([
    client.set(`user:${userId}`, JSON.stringify(data)),
    db.users.update(userId, data)
  ]);
}

// ============================================
// 3. Write-Behind（异步写入）
// ============================================

const writeQueue = [];

async function updateUserWriteBehind(userId, data) {
  // 1. 先更新缓存
  await client.set(`user:${userId}`, JSON.stringify(data));

  // 2. 加入写队列，异步写入数据库
  writeQueue.push({ table: 'users', id: userId, data });
}

// 后台任务处理写队列
async function processWriteQueue() {
  while (writeQueue.length > 0) {
    const write = writeQueue.shift();
    await db[write.table].update(write.id, write.data);
  }
}
```

### 3.6 消息队列

```javascript
// ============================================
// Redis 消息队列实现
// ============================================

// 1. 简单队列
const QUEUE_NAME = 'tasks:queue';

// 生产者 - 添加任务
async function addTask(task) {
  const taskId = crypto.randomUUID();
  const taskData = JSON.stringify({ id: taskId, ...task });

  // LPUSH 从左边入队，RPUSH 从右边入队
  await client.lPush(QUEUE_NAME, taskData);

  return taskId;
}

// 消费者 - 获取任务
async function processTask() {
  // BRPOP 阻塞等待，RPOP 非阻塞
  const result = await client.brPop(QUEUE_NAME, 10); // 10秒超时

  if (result) {
    const task = JSON.parse(result.element);
    console.log('处理任务:', task);

    // 模拟处理
    await doTask(task);

    return task;
  }

  return null;
}

// 2. 可靠队列（带确认）
const PROCESSING_QUEUE = 'tasks:processing';

async function processTaskWithAck() {
  // 从队列取任务
  const result = await client.brPop(QUEUE_NAME, 10);

  if (result) {
    const task = JSON.parse(result.element);

    // 移到处理中队列
    await client.lPush(PROCESSING_QUEUE, JSON.stringify({
      ...task,
      startedAt: Date.now()
    }));

    try {
      // 处理任务
      await doTask(task);

      // 成功，从处理中队列移除
      await client.lRem(PROCESSING_QUEUE, 1, JSON.stringify(task));
    } catch (error) {
      // 失败，移回原队列
      await client.lPush(QUEUE_NAME, JSON.stringify(task));
      await client.lRem(PROCESSING_QUEUE, 1, JSON.stringify(task));
    }
  }
}

// 3. 优先级队列
const PRIORITY_QUEUES = {
  HIGH: 'tasks:priority:high',
  MEDIUM: 'tasks:priority:medium',
  LOW: 'tasks:priority:low'
};

async function addPriorityTask(task, priority = 'MEDIUM') {
  const queue = PRIORITY_QUEUES[priority];
  await client.lPush(queue, JSON.stringify(task));
}

async function processPriorityTask() {
  // 按优先级顺序尝试获取
  for (const queue of Object.values(PRIORITY_QUEUES)) {
    const result = await client.brPop(queue, 1);
    if (result) {
      return JSON.parse(result.element);
    }
  }
  return null;
}

// 4. 发布/订阅
const PUBSUB_CHANNEL = 'notifications';

async function publishNotification(notification) {
  await client.publish(PUBSUB_CHANNEL, JSON.stringify(notification));
}

async function subscribeNotifications(callback) {
  const subscriber = client.duplicate();

  await subscriber.connect();

  await subscriber.subscribe(PUBSUB_CHANNEL, (message) => {
    const notification = JSON.parse(message);
    callback(notification);
  });

  return subscriber;
}
```

## 4. 数据库设计原则

### 4.1 设计范式

```sql
-- 第一范式 (1NF) - 原子性
-- 每个字段都是不可再分的原子值
-- ❌ 错误：address: "北京市,朝阳区,建国路"
-- ✅ 正确：city, district, street 分开

-- 第二范式 (2NF) - 消除部分依赖
-- 在满足1NF基础上，非主键字段完全依赖主键
-- ❌ 错误：(order_id, product_id) 为主键，product_name 只依赖 product_id
-- ✅ 正确：product_id 作为外键

-- 第三范式 (3NF) - 消除传递依赖
-- 在满足2NF基础上，非主键字段不依赖其他非主键字段
-- ❌ 错误：user 表中有 department_id 和 department_name（依赖 department_id）
-- ✅ 正确：department 单独建表
```

### 4.2 索引设计原则

```sql
-- 选择性高的字段优先索引
-- ❌ 低选择性：gender (只有 M/F)
-- ✅ 高选择性：user_id, email, phone

-- 复合索引顺序
-- 等值查询字段 → 排序字段 → 范围查询字段
CREATE INDEX idx_user_status_created
ON users(status, created_at DESC);

-- 覆盖索引
-- 索引包含所有查询字段，避免回表
CREATE INDEX idx_user_cover
ON users(status) INCLUDE (username, email, age);

-- 避免过多索引
-- 每次写入都要维护索引，影响性能
```

```javascript
import { RedisService } from './redis.service';

@Injectable()
export class CacheService {
  constructor(private redis: RedisService) {}

  // 用户缓存
  async getUser(userId: string) {
    const key = `user:${userId}`;
    const cached = await this.redis.get(key);

    if (cached) {
      return JSON.parse(cached);
    }

    return null;
  }

  async setUser(userId: string, user: object, ttl = 1800) {
    const key = `user:${userId}`;
    await this.redis.setEx(key, ttl, JSON.stringify(user));
  }

  async invalidateUser(userId: string) {
    const key = `user:${userId}`;
    await this.redis.del(key);
  }

  // 分页缓存
  async getPageList(cacheKey: string, page: number, limit: number) {
    const key = `list:${cacheKey}:${page}:${limit}`;
    return this.redis.get(key);
  }

  async setPageList(cacheKey: string, page: number, limit: number, data: object[], ttl = 300) {
    const key = `list:${cacheKey}:${page}:${limit}`;
    await this.redis.setEx(key, ttl, JSON.stringify(data));
  }

  // 排行榜
  async updateLeaderboard(scoreboard: string, member: string, score: number) {
    const key = `leaderboard:${scoreboard}`;
    await this.redis.zAdd(key, { score, value: member });
  }

  async getTopPlayers(scoreboard: string, topN: number) {
    const key = `leaderboard:${scoreboard}`;
    return this.redis.zRangeWithScores(key, 0, topN - 1, { reverse: true });
  }

  // 分布式锁
  async acquireLock(key: string, ttl = 10): Promise<boolean> {
    const lockKey = `lock:${key}`;
    const result = await this.redis.set(lockKey, '1', {
      NX: true,
      EX: ttl
    });
    return result === 'OK';
  }

  async releaseLock(key: string) {
    const lockKey = `lock:${key}`;
    await this.redis.del(lockKey);
  }

  // 限流
  async rateLimit(key: string, limit: number, windowSeconds: number): Promise<boolean> {
    const rateKey = `rate:${key}`;
    const current = await this.redis.incr(rateKey);

    if (current === 1) {
      await this.redis.expire(rateKey, windowSeconds);
    }

    return current <= limit;
  }
}
```

**面试题6：Redis如何实现分布式锁？**

**参考答案：**

```javascript
// Redis 分布式锁实现

class DistributedLock {
  constructor(redis, options = {}) {
    this.redis = redis;
    this.lockKey = options.lockKey || 'lock';
    this.expireMs = options.expireMs || 30000;  // 30秒过期
    this.retryDelayMs = options.retryDelayMs || 200;
    this.retryCount = options.retryCount || 3;
  }

  // 获取锁
  async acquire(value) {
    const lockValue = value || `${process.pid}-${Date.now()}`;
    for (let i = 0; i < this.retryCount; i++) {
      // SET key value NX PX milliseconds
      // NX: 仅当key不存在时设置
      // PX: 过期时间（毫秒）
      const result = await this.redis.set(
        this.lockKey,
        lockValue,
        'NX',
        'PX',
        this.expireMs
      );

      if (result === 'OK') {
        return lockValue;  // 获取成功
      }

      // 重试
      await this.sleep(this.retryDelayMs);
    }
    return null;  // 获取失败
  }

  // 释放锁（Lua脚本保证原子性）
  async release(lockValue) {
    // Lua脚本：只有持有锁的进程才能释放
    const script = `
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("del", KEYS[1])
      else
        return 0
      end
    `;
    await this.redis.eval(script, 1, this.lockKey, lockValue);
  }

  // 延长锁（自动续期）
  async extend(lockValue, extendMs) {
    const script = `
      if redis.call("get", KEYS[1]) == ARGV[1] then
        return redis.call("pexpire", KEYS[1], ARGV[2])
      else
        return 0
      end
    `;
    return await this.redis.eval(script, 1, this.lockKey, lockValue, extendMs);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// 使用示例
const lock = new DistributedLock(redis, { lockKey: 'order:123', expireMs: 10000 });

async function createOrder() {
  const lockValue = await lock.acquire();
  if (!lockValue) {
    throw new Error('获取锁失败，请重试');
  }

  try {
    // 业务逻辑
    const order = await doCreateOrder();
    return order;
  } finally {
    await lock.release(lockValue);  // 释放锁
  }
}
```

## 6. PostgreSQL连接池配置

### 6.1 连接池核心概念

连接池是数据库性能优化的关键技术，它通过复用数据库连接来减少连接创建销毁的开销。在高并发场景下，频繁创建和关闭数据库连接会产生显著的性能开销，连接池通过维护一组持久连接，使应用能够高效地复用这些连接。

连接池的工作原理是在应用启动时创建一定数量的数据库连接，并将其放入连接池中。当应用需要执行数据库操作时，从池中获取一个可用连接；操作完成后，将连接归还到池中而不是关闭。这种方式避免了频繁创建销毁连接的开销，显著提升了数据库访问性能。

PostgreSQL的连接池配置需要考虑多个参数，包括最小连接数、最大连接数、连接超时时间、空闲连接回收时间等。合理配置这些参数能够使连接池在保证性能的同时避免资源浪费。

### 6.2 Node.js连接池配置

```javascript
// PostgreSQL连接池配置详解
const { Pool } = require('pg');

// 创建连接池配置
const poolConfig = {
    // 数据库连接信息
    host: process.env.DB_HOST || 'localhost',
    port: parseInt(process.env.DB_PORT || '5432'),
    database: process.env.DB_NAME || 'mydb',
    user: process.env.DB_USER || 'admin',
    password: process.env.DB_PASSWORD || 'password',

    // 连接池大小配置
    max: 20,              // 最大连接数，超过这个数量的请求会等待可用连接
    min: 5,               // 最小连接数，池会保持这个数量的连接
    idleTimeoutMillis: 30000,  // 空闲连接超时（毫秒），超过这个时间没有使用的连接会被关闭
    connectionTimeoutMillis: 2000,  // 连接超时（毫秒），获取连接时的最大等待时间

    // 性能优化配置
    allowExitOnIdle: false,  // 设为true时，空闲时允许Node.js退出
    reapIntervalMillis: 1000,  // 检查空闲连接的间隔（毫秒）

    // SSL配置（生产环境必须启用）
    ssl: process.env.NODE_ENV === 'production' ? {
        rejectUnauthorized: true,
        cert: process.env.DB_SSL_CERT,
        key: process.env.DB_SSL_KEY,
        ca: process.env.DB_SSL_CA
    } : false
};

// 创建连接池实例
const pool = new Pool(poolConfig);

// 连接池事件监听
pool.on('connect', (client) => {
    // 新连接建立时的回调
    console.log('新数据库连接已建立，当前连接数:', pool.totalCount);
});

pool.on('acquire', (client) => {
    // 连接从池中被获取时的回调
    console.log('连接被获取');
});

pool.on('remove', (client) => {
    // 连接从池中移除时的回调
    console.log('连接已从池中移除');
});

pool.on('error', (err, client) => {
    // 连接错误时的回调
    console.error('数据库连接错误:', err.message);
});

// 查询执行函数
async function executeQuery(text, params) {
    // 从池中获取连接并执行查询
    const start = Date.now();
    const result = await pool.query(text, params);
    const duration = Date.now() - start;

    // 记录查询性能
    console.log('执行查询', { text, duration, rows: result.rowCount });
    return result;
}

// 事务执行函数
async function executeTransaction(callback) {
    const client = await pool.connect();
    try {
        // 开始事务
        await client.query('BEGIN');

        // 执行回调函数中的事务操作
        const result = await callback(client);

        // 提交事务
        await client.query('COMMIT');
        return result;
    } catch (error) {
        // 回滚事务
        await client.query('ROLLBACK');
        throw error;
    } finally {
        // 释放连接回池
        client.release();
    }
```

```python
        """
        执行查询并返回字典列表
        """

            # 使用参数化查询防止SQL注入
            rows = await connection.fetch(query, *args)
            return [dict(row) for row in rows]

        """
        执行查询并返回单条记录

            return result

        执行事务操作
        callback是异步函数，接收connection参数
```

```python
        # 注意：Redis有序集合的score是浮点数
        return await self.client.zadd(name, mapping)

    async def zrangebyscore(
        self,
        name: str,
        min_score: float,
        max_score: float,
        withscores: bool = False
    ) -> List[Any]:
        """按分数范围获取有序集合"""
        result = await self.client.zrangebyscore(
            name, min_score, max_score, withscores=withscores
        )
        if withscores:
            return [(self._deserialize(m), s) for m, s in result]
        return [self._deserialize(m) for m in result]

        )
        if withscores:

        return [self._deserialize(m) for m in result]

    async def zrank(self, name: str, member: Any) -> Optional[int]:

    # 分布式锁
    async def acquire_lock(
        self,
        lock_key: str,
        expire_ms: int = 30000,
        retry_times: int = 3,
        retry_delay_ms: int = 200
    ) -> Optional[str]:
        """
        获取分布式锁
        使用SET NX PX实现原子性加锁
        """
        lock_value = f"{asyncio.current_task().get_name()}-{id(self)}-{asyncio.get_event_loop().time()}"

        for _ in range(retry_times):
            # SET key value NX PX milliseconds
            if await self.client.set(
                lock_key, lock_value, nx=True, px=expire_ms
            ):
                return lock_value
            await asyncio.sleep(retry_delay_ms / 1000)

        return None

    async def release_lock(self, lock_key: str, lock_value: str) -> bool:
        """
        释放分布式锁
        使用Lua脚本保证原子性
        """
        lua_script = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        result = await self.client.eval(lua_script, 1, lock_key, lock_value)
        return result == 1

    # 缓存装饰器
    def cached(self, expire_seconds: int = 3600, key_prefix: str = ""):
        """缓存装饰器"""
        def decorator(func):
            @functools.wraps(func)
            async def wrapper(*args, **kwargs):
                # 生成缓存键
                cache_key = f"{key_prefix}:{func.__name__}:{str(args)}:{str(kwargs)}"

                # 尝试获取缓存
                cached_value = await self.get(cache_key)
                if cached_value is not None:
                    return cached_value

                # 调用原函数
                result = await func(*args, **kwargs)

                # 设置缓存
                await self.setex(cache_key, expire_seconds, result)

                return result
            return wrapper
        return decorator

    # 管道操作
    async def pipeline(self):
        """创建管道"""
        return self.client.pipeline()

    async def close(self):
        """关闭连接"""
        if self.client:
            await self.client.close()
        if self.pool:
            await self.pool.disconnect()
        print("Redis连接已关闭")

    @staticmethod
    def _serialize(value: Any) -> str:
        """序列化值为JSON字符串"""
        return json.dumps(value, ensure_ascii=False)

    @staticmethod
    def _deserialize(value: Optional[str]) -> Optional[Any]:
        """反序列化JSON字符串"""
        if value is None:
            return None
        try:
            return json.loads(value)
        except (json.JSONDecodeError, TypeError):
            return value
```

## 8. Elasticsearch查询优化

### 8.1 千万级数据查询背景

Elasticsearch是一个基于Lucene的分布式搜索和分析引擎，在处理大规模数据时表现出色。当数据量达到千万级别时，查询性能优化变得尤为重要。本节将深入分析千万级数据的top100查询执行过程，探讨查询优化策略。

在千万级数据场景下，查询优化需要考虑多个方面：索引设计、查询语法、分页方式、聚合优化、缓存策略等。合理的优化可以将查询时间从秒级降低到毫秒级，显著提升用户体验。

```javascript
/**
 * Elasticsearch查询执行过程详解
 *
 * 当执行一个查询请求时，ES内部会经历以下阶段：
 *
 * 1. 请求接收阶段
 *    - 客户端发送HTTP请求到协调节点
 *    - 协调节点验证请求格式和权限
 *
 * 2. 查询解析阶段
 *    - 解析查询DSL
 *    - 确定需要查询的索引和分片
 *
 * 3. 分片查询阶段
 *    - 协调节点将请求广播到所有相关分片
 *    - 每个分片独立执行查询
 *
 * 4. 取回阶段
 *    - 从各分片取回匹配的文档ID和排序值
 *    - 合并结果并排序
 *
 * 5. 计算阶段
 *    - 计算聚合（如果需要）
 *    - 应用后过滤
 *
 * 6. 返回阶段
 *    - 格式化结果
 *    - 返回给客户端

class ESQueryExecutor {
    constructor(client) {
        this.client = client;
    }

    /**
     * Top100查询：获取评分最高的前100条记录
     * 这是最常见的查询场景，用于排行榜、推荐列表等
     */
    async top100Query(indexName, query, sortField = 'score', size = 100) {
        const startTime = Date.now();

        // 第一阶段：构建查询DSL
        const searchQuery = {
            index: indexName,
            body: {
                // 查询条件
                query: {
                    bool: {
                        must: [
                            { match: { title: query } }
                        ],
                        filter: [
                            { term: { status: 'published' } },
                            { range: { publishDate: { gte: '2024-01-01' } } }
                        ]
                    }
                },
                // 排序：按评分降序
                sort: [
                    { [sortField]: { order: 'desc' } },
                    { _score: { order: 'desc' } }
                ],
                // 返回字段
                _source: [
                    'id', 'title', 'content', 'author',
                    'score', 'publishDate', 'tags'
                ],
                // 分页大小
                size: size,
                // 高亮显示
                highlight: {
                    fields: {
                        title: {},
                        content: { fragment_size: 150, number_of_fragments: 3 }
                    },
                    pre_tags: ['<em>'],
                    post_tags: ['</em>']
                },
                // 聚合统计
                aggs: {
                    // 按标签聚合
                    tags: {
                        terms: {
                            field: 'tags.keyword',
                            size: 10
                        }
                    },
                    // 平均评分
                    avgScore: {
                        avg: { field: 'score' }
                    },
                    // 评分分布
                    scoreRanges: {
                        range: {
                            field: 'score',
                            ranges: [
                                { key: '0-50', from: 0, to: 50 },
                                { key: '50-80', from: 50, to: 80 },
                                { key: '80-100', from: 80, to: 100 }
                            ]
                        }
                    }
                }
            },
            // 性能优化参数
            requestParams: {
                track_total_hits: false  // 不需要精确总数，提升性能
            }
        };

        // 第二阶段：执行查询
        const result = await this.client.search(searchQuery);

        // 第三阶段：处理结果
        const hits = result.hits.hits.map(hit => ({
            id: hit._id,
            score: hit._score,
            source: hit._source,
            highlight: hit.highlight
        }));

        const aggregations = {
            tags: result.aggregations.tags.buckets,
            avgScore: result.aggregations.avgScore.value,
            scoreRanges: result.aggregations.scoreRanges.buckets
        };

        const duration = Date.now() - startTime;

        return {
            total: result.hits.total.value,
            hits,
            aggregations,
            took: result.took,       // ES内部执行时间（毫秒）
            duration,                // 客户端总耗时
            maxScore: result.hits.max_score
        };
    }

    /**
     * 分页查询优化：深度分页问题及解决方案
     *
     * 深度分页问题：
     * - 使用from/size进行分页时，from+size过大会导致性能急剧下降
     * - 原因：每个分片需要返回from+size条数据，协调节点需要合并大量数据
     *
     * 解决方案：
     * 1. 使用search_after进行深分页（推荐）
     * 2. 使用scroll API进行全量遍历
     * 3. 控制最大分页深度
     */
    async searchAfterQuery(indexName, lastSortValues, size = 100) {
        return await this.client.search({
            index: indexName,
            body: {
                query: { match_all: {} },
                sort: [
                    { publishDate: 'desc' },
                    { _id: 'asc' }  // 第二排序字段保证唯一性
                ],
                // search_after使用上一页最后一条的排序值
                search_after: lastSortValues,
                size: size
            }
        });
    }

    /**
     * Scroll API：用于全量数据导出
     * 适用于数据迁移、批量处理等场景
     */
    async scrollExport(indexName, batchSize = 5000) {
        const scrollTimeout = '5m';  // scroll上下文保持时间

        // 初始化scroll
        let response = await this.client.search({
            index: indexName,
            scroll: scrollTimeout,
            body: {
                query: { match_all: {} },
                sort: [{ _id: 'asc' }],
                size: batchSize
            }
        });

        const allDocuments = [];
        let scrollId = response._scroll_id;
        let hits = response.hits.hits;

        // 持续滚动获取数据
        while (hits.length > 0) {
            allDocuments.push(...hits.map(hit => hit._source));

            response = await this.client.scroll({
                scroll_id: scrollId,
                scroll: scrollTimeout
            });

            scrollId = response._scroll_id;
            hits = response.hits.hits;
        }

        // 清除scroll上下文
        await this.client.clearScroll({ scroll_id: scrollId });

        return allDocuments;
    }

    /**
     * 聚合查询优化
     */
    async optimizedAggregation(indexName) {
        return await this.client.search({
            index: indexName,
            body: {
                size: 0,  // 只返回聚合结果，不返回文档
                query: {
                    bool: {
                        filter: [
                            { range: { timestamp: { gte: 'now-7d/d' } } }
                        ]
                    }
                },
                aggs: {
                    // 使用cardinality近似计数代替精确计数
                    uniqueUsers: {
                        cardinality: {
                            field: 'userId',
                            precision_threshold: 40000  // 精度/内存权衡
                        }

                    // 嵌套聚合

                        terms: { field: 'category.keyword', size: 20 },
                        aggs: {
                            avgPrice: { avg: { field: 'price' } },
                            topProducts: {
                                top_hits: {
                                    size: 5,
                                    sort: [{ sales: 'desc' }],
                                    _source: ['name', 'price', 'sales']
                                }
                            }
                        }
                    },
                    // 百分比聚合
                    pricePercentiles: {
                        percentiles: {
                            field: 'price',
                            percents: [25, 50, 75, 90, 95, 99]
                        }
                    }
                }
            }
        });
    }
}
```

### 8.3 Python ES查询优化

```python
# Python Elasticsearch查询优化详解
from elasticsearch import Elasticsearch, helpers
from typing import List, Dict, Any, Optional
import time

class ESQueryOptimizer:
    """
    Elasticsearch查询优化器
    提供高效的查询方法和性能分析功能
    """

    def __init__(self, hosts: List[str], timeout: int = 30):
        """
        初始化ES客户端

        参数:
            hosts: ES集群节点地址列表
            timeout: 请求超时时间（秒）
        """
        self.client = Elasticsearch(
            hosts,
            request_timeout=timeout,
            max_retries=3,
            retry_on_timeout=True
        )

    def top100_query(
        self,
        index: str,
        must_conditions: List[Dict],
        filter_conditions: Optional[List[Dict]] = None,
        sort_field: str = 'score',
        size: int = 100
    ) -> Dict[str, Any]:
        """
        Top100查询：获取评分最高的前100条记录

        执行过程分析：
        1. 协调节点接收请求，解析查询DSL
        2. 将查询分发到所有相关分片
        3. 每个分片执行倒排索引查询，找到匹配的文档
        4. 每个分片计算文档得分并排序
        5. 每个分片返回top N条记录的ID和得分
        6. 协调节点合并所有分片结果，再次排序
        7. 返回最终的top N结果

        参数:
            index: 索引名称
            must_conditions: 必须满足的查询条件
            filter_conditions: 过滤条件（不参与评分，更高效）
            sort_field: 排序字段
            size: 返回数量

        返回:
            包含查询结果和性能信息的字典
        """
        start_time = time.time()

        # 构建查询DSL
        query = {
            'bool': {
                'must': must_conditions
            }
        }

        # 添加过滤条件（使用filter上下文，不计算相关性得分）
        if filter_conditions:
            query['bool']['filter'] = filter_conditions

        # 构建搜索请求
        search_body = {
            'query': query,
            'sort': [
                {sort_field: {'order': 'desc'}},
                {'_score': {'order': 'desc'}}
            ],
            '_source': [
                'id', 'title', 'content', 'author',
                'score', 'publishDate', 'tags', 'category'
            ],
            'size': size,
            'highlight': {
                'fields': {
                    'title': {},
                    'content': {
                        'fragment_size': 150,
                        'number_of_fragments': 3
                    }
                },
                'pre_tags': ['<em>'],
                'post_tags': ['</em>']
            },
            'aggs': {
                'category_stats': {
                    'terms': {
                        'field': 'category.keyword',
                        'size': 20
                    },
                    'aggs': {
                        'avg_score': {
                            'avg': {'field': 'score'}
                        },
                        'max_score': {
                            'max': {'field': 'score'}
                        }
                    }
                },
                'score_distribution': {
                    'histogram': {
                        'field': 'score',
                        'interval': 10
                    }
                },
                'total_tags': {
                    'cardinality': {'field': 'tags.keyword'}
                }
            }
        }

        # 执行搜索
        response = self.client.search(
            index=index,
            body=search_body,
            # 不追踪精确总数，提升性能
            track_total_hits=False
        )

        # 处理结果
        hits = []
        for hit in response['hits']['hits']:
            hits.append({
                'id': hit['_id'],
                'score': hit['_score'],
                'source': hit['_source'],
                'highlight': hit.get('highlight', {})
            })

        # 提取聚合结果
        aggregations = {
            'by_category': response['aggregations']['category_stats']['buckets'],
            'score_histogram': response['aggregations']['score_distribution']['buckets'],
            'unique_tags': response['aggregations']['total_tags']['value']
        }

        total_time = time.time() - start_time

        return {
            'total': response['hits']['total']['value'],
            'hits': hits,
            'aggregations': aggregations,
            'es_took_ms': response['took'],
            'total_time_ms': int(total_time * 1000),
            'max_score': response['hits']['max_score']
        }

    def search_after_pagination(
        self,
        index: str,
        sort_values: Optional[List],
        page_size: int = 100
    ) -> Dict[str, Any]:
        """
        使用search_after进行深度分页

        search_after原理：
        - 使用上一页最后一条文档的排序值作为起点
        - 避免了from+size的性能问题
        - 支持无限深度分页

        适用场景：
        - 无限滚动列表
        - 实时排行榜
        - 大数据量导出
        """
        search_body = {
            'query': {'match_all': {}},
            'sort': [
                {'timestamp': 'desc'},
                {'_id': 'asc'}  # 第二排序字段确保唯一性
            ],
            'size': page_size
        }

        # 添加search_after
        if sort_values:
            search_body['search_after'] = sort_values

        response = self.client.search(index=index, body=search_body)

        hits = [hit['_source'] for hit in response['hits']['hits']]
        last_sort = response['hits']['hits'][-1]['sort'] if hits else None

        return {
            'hits': hits,
            'next_sort': last_sort,
            'has_more': len(hits) == page_size
        }

    def scroll_query(
        self,
        index: str,
        query: Dict,
        batch_size: int = 5000,
        scroll_timeout: str = '5m'
    ) -> List[Dict]:
        """
        使用scroll API进行全量数据遍历

        scroll原理：
        - 在第一次请求时创建scroll上下文
        - 后续请求使用scroll_id获取下一批数据
        - 适用于数据迁移、批量处理

        注意：
        - scroll上下文会占用内存，记得及时清理
        - 不适合实时查询，scroll_timeout不宜过长
        """
        # 初始化scroll
        response = self.client.search(
            index=index,
            scroll=scroll_timeout,
            body={
                'query': query,
                'sort': [{'_id': 'asc'}],
                'size': batch_size
            }
        )

        scroll_id = response['_scroll_id']
        all_hits = [hit['_source'] for hit in response['hits']['hits']]

        # 持续滚动获取数据
        while response['hits']['hits']:
            response = self.client.scroll(
                scroll_id=scroll_id,
                scroll=scroll_timeout
            )
            scroll_id = response['_scroll_id']
            all_hits.extend([hit['_source'] for hit in response['hits']['hits']])

        # 清除scroll上下文
        self.client.clear_scroll(scroll_id=scroll_id)

        return all_hits

    def bulk_index(self, index: str, documents: List[Dict]) -> Dict:
        """
        批量写入文档

        bulk API原理：
        - 将多个操作合并为一个HTTP请求
        - 减少网络往返次数
        - 提高写入吞吐量
        """
        def generate_actions():
            for doc in documents:
                yield {
                    '_index': index,
                    '_id': doc.get('id'),
                    '_source': doc
                }

        success, failed = helpers.bulk(
            self.client,
            generate_actions(),
            raise_on_error=False,
            stats_only=True
        )

        return {
            'success': success,
            'failed': len(failed) if isinstance(failed, list) else failed,
            'errors': failed if isinstance(failed, list) else []
        }

    def analyze_query_performance(self, index: str, query: Dict) -> Dict:
        """
        使用explain分析查询性能
        """
        # 获取查询计划
        plan = self.client.indices.analyze(index=index, body={
            'explain': True,
            'query': query
        })

        return plan

    def close(self):
        """关闭客户端连接"""
        self.client.close()

# 使用示例
def main():
    # 创建优化器实例
    optimizer = ESQueryOptimizer(
        hosts=['http://localhost:9200'],

    )
```

## 9. 数据库索引设计

### 9.1 索引设计原则

数据库索引是提升查询性能的关键技术，但不当的索引设计反而会带来负面影响。索引设计需要在查询性能和写入性能、存储空间之间取得平衡。本节将深入探讨索引设计原则和最佳实践。

索引设计的核心原则包括：选择高选择性字段、避免冗余索引、考虑查询模式、合理使用复合索引。PostgreSQL支持多种索引类型，包括B-tree、Hash、GiST、GIN等，每种索引适用于不同的场景。

### 9.2 PostgreSQL索引设计

```sql
-- PostgreSQL索引设计详解

-- ============================================
-- 1. 基础索引类型
-- ============================================

-- B-tree索引（默认）：适用于等值查询和范围查询
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_orders_created_at ON orders(created_at);
CREATE INDEX idx_products_price ON products(price);

-- 复合索引：多列组合索引，遵循最左前缀原则
CREATE INDEX idx_orders_user_status_date
ON orders(user_id, status, created_at);

-- 部分索引：只索引满足条件的行，减少索引大小
CREATE INDEX idx_orders_pending
ON orders(created_at)
WHERE status = 'pending';

-- 表达式索引：索引函数计算结果
CREATE INDEX idx_users_email_lower ON users(LOWER(email));
CREATE INDEX idx_orders_year ON orders(EXTRACT(YEAR FROM created_at));

-- ============================================
-- 2. 高级索引类型
-- ============================================

-- GiST索引：适用于几何数据类型、全文搜索
CREATE INDEX idx_products_geometry ON products USING GIST(geometry);

-- GIN索引：适用于数组、JSON、全文搜索
CREATE INDEX idx_products_tags ON products USING GIN(tags);
CREATE INDEX idx_products_attrs ON products USING GIN(attrs jsonb_path_ops);

-- Hash索引：适用于大数据的等值查询
CREATE INDEX idx_sessions_token ON sessions USING HASH(token);

-- ============================================
-- 3. 唯一索引

-- 唯一索引：保证数据唯一性
CREATE UNIQUE INDEX idx_users_email_unique ON users(email);

CREATE TABLE orders (
    id BIGSERIAL PRIMARY KEY,

-- 针对用户订单列表查询（覆盖索引）
CREATE INDEX idx_orders_user_created

-- 针对状态和时间范围查询
CREATE INDEX idx_orders_status_created
ON orders(status, created_at);

-- 针对用户状态统计（部分索引）
CREATE INDEX idx_orders_user_pending
```

### 9.3 索引设计最佳实践

```javascript
// Node.js索引管理工具
const { Pool } = require('pg');

class IndexManager {
    constructor(pool) {
        this.pool = pool;
    }

    /**
     * 分析查询并建议索引
     * 使用pg_stat_statement分析慢查询
     */
    async analyzeSlowQueries(limit = 10) {
        const query = `
            SELECT
                query,
                calls,
                mean_exec_time,
                total_exec_time,
                rows,
                shared_blks_hit,
                shared_blks_read
            FROM pg_stat_statements
            WHERE query LIKE '%SELECT%'
            ORDER BY mean_exec_time DESC
            LIMIT $1
        `;

        const result = await this.pool.query(query, [limit]);
        return result.rows;
    }

    /**
     * 生成索引建议
     * 分析查询条件，生成合适的索引
     */
    generateIndexSuggestions(queries) {
        const suggestions = [];
        const tableIndexes = new Map();

        for (const query of queries) {
            // 解析WHERE条件中的字段
            const whereFields = this.extractWhereFields(query);
            const orderFields = this.extractOrderFields(query);
            const tableName = this.extractTableName(query);

            if (!tableIndexes.has(tableName)) {
                tableIndexes.set(tableName, new Set());
            }

            // 生成单字段索引建议
            for (const field of whereFields) {
                tableIndexes.get(tableName).add(field);
            }

            // 生成复合索引建议（WHERE + ORDER）
            if (whereFields.length > 0 && orderFields.length > 0) {
                suggestions.push({
                    type: 'composite',
                    table: tableName,
                    columns: [...whereFields, ...orderFields],
                    reason: '优化范围查询和排序'
                });
            }
        }

        // 转换为索引建议
        for (const [table, fields] of tableIndexes) {
            for (const field of fields) {
                suggestions.push({
                    type: 'single',
                    table,
                    columns: [field],
                    reason: '优化等值查询'
                });
            }
        }

        return suggestions;
    }

    /**
     * 创建索引（带重命名和清理）
     */
    async recreateIndex(tableName, indexName, columns, options = {}) {
        const client = await this.pool.connect();

        try {
            await client.query('BEGIN');

            // 删除旧索引（如果存在且不同名）
            const checkQuery = `
                SELECT 1 FROM pg_indexes
                WHERE tablename = $1 AND indexname = $2
            `;
            const exists = await client.query(checkQuery, [tableName, indexName]);

            if (exists.rows.length > 0) {
                // 索引已存在，检查是否需要重建
                await client.query(`DROP INDEX IF EXISTS ${indexName}`);
            }

            // 构建CREATE INDEX语句
            let createSQL = `CREATE INDEX CONCURRENTLY ${indexName} ON ${tableName}`;

            if (options.using) {
                createSQL += ` USING ${options.using}`;
            }

            createSQL += ` (${columns.join(', ')})`;

            if (options.where) {
                createSQL += ` WHERE ${options.where}`;
            }

            if (options.unique) {
                createSQL = createSQL.replace('CREATE INDEX', 'CREATE UNIQUE INDEX');
            }

            await client.query(createSQL);
            await client.query('COMMIT');

            return { success: true, indexName };
        } catch (error) {
            await client.query('ROLLBACK');
            throw error;
        } finally {
            client.release();
        }
    }

    /**
     * 获取索引使用统计
     */
    async getIndexStats() {
        const query = `
            SELECT
                schemaname,
                tablename,
                indexname,
                idx_scan,
                idx_tup_read,
                idx_tup_fetch,
                pg_size_pretty(pg_relation_size(indexrelid)) AS index_size
            FROM pg_stat_user_indexes
            JOIN pg_index USING (indexrelid)
            WHERE idx_scan = 0
            AND NOT indisunique
            AND NOT indisprimary
            ORDER BY pg_relation_size(indexrelid) DESC
        `;

        return await this.pool.query(query);
    }

    /**
     * 简化字段提取（实际需要解析器）
     */
    extractWhereFields(query) {
        // 实际实现需要SQL解析器，这里是简化版本
        const wherePattern = /WHERE\s+(\w+)\s*=/gi;
        const fields = [];
        let match;
        while ((match = wherePattern.exec(query)) !== null) {
            fields.push(match[1]);
        }
        return [...new Set(fields)];
    }

    extractOrderFields(query) {
        const orderPattern = /ORDER\s+BY\s+(\w+)/gi;
        const fields = [];
        let match;
        while ((match = orderPattern.exec(query)) !== null) {
            fields.push(match[1]);
        }
        return [...new Set(fields)];
    }

    extractTableName(query) {
        const fromPattern = /FROM\s+(\w+)/i;
        const match = query.match(fromPattern);
        return match ? match[1] : null;
    }
}

// 使用示例
async function indexManagementDemo() {
    const pool = new Pool({ connectionString: 'postgresql://...' });
    const manager = new IndexManager(pool);

    // 分析慢查询
    const slowQueries = await manager.analyzeSlowQueries(10);
    console.log('慢查询:', slowQueries);

    // 生成索引建议
    const suggestions = manager.generateIndexSuggestions(slowQueries.map(q => q.query));
    console.log('索引建议:', suggestions);

    // 创建索引
    for (const suggestion of suggestions) {
        const indexName = `idx_${suggestion.table}_${suggestion.columns.join('_')}`;
        await manager.recreateIndex(
            suggestion.table,
            indexName,
            suggestion.columns,
            suggestion.type === 'composite' ? { where: 'status = active' } : {}
        );
    }

    // 获取未使用索引
    const unusedIndexes = await manager.getIndexStats();
    console.log('未使用索引:', unusedIndexes);

    await pool.end();
}
```
