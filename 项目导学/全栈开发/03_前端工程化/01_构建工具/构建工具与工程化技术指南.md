## 2. Vite深入指南

### 2.1 Vite 6核心特性

```typescript
// vite.config.ts 配置示例
import { defineConfig } from 'vite';
import react from '@vitejs/plugin-react';
import path from 'path';

export default defineConfig({
  // React插件支持
  plugins: [react()],

  // 环境配置
  envPrefix: ['VITE_', 'APP_'],
  envDir: 'envs',

  // 环境API支持多环境配置
  environments: {
    client: {},
    node: {},
    edge: {}
  },

  // 路径别名
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },

  // 构建优化
  build: {
    target: 'esnext',
    minify: 'esbuild',
    rollupOptions: {
      output: {
        manualChunks: {
          'react-vendor': ['react', 'react-dom'],
          'ui-vendor': ['antd', '@ant-design/icons'],
        },
      },
    },
  },

  // 开发服务器配置
  server: {
    port: 3000,
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
      },
    },
  },

  // CSS配置
  css: {
    modules: {
      localsConvention: 'camelCase',
    },
    preprocessorOptions: {
      less: {
        javascriptEnabled: true,
      },
    },
  },
});
```

### 2.3 插件开发

```typescript
// my-vite-plugin.ts
import { definePlugin } from 'vite';

export default definePlugin((config) => {
  return {
    name: 'my-custom-plugin',
    // 转换代码
    transform(code, id) {
      if (id.endsWith('.vue')) {
        return code.replace(/<script>/g, '<script setup>');
      }
    },
    // 虚拟模块
    resolveId(id) {
      if (id === 'virtual:my-module') {
        return '\0virtual:my-module';
      }
    },
    load(id) {
      if (id === '\0virtual:my-module') {
        return 'export const message = "Hello from virtual module";';
      }
    },
  };
});
```

## 3. Webpack 5与Module Federation

### 3.1 基础配置

```javascript
// webpack.config.js
module.exports = {
  mode: 'production',
  entry: './src/index.js',

  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].[contenthash].js',
    clean: true,
  },

  // 持久化缓存
  cache: {
    type: 'filesystem',
    buildDependencies: {
      config: [__filename],
    },
  },

  module: {
    rules: [
      {
        test: /\.tsx?$/,
        use: 'babel-loader',
        exclude: /node_modules/,
      },
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader'],
      },
    ],
  },

  plugins: [
    new HtmlWebpackPlugin({
      template: './src/index.html',
    }),
  ],
};
```

### 3.2 Module Federation微前端

```javascript
// host/webpack.config.js - 主应用
const { ModuleFederationPlugin } = require('webpack').container;

module.exports = {
  plugins: [
    new ModuleFederationPlugin({
      name: 'host',
      remotes: {
        // 远程应用
        remoteApp: 'remoteApp@http://localhost:3001/remoteEntry.js',
      },
      shared: {
        react: { singleton: true, eager: true },
        'react-dom': { singleton: true, eager: true },
      },
    }),
  ],
};
```

```javascript
// remote/webpack.config.js - 远程应用
const { ModuleFederationPlugin } = require('webpack').container;

module.exports = {
  plugins: [
    new ModuleFederationPlugin({
      name: 'remoteApp',
      filename: 'remoteEntry.js',
      exposes: {
        './Button': './src/Button',
        './UserCard': './src/UserCard',
      },
      shared: {
        react: { singleton: true },
        'react-dom': { singleton: true },
      },
    }),
  ],
};
```

### 3.3 共享模块配置

```javascript
module.exports = {
  plugins: [
    new ModuleFederationPlugin({
      shared: {
        // 共享React，版本一致时共享，不一致各自加载
        react: {
          singleton: true,
          requiredVersion: '^19.0.0',
          eager: true,
        },
        // 非单例，多版本共存
        'react-dom': {
          singleton: false,
        },
      },
    }),
  ],
};
```

### 4.1 Rspack配置

```javascript
// rspack.config.js
const { DefaultRspackOptions } = require('@rspack/core');

module.exports = {
  entry: './src/index.js',
  output: {
    path: path.resolve(__dirname, 'dist'),
  },
  // 兼容Webpack插件
  plugins: [
    new HtmlRspackPlugin({
      template: './src/index.html',
    }),

    rules: [
      {
        test: /\.jsx?$/,
        loader: 'builtin:swc-loader',

          jsc: {
            parser: {
              syntax: 'ecmascript',
              jsx: true,
            },
          },
        },
      },
    ],
  },
  // 开发服务器
  devServer: {
    port: 3000,
    hot: true,
  },
};
```

## 6. Monorepo架构实践

### 6.1 Monorepo概述

Monorepo（单体仓库）是一种将多个项目放在一个代码仓库中的开发模式，特别适合大型全栈应用。

**Monorepo优势**：

```
┌─────────────────────────────────────────────────────────────────────┐
│                        Monorepo vs Polyrepo                         │
├─────────────────────────────────────────────────────────────────────┤
│                                                                      │
│  Polyrepo（多仓库）：                                                │
│  ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐                 │
│  │ frontend│ │ backend │ │ mobile  │ │  docs   │                 │
│  │ repo    │ │ repo    │ │ repo    │ │ repo    │                 │
│  └────┬────┘ └────┬────┘ └────┬────┘ └────┬────┘                 │
│       │           │           │           │                       │
│       └───────────┴───────────┴───────────┘                       │
│                     难以共享代码                                     │
│                                                                      │
│  Monorepo（单仓库）：                                                │
│  ┌────────────────────────────────────────────────────────────┐   │
│  │                    my-org/                                │   │
│  │  ┌──────────┐ ┌──────────┐ ┌──────────┐ ┌──────────┐      │   │
│  │  │packages │ │packages  │ │apps/    │ │apps/    │      │   │
│  │  │ /ui     │ │ /utils  │ │ /web    │ │ /mobile │      │   │
│  │  └──────────┘ └──────────┘ └──────────┘ └──────────┘      │   │
│  │  ┌──────────┐ ┌──────────┐                                 │   │
│  │  │packages │ │packages  │                                 │   │
│  │  │ /shared │ │ /config │                                 │   │
│  │  └──────────┘ └──────────┘                                 │   │
│  └────────────────────────────────────────────────────────────┘   │
│                     轻松共享代码，原子化提交                         │
│                                                                      │
└─────────────────────────────────────────────────────────────────────┘
```

### 6.3 Turborepo配置

```json
// turbo.json
{
  "$schema": "https://turbo.build/schema.json",
  "tasks": {
    "build": {
      "dependsOn": ["^build"],  // 依赖的包先构建
      "outputs": ["dist/**", ".next/**"],
      "cache": true
    },
    "dev": {
      "cache": false,
      "persistent": true  // 常驻进程
    },
    "lint": {
      "dependsOn": ["^build"]
    },
    "test": {
      "dependsOn": ["build"],
      "outputs": ["coverage/**"]
    },
    "clean": {
      "cache": false
    }
  }
}
```

## 7. 模块联邦进阶

### 7.1 动态模块加载

```typescript
// 主机应用 - 动态加载远程模块
const RemoteButton = React.lazy(() => import('remoteApp/Button'));

function App() {
  return (
    <div>
      <h1>主机应用</h1>
      <React.Suspense fallback={<div>加载中...</div>}>
        <RemoteButton>远程按钮</RemoteButton>
      </React.Suspense>
    </div>
  );
}
```

### 7.2 共享状态管理

```javascript
// 主机应用 - 全局状态共享
const { ModuleFederationPlugin } = require('webpack').container;

module.exports = {
  plugins: [
    new ModuleFederationPlugin({
      name: 'host',
      remotes: {
        remoteApp: 'remoteApp@http://localhost:3001/remoteEntry.js',
      },
      shared: {
        // React和ReactDOM作为单例共享
        react: {
          singleton: true,
          eager: true,
          requiredVersion: '^19.0.0',
        },
        'react-dom': {
          singleton: true,
          eager: true,
          requiredVersion: '^19.0.0',
        },
        // Zustand作为单例，状态跨应用共享
        zustand: {
          singleton: true,
          requiredVersion: '^5.0.0',
        },
      },
    }),
  ],
};
```
