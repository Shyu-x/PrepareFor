## 1. 构建工具基础

### 1.1 构建工具作用

```
构建流程（Build Pipeline）

源代码（Source Code）
    ↓
┌─────────────────────────────┐
│ 1. 解析（Parsing）          │
│    - 语法分析               │
│    - 依赖树构建             │
└─────────────────────────────┘
    ↓
┌─────────────────────────────┐
│ 2. 转换（Transformation）   │
│    - TypeScript → JavaScript│
│    - JSX → createElement    │
│    - CSS → CSS Modules      │
└─────────────────────────────┘
    ↓
┌─────────────────────────────┐
│ 3. 优化（Optimization）     │
│    - 代码压缩               │
│    - Tree Shaking           │
│    - 代码分割               │
└─────────────────────────────┘
    ↓
┌─────────────────────────────┐
│ 4. 打包（Bundling）         │
│    - 模块合并               │
│    - 资源处理               │
│    - 生成产物               │
└─────────────────────────────┘
    ↓
构建产物（Build Output）
```

### 1.2 构建工具对比

| 工具 | 核心语言 | 启动速度 | HMR | 生产构建 | 适用场景 |
|------|----------|----------|-----|----------|----------|
| **Vite** | Go/Rust | ⚡ 极快 | ✅ 优秀 | 🟡 中等 | 开发体验 |
| **Webpack** | JavaScript | 🐢 较慢 | ✅ 良好 | ✅ 优秀 | 复杂项目 |
| **Rspack** | Rust | ⚡ 极快 | ✅ 优秀 | ✅ 优秀 | 企业级项目 |
| **Rollup** | JavaScript | ⚡ 快 | ❌ 无 | ✅ 优秀 | 库开发 |
| **esbuild** | Go | ⚡⚡ 最快 | ⚠️ 基础 | 🟡 中等 | 底层构建 |

### 1.3 构建工具选择指南

```javascript
// 构建工具决策树
function selectBundler(requirements) {
  // 1. 库开发
  if (requirements.type === 'library') {
    if (requirements.framework === 'React') {
      return 'Rollup';  // Rollup适合库打包
    }
    return 'esbuild';  // 纯JavaScript库
  }

  // 2. 应用开发
  if (requirements.type === 'application') {
    // 优先选择Vite
    if (requirements.complexity === 'low') {
      return 'Vite';  // 简单项目首选
    }

    // 复杂项目
    if (requirements.complexity === 'high') {
      if (requirements.ecosystem === 'Vue') {
        return 'Vite';  // Vue生态首选
      }
      if (requirements.ecosystem === 'React') {
        if (requirements.enterprise) {
          return 'Rspack';  // 企业级React项目
        }
        return 'Vite';  // 大部分React项目
      }
    }

    // 性能要求高
    if (requirements.performance === 'high') {
      return 'Rspack';  // Rust构建器
    }
  }

  // 默认选择
  return 'Vite';
}
```

### 2.2 Vite插件开发

```javascript
// plugins/my-plugin.ts
import type { Plugin } from 'vite';

export function myPlugin(): Plugin {
  return {
    name: 'my-plugin',

    // 配置钩子
    config(config, { command }) {
      if (command === 'build') {
        return {
          build: {
            // 修改构建配置
          },
        };
      }
    },

    // 转换钩子
    transform(code, id) {
      if (id.endsWith('.my-ext')) {
        // 转换自定义文件
        return {
          code: `export default ${JSON.stringify(code)}`,
          map: null,
        };
      }
    },

    // 加载钩子
    load(id) {
      if (id.endsWith('.my-ext')) {
        return 'export default "loaded from my-plugin"';
      }
    },

    // 解析钩子
    resolveId(source) {
      if (source === 'virtual-module') {
        return '\0virtual-module';  // 虚拟模块
      }
    },

    // 热更新钩子
    handleHotUpdate({ file, server }) {
      if (file.endsWith('.my-ext')) {
        server.ws.send({
          type: 'full-reload',
          path: '*',
        });
      }
    },

    // 构建后钩子
    closeBundle() {
      console.log('构建完成');
    },
  };
}

// 使用插件
// vite.config.ts
import { myPlugin } from './plugins/my-plugin';

export default defineConfig({
  plugins: [myPlugin()],
});
```

```javascript
      globals: {
        react: 'React',
        'react-dom': 'ReactDOM',
      },
    },
  ],
```

```javascript
  // 插件
  plugins: [
```

```javascript
    typescript({
      tsconfig: './tsconfig.json',

      declarationDir: 'dist',

      browser: true,
    }),

    // CommonJS兼容

    // 外部依赖
```

### 5.2 esbuild极速构建

```javascript
// esbuild.config.js
const esbuild = require('esbuild');

async function build() {
  // 开发构建
  await esbuild.context({
    entryPoints: ['src/index.tsx'],
    bundle: true,
    outfile: 'dist/index.js',
    sourcemap: true,
    loader: {
      '.ts': 'ts',
      '.tsx': 'tsx',
      '.css': 'css',
    },
    define: {
      'process.env.NODE_ENV': '"development"',
    },
    watch: true,
  }).then((ctx) => {
    ctx.watch();
  });

  // 生产构建
  await esbuild.build({
    entryPoints: ['src/index.tsx'],
    bundle: true,
    minify: true,
    outfile: 'dist/index.min.js',
    sourcemap: false,
    target: 'es2015',
    loader: {
      '.ts': 'ts',
      '.tsx': 'tsx',
    },
    define: {
      'process.env.NODE_ENV': '"production"',
    },
  });
}

build();
```
