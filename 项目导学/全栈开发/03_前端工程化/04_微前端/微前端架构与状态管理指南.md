## 1. 微前端架构概述

### 1.1 什么是微前端

```
单体前端：
┌─────────────────────────────────┐
│           主应用                   │
│  ┌─────────┐ ┌─────────┐       │
│  │ 模块A   │ │ 模块B   │       │
│  └─────────┘ └─────────┘       │
└─────────────────────────────────┘

微前端：
┌─────────────────────────────────┐
│           容器应用                │
├──────┬──────┬──────┬──────────┤
│ 子应用│ 子应用│ 子应用│  子应用   │
│  A    │  B    │  C    │    D     │
└──────┴──────┴──────┴──────────┘
```

### 1.2 微前端方案对比

| 方案 | 特点 | 适用场景 |
|------|------|----------|
| **qiankun** | 基于single-spa，HTML Entry | Vue/React混合 |
| **Module Federation** |Webpack原生，微模块 | 同构框架 |
| **single-spa** | 基础框架，灵活 | 多框架混合 |
| **iframe** | 简单隔离 | 遗留系统 |

```typescript
        state.status = 'succeeded';
```

### 6.2 跨应用通信设计

```typescript
// 自定义事件总线（发布-订阅模式）
class EventBus {
  private events: Map<string, Set<Function>> = new Map();

  // 发布事件
  emit(event: string, data: any) {
    const listeners = this.events.get(event);
    if (listeners) {
      listeners.forEach((callback) => callback(data));
    }
  }

  // 订阅事件
  on(event: string, callback: Function): () => void {
    if (!this.events.has(event)) {
      this.events.set(event, new Set());
    }
    this.events.get(event)!.add(callback);

    // 返回取消订阅函数
    return () => {
      this.events.get(event)?.delete(callback);
    };
  }

  // 一次性订阅
  once(event: string, callback: Function) {
    const unsubscribe = this.on(event, (data) => {
      callback(data);
      unsubscribe();
    });
  }
}

// 全局事件总线实例
export const eventBus = new EventBus();

// 使用示例 - 发送消息
eventBus.emit('user:login', { userId: '123', token: 'xxx' });
eventBus.emit('cart:add', { productId: '456', quantity: 2 });

// 使用示例 - 接收消息
const unsubscribe = eventBus.on('user:login', (data) => {
  console.log('用户登录:', data.userId);
});

// 组件卸载时取消订阅
useEffect(() => {
  return () => unsubscribe();
}, []);
```

### 6.3 统一鉴权方案

```typescript
// 主应用 - 权限管理
interface Permission {
  action: string;
  resource: string;
}

class AuthManager {
  private permissions: Map<string, Permission[]> = new Map();
  private currentUser: User | null = null;

  async loadPermissions(userId: string) {
    // 从服务端获取用户权限
    const permissions = await fetchPermissions(userId);
    this.permissions.set(userId, permissions);
  }

  hasPermission(action: string, resource: string): boolean {
    if (!this.currentUser) return false;

    const userPermissions = this.permissions.get(this.currentUser.id) || [];
    return userPermissions.some(
      (p) => p.action === action && p.resource === resource
    );
  }

  // 分发权限到子应用
  dispatchPermissions() {
    const permissions = this.getCurrentPermissions();
    eventBus.emit('auth:permissions', permissions);
  }
}

// 子应用 - 权限守卫
function PermissionGuard({ action, resource, children }) {
  const [hasPermission, setHasPermission] = useState(false);

  useEffect(() => {
    const unsubscribe = eventBus.on('auth:permissions', (permissions) => {
      const allowed = permissions.some(
        (p) => p.action === action && p.resource === resource
      );
      setHasPermission(allowed);
    });

    return unsubscribe;
  }, [action, resource]);

  return hasPermission ? children : <AccessDenied />;
}

// 使用
<PermissionGuard action="read" resource="user">
  <UserList />
</PermissionGuard>
```

```typescript
// 全局错误处理器
class GlobalErrorHandler {
  private errorListeners: Set<(error: Error) => void> = new Set();

  handleError(error: Error, context?: { app?: string; component?: string }) {
    console.error(`[${context?.app || 'unknown'}] Error:`, error);

    // 记录错误日志
    this.logError(error, context);

    // 通知所有监听器
    this.errorListeners.forEach((listener) => listener(error));
  }

  onError(listener: (error: Error) => void) {
    this.errorListeners.add(listener);
    return () => this.errorListeners.delete(listener);
  }

  private async logError(error: Error, context?: any) {
    await fetch('/api/logs', {
      method: 'POST',
      body: JSON.stringify({
        message: error.message,
        stack: error.stack,
        context,
        timestamp: Date.now(),
      }),
    });
  }
}
```

```typescript
    addUser: (state, action: PayloadAction<User>) => {
      state.items.push(action.payload);
    },
    updateUser: (state, action: PayloadAction<User>) => {
      const index = state.items.findIndex((u) => u.id === action.payload.id);
      if (index !== -1) {
        state.items[index] = action.payload;
      }
    },
    deleteUser: (state, action: PayloadAction<string>) => {
      state.items = state.items.filter((u) => u.id !== action.payload);
    },
  },
  extraReducers: (builder) => {
    builder
      .addCase(fetchUsers.pending, (state) => {
        state.loading = true;
        state.error = null;
      })
      .addCase(fetchUsers.fulfilled, (state, action) => {
        state.loading = false;
        state.items = action.payload;
      })
      .addCase(fetchUsers.rejected, (state, action) => {
        state.loading = false;
        state.error = action.payload as string;
      });
  },
});

export const {
  selectUser,
  clearSelection,
  setFilters,
  addUser,
  updateUser,
  deleteUser,
} = usersSlice.actions;

export default usersSlice.reducer;
```
