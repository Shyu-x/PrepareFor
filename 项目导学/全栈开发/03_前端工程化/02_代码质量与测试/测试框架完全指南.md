## 1. 测试基础

### 1.1 测试类型

```
测试金字塔（Testing Pyramid）

              E2E测试
            (少量、高成本)
            ┌──────────┐
           /  集成测试  \
          /  (中等数量)  \
         /________________\
        /                  \
       /    单元测试         \
      /    (大量、低成本)     \
     /______________________\
```

| 测试类型 | 目的 | 速度 | 成本 | 示例 |
|---------|------|------|------|------|
| **单元测试** | 测试单个函数/组件 | ⚡ 极快 | 💰 低 | 计算器函数、组件渲染 |
| **集成测试** | 测试多个模块协作 | 🚀 快 | 💵 中 | API调用、数据库操作 |
| **E2E测试** | 测试完整用户流程 | 🐢 较慢 | 💳 高 | 登录流程、购物流程 |

### 1.2 测试原则

```javascript
// ✅ 测试的最佳实践（AAA模式）
describe('计算器功能', () => {
  test('应该正确执行加法运算', () => {
    // Arrange（准备）- 设置测试数据
    const a = 5;
    const b = 3;

    // Act（执行）- 调用被测试函数
    const result = add(a, b);

    // Assert（断言）- 验证结果
    expect(result).toBe(8);
  });
});

// ✅ 测试应该具有描述性
test('用户注册时，如果邮箱已存在，应该返回错误', () => {
  // 清晰描述测试场景和预期行为
});

// ✅ 每个测试应该独立
test('测试场景1', () => {
  // 不依赖其他测试的状态
});

// ❌ 避免的实现
test('应该工作', () => {
  // 太模糊，不知道在测试什么
});
```

## 2. Jest深度解析

### 2.1 Jest核心概念

```javascript
// jest.config.js
module.exports = {
  // 测试环境
  testEnvironment: 'jsdom',  // 或 'node'

  // 测试文件匹配模式
  testMatch: [
    '**/__tests__/**/*.[jt]s?(x)',
    '**/?(*.)+(spec|test).[jt]s?(x)'
  ],

  // 转换配置
  transform: {
    '^.+\\.(ts|tsx)$': ['ts-jest', {
      tsconfig: {
        jsx: 'react',
      },
    }],
  },

  // 模块路径映射
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/src/$1',
    '\\.(css|less|scss|sass)$': 'identity-obj-proxy',
  },

  // 覆盖率收集
  collectCoverage: true,
  coverageDirectory: 'coverage',
  coverageReporters: ['text', 'lcov', 'html'],

  // 快照测试
  snapshotSerializers: [],
};
```

### 2.2 Jest断言API

```javascript
// toBe - 严格相等（===）
expect(1 + 1).toBe(2);
expect({}).toBe({});  // ❌ 失败：不同引用

// toEqual - 深度相等
expect({ name: '张三' }).toEqual({ name: '张三' });  // ✅

// toBeTruthy / toBeFalsy
expect(true).toBeTruthy();
expect(false).toBeFalsy();
expect(0).toBeFalsy();
expect('').toBeFalsy();

// toBeNull / toBeUndefined
expect(null).toBeNull();
expect(undefined).toBeUndefined();
expect(undefined).not.toBeDefined();

// toContain - 数组/字符串包含
expect([1, 2, 3]).toContain(2);
expect('hello world').toContain('world');

// toMatch - 正则匹配
expect('hello').toMatch(/ell/);

// toThrow - 抛出异常
expect(() => {
  throw new Error('错误信息');
}).toThrow('错误信息');

// toHaveLength - 长度断言
expect([1, 2, 3]).toHaveLength(3);
expect('hello').toHaveLength(5);

// toHaveProperty - 对象属性
expect({ name: '张三', age: 25 }).toHaveProperty('name');
expect({ user: { name: '张三' } }).toHaveProperty('user.name');

// toBeGreaterThan / toBeLessThan - 数字比较
expect(10).toBeGreaterThan(5);
expect(5).toBeLessThan(10);

// toBeCloseTo - 浮点数比较
expect(0.1 + 0.2).toBeCloseTo(0.3, 5);  // 精度5位小数

// toMatchObject - 部分匹配
expect({
  name: '张三',
  age: 25,
  email: 'zhangsan@example.com'
}).toMatchObject({
  name: '张三',
  age: 25
});

// resolves / rejects - Promise断言
test('异步测试', async () => {
  await expect(Promise.resolve('成功')).resolves.toBe('成功');
  await expect(Promise.reject('失败')).rejects.toThrow('失败');
});
```

### 2.3 Mock与Spy

```javascript
// 1. 函数Mock（jest.fn）
const mockFn = jest.fn();
mockFn('arg1', 'arg2');

// 检查调用次数
expect(mockFn).toHaveBeenCalled();
expect(mockFn).toHaveBeenCalledTimes(1);

// 检查调用参数
expect(mockFn).toHaveBeenCalledWith('arg1', 'arg2');

// 检查调用顺序
expect(mockFn).toHaveBeenLastCalledWith('arg1', 'arg2');

// 获取调用历史
mockFn.mock.calls;  // [['arg1', 'arg2']]
mockFn.mock.results;  // [{ type: 'return', value: undefined }]

// 设置返回值
mockFn.mockReturnValue(42);
mockFn();  // 返回 42

// 设置异步返回值
mockFn.mockResolvedValue('异步结果');
await mockFn();  // 返回 '异步结果'

// 设置抛出错误
mockFn.mockImplementation(() => {
  throw new Error('模拟错误');
});

// 2. 模块Mock（jest.mock）
jest.mock('./api', () => ({
  fetchUser: jest.fn(() => Promise.resolve({ id: 1, name: '张三' })),
}));

import { fetchUser } from './api';

test('应该使用mock的fetchUser', async () => {
  const user = await fetchUser(1);
  expect(user).toEqual({ id: 1, name: '张三' });
});

// 3. 部分Mock
jest.mock('./utils', () => ({
  ...jest.requireActual('./utils'),
  formatDate: jest.fn(() => '2024-01-01'),
}));

import { formatDate, calculateAge } from './utils';

test('应该使用部分mock', () => {
  expect(formatDate()).toBe('2024-01-01');  // mock
  expect(calculateAge(1990)).toBe(34);      // 实际函数
});

// 4. Spy（jest.spyOn）
const object = {
  method: () => '原始值',
};

const spy = jest.spyOn(object, 'method');

spy.mockReturnValue('模拟值');
object.method();  // 返回 '模拟值'

expect(spy).toHaveBeenCalled();

// 恢复原始实现
spy.mockRestore();

// 5. 定时器Mock
jest.useFakeTimers();

test('定时器测试', () => {
  const callback = jest.fn();

  setTimeout(callback, 1000);

  // 快进1秒
  jest.advanceTimersByTime(1000);

  expect(callback).toHaveBeenCalledTimes(1);

  // 快进所有定时器
  jest.runAllTimers();

  // 恢复真实定时器
  jest.useRealTimers();
});
```

### 2.4 Jest钩子函数

```javascript
describe('Jest钩子函数', () => {
  // beforeAll - 所有测试前执行一次
  beforeAll(async () => {
    // 初始化数据库连接
    await connectDatabase();
  });

  // afterAll - 所有测试后执行一次
  afterAll(async () => {
    // 关闭数据库连接
    await disconnectDatabase();
  });

  // beforeEach - 每个测试前执行
  beforeEach(() => {
    // 重置测试数据
    testData = [];
  });

  // afterEach - 每个测试后执行
  afterEach(() => {
    // 清理副作用
    jest.clearAllMocks();
  });

  test('测试1', () => {
    // 使用 beforeEach 设置的初始状态
    expect(testData).toEqual([]);
  });

  test('测试2', () => {
    // 每个测试都重新开始
    expect(testData).toEqual([]);
  });
});
```

## 3. Vitest实战

### 3.1 Vitest vs Jest对比

| 特性 | Jest | Vitest |
|------|------|--------|
| **性能** | 较慢 | ⚡ 极快（使用Vite） |
| **配置** | 复杂 | 🎯 简单（继承Vite配置） |
| **HMR** | ❌ 不支持 | ✅ 热模块替换 |
| **TypeScript** | 需要ts-jest | 🟢 原生支持 |
| **ESM** | ⚠️ 需要配置 | ✅ 原生支持 |
| **UI界面** | ❌ 无 | ✅ 可视化界面 |

### 3.2 Vitest配置

```javascript
// vitest.config.js
import { defineConfig } from 'vitest/config';

export default defineConfig({
  // 测试环境
  test: {
    environment: 'jsdom',  // 或 'node'

    // 全局API（如 describe, test）
    globals: true,

    // 覆盖率配置
    coverage: {
      provider: 'v8',
      reporter: ['text', 'json', 'html'],
      exclude: [
        'node_modules/',
        '**/*.d.ts',
        '**/*.config.*',
        '**/coverage/**',
      ],
    },

    // 测试文件
    include: ['**/*.{test,spec}.{js,mjs,cjs,ts,mts,cts,jsx,tsx}'],
    exclude: ['node_modules', 'dist', '.idea', '.git', '.cache'],

    // 测试超时
    testTimeout: 5000,

    // UI界面
    ui: true,

    // 并行测试
    pool: 'threads',  // 或 'forks'

    // 线程数
    poolOptions: {
      threads: {
        singleThread: true,  // 单线程模式
      },
    },
  },

  // 路径别名（与Vite共享）
  resolve: {
    alias: {
      '@': '/src',
    },
  },
});
```

### 3.3 Vitest API使用

```javascript
// vitest-api.test.ts
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';

describe('Vitest API', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  test('Mock函数', () => {
    const mockFn = vi.fn();
    mockFn('arg');

    expect(mockFn).toHaveBeenCalled();
    expect(mockFn).toHaveBeenCalledWith('arg');
  });

  test('定时器Mock', () => {
    vi.useFakeTimers();

    const callback = vi.fn();
    setTimeout(callback, 1000);

    vi.advanceTimersByTime(1000);
    expect(callback).toHaveBeenCalled();

    vi.useRealTimers();
  });

  test('模块Mock', async () => {
    const mockFetch = vi.fn(() => Promise.resolve({ data: 'mock' }));
    vi.stubGlobal('fetch', mockFetch);

    expect(mockFetch).toHaveBeenCalled();

    vi.unstubAllGlobals();
  });
});
```

    // 切换回原页面
    await page.bringToFront();
  });

  test('等待条件', async ({ page }) => {
    await page.goto('/products');

    await page.waitForSelector('.product-item', { state: 'visible' });

    // 等待网络空闲
    await page.waitForLoadState('networkidle');

    // 等待自定义条件
    await page.waitForFunction(() => {
      return document.querySelectorAll('.product-item').length > 0;
    });
  });

  test('响应式测试', async ({ page }) => {
    await page.goto('/');

    // 验证桌面端布局
    await expect(page.locator('.desktop-menu')).toBeVisible();
  });

  test('地理位置测试', async ({ page, context }) => {
    // 设置地理位置
    await context.setGeolocation({ latitude: 35.6895, longitude: 139.6917 });
    await page.goto('/map');

    // 验证位置

      return {
        lat: window.navigator.geolocation.coords.latitude,

      };

    expect(location.lat).toBeCloseTo(35.6895, 4);

  it('应该返回用户数据', async () => {

  });

    });

  it('应该解析为用户数据', () => {
    return expect(fetchUser(1)).resolves.toEqual({ id: 1, name: '张三' });
  });

  it('应该抛出错误', () => {
