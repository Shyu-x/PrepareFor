## 1. 单元测试概述

### 1.1 什么是单元测试

单元测试（Unit Testing）是对软件中的最小可测试单元进行验证的过程。在前端开发中，单元通常指一个函数、一个组件或一个模块。

**单元测试的核心原则（FIRST原则）：**

| 原则 | 说明 | 实践 |
|------|------|------|
| **Fast（快速）** | 测试运行速度要快 | 避免I/O操作、使用Mock |
| **Independent（独立）** | 测试之间相互独立 | 不依赖其他测试的执行顺序 |
| **Repeatable（可重复）** | 测试结果可重复 | 确定性结果、Mock外部依赖 |
| **Self-Validating（自验证）** | 测试自己判断通过与否 | 使用断言，不依赖人工判断 |
| **Timely（及时）** | 测试写在实际代码之前 | TDD（测试驱动开发） |

## 2. Vitest快速入门

### 2.1 Vitest简介

Vitest是一个基于Vite的下一代测试框架，它的特点是：

- **极速**：与Vite共享HMR技术，测试文件变化时毫秒级更新
- **兼容Jest**：API与Jest兼容，迁移成本低
- **TypeScript原生**：内置TypeScript支持
- **ESM优先**：原生支持ES Module

### 2.2 项目配置

```bash
# 安装Vitest
npm install -D vitest @vue/test-utils happy-dom

# 安装React测试工具（如使用React）
npm install -D @testing-library/react @testing-library/jest-dom
```

```typescript
// vite.config.ts 中添加Vitest配置
import { defineConfig } from 'vite';
import vue from '@vitejs/plugin-vue';
import vitest from 'vitest/config';

export default defineConfig({
    plugins: [
        vue(),
        // 集成Vitest
        vitest({
            // 全局注入测试API
            globals: true,
            // 测试环境
            environment: 'happy-dom',
            // 覆盖率配置
            coverage: {
                provider: 'v8',
                reporter: ['text', 'json', 'html'],
                exclude: [
                    'node_modules/**',
                    'dist/**',
                    '**/*.config.*',
                    '**/*.d.ts'
                ]
            },
            // 测试文件匹配模式
            include: ['**/*.test.ts', '**/*.test.tsx'],
            // 逼真计时器（测试setTimeout等）
            fakeTimers: {
                shouldClearNativeTimers: true
            }
        })
    ]
});
```

```json

        "test:coverage": "vitest run --coverage",

    }

```

### 2.3 基本测试语法

```typescript
// src/utils/calc.test.ts
import { describe, it, expect, test } from 'vitest';
import { add, subtract, multiply, divide } from './calc';

// describe用于组织相关测试
describe('计算器工具函数', () => {
    // it和test是等价的
    test('add函数：两数相加', () => {
        expect(add(1, 2)).toBe(3);
        expect(add(-1, 1)).toBe(0);
        expect(add(0, 0)).toBe(0);
    });

    test('add函数：小数相加', () => {
        expect(add(0.1, 0.2)).toBeCloseTo(0.3);
    });

    it('subtract函数：两数相减', () => {
        expect(subtract(5, 3)).toBe(2);
        expect(subtract(3, 5)).toBe(-2);
    });

    it('multiply函数：两数相乘', () => {
        expect(multiply(3, 4)).toBe(12);
        expect(multiply(-2, 3)).toBe(-6);
        expect(multiply(0, 100)).toBe(0);
    });

    it('divide函数：正常除法', () => {
        expect(divide(10, 2)).toBe(5);
        expect(divide(9, 3)).toBe(3);
    });

    it('divide函数：除以0应该抛出错误', () => {
        expect(() => divide(10, 0)).toThrow('除数不能为零');
    });
});
```

### 2.4 常用断言API

```typescript
import { expect } from 'vitest';

// === 通用匹配 ===
expect(value).toBe(expected);          // 严格相等（===）
expect(value).toEqual(expected);       // 深度相等（适合对象/数组）
expect(value).toStrictEqual(expected); // 严格深度相等

// === 布尔匹配 ===
expect(value).toBeTruthy();            // 真值
expect(value).toBeFalsy();             // 假值
expect(value).toBe(true);              // 严格为true
expect(value).toBe(false);             // 严格为false

// === 数字匹配 ===
expect(value).toBe(3);                // 严格等于3
expect(value).toEqual(3);
expect(value).toBeGreaterThan(3);       // 大于3
expect(value).toBeGreaterThanOrEqual(3); // 大于等于3
expect(value).toBeLessThan(3);         // 小于3
expect(value).toBeLessThanOrEqual(3);  // 小于等于3
expect(value).toBeCloseTo(0.3);       // 浮点数近似相等

// === 字符串匹配 ===
expect(value).toMatch(/pattern/);      // 正则匹配
expect(value).toContain('子串');       // 包含子串
expect(value).toHaveLength(5);         // 长度匹配
expect(value).toStartWith('前缀');     // 以...开头
expect(value).toEndWith('后缀');       // 以...结尾

// === 数组匹配 ===
expect(arr).toContain(item);           // 包含元素
expect(arr).toContainEqual(item);      // 包含深度相等的元素
expect(arr).toHaveLength(5);           // 长度匹配
expect(arr).toEqual(expect.arrayContaining(['a', 'b'])); // 包含某些元素

// === 对象匹配 ===
expect(obj).toHaveProperty('key');              // 拥有属性
expect(obj).toHaveProperty('nested.key', value); // 嵌套属性
expect(obj).toMatchObject({ key: value });      // 匹配部分属性
expect(obj).toStrictEqual({ ... });             // 严格相等

// === 异常匹配 ===
expect(fn).toThrow();                 // 抛出异常
expect(fn).toThrow('错误信息');        // 抛出包含特定信息的错误
expect(fn).toThrow(Error);            // 抛出特定类型错误

// === 类型/存在性匹配 ===
expect(value).toBeDefined();          // 已定义（不是undefined）
expect(value).toBeUndefined();        // 未定义
expect(value).toBeNull();             // 为null
expect(value).toBeNaN();              // 为NaN
expect(value).toBeInstanceOf(Class);  // 是某个类的实例

// === 否定 ===
expect(value).not.toBe(3);            // 不等于3
expect(arr).not.toContain(item);      // 不包含元素
```

### 4.3 Timer Mock（定时器模拟）

```typescript
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';

// 使用fake timers
beforeEach(() => {
    vi.useFakeTimers();
});

afterEach(() => {
    vi.useRealTimers();
});

describe('定时器测试', () => {
    it('setTimeout被正确调用', () => {
        const callback = vi.fn();
        setTimeout(callback, 1000);

        // 快进1秒
        vi.advanceTimersByTime(1000);

        expect(callback).toHaveBeenCalled();
    });

    it('setInterval正常工作', () => {
        vi.useFakeTimers();

        const callback = vi.fn();
        setInterval(callback, 100);

        // 快进500ms
        vi.advanceTimersByTime(500);

        // 应该被调用4次（0, 100, 200, 300, 400, 500）
        expect(callback).toHaveBeenCalledTimes(5);

        vi.useRealTimers();
    });

    it('Promise在正确时间解析', async () => {
        vi.useFakeTimers();

        const promise = new Promise((resolve) => {
            setTimeout(() => resolve('done'), 1000);
        });

        // 确保promise处于pending状态
        let settled = false;
        promise.then(() => { settled = true; });

        expect(settled).toBe(false);

        // 快进时间
        await vi.advanceTimersByTimeAsync(1000);
        expect(settled).toBe(true);

        vi.useRealTimers();
    });

    it('flush所有pending的定时器', () => {
        vi.useFakeTimers();

        const fn1 = vi.fn();
        const fn2 = vi.fn();

        setTimeout(fn1, 100);
        setTimeout(fn2, 200);

        // 快进所有时间
        vi.runAllTimers();

        expect(fn1).toHaveBeenCalled();
        expect(fn2).toHaveBeenCalled();

        vi.useRealTimers();
    });
});
```

### 4.4 Spy（监听对象方法）

```typescript
import { describe, it, expect, vi } from 'vitest';

describe('Spy监听', () => {
    it('监听对象方法', () => {
        const console = {
            log: vi.fn()
        };

        // 替换为spy
        const logSpy = vi.spyOn(console, 'log');

        // 调用方法
        console.log('Hello', 'World');

        // 验证调用
        expect(logSpy).toHaveBeenCalledWith('Hello', 'World');
        expect(logSpy).toHaveBeenCalledTimes(1);

        // 恢复原方法
        logSpy.mockRestore();
    });

    it('监听Date.now', () => {
        const now = Date.now();
        const spy = vi.spyOn(Date, 'now').mockReturnValue(123456789);

        expect(Date.now()).toBe(123456789);

        spy.mockRestore();
        expect(Date.now()).not.toBe(123456789);
```

```typescript
// src/utils/validator.ts
export interface ValidationResult {
    valid: boolean;
    error?: string;
}

export function validateEmail(email: string): ValidationResult {
    if (!email) {
        return { valid: false, error: '邮箱不能为空' };
    }

    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
        return { valid: false, error: '邮箱格式不正确' };
    }

    return { valid: true };
}

export function validatePassword(password: string): ValidationResult {
    if (!password) {
        return { valid: false, error: '密码不能为空' };
    }

    if (password.length < 8) {
        return { valid: false, error: '密码长度至少8位' };
    }

    if (!/[A-Z]/.test(password)) {
        return { valid: false, error: '密码必须包含大写字母' };
    }

    if (!/[a-z]/.test(password)) {
        return { valid: false, error: '密码必须包含小写字母' };
    }

    if (!/[0-9]/.test(password)) {
        return { valid: false, error: '密码必须包含数字' };
    }

    return { valid: true };
}

export function validateRequired(value: string, fieldName: string): ValidationResult {
    if (!value || !value.trim()) {
        return { valid: false, error: `${fieldName}不能为空` };
    }
    return { valid: true };
}
```

```typescript
// src/hooks/useDebounce.ts
import { useState, useEffect } from 'react';

export function useDebounce<T>(value: T, delay: number): T {
    const [debouncedValue, setDebouncedValue] = useState<T>(value);

    useEffect(() => {
        const timer = setTimeout(() => {
            setDebouncedValue(value);
        }, delay);

        return () => {
            clearTimeout(timer);
        };
    }, [value, delay]);

    return debouncedValue;
}
```
