// jest.config.js

module.exports = {
  // 测试环境
  testEnvironment: 'jsdom', // 或 'node'

  // 根目录
  roots: ['<rootDir>/src'],

  // 测试文件匹配
  testMatch: [
    '**/__tests__/**/*.+(ts|tsx|js)',
    '**/?(*.)+(spec|test).+(ts|tsx|js)',
  ],

  // 转换配置
  transform: {
    '^.+\\.(ts|tsx)$': 'ts-jest',
  },

  // 模块别名
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/src/$1',
    '\\.(css|less|scss|sass)$': 'identity-obj-proxy',
  },

  // 设置文件
  setupFilesAfterEnv: ['<rootDir>/jest.setup.ts'],

  // 覆盖率配置
  collectCoverageFrom: [
    'src/**/*.{ts,tsx}',
    '!src/**/*.d.ts',
    '!src/main.tsx',
  ],

  // 覆盖率阈值
  coverageThreshold: {
    global: {
      branches: 80,
      functions: 80,
      lines: 80,
      statements: 80,
    },
  },

  // 忽略转换
  transformIgnorePatterns: [
    'node_modules/(?!(react-router|@react-router)/)',
  ],
};

// jest.setup.ts
import '@testing-library/jest-dom';

// 全局Mock
global.ResizeObserver = jest.fn().mockImplementation(() => ({
  observe: jest.fn(),
  unobserve: jest.fn(),
  disconnect: jest.fn(),
}));

// 清理每个测试后的副作用

  jest.clearAllMocks();

// Mock和Spy示例

// 1. 函数Mock
describe('函数Mock', () => {
  test('jest.fn()基本用法', () => {
    const mockFn = jest.fn();

    // 调用函数
    mockFn('hello');
    mockFn('world');

    // 验证调用
    expect(mockFn).toHaveBeenCalled();
    expect(mockFn).toHaveBeenCalledTimes(2);
    expect(mockFn).toHaveBeenCalledWith('hello');
    expect(mockFn).toHaveBeenLastCalledWith('world');
  });

  test('jest.fn()返回值', () => {
    const mockFn = jest.fn()
      .mockReturnValueOnce('first')
      .mockReturnValueOnce('second')
      .mockReturnValue('default');

    expect(mockFn()).toBe('first');
    expect(mockFn()).toBe('second');
    expect(mockFn()).toBe('default');
    expect(mockFn()).toBe('default');
  });

  test('jest.fn()实现', () => {
    const mockFn = jest.fn((x: number) => x * 2);

    expect(mockFn(5)).toBe(10);
  });
});

  };

  test('spyOn基本用法', () => {
    const spy = jest.spyOn(calculator, 'add');

## 6. 面试高频问题

### 问题1：单元测试和集成测试的区别？

**答案：**
| 方面 | 单元测试 | 集成测试 |
|------|----------|----------|
| 范围 | 单个函数/组件 | 多个模块 |
| 速度 | 快 | 较慢 |
| 隔离 | 完全隔离 | 部分隔离 |
| Mock | 大量使用 | 较少使用 |

### 问题2：什么是TDD？

**答案：** TDD（测试驱动开发）是一种开发方法：
1. 先写失败的测试
2. 写最少代码使测试通过
3. 重构代码
4. 重复以上步骤

### 问题3：如何测试异步代码？

**答案：**
1. 使用 async/await
2. 使用 waitFor
3. 使用 findBy* 查询
4. Mock定时器

### 问题4：什么是Mock？为什么要用Mock？

**答案：** Mock是模拟对象，用于：
1. 隔离测试单元
2. 控制外部依赖
3. 提高测试速度
4. 测试边界情况

### 问题5：如何提高测试覆盖率？

**答案：**
1. 测试所有分支
2. 测试边界情况
3. 测试错误处理
4. 使用覆盖率工具找出未覆盖代码
5. 设置覆盖率阈值
