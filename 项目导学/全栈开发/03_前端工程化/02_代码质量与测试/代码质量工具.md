## 1. ESLint配置与规则详解

### 1.1 ESLint概述

ESLint是一个可配置的JavaScript和TypeScript代码检查工具，用于识别和修复代码中的问题。

```
┌─────────────────────────────────────────────────────────────┐
│                       ESLint 工作流程                         │
├─────────────────────────────────────────────────────────────┤
│                                                              │
│   源代码 (.js/.ts)                                          │
│         │                                                   │
│         ▼                                                   │
│   ┌──────────────┐                                         │
│   │   Parser     │ ← 解析代码为AST                         │
│   │ (espree/ts)  │                                         │
│   └──────┬───────┘                                         │
│          │                                                  │
│          ▼                                                  │
│   ┌──────────────┐                                         │
│   │  Linter      │ ← 遍历AST，应用规则                      │
│   │  Engine      │                                         │
│   └──────┬───────┘                                         │
│          │                                                  │
│          ▼                                                  │
│   ┌──────────────┐                                         │
│   │  Rules       │ ← 报告问题                              │
│   │  (no-unused) │                                         │
│   └──────────────┘                                         │
│                                                              │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 ESLint安装与基础配置

```bash
# 安装ESLint
npm install -D eslint

# 初始化配置
npx eslint --init

# 安装React相关
npm install -D eslint-plugin-react eslint-plugin-react-hooks

# 安装TypeScript相关
npm install -D @typescript-eslint/parser @typescript-eslint/eslint-plugin
```

```javascript
// .eslintrc.js
module.exports = {
    // 环境配置
    env: {
        browser: true,
        es2021: true,
        node: true

    // 解析器

            jsx: true

    // 继承的规则集
    extends: [
        'eslint:recommended',              // ESLint内置推荐规则
        'plugin:react/recommended',        // React推荐规则
        'plugin:@typescript-eslint/recommended', // TypeScript推荐规则
        'plugin:react-hooks/recommended',  // React Hooks规则
        'prettier'                         // 禁用与Prettier冲突的规则
    ],

    // 插件
    plugins: [
        'react',
        'react-hooks',
        '@typescript-eslint'
    ],

    // 自定义规则
    rules: {
        // === React规则 ===
        'react/react-in-jsx-scope': 'off',           // 关闭jsx必须导入React
        'react/prop-types': 'off',                  // 关闭prop-types检查（TypeScript用）
        'react/display-name': 'off',                 // 关闭组件displayName检查
        'react/jsx-uses-vars': 'warn',              // 标记使用的JSX变量
        'react/jsx-uses-react': 'warn',             // 标记使用的React

        // === TypeScript规则 ===
        '@typescript-eslint/no-unused-vars': ['warn', {
            argsIgnorePattern: '^_',
            varsIgnorePattern: '^_'

        '@typescript-eslint/no-explicit-any': 'warn',

        // === 代码风格规则 ===
        'no-console': 'warn',           // 警告console.log
        'no-debugger': 'error',         // 禁止debugger
        'prefer-const': 'error',         // 优先使用const

        'semi': ['error', 'always'],    // 必须使用分号

        // === 禁止规则 ===
        'no-param-reassign': 'error',   // 禁止修改函数参数
        'no-shadow': 'error',           // 禁止变量遮蔽
        'no-throw-literal': 'error'     // 禁止抛出字面量
    },

    // 全局变量
    globals: {
        __DEV__: 'readonly',
        __VERSION__: 'readonly'
    },

    // 覆盖配置
    overrides: [
        {
            files: ['*.test.ts', '*.spec.ts'],
            rules: {
                'no-unused-expressions': 'off',
                '@typescript-eslint/no-explicit-any': 'off'
            }
        },
        {
            files: ['*.config.js'],
            rules: {
                'no-unused-vars': 'off'
            }
        }

    // 共享设置

        react: {
            version: 'detect'
        }
    }
};
```

## 2. Prettier代码格式化

Prettier是一个"固执己见的"代码格式化工具，它消除了代码风格讨论，通过一致地格式化代码来保持项目风格统一。

### 2.2 Prettier配置

```bash
# 安装Prettier
npm install -D prettier

# 安装ESLint集成（解决冲突）
npm install -D eslint-config-prettier eslint-plugin-prettier
```


### 3.3 TypeScript类型守卫

```typescript
// 类型守卫示例

// typeof类型守卫
function padLeft(value: string | number, padding: string | number) {
    if (typeof padding === 'number') {
        // TypeScript知道这里是number
        return Array(padding + 1).join(' ') + value;
    }
    if (typeof padding === 'string') {
        // TypeScript知道这里是string
        return padding + value;
    }
}

// instanceof类型守卫
class Cat {
    meow() { console.log('meow'); }
}
class Dog {
    bark() { console.log('bark'); }
}
function speak(animal: Cat | Dog) {
    if (animal instanceof Cat) {
        animal.meow();
    } else {
        animal.bark();
    }
}

// 自定义类型守卫
function isString(value: unknown): value is string {
    return typeof value === 'string';
}

function isNonNullable<T>(value: T): value is NonNullable<T> {
    return value !== null && value !== undefined;
}

// in操作符类型守卫
interface Fish {
    swim: () => void;
}
interface Bird {
    fly: () => void;
}
function move(pet: Fish | Bird) {
    if ('swim' in pet) {
        pet.swim();
    } else {
        pet.fly();
    }
}

// 可辨识联合类型
type Shape =
    | { kind: 'circle'; radius: number }
    | { kind: 'rectangle'; width: number; height: number }
    | { kind: 'triangle'; base: number; height: number };

function getArea(shape: Shape): number {
    switch (shape.kind) {
        case 'circle':
            return Math.PI * shape.radius ** 2;
        case 'rectangle':
            return shape.width * shape.height;
        case 'triangle':
            return (shape.base * shape.height) / 2;
    }
}
```

## 4. Git Hooks集成

### 4.1 Husky与Lint-Staged

```bash
# 安装Husky和Lint-Staged
npm install -D husky lint-staged

# 初始化Husky
npx husky install

# 设置git hooks路径
npm pkg set scripts.prepare="husky install"
```

### 4.2 package.json配置

```json
{
    "lint-staged": {
        "*.{ts,tsx,js,jsx}": [
            "eslint --fix",
            "prettier --write"
        ],
        "*.{json,css,scss,md}": [
            "prettier --write"
        ],
        "*.ts": [
            "tsc --noEmit"
        ]
    }
}
```

### 4.3 创建Git Hook

```bash
# 创建pre-commit hook
npx husky add .husky/pre-commit "npx lint-staged"

# 创建commit-msg hook
npx husky add .husky/commit-msg "npx commitlint --edit $1"
```

## 6. CI/CD中的质量检查

### 6.1 GitHub Actions质量检查

```yaml
# .github/workflows/quality.yml
name: Code Quality

on:
    push:
        branches: [main, develop]
    pull_request:
        branches: [main]

jobs:
    lint:
        name: ESLint
        runs-on: ubuntu-latest
        steps:
            - uses: actions/checkout@v4
            - uses: actions/setup-node@v4
              with:
                  node-version: '20'
                  cache: 'npm'
            - run: npm ci
            - run: npm run lint
              env:
                  CI: true

    typecheck:
        name: TypeScript Check
        runs-on: ubuntu-latest
        steps:
            - uses: actions/checkout@v4
            - uses: actions/setup-node@v4
              with:
                  node-version: '20'
                  cache: 'npm'
            - run: npm ci
            - run: npm run typecheck

    format:
        name: Prettier Check
        runs-on: ubuntu-latest
        steps:
            - uses: actions/checkout@v4
            - uses: actions/setup-node@v4
              with:
                  node-version: '20'
                  cache: 'npm'
            - run: npm ci
            - run: npm run format:check

    quality:
        name: Full Quality Check
        runs-on: ubuntu-latest
        needs: [lint, typecheck, format]
        steps:
            - run: echo "All quality checks passed!"
```
