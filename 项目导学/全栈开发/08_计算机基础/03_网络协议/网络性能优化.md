## 1. HTTP协议与网络基础

### 1.1 HTTP请求生命周期

理解HTTP请求的完整生命周期是网络优化的基础。

```

│                     HTTP请求完整生命周期                              │
```

### 1.2 网络时间分析

```javascript
// 分析网络请求各阶段时间

function analyzeNetworkTiming() {
  const resources = performance.getEntriesByType('resource');

  return resources.map(resource => {
    const timing = {
      // 资源名称
      name: resource.name,

      // 资源类型
      type: resource.initiatorType,

      // DNS查询时间
      dns: resource.domainLookupEnd - resource.domainLookupStart,

      // TCP连接时间
      tcp: resource.connectEnd - resource.connectStart,

      // SSL握手时间
      ssl: resource.secureConnectionStart > 0
        ? resource.connectEnd - resource.secureConnectionStart
        : 0,

      // 请求时间
      request: resource.responseStart - resource.requestStart,

      // TTFB（首字节时间）
      ttfb: resource.responseStart - resource.requestStart,

      // 内容下载时间
      download: resource.responseEnd - resource.responseStart,

      // 总时间
      total: resource.responseEnd - resource.startTime,

      // 传输大小
      size: resource.transferSize,

      // 编码大小
      encodedSize: resource.encodedBodySize,
    };

    return timing;
  });
}

// 分析导航请求时间
function analyzeNavigationTiming() {
  const [navigation] = performance.getEntriesByType('navigation');

  const timing = {
    // DNS查询
    dns: navigation.domainLookupEnd - navigation.domainLookupStart,

    // TCP连接
    tcp: navigation.connectEnd - navigation.connectStart,

    // SSL握手
    ssl: navigation.secureConnectionStart > 0
      ? navigation.connectEnd - navigation.secureConnectionStart
      : 0,

    // TTFB
    ttfb: navigation.responseStart - navigation.requestStart,

    // 内容下载
    download: navigation.responseEnd - navigation.responseStart,

    // DOM解析
    domParse: navigation.domInteractive - navigation.responseEnd,

    // DOM内容加载
    domContentLoaded: navigation.domContentLoadedEventEnd - navigation.fetchStart,

    // 页面完全加载
    load: navigation.loadEventEnd - navigation.fetchStart,

    // 首次绘制
    firstPaint: getFirstPaint(),

    // 首次内容绘制
    fcp: getFCP(),
  };

  return timing;
}

function getFirstPaint() {
  const entries = performance.getEntriesByType('paint');
  const firstPaint = entries.find(e => e.name === 'first-paint');
  return firstPaint ? firstPaint.startTime : null;
}

function getFCP() {
  const entries = performance.getEntriesByType('paint');
  const fcp = entries.find(e => e.name === 'first-contentful-paint');
  return fcp ? fcp.startTime : null;
}

// 识别网络瓶颈
function identifyBottlenecks() {
  const timing = analyzeNavigationTiming();

  const bottlenecks = [];

  if (timing.dns > 100) {
    bottlenecks.push({
      type: 'dns',
      value: timing.dns,
      suggestion: '使用dns-prefetch预解析DNS'
    });
  }

  if (timing.tcp > 200) {
    bottlenecks.push({
      type: 'tcp',
      value: timing.tcp,
      suggestion: '使用preconnect预建立连接'
    });
  }

  if (timing.ttfb > 600) {
    bottlenecks.push({
      type: 'ttfb',
      value: timing.ttfb,
      suggestion: '优化服务器响应、使用CDN、启用缓存'
    });
  }

  if (timing.download > 1000) {
    bottlenecks.push({
      type: 'download',
      value: timing.download,
      suggestion: '压缩资源、使用CDN、启用Gzip/Brotli'
    });
  }

  if (timing.domParse > 1000) {
    bottlenecks.push({
      type: 'domParse',
      value: timing.domParse,
      suggestion: '减少DOM节点数量、优化HTML结构'
    });
  }

  return bottlenecks;
}
```

### 1.3 网络请求优化基础

```javascript
// 网络请求优化技巧

// 1. 并行请求 vs 串行请求
async function requestOptimization() {
  // ❌ 串行请求（慢）
  const user = await fetch('/api/user').then(r => r.json());
  const posts = await fetch('/api/posts').then(r => r.json());
  const comments = await fetch('/api/comments').then(r => r.json());

  // ✅ 并行请求（快）
  const [userData, postsData, commentsData] = await Promise.all([
    fetch('/api/user').then(r => r.json()),
    fetch('/api/posts').then(r => r.json()),
    fetch('/api/comments').then(r => r.json()),
  ]);

  return { user: userData, posts: postsData, comments: commentsData };
}

// 2. 请求取消
function cancellableRequest() {
  const controller = new AbortController();

  fetch('/api/data', { signal: controller.signal })
    .then(response => response.json())
    .then(data => console.log(data));

  // 取消请求
  controller.abort();

  // 超时取消
  const timeoutId = setTimeout(() => controller.abort(), 5000);
}

// 3. 请求重试
async function retryRequest(url, options = {}) {
  const { maxRetries = 3, retryDelay = 1000 } = options;

  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(url);
      if (response.ok) {
        return await response.json();
      }
      throw new Error(`HTTP ${response.status}`);
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(resolve => setTimeout(resolve, retryDelay * Math.pow(2, i)));
    }
  }
}

// 4. 请求缓存
const cache = new Map();

async function cachedFetch(url) {
  if (cache.has(url)) {
    return cache.get(url);
  }

  const response = await fetch(url);
  const data = await response.json();

  cache.set(url, data);
  return data;
}
```

## 2. HTTP/2与HTTP/3

### 2.1 HTTP/1.1的问题

HTTP/1.1虽然简单可靠，但存在严重的性能问题：

```
┌─────────────────────────────────────────────────────────────────────┐
│                    HTTP/1.1性能问题                                   │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  1. 队头阻塞（Head-of-Line Blocking）                                │
│     ├── 每个TCP连接同时只能处理一个请求                                 │
│     ├── 当前请求完成前，后续请求必须等待                                │
│     └── 即使浏览器允许6个并发连接，仍会出现排队                         │
│                                                                     │
│  2. 重复传输头部                                                     │
│     ├── 每个请求都要携带完整的HTTP头                                   │
│     ├── Cookie等信息在每次请求中重复传输                                │
│     └── 浪费带宽                                                      │
│                                                                     │
│  3. 资源内联                                                         │
│     └── 为减少请求，将CSS/JS内联到HTML中                              │
│                                                                     │
│  4. 域名分片                                                         │
│     └── 将资源分散到多个域名以突破并发限制                              │
│                                                                     │
│  请求示例：                                                           │
│     GET /index.html HTTP/1.1                                         │
│     Host: example.com                                                 │
│     Cookie: session=abc123...                                         │
│                                                                     │
│     HTTP/1.1 200 OK                                                   │
│     Content-Type: text/html                                           │
│     Content-Length: 1234                                               │
│     Set-Cookie: ...                                                   │
│                                                                     │
└─────────────────────────────────────────────────────────────────────┘
```

### 2.2 HTTP/2核心特性

HTTP/2通过多路复用、头部压缩等特性显著提升了性能：

```javascript
// HTTP/2核心特性

// 1. 多路复用（Multiplexing）
// 单一TCP连接上可以并行传输多个请求和响应
/*
 * HTTP/1.1：
 * GET /index.html    |____|
 * GET /style.css     |________|
 * GET /script.js     |______________|
 *
 * HTTP/2：
 * GET /index.html    |||
 * GET /style.css     |||||||||||
 * GET /script.js     |||||||||||||||||
 */

// 2. 服务器推送（Server Push）
// 服务器可以主动推送资源，而不需要客户端请求
// 服务器配置示例（Nginx）：
/*
server {
    http2_push_preload on;

    location / {
        index index.html;
    }
}
*/

// 手动推送资源
// 在HTTP头中指定要推送的资源
/*
Link: </style.css>; rel=preload; as=style
Link: </script.js>; rel=preload; as=script
*/

// 3. 头部压缩（HPACK）
// 使用HPACK算法压缩HTTP头
/*
 * 静态表：常用头部名称和值（如:method GET）
 * 动态表：本次连接中使用的具体值
 * 哈夫曼编码：字符串压缩
 */

// 4. 流优先级
// 可以为请求设置优先级
// 高优先级：HTML文档
// 中优先级：CSS、初始JS
// 低优先级：图片、懒加载资源

// 检测HTTP/2支持
function checkHTTP2Support() {
  const connection = performance.getEntriesByType('resource')[0];
  if (connection && connection.transferSize > 0) {
    // 浏览器会显示协议信息（在DevTools中查看）
    console.log('检查Network面板的Protocol列');
  }
}
```

### 2.3 HTTP/3与QUIC

HTTP/3使用QUIC协议，完全解决了TCP+TLS的握手延迟问题：

```javascript
// HTTP/3核心特性

// 1. 0-RTT / 1-RTT连接建立
/*
 * HTTP/1.1 + TLS：
 * TCP握手(3步) + TLS握手(2步) = 5步延迟
 *
 * HTTP/2 + TLS：
 * TCP握手(3步) + TLS握手(2步) = 5步延迟
 *
 * HTTP/3 (QUIC)：
 * QUIC握手(1步) = 1步延迟（首次连接）
 * 0-RTT = 0步延迟（重复连接）
 */

// 2. 独立流
/*
 * HTTP/2：
 * 单一TCP连接上的多个流
 * TCP层丢包会影响所有流
 *
 * HTTP/3：
 * QUIC连接上的多个独立流
 * 单个流丢包只影响该流
 */

// 3. 连接迁移
/*
 * 从WiFi切换到移动网络时
 * HTTP/1.1/2：需要重新建立连接
 * HTTP/3：连接ID保持，连接迁移
 */

// 4. 前向纠错
// 丢包时不需要重传，接收方可以直接恢复数据

// 检测HTTP/3支持
function checkHTTP3Support() {
  // 检查浏览器是否支持HTTP/3
  if (navigator.connection && navigator.connection.effectiveType === '5g') {
    console.log('5G网络环境可能使用HTTP/3');
  }

  // 在DevTools Network面板查看Protocol列
  // h3 = HTTP/3, h2 = HTTP/2, http/1.1 = HTTP/1.1
}

// HTTP/3配置（Nginx示例）
/*
http3 on;
listen 443 ssl http2;
listen 443 quic reuseport;

ssl_protocols TLSv1.3;
ssl_prefer_server_ciphers off;
*/
```

### 3.2 Tree Shaking

Tree Shaking是移除未使用代码的优化技术。

```javascript
// Tree Shaking实现原理

// 1. ES Modules静态结构
// math.js
export function add(a, b) {
  return a + b;
}

export function subtract(a, b) {
  return a - b;
}

// index.js
import { add } from './math';

add(1, 2); // subtract会被Tree Shaking移除

// 2. sideEffects配置
// package.json
{
  "sideEffects": [
    "**/*.css",
    "**/*.scss",
    "./src/polyfills.js"
  ]
}

// sideEffects: false 表示所有模块都是纯净的，可以安全删除未使用的导出

// 3. 常见问题及解决方案
// 问题1：CommonJS模块无法Tree Shaking
// 解决：使用ES Modules格式的库（如lodash-es）

// 问题2：动态导入可能无法Tree Shaking
// 解决：确保动态导入的模块也是ES Modules

// 问题3：副作用
// .css文件有副作用，不能删除
// 某些函数执行时有副作用（如修改全局状态）

// 4. 检测Tree Shaking效果
// webpack-bundle-analyzer
import { bundleAnalysis } from './tools/analyzer';

// 在生产构建时分析
bundleAnalysis.analyze({
  analyzerMode: 'static',
  reportFilename: 'bundle-report.html',
});
```

```javascript
// 1. 懒加载实现
function lazyLoadImages() {
  const images = document.querySelectorAll('img[data-src]');

  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const img = entry.target;
        img.src = img.dataset.src;
        img.removeAttribute('data-src');
        observer.unobserve(img);
      }
    });
  }, {
    rootMargin: '50px', // 提前50px开始加载
    threshold: 0.01,
  });

  images.forEach(img => observer.observe(img));
}

// 2. 图片压缩（使用canvas）
async function compressImage(file, quality = 0.8, maxWidth = 1920) {
  return new Promise((resolve) => {
    const img = new Image();
    img.onload = () => {
      // 计算新尺寸
      let width = img.width;
      let height = img.height;

      if (width > maxWidth) {
        height = (height * maxWidth) / width;
        width = maxWidth;
      }

      // 创建canvas
      const canvas = document.createElement('canvas');
      canvas.width = width;
      canvas.height = height;

      const ctx = canvas.getContext('2d');
      ctx.drawImage(img, 0, 0, width, height);

      // 输出
      canvas.toBlob(
        (blob) => resolve(blob),
        'image/webp',
        quality
      );
    };
    img.src = URL.createObjectURL(file);
  });
}

// 3. 响应式图片srcset解析
function parseSrcset(srcset) {
  return srcset.split(',').map(item => {
    const [url, descriptor] = item.trim().split(' ');
    const parsed = {
      url,
      width: parseInt(descriptor.replace('w', ''), 10),
    };
    return parsed;
  });
}

// 4. 选择合适的图片尺寸
function selectBestImage(images, viewportWidth) {
  const sorted = images.sort((a, b) => a.width - b.width);

  for (let i = 0; i < sorted.length; i++) {
    if (sorted[i].width >= viewportWidth * window.devicePixelRatio) {
      return sorted[i].url;
    }
  }

  // 返回最大的图片
  return sorted[sorted.length - 1].url;
}
```

```javascript
  const { request } = event;
  const url = new URL(request.url);

  // API请求：网络优先
  if (url.pathname.startsWith('/api/')) {
    event.respondWith(
      fetch(request)
        .then(response => {
          const clone = response.clone();
          caches.open(CACHE_NAME)
            .then(cache => cache.put(request, clone));
          return response;
        })
        .catch(() => caches.match(request))
    );
    return;
  }

  // 静态资源：缓存优先
  event.respondWith(
    caches.match(request)
      .then(response => {
        if (response) {
          return response;
        }
        return fetch(request)
          .then(serverResponse => {
            const clone = serverResponse.clone();
            caches.open(CACHE_NAME)
              .then(cache => cache.put(request, clone));
            return serverResponse;
          });
      })
  );
```

```javascript
// 4. 请求去重
class RequestDeduplicator {
  constructor() {
    this.pending = new Map();
  }

  async fetch(url, options) {
    const key = url + JSON.stringify(options);

    if (this.pending.has(key)) {
      return this.pending.get(key);
    }

    const promise = fetch(url, options)
      .then(r => r.json())
      .finally(() => this.pending.delete(key));

    this.pending.set(key, promise);
    return promise;
  }
}

    const cached = this.memory.get(url);
```

```javascript
class WebSocketManager {
  constructor(url) {
    this.url = url;
    this.ws = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
  }

    this.ws = new WebSocket(this.url);

    this.ws.onopen = () => {
      console.log('WebSocket连接已建立');
      this.reconnectAttempts = 0;
    };

    this.ws.onmessage = (event) => {
      this.handleMessage(JSON.parse(event.data));
    };

      console.error('WebSocket错误:', error);
    };

    this.ws.onclose = () => {
      this.reconnect();
    };
  }

  reconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempts), 30000);

      setTimeout(() => {
        console.log(`尝试重连 (${this.reconnectAttempts}/${this.maxReconnectAttempts})`);
        this.connect();
      }, delay);
    }
  }

  handleMessage(data) {
    console.log('收到消息:', data);
  }

  send(data) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(data));
    }
  }
}
```

## 9. 面试高频问题

### 问题1：HTTP/2和HTTP/1.1的主要区别是什么？

**答案：**

| 特性 | HTTP/1.1 | HTTP/2 |
|------|----------|--------|
| **多路复用** | 每个请求需要单独TCP连接 | 单一TCP连接并行多请求 |
| **头部压缩** | 每次请求重复传输 | HPACK压缩 |
| **服务器推送** | 不支持 | 支持 |
| **二进制传输** | 文本格式 | 二进制帧 |
| **流优先级** | 不支持 | 支持 |
| **队头阻塞** | 存在 | 减少 |

**核心优势：**
- 消除队头阻塞
- 减少TCP/TLS握手开销
- 更高效的利用连接

### 问题3：强缓存和协商缓存的区别？

**答案：**

**强缓存：**
- 不发送请求到服务器
- 通过`Cache-Control`和`Expires`控制
- `Cache-Control: max-age=3600`（设置缓存时长）
- `Expires: Wed, 21 Oct 2025 07:28:00 GMT`（设置过期时间）

**协商缓存：**
- 发送请求到服务器验证
- 通过`Last-Modified`和`ETag`控制
- 服务器返回`304 Not Modified`表示缓存有效
- `ETag`比`Last-Modified`更精确

**最佳实践：**
- HTML文件：使用协商缓存（no-cache）
- 静态资源：使用强缓存（max-age=1y, immutable）
- API响应：根据数据时效性选择
