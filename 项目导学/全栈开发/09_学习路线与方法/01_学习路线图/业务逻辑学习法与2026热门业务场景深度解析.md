**错误做法**：

- ❌ 直接看代码，不理解业务背景
- ❌ 只关注功能实现，不关注业务规则

## 二、核心概念

### 2.1 业务建模（Event Storming）

#### 2.1.1 事件风暴原理

事件风暴（Event Storming）是一种协作式建模技术，通过识别业务事件、命令和聚合来理解业务流程。

**核心概念**：

- **事件（Event）**：过去发生的、不可改变的事实（如：用户注册、订单创建）
- **命令（Command）**：期望发生的动作（如：注册用户、创建订单）
- **聚合（Aggregate）**：一组相关对象，保证业务规则的一致性
- **领域服务（Domain Service）**：跨聚合的业务逻辑

#### 2.1.2 事件风暴实践

// 3. 识别聚合
class Order {
  private id: string;
  private userId: string;
  private items: OrderItem[];
  private status: OrderStatus;
  private total: number;
  private createdAt: string;
  private paidAt?: string;
  private shippedAt?: string;
  private deliveredAt?: string;
  private cancelledAt?: string;

  constructor(id: string, userId: string, items: OrderItem[]) {
    this.id = id;
    this.userId = userId;
    this.items = items;
    this.status = 'pending';
    this.total = items.reduce((sum, item) => sum + item.price * item.quantity, 0);
    this.createdAt = new Date().toISOString();
  }

  // 领域方法：支付订单
  pay(paymentId: string, amount: number): void {
    if (this.status !== 'pending') {
      throw new Error('订单状态不允许支付');
    }
    if (amount !== this.total) {
      throw new Error('支付金额不匹配');
    }
    
    this.status = 'paid';
    this.paidAt = new Date().toISOString();
  }

  // 领域方法：发货
  ship(trackingNumber: string): void {
    if (this.status !== 'paid') {
      throw new Error('只有已支付订单可以发货');
    }
    
    this.status = 'shipped';
    this.shippedAt = new Date().toISOString();
  }

  // 领域方法：确认收货
  deliver(): void {
    if (this.status !== 'shipped') {
      throw new Error('只有已发货订单可以确认收货');
    }
    
    this.status = 'delivered';
    this.deliveredAt = new Date().toISOString();
  }

  // 领域方法：取消订单
  cancel(reason: string): void {
    if (this.status === 'delivered') {
      throw new Error('已收货订单不能取消');
    }
    
    this.status = 'cancelled';
    this.cancelledAt = new Date().toISOString();
  }
}

// 4. 识别领域服务
class OrderService {
  private orderRepository: OrderRepository;
  private paymentService: PaymentService;

  constructor(
    orderRepository: OrderRepository,
    paymentService: PaymentService
  ) {
    this.orderRepository = orderRepository;
    this.paymentService = paymentService;
  }

  // 创建订单
  async createOrder(userId: string, items: OrderItem[]): Promise<Order> {
    const order = new Order(
      uuid.v4(),
      userId,
      items
    );
    
    await this.orderRepository.save(order);
    return order;
  }

  // 支付订单
  async payOrder(orderId: string, paymentMethod: string): Promise<void> {
    const order = await this.orderRepository.findById(orderId);
    
    if (!order) {
      throw new Error('订单不存在');
    }

    // 调用支付服务
    const payment = await this.paymentService.createPayment(
      order.userId,
      order.total,
      paymentMethod
    );

    // 支付成功后更新订单
    order.pay(payment.id, payment.amount);
    await this.orderRepository.save(order);
  }

  // 发货
  async shipOrder(orderId: string, trackingNumber: string): Promise<void> {
    const order = await this.orderRepository.findById(orderId);
    
    if (!order) {
      throw new Error('订单不存在');
    }

    order.ship(trackingNumber);
    await this.orderRepository.save(order);
  }

  // 取消订单
  async cancelOrder(orderId: string, reason: string): Promise<void> {
    const order = await this.orderRepository.findById(orderId);
    
    if (!order) {
      throw new Error('订单不存在');
    }

    order.cancel(reason);
    await this.orderRepository.save(order);
  }
}

### 2.2 数据生命周期（Data Flow）

#### 2.2.1 数据生命周期模型

数据在系统中的生命周期可以分为五个阶段：

1. **创建（Create）**：数据首次生成
2. **读取（Read）**：数据被读取和使用
3. **更新（Update）**：数据被修改
4. **删除（Delete）**：数据被删除
5. **归档（Archive）**：数据被归档保存

#### 2.2.2 数据生命周期管理

```typescript
// 数据生命周期管理示例
interface DataLifecycle<T> {
  // 创建数据
  create(data: T): Promise<T>;
  
  // 读取数据
  read(id: string): Promise<T | null>;
  readAll(): Promise<T[]>;
  
  // 更新数据
  update(id: string, data: Partial<T>): Promise<T>;
  
  // 删除数据（软删除）
  delete(id: string): Promise<void>;
  
  // 归档数据
  archive(id: string): Promise<void>;
  
  // 恢复数据
  restore(id: string): Promise<void>;
}

// 实现数据生命周期
class UserRepository implements DataLifecycle<User> {
  private db: Database;
  private cache: Cache;

  async create(user: User): Promise<User> {
    // 1. 验证数据
    this.validateUser(user);
    
    // 2. 保存到数据库
    const savedUser = await this.db.users.insert(user);
    
    // 3. 更新缓存
    await this.cache.set(`user:${savedUser.id}`, savedUser);
    
    return savedUser;
  }

  async read(id: string): Promise<User | null> {
    // 1. 尝试从缓存读取
    const cached = await this.cache.get<User>(`user:${id}`);
    if (cached) {
      return cached;
    }
    
    // 2. 从数据库读取
    const user = await this.db.users.findById(id);
    
    // 3. 更新缓存
    if (user) {
      await this.cache.set(`user:${id}`, user);
    }
    
    return user;
  }

  async update(id: string, userData: Partial<User>): Promise<User> {
    // 1. 读取现有数据
    const existingUser = await this.read(id);
    if (!existingUser) {
      throw new Error('用户不存在');
    }
    
    // 2. 合并数据
    const updatedUser = { ...existingUser, ...userData };
    
    // 3. 验证数据
    this.validateUser(updatedUser);
    
    // 4. 保存到数据库
    await this.db.users.update(id, updatedUser);
    
    // 5. 更新缓存
    await this.cache.set(`user:${id}`, updatedUser);
    
    return updatedUser;
  }

  async delete(id: string): Promise<void> {
    // 1. 读取现有数据
    const user = await this.read(id);
    if (!user) {
      throw new Error('用户不存在');
    }
    
    // 2. 软删除（设置删除时间）
    await this.db.users.update(id, {
      ...user,
      deletedAt: new Date().toISOString()
    });
    
    // 3. 更新缓存
    await this.cache.set(`user:${id}`, {
      ...user,
      deletedAt: new Date().toISOString()
    });
  }

  async archive(id: string): Promise<void> {
    // 1. 读取现有数据
    const user = await this.read(id);
    if (!user) {
      throw new Error('用户不存在');
    }
    
    // 2. 归档到归档表
    await this.db.archivedUsers.insert({
      ...user,
      archivedAt: new Date().toISOString()
    });
    
    // 3. 删除主表数据
    await this.db.users.delete(id);
    
    // 4. 清除缓存
    await this.cache.delete(`user:${id}`);
  }

  async restore(id: string): Promise<void> {
    // 1. 从归档表读取数据
    const archivedUser = await this.db.archivedUsers.findById(id);
    if (!archivedUser) {
      throw new Error('归档用户不存在');
    }
    
    // 2. 恢复到主表
    await this.db.users.insert({
      ...archivedUser,
      deletedAt: undefined,
      restoredAt: new Date().toISOString()
    });
    
    // 3. 删除归档数据
    await this.db.archivedUsers.delete(id);
    
    // 4. 更新缓存
    await this.cache.delete(`user:${id}`);
  }

  // 验证用户数据
  private validateUser(user: User): void {
    if (!user.name || user.name.length < 2) {
      throw new Error('用户名至少2个字符');
    }
    
    if (!user.email || !this.isValidEmail(user.email)) {
      throw new Error('邮箱格式不正确');
    }
  }

  private isValidEmail(email: string): boolean {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }
}
```

### 2.3 代码映射（Code Mapping）

#### 2.3.1 代码映射原则

将业务逻辑映射到代码时，需要遵循以下原则：

1. **单一职责**：每个类/函数只做一件事
2. **高内聚**：相关功能聚合在一起
3. **低耦合**：模块间通过接口通信
4. **可测试**：代码易于测试和维护

#### 2.3.2 代码映射模式

```typescript
// 代码映射模式示例：订单状态管理
// 1. 使用状态机管理业务状态
type OrderStatus = 'pending' | 'paid' | 'shipped' | 'delivered' | 'cancelled';

interface OrderState {
  status: OrderStatus;
  statusHistory: { status: OrderStatus; timestamp: string }[];
}

class OrderStateMachine {
  private state: OrderState;

  constructor(initialStatus: OrderStatus) {
    this.state = {
      status: initialStatus,
      statusHistory: [{ status: initialStatus, timestamp: new Date().toISOString() }]
    };
  }

  // 状态转换
  transition(toStatus: OrderStatus): void {
    const allowedTransitions = this.getAllowedTransitions(this.state.status);
    
    if (!allowedTransitions.includes(toStatus)) {
      throw new Error(`不允许从${this.state.status}转换到${toStatus}`);
    }
    
    this.state.status = toStatus;
    this.state.statusHistory.push({
      status: toStatus,
      timestamp: new Date().toISOString()
    });
  }

  // 获取允许的状态转换
  private getAllowedTransitions(currentStatus: OrderStatus): OrderStatus[] {
    const transitions: Record<OrderStatus, OrderStatus[]> = {
      pending: ['paid', 'cancelled'],
      paid: ['shipped', 'cancelled'],
      shipped: ['delivered'],
      delivered: [],
      cancelled: []
    };
    
    return transitions[currentStatus] || [];
  }

  // 获取当前状态
  getStatus(): OrderStatus {
    return this.state.status;
  }

  // 获取状态历史
  getStatusHistory(): { status: OrderStatus; timestamp: string }[] {
    return this.state.statusHistory;
  }
}

// 2. 使用策略模式处理不同状态的业务逻辑
interface OrderStrategy {
  canPay(): boolean;
  canShip(): boolean;
  canCancel(): boolean;
}

class PendingOrderStrategy implements OrderStrategy {
  canPay(): boolean { return true; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return true; }
}

class PaidOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return true; }
  canCancel(): boolean { return true; }
}

class ShippedOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return false; }
}

class DeliveredOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return false; }
}

class CancelledOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return false; }
}

// 3. 使用工厂模式创建策略
class OrderStrategyFactory {
  static createStrategy(status: OrderStatus): OrderStrategy {
    switch (status) {
      case 'pending':
        return new PendingOrderStrategy();
      case 'paid':
        return new PaidOrderStrategy();
      case 'shipped':
        return new ShippedOrderStrategy();
      case 'delivered':
        return new DeliveredOrderStrategy();
      case 'cancelled':
        return new CancelledOrderStrategy();
      default:
        throw new Error(`未知状态: ${status}`);
    }
  }
}

// 4. 组合使用
class Order {
  private stateMachine: OrderStateMachine;
  private strategyFactory: OrderStrategyFactory;

  constructor() {
    this.stateMachine = new OrderStateMachine('pending');
    this.strategyFactory = new OrderStrategyFactory();
  }

  pay(): void {
    const strategy = this.strategyFactory.createStrategy(this.stateMachine.getStatus());
    
    if (!strategy.canPay()) {
      throw new Error('当前状态不允许支付');
    }
    
    this.stateMachine.transition('paid');
  }

  ship(): void {
    const strategy = this.strategyFactory.createStrategy(this.stateMachine.getStatus());
    
    if (!strategy.canShip()) {
      throw new Error('当前状态不允许发货');
    }
    
    this.stateMachine.transition('shipped');
  }

  cancel(): void {
    const strategy = this.strategyFactory.createStrategy(this.stateMachine.getStatus());
    
    if (!strategy.canCancel()) {
      throw new Error('当前状态不允许取消');
    }
    
    this.stateMachine.transition('cancelled');
  }

  getStatus(): OrderStatus {
    return this.stateMachine.getStatus();
  }

  getStatusHistory(): { status: OrderStatus; timestamp: string }[] {
    return this.stateMachine.getStatusHistory();
  }
}
```

### 3.1 AI Agent编排系统

#### 3.1.1 业务本质

AI Agent编排系统不再是简单的CRUD，而是**"长耗时、状态化的工作流（Stateful Workflows）"**。

#### 3.1.2 核心挑战

- **执行时间不可控**：LLM推理可能需要30秒
- **重试逻辑复杂**：API调用失败了，是重跑整个工作流还是断点重续？

#### 3.1.3 技术实现

```typescript
// AI Agent编排系统实现
// 1. 定义Agent状态
type AgentStatus = 'idle' | 'processing' | 'completed' | 'failed';

interface AgentState {
  status: AgentStatus;
  currentStep: number;
  totalSteps: number;
  result: any;
  error?: string;
  logs: AgentLog[];
}

interface AgentLog {
  timestamp: string;
  level: 'info' | 'warn' | 'error';
  message: string;
  details?: any;
}

// 2. 定义Agent任务
interface AgentTask {
  id: string;
  name: string;
  description: string;
  steps: AgentStep[];
  timeout: number;
  retryCount: number;
}

interface AgentStep {
  id: string;
  type: 'llm' | 'tool' | 'condition' | 'loop';
  name: string;
  description: string;
  config: any;
}

// 3. 实现Agent执行器
class AgentExecutor {
  private state: AgentState;
  private task: AgentTask;
  private llmService: LLMService;
  private toolService: ToolService;

  constructor(task: AgentTask) {
    this.task = task;
    this.state = {
      status: 'idle',
      currentStep: 0,
      totalSteps: task.steps.length,
      result: null,
      logs: []
    };
    
    this.llmService = new LLMService();
    this.toolService = new ToolService();
  }

  // 执行Agent任务
  async execute(): Promise<AgentState> {
    this.log('info', '开始执行Agent任务');
    this.state.status = 'processing';

    try {
      for (let i = 0; i < this.task.steps.length; i++) {
        this.state.currentStep = i;
        const step = this.task.steps[i];

        this.log('info', `开始执行步骤: ${step.name}`, { step });

        // 执行步骤
        const result = await this.executeStep(step);

        this.log('info', `步骤执行完成: ${step.name}`, { result });

        // 检查是否需要中断
        if (this.shouldInterrupt()) {
          this.state.status = 'failed';
          this.state.error = '任务被中断';
          return this.state;
        }
      }

      this.state.status = 'completed';
      this.state.result = this.state.logs;
      this.log('info', 'Agent任务执行完成');

    } catch (error) {
      this.state.status = 'failed';
      this.state.error = error.message;
      this.log('error', 'Agent任务执行失败', { error });
    }

    return this.state;
  }

  // 执行单个步骤
  private async executeStep(step: AgentStep): Promise<any> {
    // 重试机制
    for (let i = 0; i < this.task.retryCount; i++) {
      try {
        switch (step.type) {
          case 'llm':
            return await this.executeLLMStep(step);
          case 'tool':
            return await this.executeToolStep(step);
          case 'condition':
            return await this.executeConditionStep(step);
          case 'loop':
            return await this.executeLoopStep(step);
          default:
            throw new Error(`未知步骤类型: ${step.type}`);
        }
      } catch (error) {
        this.log('warn', `步骤执行失败，重试中 (${i + 1}/${this.task.retryCount})`, { error });
        
        if (i === this.task.retryCount - 1) {
          throw error;
        }
        
        // 指数退避
        await new Promise(resolve => 
          setTimeout(resolve, Math.pow(2, i) * 1000)
        );
      }
    }
  }

  // 执行LLM步骤
  private async executeLLMStep(step: AgentStep): Promise<any> {
    const { prompt, model, temperature } = step.config;
    
    this.log('info', '调用LLM服务', { model, temperature });
    
    const result = await this.llmService.generate(prompt, {
      model,
      temperature,
      timeout: this.task.timeout
    });

    return result;
  }

  // 执行工具步骤
  private async executeToolStep(step: AgentStep): Promise<any> {
    const { toolName, input } = step.config;
    
    this.log('info', '调用工具', { toolName, input });
    
    const result = await this.toolService.execute(toolName, input);
    
    return result;
  }

  // 执行条件步骤
  private async executeConditionStep(step: AgentStep): Promise<boolean> {
    const { condition, ifTrue, ifFalse } = step.config;
    
    this.log('info', '执行条件判断', { condition });
    
    const result = this.evaluateCondition(condition);
    
    if (result) {
      this.log('info', '条件为真，执行ifTrue分支');
      return true;
    } else {
      this.log('info', '条件为假，执行ifFalse分支');
      return false;
    }
  }

  // 执行循环步骤
  private async executeLoopStep(step: AgentStep): Promise<any[]> {
    const { items, loopBody } = step.config;
    
    this.log('info', '开始循环', { itemCount: items.length });
    
    const results = [];
    
    for (const item of items) {
      this.log('info', '处理循环项', { item });
      
      const result = await this.executeStep({
        ...loopBody,
        config: {
          ...loopBody.config,
          item
        }
      });
      
      results.push(result);
    }
    
    return results;
  }

  // 评估条件

#### 3.2.1 业务本质

// 实时协同编辑器实现
// 1. 定义文档操作
type DocumentOperation = 
  | { type: 'insert'; position: number; text: string }
  | { type: 'delete'; position: number; length: number }
  | { type: 'update'; position: number; text: string };

// 2. 实现OT算法
class OTAlgorithm {
  // 转换操作
  transform(op1: DocumentOperation, op2: DocumentOperation): [DocumentOperation, DocumentOperation] {
    if (op1.type === 'insert' && op2.type === 'insert') {
      // 两个插入操作
      if (op1.position <= op2.position) {
        return [op1, { ...op2, position: op2.position + op1.text.length }];
      } else {
        return [{ ...op1, position: op1.position + op2.text.length }, op2];
      }
    } else if (op1.type === 'delete' && op2.type === 'delete') {
      // 两个删除操作
      if (op1.position <= op2.position) {
        if (op1.position + op1.length <= op2.position) {
          return [op1, { ...op2, position: op2.position - op1.length }];
        } else {
          // 重叠删除
          return [op1, { ...op2, position: op2.position, length: Math.max(0, op2.length - (op1.position + op1.length - op2.position)) }];
        }
      } else {
        if (op2.position + op2.length <= op1.position) {
          return [{ ...op1, position: op1.position - op2.length }, op2];
        } else {
          // 重叠删除
          return [{ ...op1, position: op1.position, length: Math.max(0, op1.length - (op2.position + op2.length - op1.position)) }, op2];
        }
      }
    } else if (op1.type === 'insert' && op2.type === 'delete') {
      // 插入和删除操作
      if (op1.position <= op2.position) {
        return [op1, { ...op2, position: op2.position + op1.text.length }];
      } else {
        if (op2.position + op2.length <= op1.position) {
          return [{ ...op1, position: op1.position - op2.length }, op2];
        } else {
          // 插入在删除范围内
          return [op1, { ...op2, length: Math.max(0, op2.length - op1.text.length) }];
        }
      }
    } else if (op1.type === 'delete' && op2.type === 'insert') {
      // 删除和插入操作
      if (op1.position + op1.length <= op2.position) {
        return [op1, { ...op2, position: op2.position + op1.length }];
      } else {
        if (op1.position <= op2.position) {
          return [{ ...op1, length: op1.length + op2.text.length }, op2];
        } else {
          return [{ ...op1, position: op1.position - op2.text.length }, op2];
        }
      }
    }
    
    return [op1, op2];
  }

  // 组合操作
  compose(ops: DocumentOperation[]): DocumentOperation[] {
    if (ops.length === 0) return [];
    if (ops.length === 1) return ops;

    const [first, ...rest] = ops;
    
    if (first.type === 'insert' && rest[0]?.type === 'insert') {
      // 合并连续的插入操作
      return [{ ...first, text: first.text + rest[0].text }, ...rest.slice(1)];
    }

    return [first, ...this.compose(rest)];
  }
}

// 3. 实现CRDT算法
class CRDTAlgorithm {
  // 为每个字符分配唯一ID
  private idCounter = 0;
  private idPrefix = Date.now().toString(36);

  private generateId(): string {
    return `${this.idPrefix}-${this.idCounter++}`;
  }

  // 定义文档元素
  interface DocumentElement {
    id: string;
    text: string;
    deleted: boolean;
    createdAt: number;
    deletedAt?: number;
  }

  // 插入操作
  insert(text: string, position: number, elements: DocumentElement[]): DocumentElement[] {
    const newElements: DocumentElement[] = text.split('').map(char => ({
      id: this.generateId(),
      text: char,
      deleted: false,
      createdAt: Date.now()
    }));

    return [
      ...elements.slice(0, position),
      ...newElements,
      ...elements.slice(position)
    ];
  }

  // 删除操作
  delete(position: number, length: number, elements: DocumentElement[]): DocumentElement[] {
    return elements.map((element, index) => {
      if (index >= position && index < position + length) {
        return { ...element, deleted: true, deletedAt: Date.now() };
      }
      return element;
    });
  }

  // 合并文档
  merge(elements1: DocumentElement[], elements2: DocumentElement[]): DocumentElement[] {
    // 使用CRDT合并算法
    const merged = new Map<string, DocumentElement>();

    // 添加所有元素
    elements1.forEach(element => merged.set(element.id, element));
    elements2.forEach(element => {
      const existing = merged.get(element.id);
      if (!existing || element.createdAt > existing.createdAt) {
        merged.set(element.id, element);
      }
    });

    // 按创建时间排序
    return Array.from(merged.values()).sort((a, b) => a.createdAt - b.createdAt);
  }

  // 获取文档内容
  getContent(elements: DocumentElement[]): string {
    return elements
      .filter(element => !element.deleted)
      .map(element => element.text)
      .join('');
  }
}

// 4. 实现协同编辑器
class CollaborativeEditor {
  private content: string;
  private ot: OTAlgorithm;
  private crdt: CRDTAlgorithm;
  private operations: DocumentOperation[] = [];
  private peers: Map<string, DocumentOperation[]> = new Map();

  constructor(initialContent: string = '') {
    this.content = initialContent;
    this.ot = new OTAlgorithm();
    this.crdt = new CRDTAlgorithm();
  }

  // 本地操作
  localOperation(op: DocumentOperation): void {
    this.operations.push(op);
    this.applyOperation(op);
  }

  // 应用操作
  private applyOperation(op: DocumentOperation): void {
    switch (op.type) {
      case 'insert':
        this.content = this.content.slice(0, op.position) + op.text + this.content.slice(op.position);
        break;
      case 'delete':
        this.content = this.content.slice(0, op.position) + this.content.slice(op.position + op.length);
        break;
      case 'update':
        this.content = this.content.slice(0, op.position) + op.text + this.content.slice(op.position + op.text.length);
        break;
    }
  }

  // 接收远程操作
  remoteOperation(peerId: string, op: DocumentOperation): void {
    // 转换操作
    for (const [id, ops] of this.peers) {
      if (id !== peerId) {
        const [transformedOp, _] = this.ot.transform(op, ops[0]);
        ops[0] = transformedOp;
      }
    }

    // 应用操作
    this.applyOperation(op);

    // 广播操作
    this.peers.set(peerId, [op]);
  }

  // 合并文档
  merge(peerId: string, remoteContent: string): void {
    const remoteElements = this.crdt.crdt.parse(remoteContent);
    const localElements = this.crdt.crdt.parse(this.content);

    const mergedElements = this.crdt.crdt.merge(localElements, remoteElements);
    this.content = this.crdt.crdt.getContent(mergedElements);
  }

  // 获取文档内容
  getContent(): string {
    return this.content;
  }

  // 获取操作历史
  getOperations(): DocumentOperation[] {
    return this.operations;
  }
}

      deletedAt: new Date().toISOString()
    });
    
    await this.cache.set(`user:${id}`, {
      ...user,
      deletedAt: new Date().toISOString()
    });
  }

  async archive(id: string): Promise<void> {
    const user = await this.read(id);
    if (!user) throw new Error('用户不存在');
    
    await this.db.archivedUsers.insert({
      ...user,
      archivedAt: new Date().toISOString()
    });
    
    await this.db.users.delete(id);
    await this.cache.delete(`user:${id}`);
  }

  async restore(id: string): Promise<void> {
    const archivedUser = await this.db.archivedUsers.findById(id);
    if (!archivedUser) throw new Error('归档用户不存在');
    
    await this.db.users.insert({
      ...archivedUser,
      deletedAt: undefined,
      restoredAt: new Date().toISOString()
    });
    
    await this.db.archivedUsers.delete(id);
    await this.cache.delete(`user:${id}`);
  }

  private validateUser(user: User): void {
    if (!user.name || user.name.length < 2) {
      throw new Error('用户名至少2个字符');
    }
    
    if (!user.email || !this.isValidEmail(user.email)) {
      throw new Error('邮箱格式不正确');
    }
  }

  private isValidEmail(email: string): boolean {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }
}

// 订单状态管理
type OrderStatus = 'pending' | 'paid' | 'shipped' | 'delivered' | 'cancelled';

interface OrderState {
  status: OrderStatus;
  statusHistory: { status: OrderStatus; timestamp: string }[];
}

class OrderStateMachine {
  private state: OrderState;

  constructor(initialStatus: OrderStatus) {
    this.state = {
      status: initialStatus,
      statusHistory: [{ status: initialStatus, timestamp: new Date().toISOString() }]
    };
  }

  transition(toStatus: OrderStatus): void {
    const allowedTransitions = this.getAllowedTransitions(this.state.status);
    
    if (!allowedTransitions.includes(toStatus)) {
      throw new Error(`不允许从${this.state.status}转换到${toStatus}`);
    }
    
    this.state.status = toStatus;
    this.state.statusHistory.push({
      status: toStatus,
      timestamp: new Date().toISOString()
    });
  }

  private getAllowedTransitions(currentStatus: OrderStatus): OrderStatus[] {
    const transitions: Record<OrderStatus, OrderStatus[]> = {
      pending: ['paid', 'cancelled'],
      paid: ['shipped', 'cancelled'],
      shipped: ['delivered'],
      delivered: [],
      cancelled: []
    };
    
    return transitions[currentStatus] || [];
  }

  getStatus(): OrderStatus {
    return this.state.status;
  }

  getStatusHistory(): { status: OrderStatus; timestamp: string }[] {
    return this.state.statusHistory;
  }
}

interface OrderStrategy {
  canPay(): boolean;
  canShip(): boolean;
  canCancel(): boolean;
}

class PendingOrderStrategy implements OrderStrategy {
  canPay(): boolean { return true; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return true; }
}

class PaidOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return true; }
  canCancel(): boolean { return true; }
}

class ShippedOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return false; }
}

class DeliveredOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return false; }
}

class CancelledOrderStrategy implements OrderStrategy {
  canPay(): boolean { return false; }
  canShip(): boolean { return false; }
  canCancel(): boolean { return false; }
}

class OrderStrategyFactory {
  static createStrategy(status: OrderStatus): OrderStrategy {
    switch (status) {
      case 'pending':
        return new PendingOrderStrategy();
      case 'paid':
        return new PaidOrderStrategy();
      case 'shipped':
        return new ShippedOrderStrategy();
      case 'delivered':
        return new DeliveredOrderStrategy();
      case 'cancelled':
        return new CancelledOrderStrategy();
      default:
        throw new Error(`未知状态: ${status}`);
    }
  }
}

class Order {
  private stateMachine: OrderStateMachine;
  private strategyFactory: OrderStrategyFactory;

  constructor() {
    this.stateMachine = new OrderStateMachine('pending');
    this.strategyFactory = new OrderStrategyFactory();
  }

  pay(): void {
    const strategy = this.strategyFactory.createStrategy(this.stateMachine.getStatus());
    
    if (!strategy.canPay()) {
      throw new Error('当前状态不允许支付');
    }
    
    this.stateMachine.transition('paid');
  }

  ship(): void {
    const strategy = this.strategyFactory.createStrategy(this.stateMachine.getStatus());
    
    if (!strategy.canShip()) {
      throw new Error('当前状态不允许发货');
    }
    
    this.stateMachine.transition('shipped');
  }

  cancel(): void {
    const strategy = this.strategyFactory.createStrategy(this.stateMachine.getStatus());
    
    if (!strategy.canCancel()) {
      throw new Error('当前状态不允许取消');
    }
    
    this.stateMachine.transition('cancelled');
  }

  getStatus(): OrderStatus {
    return this.stateMachine.getStatus();
  }

  getStatusHistory(): { status: OrderStatus; timestamp: string }[] {
    return this.stateMachine.getStatusHistory();
  }
}

  // 1. 验证邮箱格式
  if (!isValidEmail(email)) {
    throw new ValidationError('无效的邮箱格式');
  }

  // 2. 检查邮箱是否已存在

    throw new ConflictError('邮箱已被注册');
  }

  const salt = await bcrypt.genSalt(10);

  return user;

async function login(email: string, password: string) {
  // 1. 查找用户
  const user = await User.findOne({ email });
  if (!user) {
    throw new AuthError('邮箱或密码错误');
  }

  // 2. 验证密码
  const isValid = await bcrypt.compare(password, user.password);
  if (!isValid) {
    throw new AuthError('邮箱或密码错误');
  }

  // 3. 检查邮箱是否验证
  if (!user.emailVerified) {
    throw new AuthError('请先验证邮箱');
  }

  // 4. 生成JWT
  const accessToken = jwt.sign(
    { userId: user.id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: '1h' }
  );

  const refreshToken = jwt.sign(
    { userId: user.id },
    process.env.JWT_REFRESH_SECRET,
    { expiresIn: '7d' }
  );

  return { accessToken, refreshToken };
}

// 通知服务
class NotificationService {
  // 发送通知
  async send(notification: {
    userId: string;
    type: 'system' | 'message' | 'push' | 'email';
    title: string;
    content: string;
    data?: Record<string, any>;
  }) {
    // 1. 创建通知记录
    const record = await Notification.create({
      userId: notification.userId,
      type: notification.type,
      title: notification.title,
      content: notification.content,
      data: notification.data,
      read: false,
    });

    // 2. 根据类型发送
    switch (notification.type) {
      case 'system':
        await this.sendSystemNotification(record);
        break;
      case 'message':
        await this.sendMessage(record);
        break;
      case 'push':
        await this.sendPushNotification(record);
        break;
      case 'email':
        await this.sendEmail(record);
        break;
    }

    return record;
  }

  // 批量发送
  async sendBatch(notifications: NotificationInput[]) {
    // 使用消息队列批量处理
    await this.queue.add('notification:sendBatch', notifications, {
      attempts: 3,
      backoff: {
        type: 'exponential',
        delay: 1000,
      },
    });
  }
}
```
