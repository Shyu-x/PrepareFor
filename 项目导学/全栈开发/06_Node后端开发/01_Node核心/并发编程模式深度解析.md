# 并发编程模式深度解析

## 一、并发编程基础概念

### 1.1 为什么需要并发编程

在现代Web开发中，并发处理能力直接影响系统的性能和吞吐量。Node.js的单线程事件循环模型天然适合I/O密集型场景，但CPU密集型任务会阻塞整个事件循环。因此，掌握各种并发编程模式对于构建高效、稳定的Node.js应用至关重要。

```
同步阻塞模型:
┌─────────────────────────────────────────────────────────────┐
│  请求1 ──→ [处理中...............................] ──→ 响应1│
│  请求2 ──→ [等待请求1完成] ────────────────────────────→ 响应2│
│  请求3 ──→ [等待请求2完成] ────────────────────────────→ 响应3│
└─────────────────────────────────────────────────────────────┘

并发处理模型:
┌─────────────────────────────────────────────────────────────┐
│  请求1 ──→ [处理中...] ──→ 响应1                          │
│  请求2 ──→ [处理中...] ──→ 响应2  (并行处理)               │
│  请求3 ──→ [处理中...] ──→ 响应3                          │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 并发与并行的区别

- **并发（Concurrency）**：多个任务交替执行，在时间片内共享CPU资源
- **并行（Parallelism）**：多个任务真正同时执行，需要多核CPU支持

Node.js通过事件循环实现并发，但在多核环境下可以通过子进程实现并行处理。

## 二、生产者-消费者模式

### 2.1 模式概述

生产者-消费者模式是最经典的并发模式之一，用于解耦数据产生和数据处理的速率差异。在Node.js中，常用于任务队列、消息处理等场景。

```
┌──────────────┐     ┌──────────────┐     ┌──────────────┐
│   生产者     │ ──→ │   任务队列    │ ──→ │   消费者     │
│ (Producer)   │     │   (Queue)    │     │ (Consumer)   │
└──────────────┘     └──────────────┘     └──────────────┘
     产生数据             缓冲           处理数据
```

### 2.2 基础实现

```javascript
// task-queue.js - 生产者-消费者模式基础实现

/**
 * 任务队列类
 * 核心功能：存储待处理任务，按顺序分发任务给消费者
 */
class TaskQueue {
  constructor(options = {}) {
    this.maxSize = options.maxSize || 1000;    // 队列最大容量
    this.queue = [];                             // 任务队列
    this.processing = 0;                        // 正在处理的任务数
    this.concurrency = options.concurrency || 5; // 并发处理数
    this.handlers = [];                          // 消费者处理器
    this.isPaused = false;                      // 暂停状态
    this.stats = {
      total: 0,         // 总任务数
      completed: 0,     // 已完成任务数
      failed: 0,        // 失败任务数
      processed: 0,     // 已处理任务数（成功+失败）
    };
  }

  // 注册任务处理器
  registerHandler(handler) {
    if (typeof handler !== 'function') {
      throw new Error('处理器必须是函数');
    }
    this.handlers.push(handler);
  }

  // 生产者：添加任务到队列
  async produce(task) {
    return new Promise((resolve, reject) => {
      // 检查队列是否已满
      if (this.queue.length >= this.maxSize) {
        // 队列满时的策略：等待或拒绝
        if (options.rejectOnFull) {
          reject(new Error('队列已满'));
          return;
        }
        // 等待队列有空间
        setImmediate(() => this.produce(task).then(resolve).catch(reject));
        return;
      }

      // 创建任务对象
      const taskItem = {
        id: `task-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
        data: task,
        timestamp: Date.now(),
        status: 'pending',
        resolve,
        reject,
      };

      this.queue.push(taskItem);
      this.stats.total++;

      // 尝试处理任务
      this.processNext();
    });
  }

  // 消费者：处理下一个任务
  async processNext() {
    // 暂停时或达到并发上限时停止
    if (this.isPaused || this.processing >= this.concurrency) {
      return;
    }

    // 从队列取出任务
    const task = this.queue.shift();
    if (!task) {
      return;
    }

    this.processing++;
    task.status = 'processing';

    try {
      // 执行所有注册的处理器
      let result;
      for (const handler of this.handlers) {
        result = await handler(task.data);
      }

      this.stats.completed++;
      task.status = 'completed';
      task.resolve(result);

    } catch (error) {
      this.stats.failed++;
      task.status = 'failed';
      task.reject(error);
    } finally {
      this.processing--;

      // 继续处理下一个任务
      setImmediate(() => this.processNext());
    }
  }

  // 批量添加任务
  async produceBatch(tasks) {
    return Promise.all(tasks.map(task => this.produce(task)));
  }

  // 暂停队列
  pause() {
    this.isPaused = true;
    console.log('队列已暂停');
  }

  // 恢复队列
  resume() {
    this.isPaused = false;
    console.log('队列已恢复');
    // 恢复后尝试处理积压任务
    this.processNext();
  }

  // 获取队列状态
  getStats() {
    return {
      ...this.stats,
      queueLength: this.queue.length,
      processing: this.processing,
      isPaused: this.isPaused,
      available: this.concurrency - this.processing,
    };
  }

  // 清空队列
  clear() {
    const cleared = this.queue.length;
    this.queue.forEach(task => {
      task.reject(new Error('队列已清空'));
    });
    this.queue = [];
    console.log(`已清空队列，${cleared}个任务被取消`);
    return cleared;
  }
}

// 使用示例
async function demo() {
  // 创建任务队列，并发数设为3
  const queue = new TaskQueue({
    concurrency: 3,
    maxSize: 100,
  });

  // 注册任务处理器
  queue.registerHandler(async (taskData) => {
    console.log(`处理任务: ${taskData.name}`);
    // 模拟任务处理
    await new Promise(resolve => setTimeout(resolve, 1000));
    return { success: true, data: taskData };
  });

  // 生产者：模拟持续产生任务
  const producer = setInterval(() => {
    const task = { name: `任务-${Date.now()}`, data: Math.random() };
    queue.produce(task)
      .then(result => console.log('任务完成:', result))
      .catch(err => console.error('任务失败:', err));
  }, 500);

  // 监控队列状态
  const monitor = setInterval(() => {
    console.log('队列状态:', queue.getStats());
  }, 2000);

  // 10秒后停止
  setTimeout(() => {
    clearInterval(producer);
    clearInterval(monitor);
    console.log('最终状态:', queue.getStats());
  }, 10000);
}

module.exports = TaskQueue;
```

### 2.3 带优先级的任务队列

```javascript
// priority-queue.js - 优先级队列实现

/**
 * 优先级任务队列
 * 支持任务优先级排序，高优先级任务优先处理
 */
class PriorityTaskQueue {
  constructor() {
    // 使用最小堆存储任务（便于快速获取最高优先级）
    this.heap = [];
    this.comparator = (a, b) => a.priority - b.priority; // 数值越小优先级越高
  }

  // 获取父节点索引
  parentIndex(i) {
    return Math.floor((i - 1) / 2);
  }

  // 获取左子节点索引
  leftChildIndex(i) {
    return 2 * i + 1;
  }

  // 获取右子节点索引
  rightChildIndex(i) {
    return 2 * i + 2;
  }

  // 交换两个元素
  swap(i, j) {
    [this.heap[i], this.heap[j]] = [this.heap[j], this.heap[i]];
  }

  // 上浮操作：用于添加元素后恢复堆性质
  siftUp(index) {
    while (index > 0) {
      const parent = this.parentIndex(index);
      if (this.comparator(this.heap[index], this.heap[parent]) < 0) {
        this.swap(index, parent);
        index = parent;
      } else {
        break;
      }
    }
  }

  // 下沉操作：用于删除元素后恢复堆性质
  siftDown(index) {
    const length = this.heap.length;

    while (true) {
      const left = this.leftChildIndex(index);
      const right = this.rightChildIndex(index);
      let smallest = index;

      if (left < length && this.comparator(this.heap[left], this.heap[smallest]) < 0) {
        smallest = left;
      }

      if (right < length && this.comparator(this.heap[right], this.heap[smallest]) < 0) {
        smallest = right;
      }

      if (smallest !== index) {
        this.swap(index, smallest);
        index = smallest;
      } else {
        break;
      }
    }
  }

  // 添加任务
  enqueue(taskData, priority = 5) {
    const task = {
      id: `task-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
      data: taskData,
      priority,
      timestamp: Date.now(),
    };

    this.heap.push(task);
    this.siftUp(this.heap.length - 1);

    return task.id;
  }

  // 取出最高优先级任务
  dequeue() {
    if (this.heap.length === 0) {
      return null;
    }

    // 取出堆顶（优先级最高）
    const min = this.heap[0];

    // 将最后一个元素移到堆顶
    const last = this.heap.pop();

    if (this.heap.length > 0) {
      this.heap[0] = last;
      this.siftDown(0);
    }

    return min;
  }

  // 查看堆顶任务（不取出）
  peek() {
    return this.heap[0] || null;
  }

  // 获取队列长度
  size() {
    return this.heap.length;
  }

  // 判断是否为空
  isEmpty() {
    return this.heap.length === 0;
  }
}

// 使用示例
function priorityDemo() {
  const queue = new PriorityTaskQueue();

  // 添加不同优先级的任务
  queue.enqueue({ type: '邮件通知', content: '用户注册成功' }, 1); // 最高优先级
  queue.enqueue({ type: '数据统计', content: '生成日报' }, 10);      // 低优先级
  queue.enqueue({ type: '缓存清理', content: '清理过期缓存' }, 5);     // 中优先级
  queue.enqueue({ type: '紧急告警', content: '服务器异常' }, 1);     // 最高优先级

  console.log('队列大小:', queue.size());

  // 按优先级顺序取出任务
  console.log('按优先级处理任务:');
  while (!queue.isEmpty()) {
    const task = queue.dequeue();
    console.log(`优先级${task.priority}:`, task.data);
  }
}

module.exports = PriorityTaskQueue;
```

## 三、读写锁实现

### 3.1 读写锁原理

读写锁是一种特殊的锁策略，允许多个读操作同时进行，但写操作需要独占访问。这在读多写少的场景下能显著提升并发性能。

```
读锁状态:
┌─────────────────────────────────────────────────────────────┐
│  读锁生效时:                                                  │
│  ✓ 多个读操作可以同时进行                                      │
│  ✗ 写操作必须等待所有读操作完成                                │
└─────────────────────────────────────────────────────────────┘

写锁状态:
┌─────────────────────────────────────────────────────────────┐
│  写锁生效时:                                                  │
│  ✗ 所有读操作必须等待                                        │
│  ✗ 所有写操作必须等待                                        │
│  ✓ 只有一个写操作在进行                                      │
└─────────────────────────────────────────────────────────────┘
```

### 3.2 读写锁实现

```javascript
// read-write-lock.js - 读写锁实现

/**
 * 读写锁类
 * 特性：
 * - 多个读操作可以并发执行
 * - 写操作独占访问
 * - 写操作优先（防止写饥饿）
 */
class ReadWriteLock {
  constructor() {
    this.readers = 0;           // 当前读者数量
    this.writing = false;       // 是否有写操作进行
    this.readWaitQueue = [];    // 等待的读操作
    this.writeWaitQueue = [];   // 等待的写操作
    this.preferWrite = true;    // 写操作优先策略
  }

  // 获取读锁
  async readLock() {
    return new Promise((resolve, reject) => {
      const acquire = () => {
        // 如果没有写操作在进行，且写队列没有等待的写操作，则可以获取读锁
        if (!this.writing &&
            (this.writeWaitQueue.length === 0 || !this.preferWrite)) {
          this.readers++;
          resolve();
        } else {
          // 加入等待队列
          this.readWaitQueue.push(acquire);
        }
      };

      acquire();
    });
  }

  // 释放读锁
  readUnlock() {
    if (this.readers <= 0) {
      throw new Error('没有持有读锁');
    }

    this.readers--;

    // 如果所有读者都已离开，唤醒一个写操作
    if (this.readers === 0 && this.writeWaitQueue.length > 0) {
      const nextWrite = this.writeWaitQueue.shift();
      this.writing = true;
      nextWrite();
    }
  }

  // 获取写锁
  async writeLock() {
    return new Promise((resolve, reject) => {
      const acquire = () => {
        // 如果没有读者和写操作，可以获取写锁
        if (this.readers === 0 && !this.writing) {
          this.writing = true;
          resolve();
        } else {
          // 加入等待队列
          this.writeWaitQueue.push(acquire);
        }
      };

      acquire();
    });
  }

  // 释放写锁
  writeUnlock() {
    if (!this.writing) {
      throw new Error('没有持有写锁');
    }

    this.writing = false;

    // 优先唤醒写操作
    if (this.writeWaitQueue.length > 0) {
      const nextWrite = this.writeWaitQueue.shift();
      this.writing = true;
      nextWrite();
    } else if (this.readWaitQueue.length > 0) {
      // 唤醒所有等待的读操作
      const waitingReaders = [...this.readWaitQueue];
      this.readWaitQueue = [];
      waitingReaders.forEach(() => {
        this.readers++;
      });
      waitingReaders.forEach(resolve => resolve());
    }
  }

  // 获取状态信息
  getStatus() {
    return {
      readers: this.readers,
      writing: this.writing,
      readWaitCount: this.readWaitQueue.length,
      writeWaitCount: this.writeWaitQueue.length,
    };
  }
}

// 使用示例
async function rwLockDemo() {
  const lock = new ReadWriteLock();
  let sharedResource = 0;

  // 模拟读操作
  async function read(key) {
    await lock.readLock();
    console.log(`[读 ${key}] 开始读取，值为: ${sharedResource}`);
    await new Promise(r => setTimeout(r, 100)); // 模拟读取耗时
    console.log(`[读 ${key}] 读取完成`);
    lock.readUnlock();
    return sharedResource;
  }

  // 模拟写操作
  async function write(key, value) {
    await lock.writeLock();
    console.log(`[写 ${key}] 开始写入，值为: ${value}`);
    await new Promise(r => setTimeout(r, 100)); // 模拟写入耗时
    sharedResource = value;
    console.log(`[写 ${key}] 写入完成`);
    lock.writeUnlock();
  }

  // 并发测试
  console.log('=== 并发测试开始 ===');

  // 多个读操作同时进行
  await Promise.all([
    read('A'),
    read('B'),
    read('C'),
  ]);

  // 写操作会等待所有读操作完成
  await write('X', 100);

  // 再次并发读
  await Promise.all([
    read('D'),
    read('E'),
  ]);

  console.log('=== 并发测试结束 ===');
  console.log('最终值:', sharedResource);
}

module.exports = ReadWriteLock;
```

## 四、信号量模式

### 4.1 信号量概念

信号量是一种更通用的锁机制，它维护一个计数器，通过原子操作控制对资源的访问。计数器表示可用资源数量，当计数器为0时，请求资源的操作会阻塞。

```
信号量工作原理:
┌─────────────────────────────────────────────────────────────┐
│  初始化: 信号量 = N                                          │
│                                                             │
│  P操作 (wait/procure):                                     │
│    如果 信号量 > 0:                                          │
│        信号量--                                             │
│        继续执行                                             │
│    否则:                                                    │
│        阻塞等待                                             │
│                                                             │
│  V操作 (signal/post):                                      │
│        信号量++                                             │
│        唤醒等待的线程                                        │
└─────────────────────────────────────────────────────────────┘
```

### 4.2 信号量实现

```javascript
// semaphore.js - 信号量实现

/**
 * 信号量类
 * 用于控制对共享资源的并发访问数量
 */
class Semaphore {
  constructor(initialCount = 1) {
    if (initialCount < 0) {
      throw new Error('信号量初始值不能为负数');
    }
    this.count = initialCount;     // 可用资源数
    this.waitQueue = [];           // 等待队列
  }

  // P操作：获取信号量
  async acquire() {
    if (this.count > 0) {
      // 有可用资源，直接获取
      this.count--;
      return;
    }

    // 没有可用资源，加入等待队列
    return new Promise((resolve) => {
      this.waitQueue.push(resolve);
    });
  }

  // V操作：释放信号量
  release() {
    if (this.waitQueue.length > 0) {
      // 唤醒一个等待的请求
      const resolve = this.waitQueue.shift();
      resolve();
    } else {
      // 没有等待的请求，资源数+1
      this.count++;
    }
  }

  // 批量获取
  async acquireMultiple(n) {
    const promises = [];
    for (let i = 0; i < n; i++) {
      promises.push(this.acquire());
    }
    await Promise.all(promises);
  }

  // 批量释放
  releaseMultiple(n) {
    for (let i = 0; i < n; i++) {
      this.release();
    }
  }

  // 尝试获取（不阻塞）
  tryAcquire() {
    if (this.count > 0) {
      this.count--;
      return true;
    }
    return false;
  }

  // 获取当前计数
  getCount() {
    return this.count;
  }

  // 获取等待数量
  getWaitCount() {
    return this.waitQueue.length;
  }
}

// 使用示例：数据库连接池
async function databasePoolDemo() {
  // 模拟数据库连接池，最多同时5个连接
  const MAX_CONNECTIONS = 5;
  const pool = new Semaphore(MAX_CONNECTIONS);

  // 模拟数据库查询
  async function query(sql) {
    console.log(`[查询开始] ${sql}`);

    // 获取连接
    await pool.acquire();

    try {
      // 模拟查询执行
      await new Promise(r => setTimeout(r, Math.random() * 1000));
      console.log(`[查询完成] ${sql}`);
      return { success: true, sql };
    } finally {
      // 释放连接
      pool.release();
    }
  }

  // 模拟10个并发查询
  const queries = [
    'SELECT * FROM users',
    'SELECT * FROM orders',
    'SELECT * FROM products',
    'INSERT INTO logs ...',
    'UPDATE users SET ...',
    'SELECT * FROM users WHERE id=1',
    'SELECT COUNT(*) FROM orders',
    'DELETE FROM logs WHERE ...',
    'SELECT * FROM products ORDER BY id',
    'SELECT SUM(amount) FROM orders',
  ];

  console.log(`发送 ${queries.length} 个并发查询，连接池大小: ${MAX_CONNECTIONS}`);
  console.log('当前可用连接:', pool.getCount());

  await Promise.all(queries.map(sql => query(sql)));

  console.log('所有查询完成');
  console.log('当前可用连接:', pool.getCount());
}

module.exports = Semaphore;
```

## 五、Promise池

### 5.1 Promise池原理

Promise池用于控制同时执行的异步任务数量，避免一次性创建过多Promise导致资源耗尽。这在处理批量API请求、文件操作等场景中非常有用。

```
Promise池工作流程:
┌─────────────────────────────────────────────────────────────┐
│  任务队列: [T1] [T2] [T3] [T4] [T5] [T6] [T7] [T8]         │
│                                                             │
│  池大小: 3                                                  │
│  ┌──────────────────────────────────────────────────────┐  │
│  │  活跃任务: [T1] [T2] [T3]                             │  │
│  │  等待队列: [T4] [T5] [T6] [T7] [T8]                   │  │
│  └──────────────────────────────────────────────────────┘  │
│                                                             │
│  T1完成 → T4开始                                            │
│  T2完成 → T5开始                                            │
│  T3完成 → T6开始                                            │
└─────────────────────────────────────────────────────────────┘
```

### 5.2 Promise池实现

```javascript
// promise-pool.js - Promise池实现

/**
 * Promise池类
 * 控制同时执行的异步任务数量
 */
class PromisePool {
  constructor(concurrency = 5) {
    this.concurrency = concurrency;  // 最大并发数
    this.activeCount = 0;            // 当前活跃任务数
    this.taskQueue = [];             // 任务队列
    this.results = [];              // 执行结果
    this.errors = [];                // 执行错误
    this.isRunning = false;
    this.isStopped = false;
  }

  // 添加任务
  add(taskFn) {
    return new Promise((resolve, reject) => {
      this.taskQueue.push({ taskFn, resolve, reject });

      if (!this.isStopped) {
        this.processNext();
      }
    });
  }

  // 添加多个任务
  addMany(taskFns) {
    return Promise.all(taskFns.map(fn => this.add(fn)));
  }

  // 处理下一个任务
  async processNext() {
    // 达到并发上限或队列为空时停止
    if (this.activeCount >= this.concurrency || this.taskQueue.length === 0) {
      return;
    }

    const { taskFn, resolve, reject } = this.taskQueue.shift();
    this.activeCount++;

    // 异步执行任务
    this.runTask(taskFn, resolve, reject)
      .finally(() => {
        this.activeCount--;
        // 继续处理下一个任务
        if (this.taskQueue.length > 0 && !this.isStopped) {
          setImmediate(() => this.processNext());
        }
      });
  }

  // 运行单个任务
  async runTask(taskFn, resolve, reject) {
    try {
      const result = await taskFn();
      this.results.push(result);
      resolve(result);
    } catch (error) {
      this.errors.push(error);
      reject(error);
    }
  }

  // 开始执行
  async start() {
    this.isRunning = true;
    this.isStopped = false;

    // 填充并发槽
    const promises = [];
    const slots = Math.min(this.concurrency, this.taskQueue.length);

    for (let i = 0; i < slots; i++) {
      promises.push(this.processNext());
    }

    await Promise.all(promises);
    this.isRunning = false;
  }

  // 停止接受新任务
  stop() {
    this.isStopped = true;
    this.isRunning = false;

    // 取消队列中的任务
    this.taskQueue.forEach(({ reject }) => {
      reject(new Error('Promise池已停止'));
    });
    this.taskQueue = [];
  }

  // 获取状态
  getStatus() {
    return {
      active: this.activeCount,
      queued: this.taskQueue.length,
      results: this.results.length,
      errors: this.errors.length,
      isRunning: this.isRunning,
      isStopped: this.isStopped,
    };
  }

  // 等待所有任务完成
  async waitForCompletion() {
    while (this.activeCount > 0 || this.taskQueue.length > 0) {
      await new Promise(resolve => setTimeout(resolve, 10));
    }
  }

  // 获取所有结果
  getResults() {
    return {
      results: this.results,
      errors: this.errors,
    };
  }
}
```

## 六、并发限流器

### 6.1 限流器概念

限流器用于控制请求的速率，防止系统过载。在API调用、数据库访问等场景中广泛使用。

```
限流器策略对比:
┌─────────────────────────────────────────────────────────────┐
│  固定窗口限流:                                               │
│  ─────────────────────────────────                         │
│  在固定时间窗口内限制请求数                                   │
│  问题: 窗口边界可能出现突发流量                               │
│                                                             │
│  滑动窗口限流:                                               │
│  ─────────────────────────────────                         │
│  基于滑动时间窗口精确控制请求速率                             │
│  更平滑，但实现更复杂                                        │
│                                                             │
│  令牌桶算法:                                                 │
│  ─────────────────────────────────                         │
│  以固定速率添加令牌，请求消耗令牌                             │
│  支持突发流量，是最常用的算法                                 │
│                                                             │
│  漏桶算法:                                                   │
│  ─────────────────────────────────                         │
│  请求以任意速率进入，以固定速率处理                           │
│  平滑输出，但不支持突发                                       │
└─────────────────────────────────────────────────────────────┘
```

### 6.2 令牌桶限流器实现

```javascript
// rate-limiter.js - 令牌桶限流器实现

/**
 * 令牌桶限流器
 * 特性：
 * - 以固定速率生成令牌
 * - 桶有最大容量
 * - 请求需要获取令牌才能执行
 * - 支持突发流量（最多桶满时的令牌数）
 */
class TokenBucketRateLimiter {
  constructor(options = {}) {
    this.capacity = options.capacity || 100;        // 令牌桶容量
    this.refillRate = options.refillRate || 10;     // 每秒补充的令牌数
    this.tokens = this.capacity;                     // 当前令牌数
    this.lastRefillTime = Date.now();               // 上次补充时间

    // 可选：限制突发大小
    this.maxBurst = options.maxBurst || this.capacity;
  }

  // 补充令牌
  refill() {
    const now = Date.now();
    const elapsed = (now - this.lastRefillTime) / 1000; // 秒
    const tokensToAdd = elapsed * this.refillRate;

    this.tokens = Math.min(this.capacity, this.tokens + tokensToAdd);
    this.lastRefillTime = now;

    return this.tokens;
  }

  // 尝试获取令牌
  tryConsume(tokensNeeded = 1) {
    this.refill();

    if (this.tokens >= tokensNeeded) {
      this.tokens -= tokensNeeded;
      return true;
    }

    return false;
  }

  // 异步获取令牌（等待可用令牌）
  async acquire(tokensNeeded = 1) {
    return new Promise((resolve, reject) => {
      const attempt = () => {
        if (this.tryConsume(tokensNeeded)) {
          resolve();
        } else {
          // 计算需要等待的时间
          const tokensNeededForOne = tokensNeeded - this.tokens;
          const waitTime = (tokensNeededForOne / this.refillRate) * 1000;

          // 等待后重试
          setTimeout(attempt, Math.min(waitTime, 100));
        }
      };

      attempt();
    });
  }

  // 获取当前状态
  getStatus() {
    this.refill();
    return {
      tokens: this.tokens.toFixed(2),
      capacity: this.capacity,
      refillRate: this.refillRate,
      canConsume: this.tokens >= 1,
    };
  }
}

/**
 * 滑动窗口限流器
 * 更精确的限流实现
 */
class SlidingWindowRateLimiter {
  constructor(options = {}) {
    this.maxRequests = options.maxRequests || 100;  // 时间窗口内最大请求数
    this.windowSize = options.windowSize || 60000;   // 时间窗口大小（毫秒）
    this.requests = [];                             // 请求时间戳记录
  }

  // 清理过期请求记录
  cleanOldRequests() {
    const now = Date.now();
    const windowStart = now - this.windowSize;

    // 移除窗口外的记录
    this.requests = this.requests.filter(time => time > windowStart);
  }

  // 尝试执行请求
  tryAcquire() {
    this.cleanOldRequests();

    if (this.requests.length < this.maxRequests) {
      this.requests.push(Date.now());
      return true;
    }

    return false;
  }

  // 异步获取许可
  async acquire() {
    return new Promise((resolve, reject) => {
      const attempt = () => {
        if (this.tryAcquire()) {
          resolve();
        } else {
          // 等待一个时间窗口后重试
          setTimeout(attempt, this.windowSize / this.maxRequests);
        }
      };

      attempt();
    });
  }

  // 获取状态
  getStatus() {
    this.cleanOldRequests();
    return {
      currentRequests: this.requests.length,
      maxRequests: this.maxRequests,
      windowSize: this.windowSize,
      available: this.maxRequests - this.requests.length,
    };
  }
}

// 使用示例
async function rateLimiterDemo() {
  // 令牌桶示例
  console.log('=== 令牌桶限流器 ===');
  const tokenBucket = new TokenBucketRateLimiter({
    capacity: 5,       // 最多5个令牌
    refillRate: 2,     // 每秒补充2个令牌
  });

  async function simulateTokenBucket() {
    for (let i = 0; i < 10; i++) {
      const allowed = tokenBucket.tryConsume();
      console.log(`请求 ${i + 1}: ${allowed ? '允许' : '拒绝'}`, tokenBucket.getStatus());

      if (!allowed) {
        await tokenBucket.acquire();
        console.log(`请求 ${i + 1}: 等待后允许`);
      }

      await new Promise(r => setTimeout(r, 200));
    }
  }

  await simulateTokenBucket();

  // 滑动窗口示例
  console.log('\n=== 滑动窗口限流器 ===');
  const slidingWindow = new SlidingWindowRateLimiter({
    maxRequests: 5,
    windowSize: 5000,
  });

  for (let i = 0; i < 10; i++) {
    const allowed = slidingWindow.tryAcquire();
    console.log(`请求 ${i + 1}: ${allowed ? '允许' : '拒绝'}`, slidingWindow.getStatus());
    await new Promise(r => setTimeout(r, 300));
  }
}
```

## 七、缓存击穿、穿透、雪崩

### 7.1 三大缓存问题概述

```
┌─────────────────────────────────────────────────────────────┐
│  缓存击穿 (Cache Breakdown)                                  │
│  ─────────────────────────────────────                      │
│  热点key过期瞬间，大量请求直接打到数据库                       │
│  原因: 某个热点数据过期，但大量并发请求恰好访问这个key         │
│                                                             │
│  缓存穿透 (Cache Penetration)                                │
│  ─────────────────────────────────────                      │
│  请求的数据既不在缓存也不在数据库中                           │
│  原因: 恶意请求或业务逻辑问题，查询不存在的数据               │
│                                                             │
│  缓存雪崩 (Cache Avalanche)                                  │
│  ─────────────────────────────────────                      │
│  大量缓存同时过期，导致大量请求打到数据库                       │
│  原因: 缓存服务宕机或大量key同时过期                          │
└─────────────────────────────────────────────────────────────┘
```

### 7.2 解决方案实现

```javascript
// cache-problems.js - 缓存三大问题解决方案

/**
 * 缓存击穿解决方案：互斥锁 + 永不过期
 */
class CacheBreakdownSolution {
  constructor(cache, options = {}) {
    this.cache = cache;
    this.mutex = new Map();        // 互斥锁
    this.lockTimeout = options.lockTimeout || 5000;  // 锁超时时间
  }

  // 获取互斥锁
  async acquireLock(key) {
    const lockKey = `lock:${key}`;

    // 尝试获取锁
    const locked = this.cache.get(lockKey);
    if (locked) {
      return false;
    }

    // 设置锁，带过期时间防止死锁
    this.cache.set(lockKey, '1', this.lockTimeout);
    return true;
  }

  // 释放锁
  releaseLock(key) {
    const lockKey = `lock:${key}`;
    this.cache.delete(lockKey);
  }

  // 带击穿保护的获取
  async get(key, fetchFn) {
    // 1. 先从缓存获取
    let value = await this.cache.get(key);
    if (value !== undefined) {
      return value;
    }

    // 2. 尝试获取互斥锁
    const lockAcquired = await this.acquireLock(key);

    if (lockAcquired) {
      try {
        // 3. 获取锁成功，再次检查缓存（可能其他线程已填充）
        value = await this.cache.get(key);
        if (value !== undefined) {
          return value;
        }

        // 4. 从数据源获取
        value = await fetchFn();

        // 5. 写入缓存
        await this.cache.set(key, value);

        return value;
      } finally {
        // 6. 释放锁
        this.releaseLock(key);
      }
    } else {
      // 7. 未获取到锁，等待后重试
      await new Promise(r => setTimeout(r, 100));
      return this.get(key, fetchFn);
    }
  }
}

/**
 * 缓存穿透解决方案：布隆过滤器 + 空值缓存
 */
class CachePenetrationSolution {
  constructor(options = {}) {
    this.cache = new Map();        // 缓存
    this.bloomFilter = new Set(); // 简化版布隆过滤器
    this.nullCacheTTL = options.nullCacheTTL || 60000;  // 空值缓存时间
  }

  // 添加到布隆过滤器
  addToFilter(key) {
    this.bloomFilter.add(key);
  }

  // 检查是否可能在过滤器中
  mayExist(key) {
    return this.bloomFilter.has(key);
  }

  // 设置缓存
  async set(key, value) {
    this.cache.set(key, {
      value,
      timestamp: Date.now(),
    });
    this.addToFilter(key);
  }

  // 获取缓存
  async get(key) {
    // 1. 先检查布隆过滤器
    if (!this.mayExist(key)) {
      // 一定不存在，直接返回
      return null;
    }

    // 2. 检查缓存
    const cached = this.cache.get(key);
    if (cached) {
      // 3. 检查是否是空值缓存
      if (cached.value === null) {
        // 空值缓存未过期
        return null;
      }
      return cached.value;
    }

    return undefined; // 缓存不存在
  }

  // 设置空值缓存
  setNullValue(key) {
    this.cache.set(key, {
      value: null,
      timestamp: Date.now(),
    });
    this.addToFilter(key);
  }

  // 获取（支持空值缓存）
  async getWithNullProtection(key, fetchFn) {
    // 1. 先检查布隆过滤器
    if (!this.mayExist(key)) {
      // 查询数据库
      const value = await fetchFn();

      if (value === null || value === undefined) {
        // 数据不存在，缓存空值
        this.setNullValue(key);
      } else {
        // 数据存在，添加到过滤器和缓存
        this.set(key, value);
      }

      return value;
    }

    // 2. 检查缓存
    const cached = await this.get(key);
    if (cached !== undefined) {
      return cached;
    }

    // 3. 缓存不存在，查询数据库
    const value = await fetchFn();

    if (value === null || value === undefined) {
      this.setNullValue(key);
    } else {
      await this.set(key, value);
    }

    return value;
  }
}

/**
 * 缓存雪崩解决方案：随机过期时间 + 缓存预热 + 多级缓存
 */
class CacheAvalancheSolution {
  constructor(options = {}) {
    this.cache = new Map();
    this.baseTTL = options.baseTTL || 3600000;        // 基础过期时间（1小时）
    this.jitterMax = options.jitterMax || 300000;     // 最大随机抖动（5分钟）
  }

  // 计算带随机抖动的过期时间
  calculateTTL() {
    const jitter = Math.random() * this.jitterMax;
    return this.baseTTL + jitter;
  }

  // 设置缓存（带随机过期时间）
  set(key, value, customTTL = null) {
    const ttl = customTTL || this.calculateTTL();
    const expireAt = Date.now() + ttl;

    this.cache.set(key, {
      value,
      expireAt,
    });
  }

  // 获取缓存
  async get(key) {
    const item = this.cache.get(key);

    if (!item) {
      return undefined;
    }

    // 检查是否过期
    if (Date.now() > item.expireAt) {
      this.cache.delete(key);
      return undefined;
    }

    return item.value;
  }

  // 缓存预热
  async preheat(keys, fetchFn) {
    console.log(`开始缓存预热，共 ${keys.length} 个key...`);

    for (const key of keys) {
      try {
        const value = await fetchFn(key);
        this.set(key, value);
        console.log(`预热完成: ${key}`);
      } catch (error) {
        console.error(`预热失败: ${key}`, error);
      }
    }

    console.log('缓存预热完成');
  }

  // 批量设置（用于数据更新时的缓存重建）
  async batchSet(items, fetchFn) {
    const keys = items.map(item => item.key);

    // 清空相关缓存
    for (const key of keys) {
      this.cache.delete(key);
    }

    // 重新预热
    await this.preheat(keys, fetchFn);
  }
}
```

## 总结

本文深入解析了Node.js中六种核心并发编程模式：

| 模式 | 适用场景 | 核心价值 |
|------|---------|---------|
| 生产者-消费者 | 任务队列、消息处理 | 解耦生产与处理速率 |
| 读写锁 | 读多写少场景 | 提高读并发性能 |
| 信号量 | 资源池控制 | 限制并发访问数量 |
| Promise池 | 批量异步任务 | 控制并发总数 |
| 限流器 | API调用限制 | 防止系统过载 |
| 缓存问题解决 | 缓存系统 | 保护数据库稳定 |

掌握这些模式，能够有效应对各种高并发场景，构建稳定、高效的Node.js应用。
