## 1. fs 文件系统操作

fs 模块是 Node.js 最常用的模块之一，提供文件系统操作功能，包括文件读写、目录操作、文件属性修改等。

### 1.1 文件读取

#### 1.1.1 同步读取

```javascript
const fs = require('fs');

// 同步读取文件（阻塞）
const content = fs.readFileSync('./file.txt', 'utf8');
console.log(content);

// 同步读取二进制
const buffer = fs.readFileSync('./file.txt');
console.log(buffer);
```

#### 1.1.2 异步读取

```javascript
const fs = require('fs');

// 异步读取（回调方式）
fs.readFile('./file.txt', 'utf8', (err, data) => {
  if (err) {
    console.error('读取失败:', err);
    return;
  }
  console.log(data);
});

// 异步读取二进制
fs.readFile('./file.txt', (err, buffer) => {
  if (err) throw err;
  console.log(buffer);
});
```

#### 1.1.3 Promise 方式

```javascript
const fs = require('fs').promises;

// 异步读取（Promise 方式）
async function readFile() {
  try {
    const data = await fs.readFile('./file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error('读取失败:', err);
  }
}

readFile();
```

#### 1.1.4 流式读取

```javascript
const fs = require('fs');

// 流式读取大文件
const readStream = fs.createReadStream('./large-file.txt', {
  encoding: 'utf8',
  highWaterMark: 1024 * 1024  // 1MB
});

let totalData = '';

readStream.on('data', (chunk) => {
  totalData += chunk;
  console.log('接收到:', chunk.length, 'bytes');
});

readStream.on('end', () => {
  console.log('总大小:', totalData.length);
});

readStream.on('error', (err) => {
  console.error('错误:', err);
});
```

### 1.2 文件写入

#### 1.2.1 同步写入

```javascript
const fs = require('fs');

// 同步写入（覆盖）
fs.writeFileSync('./output.txt', 'Hello World', 'utf8');

// 同步追加
fs.appendFileSync('./output.txt', '\n新内容', 'utf8');
```

#### 1.2.2 异步写入

```javascript
const fs = require('fs').promises;

// 异步写入
async function writeFile() {
  await fs.writeFile('./output.txt', 'Hello World', 'utf8');
  console.log('写入完成');
}

writeFile();

// 异步追加
async function appendFile() {
  await fs.appendFile('./output.txt', '\n新内容', 'utf8');
  console.log('追加完成');
}
```

#### 1.2.3 流式写入

```javascript
const fs = require('fs');

// 创建写入流
const writeStream = fs.createWriteStream('./output.txt');

writeStream.write('第一行\n');
writeStream.write('第二行\n');
writeStream.write('第三行\n');

writeStream.end('结束\n');

writeStream.on('finish', () => {
  console.log('写入完成');
});

writeStream.on('error', (err) => {
  console.error('错误:', err);
});
```

#### 1.2.4 使用 pipe 复制文件

```javascript
const fs = require('fs');

const readStream = fs.createReadStream('./source.txt');
const writeStream = fs.createWriteStream('./destination.txt');

readStream.pipe(writeStream);

writeStream.on('finish', () => {
  console.log('复制完成');
});
```

### 1.6 完整文件操作示例

```javascript
const fs = require('fs').promises;
const path = require('path');

class FileManager {
  constructor(basePath) {
    this.basePath = basePath;
  }

  // 读取文件
  async readFile(filePath) {
    const fullPath = path.join(this.basePath, filePath);
    return await fs.readFile(fullPath, 'utf8');
  }

  // 写入文件
  async writeFile(filePath, content) {
    const fullPath = path.join(this.basePath, filePath);
    await fs.writeFile(fullPath, content, 'utf8');
  }

  // 追加写入
  async appendFile(filePath, content) {
    const fullPath = path.join(this.basePath, filePath);
    await fs.appendFile(fullPath, content, 'utf8');
  }

  // 删除文件
  async deleteFile(filePath) {
    const fullPath = path.join(this.basePath, filePath);
    await fs.unlink(fullPath);
  }

  // 列出目录
  async listFiles(dirPath = '') {
    const fullPath = path.join(this.basePath, dirPath);
    return await fs.readdir(fullPath);
  }

  // 创建目录
  async createDir(dirPath) {
    const fullPath = path.join(this.basePath, dirPath);
    await fs.mkdir(fullPath, { recursive: true });
  }

  // 删除目录
  async deleteDir(dirPath) {
    const fullPath = path.join(this.basePath, dirPath);
    await fs.rm(fullPath, { recursive: true, force: true });
  }

  // 检查是否存在
  async exists(filePath) {
    const fullPath = path.join(this.basePath, filePath);
    try {
      await fs.access(fullPath);
      return true;
    } catch {
      return false;
    }
  }

  // 获取文件信息
  async getInfo(filePath) {
    const fullPath = path.join(this.basePath, filePath);
    const stats = await fs.stat(fullPath);
    return {
      size: stats.size,
      created: stats.birthtime,
      modified: stats.mtime,
      isFile: stats.isFile(),
      isDirectory: stats.isDirectory()
    };
  }
}

// 使用示例
async function main() {
  const fileManager = new FileManager('./data');

  // 写入文件
  await fileManager.writeFile('test.txt', 'Hello World');

  // 读取文件
  const content = await fileManager.readFile('test.txt');
  console.log(content);

  // 列出文件
  const files = await fileManager.listFiles();
  console.log(files);
}

main();
```

## 2. path 路径处理

path 模块提供处理文件和目录路径的工具函数。

### 2.1 路径解析

```javascript
const path = require('path');

// 获取路径中的各个部分
const filePath = '/home/user/project/src/app.js';

console.log('目录:', path.dirname(filePath));  // /home/user/project/src
console.log('文件名:', path.basename(filePath));  // app.js
console.log('扩展名:', path.extname(filePath));  // .js
console.log('文件名(无扩展名):', path.basename(filePath, '.js'));  // app

// Windows 路径处理
const winPath = 'C:\\Users\\Admin\\Documents\\file.txt';
console.log('目录:', path.dirname(winPath));  // C:\Users\Admin\Documents
console.log('文件名:', path.basename(winPath));  // file.txt
```

### 2.2 路径拼接

```javascript
const path = require('path');

// 拼接路径
const fullPath = path.join('/home', 'user', 'project', 'src', 'app.js');
console.log(fullPath);  // /home/user/project/src/app.js

// 解析相对路径
const relativePath = path.join(__dirname, '..', 'config', 'app.json');
console.log(relativePath);

// 使用 path.resolve 获取绝对路径
const absolutePath = path.resolve('config', 'app.json');
console.log(absolutePath);  // /current/working/dir/config/app.json

// resolve 会将相对路径转为绝对路径
console.log(path.resolve('./app.js'));  // /current/working/dir/app.js
```

### 2.3 路径解析

```javascript
const path = require('path');

// 解析路径为对象
const parsed = path.parse('/home/user/project/src/app.js');

console.log(parsed);
// {
//   root: '/',
//   dir: '/home/user/project/src',
//   base: 'app.js',
//   ext: '.js',
//   name: 'app'
// }

// 从对象构建路径
const newPath = path.format({
  root: '/',
  dir: '/home/user/project/src',
  base: 'app.js',
  ext: '.js',
  name: 'app'
});
console.log(newPath);  // /home/user/project/src/app.js
```

### 2.4 路径判断

```javascript
const path = require('path');

// 判断是否为绝对路径
console.log(path.isAbsolute('/home/user'));  // true
console.log(path.isAbsolute('./user'));  // false
console.log(path.isAbsolute('C:\\Users'));  // true

// 路径比较
console.log(path.normalize('/home//user/../user'));  // /home/user
console.log(path.normalize('./src/../config/app.json'));  // config/app.json

// 相对路径计算
console.log(path.relative('/home/user/a', '/home/user/b'));  // ../b
console.log(path.relative('/home/user/project', '/home/user/project/src/app.js'));  // src/app.js
```

### 2.5 Windows 与 POSIX 路径

```javascript
const path = require('path');

// 在不同系统间处理路径
// Windows 使用反斜杠 \
// POSIX 使用正斜杠 /

// 跨平台路径分隔符
console.log(path.sep);  // \ (Windows) 或 / (POSIX)

// 路径分隔符列表
console.log(path.delimiter);  // ; (Windows) 或 : (POSIX)

// 转换为平台特定路径
const posixPath = '/home/user/project';
const winPath = path.win32.toNamespacedPath(posixPath);
console.log(winPath);

// 正确处理 Windows 路径
function processPath(inputPath) {
  // 将反斜杠转换为正斜杠
  return inputPath.replace(/\\/g, '/');
}
```

## 3. http/https 网络编程

### 3.1 创建 HTTP 服务器

```javascript
const http = require('http');

// 创建服务器
const server = http.createServer((req, res) => {
  // 设置响应头
  res.writeHead(200, {
    'Content-Type': 'text/html; charset=utf-8'
  });

  // 发送响应
  res.end('<h1>Hello World</h1>');
});

// 监听端口
server.listen(3000, () => {
  console.log('服务器运行在 http://localhost:3000');
});
```

### 3.2 HTTP 请求

```javascript
const http = require('http');

// GET 请求
const options = {
  hostname: 'example.com',
  port: 80,
  path: '/api/users',
  method: 'GET',
  headers: {
    'Content-Type': 'application/json'
  }
};

const req = http.request(options, (res) => {
  let data = '';

  res.on('data', (chunk) => {
    data += chunk;
  });

  res.on('end', () => {
    console.log('响应:', data);
  });
});

req.on('error', (err) => {
  console.error('请求错误:', err);
});

req.end();

// 使用 http.get 简化 GET 请求
http.get('http://example.com/api/users', (res) => {
  let data = '';

  res.on('data', (chunk) => {
    data += chunk;
  });

  res.on('end', () => {
    console.log(data);
  });
});
```

### 3.4 处理请求路由

```javascript
const http = require('http');
const { URL } = require('url');

const server = http.createServer((req, res) => {
  const url = new URL(req.url, `http://${req.headers.host}`);
  const pathname = url.pathname;
  const method = req.method;

  // 设置 CORS 头
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');

  // 路由处理
  if (method === 'GET' && pathname === '/api/users') {
    // 返回用户列表
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify([
      { id: 1, name: '张三' },
      { id: 2, name: '李四' }
    ]));
  } else if (method === 'GET' && pathname.startsWith('/api/users/')) {
    // 获取单个用户
    const id = pathname.split('/').pop();
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ id, name: '用户' + id }));
  } else if (method === 'POST' && pathname === '/api/users') {
    // 创建用户
    let body = '';
    req.on('data', chunk => body += chunk);
    req.on('end', () => {
      const user = JSON.parse(body);
      res.writeHead(201, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ success: true, user }));
    });
  } else {
    // 404
    res.writeHead(404, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ error: 'Not Found' }));
  }
});

server.listen(3000, () => {
  console.log('服务器运行在 http://localhost:3000');
});
```

### 3.7 HTTP 客户端封装

```javascript
const http = require('http');

class HttpClient {
  static request(options) {
    return new Promise((resolve, reject) => {
      const req = http.request(options, (res) => {
        let data = '';

        res.on('data', (chunk) => {
          data += chunk;
        });

        res.on('end', () => {
          try {
            const result = JSON.parse(data);
            resolve(result);
          } catch {
            resolve(data);
          }
        });
      });

      req.on('error', reject);

      if (options.body) {
        req.write(options.body);
      }

      req.end();
    });
  }

  static get(url, headers = {}) {
    const options = {
      hostname: new URL(url).hostname,
      port: new URL(url).port || 80,
      path: new URL(url).pathname,
      method: 'GET',
      headers
    };
    return this.request(options);
  }

  static post(url, body, headers = {}) {
    const jsonBody = JSON.stringify(body);
    const options = {
      hostname: new URL(url).hostname,
      port: new URL(url).port || 80,
      path: new URL(url).pathname,
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(jsonBody),
        ...headers
      },
      body: jsonBody
    };
    return this.request(options);
  }
}

// 使用示例
async function main() {
  const users = await HttpClient.get('http://localhost:3000/api/users');
  console.log(users);

  const newUser = await HttpClient.post('http://localhost:3000/api/users', {
    name: '张三',
    email: 'zhangsan@example.com'
  });
  console.log(newUser);
}
```

## 4. crypto 加密解密

crypto 模块提供加密功能，包括哈希算法、对称加密、非对称加密等。

### 4.1 哈希算法

```javascript
const crypto = require('crypto');

// MD5（不安全，仅用于演示）
const md5Hash = crypto.createHash('md5').update('Hello World').digest('hex');
console.log('MD5:', md5Hash);

// SHA-256
const sha256Hash = crypto.createHash('sha256').update('Hello World').digest('hex');
console.log('SHA-256:', sha256Hash);

// SHA-512
const sha512Hash = crypto.createHash('sha512').update('Hello World').digest('hex');
console.log('SHA-512:', sha512Hash);

// Base64 编码
const base64Hash = crypto.createHash('sha256').update('Hello World').digest('base64');
console.log('Base64:', base64Hash);

// 增量哈希
const hash = crypto.createHash('sha256');
hash.update('Hello');
hash.update(' ');
hash.update('World');
console.log('增量:', hash.digest('hex'));
```

### 4.2 HMAC（密钥哈希）

```javascript
const crypto = require('crypto');

// HMAC-SHA256
const key = 'my-secret-key';
const hmac = crypto.createHmac('sha256', key);

hmac.update('Hello World');
const result = hmac.digest('hex');

console.log('HMAC-SHA256:', result);

// 简化写法
const hmacResult = crypto.createHmac('sha256', 'secret')
  .update('Hello World')
  .digest('hex');
console.log('HMAC:', hmacResult);
```

### 4.3 对称加密

```javascript
const crypto = require('crypto');

// AES-256-CBC 加密
function encrypt(text, key, iv) {
  const cipher = crypto.createCipheriv(
    'aes-256-cbc',
    Buffer.from(key, 'hex'),
    Buffer.from(iv, 'hex')
  );

  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');

  return encrypted;
}

// AES-256-CBC 解密
function decrypt(encrypted, key, iv) {
  const decipher = crypto.createDecipheriv(
    'aes-256-cbc',
    Buffer.from(key, 'hex'),
    Buffer.from(iv, 'hex')
  );

  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');

  return decrypted;
}

// 生成随机密钥和 IV
const key = crypto.randomBytes(32).toString('hex');  // 256 位密钥
const iv = crypto.randomBytes(16).toString('hex');   // 128 位 IV

console.log('密钥:', key);
console.log('IV:', iv);

// 加密
const encrypted = encrypt('Hello World', key, iv);
console.log('加密:', encrypted);

// 解密
const decrypted = decrypt(encrypted, key, iv);
console.log('解密:', decrypted);
```

### 4.4 非对称加密

```javascript
const crypto = require('crypto');

// 生成 RSA 密钥对
const { publicKey, privateKey } = crypto.generateKeyPairSync('rsa', {
  modulusLength: 2048,
  publicKeyEncoding: {
    type: 'spki',
    format: 'pem'
  },
  privateKeyEncoding: {
    type: 'pkcs8',
    format: 'pem'
  }
});

console.log('公钥:', publicKey);
console.log('私钥:', privateKey);

// 使用公钥加密
function encryptWithPublicKey(text, publicKey) {
  const encrypted = crypto.publicEncrypt(
    {
      key: publicKey,
      padding: crypto.constants.RSA_PKCS1_OAEP_PADDING
    },
    Buffer.from(text)
  );
  return encrypted.toString('base64');
}

// 使用私钥解密
function decryptWithPrivateKey(encrypted, privateKey) {
  const decrypted = crypto.privateDecrypt(
    {
      key: privateKey,
      padding: crypto.constants.RSA_PKCS1_OAEP_PADDING
    },
    Buffer.from(encrypted, 'base64')
  );
  return decrypted.toString('utf8');
}

// 使用私钥签名
function sign(text, privateKey) {
  const sign = crypto.createSign('RSA-SHA256');
  sign.update(text);
  return sign.sign(privateKey, 'base64');
}

// 使用公钥验证签名
function verify(text, signature, publicKey) {
  const verify = crypto.createVerify('RSA-SHA256');
  verify.update(text);
  return verify.verify(publicKey, signature, 'base64');
}
```

### 4.5 数字签名


## 常见面试问题

### 问题 1：fs 模块的同步和异步如何选择？

**答案：** 在服务器环境中，优先使用异步方法（Promise 或回调），避免阻塞事件循环。同步方法只在启动脚本或 CLI 工具中使用。

### 问题 2：path.join 和 path.resolve 的区别？

**答案：** path.join 只是简单地拼接路径，不会解析为绝对路径；path.resolve 会将相对路径解析为绝对路径，类似于在终端中执行 cd 命令。

### 问题 3：crypto 模块常见的加密算法有哪些？

**答案：** 常见哈希算法有 MD5、SHA-1、SHA-256；对称加密有 AES；非对称加密有 RSA；HMAC 用于密钥哈希。

## 最佳实践

1. **文件操作**：优先使用 Promise API，避免回调地狱
2. **大文件处理**：使用流（Stream）处理，避免内存溢出
3. **路径处理**：使用 path 模块，确保跨平台兼容性
4. **敏感数据**：不要使用 MD5 存储密码，使用 bcrypt 或 scrypt
5. **HTTPS**：生产环境必须使用 HTTPS，使用 TLS 1.2+
6. **压缩**：对文本内容启用压缩，节省带宽

```javascript
const http = require('http');

// HTTP 请求封装
class HttpClient {
  static request(url, options = {}) {
    return new Promise((resolve, reject) => {
      const urlObj = new URL(url);
      const client = urlObj.protocol === 'https:' ? https : http;

      const requestOptions = {
        hostname: urlObj.hostname,
        port: urlObj.port || (urlObj.protocol === 'https:' ? 443 : 80),
        path: urlObj.pathname + urlObj.search,
        method: options.method || 'GET',
        headers: options.headers || {},
        timeout: options.timeout || 30000,
      };

      const req = client.request(requestOptions, (res) => {
        let data = '';

        res.on('data', (chunk) => {
          data += chunk;
        });

        res.on('end', () => {
          resolve({
            status: res.statusCode,
            headers: res.headers,
            data,
          });
        });
      });

      req.on('error', (error) => {
        reject(new Error(`HTTP 请求失败: ${error.message}`));
      });

      req.on('timeout', () => {
        req.destroy();
        reject(new Error('请求超时'));
      });

      if (options.body) {
        req.write(options.body);
      }

      req.end();
    });
  }

  // 带重试的请求
  static async requestWithRetry(url, options = {}, maxRetries = 3) {
    let lastError;

    for (let i = 0; i < maxRetries; i++) {
      try {
        const response = await this.request(url, options);

        // 检查状态码
        if (response.status >= 500) {
          throw new Error(`服务器错误: ${response.status}`);
        }

        return response;
      } catch (error) {
        lastError = error;

        // 指数退避
        const delay = Math.min(1000 * Math.pow(2, i), 10000);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }

    throw lastError;
  }
}

    );

  } catch (error) {
```

```javascript
const crypto = require('crypto');

// 加密错误处理
class CryptoError extends Error {
  constructor(message, code) {
    super(message);
    this.code = code;
    this.name = 'CryptoError';
  }
}

// 安全的加密类
class SecureEncryption {
  constructor(key) {
    if (!key || key.length !== 64) {
      throw new CryptoError('密钥必须是64字符的十六进制字符串', 'INVALID_KEY');
    }

    this.algorithm = 'aes-256-gcm';
    this.key = Buffer.from(key, 'hex');
  }

  encrypt(text) {
    try {
      const iv = crypto.randomBytes(16);
      const cipher = crypto.createCipheriv(this.algorithm, this.key, iv);

      let encrypted = cipher.update(text, 'utf8', 'hex');
      encrypted += cipher.final('hex');

      const authTag = cipher.getAuthTag();

      return {
        iv: iv.toString('hex'),
        data: encrypted,
        authTag: authTag.toString('hex'),
      };
    } catch (error) {
      throw new CryptoError(`加密失败: ${error.message}`, 'ENCRYPTION_FAILED');
    }
  }

  decrypt(encrypted) {
    try {
      if (!encrypted.iv || !encrypted.data || !encrypted.authTag) {
        throw new CryptoError('无效的加密数据格式', 'INVALID_FORMAT');
      }

      const decipher = crypto.createDecipheriv(
        this.algorithm,
        this.key,
        Buffer.from(encrypted.iv, 'hex')
      );

      decipher.setAuthTag(Buffer.from(encrypted.authTag, 'hex'));

      let decrypted = decipher.update(encrypted.data, 'hex', 'utf8');
      decrypted += decipher.final('utf8');

      return decrypted;
    } catch (error) {
      if (error instanceof CryptoError) {
        throw error;
      }
      throw new CryptoError(`解密失败: ${error.message}`, 'DECRYPTION_FAILED');
    }
  }
}

// 使用示例

  try {
    const encrypted = encryption.encrypt(JSON.stringify(userData));
    return encrypted;
  } catch (error) {
    console.error('加密用户数据失败:', error.message);
    throw error;
  }
}
```

```javascript
async function hashPassword(password) {
```

```javascript
    return (req, res, next) => {
      const key = req.url;
      const cached = this.cache.get(key);

      if (cached && Date.now() - cached.time < this.ttl) {
        res.setHeader('X-Cache', 'HIT');
        res.end(cached.data);
        return;
      }

      res.setHeader('X-Cache', 'MISS');

      // 保存原始方法
      const originalEnd = res.end;
      res.end = function(data) {
        if (res.statusCode === 200) {
          this.cache.set(key, { data, time: Date.now() });
        }
        originalEnd.call(res, data);
      }.bind(this);

      next();
    };
  }
}
```

```javascript
    this.chunkDir = options.chunkDir || './chunks';

  ensureDirectories() {
    [this.uploadDir, this.chunkDir].forEach(dir => {
      if (!fs.existsSync(dir)) {
        fs.mkdirSync(dir, { recursive: true });
      }
    });
  }

  // 上传分片
  async uploadChunk(fileId, chunkIndex, chunkData) {
    const chunkPath = path.join(this.chunkDir, fileId, `${chunkIndex}.chunk`);

    await fs.promises.mkdir(path.dirname(chunkPath), { recursive: true });
    await fs.promises.writeFile(chunkPath, chunkData);

    return { success: true, chunkIndex };
  }

  // 合并分片
  async mergeChunks(fileId, totalChunks, fileName) {
    const filePath = path.join(this.uploadDir, fileId + path.extname(fileName));
    const writeStream = fs.createWriteStream(filePath);

    for (let i = 0; i < totalChunks; i++) {
      const chunkPath = path.join(this.chunkDir, fileId, `${i}.chunk`);
      const chunkData = await fs.promises.readFile(chunkPath);
      writeStream.write(chunkData);
    }

    writeStream.end();

    // 清理分片
    await fs.promises.rm(path.join(this.chunkDir, fileId), { recursive: true });

    return { success: true, filePath };
  }

  // 计算文件哈希
  async calculateHash(filePath) {
    return new Promise((resolve, reject) => {
      const hash = crypto.createHash('md5');
      const stream = fs.createReadStream(filePath);

      stream.on('data', chunk => hash.update(chunk));
      stream.on('end', () => resolve(hash.digest('hex')));
      stream.on('error', reject);
    });
  }
}
```

```javascript
const https = require('https');

class ProxyServer {
  constructor(options = {}) {
    this.port = options.port || 8080;
    this.target = options.target;
    this.timeout = options.timeout || 30000;
  }

  start() {
    const server = http.createServer((req, res) => {
      this.handleRequest(req, res);
    });

    server.listen(this.port, () => {
      console.log(`代理服务器运行在端口 ${this.port}`);
    });

    return server;
  }

  async handleRequest(req, res) {
    const targetUrl = new URL(this.target || req.url);
    const client = targetUrl.protocol === 'https:' ? https : http;

    const options = {
      hostname: targetUrl.hostname,
      port: targetUrl.port || (targetUrl.protocol === 'https:' ? 443 : 80),
      path: targetUrl.pathname + targetUrl.search,
      method: req.method,
      headers: {
        ...req.headers,
        host: targetUrl.host,
      },
    };

    const proxyReq = client.request(options, (proxyRes) => {
      res.writeHead(proxyRes.statusCode, proxyRes.headers);
      proxyRes.pipe(res);
    });

    proxyReq.on('error', (err) => {
      res.statusCode = 502;
      res.end(`代理错误: ${err.message}`);
    });

    req.pipe(proxyReq);
  }
}
```

```javascript
class SecureFileDownloader {
  constructor(baseDir) {
    this.baseDir = path.resolve(baseDir);
  }

  // 验证路径安全
  validatePath(userPath) {
    const resolvedPath = path.resolve(this.baseDir, userPath);

    if (!resolvedPath.startsWith(this.baseDir)) {
      throw new Error('非法路径');
    }

    return resolvedPath;
  }

  // 下载文件
  async download(res, userPath, options = {}) {
    const filePath = this.validatePath(userPath);

    // 检查文件是否存在
    const stats = await fs.promises.stat(filePath);

    if (!stats.isFile()) {
      throw new Error('不是文件');
    }

    // 设置响应头
    const fileName = options.fileName || path.basename(filePath);
    const mimeType = this.getMimeType(filePath);

    res.setHeader('Content-Type', mimeType);
    res.setHeader('Content-Disposition', `attachment; filename="${encodeURIComponent(fileName)}"`);
    res.setHeader('Content-Length', stats.size);

    // 流式传输
    const readStream = fs.createReadStream(filePath);
    readStream.pipe(res);

    return new Promise((resolve, reject) => {
      readStream.on('end', resolve);
      readStream.on('error', reject);
    });
  }

  // 获取 MIME 类型
  getMimeType(filePath) {
    const ext = path.extname(filePath).toLowerCase();
    const mimeTypes = {
      '.pdf': 'application/pdf',
      '.zip': 'application/zip',
      '.jpg': 'image/jpeg',
      '.png': 'image/png',
      '.txt': 'text/plain',
      '.json': 'application/json',
    };

    return mimeTypes[ext] || 'application/octet-stream';
  }
```

```javascript
const fs = require('fs').promises;
const path = require('path');
const crypto = require('crypto');

class FileService {
  constructor(baseDir) {
    this.baseDir = path.resolve(baseDir);
    this.init();
  }

  async init() {
    try {
      await fs.mkdir(this.baseDir, { recursive: true });
    } catch {}
  }

  // 列出文件
  async list(dir = '') {
    const targetDir = path.join(this.baseDir, dir);
    const files = await fs.readdir(targetDir, { withFileTypes: true });

    return files.map(file => ({
      name: file.name,
      isDirectory: file.isDirectory(),
      isFile: file.isFile(),
      path: path.join(dir, file.name),
    }));
  }

  // 读取文件
  async read(filePath) {
    const fullPath = path.join(this.baseDir, filePath);
    return fs.readFile(fullPath, 'utf8');
  }

  // 写入文件
  async write(filePath, content) {
    const fullPath = path.join(this.baseDir, filePath);
    await fs.mkdir(path.dirname(fullPath), { recursive: true });
    await fs.writeFile(fullPath, content, 'utf8');
  }

  // 删除文件
  async delete(filePath) {
    const fullPath = path.join(this.baseDir, filePath);
    await fs.unlink(fullPath);
  }

  // 移动文件
  async move(source, destination) {
    const srcPath = path.join(this.baseDir, source);
    const destPath = path.join(this.baseDir, destination);
    await fs.mkdir(path.dirname(destPath), { recursive: true });
    await fs.rename(srcPath, destPath);
  }

  // 复制文件
  async copy(source, destination) {
    const srcPath = path.join(this.baseDir, source);
    const destPath = path.join(this.baseDir, destination);
    await fs.mkdir(path.dirname(destPath), { recursive: true });
    await fs.copyFile(srcPath, destPath);
  }

  // 获取文件信息
  async stat(filePath) {
    const fullPath = path.join(this.baseDir, filePath);
    const stats = await fs.stat(fullPath);

    return {
      size: stats.size,
      created: stats.birthtime,
      modified: stats.mtime,
      isDirectory: stats.isDirectory(),
      isFile: stats.isFile(),
    };
  }

  // 计算文件哈希
  async hash(filePath, algorithm = 'sha256') {
    const fullPath = path.join(this.baseDir, filePath);
    const content = await fs.readFile(fullPath);
    return crypto.createHash(algorithm).update(content).digest('hex');
  }
}
```
