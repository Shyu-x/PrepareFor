## 1. 关系型数据库概述

### 1.1 什么是关系型数据库

关系型数据库是基于关系模型的数据库，使用SQL（结构化查询语言）进行数据操作。数据以表（Table）的形式存储，表之间通过关系（Relationship）相互关联。

```
关系型数据库核心概念：
┌─────────────────┐     ┌─────────────────┐
│    Users        │     │    Orders      │
├─────────────────┤     ├─────────────────┤
│ id (PK)        │     │ id (PK)        │
│ username        │     │ user_id (FK)    │
│ email           │────<│ product_id      │
│ password_hash   │     │ quantity        │
│ created_at     │     │ total_price     │
└─────────────────┘     │ status          │
                       │ created_at      │
                       └─────────────────┘
```

### 1.2 核心特性

| 特性 | 说明 |
|------|------|
| **ACID事务** | 原子性、一致性、隔离性、持久性 |
| **SQL语言** | 标准化查询语言 |
| **关系完整性** | 主键、外键、约束 |
| **成熟稳定** |  decades of 优化和稳定性 |

### 1.3 主流关系型数据库对比

| 数据库 | 特点 | 适用场景 |
|--------|------|----------|
| **PostgreSQL** | 功能最强大的开源数据库 | 复杂查询、GIS、数据仓库 |
| **MySQL** | 最流行的开源数据库 | Web应用、日志、电商 |
| **Oracle** | 企业级功能 | 金融、大型企业应用 |
| **SQL Server** | 微软生态集成 | .NET企业应用 |

## 2. PostgreSQL完全指南

### 2.1 什么是PostgreSQL

PostgreSQL是一个功能强大的开源对象关系型数据库系统，支持SQL标准并提供了许多高级特性，如复杂查询、外键、触发器、视图、事务完整性、多版本并发控制等。

### 2.2 核心特性

- **ACID完全支持**：确保事务的原子性、一致性、隔离性、持久性
- **复杂查询支持**：窗口函数、CTE（公共表表达式）、递归查询
- **JSON支持**：原生JSON和JSONB数据类型
- **PostGIS**：强大的地理空间数据支持
- **扩展性**：支持自定义函数、数据类型、操作符
- **MVCC**：多版本并发控制，高并发性能好

### 2.3 基础操作

```sql
-- 创建数据库
CREATE DATABASE myapp;

-- 连接到数据库
\c myapp

-- 创建表
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建带约束的表
CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    total_amount DECIMAL(10, 2) NOT NULL,
    status VARCHAR(20) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT positive_amount CHECK (total_amount > 0)
);

-- 插入数据
INSERT INTO users (username, email, password_hash)
VALUES ('张三', 'zhangsan@example.com', 'hash123');

-- 查询数据
SELECT id, username, email, created_at
FROM users
WHERE id > 10
ORDER BY created_at DESC
LIMIT 20;

-- 更新数据
UPDATE users
SET email = 'newemail@example.com', updated_at = CURRENT_TIMESTAMP
WHERE id = 1;

-- 删除数据
DELETE FROM users WHERE id = 1;

-- 关联查询
SELECT o.id, u.username, o.total_amount, o.status
FROM orders o
JOIN users u ON o.user_id = u.id
WHERE o.status = 'pending'
ORDER BY o.created_at DESC;
```

### 2.4 高级查询

```sql
-- 窗口函数
SELECT
    user_id,
    order_date,
    amount,
    SUM(amount) OVER (PARTITION BY user_id ORDER BY order_date) as running_total,
    AVG(amount) OVER (PARTITION BY user_id) as user_avg,
    RANK() OVER (PARTITION BY user_id ORDER BY amount DESC) as amount_rank
FROM orders;

-- CTE（公共表表达式）
WITH monthly_sales AS (
    SELECT
        DATE_TRUNC('month', created_at) as month,
        SUM(total_amount) as revenue,
        COUNT(*) as order_count
    FROM orders
    GROUP BY DATE_TRUNC('month', created_at)
)
SELECT * FROM monthly_sales
WHERE revenue > 10000
ORDER BY month;

-- 递归查询（树形结构）
WITH RECURSIVE org_tree AS (
    -- 根节点
    SELECT id, name, manager_id, 1 as level
    FROM employees
    WHERE manager_id IS NULL

    UNION ALL

    -- 递归部分
    SELECT e.id, e.name, e.manager_id, ot.level + 1
    FROM employees e
    JOIN org_tree ot ON e.manager_id = ot.id
)
SELECT * FROM org_tree;

-- 聚合查询
SELECT
    u.username,
    COUNT(o.id) as order_count,
    SUM(o.total_amount) as total_spent,
    AVG(o.total_amount) as avg_order_value,
    MAX(o.created_at) as last_order_date
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.id, u.username
HAVING COUNT(o.id) > 5
ORDER BY total_spent DESC;
```

### 2.5 索引优化

```sql
-- 单列索引
CREATE INDEX idx_users_email ON users(email);

-- 多列索引
CREATE INDEX idx_orders_user_status ON orders(user_id, status);

-- 唯一索引
CREATE UNIQUE INDEX idx_users_username ON users(username);

-- 表达式索引
CREATE INDEX idx_users_lower_email ON users(LOWER(email));

-- 部分索引
CREATE INDEX idx_active_users ON users(email)
WHERE status = 'active';

-- 复合索引使用原则
-- 索引顺序：(a, b, c)
-- 优化查询：WHERE a = ? AND b = ? AND c = ?
--         WHERE a = ? AND b = ?
--         WHERE a = ?
-- 不适用：WHERE b = ? OR WHERE c = ?
```

### 2.6 事务控制

```sql
-- 开始事务
BEGIN;

-- 或者使用
START TRANSACTION;

-- 执行操作
INSERT INTO users (username, email, password_hash)
VALUES ('新用户', 'new@example.com', 'hash');

INSERT INTO orders (user_id, total_amount)
VALUES (currval('users_id_seq'), 100.00);

-- 提交事务
COMMIT;

-- 回滚事务
ROLLBACK;

-- 保存点
SAVEPOINT sp1;

-- 回滚到保存点
ROLLBACK TO SAVEPOINT sp1;

-- 事务隔离级别
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
```

## 3. MySQL数据库指南

### 3.1 MySQL基础

```sql
-- 创建数据库
CREATE DATABASE myapp CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建表
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

) ENGINE=InnoDB;

-- 插入数据
INSERT INTO users (username, email, password_hash)

-- 查询
SELECT * FROM users WHERE id = 1;

UPDATE users SET email = 'new@example.com' WHERE id = 1;

-- 删除
DELETE FROM users WHERE id = 1;

-- 关联查询
SELECT u.username, o.order_id, o.total
FROM users u
INNER JOIN orders o ON u.id = o.user_id;
```

| **MyISAM** | 全文索引、表级锁 | 只读、静态表 |

## 4. NoSQL数据库

### 4.1 NoSQL概述

NoSQL（Not Only SQL）数据库是非关系型数据库的总称，特点是灵活的模式、高可扩展性、高性能。

```
NoSQL数据库分类：
┌─────────────────────────────────────────────────────────────┐
│                        NoSQL                                │
├──────────────┬──────────────┬─────────────┬───────────────┤
│  文档数据库   │  键值数据库   │  列式数据库  │   图形数据库   │
├──────────────┼──────────────┼─────────────┼───────────────┤
│  MongoDB     │  Redis       │  Cassandra  │  Neo4j        │
│  CouchDB     │  Memcached   │  HBase     │  ArangoDB     │
│  PostgreSQL  │  DynamoDB    │  BigTable  │  TigerGraph   │
│  (JSON)      │  etcd        │  Druid     │               │
└──────────────┴──────────────┴─────────────┴───────────────┘
```

### 4.2 NoSQL vs 关系型数据库

| 特性 | 关系型数据库 | NoSQL数据库 |
|------|-------------|-------------|
| 数据模型 | 表、行、列 | 文档、键值、图 |
| 模式 | 固定模式 | 灵活模式 |
| 事务 | ACID | 最终一致性 |
| 扩展性 | 垂直扩展 | 水平扩展 |
| 查询语言 | SQL | 各不相同 |
| 适用场景 | 复杂查询、事务 | 大数据、高并发 |

## 5. MongoDB完全指南

### 5.1 什么是MongoDB

MongoDB是一个基于分布式文件存储的文档数据库，使用BSON（Binary JSON）格式存储数据，具有高性能、高可用性和易扩展性。

```
MongoDB数据结构：
┌─────────────────────────────────────────────────┐
│                  Database                         │
│  ┌───────────────────────────────────────────┐  │
│  │              Collection (集合)              │  │
│  │  ┌─────────────────────────────────────┐  │  │
│  │  │         Document (文档)              │  │  │
│  │  │  {                                   │  │  │
│  │  │    "_id": ObjectId("..."),         │  │  │
│  │  │    "name": "张三",                  │  │  │
│  │  │    "email": "zhangsan@example.com",│  │  │
│  │  │    "age": 25,                       │  │  │
│  │  │    "address": {                     │  │  │
│  │  │      "city": "北京",                │  │  │
│  │  │      "district": "朝阳区"           │  │  │
│  │  │    },                                │  │  │
│  │  │    "tags": ["developer", "coder"], │  │  │
│  │  │    "createdAt": Date(...)          │  │  │
│  │  │  }                                   │  │  │
│  │  └─────────────────────────────────────┘  │  │

└─────────────────────────────────────────────────┘
```

### 5.7 事务（MongoDB 4.0+）

```javascript
// 启动事务

    readConcern: { level: "snapshot" },
    writeConcern: { level: "majority" }
});

try {
    const usersCollection = session.getDatabase("myapp").users;
    const ordersCollection = session.getDatabase("myapp").orders;

    // 操作1：创建用户

    session.commitTransaction();
} catch (error) {
    // 回滚事务
    session.abortTransaction();
    throw error;
} finally {
    session.endSession();
}
```

## 6. Redis缓存数据库

### 6.1 什么是Redis

Redis（Remote Dictionary Server）是一个开源的、内存中的数据结构存储系统，可用作数据库、缓存和消息队列。它支持多种类型的数据结构，如字符串、哈希、列表、集合、有序集合等。

### 6.2 数据类型

```

│  数据类型    │  示例命令              │  典型用途       │
├─────────────────────────────────────────────────────────┤
│  STRING     │  SET, GET, INCR       │  缓存、计数器    │
│  HASH       │  HSET, HGET, HGETALL  │  对象存储        │
│  LIST       │  LPUSH, RPUSH, LRANGE │  消息队列        │
│  SET        │  SADD, SMEMBERS       │  标签、去重      │
│  ZSET       │  ZADD, ZRANGE         │  排行榜、排序    │
│  BITMAP     │  SETBIT, BITCOUNT     │  用户行为统计    │
```


### 6.5 JavaScript/Node.js 使用

```javascript
// 使用 ioredis
import Redis from 'ioredis';

const redis = new Redis({
    host: 'localhost',
    port: 6379,
    password: '',
    // 密码重试
    retryStrategy: (times) => {
        return Math.min(times * 50, 2000);
    }
});

// 字符串操作
await redis.set('user:1', JSON.stringify({ name: '张三' }));
const user = JSON.parse(await redis.get('user:1'));

// 缓存模式示例
async function getUser(id) {
    const cacheKey = `user:${id}`;

    // 先从缓存获取
    const cached = await redis.get(cacheKey);
    if (cached) {
        return JSON.parse(cached);
    }

    // 缓存未命中，从数据库获取
    const user = await database.users.findById(id);

    if (user) {
        // 写入缓存，设置1小时过期
        await redis.setex(cacheKey, 3600, JSON.stringify(user));
    }

    return user;
}

// 分布式锁
async function acquireLock(key, ttl = 10) {
    const lockKey = `lock:${key}`;
    const result = await redis.set(lockKey, '1', 'EX', ttl, 'NX');
    return result === 'OK';
}

async function releaseLock(key) {
    const lockKey = `lock:${key}`;
    await redis.del(lockKey);
}

// 消息发布/订阅
const subscriber = new Redis();
const publisher = new Redis();

subscriber.subscribe('notifications');

subscriber.on('message', (channel, message) => {
    if (channel === 'notifications') {
        console.log('收到通知:', message);
    }
});

await publisher.publish('notifications', JSON.stringify({
    type: 'user_registered',
    userId: 123
}));

// 限流示例
async function rateLimit(key, limit, window) {
    const now = Date.now();
    const windowKey = `ratelimit:${key}:${Math.floor(now / window)}`;

    const count = await redis.incr(windowKey);

    if (count === 1) {
        await redis.expire(windowKey, window);
    }

    return {
        allowed: count <= limit,
        remaining: Math.max(0, limit - count)
    };
}
```

### 6.6 Redis使用场景

| 场景 | 数据类型 | 说明 |
|------|----------|------|

| **分布式锁** | STRING | 控制并发、防止重复提交 |
| **消息队列** | LIST/STREAM | 异步任务处理 |

| **最新列表** | LIST | 最新动态、最新商品 |
| **去重** | SET | 用户标签、投票记录 |
| **限流** | STRING | API调用频率限制 |
