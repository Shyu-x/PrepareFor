# 进程间通信与协程：操作系统内核深度解析 (2026版)

## 1. 概述：为什么需要进程间通信？

在 2026 年的分布式系统中，进程间通信 (IPC) 是构建高并发、高可用系统的基石。无论是微服务架构还是单体应用，进程之间都需要交换数据、协调工作。

## 2. 进程间通信 (IPC) 机制

### 2.1 管道 (Pipe)

```c
// 匿名管道：C 语言实现
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>

int main() {
    int pipefd[2];  // pipefd[0] 读端, pipefd[1] 写端
    pid_t pid;

    // 创建管道
    if (pipe(pipefd) == -1) {
        perror("pipe");
        exit(EXIT_FAILURE);
    }

    pid = fork();

    if (pid == -1) {
        perror("fork");
        exit(EXIT_FAILURE);
    }

    if (pid == 0) {
        // 子进程：关闭读端，写入数据
        close(pipefd[0]);  // 关闭读端

        const char *msg = "Hello from child!";
        write(pipefd[1], msg, strlen(msg));

        close(pipefd[1]);  // 关闭写端
        exit(EXIT_SUCCESS);
    } else {
        // 父进程：关闭写端，读取数据
        close(pipefd[1]);  // 关闭写端

        char buf[128];
        ssize_t n = read(pipefd[0], buf, sizeof(buf));
        buf[n] = '\0';

        printf("Parent received: %s\n", buf);

        close(pipefd[0]);  // 关闭读端
        wait(NULL);  // 等待子进程
    }

    return 0;
}
```

import os

fifo_path = '/tmp/my_fifo'

# 创建 FIFO（如果不存在）
if not os.path.exists(fifo_path):
    os.mkfifo(fifo_path)

# 写入进程
def write_to_fifo():
    with open(fifo_path, 'w') as f:
        f.write("Message from writer\n")

# 读取进程
def read_from_fifo():
    with open(fifo_path, 'r') as f:
        while True:
            line = f.readline()
            if line:
                print(f"Received: {line}")

### 2.3 共享内存 (Shared Memory)

```c
// 共享内存：最高效的 IPC 方式
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>

#define SHM_NAME "/my_shm"
#define SHM_SIZE 4096

int main() {
    int shm_fd;
    void *ptr;

    // 创建共享内存对象
    shm_fd = shm_open(SHM_NAME, O_CREAT | O_RDWR, 0666);
    if (shm_fd == -1) {
        perror("shm_open");
        exit(EXIT_FAILURE);
    }

    // 设置大小
    ftruncate(shm_fd, SHM_SIZE);

    // 映射到进程地址空间
    ptr = mmap(0, SHM_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, shm_fd, 0);
    if (ptr == MAP_FAILED) {
        perror("mmap");
        exit(EXIT_FAILURE);
    }

    pid_t pid = fork();

    if (pid == 0) {
        // 子进程：写入数据
        sprintf(ptr, "Hello from child process!");
    } else {
        // 父进程：读取数据
        wait(NULL);
        printf("Parent read: %s\n", (char *)ptr);
    }

    // 清理
    munmap(ptr, SHM_SIZE);
    close(shm_fd);
    shm_unlink(SHM_NAME);

    return 0;
}
```

### 2.4 消息队列 (Message Queue)

```python
import posix.ipc as ipc
import struct

# 创建消息队列
mq = ipc.MessageQueue('/my_mq', ipc.O_CREAT, mode=0o666, max_messages=100, max_message_size=1024)

# 发送消息
def send_message(msg_type, data):
    # 格式：消息类型(4字节) + 数据
    message = struct.pack('i', msg_type) + data.encode()
    mq.send(message)

# 接收消息
def receive_message():
    msg = mq.receive()
    msg_type, = struct.unpack('i', msg[:4])
    data = msg[4:].decode()
    return msg_type, data

# 示例
send_message(1, "Hello from sender!")
msg_type, data = receive_message()
print(f"Type: {msg_type}, Data: {data}")
```

### 2.5 Socket 通信

```python
# Unix Domain Socket（本地通信，比 TCP 更快）
import socket
import os

socket_path = '/tmp/my_socket.sock'

# 服务器
def server():
    if os.path.exists(socket_path):
        os.unlink(socket_path)

    server = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
    server.bind(socket_path)
    server.listen(1)

    print(f"Server listening on {socket_path}")

    conn, addr = server.accept()
    data = conn.recv(1024)
    print(f"Received: {data.decode()}")

    conn.send(b"Hello from server!")
    conn.close()
    server.close()

# 客户端
def client():
    client = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
    client.connect(socket_path)

    client.send(b"Hello from client!")
    data = client.recv(1024)
    print(f"Received: {data.decode()}")

    client.close()
```

### 2.6 IPC 机制对比

| 机制 | 速度 | 数据量 | 复杂度 | 适用场景 |
|------|------|--------|--------|----------|
| **管道** | 快 | 小 | 低 | 父子进程通信 |
| **FIFO** | 快 | 小 | 低 | 无亲缘关系进程 |
| **共享内存** | 最快 | 大 | 中 | 高性能数据交换 |
| **消息队列** | 中 | 中 | 中 | 异步任务队列 |
| **Socket** | 慢 | 任意 | 高 | 网络通信、跨机器 |

## 3. 线程同步机制

### 3.1 互斥锁 (Mutex)

```c
#include <pthread.h>

typedef struct {
    pthread_mutex_t mutex;
    int shared_data;
} SharedData;

void init_shared_data(SharedData *data) {
    pthread_mutex_init(&data->mutex, NULL);
    data->shared_data = 0;
}

void *producer(void *arg) {
    SharedData *data = (SharedData *)arg;

    for (int i = 0; i < 1000000; i++) {
        pthread_mutex_lock(&data->mutex);
        data->shared_data++;  // 临界区
        pthread_mutex_unlock(&data->mutex);
    }

    return NULL;
}

int main() {
    SharedData data;
    init_shared_data(&data);

    pthread_t t1, t2;

    // 创建两个生产者线程
    pthread_create(&t1, NULL, producer, &data);
    pthread_create(&t2, NULL, producer, &data);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    printf("Final value: %d\n", data.shared_data);

    pthread_mutex_destroy(&data.mutex);

    return 0;
}
```

### 3.2 读写锁 (RWLock)

```c
#include <pthread.h>

typedef struct {
    pthread_rwlock_t rwlock;
    int data;
} DataStore;

// 读操作：多个线程可以同时读
void *reader(void *arg) {
    DataStore *store = (DataStore *)arg;

    pthread_rwlock_rdlock(&store->rwlock);
    // 读操作，多个线程可以同时执行
    int value = store->data;
    printf("Reader: %d\n", value);
    pthread_rwlock_unlock(&store->rwlock);

    return NULL;
}

// 写操作：独占访问
void *writer(void *arg) {
    DataStore *store = (DataStore *)arg;

    pthread_rwlock_wrlock(&store->rwlock);
    // 写操作，独占
    store->data++;
    pthread_rwlock_unlock(&store->rwlock);

    return NULL;
}
```

### 3.3 条件变量 (Condition Variable)

```c
#include <pthread.h>

typedef struct {
    pthread_mutex_t mutex;
    pthread_cond_t cond;
    int ready;
} Context;

void wait_for_ready(Context *ctx) {
    pthread_mutex_lock(&ctx->mutex);

    while (!ctx->ready) {
        // 释放锁并等待，直到另一个线程 signal
        pthread_cond_wait(&ctx->cond, &ctx->mutex);
    }

    pthread_mutex_unlock(&ctx->mutex);
}

void set_ready(Context *ctx) {
    pthread_mutex_lock(&ctx->mutex);
    ctx->ready = 1;
    pthread_cond_signal(&ctx->cond);  // 唤醒一个等待的线程
    // 或 pthread_cond_broadcast(&ctx->cond); 唤醒所有
    pthread_mutex_unlock(&ctx->mutex);
}
```

### 3.4 信号量 (Semaphore)

```c
#include <semaphore.h>

#define BUFFER_SIZE 10
int buffer[BUFFER_SIZE];
int count = 0;

// 空信号量（生产者等待空槽）
sem_t empty;
// 满信号量（消费者等待数据）
sem_t full;
// 互斥锁（保护缓冲区）
pthread_mutex_t mutex;

void producer(int item) {
    sem_wait(&empty);           // 等待空槽
    pthread_mutex_lock(&mutex);  // 进入临界区

    buffer[count++] = item;

    pthread_mutex_unlock(&mutex);
    sem_post(&full);            // 增加可用数据
}

int consumer() {
    sem_wait(&full);            // 等待数据
    pthread_mutex_lock(&mutex); // 进入临界区

    int item = buffer[--count];

    pthread_mutex_unlock(&mutex);
    sem_post(&empty);          // 增加空槽

    return item;
}
```

## 4. 协程 (Coroutine)：用户态轻量级并发

### 4.1 协程 vs 线程

```
┌─────────────────────────────────────────────────────────┐
│                    线程模型                              │
├─────────────────────────────────────────────────────────┤
│                                                         │
│   线程1 (内核态)                                        │
│   ┌─────────────────────────────────────────────────┐ │
│   │  协程A (用户态)  │  协程B (用户态)  │  协程C    │ │
│   │  ┌─────────┐    │  ┌─────────┐     │  ┌───────┐ │ │
│   │  │ await() │ ──▶│  │ yield() │ ──▶│  │ work  │ │ │
│   │  └─────────┘    │  └─────────┘     │  └───────┘ │ │
│   └─────────────────────────────────────────────────┘ │
│                                                         │
│   内核切换（上下文保存/恢复）= 数百微秒                  │
└─────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────┐
│                    协程模型                              │
├─────────────────────────────────────────────────────────┤
│                                                         │
│   协程切换（用户态）= 微秒级甚至更少                     │
│                                                         │
│   协程A ──await──▶ 协程B ──yield──▶ 协程C ──resume──▶ 协程A │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

### 4.2 libuv 的协程实现

```javascript
// libuv (Node.js 底层) 使用 continuation-passing style
// 但我们可以使用 async/await 作为更高级的抽象

async function fetchUser(userId) {
  // 这里的 await 会让出执行权给事件循环
  const response = await fetch(`/api/users/${userId}`);
  const user = await response.json();
  return user;
}

async function fetchAllUsers() {
  // 并发执行：启动所有请求，但不等待完成
  const promises = [1, 2, 3, 4, 5].map(id => fetchUser(id));

  // 等待所有完成
  const users = await Promise.all(promises);
  return users;
}
```

### 4.3 Go 协程

```go
package main

import (
    "fmt"
    "time"
)

// goroutine：轻量级协程，由 Go 运行时调度
func worker(id int, jobs <-chan int, results chan<- int) {
    for job := range jobs {
        fmt.Printf("Worker %d processing job %d\n", id, job)
        time.Sleep(time.Millisecond * 100) // 模拟工作
        results <- job * 2
    }
}

func main() {
    // 创建通道
    jobs := make(chan int, 100)
    results := make(chan int, 100)

    // 启动 3 个 worker 协程
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }

    // 发送 5 个任务
    for j := 1; j <= 5; j++ {
        jobs <- j
    }
    close(jobs)  // 关闭通道

    // 收集结果
    for a := 1; a <= 5; a++ {
        <-results
    }
}
```

### 4.4 Python 协程 (asyncio)

```python
import asyncio

# 定义协程
async def fetch_data(delay, id):
    print(f"Coroutine {id} starting")
    await asyncio.sleep(delay)  # 让出执行权
    print(f"Coroutine {id} finished")
    return f"Data {id}"

async def main():
    # 并发执行多个协程
    tasks = [
        fetch_data(1, "A"),
        fetch_data(2, "B"),
        fetch_data(0.5, "C"),
    ]

    # 等待所有完成
    results = await asyncio.gather(*tasks)
    print(f"Results: {results}")

# 运行事件循环
asyncio.run(main())
```

### 4.5 C++20 协程

```cpp
#include <coroutine>
#include <future>
#include <iostream>

// 定义协程返回类型
struct Task {
    struct promise_type {
        int value;

        Task get_return_object() {
            return Task{std::coroutine_handle<promise_type>::from_promise(*this)};
        }

        std::suspend_never initial_suspend() { return {}; }
        std::suspend_never final_suspend() noexcept { return {}; }

        void return_value(int v) { value = v; }
        void unhandled_exception() {}
    };

    std::coroutine_handle<promise_type> handle;

    Task(std::coroutine_handle<promise_type> h) : handle(h) {}
    ~Task() { if (handle) handle.destroy(); }

    int get() {
        handle.resume();  // 执行协程
        return handle.promise().value;
    }
};

// 协程函数
Task async_operation(int value) {
    std::cout << "Starting operation\n";
    co_await std::suspend_always{};  // 让出执行权
    std::cout << "Resuming operation\n";
    co_return value * 2;  // 返回值
}

int main() {
    Task task = async_operation(42);
    std::cout << "Main before get()\n";
    int result = task.get();
    std::cout << "Result: " << result << "\n";
    return 0;
}
```

## 6. 面试高频问题

**Q1：进程和线程的本质区别是什么？**

**答：** 进程是资源分配的最小单位，拥有独立的地址空间；线程是 CPU 调度的最小单位，同一进程内的线程共享堆和全局变量。进程间通信需要 IPC 机制，而线程间可以直接通过共享内存通信（需要同步）。

**Q2：为什么协程比线程高效？**

**答：** 线程切换需要进入内核态（上下文保存/恢复约数百微秒），而协程切换在用户态完成（微秒级）。此外，线程数量受限于系统调度，协程可以在单线程内创建数万个。协程通过主动让出执行权实现协作式多任务，避免了线程切换的开销。

**Q3：互斥锁和读写锁的使用场景？**

**答：** 互斥锁适用于写操作占主导的场景。读写锁在读多写少场景下性能更优：多个线程可以同时持有读锁并行读取，但写锁必须独占。Redis 的内部实现大量使用读写锁优化并发读取性能。
