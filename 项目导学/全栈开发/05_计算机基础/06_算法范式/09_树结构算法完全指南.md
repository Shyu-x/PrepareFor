# 树结构算法完全指南

## 一、树的基础概念

### 1.1 树的定义与分类

**树（Tree）** 是由 n 个节点组成的有限集合，满足以下条件：
- 有且只有一个根节点（Root）
- 除根节点外，其余节点分为 m 个互不相交的有限集合，每个集合都是一棵树
- 树中每个节点有零个或多个子节点

**类比理解——家族族谱：**

想象一个家族族谱：
- 祖先是根节点
- 父亲和儿子之间的关系就像树的父子关系
- 叔叔、伯伯等就是"兄弟"关系
- 族谱中的每一代就像树的层级

### 1.2 树的分类

```
树的分类

二叉树
├── BST（二叉搜索树）
├── AVL树（平衡二叉搜索树）

└── 完全二叉树/满二叉树

多叉树

特殊树

└── 表达式树
```

## 二、二叉搜索树（BST）

### 2.1 BST基础

**二叉搜索树（Binary Search Tree）** 是一种特殊的二叉树，满足以下性质：
- 左子树上所有节点的值都小于根节点的值
- 右子树上所有节点的值都大于根节点的值
- 左右子树也分别是二叉搜索树

```typescript
/**
 * 二叉搜索树（BST）实现
 *
 * 时间复杂度：
 * - 搜索/插入/删除：平均O(log n)，最差O(n)（退化成链表）
 * - 空间复杂度：O(n)
 */

interface BSTNode {
  val: number;
  left: BSTNode | null;
  right: BSTNode | null;
}

class BinarySearchTree {
  private root: BSTNode | null = null;

  /**
   * 插入
   */
  insert(val: number): void {
    const newNode: BSTNode = {val, left: null, right: null};

    if (!this.root) {
      this.root = newNode;
      return;
    }

    let current = this.root;
    while (true) {
      if (val < current.val) {
        if (!current.left) {
          current.left = newNode;
          return;
        }
        current = current.left;
      } else {
        if (!current.right) {
          current.right = newNode;
          return;
        }
        current = current.right;
      }
    }
  }

  /**
   * 搜索
   */
  search(val: number): boolean {
    let current = this.root;

    while (current) {
      if (val === current.val) return true;
      if (val < current.val) {
        current = current.left;
      } else {
        current = current.right;
      }
    }

    return false;
  }

  /**
   * 删除
   */
  delete(val: number): void {
    this.root = this.deleteNode(this.root, val);
  }

  private deleteNode(node: BSTNode | null, val: number): BSTNode | null {
    if (!node) return null;

    if (val < node.val) {
      node.left = this.deleteNode(node.left, val);
      return node;
    }

    if (val > node.val) {
      node.right = this.deleteNode(node.right, val);
      return node;
    }

    // 找到要删除的节点
    if (!node.left) return node.right;
    if (!node.right) return node.left;

    // 有两个子节点，找到右子树的最小节点
    const minNode = this.findMin(node.right);
    node.val = minNode.val;
    node.right = this.deleteNode(node.right, minNode.val);
    return node;
  }

  private findMin(node: BSTNode): BSTNode {
    while (node.left) {
      node = node.left;
    }
    return node;
  }

  /**
   * 中序遍历（得到有序序列）
   */
  inorder(): number[] {
    const result: number[] = [];
    this.inorderHelper(this.root, result);
    return result;
  }

  private inorderHelper(node: BSTNode | null, result: number[]): void {
    if (!node) return;
    this.inorderHelper(node.left, result);
    result.push(node.val);
    this.inorderHelper(node.right, result);
  }

  /**
   * 前序遍历
   */
  preorder(): number[] {
    const result: number[] = [];
    this.preorderHelper(this.root, result);
    return result;
  }

  private preorderHelper(node: BSTNode | null, result: number[]): void {
    if (!node) return;
    result.push(node.val);
    this.preorderHelper(node.left, result);
    this.preorderHelper(node.right, result);
  }

  /**
   * 后序遍历
   */
  postorder(): number[] {
    const result: number[] = [];
    this.postorderHelper(this.root, result);
    return result;
  }

  private postorderHelper(node: BSTNode | null, result: number[]): void {
    if (!node) return;
    this.postorderHelper(node.left, result);
    this.postorderHelper(node.right, result);
    result.push(node.val);
  }
}
```

### 2.2 BST的变体操作

```typescript
/**
 * BST相关操作
 */

// 找到第k小的元素
function kthSmallest(root: BSTNode | null, k: number): number {
  const result: number[] = [];
  inorder(root, result);
  return result[k - 1];
}

function inorder(node: BSTNode | null, result: number[]): void {
  if (!node) return;
  inorder(node.left, result);
  result.push(node.val);
  inorder(node.right, result);
}

// 验证BST
function isValidBST(root: BSTNode | null): boolean {
  function validate(node: BSTNode | null, min: number, max: number): boolean {
    if (!node) return true;
    if (node.val <= min || node.val >= max) return false;
    return validate(node.left, min, node.val) && validate(node.right, node.val, max);
  }

  return validate(root, -Infinity, Infinity);
}

// BST的最近公共祖先
function lowestCommonAncestorBST(
  root: BSTNode,
  p: BSTNode,
  q: BSTNode
): BSTNode {
  while (root) {
    if (p.val < root.val && q.val < root.val) {
      root = root.left;
    } else if (p.val > root.val && q.val > root.val) {
      root = root.right;
    } else {
      return root;
    }
  }
  return root!;
}
```

## 三、平衡二叉树（AVL）

### 3.1 AVL树原理

**AVL树** 是一种自平衡的二叉搜索树，任何节点的两个子树高度差（平衡因子）最多为1。

**平衡因子：** 左子树高度 - 右子树高度

**四种旋转操作：**
1. **左旋（Left Rotation）：** 用于处理右右失衡
2. **右旋（Right Rotation）：** 用于处理左左失衡
3. **左右旋（Left-Right Rotation）：** 用于处理左右失衡
4. **右左旋（Right-Left Rotation）：** 用于处理右左失衡

```typescript
/**
 * AVL树实现
 */

interface AVLNode {
  val: number;
  height: number;
  left: AVLNode | null;
  right: AVLNode | null;
}

class AVLTree {
  private root: AVLNode | null = null;

  /**
   * 获取高度
   */
  private height(node: AVLNode | null): number {
    return node ? node.height : 0;
  }

  /**
   * 获取平衡因子
   */
  private getBalance(node: AVLNode | null): number {
    return node ? this.height(node.left) - this.height(node.right) : 0;
  }

  /**
   * 右旋
   */
  private rightRotate(y: AVLNode): AVLNode {
    const x = y.left!;
    const t2 = x.right;

    x.right = y;
    y.left = t2;

    y.height = Math.max(this.height(y.left), this.height(y.right)) + 1;
    x.height = Math.max(this.height(x.left), this.height(x.right)) + 1;

    return x;
  }

  /**
   * 左旋
   */
  private leftRotate(x: AVLNode): AVLNode {
    const y = x.right!;
    const t2 = y.left;

    y.left = x;
    x.right = t2;

    x.height = Math.max(this.height(x.left), this.height(x.right)) + 1;
    y.height = Math.max(this.height(y.left), this.height(y.right)) + 1;

    return y;
  }

  /**
   * 插入
   */
  insert(val: number): void {
    this.root = this.insertNode(this.root, val);
  }

  private insertNode(node: AVLNode | null, val: number): AVLNode {
    // 普通BST插入
    if (!node) {
      return {val, height: 1, left: null, right: null};
    }

    if (val < node.val) {
      node.left = this.insertNode(node.left, val);
    } else if (val > node.val) {
      node.right = this.insertNode(node.right, val);
    } else {
      return node;  // 不允许重复
    }

    // 更新高度
    node.height = Math.max(this.height(node.left), this.height(node.right)) + 1;

    // 获取平衡因子，进行旋转
    const balance = this.getBalance(node);

    // 左左失衡 - 右旋
    if (balance > 1 && val < node.left!.val) {
      return this.rightRotate(node);
    }

    // 右右失衡 - 左旋
    if (balance < -1 && val > node.right!.val) {
      return this.leftRotate(node);
    }

    // 左右失衡 - 先左旋后右旋
    if (balance > 1 && val > node.left!.val) {
      node.left = this.leftRotate(node.left!);
      return this.rightRotate(node);
    }

    // 右左失衡 - 先右旋后左旋
    if (balance < -1 && val < node.right!.val) {
      node.right = this.rightRotate(node.right!);
      return this.leftRotate(node);
    }

    return node;
  }

  /**
   * 删除
   */
  delete(val: number): void {
    this.root = this.deleteNode(this.root, val);
  }

  private deleteNode(node: AVLNode | null, val: number): AVLNode | null {
    if (!node) return null;

    if (val < node.val) {
      node.left = this.deleteNode(node.left, val);
    } else if (val > node.val) {
      node.right = this.deleteNode(node.right, val);
    } else {
      if (!node.left || !node.right) {
        return node.left || node.right;
      }

      const temp = this.findMin(node.right);
      node.val = temp.val;
      node.right = this.deleteNode(node.right, temp.val);
    }

    if (!node) return null;

    node.height = Math.max(this.height(node.left), this.height(node.right)) + 1;

    const balance = this.getBalance(node);

    // 与插入类似的旋转操作
    if (balance > 1 && this.getBalance(node.left) >= 0) {
      return this.rightRotate(node);
    }

    if (balance > 1 && this.getBalance(node.left) < 0) {
      node.left = this.leftRotate(node.left!);
      return this.rightRotate(node);
    }

    if (balance < -1 && this.getBalance(node.right) <= 0) {
      return this.leftRotate(node);
    }

    if (balance < -1 && this.getBalance(node.right) > 0) {
      node.right = this.rightRotate(node.right!);
      return this.leftRotate(node);
    }

    return node;
  }

  private findMin(node: AVLNode): AVLNode {
    while (node.left) {
      node = node.left;
    }
    return node;
  }
}
```

## 四、红黑树

### 4.1 红黑树性质

**红黑树** 是一种自平衡的二叉搜索树，具有以下性质：
1. 每个节点非红即黑
2. 根节点是黑色
3. 每个叶节点（NIL）是黑色
4. 红节点的子节点必须是黑色
5. 从任一节点到其每个叶子的路径上，黑节点数量相同

**性质保证：** 红黑树的高度最多是 2*log(n+1)，因此搜索、插入、删除操作都是 O(log n)

```typescript
/**
 * 红黑树实现（简化版）
 */

enum Color {
  RED,
  BLACK
}

interface RBNode {
  val: number;
  color: Color;
  left: RBNode | null;
  right: RBNode | null;
  parent: RBNode | null;
}

class RedBlackTree {
  private NIL: RBNode;
  private root: RBNode;

  constructor() {
    this.NIL = {
      val: 0,
      color: Color.BLACK,
      left: null,
      right: null,
      parent: null
    };
    this.root = this.NIL;
  }

  /**
   * 左旋
   */
  private leftRotate(x: RBNode): void {
    const y = x.right!;
    x.right = y.left;

    if (y.left !== this.NIL) {
      y.left!.parent = x;
    }

    y.parent = x.parent;

    if (x.parent === this.NIL) {
      this.root = y;
    } else if (x === x.parent.left) {
      x.parent.left = y;
    } else {
      x.parent.right = y;
    }

    y.left = x;
    x.parent = y;
  }

  /**
   * 右旋
   */
  private rightRotate(x: RBNode): void {
    const y = x.left!;
    x.left = y.right;

    if (y.right !== this.NIL) {
      y.right!.parent = x;
    }

    y.parent = x.parent;

    if (x.parent === this.NIL) {
      this.root = y;
    } else if (x === x.parent.right) {
      x.parent.right = y;
    } else {
      x.parent.left = y;
    }

    y.right = x;
    x.parent = y;
  }

  /**
   * 插入
   */
  insert(val: number): void {
    const node: RBNode = {
      val,
      color: Color.RED,
      left: this.NIL,
      right: this.NIL,
      parent: this.NIL
    };

    let y = this.NIL;
    let x = this.root;

    while (x !== this.NIL) {
      y = x;
      if (node.val < x.val) {
        x = x.left;
      } else {
        x = x.right;
      }
    }

    node.parent = y;

    if (y === this.NIL) {
      this.root = node;
    } else if (node.val < y.val) {
      y.left = node;
    } else {
      y.right = node;
    }

    this.insertFixup(node);
  }

  /**
   * 插入修复
   */
  private insertFixup(node: RBNode): void {
    while (node.parent.color === Color.RED) {
      if (node.parent === node.parent.parent?.left) {
        const uncle = node.parent.parent.right;

        if (uncle?.color === Color.RED) {
          node.parent.color = Color.BLACK;
          uncle.color = Color.BLACK;
          node.parent.parent.color = Color.RED;
          node = node.parent.parent;
        } else {
          if (node === node.parent.right) {
            node = node.parent;
            this.leftRotate(node);
          }
          node.parent.color = Color.BLACK;
          node.parent.parent.color = Color.RED;
          this.rightRotate(node.parent.parent);
        }
      } else {
        const uncle = node.parent.parent?.left;

        if (uncle?.color === Color.RED) {
          node.parent.color = Color.BLACK;
          uncle.color = Color.BLACK;
          node.parent.parent.color = Color.RED;
          node = node.parent.parent;
        } else {
          if (node === node.parent.left) {
            node = node.parent;
            this.rightRotate(node);
          }
          node.parent.color = Color.BLACK;
          node.parent.parent.color = Color.RED;
          this.leftRotate(node.parent.parent);
        }
      }
    }

    this.root.color = Color.BLACK;
  }

  /**
   * 搜索
   */
  search(val: number): boolean {
    let current = this.root;

    while (current !== this.NIL) {
      if (val === current.val) return true;
      if (val < current.val) {
        current = current.left;
      } else {
        current = current.right;
      }
    }

    return false;
  }
}
```

## 五、线段树

### 5.1 线段树原理

**线段树（Segment Tree）** 是一种用于处理区间查询和区间修改的数据结构。

**类比理解：** 想象你有一个数组，需要经常查询某个区间的和、最大值、最小值等。线段树就是把这些区间信息组织成一棵树，每个节点存储一个区间的聚合信息。

```typescript
/**
 * 线段树实现
 *
 * 支持操作：
 * - 点更新
 * - 区间查询
 * - 区间修改
 *
 * 时间复杂度：O(log n)
 * 空间复杂度：O(n)
 */

class SegmentTree {
  private tree: number[];
  private n: number;
  private operation: (a: number, b: number) => number;
  private defaultValue: number;

  /**
   * @param arr 原始数组
   * @param operation 区间合并操作（如 sum, max, min）
   * @param defaultValue 单位元（如 sum 为 0，max 为 -Infinity）
   */
  constructor(
    arr: number[],
    operation: (a: number, b: number) => number = (a, b) => a + b,
    defaultValue: number = 0
  ) {
    this.n = arr.length;
    this.operation = operation;
    this.defaultValue = defaultValue;
    this.tree = new Array(4 * this.n).fill(defaultValue);

    if (this.n > 0) {
      this.build(arr, 0, 0, this.n - 1);
    }
  }

  /**
   * 构建线段树
   */
  private build(arr: number[], node: number, start: number, end: number): void {
    if (start === end) {
      this.tree[node] = arr[start];
      return;
    }

    const mid = Math.floor((start + end) / 2);
    this.build(arr, 2 * node + 1, start, mid);
    this.build(arr, 2 * node + 2, mid + 1, end);
    this.tree[node] = this.operation(
      this.tree[2 * node + 1],
      this.tree[2 * node + 2]
    );
  }

  /**
   * 点更新
   */
  update(idx: number, value: number): void {
    this.updateHelper(0, 0, this.n - 1, idx, value);
  }

  private updateHelper(
    node: number,
    start: number,
    end: number,
    idx: number,
    value: number
  ): void {
    if (start === end) {
      this.tree[node] = value;
      return;
    }

    const mid = Math.floor((start + end) / 2);
    if (idx <= mid) {
      this.updateHelper(2 * node + 1, start, mid, idx, value);
    } else {
      this.updateHelper(2 * node + 2, mid + 1, end, idx, value);
    }

    this.tree[node] = this.operation(
      this.tree[2 * node + 1],
      this.tree[2 * node + 2]
    );
  }

  /**
   * 区间查询
   */
  query(left: number, right: number): number {
    return this.queryHelper(0, 0, this.n - 1, left, right);
  }

  private queryHelper(
    node: number,
    start: number,
    end: number,
    left: number,
    right: number
  ): number {
    // 完全在查询范围内
    if (left <= start && end <= right) {
      return this.tree[node];
    }

    // 完全不在查询范围内
    if (right < start || left > end) {
      return this.defaultValue;
    }

    // 部分重叠
    const mid = Math.floor((start + end) / 2);
    const leftResult = this.queryHelper(2 * node + 1, start, mid, left, right);
    const rightResult = this.queryHelper(2 * node + 2, mid + 1, end, left, right);
    return this.operation(leftResult, rightResult);
  }
}

/**
 * 线段树应用：范围求和
 */
function segmentTreeSum(): void {
  const arr = [1, 3, 5, 7, 9, 11];
  const segTree = new SegmentTree(arr, (a, b) => a + b, 0);

  console.log("区间 [1,3] 的和:", segTree.query(1, 3)); // 3+5+7=15
  console.log("区间 [0,5] 的和:", segTree.query(0, 5)); // 1+3+5+7+9+11=36

  segTree.update(2, 10); // 将 arr[2] 从 5 改成 10
  console.log("更新后区间 [1,3] 的和:", segTree.query(1, 3)); // 3+10+7=20
}

/**
 * 线段树应用：范围最小值
 */
function segmentTreeMin(): void {
  const arr = [5, 2, 8, 1, 9];
  const segTree = new SegmentTree(arr, Math.min, Infinity);

  console.log("区间 [0,2] 的最小值:", segTree.query(0, 2)); // min(5,2,8)=2
  console.log("区间 [1,4] 的最小值:", segTree.query(1, 4)); // min(2,1,9)=1
}
```

## 六、树状数组（Fenwick Tree）

### 6.1 树状数组原理

**树状数组（Binary Indexed Tree / Fenwick Tree）** 是另一种用于处理区间查询和点更新的数据结构，比线段树更简洁。

**核心思想：** 利用二进制表示，将数组组织成树形结构。

```typescript
/**
 * 树状数组（Fenwick Tree）
 *
 * 时间复杂度：
 * - 点更新：O(log n)
 * - 前缀查询：O(log n)
 * - 区间查询：O(log n)
 *
 * 空间复杂度：O(n)
 */

class FenwickTree {
  private tree: number[];
  private n: number;

  constructor(n: number) {
    this.n = n;
    this.tree = new Array(n + 1).fill(0);
  }

  /**
   * 点更新：将 index 位置的值增加 delta
   */
  update(index: number, delta: number): void {
    for (let i = index + 1; i <= this.n; i += i & -i) {
      this.tree[i] += delta;
    }
  }

  /**
   * 前缀和：计算 [0, index] 的和
   */
  prefixSum(index: number): number {
    let sum = 0;
    for (let i = index + 1; i > 0; i -= i & -i) {
      sum += this.tree[i];
    }
    return sum;
  }

  /**
   * 区间和：计算 [left, right] 的和
   */
  rangeSum(left: number, right: number): number {
    if (left === 0) {
      return this.prefixSum(right);
    }
    return this.prefixSum(right) - this.prefixSum(left - 1);
  }

  /**
   * 构造：直接从数组构建
   */
  static fromArray(arr: number[]): FenwickTree {
    const tree = new FenwickTree(arr.length);
    for (let i = 0; i < arr.length; i++) {
      tree.update(i, arr[i]);
    }
    return tree;
  }
}

/**
 * 树状数组应用：动态频率统计
 */
function fenwickTreeExample(): void {
  const freq = new FenwickTree(10);

  // 初始化一些频率
  freq.update(0, 5);  // 位置0出现5次
  freq.update(3, 2); // 位置3出现2次
  freq.update(5, 7); // 位置5出现7次

  console.log("位置0-3的出现次数:", freq.prefixSum(3)); // 5+2=7
  console.log("位置0-5的出现次数:", freq.prefixSum(5)); // 5+2+7=14
  console.log("位置3-5的出现次数:", freq.rangeSum(3, 5)); // 2+7=9
}
```

```typescript
/**
 * 二叉树的BFS和DFS遍历
 */

interface TreeNode {
  val: number;
  left: TreeNode | null;
  right: TreeNode | null;
}

/**
 * BFS层序遍历
 */
function levelOrder(root: TreeNode | null): number[][] {
  const result: number[][] = [];
  if (!root) return result;

  const queue: TreeNode[] = [root];

  while (queue.length > 0) {
    const levelSize = queue.length;
    const currentLevel: number[] = [];

    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift()!;
      currentLevel.push(node.val);

      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }

    result.push(currentLevel);
  }

  return result;
}

/**
 * BFS层序遍历（每层单独一个数组）
 */
function levelOrderSingleLevels(root: TreeNode | null): number[] {
  const result: number[] = [];
  if (!root) return result;

  const queue: TreeNode[] = [root];

  while (queue.length > 0) {
    const node = queue.shift()!;
    result.push(node.val);

    if (node.left) queue.push(node.left);
    if (node.right) queue.push(node.right);
  }

  return result;
}

/**
 * DFS前序遍历（递归）
 */
function preorderTraversal(root: TreeNode | null): number[] {
  const result: number[] = [];
  preorder(root, result);
  return result;
}

function preorder(node: TreeNode | null, result: number[]): void {
  if (!node) return;
  result.push(node.val);
  preorder(node.left, result);
  preorder(node.right, result);
}

/**
 * DFS中序遍历（递归）
 */
function inorderTraversal(root: TreeNode | null): number[] {
  const result: number[] = [];
  inorder(root, result);
  return result;
}

function inorder(node: TreeNode | null, result: number[]): void {
  if (!node) return;
  inorder(node.left, result);
  result.push(node.val);
  inorder(node.right, result);
}

/**
 * DFS后序遍历（递归）
 */
function postorderTraversal(root: TreeNode | null): number[] {
  const result: number[] = [];
  postorder(root, result);
  return result;
}

function postorder(node: TreeNode | null, result: number[]): void {
  if (!node) return;
  postorder(node.left, result);
  postorder(node.right, result);
  result.push(node.val);
}

/**
 * DFS前序遍历（迭代）
 */
function preorderIterative(root: TreeNode | null): number[] {
  const result: number[] = [];
  if (!root) return result;

  const stack: TreeNode[] = [root];

  while (stack.length > 0) {
    const node = stack.pop()!;
    result.push(node.val);

    // 先压右子树，再压左子树
    if (node.right) stack.push(node.right);
    if (node.left) stack.push(node.left);
  }

  return result;
}

/**
 * DFS中序遍历（迭代）
 */
function inorderIterative(root: TreeNode | null): number[] {
  const result: number[] = [];
  const stack: TreeNode[] = [];
  let current = root;

  while (current || stack.length > 0) {
    while (current) {
      stack.push(current);
      current = current.left;
    }

    current = stack.pop()!;
    result.push(current.val);
    current = current.right;
  }

  return result;
}

/**
 * DFS后序遍历（迭代）
 */
function postorderIterative(root: TreeNode | null): number[] {
  const result: number[] = [];
  if (!root) return result;

  const stack1: TreeNode[] = [root];
  const stack2: TreeNode[] = [];

  while (stack1.length > 0) {
    const node = stack1.pop()!;
    stack2.push(node);

    if (node.left) stack1.push(node.left);
    if (node.right) stack1.push(node.right);
  }

  while (stack2.length > 0) {
    result.push(stack2.pop()!.val);
  }

  return result;
}
```

```typescript
/**

// 二叉树最大深度
function maxDepth(root: TreeNode | null): number {
  if (!root) return 0;
  return 1 + Math.max(maxDepth(root.left), maxDepth(root.right));
}

// 二叉树最小深度
function minDepth(root: TreeNode | null): number {
  if (!root) return 0;
  if (!root.left) return 1 + minDepth(root.right);
  if (!root.right) return 1 + minDepth(root.left);
  return 1 + Math.min(minDepth(root.left), minDepth(root.right));
}

// 判断是否平衡二叉树
function isBalanced(root: TreeNode | null): boolean {
  return checkBalance(root) !== -1;
}

function checkBalance(node: TreeNode | null): number {
  if (!node) return 0;

  const left = checkBalance(node.left);
  if (left === -1) return -1;

  const right = checkBalance(node.right);
  if (right === -1) return -1;

  if (Math.abs(left - right) > 1) return -1;

  return Math.max(left, right) + 1;
}

// 二叉树的最近公共祖先
function lowestCommonAncestor(
  root: TreeNode,
  p: TreeNode,
  q: TreeNode
): TreeNode {
  if (!root || root === p || root === q) return root;

  const left = lowestCommonAncestor(root.left!, p, q);
  const right = lowestCommonAncestor(root.right!, p, q);

  if (!left) return right;
  if (!right) return left;

  return root;
}

// 二叉树直径（任意两个节点最长路径）
function diameterOfBinaryTree(root: TreeNode | null): number {
  let diameter = 0;

  function depth(node: TreeNode | null): number {
    if (!node) return 0;

    const left = depth(node.left);
    const right = depth(node.right);

    diameter = Math.max(diameter, left + right);

    return Math.max(left, right) + 1;
  }

  depth(root);
  return diameter;
}

// 验证对称二叉树
function isSymmetric(root: TreeNode | null): boolean {
  if (!root) return true;
  return isMirror(root.left, root.right);
}

function isMirror(t1: TreeNode | null, t2: TreeNode | null): boolean {
  if (!t1 && !t2) return true;
  if (!t1 || !t2) return false;
  return (
    t1.val === t2.val &&
    isMirror(t1.left, t2.right) &&
    isMirror(t1.right, t2.left)
  );
}

// 路径总和
function hasPathSum(root: TreeNode | null, targetSum: number): boolean {
  if (!root) return false;

  if (!root.left && !root.right) {
    return targetSum === root.val;
  }

  return (
    hasPathSum(root.left, targetSum - root.val) ||
    hasPathSum(root.right, targetSum - root.val)
  );
}

// 二叉树展开为链表
function flatten(root: TreeNode | null): void {
  if (!root) return;

  const stack: TreeNode[] = [root];

  while (stack.length > 0) {
    const node = stack.pop()!;

    if (node.right) stack.push(node.right);
    if (node.left) stack.push(node.left);

    node.left = null;
    node.right = stack.length > 0 ? stack[stack.length - 1] : null;
  }
}
```

## 九、总结

### 9.1 树结构核心要点

```
树结构算法要点

二叉搜索树（BST）：
- 左 < 根 < 右
- 搜索/插入/删除：平均O(log n)
- 有序遍历：O(n)

平衡树（AVL/红黑树）：
- 自平衡，保证O(log n)
- AVL：严格平衡（高度差<=1）
- 红黑树：颜色约束，近似平衡

线段树：
- 区间查询/修改
- O(log n)
- 空间O(n)

树状数组：
- 点更新 + 前缀查询
- 比线段树更简洁
- O(log n)

树的遍历：
- BFS：层序，使用队列
- DFS：前/中/后序，使用栈或递归
```

### 9.2 数据结构选择指南

```
问题类型 → 推荐数据结构

只做搜索/插入/删除：
- BST（数据有序）

需要维护平衡：
- AVL（读多写少）
- 红黑树（读写均衡）

区间查询/修改：
- 线段树（通用）
- 树状数组（简单前缀）

前缀/后缀统计：
- 树状数组

字符串前缀问题：
- Trie树
```
