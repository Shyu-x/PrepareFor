# 递归与迭代转换

## 一、递归与迭代基础

### 1.1 递归与迭代的概念

```
递归（Recursion）：
- 函数调用自身
- 将大问题分解为小问题
- 包含基准情形和递归情形

迭代（Iteration）：
- 使用循环重复执行代码
- 显式维护循环变量
- 不涉及函数调用自身
```

### 1.2 递归的组成

```typescript
/**
 * 递归函数的三要素
 */
function recursiveFunction(n: number): number {
  // 1. 基准情形（Base Case）：递归终止条件
  if (n <= 1) {
    return n;
  }

  // 2. 递归情形（Recursive Case）：分解问题
  const smallerProblem = n - 1;

  // 3. 组合：将子问题的解组合成原问题的解
  return smallerProblem + 1;
}
```

### 1.3 递归 vs 迭代对比

```typescript
/**
 * 递归计算阶乘
 */
function factorialRecursive(n: number): number {
  // 基准情形
  if (n <= 1) return 1;

  // 递归调用
  return n * factorialRecursive(n - 1);
}

/**
 * 迭代计算阶乘
 */
function factorialIterative(n: number): number {
  let result = 1;
  for (let i = 2; i <= n; i++) {
    result *= i;
  }
  return result;
}

/**
 * 递归计算斐波那契数列
 */
function fibRecursive(n: number): number {
  if (n <= 1) return n;
  return fibRecursive(n - 1) + fibRecursive(n - 2);
}

/**
 * 迭代计算斐波那契数列
 */
function fibIterative(n: number): number {
  if (n <= 1) return n;
  let prev = 0, curr = 1;
  for (let i = 2; i <= n; i++) {
    const next = prev + curr;
    prev = curr;
    curr = next;
  }
  return curr;
}
```

## 二、递归改迭代的基本方法

### 2.1 方法一：使用显式栈模拟递归

```typescript
/**
 * 递归改迭代的核心方法：使用栈模拟系统调用栈
 *
 * 场景：简单的递归函数
 * 原理：将递归调用的参数和返回地址压入栈中
 */

/**
 * 递归版本：二叉树前序遍历
 */
function preorderTraversalRecursive(root: TreeNode | null): number[] {
  const result: number[] = [];

  function traverse(node: TreeNode | null) {
    if (!node) return;
    result.push(node.val);     // 访问根
    traverse(node.left);       // 遍历左
    traverse(node.right);      // 遍历右
  }

  traverse(root);
  return result;
}

/**
 * 迭代版本：使用栈模拟
 */
function preorderTraversalIterative(root: TreeNode | null): number[] {
  const result: number[] = [];
  const stack: TreeNode[] = [];

  if (!root) return result;

  stack.push(root);

  while (stack.length > 0) {
    const node = stack.pop()!;
    result.push(node.val);

    // 先压右子树，再压左子树（出栈时先访问左子树）
    if (node.right) stack.push(node.right);
    if (node.left) stack.push(node.left);
  }

  return result;
}

/**
 * 二叉树节点定义
 */
class TreeNode {
  val: number;
  left: TreeNode | null;
  right: TreeNode | null;
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val;
    this.left = left === undefined ? null : left;
    this.right = right === undefined ? null : right;
  }
}

// 测试
const root = new TreeNode(1);
root.left = new TreeNode(2);
root.right = new TreeNode(3);
root.left.left = new TreeNode(4);
root.left.right = new TreeNode(5);

console.log(preorderTraversalRecursive(root)); // [1, 2, 4, 5, 3]
console.log(preorderTraversalIterative(root)); // [1, 2, 4, 5, 3]
```

### 2.2 方法二：使用栈模拟深度优先搜索

```typescript
/**
 * 中序遍历递归改迭代
 */
function inorderTraversalIterative(root: TreeNode | null): number[] {
  const result: number[] = [];
  const stack: TreeNode[] = [];
  let current = root;

  while (current || stack.length > 0) {
    // 先遍历到最左节点
    while (current) {
      stack.push(current);
      current = current.left;
    }

    // 弹出栈顶节点
    current = stack.pop()!;
    result.push(current.val);

    // 转向右子树
    current = current.right;
  }

  return result;
}

/**
 * 后序遍历递归改迭代
 */
function postorderTraversalIterative(root: TreeNode | null): number[] {
  const result: number[] = [];
  const stack: TreeNode[] = [];
  let lastVisited: TreeNode | null = null;

  if (!root) return result;

  stack.push(root);

  while (stack.length > 0) {
    const node = stack[stack.length - 1];

    // 如果是叶子节点或子节点都已访问
    if (!node.left && !node.right ||
        (lastVisited && (node.left === lastVisited || node.right === lastVisited))) {
      result.push(node.val);
      stack.pop();
      lastVisited = node;
    } else {
      // 先压右子树，再压左子树
      if (node.right) stack.push(node.right);
      if (node.left) stack.push(node.left);
    }
  }

  return result;
}
```

### 2.3 方法三：广度优先搜索改迭代

```typescript
/**
 * 层序遍历（使用队列）
 */
function levelOrderTraversal(root: TreeNode | null): number[][] {
  const result: number[][] = [];
  const queue: TreeNode[] = [];

  if (!root) return result;

  queue.push(root);

  while (queue.length > 0) {
    const levelSize = queue.length;
    const currentLevel: number[] = [];

    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift()!;
      currentLevel.push(node.val);

      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }

    result.push(currentLevel);
  }

  return result;
}
```

## 三、尾递归优化

### 3.1 什么是尾递归？

```typescript
/**
 * 尾递归定义：
 * 递归调用是函数的最后一个操作（return 语句中只有递归调用）
 *
 * 普通递归：
 * function fact(n) {
 *   if (n <= 1) return 1;
 *   return n * fact(n - 1);  // 最后一个操作是乘法，不是递归调用
 * }
 *
 * 尾递归：
 * function factTail(n, acc = 1) {
 *   if (n <= 1) return acc;
 *   return factTail(n - 1, n * acc);  // 最后一个操作是递归调用
 * }
 */

/**
 * 普通递归：计算阶乘
 */
function factorialNormal(n: number): number {
  if (n <= 1) return 1;
  return n * factorialNormal(n - 1);
}

/**
 * 尾递归：计算阶乘
 * 编译器可以将其优化为循环
 */
function factorialTail(n: number, acc: number = 1): number {
  if (n <= 1) return acc;
  return factorialTail(n - 1, n * acc);
}

// JavaScript 引擎的尾递归优化（ES6）：
// 现代浏览器和 Node.js 已经支持尾递归优化
// 但需要确保递归调用是 return 语句的唯一操作
```

### 3.2 尾递归转迭代

```typescript
/**
 * 尾递归转换为迭代的通用方法
 *
 * 原理：将递归调用的参数作为迭代变量
 */

/**
 * 尾递归版本：计算阶乘
 */
function factorialTailRec(n: number, acc: number = 1): number {
  if (n <= 1) return acc;
  return factorialTailRec(n - 1, n * acc);
}

/**
 * 转换为迭代版本
 */
function factorialToLoop(n: number): number {
  let acc = 1;
  while (n > 1) {
    acc = acc * n;
    n--;
  }
  return acc;
}

/**
 * 尾递归版本：反转链表
 */
function reverseListTailRec(head: ListNode | null, prev: ListNode | null = null): ListNode | null {
  if (!head) return prev;
  const next = head.next;
  head.next = prev;
  return reverseListTailRec(next, head);
}

/**
 * 转换为迭代版本
 */
function reverseListToLoop(head: ListNode | null): ListNode | null {
  let prev = null;
  let current = head;

  while (current) {
    const next = current.next;
    current.next = prev;
    prev = current;
    current = next;
  }

  return prev;
}

/**
 * 链表节点定义
 */
class ListNode {
  val: number;
  next: ListNode | null;
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val;
    this.next = next === undefined ? null : next;
  }
}
```

## 四、记忆化递归（备忘录）

### 4.1 什么是记忆化递归？

```typescript
/**
 * 记忆化递归（Memoization）
 *
 * 思想：将已计算过的结果存储在缓存中
 *      避免重复计算重叠子问题
 *
 * 本质：自顶向下的动态规划
 */

/**
 * 普通递归：斐波那契数列（指数级时间）
 */
function fibMemoNaive(n: number): number {
  if (n <= 1) return n;
  return fibMemoNaive(n - 1) + fibMemoNaive(n - 2);
}
// fib(5) = fib(4) + fib(3)
//       = (fib(3) + fib(2)) + (fib(2) + fib(1))
//       = ... 大量重复计算

/**
 * 记忆化递归：斐波那契数列（线性时间）
 */
function fibMemo(n: number, memo: Map<number, number> = new Map()): number {
  // 基准情形
  if (n <= 1) return n;

  // 查表：避免重复计算
  if (memo.has(n)) {
    return memo.get(n)!;
  }

  // 计算并存储
  const result = fibMemo(n - 1, memo) + fibMemo(n - 2, memo);
  memo.set(n, result);

  return result;
}

/**
 * 记忆化递归：爬楼梯（LeetCode 70）
 */
function climbStairsMemo(n: number, memo: Map<number, number> = new Map()): number {
  if (n <= 2) return n;

  if (memo.has(n)) {
    return memo.get(n)!;
  }

  const result = climbStairsMemo(n - 1, memo) + climbStairsMemo(n - 2, memo);
  memo.set(n, result);

  return result;
}

/**
 * 记忆化递归：硬币找零（LeetCode 322）
 */
function coinChangeMemo(
  coins: number[],
  amount: number,
  memo: Map<number, number> = new Map()
): number {
  if (amount === 0) return 0;
  if (amount < 0) return -1;

  if (memo.has(amount)) {
    return memo.get(amount)!;
  }

  let minCoins = Infinity;

  for (const coin of coins) {
    const remainder = amount - coin;
    const subProblem = coinChangeMemo(coins, remainder, memo);
    if (subProblem >= 0) {
      minCoins = Math.min(minCoins, subProblem + 1);
    }
  }

  const result = minCoins === Infinity ? -1 : minCoins;
  memo.set(amount, result);

  return result;
}
```

### 4.2 记忆化 vs 动态规划

```typescript
/**
 * 记忆化递归 vs 动态规划
 *
 * 记忆化递归：
 * - 自顶向下
 * - 递归 + 缓存
 * - 适合：需要时才计算
 *
 * 动态规划：
 * - 自底向上
 * - 迭代 + 数组
 * - 适合：预先知道计算顺序
 */

/**
 * 斐波那契：动态规划版本
 */
function fibDP(n: number): number {
  if (n <= 1) return n;

  const dp: number[] = new Array(n + 1).fill(0);
  dp[1] = 1;

  for (let i = 2; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2];
  }

  return dp[n];
}

/**
 * 斐波那契：空间优化
 */
function fibDPOptimized(n: number): number {
  if (n <= 1) return n;

  let prev = 0, curr = 1;

  for (let i = 2; i <= n; i++) {
    const next = prev + curr;
    prev = curr;
    curr = next;
  }

  return curr;
}

// 测试：对比三种方法的效率
console.time("普通递归 fib(40)");
// fibMemoNaive(40); // 太慢，不建议测试
console.timeEnd("普通递归 fib(40)");

console.time("记忆化递归 fib(40)");
fibMemo(40);
console.timeEnd("记忆化递归 fib(40)");

console.time("动态规划 fib(40)");
fibDP(40);
console.timeEnd("动态规划 fib(40)");
```

## 五、复杂递归问题的迭代转换

### 5.1 汉诺塔问题

```typescript
/**
 * LeetCode面试题 08.06. 汉诺塔问题
 *
 * 问题：将 A 柱子上的盘子移动到 C 柱子
 *
 * 递归版本：
 * 1. 将 n-1 个盘子从 A 移动到 B
 * 2. 将最大的盘子从 A 移动到 C
 * 3. 将 n-1 个盘子从 B 移动到 C
 */
function hanotaRecursive(A: number[], B: number[], C: number[]): void {
  const move = (n: number, from: number[], to: number[], via: number[]) => {
    if (n === 1) {
      to.push(from.pop()!);
      return;
    }
    move(n - 1, from, via, to);
    to.push(from.pop()!);
    move(n - 1, via, to, from);
  };

  move(A.length, A, C, B);
}

/**
 * 迭代版本：使用显式栈
 * 关键：记录状态 (n, from, to, via)
 */
function hanotaIterative(A: number[], B: number[], C: number[]): void {
  interface State {
    n: number;
    from: number[];
    to: number[];
    via: number[];
  }

  const stack: State[] = [];
  stack.push({ n: A.length, from: A, to: C, via: B });

  while (stack.length > 0) {
    const state = stack.pop()!;
    const { n, from, to, via } = state;

    if (n === 1) {
      to.push(from.pop()!);
    } else {
      // 逆序压入（模拟递归的执行顺序）
      // 1. move(n-1, via, to, from)
      stack.push({
        n: n - 1,
        from: via,
        to: to,
        via: from
      });
      // 2. 移动最大的盘子
      stack.push({
        n: 1,
        from: from,
        to: to,
        via: via
      });
      // 3. move(n-1, from, via, to)
      stack.push({
        n: n - 1,
        from: from,
        to: via,
        via: to
      });
    }
  }
}

// 测试
const A1 = [3, 2, 1];
const B1: number[] = [];
const C1: number[] = [];
hanotaRecursive(A1, B1, C1);
console.log("递归:", C1); // [3, 2, 1]

const A2 = [3, 2, 1];
const B2: number[] = [];
const C2: number[] = [];
hanotaIterative(A2, B2, C2);
console.log("迭代:", C2); // [3, 2, 1]
```

### 5.2 二叉树的所有路径

```typescript
/**
 * LeetCode 257. 二叉树的所有路径
 *
 * 问题：返回所有从根到叶子的路径
 */

/**
 * 递归版本
 */
function binaryTreePathsRecursive(root: TreeNode | null): string[] {
  const result: string[] = [];

  function dfs(node: TreeNode | null, path: string) {
    if (!node) return;

    path += node.val.toString();

    if (!node.left && !node.right) {
      result.push(path);
    } else {
      path += "->";
      dfs(node.left, path);
      dfs(node.right, path);
    }
  }

  dfs(root, "");
  return result;
}

/**
 * 迭代版本：使用栈
 */
function binaryTreePathsIterative(root: TreeNode | null): string[] {
  const result: string[] = [];

  if (!root) return result;

  interface StackItem {
    node: TreeNode;
    path: string;
  }

  const stack: StackItem[] = [];
  stack.push({ node: root, path: root.val.toString() });

  while (stack.length > 0) {
    const { node, path } = stack.pop()!;

    if (!node.left && !node.right) {
      result.push(path);
    } else {
      if (node.right) {
        stack.push({ node: node.right, path: path + "->" + node.right.val });
      }
      if (node.left) {
        stack.push({ node: node.left, path: path + "->" + node.left.val });
      }
    }
  }

  return result;
}
```

## 六、递归思维训练

### 6.1 如何分析递归问题

```typescript
/**
 * 分析递归问题的步骤：
 *
 * 1. 确定问题的输入和输出
 * 2. 找出最小问题（基准情形）
 * 3. 找出如何将问题分解为子问题
 * 4. 确定子问题和原问题的关系
 * 5. 确定递归调用的顺序
 */

/**
 * 示例1：数组求和
 *
 * 输入：数组 arr
 * 输出：所有元素的和
 *
 * 分解：
 * - 基准：arr.length === 0, 返回 0
 * - 分解：sum(arr) = arr[0] + sum(arr[1:])
 * - 递归：先求剩余元素的和，再加上当前元素
 */
function sumArray(arr: number[]): number {
  if (arr.length === 0) return 0;
  return arr[0] + sumArray(arr.slice(1));
}

/**
 * 示例2：字符串反转
 *
 * 输入：字符串 s
 * 输出：反转后的字符串
 *
 * 分解：
 * - 基准：s.length <= 1, 返回 s
 * - 分解：reverse(s) = reverse(s[1:]) + s[0]
 */
function reverseStringRec(s: string): string {
  if (s.length <= 1) return s;
  return reverseStringRec(s.slice(1)) + s[0];
}

/**
 * 示例3：检查括号匹配
 *
 * 递归版本
 */
function isValidRec(s: string): boolean {
  /**
   * 递归辅助函数
   */
  function helper(s: string, depth: number): string {
    if (s.length === 0) return s;

    if (s[0] === '(') depth++;
    else if (s[0] === ')') depth--;

    if (depth < 0) return ""; // 提前终止

    return s[0] + helper(s.slice(1), depth);
  }

  return helper(s, 0) === "";
}

/**
 * 示例4：幂运算
 *
 * 计算 x^n
 *
 * 分解：
 * - 基准：n === 0, 返回 1
 * - 分解：x^n = x^(n/2) * x^(n/2) （n为偶数）
 *       = x^(n-1/2) * x^(n-1/2) * x （n为奇数）
 */
function power(x: number, n: number): number {
  if (n === 0) return 1;
  if (n < 0) {
    x = 1 / x;
    n = -n;
  }

  if (n === 1) return x;

  if (n % 2 === 0) {
    const half = power(x, n / 2);
    return half * half;
  } else {
    const half = power(x, Math.floor(n / 2));
    return half * half * x;
  }
}
```

## 八、本章小结

本章我们学习了递归与迭代的转换：

1. **递归基础**：
   - 基准情形：递归终止条件
   - 递归情形：分解问题，调用自身
   - 组合：合并子问题的解

2. **递归改迭代方法**：
   - 使用显式栈模拟系统调用栈
   - 广度优先用队列，深度优先用栈

3. **尾递归优化**：
   - 递归调用是 return 的唯一操作
   - 编译器可优化为循环
   - 手动转换为迭代

4. **记忆化递归**：
   - 存储已计算结果避免重复
   - 本质是自顶向下的动态规划
   - 可转为自底向上的动态规划

5. **选择建议**：
   - 问题本身递归定义清晰：用递归
   - 需要避免栈溢出：用迭代
   - 重叠子问题：记忆化或 DP
