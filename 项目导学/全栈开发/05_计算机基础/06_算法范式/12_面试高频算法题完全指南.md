## 二、双指针技术

### 2.1 双指针基础

**双指针（Two Pointers）** 是面试中最常考的技术之一。

**核心思想：** 使用两个指针遍历数组，根据指针指向的元素关系移动指针。

```typescript
/**
 * 双指针技术
 */

// 类型1：对撞指针（左右指针向中间移动）
function twoSumSorted(arr: number[], target: number): [number, number] | null {
  let left = 0;
  let right = arr.length - 1;

  while (left < right) {
    const sum = arr[left] + arr[right];
    if (sum === target) {
      return [left, right];
    } else if (sum < target) {
      left++;
    } else {
      right--;
    }
  }

  return null;
}

// 类型2：快慢指针（快指针先遍历，慢指针跟随）
function removeDuplicates(arr: number[]): number {
  let slow = 0;

  for (let fast = 1; fast < arr.length; fast++) {
    if (arr[fast] !== arr[slow]) {
      slow++;
      arr[slow] = arr[fast];
    }
  }

  return slow + 1;
}

/**
 * 三数之和
 * 给你一个整数数组nums，判断是否存在a + b + c = 0
 */
function threeSum(nums: number[]): number[][] {
  const result: number[][] = [];
  nums.sort((a, b) => a - b);

  for (let i = 0; i < nums.length - 2; i++) {
    if (i > 0 && nums[i] === nums[i - 1]) continue;

    let left = i + 1;
    let right = nums.length - 1;

    while (left < right) {
      const sum = nums[i] + nums[left] + nums[right];

      if (sum === 0) {
        result.push([nums[i], nums[left], nums[right]]);
        left++;
        right--;

        while (left < right && nums[left] === nums[left - 1]) left++;
        while (left < right && nums[right] === nums[right + 1]) right--;
      } else if (sum < 0) {
        left++;
      } else {
        right--;
      }
    }
  }

  return result;
}

/**
 * 接雨水问题
 * 给定 n 个非负整数表示每个宽度为 1 的柱子的高度图，
 * 计算按此排列的柱子，下雨之后能接多少雨水
 */
function trap(height: number[]): number {
  if (height.length <= 2) return 0;

  let left = 0;
  let right = height.length - 1;
  let leftMax = 0;
  let rightMax = 0;
  let water = 0;

  while (left < right) {
    if (height[left] < height[right]) {
      if (height[left] >= leftMax) {
        leftMax = height[left];
      } else {
        water += leftMax - height[left];
      }
      left++;
    } else {
      if (height[right] >= rightMax) {
        rightMax = height[right];
      } else {
        water += rightMax - height[right];
      }
      right--;
    }
  }

  return water;
}
```

### 2.2 滑动窗口

**滑动窗口（Sliding Window）** 是双指针的扩展。

```typescript
/**
 * 滑动窗口技术
 */

// 最大滑动窗口
function maxSlidingWindow(nums: number[], k: number): number[] {
  if (nums.length === 0) return [];

  const result: number[] = [];
  const deque: number[] = [];  // 存储索引

  for (let i = 0; i < nums.length; i++) {
    // 移除超出窗口范围的元素
    while (deque.length > 0 && deque[0] < i - k + 1) {
      deque.shift();
    }

    // 保持 deque 递减（队头是最大值）
    while (deque.length > 0 && nums[deque[deque.length - 1]] < nums[i]) {
      deque.pop();
    }
    deque.push(i);

    // 记录结果
    if (i >= k - 1) {
      result.push(nums[deque[0]]);
    }
  }

  return result;
}

/**
 * 无重复字符的最长子串
 */
function lengthOfLongestSubstring(s: string): number {
  const charIndex = new Map<string, number>();
  let maxLen = 0;
  let left = 0;

  for (let right = 0; right < s.length; right++) {
    const char = s[right];

    if (charIndex.has(char) && charIndex.get(char)! >= left) {
      left = charIndex.get(char)! + 1;
    }

    charIndex.set(char, right);
    maxLen = Math.max(maxLen, right - left + 1);
  }

  return maxLen;
}

/**
 * 最小覆盖子串
 * 给你一个字符串 S，一个字符串 T，
 * 请你找出 S 中涵盖 T 所有字符的最小子串
 */
function minWindow(s: string, t: string): string {
  if (s.length < t.length) return '';

  const need = new Map<string, number>();
  for (const c of t) {
    need.set(c, (need.get(c) || 0) + 1);
  }

  let left = 0;
  let right = 0;
  let count = t.length;
  let minLen = Infinity;
  let minStart = 0;

  while (right < s.length) {
    const char = s[right];
    if (need.has(char)) {
      if (need.get(char)! > 0) {
        count--;
      }
      need.set(char, need.get(char)! - 1);
    }

    while (count === 0) {
      // 更新最小窗口
      if (right - left + 1 < minLen) {
        minLen = right - left + 1;
        minStart = left;
      }

      const leftChar = s[left];
      if (need.has(leftChar)) {
        if (need.get(leftChar)! === 0) {
          count++;
        }
        need.set(leftChar, need.get(leftChar)! + 1);
      }
      left++;
    }

    right++;
  }

  return minLen === Infinity ? '' : s.substring(minStart, minStart + minLen);
}

/**
 * 找到所有字母异位词
 * 给定两个字符串 s 和 p，找到 s 中所有 p 的变位词所在的起始索引
 */
function findAnagrams(s: string, p: string): number[] {
  const result: number[] = [];
  const need = new Map<string, number>();

  for (const c of p) {
    need.set(c, (need.get(c) || 0) + 1);
  }

  let left = 0;
  let right = 0;
  let count = p.length;

  while (right < s.length) {
    const char = s[right];
    if (need.has(char)) {
      if (need.get(char)! > 0) {
        count--;
      }
      need.set(char, need.get(char)! - 1);
    }

    while (count === 0) {
      if (right - left + 1 === p.length) {
        result.push(left);
      }

      const leftChar = s[left];
      if (need.has(leftChar)) {
        if (need.get(leftChar)! === 0) {
          count++;
        }
        need.set(leftChar, need.get(leftChar)! + 1);
      }
      left++;
    }

    right++;
  }

  return result;
}
```

## 三、TopK问题

### 3.1 TopK基础

```typescript
/**
 * TopK问题
 *
 * 问题定义：找到第k大的元素/前k个最大元素
 *
 * 解法：
 * 1. 排序：O(n log n)
 * 2. 堆：O(n log k)
 * 3. 快速选择：O(n)（平均）
 */

// 方法1：使用堆
function findKthLargest(nums: number[], k: number): number {
  // 最小堆
  const heap: number[] = [];

  for (const num of nums) {
    heap.push(num);

    if (heap.length > k) {
      heap.shift();  // 移除堆顶（最小的）
    }
  }

  return heap[0];
}

// 方法2：快速选择（基于快速排序）
function findKthLargestQuickSelect(nums: number[], k: number): number {
  const target = nums.length - k;

  function quickSelect(left: number, right: number): number {
    const pivot = nums[right];
    let i = left;

    for (let j = left; j < right; j++) {
      if (nums[j] <= pivot) {
        [nums[i], nums[j]] = [nums[j], nums[i]];
        i++;
      }
    }

    [nums[i], nums[right]] = [nums[right], nums[i]];

    if (i === target) {
      return nums[i];
    } else if (i < target) {
      return quickSelect(i + 1, right);
    } else {
      return quickSelect(left, i - 1);
    }
  }

  return quickSelect(0, nums.length - 1);
}

/**
 * 前K个高频元素
 */
function topKFrequent(nums: number[], k: number): number[] {
  // 统计频率
  const freq = new Map<number, number>();
  for (const num of nums) {
    freq.set(num, (freq.get(num) || 0) + 1);
  }

  // 桶排序
  const buckets: number[][] = [];
  for (const [num, count] of freq) {
    if (!buckets[count]) buckets[count] = [];
    buckets[count].push(num);
  }

  // 从高频开始取
  const result: number[] = [];
  for (let i = buckets.length - 1; i >= 0 && result.length < k; i--) {
    if (buckets[i]) {
      result.push(...buckets[i]);
    }
  }

  return result;
}
```

### 3.2 LRU与LFU

```typescript
/**
 * LRU缓存
 *
 * 最近最少使用（Least Recently Used）
 * 容量满时，淘汰最久未使用的元素
 */

/**
 * 使用 Map 实现（Map保持插入顺序）
 */
class LRUCache {
  private capacity: number;
  private cache: Map<number, number>;

  constructor(capacity: number) {
    this.capacity = capacity;
    this.cache = new Map();
  }

  get(key: number): number {
    if (!this.cache.has(key)) return -1;

    // 将key移到末尾（表示最近使用）
    const value = this.cache.get(key)!;
    this.cache.delete(key);
    this.cache.set(key, value);

    return value;
  }

  put(key: number, value: number): void {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else if (this.cache.size >= this.capacity) {
      // 删除最久未使用的（Map的第一个元素）
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    this.cache.set(key, value);
  }
}

/**
 * 使用双向链表 + Map 实现（面试要求）
 */
class LRUNode {
  key: number;
  value: number;
  prev: LRUNode | null = null;
  next: LRUNode | null = null;

  constructor(key: number, value: number) {
    this.key = key;
    this.value = value;
  }
}

class LRUCacheLinkedList {
  private capacity: number;
  private map: Map<number, LRUNode>;
  private head: LRUNode;  // 哑头节点
  private tail: LRUNode;  // 哑尾节点

  constructor(capacity: number) {
    this.capacity = capacity;
    this.map = new Map();
    this.head = new LRUNode(0, 0);
    this.tail = new LRUNode(0, 0);
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }

  get(key: number): number {
    if (!this.map.has(key)) return -1;

    const node = this.map.get(key)!;
    this.moveToHead(node);

    return node.value;
  }

  put(key: number, value: number): void {
    if (this.map.has(key)) {
      const node = this.map.get(key)!;
      node.value = value;
      this.moveToHead(node);
    } else {
      const node = new LRUNode(key, value);
      this.map.set(key, node);
      this.addToHead(node);

      if (this.map.size > this.capacity) {
        const tailNode = this.removeTail();
        this.map.delete(tailNode.key);
      }
    }
  }

  private addToHead(node: LRUNode): void {
    node.next = this.head.next;
    node.prev = this.head;
    this.head.next!.prev = node;
    this.head.next = node;
  }

  private removeNode(node: LRUNode): void {
    node.prev!.next = node.next;
    node.next!.prev = node.prev;
  }

  private moveToHead(node: LRUNode): void {
    this.removeNode(node);
    this.addToHead(node);
  }

  private removeTail(): LRUNode {
    const node = this.tail.prev!;
    this.removeNode(node);
    return node;
  }
}

/**
 * LFU缓存
 *
 * 最不经常使用（Least Frequently Used）
 * 容量满时，淘汰使用频率最低的元素
 */

class LFUCache {
  private capacity: number;
  private minFreq: number;
  private keyToNode: Map<number, {freq: number, value: number}>;
  private freqToKeys: Map<number, Set<number>>;

  constructor(capacity: number) {
    this.capacity = capacity;
    this.minFreq = 0;
    this.keyToNode = new Map();
    this.freqToKeys = new Map();
  }

  get(key: number): number {
    if (!this.keyToNode.has(key)) return -1;

    const node = this.keyToNode.get(key)!;
    const freq = node.freq;

    // 从当前频率集合中移除
    this.freqToKeys.get(freq)!.delete(key);
    if (this.freqToKeys.get(freq)!.size === 0) {
      this.freqToKeys.delete(freq);
      if (this.minFreq === freq) {
        this.minFreq++;
      }
    }

    // 增加频率
    node.freq++;
    if (!this.freqToKeys.has(node.freq)) {
      this.freqToKeys.set(node.freq, new Set());
    }
    this.freqToKeys.get(node.freq)!.add(key);

    return node.value;
  }

  put(key: number, value: number): void {
    if (this.capacity === 0) return;

    if (this.keyToNode.has(key)) {
      const node = this.keyToNode.get(key)!;
      this.get(key);  // 先更新频率
      node.value = value;
      return;
    }

    if (this.keyToNode.size >= this.capacity) {
      // 移除频率最低的
      const keysToRemove = this.freqToKeys.get(this.minFreq)!;
      const keyToRemove = keysToRemove.values().next().value;
      keysToRemove.delete(keyToRemove);
      if (keysToRemove.size === 0) {
        this.freqToKeys.delete(this.minFreq);
      }
      this.keyToNode.delete(keyToRemove);
    }

    this.keyToNode.set(key, {freq: 1, value});
    if (!this.freqToKeys.has(1)) {
      this.freqToKeys.set(1, new Set());
    }
    this.freqToKeys.get(1)!.add(key);
    this.minFreq = 1;
  }
}
```

## 四、单调栈

### 4.1 单调栈基础

**单调栈（Monotonic Stack）** 是一种特殊的栈，用于快速找到每个元素最近的更大/更小元素。

```typescript
/**
 * 单调栈
 *
 * 应用场景：
 * - 寻找每个元素右边/左边第一个比它大/小的元素
 * - 去除重复元素
 * - 柱状图中最大矩形
 */

// 基础单调栈：找右边第一个更大的元素
function nextGreaterElement(nums: number[]): number[] {
  const n = nums.length;
  const result = new Array(n).fill(-1);
  const stack: number[] = [];  // 存储索引

  for (let i = 0; i < n; i++) {
    while (stack.length > 0 && nums[stack[stack.length - 1]] < nums[i]) {
      const index = stack.pop()!;
      result[index] = nums[i];
    }
    stack.push(i);
  }

  return result;
}

/**
 * 每日温度
 * 给定每天的温度，返回需要等待多少天才能等到更暖和的一天
 */
function dailyTemperatures(temperatures: number[]): number[] {
  const n = temperatures.length;
  const result = new Array(n).fill(0);
  const stack: number[] = [];

  for (let i = 0; i < n; i++) {
    while (stack.length > 0 &&
           temperatures[stack[stack.length - 1]] < temperatures[i]) {
      const index = stack.pop()!;
      result[index] = i - index;
    }
    stack.push(i);
  }

  return result;
}

/**
 * 柱状图中最大的矩形
 */
function largestRectangleArea(heights: number[]): number {
  const n = heights.length;
  const stack: number[] = [];
  let maxArea = 0;

  for (let i = 0; i <= n; i++) {
    const h = i === n ? 0 : heights[i];

    while (stack.length > 0 && heights[stack[stack.length - 1]] > h) {
      const height = heights[stack.pop()!];
      const width = stack.length === 0 ? i : i - stack[stack.length - 1] - 1;
      maxArea = Math.max(maxArea, height * width);
    }

    stack.push(i);
  }

  return maxArea;
}

/**
 * 字符串解码
 * 如 3[a2[c]] -> accaccacc
 */
function decodeString(s: string): string {
  const stack: {count: number, result: string}[] = [];
  let result = '';
  let count = 0;

  for (const char of s) {
    if (char >= '0' && char <= '9') {
      count = count * 10 + parseInt(char);
    } else if (char === '[') {
      stack.push({count, result});
      result = '';
      count = 0;
    } else if (char === ']') {
      const {count: prevCount, result: prevResult} = stack.pop()!;
      result = prevResult + result.repeat(prevCount);
    } else {
      result += char;
    }
  }

  return result;
}
```

### 5.1 快慢指针

```typescript
/**
 * 快慢指针解决链表问题
 */

interface ListNode {
  val: number;
  next: ListNode | null;
}

/**
 * 判断链表是否有环
 */
function hasCycle(head: ListNode | null): boolean {
  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow!.next;
    fast = fast.next.next;

    if (slow === fast) {
      return true;
    }
  }

  return false;
}

/**
 * 找到环的入口节点
 */
function detectCycle(head: ListNode | null): ListNode | null {
  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow!.next;
    fast = fast.next.next;

    if (slow === fast) {
      // 找到相遇点，从头节点和相遇点同时出发，再次相遇即为入口
      slow = head;
      while (slow !== fast) {
        slow = slow!.next;
        fast = fast!.next;
      }
      return slow;
    }
  }

  return null;
}

/**
 * 找到链表的中间节点
 */
function middleNode(head: ListNode | null): ListNode | null {
  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow!.next;
    fast = fast.next.next;
  }

  return slow;
}

/**
 * 删除链表的倒数第N个节点
 */
function removeNthFromEnd(head: ListNode | null, n: number): ListNode | null {
  const dummy = new ListNode(0, head);
  let slow: ListNode | null = dummy;
  let fast: ListNode | null = dummy;

  // fast先走n+1步
  for (let i = 0; i <= n; i++) {
    fast = fast!.next;
  }

  // 同时移动直到fast到达末尾
  while (fast) {
    slow = slow!.next;
    fast = fast.next;
  }

  // 删除节点
  slow!.next = slow!.next!.next;

  return dummy.next;
}

/**
 * 合并两个有序链表
 */
function mergeTwoLists(
  l1: ListNode | null,
  l2: ListNode | null
): ListNode | null {
  const dummy = new ListNode(0);
  let current = dummy;

  while (l1 && l2) {
    if (l1.val <= l2.val) {
      current.next = l1;
      l1 = l1.next;
    } else {
      current.next = l2;
      l2 = l2.next;
    }
    current = current.next;
  }

  current.next = l1 || l2;

  return dummy.next;
}

/**
 * 反转链表
 */
function reverseList(head: ListNode | null): ListNode | null {
  let prev: ListNode | null = null;
  let current = head;

  while (current) {
    const next = current.next;
    current.next = prev;
    prev = current;
    current = next;
  }

  return prev;
}

/**
 * 反转链表II
 * 反转从位置 left 到 right 的链表节点
 */
function reverseBetween(
  head: ListNode | null,
  left: number,
  right: number
): ListNode | null {
  const dummy = new ListNode(0, head);
  let prev = dummy;

  // 找到left前一个节点
  for (let i = 1; i < left; i++) {
    prev = prev!.next!;
  }

  // 开始反转
  let current = prev!.next;
  for (let i = 0; i < right - left; i++) {
    const next = current!.next;
    current!.next = next!.next;
    next!.next = prev!.next;
    prev!.next = next;
  }

  return dummy.next;
}
```

```typescript
/**
 * 最长回文子串
 */
function longestPalindrome(s: string): string {
  if (s.length <= 1) return s;

  let start = 0;
  let maxLen = 1;

  function expand(left: number, right: number): void {
    while (left >= 0 && right < s.length && s[left] === s[right]) {
      const len = right - left + 1;
      if (len > maxLen) {
        maxLen = len;
        start = left;
      }
      left--;
      right++;
    }
  }

  for (let i = 0; i < s.length; i++) {
    expand(i, i);       // 奇数长度
    expand(i, i + 1);   // 偶数长度
  }

  return s.substring(start, start + maxLen);
}

/**
 * 正则表达式匹配
 */
function isMatch(s: string, p: string): boolean {
  const m = s.length;
  const n = p.length;

  const dp = Array.from({length: m + 1}, () => new Array(n + 1).fill(false));
  dp[0][0] = true;

  // 处理模式开头的*
  for (let j = 1; j <= n; j++) {
    if (p[j - 1] === '*') {
      dp[0][j] = dp[0][j - 2];
    }
  }

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (p[j - 1] === '*') {
        // 匹配0次
        dp[i][j] = dp[i][j - 2];
        // 匹配1次
        if (p[j - 2] === '.' || p[j - 2] === s[i - 1]) {
          dp[i][j] = dp[i][j] || dp[i - 1][j];
        }
      } else if (p[j - 1] === '.' || p[j - 1] === s[i - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      }
    }
  }

  return dp[m][n];
}

/**
 * 字符串相乘
 */
function multiply(num1: string, num2: string): string {
  if (num1 === '0' || num2 === '0') return '0';

  const m = num1.length;
  const n = num2.length;
  const result = new Array(m + n).fill(0);

  for (let i = m - 1; i >= 0; i--) {
    for (let j = n - 1; j >= 0; j--) {
      const mul = (num1[i] - '0') * (num2[j] - '0');
      const p1 = i + j;
      const p2 = i + j + 1;
      const sum = mul + result[p2];

      result[p2] = sum % 10;
      result[p1] += Math.floor(sum / 10);
    }
  }

  // 去除前导零
  let i = 0;
  while (i < result.length && result[i] === 0) i++;

  return result.slice(i).join('');
}

/**
 * 括号生成
 */
function generateParenthesis(n: number): string[] {
  const result: string[] = [];

  function backtrack(current: string, open: number, close: number): void {
    if (current.length === 2 * n) {
      result.push(current);
      return;
    }

    if (open < n) {
      backtrack(current + '(', open + 1, close);
    }

    if (close < open) {
      backtrack(current + ')', open, close + 1);
    }
  }

  backtrack('', 0, 0);
  return result;
}
```

```typescript
// 1. 清晰的变量命名
function findMaxSubArraySum(nums: number[], k: number): number {
  let currentSum = 0;  // 清晰说明用途
  let maxSum = -Infinity;

  for (const num of nums) {
    currentSum = Math.max(num, currentSum + num);
    maxSum = Math.max(maxSum, currentSum);
  }

  return maxSum;

// 2. 适当的注释
function binarySearch(arr: number[], target: number): number {
  let left = 0;
  let right = arr.length - 1;

  while (left <= right) {  // 使用 <= 处理相等情况
    const mid = left + Math.floor((right - left) / 2);  // 防止整数溢出

    if (arr[mid] === target) {
      return mid;
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  return -1;
}

// 3. 错误处理
function divide(a: number, b: number): number {
  if (b === 0) {
    throw new Error('除数不能为0');
  }
  return a / b;
```
