# 双指针与滑动窗口

## 一、双指针概述

### 1.1 什么是双指针？

双指针是一种通过**两个指针协同工作**来遍历数组/字符串的技巧，可以将 O(n²) 的问题优化到 O(n)。

```
传统双循环：O(n²)
for (int i = 0; i < n; i++) {
  for (int j = 0; j < n; j++) {
    // 处理
  }
}

双指针优化：O(n)
int left = 0, right = n - 1;
while (left < right) {
  // 处理
  left++;
  right--;
}
```

### 1.2 双指针类型

```
1. 对撞指针（左右指针）
   - 一个指针从左开始，一个从右开始
   - 相向移动或背向移动
   - 用于有序数组的两数之和、三数之和等问题

2. 快慢指针
   - 快指针每次走两步，慢指针每次走一步
   - 用于检测环、找中点等

3. 滑动窗口
   - 两个指针确定一个窗口
   - 窗口可以扩大或缩小
   - 用于子串/子数组问题
```

## 二、对撞指针

### 2.1 两数之和

```typescript
/**
 * LeetCode 167. 两数之和 II - 输入有序数组
 *
 * 问题：给定已排序数组 numbers，返回两数之和等于目标值的下标
 *       (下标从 1 开始)
 *
 * 思路：对撞指针
 * - left 从左边开始，right 从右边开始
 * - 如果 sum > target，right 左移
 * - 如果 sum < target，left 右移
 * - 如果 sum == target，返回结果
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function twoSumSorted(numbers: number[], target: number): number[] {
  let left = 0;           // 左指针
  let right = numbers.length - 1; // 右指针

  while (left < right) {
    const sum = numbers[left] + numbers[right];

    if (sum === target) {
      // 找到结果（题目要求下标从1开始）
      return [left + 1, right + 1];
    } else if (sum < target) {
      // 太小，左指针右移，增大和
      left++;
    } else {
      // 太大，右指针左移，减小和
      right--;
    }
  }

  return [-1, -1];
}

// 测试
console.log(twoSumSorted([2, 7, 11, 15], 9)); // [1, 2]
console.log(twoSumSorted([2, 3, 4], 6));      // [1, 3]
```

### 2.2 三数之和

```typescript
/**
 * LeetCode 15. 三数之和
 *
 * 问题：找出数组中所有和为0的三元组
 *
 * 思路：排序 + 双指针
 * 1. 先排序数组
 * 2. 固定一个数 a，然后用双指针找 b + c = -a
 * 3. 跳过重复元素避免重复三元组
 *
 * 时间复杂度: O(n²)
 * 空间复杂度: O(1)（不计结果空间）
 */
function threeSum(nums: number[]): number[][] {
  const result: number[][] = [];
  const n = nums.length;

  if (n < 3) return result;

  // 排序
  nums.sort((a, b) => a - b);

  for (let i = 0; i < n - 2; i++) {
    // 剪枝：如果最小的数大于0，三数之和不可能为0
    if (nums[i] > 0) break;

    // 跳过重复元素
    if (i > 0 && nums[i] === nums[i - 1]) continue;

    // 双指针找两数之和
    let left = i + 1;
    let right = n - 1;
    const target = -nums[i];

    while (left < right) {
      const sum = nums[left] + nums[right];

      if (sum === target) {
        result.push([nums[i], nums[left], nums[right]]);

        // 跳过重复元素
        while (left < right && nums[left] === nums[left + 1]) left++;
        while (left < right && nums[right] === nums[right - 1]) right--;

        left++;
        right--;
      } else if (sum < target) {
        left++;
      } else {
        right--;
      }
    }
  }

  return result;
}

// 测试
console.log(threeSum([-1, 0, 1, 2, -1, -4]));
// 输出: [[-1,-1,2], [-1,0,1]]
```

### 2.3 验证回文串

```typescript
/**
 * LeetCode 125. 验证回文串
 *
 * 问题：判断字符串是否是回文串（忽略字母和数字，只看字符）
 *
 * 思路：对撞指针
 * - left 从左开始，right 从右开始
 * - 跳过非字母数字字符
 * - 比较两个字符（忽略大小写）
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function isPalindrome(s: string): boolean {
  let left = 0;
  let right = s.length - 1;

  /**
   * 判断字符是否是字母或数字
   */
  function isAlphanumeric(c: string): boolean {
    return /[a-zA-Z0-9]/.test(c);
  }

  /**
   * 转换为小写
   */
  function toLower(c: string): string {
    return c.toLowerCase();
  }

  while (left < right) {
    // 跳过左边的非字母数字
    while (left < right && !isAlphanumeric(s[left])) {
      left++;
    }

    // 跳过右边的非字母数字
    while (left < right && !isAlphanumeric(s[right])) {
      right--;
    }

    if (left < right) {
      // 比较两个字符
      if (toLower(s[left]) !== toLower(s[right])) {
        return false;
      }
      left++;
      right--;
    }
  }

  return true;
}

// 测试
console.log(isPalindrome("A man, a plan, a canal: Panama")); // true
console.log(isPalindrome("race a car")); // false
```

### 2.4 反转字符串

```typescript
/**
 * LeetCode 344. 反转字符串
 *
 * 问题：原地反转字符串
 *
 * 思路：对撞指针，交换左右字符
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function reverseString(s: string[]): void {
  let left = 0;
  let right = s.length - 1;

  while (left < right) {
    // 交换 s[left] 和 s[right]
    [s[left], s[right]] = [s[right], s[left]];
    left++;
    right--;
  }
}

// 测试
const arr1 = ["h", "e", "l", "l", "o"];
reverseString(arr1);
console.log(arr1); // ["o","l","l","e","h"]
```

## 三、快慢指针

### 3.1 检测环形链表

```typescript
/**
 * LeetCode 141. 环形链表
 *
 * 问题：判断链表是否有环
 *
 * 思路：快慢指针（Floyd 判圈算法）
 * - 慢指针每次走一步，快指针每次走两步
 * - 如果有环，快慢指针最终会相遇
 * - 如果无环，快指针会到达 null
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function hasCycle(head: ListNode | null): boolean {
  if (!head || !head.next) return false;

  let slow = head;      // 慢指针
  let fast = head;      // 快指针

  while (fast && fast.next) {
    slow = slow.next!;      // 慢指针走一步
    fast = fast.next.next; // 快指针走两步

    if (slow === fast) {
      return true; // 相遇，有环
    }
  }

  return false; // 无环
}

/**
 * LeetCode 142. 环形链表 II
 *
 * 问题：返回链表环的起始节点
 *
 * 数学证明：
 * - 设起点到环起始点的距离为 a
 * - 环起始点到相遇点的距离为 b
 * - 相遇点到环起始点的距离为 c
 * - 慢指针走了 a + b，快指针走了 a + b + c + b = a + 2b + c
 * - 快指针速度是慢指针的2倍：a + 2b + c = 2(a + b)
 * - 化简得：a = c
 * - 即：从起点出发的指针和从相遇点出发的指针，会在环起始点相遇
 */
function detectCycle(head: ListNode | null): ListNode | null {
  if (!head || !head.next) return null;

  let slow = head;
  let fast = head;

  // 第一步：检测是否有环
  while (fast && fast.next) {
    slow = slow.next!;
    fast = fast.next.next;
    if (slow === fast) break;
  }

  if (!fast || !fast.next) return null; // 无环

  // 第二步：找环起始点
  slow = head;
  while (slow !== fast) {
    slow = slow.next!;
    fast = fast.next!;
  }

  return slow;
}

// 链表节点定义
class ListNode {
  val: number;
  next: ListNode | null;
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val;
    this.next = next === undefined ? null : next;
  }
}
```

### 3.2 找链表中点

```typescript
/**
 * 找链表中点
 *
 * 思路：快慢指针
 * - 慢指针走一步，快指针走两步
 * - 当快指针到达末尾时，慢指针正好在中点
 *
 * 注意：
 * - 偶数个节点：中点有两个，取前一个还是后一个视情况而定
 */
function middleNode(head: ListNode): ListNode {
  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow.next!;
    fast = fast.next.next;
  }

  // 如果 fast 不为空（fast.next != null），说明是奇数个节点
  // 此时 slow 正好在中点
  return slow;
}

/**
 * LeetCode 876. 链表的中间结点
 */
function middleNode(head: ListNode | null): ListNode | null {
  if (!head) return null;

  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow.next!;
    fast = fast.next.next;
  }

  return slow;
}

/**
 * 找链表中点的前一个节点（偶数个节点时）
 */
function middleNodePrev(head: ListNode): ListNode | null {
  let slow = head;
  let fast = head.next; // 从第二个节点开始

  while (fast && fast.next) {
    slow = slow.next!;
    fast = fast.next.next;
  }

  return slow;
}
```

### 3.3 删除链表的倒数第 N 个节点

```typescript
/**
 * LeetCode 19. 删除链表的倒数第 N 个结点
 *
 * 问题：删除链表的倒数第 n 个节点
 *
 * 思路：快慢指针
 * - 让快指针先走 n 步
 * - 然后快慢指针同时移动
 * - 当快指针到达末尾时，慢指针的下一个节点就是倒数第 n 个
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function removeNthFromEnd(head: ListNode | null, n: number): ListNode | null {
  if (!head) return null;

  // 创建一个哑节点，简化边界处理
  const dummy = new ListNode(0, head);

  let fast: ListNode | null = head;
  let slow: ListNode | null = dummy;

  // 快指针先走 n 步
  for (let i = 0; i < n; i++) {
    if (fast) fast = fast.next;
  }

  // 快慢指针同时移动
  while (fast) {
    slow = slow!.next;
    fast = fast.next;
  }

  // 删除节点
  if (slow && slow.next) {
    slow.next = slow.next.next;
  }

  return dummy.next;
}

// 测试
// 创建链表 1 -> 2 -> 3 -> 4 -> 5
const head = new ListNode(1);
head.next = new ListNode(2);
head.next.next = new ListNode(3);
head.next.next.next = new ListNode(4);
head.next.next.next.next = new ListNode(5);

const result = removeNthFromEnd(head, 2);
// 结果：1 -> 2 -> 3 -> 5
```

## 四、滑动窗口

### 4.1 滑动窗口基础

```
滑动窗口是一种特殊的双指针技巧

原理：
┌─────────────────────────────────────────────────────┐
│  left                                              │
│   │                                               │
│   ▼                                               │
│  ┌────┬────┬────┬────┬────┬────┬────┬────┐        │
│  │ 2  │ 3  │ 1  │ 5  │ 8  │ 4  │ 2  │ 6  │        │
│  └────┴────┴────┴────┴────┴────┴────┴────┘        │
│                ▲                                   │
│                │                                   │
│              right                                  │
│                                                     │
│  窗口范围: [left, right)                           │
│  窗口大小: right - left                             │
└─────────────────────────────────────────────────────┘

基本模板：
left = 0, right = 0
while (right < len) {
  // 扩大窗口
  add arr[right]
  right++

  // 缩小窗口（当满足某个条件时）
  while (condition) {
    // 更新答案
    remove arr[left]
    left++
  }

  // 更新答案（可选）
  update answer
}
```

### 4.2 长度最大的子数组

```typescript
/**
 * LeetCode 713. 乘积小于 K 的子数组
 *
 * 问题：给定正整数数组 nums 和整数 k，
 *       返回元素乘积小于 k 的连续子数组的个数
 *
 * 思路：滑动窗口
 * - 右指针扩大窗口，计算窗口内乘积
 * - 当乘积 >= k 时，收缩左指针
 * - 对于每个 right，窗口内的子数组个数 = right - left + 1
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function numSubarrayProductLessThanK(nums: number[], k: number): number {
  if (k <= 1) return 0; // 特殊情况

  let left = 0;
  let product = 1;
  let count = 0;

  for (let right = 0; right < nums.length; right++) {
    // 扩大窗口
    product *= nums[right];

    // 收缩窗口直到乘积小于 k
    while (product >= k && left <= right) {
      product /= nums[left];
      left++;
    }

    // 计数：以 right 为结尾、乘积小于 k 的子数组个数
    count += right - left + 1;
  }

  return count;
}

// 测试
console.log(numSubarrayProductLessThanK([10, 5, 2, 6], 100)); // 8
console.log(numSubarrayProductLessThanK([1, 2, 3], 0));       // 0
```

### 4.3 最小覆盖子串

```typescript
/**
 * LeetCode 76. 最小覆盖子串
 *
 * 问题：找出 s 中涵盖 t 所有字符的最小子串
 *
 * 思路：滑动窗口
 * 1. 右指针扩大窗口，添加字符
 * 2. 当窗口包含所有 t 的字符时，收缩左指针尝试优化
 * 3. 记录最小窗口
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)（字符集合大小固定）
 */
function minWindow(s: string, t: string): string {
  // 边界情况
  if (s.length < t.length) return "";

  // 需要的字符计数
  const need = new Map<string, number>();
  for (const c of t) {
    need.set(c, (need.get(c) || 0) + 1);
  }

  // 窗口内的字符计数
  const window = new Map<string, number>();

  let left = 0;
  let right = 0;
  let valid = 0; // 已满足的字符种类数
  let minLen = Infinity;
  let minStart = 0;

  while (right < s.length) {
    const c = s[right];
    right++;

    // 扩大窗口
    if (need.has(c)) {
      window.set(c, (window.get(c) || 0) + 1);
      if (window.get(c) === need.get(c)) {
        valid++;
      }
    }

    // 收缩窗口：当窗口包含所有需要的字符时
    while (valid === need.size) {
      // 更新最小窗口
      if (right - left < minLen) {
        minLen = right - left;
        minStart = left;
      }

      const d = s[left];
      left++;

      // 收缩窗口
      if (need.has(d)) {
        if (window.get(d) === need.get(d)) {
          valid--;
        }
        window.set(d, window.get(d)! - 1);
      }
    }
  }

  return minLen === Infinity ? "" : s.substring(minStart, minStart + minLen);
}

// 测试
console.log(minWindow("ADOBECODEBANC", "ABC")); // "BANC"
console.log(minWindow("a", "a"));               // "a"
```

### 4.4 字符串排列

```typescript
/**
 * LeetCode 567. 字符串的排列
 *
 * 问题：判断 s2 是否包含 s1 的排列（s1 的某种排列）
 *
 * 思路：滑动窗口 + 字符计数
 * 1. 用固定长度的滑动窗口（长度为 s1.length）
 * 2. 比较两个窗口的字符计数是否相同
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function checkInclusion(s1: string, s2: string): boolean {
  const n1 = s1.length;
  const n2 = s2.length;

  if (n1 > n2) return false;

  // 字符计数
  const count = new Array(26).fill(0);

  // s1 的字符计数
  for (const c of s1) {
    count[c.charCodeAt(0) - 'a'.charCodeAt(0)]++;
  }

  let left = 0;

  for (let right = 0; right < n2; right++) {
    // 扩大窗口
    const idx = s2[right].charCodeAt(0) - 'a'.charCodeAt(0);
    count[idx]--;

    // 当窗口长度超过 s1 时，收缩左边界
    if (right - left + 1 > n1) {
      const leftIdx = s2[left].charCodeAt(0) - 'a'.charCodeAt(0);
      count[leftIdx]++;
      left++;
    }

    // 检查是否所有计数都为 0
    if (isAllZero(count)) {
      return true;
    }
  }

  return false;
}

function isAllZero(count: number[]): boolean {
  return count.every(c => c === 0);
}

// 测试
console.log(checkInclusion("ab", "eidbaooo")); // true
console.log(checkInclusion("ab", "eidboaoo")); // false
```

### 5.2 单调队列解法

```typescript
/**
 * LeetCode 239. 滑动窗口最大值
 *
 * 思路：单调递减队列
 * - 队列中存储数组元素的索引（而非值）
 * - 队列按值单调递减
 * - 队首始终是当前窗口的最大值
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(k)
 */

/**
 * 单调递减队列
 */
class MonotonicQueue {
  private queue: number[] = []; // 存储索引

  /**
   * 入队：将小于当前元素的值都移除（因为它们不可能成为最大值）
   */
  push(idx: number, nums: number[]) {
    while (this.queue.length > 0 && nums[this.queue[this.queue.length - 1]] <= nums[idx]) {
      this.queue.pop();
    }
    this.queue.push(idx);
  }

  /**
   * 出队：如果队首是过期的索引（不在窗口内），移除
   */
  pop(idx: number) {
    if (this.queue.length > 0 && this.queue[0] === idx) {
      this.queue.shift();
    }
  }

  /**
   * 获取最大值（队首元素的值）
   */
  max(nums: number[]): number {
    return nums[this.queue[0]];
  }

  isEmpty(): boolean {
    return this.queue.length === 0;
  }
}

function maxSlidingWindow(nums: number[], k: number): number[] {
  const result: number[] = [];
  const deque = new MonotonicQueue();

  // 初始化第一个窗口
  for (let i = 0; i < k; i++) {
    deque.push(i, nums);
  }
  result.push(deque.max(nums));

  // 滑动窗口
  for (let i = k; i < nums.length; i++) {
    // 先移除窗口最左边的元素
    deque.pop(i - k);

    // 添加新元素
    deque.push(i, nums);

    // 记录当前窗口的最大值
    result.push(deque.max(nums));
  }

  return result;
}

// 测试
console.log(maxSlidingWindow([1, 3, -1, -3, 5, 3, 6, 7], 3));
// 输出: [3, 3, 5, 5, 6, 7]
```

```typescript
/**
 * LeetCode 862. 和至少为 K 的最短子数组
 *
 * 问题：返回和非小于 k 的最短子数组长度
 *
 * 思路：前缀和 + 单调队列
 * 1. 计算前缀和
 * 2. 使用单调队列找到最小的 j 使得 prefixSum[j] >= prefixSum[i] + k
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(n)
 */
function shortestSubarray(nums: number[], k: number): number {
  const n = nums.length;
  const prefixSum = new Array(n + 1).fill(0);

  // 计算前缀和
  for (let i = 0; i < n; i++) {
    prefixSum[i + 1] = prefixSum[i] + nums[i];
  }

  const deque: number[] = []; // 存储前缀和的索引
  let result = Infinity;

  for (let i = 0; i <= n; i++) {
    // 单调队列：保持 deque 中前缀和递增
    while (deque.length > 0 && prefixSum[deque[deque.length - 1]] >= prefixSum[i]) {
      deque.pop();
    }

    // 查找满足条件的最小 j
    while (deque.length > 0 && prefixSum[i] - prefixSum[deque[0]] >= k) {
      result = Math.min(result, i - deque.shift()!);
    }

    deque.push(i);
  }

  return result === Infinity ? -1 : result;
}

// 测试
console.log(shortestSubarray([1], 1));      // 1
console.log(shortestSubarray([1, 2], 4));   // -1
console.log(shortestSubarray([2, -1, 2], 3)); // 3
```

### 6.2 无重复字符的最长子串

```typescript
/**
 * LeetCode 3. 无重复字符的最长子串
 *
 * 问题：找出不含有重复字符的最长子串的长度
 *
 * 思路：滑动窗口 + 哈希集合
 * - 右指针扩大窗口，添加字符
 * - 如果字符已存在，收缩左指针直到不重复
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(min(m, n))，m 为字符集大小
 */
function lengthOfLongestSubstring(s: string): number {
  const seen = new Set<string>();
  let left = 0;
  let maxLen = 0;

  for (let right = 0; right < s.length; right++) {
    const c = s[right];

    // 收缩左边界直到没有重复
    while (seen.has(c)) {
      seen.delete(s[left]);
      left++;
    }

    // 添加当前字符
    seen.add(c);

    // 更新最大长度
    maxLen = Math.max(maxLen, right - left + 1);
  }

  return maxLen;
}

/**
 * 优化版本：记录字符的最新位置
 */
function lengthOfLongestSubstringOptimized(s: string): number {
  const lastSeen = new Map<string, number>();
  let left = 0;
  let maxLen = 0;

  for (let right = 0; right < s.length; right++) {
    const c = s[right];

    // 如果字符已存在且在当前窗口内，移动左边界
    if (lastSeen.has(c) && lastSeen.get(c)! >= left) {
      left = lastSeen.get(c)! + 1;
    }

    lastSeen.set(c, right);
    maxLen = Math.max(maxLen, right - left + 1);
  }

  return maxLen;
}

// 测试
console.log(lengthOfLongestSubstring("abcabcbb")); // 3 ("abc")
console.log(lengthOfLongestSubstring("bbbbb"));    // 1 ("b")
console.log(lengthOfLongestSubstring("pwwkew"));   // 3 ("wke")
```

## 九、本章小结

本章我们学习了双指针和滑动窗口技巧：

1. **对撞指针**：
   - 左右指针从两端向中间移动
   - 适用于有序数组问题
   - O(n) 时间复杂度

2. **快慢指针**：
   - 快指针走两步，慢指针走一步
   - 适用于链表问题
   - O(n) 时间复杂度

3. **滑动窗口**：
   - 维护一个可变大小的窗口
   - 右指针扩大，左指针收缩
   - 适用于子串/子数组问题
   - O(n) 时间复杂度

4. **单调队列**：
   - 队列内元素单调
   - 用于滑动窗口最大值问题
   - O(n) 时间复杂度
