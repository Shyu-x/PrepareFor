# 动态规划DP完全指南

## 一、动态规划基础

### 1.1 什么是动态规划？

**动态规划（Dynamic Programming，简称DP）** 是一种用于解决具有最优子结构 和 重叠子问题 特性的复杂问题的算法思想。

**类比理解——记忆化学习法：**

想象你是一名学生做数学作业。有一天，老师布置了一道题目："计算斐波那契数列第30项"。

你可能会这样想：

1. **方法一（暴力递归）：** 每次遇到新题目都重新算一遍
   - 计算F(30) = F(29) + F(30)
   - F(29) = F(28) + F(29)
   - F(28) = F(27) + F(28)
   - ...算到天荒地老

2. **方法二（动态规划/记忆化）：** 用一个笔记本记住算过的题目
   - F(1) = 1，记在笔记本上
   - F(2) = 1，记在笔记本上
   - F(3) = F(2) + F(1) = 2，记在笔记本上
   - F(4) = F(3) + F(2) = 3，记在笔记本上
   - ...下次遇到直接查表

**动态规划就是那个"笔记本"**——它把已经计算过的子问题答案存储起来，下次遇到直接查表，而不是重新计算。

### 1.2 动态规划的两个核心条件

| 条件 | 说明 | 类比 |
|------|------|------|
| **最优子结构** | 问题的最优解包含子问题的最优解 | 班长是班里成绩最好的，宿舍长是宿舍成绩最好的，但不意味着班长就是宿舍长 |
| **重叠子问题** | 解的过程中会反复遇到相同的子问题 | 算F(5)时要算F(4)，算F(6)时也要算F(4)，F(4)被重复计算 |

### 1.3 动态规划的三种实现方式

```typescript
/**
 * 动态规划三种实现方式对比
 * 以斐波那契数列为例：F(n) = F(n-1) + F(n-2), F(1)=F(2)=1
 */

// 方式一：自顶向下（记忆化递归）
// 优点：代码直观，易于理解
// 缺点：需要递归栈，可能有栈溢出风险
const memo = new Map<number, number>();
function fibMemo(n: number): number {
  if (n <= 2) return 1;

  if (memo.has(n)) {
    return memo.get(n)!;  // 查表，直接返回
  }

  const result = fibMemo(n - 1) + fibMemo(n - 2);
  memo.set(n, result);    // 记录结果
  return result;
}

// 方式二：自底向上（填表法）
// 优点：无需递归栈，空间效率高
// 缺点：需要明确计算顺序
function fibTabulation(n: number): number {
  if (n <= 2) return 1;

  const dp = new Array(n + 1).fill(0);
  dp[1] = 1;
  dp[2] = 1;

  for (let i = 3; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2];  // 从小到大计算
  }

  return dp[n];
}

// 方式三：空间优化（压缩状态）
// 优点：只需O(1)空间
// 缺点：只能获取最终结果，无法回溯
function fibOptimized(n: number): number {
  if (n <= 2) return 1;

  let prev = 1, curr = 1;

  for (let i = 3; i <= n; i++) {
    const next = prev + curr;
    prev = curr;
    curr = next;
  }

  return curr;
}
```

### 1.4 动态规划解题四步法

```
第一步：定义状态
   - dp[i] 代表什么？
   - dp[i][j] 代表什么？

第二步：找出状态转移方程
   - dp[i] = ?
   - dp[i][j] = ?

第三步：确定初始化
   - dp[0] = ?
   - dp[1] = ?
   - 边界情况如何处理？

第四步：确定计算顺序
   - 从小到大？从大到小？
   - 如何保证使用状态时已经计算过？
```

## 二、基础动态规划问题

### 2.1 爬楼梯问题

**问题描述：** 假设你正在爬楼梯。需要 n 阶你才能到达楼顶。每次你可以爬 1 或 2 个台阶。问有多少种不同的方法可以爬到楼顶？

**问题分析：**
- 第n阶楼梯可以从第n-1阶爬1步上来，也可以从第n-2阶爬2步上来
- 状态定义：dp[i] = 爬到第i阶的方法数
- 状态转移：dp[i] = dp[i-1] + dp[i-2]

```typescript
/**
 * 爬楼梯问题
 *
 * 状态定义：dp[i] = 爬到第i阶的方法数
 * 状态转移：dp[i] = dp[i-1] + dp[i-2]
 * 初始值：dp[1] = 1, dp[2] = 2
 *
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)，可优化到O(1)
 */

function climbStairs(n: number): number {
  if (n <= 2) return n;

  // 空间优化版本
  let prev = 1;   // dp[i-2]
  let curr = 2;   // dp[i-1]

  for (let i = 3; i <= n; i++) {
    const next = prev + curr;
    prev = curr;
    curr = next;
  }

  return curr;
}

/**
 * 爬楼梯（变形：可以爬1、2、3步）
 */
function climbStairsVariants(n: number, maxStep: number = 3): number {
  if (n <= 2) return n;
  if (maxStep < 2) return 1;

  const dp = new Array(n + 1).fill(0);
  dp[0] = 1;  // 0阶有一种方法（不动）

  for (let i = 1; i <= n; i++) {
    for (let step = 1; step <= maxStep && i - step >= 0; step++) {
      dp[i] += dp[i - step];
    }
  }

  return dp[n];
}

/**
 * 爬楼梯（变形：每步代价不同）
 * 使用最小代价爬楼梯
 */
function minCostClimbingStairs(cost: number[]): number {
  const n = cost.length;

  // dp[i] = 爬到第i阶的最小代价
  // dp[i] = min(dp[i-1] + cost[i-1], dp[i-2] + cost[i-2])
  const dp = new Array(n + 1).fill(0);
  dp[0] = 0;  // 起点，代价为0
  dp[1] = 0;  // 第一阶可以从0或1出发，代价都是0

  for (let i = 2; i <= n; i++) {
    dp[i] = Math.min(
      dp[i - 1] + cost[i - 1],  // 从第i-1阶爬1步上来
      dp[i - 2] + cost[i - 2]   // 从第i-2阶爬2步上来
    );
  }

  return dp[n];
}
```

### 2.2 打家劫舍问题

**问题描述：** 你是一个专业的小偷，试图从一条街上偷东西。每间屋子都有一定数量的现金，唯一的限制是相邻的两间屋子不能同时被闯入，因为它们是相邻的。

```typescript
/**
 * 打家劫舍问题
 *
 * 状态定义：dp[i] = 偷到第i间屋子的最大金额
 * 状态转移：
 *   - 不偷第i间：dp[i] = dp[i-1]
 *   - 偷第i间：dp[i] = dp[i-2] + nums[i-1]
 *   dp[i] = max(上述两种情况)
 *
 * 时间复杂度：O(n)
 * 空间复杂度：O(n)，可优化到O(1)
 */

function rob(nums: number[]): number {
  const n = nums.length;
  if (n === 0) return 0;
  if (n === 1) return nums[0];

  // 空间优化版本
  let prev2 = 0;    // dp[i-2]
  let prev1 = 0;    // dp[i-1]

  for (let i = 0; i < n; i++) {
    const curr = Math.max(prev1, prev2 + nums[i]);
    prev2 = prev1;
    prev1 = curr;
  }

  return prev1;
}

/**
 * 打家劫舍II（房屋围成一圈）
 * 第一个和最后一个不能同时偷
 */
function robII(nums: number[]): number {
  const n = nums.length;
  if (n === 0) return 0;
  if (n === 1) return nums[0];
  if (n === 2) return Math.max(nums[0], nums[1]);

  // 情况1：不偷第一间，偷最后一间
  // 情况2：偷第一间，不偷最后一间
  // 取两种情况的最大值

  function robRange(start: number, end: number): number {
    let prev2 = 0;
    let prev1 = 0;

    for (let i = start; i <= end; i++) {
      const curr = Math.max(prev1, prev2 + nums[i]);
      prev2 = prev1;
      prev1 = curr;
    }

    return prev1;
  }

  return Math.max(
    robRange(0, n - 2),   // 不偷最后一间
    robRange(1, n - 1)    // 不偷第一间
  );
}

/**
 * 打家劫舍III（树形结构）
 * 使用二叉树后序遍历 + 记忆化
 */
interface TreeNode {
  val: number;
  left: TreeNode | null;
  right: TreeNode | null;
}

function robIII(root: TreeNode | null): number {
  const memo = new Map<TreeNode, number>();

  function dfs(node: TreeNode | null): number {
    if (!node) return 0;

    if (memo.has(node)) {
      return memo.get(node)!;
    }

    // 不偷当前节点
    const notRob = dfs(node.left) + dfs(node.right);

    // 偷当前节点（不能偷子节点）
    let rob = node.val;
    if (node.left) {
      rob += dfs(node.left.left) + dfs(node.left.right);
    }
    if (node.right) {
      rob += dfs(node.right.left) + dfs(node.right.right);
    }

    const result = Math.max(notRob, rob);
    memo.set(node, result);

    return result;
  }

  return dfs(root);
}
```

### 2.3 股票买卖问题

**问题描述：** 给定一个数组 prices ，它的第 i 个元素 prices[i] 是一支给定的股票在第 i 天的价格。设计算法求出最大利润。

```typescript
/**
 * 股票买卖问题系列
 */

// 问题1：只允许交易一次
function maxProfitOnce(prices: number[]): number {
  let minPrice = Infinity;
  let maxProfit = 0;

  for (const price of prices) {
    // 更新最低价格
    minPrice = Math.min(minPrice, price);
    // 更新最大利润
    maxProfit = Math.max(maxProfit, price - minPrice);
  }

  return maxProfit;
}

// 问题2：允许交易无数次（但有冷冻期）
function maxProfitUnlimited(prices: number[]): number {
  const n = prices.length;
  if (n <= 1) return 0;

  // dp[i][0] = 第i天手中没有股票的最大利润
  // dp[i][1] = 第i天手中有股票的最大利润
  const dp = Array.from({length: n}, () => [0, 0]);

  dp[0][0] = 0;           // 第一天不买
  dp[0][1] = -prices[0]; // 第一天买了

  for (let i = 1; i < n; i++) {
    dp[i][0] = Math.max(dp[i-1][0], dp[i-1][1] + prices[i]);  // 卖出或保持
    dp[i][1] = Math.max(dp[i-1][1], dp[i-1][0] - prices[i]); // 买入或保持
  }

  return dp[n-1][0];
}

// 问题3：最多交易k次
function maxProfitK(k: number, prices: number[]): number {
  const n = prices.length;
  if (n === 0 || k === 0) return 0;

  // 如果k >= n/2，可以进行无限次交易
  if (k >= n / 2) {
    let profit = 0;
    for (let i = 1; i < n; i++) {
      profit += Math.max(0, prices[i] - prices[i - 1]);
    }
    return profit;
  }

  // dp[i][j] = 第i天最多交易j次且手中无股票的最大利润
  // dp[i][j] = max(dp[i-1][j], prices[i] + maxDiff)
  // maxDiff = max(maxDiff, dp[i-1][j-1] - prices[i])

  const dp = Array.from({length: n}, () => new Array(k + 1).fill(0));

  for (let j = 1; j <= k; j++) {
    let maxDiff = -prices[0];
    for (let i = 1; i < n; i++) {
      dp[i][j] = Math.max(dp[i-1][j], prices[i] + maxDiff);
      maxDiff = Math.max(maxDiff, dp[i-1][j-1] - prices[i]);
    }
  }

  return dp[n-1][k];
}

// 问题4：带有冷冻期
function maxProfitWithCoolDown(prices: number[]): number {
  const n = prices.length;
  if (n <= 1) return 0;

  // dp[i][0] = 第i天手中没有股票且处于冷冻期
  // dp[i][1] = 第i天手中没有股票且不处于冷冻期
  // dp[i][2] = 第i天手中持有股票
  const dp = Array.from({length: n}, () => [0, 0, 0]);

  dp[0][0] = 0;
  dp[0][1] = 0;
  dp[0][2] = -prices[0];

  for (let i = 1; i < n; i++) {
    dp[i][0] = dp[i-1][2] + prices[i];      // 卖出股票，进入冷冻期
    dp[i][1] = Math.max(dp[i-1][1], dp[i-1][0]); // 保持不持有，且不处于冷冻期
    dp[i][2] = Math.max(dp[i-1][2], dp[i-1][1] - prices[i]); // 买入或保持
  }

  return Math.max(dp[n-1][0], dp[n-1][1]);
}

// 问题5：带有交易手续费
function maxProfitWithFee(prices: number[], fee: number): number {
  const n = prices.length;
  if (n <= 1) return 0;

  let hold = -prices[0];   // 持有股票
  let cash = 0;           // 不持有股票

  for (let i = 1; i < n; i++) {
    // 更新不持有股票的现金
    cash = Math.max(cash, hold + prices[i] - fee);
    // 更新持有股票的状态
    hold = Math.max(hold, cash - prices[i]);
  }

  return cash;
}
```

## 三、背包问题

### 3.1 0-1背包问题

**问题描述：** 给定 n 个物品和一个容量为 W 的背包，每个物品 i 都有重量 weight[i] 和价值 value[i]，每个物品只能选择装入或不装入，求解将哪些物品装入背包使得所装物品的重量不超过背包容量且价值最大。

```typescript
/**
 * 0-1背包问题
 *
 * 状态定义：dp[i][w] = 前i个物品，背包容量为w时的最大价值
 * 状态转移：
 *   - 不选第i个物品：dp[i][w] = dp[i-1][w]
 *   - 选第i个物品：dp[i][w] = dp[i-1][w-weight[i]] + value[i]
 *   dp[i][w] = max(上述两种情况)
 *
 * 时间复杂度：O(n × W)
 * 空间复杂度：O(n × W)，可优化到O(W)
 */

function knapsack01(weights: number[], values: number[], W: number): number {
  const n = weights.length;

  // dp[w] = 容量为w时的最大价值
  // 初始化为0
  const dp = new Array(W + 1).fill(0);

  // 遍历每个物品
  for (let i = 0; i < n; i++) {
    // 注意：需要从大到小遍历W
    // 因为每个物品只能选一次，要避免重复选择
    for (let w = W; w >= weights[i]; w--) {
      dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
    }
  }

  return dp[W];
}

/**
 * 0-1背包问题（输出具体选择）
 */
function knapsack01WithChoice(
  weights: number[],
  values: number[],
  W: number
): {maxValue: number, choice: boolean[]} {
  const n = weights.length;

  // dp[i][w] = 前i个物品，背包容量为w时的最大价值
  const dp = Array.from({length: n + 1}, () => new Array(W + 1).fill(0));

  // 填表
  for (let i = 1; i <= n; i++) {
    for (let w = 0; w <= W; w++) {
      // 不选第i个物品
      dp[i][w] = dp[i - 1][w];

      // 选第i个物品（如果能装下）
      if (w >= weights[i - 1]) {
        dp[i][w] = Math.max(
          dp[i][w],
          dp[i - 1][w - weights[i - 1]] + values[i - 1]
        );
      }
    }
  }

  // 回溯找出具体选择
  const choice = new Array(n).fill(false);
  let w = W;
  for (let i = n; i > 0; i--) {
    if (dp[i][w] !== dp[i - 1][w]) {
      choice[i - 1] = true;
      w -= weights[i - 1];
    }
  }

  return {maxValue: dp[n][W], choice};
}

/**
 * 0-1背包问题（恰好装满）
 * 要求恰好装满背包，返回最大价值（或-1表示无解）
 */
function knapsack01Exactly(
  weights: number[],
  values: number[],
  W: number
): number {
  const n = weights.length;

  // dp[w] = 容量为w时能获得的最大价值，-1表示无解
  const dp = new Array(W + 1).fill(-Infinity);
  dp[0] = 0;  // 容量为0，只有一种方式（什么都不装），价值为0

  for (let i = 0; i < n; i++) {
    for (let w = W; w >= weights[i]; w--) {
      if (dp[w - weights[i]] !== -Infinity) {
        dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
      }
    }
  }

  return dp[W] === -Infinity ? -1 : dp[W];
}
```

### 3.2 完全背包问题

**问题描述：** 与0-1背包类似，但每个物品可以选择无限次（完全背包）。

```typescript
/**
 * 完全背包问题
 *
 * 与0-1背包的区别：
 * - 0-1背包：每种物品只能选一次，遍历容量时从大到小
 * - 完全背包：每种物品可以选无限次，遍历容量时从小到大
 *
 * 时间复杂度：O(n × W)
 * 空间复杂度：O(W)
 */

function completeKnapsack(weights: number[], values: number[], W: number): number {
  const n = weights.length;
  const dp = new Array(W + 1).fill(0);

  for (let i = 0; i < n; i++) {
    // 完全背包：从容量小到大遍历
    for (let w = weights[i]; w <= W; w++) {
      dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
    }
  }

  return dp[W];
}

/**
 * 零钱兑换问题（完全背包变体）
 * 给定不同面额的硬币和总金额，计算凑成总金额的最少硬币数
 */
function coinChange(coins: number[], amount: number): number {
  // dp[w] = 凑成金额w所需的最少硬币数
  const dp = new Array(amount + 1).fill(Infinity);
  dp[0] = 0;  // 金额为0需要0个硬币

  for (const coin of coins) {
    for (let w = coin; w <= amount; w++) {
      dp[w] = Math.min(dp[w], dp[w - coin] + 1);
    }
  }

  return dp[amount] === Infinity ? -1 : dp[amount];
}

/**
 * 零钱兑换II（计算组合数）
 * 计算凑成总金额的硬币组合数
 */
function change(coins: number[], amount: number): number {
  const dp = new Array(amount + 1).fill(0);
  dp[0] = 1;  // 金额为0有1种组合（什么都不选）

  for (const coin of coins) {
    for (let w = coin; w <= amount; w++) {
      dp[w] += dp[w - coin];
    }
  }

  return dp[amount];
}
```

### 3.3 多重背包问题

**问题描述：** 每种物品有有限个，而不是0-1背包的最多1个或完全背包的无限个。

```typescript
/**
 * 多重背包问题
 *
 * 方法：将多重背包转换成0-1背包
 * 对于第i种物品，有count[i]个，可以展开成count[i]个物品
 * 但这样时间复杂度太高，优化方法：二进制优化
 *
 * 二进制优化：
 * 将count[i]拆分成：1, 2, 4, 8, ..., 2^k, count[i] - (2^k - 1)
 * 这样可以用O(log count[i])个物品表示原来的count[i]个物品
 */

// 基本多重背包（直接展开，不推荐使用）
function multipleKnapsackBasic(
  weights: number[],
  values: number[],
  counts: number[],
  W: number
): number {
  const dp = new Array(W + 1).fill(0);

  for (let i = 0; i < weights.length; i++) {
    // 展开成count[i]个物品
    for (let c = 0; c < counts[i]; c++) {
      for (let w = W; w >= weights[i]; w--) {
        dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
      }
    }
  }

  return dp[W];
}

// 多重背包（二进制优化）
function multipleKnapsackOptimized(
  weights: number[],
  values: number[],
  counts: number[],
  W: number
): number {
  const dp = new Array(W + 1).fill(0);

  for (let i = 0; i < weights.length; i++) {
    // 二进制拆分
    let count = counts[i];
    let w = weights[i];
    let v = values[i];

    // 拆分成 1, 2, 4, 8, ...
    let k = 1;
    while (k <= count) {
      for (let cap = W; cap >= k * w; cap--) {
        dp[cap] = Math.max(dp[cap], dp[cap - k * w] + k * v);
      }
      count -= k;
      k *= 2;
    }

    // 处理剩余的
    if (count > 0) {
      for (let cap = W; cap >= count * w; cap--) {
        dp[cap] = Math.max(dp[cap], dp[cap - count * w] + count * v);
      }
    }
  }

  return dp[W];
}
```

## 四、最长公共子序列与编辑距离

### 4.1 最长公共子序列（LCS）

**问题描述：** 给定两个字符串 text1 和 text2，返回它们的最长公共子序列的长度。

```typescript
/**
 * 最长公共子序列
 *
 * 状态定义：dp[i][j] = text1[0..i-1]和text2[0..j-1]的最长公共子序列长度
 * 状态转移：
 *   - 如果text1[i-1] == text2[j-1]：dp[i][j] = dp[i-1][j-1] + 1
 *   - 否则：dp[i][j] = max(dp[i-1][j], dp[i][j-1])
 *
 * 时间复杂度：O(m × n)
 * 空间复杂度：O(m × n)，可优化到O(min(m,n))
 */

function longestCommonSubsequence(text1: string, text2: string): number {
  const m = text1.length;
  const n = text2.length;

  // dp[i][j] = text1[0..i-1]和text2[0..j-1]的LCS长度
  const dp = Array.from({length: m + 1}, () => new Array(n + 1).fill(0));

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }

  return dp[m][n];
}

/**
 * 最长公共子序列（空间优化）
 */
function longestCommonSubsequenceOptimized(text1: string, text2: string): number {
  const m = text1.length;
  const n = text2.length;

  // 使用两行滚动数组优化空间
  // 确保text1是较短的字符串
  if (m < n) {
    [text1, text2] = [text2, text1];
  }

  const dp = new Array(n + 1).fill(0);
  let prevRow = new Array(n + 1).fill(0);

  for (let i = 1; i <= m; i++) {
    [dp, prevRow] = [prevRow, dp];
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[j] = prevRow[j - 1] + 1;
      } else {
        dp[j] = Math.max(prevRow[j], dp[j - 1]);
      }
    }
  }

  return prevRow[n];
}

/**
 * 最长公共子序列（输出具体序列）
 */
function longestCommonSubsequenceWithSequence(
  text1: string,
  text2: string
): {length: number, sequence: string} {
  const m = text1.length;
  const n = text2.length;

  const dp = Array.from({length: m + 1}, () => new Array(n + 1).fill(0));

  // 填表
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }

  // 回溯找出具体序列
  let i = m, j = n;
  const sequence: string[] = [];

  while (i > 0 && j > 0) {
    if (text1[i - 1] === text2[j - 1]) {
      sequence.push(text1[i - 1]);
      i--;
      j--;
    } else if (dp[i - 1][j] > dp[i][j - 1]) {
      i--;
    } else {
      j--;
    }
  }

  return {
    length: dp[m][n],
    sequence: sequence.reverse().join('')
  };
}
```

### 4.2 编辑距离

**问题描述：** 给你两个单词 word1 和 word2，请你计算出将 word1 转换成 word2 最少需要多少步操作。允许的操作包括：插入一个字符、删除一个字符、替换一个字符。

```typescript
/**
 * 编辑距离（Levenshtein Distance）
 *
 * 状态定义：dp[i][j] = word1[0..i-1]转换成word2[0..j-1]的最少操作数
 * 状态转移：
 *   - 如果word1[i-1] == word2[j-1]：dp[i][j] = dp[i-1][j-1]（不需要操作）
 *   - 否则：dp[i][j] = 1 + min(
 *       dp[i-1][j] + 1,     // 删除word1[i-1]
 *       dp[i][j-1] + 1,     // 插入word2[j-1]
 *       dp[i-1][j-1] + 1    // 替换word1[i-1]为word2[j-1]
 *     )
 *
 * 时间复杂度：O(m × n)
 * 空间复杂度：O(m × n)，可优化到O(min(m,n))
 */

function minDistance(word1: string, word2: string): number {
  const m = word1.length;
  const n = word2.length;

  // dp[i][j] = word1[0..i-1]转换成word2[0..j-1]的最少操作数
  const dp = Array.from({length: m + 1}, () => new Array(n + 1).fill(0));

  // 初始化边界
  // word1空串转换成word2，需要插入n个字符
  for (let j = 0; j <= n; j++) {
    dp[0][j] = j;
  }

  // word2空串转换成word1，需要删除m个字符
  for (let i = 0; i <= m; i++) {
    dp[i][0] = i;
  }

  // 填表
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (word1[i - 1] === word2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      } else {
        dp[i][j] = 1 + Math.min(
          dp[i - 1][j],      // 删除
          dp[i][j - 1],      // 插入
          dp[i - 1][j - 1]   // 替换
        );
      }
    }
  }

  return dp[m][n];
}

/**
 * 编辑距离（空间优化）
 */
function minDistanceOptimized(word1: string, word2: string): number {
  const m = word1.length;
  const n = word2.length;

  // 确保m <= n，用word2作为行以节省空间
  if (m > n) {
    [word1, word2] = [word2, word1];
  }

  const dp = new Array(m + 1).fill(0);

  // 初始化
  for (let i = 0; i <= m; i++) {
    dp[i] = i;
  }

  for (let j = 1; j <= n; j++) {
    let prev = dp[0];  // 保存dp[i-1][j-1]
    dp[0] = j;         // dp[0][j] = j

    for (let i = 1; i <= m; i++) {
      const temp = dp[i];  // 保存dp[i-1][j]，供下次循环使用
      if (word1[i - 1] === word2[j - 1]) {
        dp[i] = prev;
      } else {
        dp[i] = 1 + Math.min(prev, dp[i], dp[i - 1]);
      }
      prev = temp;
    }
  }

  return dp[m];
}
```

## 五、状态压缩动态规划

### 5.1 状态压缩DP简介

**状态压缩DP** 是一种利用二进制表示状态来节省空间的动态规划技巧。

**适用场景：**
- 问题的状态可以用二进制位表示
- 状态数不超过 20-30 个（否则 2^n 会太大）

**类比理解：** 想象你有10个开关，每个开关有开和关两种状态。你可以用一个10位的二进制数来表示所有开关的状态：0000000000 表示全关，1111111111 表示全开。状态压缩DP就是用这样的思想，用一个整数来表示整个状态。

### 5.2 旅行商问题（TSP）

**问题描述：** 给定 n 个城市和它们之间的距离，求访问所有城市一次并返回起点的最短路径。

```typescript
/**
 * 旅行商问题（TSP）- 状态压缩DP
 *
 * 状态定义：dp[mask][i] = 当前在城市i，已访问城市集合为mask的最短路径
 * mask是一个二进制数，第j位为1表示城市j已被访问
 *
 * 状态转移：
 * dp[mask][i] = min(dp[mask ^ (1 << i)][j] + dist[j][i])
 * 其中j是mask中除i外的另一个已访问城市
 *
 * 时间复杂度：O(n² × 2^n)
 * 空间复杂度：O(n × 2^n)
 */

function tsp(dist: number[][]): number {
  const n = dist.length;
  if (n <= 1) return 0;

  const INF = Infinity;
  const fullMask = (1 << n) - 1;  // 所有位都为1的掩码

  // dp[mask][i] = 访问集合mask中的城市，以i结尾的最短路径
  // mask中必须包含i
  const dp = Array.from({length: 1 << n}, () => new Array(n).fill(INF));

  // 起点是城市0
  dp[1 << 0][0] = 0;

  // 枚举所有状态
  for (let mask = 1; mask <= fullMask; mask++) {
    for (let i = 0; i < n; i++) {
      // 如果mask中不包含i，跳过
      if (!(mask & (1 << i))) continue;

      // 如果mask中只有i，继续
      if (mask === (1 << i)) continue;

      // 从j走到i
      const prevMask = mask ^ (1 << i);
      for (let j = 0; j < n; j++) {
        // j必须在prevMask中
        if (prevMask & (1 << j)) {
          dp[mask][i] = Math.min(
            dp[mask][i],
            dp[prevMask][j] + dist[j][i]
          );
        }
      }
    }
  }

  // 返回从所有城市出发回到0的最短路径
  let ans = INF;
  for (let i = 1; i < n; i++) {
    ans = Math.min(ans, dp[fullMask][i] + dist[i][0]);
  }

  return ans;
}

/**
 * TSP（输出具体路径）
 */
function tspWithPath(dist: number[][]): {distance: number, path: number[]} {
  const n = dist.length;
  if (n <= 1) return {distance: 0, path: [0]};

  const INF = Infinity;
  const fullMask = (1 << n) - 1;

  // dp[mask][i]
  const dp = Array.from({length: 1 << n}, () => new Array(n).fill(INF));
  const next = Array.from({length: 1 << n}, () => new Array(n).fill(-1));

  dp[1 << 0][0] = 0;

  for (let mask = 1; mask <= fullMask; mask++) {
    for (let i = 0; i < n; i++) {
      if (!(mask & (1 << i))) continue;
      if (mask === (1 << i)) continue;

      const prevMask = mask ^ (1 << i);
      for (let j = 0; j < n; j++) {
        if (prevMask & (1 << j)) {
          const newDist = dp[prevMask][j] + dist[j][i];
          if (newDist < dp[mask][i]) {
            dp[mask][i] = newDist;
            next[mask][i] = j;
          }
        }
      }
    }
  }

  // 回溯路径
  let mask = fullMask;
  let last = 0;
  let ans = INF;
  for (let i = 1; i < n; i++) {
    const d = dp[fullMask][i] + dist[i][0];
    if (d < ans) {
      ans = d;
      last = i;
    }
  }

  const path: number[] = [0];
  while (mask !== (1 << last)) {
    path.push(last);
    const prev = next[mask][last];
    mask = mask ^ (1 << last);
    last = prev;
  }
  path.push(0);

  return {distance: ans, path};
}
```

### 5.3 蒙德里安梦想问题

**问题描述：** 用 1×2 的多米诺骨牌铺满 n×m 的棋盘，问有多少种铺法。

```typescript
/**
 * 蒙德里安的梦想 - 状态压缩DP
 *
 * 问题：用1×2的多米诺骨牌铺满n×m的棋盘
 *
 * 状态定义：dp[i][state] = 第i行，状态为state的方案数
 * state是一个二进制数，表示第i列是否被上一行的骨牌占用
 *
 * 时间复杂度：O(n × m × 2^m × 状态转移)
 *
 * @param n 行数
 * @param m 列数
 */
function dominosCount(n: number, m: number): number {
  // 确保m <= n以优化空间
  if (m > n) {
    [n, m] = [m, n];
  }

  const fullMask = (1 << m) - 1;

  // valid[mask] = mask是否是有效状态（没有连续奇数个0）
  const valid: boolean[] = [];
  for (let mask = 0; mask <= fullMask; mask++) {
    let cnt = 0;
    let isValid = true;
    for (let i = 0; i < m; i++) {
      if (mask & (1 << i)) {
        // 如果当前位是1，检查前面的0的个数
        if (cnt % 2 === 1) {
          isValid = false;
          break;
        }
        cnt = 0;
      } else {
        cnt++;
      }
    }
    // 检查最后一段0
    if (cnt % 2 === 1) isValid = false;
    valid[mask] = isValid;
  }

  //兼容状态预计算：state1和state2兼容（在同一列不冲突）
  const compatible: boolean[][] = [];
  for (let state1 = 0; state1 <= fullMask; state1++) {
    compatible[state1] = [];
    for (let state2 = 0; state2 <= fullMask; state2++) {
      // state1和state2兼容的条件：
      // 1. 两者都是有效状态
      // 2. 两者没有重叠的1（state1 & state2 === 0）
      if (valid[state1] && valid[state2] && (state1 & state2) === 0) {
        // 合并后的状态也必须是有效的
        const merged = state1 | state2;
        compatible[state1][state2] = valid[merged];
      }
    }
  }

  // dp[i][state] = 第i行状态为state的方案数
  const dp = Array.from({length: n + 1}, () => new Array(1 << m).fill(0));
  dp[0][fullMask] = 1;  // 第0行全为1（假设有虚拟行）

  for (let i = 1; i <= n; i++) {
    for (let state = 0; state <= fullMask; state++) {
      if (!valid[state]) continue;

      let sum = 0;
      for (let prevState = 0; prevState <= fullMask; prevState++) {
        if (compatible[prevState][state]) {
          sum += dp[i - 1][prevState];
        }
      }
      dp[i][state] = sum;
    }
  }

  return dp[n][fullMask];
}
```
