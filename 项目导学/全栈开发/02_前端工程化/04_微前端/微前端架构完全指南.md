## 1. 微前端概述

### 1.1 什么是微前端？

微前端是一种将前端应用分解成更小、更简单的独立部分的架构风格，每个部分可以独立开发、测试和部署。

```typescript
function selectMicroFrontendSolution(requirements: {
  teamSize: 'small' | 'medium' | 'large';
  techStack: 'same' | 'mixed';
  isolation: 'high' | 'medium' | 'low';
  legacy: boolean;
}): string {
  // 小团队，技术栈相同
  if (requirements.teamSize === 'small' && requirements.techStack === 'same') {
    return 'Module Federation';
  }

  // 大团队，技术栈混合
  if (requirements.teamSize === 'large' && requirements.techStack === 'mixed') {
    return 'qiankun';
  }

  // 需要高度隔离
  if (requirements.isolation === 'high') {
    return 'iframe';
  }

  // 有遗留系统
  if (requirements.legacy) {
    return 'qiankun';
  }

  // 默认推荐
  return 'qiankun';
}
```

### 4.2 动态远程模块

```typescript
// 动态加载远程模块

// 1. 动态远程模块加载器
class RemoteLoader {
  private loadedRemotes: Map<string, any> = new Map();

  async load(remoteUrl: string, scope: string, module: string) {
    const key = `${scope}/${module}`;

    if (this.loadedRemotes.has(key)) {
      return this.loadedRemotes.get(key);
    }

    // 加载远程入口
    await this.loadScript(remoteUrl);

    // 获取容器
    const container = (window as any)[scope];

    // 初始化共享作用域
    await container.init(__webpack_share_scopes__.default);

    // 获取模块
    const factory = await container.get(module);
    const Module = factory();

    this.loadedRemotes.set(key, Module);

    return Module;
  }

  private loadScript(url: string): Promise<void> {
    return new Promise((resolve, reject) => {
      if (document.querySelector(`script[src="${url}"]`)) {
        resolve();
        return;
      }

      const script = document.createElement('script');
      script.src = url;
      script.type = 'text/javascript';
      script.async = true;

      script.onload = () => resolve();
      script.onerror = reject;

      document.head.appendChild(script);
    });
  }
}

// 2. 使用动态加载器
const loader = new RemoteLoader();

async function loadRemoteComponent() {
  const module = await loader.load(
    'http://localhost:3001/remoteEntry.js',
    'remoteApp',
    './Button'
  );

  return module.default;
}

// 3. React Hook封装
function useRemoteModule<T>(
  remoteUrl: string,
  scope: string,
  module: string
): { data: T | null; loading: boolean; error: Error | null } {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    const loader = new RemoteLoader();

    loader.load<T>(remoteUrl, scope, module)
      .then(setData)
      .catch(setError)
      .finally(() => setLoading(false));
  }, [remoteUrl, scope, module]);

  return { data, loading, error };
}

// 使用
function RemoteComponent() {
  const { data: Button, loading, error } = useRemoteModule(
    'http://localhost:3001/remoteEntry.js',
    'remoteApp',
    './Button'
  );

  if (loading) return <div>加载中...</div>;
  if (error) return <div>加载失败: {error.message}</div>;

  return <Button>远程按钮</Button>;
}
```

## 5. 沙箱隔离机制

### 5.1 JS沙箱实现

```typescript
// qiankun JS沙箱实现原理

// 1. 快照沙箱（SnapshotSandbox）
class SnapshotSandbox {
  private windowSnapshot: any = {};
  private modifyPropsMap: any = {};

  active() {
    // 保存当前window状态
    for (const prop in window) {
      this.windowSnapshot[prop] = window[prop];
    }

    // 恢复之前的修改
    Object.keys(this.modifyPropsMap).forEach(prop => {
      window[prop] = this.modifyPropsMap[prop];
    });
  }

  inactive() {
    // 记录修改
    for (const prop in window) {
      if (window[prop] !== this.windowSnapshot[prop]) {
        this.modifyPropsMap[prop] = window[prop];
        // 恢复原始值
        window[prop] = this.windowSnapshot[prop];
      }
    }
  }
}

// 2. Proxy沙箱（LegacySandbox）
class LegacySandbox {
  private addedPropsMapInSandbox: Map<string, any> = new Map();
  private modifiedPropsOriginalValueMapInSandbox: Map<string, any> = new Map();
  private currentUpdatedPropsValueMap: Map<string, any> = new Map();

  private proxyWindow: Window;

  constructor() {
    const fakeWindow = Object.create(null) as Window;

    this.proxyWindow = new Proxy(fakeWindow, {
      set: (target, prop, value) => {
        // 记录新增属性
        if (!window.hasOwnProperty(prop)) {
          this.addedPropsMapInSandbox.set(prop as string, value);
        }
        // 记录修改的原始值
        else if (!this.modifiedPropsOriginalValueMapInSandbox.has(prop as string)) {
          this.modifiedPropsOriginalValueMapInSandbox.set(
            prop as string,
            window[prop as string]
          );
        }

        this.currentUpdatedPropsValueMap.set(prop as string, value);
        window[prop as string] = value;

        return true;
      },

      get: (target, prop) => {
        return window[prop as string];
      },
    }) as Window;
  }

  getProxyWindow() {
    return this.proxyWindow;
  }
}

// 3. Proxy沙箱（ProxySandbox）- 最完善
class ProxySandbox {
  private sandboxRunning: boolean = false;

  private proxyWindow: Window;

  constructor() {
    const fakeWindow = Object.create(null) as Window;

    this.proxyWindow = new Proxy(fakeWindow, {
      set: (target, prop, value, receiver) => {
        if (this.sandboxRunning) {
          target[prop as string] = value;
        }
        return true;
      },

      get: (target, prop, receiver) => {
        // 优先从fakeWindow获取
        if (target.hasOwnProperty(prop)) {
          return target[prop as string];
        }

        // 从真实window获取
        const value = window[prop as string];

        // 如果是函数，绑定this到window
        if (typeof value === 'function') {
          return value.bind(window);
        }

        return value;
      },

      has: (target, prop) => {
        return prop in target || prop in window;
      },

      deleteProperty: (target, prop) => {
        if (target.hasOwnProperty(prop)) {
          delete target[prop as string];
        }
        return true;
      },
    }) as Window;
  }

  active() {
    this.sandboxRunning = true;
  }

  inactive() {
    this.sandboxRunning = false;
  }

  getProxyWindow() {
    return this.proxyWindow;
  }
}
```

### 5.2 沙箱逃逸防护

```typescript
// 沙箱逃逸防护

// 1. 防止通过原型链逃逸
const proxy = new Proxy(fakeWindow, {
  get: (target, prop, receiver) => {
    // 阻止访问危险属性
    const dangerousProps = [
      'constructor',
      '__proto__',
      'prototype',
      'eval',
      'Function',
    ];

    if (dangerousProps.includes(prop as string)) {
      throw new Error(`禁止访问: ${String(prop)}`);
    }

    return Reflect.get(target, prop, receiver);
  },
});

// 2. 防止通过this逃逸
function safeFunction(fn: Function) {
  return new Proxy(fn, {
    apply: (target, thisArg, args) => {
      // 强制this指向proxy
      return Reflect.apply(target, proxy, args);
    },
  });
}

// 使用严格模式禁止with语句
'use strict';

// 4. 防止通过document逃逸
const proxyDocument = new Proxy(document, {
  get: (target, prop) => {
    // 限制危险操作
    if (prop === 'write' || prop === 'writeln') {
      return () => {
        throw new Error('禁止使用document.write');
      };
    }

    return Reflect.get(target, prop);
  },
});
```

```typescript
// 动态样式隔离

// 1. 动态添加样式前缀
function addStylePrefix(css: string, prefix: string): string {
  // 为选择器添加前缀
  return css.replace(/(^|})\s*([^{]+)\s*\{/g, (match, start, selectors) => {
    const prefixedSelectors = selectors
      .split(',')
      .map(s => `${prefix} ${s.trim()}`)
      .join(', ');
    return `${start} ${prefixedSelectors} {`;
  });
}

// 2. 样式隔离中间件
function styleIsolationMiddleware(prefix: string) {
  return (req: any, res: any, next: any) => {
    const originalSend = res.send;

    res.send = function(body: string) {
      if (typeof body === 'string' && body.includes('<style>')) {
        // 处理内联样式
        body = body.replace(/<style>([\s\S]*?)<\/style>/g, (match, css) => {
          const prefixedCss = addStylePrefix(css, prefix);
          return `<style>${prefixedCss}</style>`;
        });
      }

      originalSend.call(this, body);
    };

    next();
  };
}

// 3. 运行时样式隔离
class StyleIsolator {
  private styleMap: Map<string, HTMLStyleElement> = new Map();

  // 添加样式
  addStyle(id: string, css: string, prefix: string) {
    const prefixedCss = addStylePrefix(css, prefix);

    const style = document.createElement('style');
    style.id = id;
    style.textContent = prefixedCss;

    document.head.appendChild(style);
    this.styleMap.set(id, style);
  }

  // 移除样式
  removeStyle(id: string) {
    const style = this.styleMap.get(id);
    if (style) {
      style.remove();
      this.styleMap.delete(id);
    }
  }

  // 清除所有样式
  clearAll() {
    this.styleMap.forEach(style => style.remove());
    this.styleMap.clear();
  }
}
```

```typescript
class EventBus {
  private events: Map<string, Set<Function>> = new Map();

  on(event: string, callback: Function) {
    if (!this.events.has(event)) {
      this.events.set(event, new Set());
    }
    this.events.get(event)!.add(callback);
  }

  off(event: string, callback: Function) {
    this.events.get(event)?.delete(callback);
  }

  emit(event: string, data: any) {
    this.events.get(event)?.forEach(callback => callback(data));
  }
}

// 挂载到window
(window as any).eventBus = new EventBus();
```

## 8. 面试高频问题

### 问题1：微前端有什么优势？

**答案：**
1. 技术栈无关，不同团队可以使用不同框架
2. 独立开发部署，降低发布风险
3. 增量升级，逐步替换老旧系统
4. 团队自治，提高开发效率

### 问题2：qiankun的沙箱机制是什么？

**答案：**
qiankun提供了三种JS沙箱：
1. **快照沙箱**：激活时保存window状态，失活时恢复
2. **LegacySandbox**：使用Proxy记录修改
3. **ProxySandbox**：最完善的方案，完全隔离

### 问题3：如何解决样式冲突？

**答案：**
1. CSS Scoped / CSS Modules
2. Shadow DOM
3. CSS命名空间
4. qiankun的样式隔离功能
5. CSS-in-JS

### 问题4：Module Federation和qiankun的区别？

**答案：**
| 方面 | Module Federation | qiankun |
|------|-------------------|---------|
| 隔离 | 无隔离 | JS/CSS沙箱 |
| 集成方式 | 构建时 | 运行时 |
| 依赖共享 | 支持 | 不支持 |
| 适用场景 | 同技术栈 | 跨技术栈 |

### 问题5：微前端如何处理公共依赖？

**答案：**
1. Module Federation的shared配置
2. CDN加载公共库
3. externals配置
4. 主应用统一加载
