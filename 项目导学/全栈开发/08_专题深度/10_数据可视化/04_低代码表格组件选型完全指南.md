# 低代码表格组件选型完全指南

## 前言

在企业级应用开发中，表格组件是最常用的数据展示和编辑组件之一。无论是管理后台、数据看板还是配置页面，表格都是不可或缺的核心组件。

本章将对目前市面上最流行的几款前端表格组件进行深入对比分析，包括 Handsontable、AG-Grid、SpreadJS（葡萄城）和 FineUI 等。通过本指南，你将能够根据项目需求选择最合适的表格组件，并了解各组件的核心特性、使用方法和性能优化技巧。

## 二、Handsontable 深度分析

### 2.1 Handsontable 概述

Handsontable 是一款专为 Excel-like 数据编辑场景设计的表格组件。它提供了类似 Excel 的编辑体验，包括单元格编辑、填充序列、公式计算等功能。

```typescript
/**
 * Handsontable 核心概念
 *
 * Handsontable 专注于电子表格场景，特别适合：
 * 1. 数据录入和编辑场景
 * 2. 需要 Excel 导入导出功能
 * 3. 简单的数据展示和操作
 *
 * 优点：
 * - Excel 体验好
 * - 公式支持较好
 * - 文档完善
 *
 * 缺点：
 * - 虚拟滚动性能一般
 * - 自定义能力有限
 * - 商业版本价格较高
 */

// 安装：npm install handsontable @handsontable/react
import Handsontable from 'handsontable';
import { HotTable } from '@handsontable/react';

  // 列配置
  const columns = [
    { type: 'text', title: '品牌' },
    { type: 'numeric', title: '2019' },
    { type: 'numeric', title: '2020' },
    { type: 'numeric', title: '2021' },
    { type: 'numeric', title: '2022' },
    { type: 'numeric', title: '2023' },

  // Handsontable 配置
  const hotSettings: Handsontable.Settings = {
    data: hotData,
    columns: columns,
    // 行头设置
    rowHeaders: true,
    // 列头设置

    // 粘贴设置

      const cellMeta: Handsontable.CellMetadata = {};

        cellMeta.className = 'htCenter htMiddle header-cell';
      }
      // 数字列右对齐
      if (col > 0) {
        cellMeta.className = 'numeric-cell';
      }
      return cellMeta;
    },
    // 上下文菜单
    contextMenu: {
      items: {
        'row_above': { name: '在上方插入行' },
        'row_below': { name: '在下方插入行' },

      settings={hotSettings}
```

## 三、AG-Grid 深度分析

```typescript

 * AG-Grid 核心概念

 * 2. 复杂的企业级表格应用
 * 3. 需要丰富交互的数据分析

 * 优点：
 * - 性能极佳（Canvas 渲染）
```

```typescript
                {flexRender(header.column.columnDef.header, header.getContext())}
```

```typescript
            ))}
          </tr>
        </thead>
        <tbody>
          {paddingTop > 0 && (
            <tr>
              <td style={{ height: `${paddingTop}px` }} />

                  <td key={cell.id} style={{ width: cell.column.getSize() }}>

              <td style={{ height: `${paddingBottom}px` }} />
```

### 6.1 虚拟滚动核心原理

```typescript
/**
 * 虚拟滚动核心原理
 *
 * 虚拟滚动的核心思想是：只渲染可见区域的行
 * 当滚动时，动态计算可见范围，只渲染该范围内的行
 *
 * 关键计算：
 * 1. 可见起始行 = floor(scrollTop / rowHeight)
 * 2. 可见结束行 = ceil((scrollTop + viewportHeight) / rowHeight)
 * 3. 缓冲区行数 = 通常为可见行数的 1-2 倍
 *
 * 性能提升：
 * - 10000 行数据：渲染 10000 个 DOM -> 渲染 ~20 个 DOM
 * - 性能提升：100x+
 */

/**
 * 虚拟滚动实现示例
 */
class VirtualScrollManager {
  /** 容器高度 */
  private containerHeight: number = 0;
  /** 滚动位置 */
  private scrollTop: number = 0;
  /** 总行数 */
  private totalRows: number = 0;
  /** 行高（固定行高场景） */
  private rowHeight: number = 40;
  /** 缓冲区行数 */
  private bufferSize: number = 5;

  /**
   * 计算可见范围
   */
  getVisibleRange(): { startIndex: number; endIndex: number } {
    const startIndex = Math.max(
      0,
      Math.floor(this.scrollTop / this.rowHeight) - this.bufferSize
    );

    const visibleCount = Math.ceil(this.containerHeight / this.rowHeight);
    const endIndex = Math.min(
      this.totalRows - 1,
      Math.floor(this.scrollTop / this.rowHeight) + visibleCount + this.bufferSize
    );

    return { startIndex, endIndex };
  }

  /**
   * 计算偏移量（用于滚动条）
   */
  getScrollHeight(): number {
    return this.totalRows * this.rowHeight;
  }

  /**
   * 计算偏移量（用于定位可见行）
   */
  getOffsetY(index: number): number {
    return index * this.rowHeight;
  }
}

/**
 * React 虚拟滚动 Hook
 */
function useVirtualScroll(
  containerRef: React.RefObject<HTMLDivElement>,
  data: unknown[],
  rowHeight: number = 40
): {
  visibleData: unknown[];
  totalHeight: number;
  offsetY: number;
  containerProps: {
    onScroll: (e: React.UIEvent<HTMLDivElement>) => void;
    style: { height: string; overflow: 'auto' };
  };
} {
  const [scrollTop, setScrollTop] = useState(0);
  const [containerHeight, setContainerHeight] = useState(0);

  // 监听容器尺寸变化
  useEffect(() => {
    if (!containerRef.current) return;

    const resizeObserver = new ResizeObserver((entries) => {
      const entry = entries[0];
      setContainerHeight(entry.contentRect.height);
    });

    resizeObserver.observe(containerRef.current);

    return () => resizeObserver.disconnect();
  }, [containerRef]);

  // 处理滚动
  const handleScroll = (e: React.UIEvent<HTMLDivElement>) => {
    setScrollTop(e.currentTarget.scrollTop);
  };

  // 计算可见范围
  const bufferSize = 5;
  const startIndex = Math.max(0, Math.floor(scrollTop / rowHeight) - bufferSize);
  const visibleCount = Math.ceil(containerHeight / rowHeight);
  const endIndex = Math.min(data.length - 1, startIndex + visibleCount + bufferSize * 2);

  // 获取可见数据
  const visibleData = data.slice(startIndex, endIndex + 1);

  // 总高度
  const totalHeight = data.length * rowHeight;

  // 偏移量
  const offsetY = startIndex * rowHeight;

  return {
    visibleData,
    totalHeight,
    offsetY,
    containerProps: {
      onScroll: handleScroll,
      style: { height: '100%', overflow: 'auto' },
    },
  };
}
```
