# 多维表格设计原理与实现完全指南

## 前言

多维表格（Spreadsheet/Grid）是一种强大的数据组织和可视化工具，它允许用户以表格形式存储数据，同时支持公式计算、筛选排序、分组透视等高级功能。与传统电子表格相比，现代多维表格更注重**协同编辑**和**可视化展示**，成为企业级应用和数据管理系统的核心组件。

本章将从实战角度出发，深入剖析多维表格的核心设计原理，包括单元格类型系统、公式引擎、筛选排序机制、分组透视功能以及协同编辑方案。无论你是想构建一个类似 Notion Database、Airtable 或腾讯文档表格的多维表格系统，还是需要在项目中集成表格组件，本指南都将提供全面的技术指导。

## 一、整体架构设计

### 1.2 数据模型设计

多维表格的数据模型是整个系统的基础。一个典型的数据模型需要支持以下结构：

```typescript
/**
 * 多维表格数据模型
 *
 * 设计思路：
 * 1. 采用稀疏矩阵存储，只存储有数据的单元格
 * 2. 每个单元格包含值、公式、样式三个维度
 * 3. 列定义与数据分离，便于扩展和维护
 */

// 单元格数据类型枚举
type CellValueType = 'text' | 'number' | 'boolean' | 'date' | 'select' | 'multiSelect' | 'user' | 'file' | 'formula' | 'link';

// 单元格数据结构
interface CellData {
  /** 单元格原始值 */
  v: string | number | boolean | null;
  /** 单元格显示值（公式计算后） */
  displayValue?: string | number | boolean | null;
  /** 单元格公式（如果有） */
  formula?: string;
  /** 单元格类型 */
  type: CellValueType;
  /** 单元格样式ID */
  styleId?: string;
  /** 单元格元数据 */
  meta?: Record<string, unknown>;
  /** 备注/注释 */
  comment?: string;
  /** 链接到其他单元格的引用 */
  references?: CellReference[];
  /** 最后修改时间 */
  updatedAt: number;
  /** 最后修改人 */
  updatedBy?: string;
}

/**
 * 单元格引用 - 用于追踪公式依赖关系
 * 例如：A1 单元格引用了 B1 和 C1，
 * 当 B1 或 C1 变化时，A1 需要重新计算
 */
interface CellReference {
  /** 引用的工作表名称 */
  sheetName: string;
  /** 引用的列 */
  col: number;
  /** 引用的行 */
  row: number;
}

/**
 * 列定义结构
 *
 * 设计思路：
 * 1. 列定义包含了该列的所有元信息
 * 2. 支持多种列类型，每种类型有不同的属性
 * 3. 验证规则用于数据校验
 */
interface ColumnDefinition {
  /** 列ID，唯一标识 */
  id: string;
  /** 列名称，用户可见 */
  name: string;
  /** 列类型 */
  type: CellValueType;
  /** 列宽度 */
  width: number;
  /** 是否隐藏 */
  hidden: boolean;
  /** 是否锁定（锁定后不可编辑） */
  locked: boolean;
  /** 列顺序 */
  order: number;
  /** 验证规则 */
  validation?: ValidationRule;
  /** 格式化配置 */
  format?: CellFormat;
  /** 枚举值（用于 select/multiSelect 类型） */
  options?: SelectOption[];
  /** 默认值 */
  defaultValue?: unknown;
  /** 权限配置 */
  permission?: ColumnPermission;
}

/**
 * 验证规则 - 用于数据校验
 * 例如：数字列可以设置最小值、最大值
 * 文本列可以设置正则表达式
 */
interface ValidationRule {
  /** 验证类型 */
  type: 'required' | 'min' | 'max' | 'pattern' | 'custom' | 'unique';
  /** 验证失败时的错误信息 */
  errorMessage: string;
  /** 验证值（根据类型不同而不同） */
  value?: unknown;
  /** 自定义验证函数 */
  customValidator?: (value: unknown) => boolean;
}

/**
 * 单元格格式化配置
 * 用于控制单元格的显示格式
 */
interface CellFormat {
  /** 数字格式 */
  number?: {
    /** 小数位数 */
    precision?: number;
    /** 是否使用千分位分隔符 */
    thousandSeparator?: boolean;
    /** 货币符号 */
    prefix?: string;
    /** 后缀 */
    suffix?: string;
  };
  /** 日期格式 */
  date?: {
    /** 日期格式字符串，如 'YYYY-MM-DD' */
    format: string;
    /** 是否显示时间 */
    showTime?: boolean;
  };
  /** 百分比格式 */
  percent?: {
    /** 小数位数 */
    precision?: number;
  };
}

/**
 * 选择项选项（用于下拉单选/多选）
 */
interface SelectOption {
  /** 选项ID */
  id: string;
  /** 选项名称 */
  name: string;
  /** 选项颜色 */
  color?: string;
  /** 选项图标 */
  icon?: string;
}

/**
 * 列权限配置
 */
interface ColumnPermission {
  /** 可编辑的用户/角色 */
  editableBy: string[];
  /** 可查看的用户/角色 */
  viewableBy: string[];
}

/**
 * 工作表（Sheet）数据结构
 *
 * 设计思路：
 * 1. 工作表是表格数据的容器
 * 2. 采用行对象数组而非列对象数组存储数据，更符合业务习惯
 * 3. 行数据使用 Map 存储，便于快速查找和更新
 */
interface Sheet {
  /** 工作表ID */
  id: string;
  /** 工作表名称 */
  name: string;
  /** 工作表顺序 */
  order: number;
  /** 是否激活 */
  active: boolean;
  /** 行数据 Map，key 为行ID */
  rows: Map<string, RowData>;
  /** 列定义 Map，key 为列ID */
  columns: Map<string, ColumnDefinition>;
  /** 合并单元格区域列表 */
  merges: MergeRange[];
  /** 冻结行列配置 */
  frozenRows: number;   // 冻结前几行
  frozenCols: number;   // 冻结前几列
  /** 筛选区域 */
  filterArea?: Range;
  /** 排序状态 */
  sortState?: SortState;
  /** 分组状态 */
  groupState?: GroupState;
  /** 视图配置 */
  viewSettings: ViewSettings;
}

/**
 * 行数据结构

  height: number;

/**

  /** 子分组 */

  /** 分组键值 */

  /** 该组包含的行数 */
  count: number;

  expanded: boolean;
  /** 子分组 */
  subgroups?: GroupInfo[];

  type: 'table' | 'kanban' | 'gallery';

  colorField?: string;
```

### 1.3 核心类设计

```typescript
/**
 * 多维表格核心类
 *
 * 设计思路：
 * 1. 采用命令模式封装所有数据修改操作，便于撤销/重做
 * 2. 事件驱动架构，UI 层订阅数据变化事件
 * 3. 计算图追踪依赖，自动确定重新计算顺序
 */

// 事件类型枚举
type SpreadsheetEventType =
  | 'cell:change'      // 单元格值变化
  | 'cell:format'      // 单元格格式变化
  | 'row:add'          // 行添加
  | 'row:delete'       // 行删除
  | 'row:move'         // 行移动
  | 'col:add'          // 列添加
  | 'col:delete'       // 列删除
  | 'col:resize'       // 列宽变化
  | 'selection:change' // 选中区域变化
  | 'sort:change'      // 排序变化
  | 'filter:change'    // 筛选变化
  | 'view:change';     // 视图变化

/**
 * 事件对象
 */
interface SpreadsheetEvent {
  /** 事件类型 */
  type: SpreadsheetEventType;
  /** 事件数据 */
  data: unknown;
  /** 触发时间 */
  timestamp: number;
  /** 触发者ID */
  sourceId: string;
  /** 是否可撤销 */
  undoable: boolean;
}

/**
 * 单元格坐标
 */
interface CellPosition {
  sheetId: string;
  row: number;
  col: number;
}

/**
 * 单元格范围
 */
interface CellRange {
  sheetId: string;
  startRow: number;
  startCol: number;
  endRow: number;
  endCol: number;
}

/**
 * 选择管理器 - 负责管理当前选中区域
 *
 * 设计思路：
 * 1. 支持多选区域（Ctrl+点击）
 * 2. 维护选区的层级结构（主选区、从选区）
 * 3. 选区变化时触发事件，UI 层订阅并更新高亮
 */
class SelectionManager {
  /** 当前激活的工作表ID */
  private activeSheetId: string = '';
  /** 主选区（第一个选中的区域） */
  private primarySelection: CellRange | null = null;
  /** 从选区列表（额外选中的区域） */
  private secondarySelections: CellRange[] = [];
  /** 拖拽选择状态 */
  private isDragSelecting: boolean = false;
  /** 拖拽起始位置 */
  private dragStartCell: CellPosition | null = null;
  /** 事件总线 */
  private eventBus: EventBus;
  /** 事件订阅列表（用于清理） */
  private subscriptions: Map<string, Function> = new Map();

  constructor(eventBus: EventBus) {
    this.eventBus = eventBus;
  }

  /**
   * 设置激活的工作表
   * @param sheetId 工作表ID
   */
  setActiveSheet(sheetId: string): void {
    this.activeSheetId = sheetId;
    // 切换工作表时清空选区
    this.primarySelection = null;
    this.secondarySelections = [];
    this.notifySelectionChange();
  }

  /**
   * 设置主选区
   * @param range 选区范围
   */
  setPrimarySelection(range: CellRange): void {
    if (range.sheetId !== this.activeSheetId) {
      console.warn('选区所在工作表与当前激活工作表不一致');
      return;
    }
    this.primarySelection = this.normalizeRange(range);
    this.notifySelectionChange();
  }

  /**
   * 添加从选区（Ctrl+点击时）
   * @param range 选区范围
   */
  addSecondarySelection(range: CellRange): void {
    if (range.sheetId !== this.activeSheetId) {
      return;
    }
    const normalizedRange = this.normalizeRange(range);
    // 检查是否与现有选区重叠
    const isOverlap = this.secondarySelections.some(sel =>
      this.rangesOverlap(sel, normalizedRange)
    );
    if (!isOverlap) {
      this.secondarySelections.push(normalizedRange);
      this.notifySelectionChange();
    }
  }

  /**
   * 开始拖拽选择
   * @param cell 起始单元格
   */
  startDrag(cell: CellPosition): void {
    if (cell.sheetId !== this.activeSheetId) {
      return;
    }
    this.isDragSelecting = true;
    this.dragStartCell = cell;
    // 拖拽开始时，清空从选区
    this.secondarySelections = [];
    this.primarySelection = {
      sheetId: cell.sheetId,
      startRow: cell.row,
      startCol: cell.col,
      endRow: cell.row,
      endCol: cell.col,
    };
    this.notifySelectionChange();
  }

  /**
   * 更新拖拽选择（鼠标移动时）
   * @param cell 当前单元格
   */
  updateDrag(cell: CellPosition): void {
    if (!this.isDragSelecting || !this.dragStartCell) {
      return;
    }
    if (cell.sheetId !== this.activeSheetId) {
      return;
    }
    this.primarySelection = {
      sheetId: cell.sheetId,
      startRow: Math.min(this.dragStartCell.row, cell.row),
      startCol: Math.min(this.dragStartCell.col, cell.col),
      endRow: Math.max(this.dragStartCell.row, cell.row),
      endCol: Math.max(this.dragStartCell.col, cell.col),
    };
    this.notifySelectionChange();
  }

  /**
   * 结束拖拽选择
   */
  endDrag(): void {
    this.isDragSelecting = false;
    this.dragStartCell = null;
  }

  /**
   * 全选当前工作表
   */
  selectAll(): void {
    if (!this.activeSheetId) {
      return;
    }
    this.primarySelection = {
      sheetId: this.activeSheetId,
      startRow: 0,
      startCol: 0,
      endRow: 10000, // 假设最大10000行
      endCol: 100,   // 假设最大100列
    };
    this.secondarySelections = [];
    this.notifySelectionChange();
  }

  /**
   * 清除所有选区
   */
  clearSelection(): void {
    this.primarySelection = null;
    this.secondarySelections = [];
    this.notifySelectionChange();
  }

  /**
   * 获取主选区
   */
  getPrimarySelection(): CellRange | null {
    return this.primarySelection;
  }

  /**
   * 获取所有选区（包括主选区和从选区）
   */
  getAllSelections(): CellRange[] {
    const selections: CellRange[] = [];
    if (this.primarySelection) {
      selections.push(this.primarySelection);
    }
    selections.push(...this.secondarySelections);
    return selections;
  }

  /**
   * 判断单元格是否在选区中
   * @param cell 单元格位置
   */
  isCellSelected(cell: CellPosition): boolean {
    if (cell.sheetId !== this.activeSheetId) {
      return false;
    }
    if (this.primarySelection && this.isCellInRange(cell, this.primarySelection)) {
      return true;
    }
    return this.secondarySelections.some(sel => this.isCellInRange(cell, sel));
  }

  /**
   * 获取选中的所有单元格坐标
   */
  getSelectedCells(): CellPosition[] {
    const cells: CellPosition[] = [];
    const processRange = (range: CellRange) => {
      for (let row = range.startRow; row <= range.endRow; row++) {
        for (let col = range.startCol; col <= range.endCol; col++) {
          cells.push({ sheetId: range.sheetId, row, col });
        }
      }
    };
    if (this.primarySelection) {
      processRange(this.primarySelection);
    }
    this.secondarySelections.forEach(processRange);
    return cells;
  }

  // --- 私有辅助方法 ---

  /**
   * 规范化范围（确保 start <= end）
   */
  private normalizeRange(range: CellRange): CellRange {
    return {
      sheetId: range.sheetId,
      startRow: Math.min(range.startRow, range.endRow),
      startCol: Math.min(range.startCol, range.endCol),
      endRow: Math.max(range.startRow, range.endRow),
      endCol: Math.max(range.startCol, range.endCol),
    };
  }

  /**
   * 判断两个范围是否重叠
   */
  private rangesOverlap(a: CellRange, b: CellRange): boolean {
    return !(
      a.endCol < b.startCol ||
      a.startCol > b.endCol ||
      a.endRow < b.startRow ||
      a.startRow > b.endRow
    );
  }

  /**
   * 判断单元格是否在范围内
   */
  private isCellInRange(cell: CellPosition, range: CellRange): boolean {
    return (
      cell.row >= range.startRow &&
      cell.row <= range.endRow &&
      cell.col >= range.startCol &&
      cell.col <= range.endCol
    );
  }

  /**
   * 通知选区变化
   */
  private notifySelectionChange(): void {
    this.eventBus.emit({
      type: 'selection:change',
      data: this.getAllSelections(),
      timestamp: Date.now(),
      sourceId: 'SelectionManager',
      undoable: false,
    });
  }
}

/**
 * 剪贴板管理器 - 负责处理复制/粘贴/剪切操作
 *
 * 设计思路：
 * 1. 支持普通粘贴和特殊粘贴（只粘贴格式、只粘贴公式等）
 * 2. 粘贴时处理相对引用和绝对引用
 * 3. 支持跨工作表、跨文档粘贴
 */
class ClipboardManager {
  /** 剪贴板数据 */
  private clipboardData: {
    /** 复制的单元格数据 */
    cells: Map<string, CellData>;
    /** 复制的范围 */
    range: CellRange;
    /** 是否是剪切 */
    isCut: boolean;
    /** 复制的原始位置 */
    sourceRange: CellRange;
  } | null = null;

  /** 剪贴板格式类型 */
  private clipboardFormat: 'cell' | 'value' | 'formula' | 'format' = 'cell';

  /**
   * 复制选区内容
   * @param selections 选区列表
   * @param sheet 当前工作表
   */
  copy(selections: CellRange[], sheet: Sheet): void {
    if (selections.length === 0) {
      return;
    }

    const primarySelection = selections[0];
    const cells = new Map<string, CellData>();

    // 遍历选区收集单元格数据
    for (let row = primarySelection.startRow; row <= primarySelection.endRow; row++) {
      for (let col = primarySelection.startCol; col <= primarySelection.endCol; col++) {
        const cellId = this.getCellId(row, col);
        const cell = sheet.rows.get(String(row))?.cells.get(String(col));
        if (cell) {
          cells.set(cellId, { ...cell });
        }
      }
    }

    this.clipboardData = {
      cells,
      range: primarySelection,
      isCut: false,
      sourceRange: { ...primarySelection },
    };
  }

  /**
   * 剪切选区内容
   * @param selections 选区列表
   * @param sheet 当前工作表
   */
  cut(selections: CellRange[], sheet: Sheet): void {
    this.copy(selections, sheet);
    if (this.clipboardData) {
      this.clipboardData.isCut = true;
    }
  }

  /**
   * 粘贴内容
   * @param targetRow 目标起始行
   * @param targetCol 目标起始列
   * @param pasteType 粘贴类型
   * @param sheet 当前工作表
   */
  paste(
    targetRow: number,
    targetCol: number,
    pasteType: 'all' | 'value' | 'formula' | 'format' = 'all',
    sheet: Sheet
  ): { updatedCells: CellPosition[]; deletedCells?: CellPosition[] } {
    if (!this.clipboardData) {
      return { updatedCells: [] };
    }

    const { cells, range, isCut, sourceRange } = this.clipboardData;
    const updatedCells: CellPosition[] = [];
    const deletedCells: CellPosition[] = [];

    // 计算行列偏移量
    const rowOffset = targetRow - range.startRow;
    const colOffset = targetCol - range.startCol;

    // 如果是剪切，先清除原位置数据
    if (isCut) {
      for (let row = sourceRange.startRow; row <= sourceRange.endRow; row++) {
        for (let col = sourceRange.startCol; col <= sourceRange.endCol; col++) {
          deletedCells.push({ sheetId: range.sheetId, row, col });
        }
      }
    }

    // 粘贴到目标位置
    cells.forEach((cellData, cellId) => {
      const [row, col] = cellId.split(':').map(Number);
      const newRow = row + rowOffset;
      const newCol = col + colOffset;

      const newCell: CellData = { ...cellData };

      // 根据粘贴类型决定要复制的内容
      if (pasteType === 'value') {
        newCell.formula = undefined;
      } else if (pasteType === 'formula') {
        // 只复制公式，值需要重新计算
        if (!newCell.formula) {
          return; // 如果没有公式，不复制
        }
      } else if (pasteType === 'format') {
        // 只复制格式，清空值
        return; // 格式粘贴需要单独处理
      }

      // 处理公式中的相对引用
      if (newCell.formula) {
        newCell.formula = this.adjustFormulaReferences(
          newCell.formula,
          rowOffset,
          colOffset
        );
      }

      updatedCells.push({ sheetId: range.sheetId, row: newRow, col: newCol });
    });

    // 返回更新和删除的单元格列表
    return {
      updatedCells,
      deletedCells: isCut ? deletedCells : undefined,
    };
  }

  /**
   * 获取剪贴板内容
   */
  getClipboardContent() {
    return this.clipboardData;
  }

  /**
   * 检查剪贴板是否为空
   */
  isEmpty(): boolean {
    return this.clipboardData === null;
  }

  /**
   * 调整公式中的相对引用
   *
   * 例如：如果从 A1:B2 复制到 C3:D4，
   * 则公式中的 A1 应该变成 C3
   *
   * @param formula 原始公式
   * @param rowOffset 行偏移量
   * @param colOffset 列偏移量
   */
  private adjustFormulaReferences(
    formula: string,
    rowOffset: number,
    colOffset: number
  ): string {
    // 使用正则表达式匹配单元格引用
    // 支持 A1, $A1, A$1, $A$1 等形式
    const cellRefRegex = /\$?([A-Z]+)\$?(\d+)/g;

    return formula.replace(cellRefRegex, (match, colStr, rowStr) => {
      // 解析列字母为数字
      const col = this.columnLetterToIndex(colStr);
      const row = parseInt(rowStr, 10);

      // 计算新的行列位置
      const newCol = col + colOffset;
      const newRow = row + rowOffset;

      // 检查是否是绝对引用
      const isAbsoluteCol = match.startsWith('$');
      const isAbsoluteRow = match.includes('$') && match.lastIndexOf('$') > 0;

      // 转换回字母形式
      const newColStr = this.columnIndexToLetter(
        isAbsoluteCol ? col : newCol
      );
      const newRowStr = isAbsoluteRow ? rowStr : String(newRow);

      return `${isAbsoluteCol ? '$' : ''}${newColStr}${isAbsoluteRow ? '$' : ''}${newRowStr}`;
    });
  }

  /**
   * 列字母转数字索引
   * A -> 0, B -> 1, ..., Z -> 25, AA -> 26, ...
   */
  private columnLetterToIndex(letters: string): number {
    let index = 0;
    for (let i = 0; i < letters.length; i++) {
      index = index * 26 + (letters.charCodeAt(i) - 64);
    }
    return index - 1;
  }

  /**
   * 数字索引转列字母
   * 0 -> A, 1 -> B, ..., 25 -> Z, 26 -> AA, ...
   */
  private columnIndexToLetter(index: number): string {
    let letters = '';
    while (index >= 0) {
      letters = String.fromCharCode((index % 26) + 65) + letters;
      index = Math.floor(index / 26) - 1;
    }
    return letters;
  }

  /**
   * 生成单元格ID
   */
  private getCellId(row: number, col: number): string {
    return `${row}:${col}`;
  }
}

/**
 * 撤销/重做管理器 - 使用命令模式
 *
 * 设计思路：
 * 1. 每个操作都封装为一个命令对象
 * 2. 命令对象包含 execute（执行）和 undo（撤销）方法
 * 3. 使用两个栈分别存储已执行和已撤销的命令
 * 4. 支持宏命令（将多个命令组合为一个）
 */
class UndoRedoManager {
  /** 已执行命令栈 */
  private executeStack: Command[] = [];
  /** 已撤销命令栈 */
  private undoneStack: Command[] = [];
  /** 最大历史记录数 */
  private maxHistorySize: number = 100;
  /** 是否正在执行宏命令 */
  private isExecutingMacro: boolean = false;
  /** 当前宏命令缓冲区 */
  private macroBuffer: Command[] = [];

  /**
   * 执行命令
   * @param command 要执行的命令
   */
  execute(command: Command): void {
    if (this.isExecutingMacro) {
      // 宏命令执行中，将命令加入缓冲区
      this.macroBuffer.push(command);
    } else {
      // 普通执行
      command.execute();
      this.executeStack.push(command);

      // 清除已撤销栈（新命令执行后，重做历史失效）
      this.undoneStack = [];

      // 限制历史记录数量
      if (this.executeStack.length > this.maxHistorySize) {
        this.executeStack.shift();
      }
    }
  }

  /**
   * 撤销最近一个命令
   * @returns 是否成功撤销
   */
  undo(): boolean {
    const command = this.executeStack.pop();
    if (command) {
      command.undo();
      this.undoneStack.push(command);
      return true;
    }
    return false;
  }

  /**
   * 重做最近一个撤销的命令
   * @returns 是否成功重做
   */
  redo(): boolean {
    const command = this.undoneStack.pop();
    if (command) {
      command.execute();
      this.executeStack.push(command);
      return true;
    }
    return false;
  }

  /**
   * 开始宏命令
   * 之后的命令将被组合为一个撤销单位
   */
  beginMacro(): void {
    this.isExecutingMacro = true;
    this.macroBuffer = [];
  }

  /**
   * 结束宏命令
   * 将缓冲区中的命令组合为一个宏命令并执行
   * @returns 组合后的宏命令
   */
  endMacro(): Command | null {
    this.isExecutingMacro = false;

    if (this.macroBuffer.length === 0) {
      return null;
    }

    const macro = new MacroCommand(this.macroBuffer);
    this.executeStack.push(macro);
    this.undoneStack = [];

    return macro;
  }

  /**
   * 清空所有历史记录
   */
  clear(): void {
    this.executeStack = [];
    this.undoneStack = [];
  }

  /**
   * 检查是否可以撤销
   */
  canUndo(): boolean {
    return this.executeStack.length > 0;
  }

  /**
   * 检查是否可以重做
   */
  canRedo(): boolean {
    return this.undoneStack.length > 0;
  }
}

/**
 * 命令接口
 */
interface Command {
  /** 执行命令 */
  execute(): void;
  /** 撤销命令 */
  undo(): void;
}

/**
 * 设置单元格值命令
 */
class SetCellValueCommand implements Command {
  private sheet: Sheet;
  private row: number;
  private col: number;
  private oldValue: CellData | null;
  private newValue: CellData;
  private formulaEngine: FormulaEngine;

  constructor(
    sheet: Sheet,
    row: number,
    col: number,
    newValue: CellData,
    formulaEngine: FormulaEngine
  ) {
    this.sheet = sheet;
    this.row = row;
    this.col = col;
    this.newValue = newValue;
    this.formulaEngine = formulaEngine;

    // 获取旧值
    const rowData = sheet.rows.get(String(row));
    this.oldValue = rowData?.cells.get(String(col)) || null;
  }

  execute(): void {
    const rowData = this.sheet.rows.get(String(this.row));
    if (rowData) {
      rowData.cells.set(String(this.col), this.newValue);
      // 如果新值包含公式，重新计算
      if (this.newValue.formula) {
        this.formulaEngine.markForRecalculation(
          this.sheet.id,
          this.row,
          this.col
        );
      }
    }
  }

  undo(): void {
    const rowData = this.sheet.rows.get(String(this.row));
    if (rowData) {
      if (this.oldValue) {
        rowData.cells.set(String(this.col), this.oldValue);
      } else {
        rowData.cells.delete(String(this.col));
      }
    }
  }
}

/**
 * 批量设置单元格值命令（用于粘贴等操作）
 */
class BatchSetCellsCommand implements Command {
  private sheet: Sheet;
  private changes: Array<{
    row: number;
    col: number;
    oldValue: CellData | null;
    newValue: CellData;
  }>;
  private formulaEngine: FormulaEngine;

  constructor(
    sheet: Sheet,
    changes: Array<{
      row: number;
      col: number;
      newValue: CellData;
    }>,
    formulaEngine: FormulaEngine
  ) {
    this.sheet = sheet;
    this.changes = changes.map(change => ({
      ...change,
      oldValue: sheet.rows.get(String(change.row))?.cells.get(String(change.col)) || null,
    }));
    this.formulaEngine = formulaEngine;
  }

  execute(): void {
    const recalcCells: Array<{ sheetId: string; row: number; col: number }> = [];

    this.changes.forEach(change => {
      const rowData = this.sheet.rows.get(String(change.row));
      if (rowData) {
        rowData.cells.set(String(change.col), change.newValue);
        if (change.newValue.formula) {
          recalcCells.push({
            sheetId: this.sheet.id,
            row: change.row,
            col: change.col,
          });
        }
      }
    });

    // 批量重新计算
    recalcCells.forEach(cell => {
      this.formulaEngine.markForRecalculation(cell.sheetId, cell.row, cell.col);
    });
  }

  undo(): void {
    this.changes.forEach(change => {
      const rowData = this.sheet.rows.get(String(change.row));
      if (rowData) {
        if (change.oldValue) {
          rowData.cells.set(String(change.col), change.oldValue);
        } else {
          rowData.cells.delete(String(change.col));
        }
      }
    });
  }
}

/**
 * 宏命令 - 将多个命令组合为一个
 */
class MacroCommand implements Command {
  private commands: Command[];

  constructor(commands: Command[]) {
    this.commands = commands;
  }

  execute(): void {
    this.commands.forEach(cmd => cmd.execute());
  }

  undo(): void {
    // 逆向撤销
    for (let i = this.commands.length - 1; i >= 0; i--) {
      this.commands[i].undo();
    }
  }
}
```

```typescript
/**
 * 单元格类型注册表类

class CellTypeRegistry {
  /** 类型注册表 Map */
  private types: Map<CellValueType, CellTypeConfig> = new Map();
  /** 类型别名映射 */
  private aliases: Map<string, CellValueType> = new Map();

  constructor() {
    // 注册内置类型
    this.registerBuiltinTypes();
  }

  /**
   * 注册内置单元格类型
   */
  private registerBuiltinTypes(): void {
    // 文本类型
    this.register({
      type: 'text',
      label: '文本',
      icon: '📝',
      renderer: TextCellRenderer,
      editor: TextCellEditor,
      defaultValueFactory: () => '',
      validator: (value) => {
        if (typeof value !== 'string' && value !== null && value !== undefined) {
          return { valid: false, errorMessage: '值必须是文本类型' };
        }
        return { valid: true };
      },
      formatter: (value) => String(value ?? ''),
      serializer: (value) => value,
      deserializer: (data) => data,
      comparator: (a, b) => String(a).localeCompare(String(b)),
    });

    // 数字类型
    this.register({
      type: 'number',
      label: '数字',
      icon: '🔢',
      renderer: NumberCellRenderer,
      editor: NumberCellEditor,
      defaultValueFactory: () => 0,
      validator: (value, context) => {
        if (value === null || value === undefined || value === '') {
          return { valid: true }; // 空值通过验证
        }
        const num = Number(value);
        if (isNaN(num)) {
          return { valid: false, errorMessage: '值必须是数字' };
        }
        // 检查列定义的验证规则
        if (context.column.validation) {
          const rule = context.column.validation;
          if (rule.type === 'min' && num < (rule.value as number)) {
            return { valid: false, errorMessage: rule.errorMessage };
          }
          if (rule.type === 'max' && num > (rule.value as number)) {
            return { valid: false, errorMessage: rule.errorMessage };
          }
        }
        return { valid: true };
      },
      formatter: (value, format) => {
        if (value === null || value === undefined) return '';
        const num = Number(value);
        if (isNaN(num)) return String(value);

        if (format?.number) {
          const { precision, thousandSeparator, prefix, suffix } = format.number;
          let formatted = num.toFixed(precision ?? 2);
          if (thousandSeparator) {
            const parts = formatted.split('.');
            parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ',');
            formatted = parts.join('.');
          }
          return `${prefix ?? ''}${formatted}${suffix ?? ''}`;
        }
        return String(num);
      },
      serializer: (value) => (value === '' ? null : Number(value)),
      deserializer: (data) => (data === null ? null : Number(data)),
      comparator: (a, b) => Number(a) - Number(b),
      aggregator: (values) => {
        const nums = values.filter(v => v !== null && v !== undefined).map(Number);
        return nums.reduce((sum, n) => sum + n, 0);
      },
    });

    // 布尔类型
    this.register({
      type: 'boolean',
      label: '复选框',
      icon: '✅',
      renderer: CheckboxCellRenderer,
      editor: CheckboxCellEditor,
      defaultValueFactory: () => false,
      formatter: (value) => (value ? '是' : '否'),
      serializer: (value) => Boolean(value),
      deserializer: (data) => Boolean(data),
      comparator: (a, b) => (Boolean(a) === Boolean(b) ? 0 : Boolean(a) ? 1 : -1),
    });

    // 日期类型
    this.register({
      type: 'date',
      label: '日期',
      icon: '📅',
      renderer: DateCellRenderer,
      editor: DateCellEditor,
      defaultValueFactory: () => null,
      validator: (value) => {
        if (value === null || value === undefined || value === '') {
          return { valid: true };
        }
        const date = new Date(value as string);
        if (isNaN(date.getTime())) {
          return { valid: false, errorMessage: '日期格式不正确' };
        }
        return { valid: true };
      },
      formatter: (value, format) => {
        if (!value) return '';
        const date = new Date(value as string);
        if (isNaN(date.getTime())) return String(value);
        if (format?.date) {
          return formatDate(date, format.date.format);
        }
```

## 三、公式系统

### 3.1 公式引擎概述

公式系统是多维表格的核心功能之一，它允许用户使用公式对数据进行计算、分析和处理。一个强大的公式系统需要支持：

1. **丰富的内置函数**：数学、统计、文本、日期、逻辑、查找等
2. **单元格引用**：相对引用、绝对引用、跨工作表引用
3. **数组公式**：支持数组操作和广播
4. **自定义函数**：允许用户注册自定义函数
5. **依赖追踪**：自动追踪公式依赖关系，优化重新计算
6. **错误处理**：优雅地处理各种错误情况

```typescript
/**
 * 词法分析器
 * 将公式字符串转换为Token流
 */
class FormulaLexer {
  private formula: string;
  private position: number = 0;
  private currentChar: string | null;

  constructor(formula: string) {
    // 去掉前导的 = 号
    this.formula = formula.startsWith('=') ? formula.slice(1) : formula;
    this.currentChar = this.formula.length > 0 ? this.formula[0] : null;
  }

  /**
   * 获取下一个Token
   */
  nextToken(): Token {
    // 跳过空白字符
    while (this.isWhitespace(this.currentChar)) {
      this.advance();
    }

    // 到达字符串末尾
    if (this.currentChar === null) {
      return { type: TokenType.EOF, value: '', position: this.position };
    }

    const pos = this.position;

    // 数字
    if (this.isDigit(this.currentChar) ||
        (this.currentChar === '.' && this.isDigit(this.peek()))) {
      return this.readNumber();
    }

    // 字符串（双引号）
    if (this.currentChar === '"') {
      return this.readString();
    }

    // 字符串（单引号，用于某些本地化）
    if (this.currentChar === "'") {
      return this.readSingleQuotedString();
    }

    // 单元格引用或函数名
    if (this.isLetter(this.currentChar) || this.currentChar === '_') {
      return this.readIdentifier();
    }

    // 运算符和标点符号
    return this.readOperator();
  }

  /**
   * 读取数字
   */
  private readNumber(): Token {
    const start = this.position;
    let hasDecimal = false;

    while (this.currentChar !== null) {
      if (this.isDigit(this.currentChar)) {
        this.advance();
      } else if (this.currentChar === '.' && !hasDecimal) {
        hasDecimal = true;
        this.advance();
      } else if (this.currentChar === 'E' || this.currentChar === 'e') {
        // 科学计数法，如 1.5E+10
        this.advance();
        if (this.currentChar === '+' || this.currentChar === '-') {
          this.advance();
        }
        while (this.isDigit(this.currentChar)) {
          this.advance();
        }
        break;
      } else {
        break;
      }
    }

    return {
      type: TokenType.Number,
      value: this.formula.slice(start, this.position),
      position: start,
    };
  }

  /**
   * 读取字符串（双引号）
   */
  private readString(): Token {
    const start = this.position;
    this.advance(); // 跳过开始的双引号

    while (this.currentChar !== null && this.currentChar !== '"') {
      // 处理转义字符
      if (this.currentChar === '\\' && this.peek() === '"') {
        this.advance();
        this.advance();
      } else {
        this.advance();
      }
    }

    if (this.currentChar === '"') {
      this.advance(); // 跳过结束的双引号
    }

    return {
      type: TokenType.String,
      value: this.formula.slice(start + 1, this.position - 1),
      position: start,
    };
  }

  /**
   * 读取单引号字符串（用于工作表名称）
   */
  private readSingleQuotedString(): Token {
    const start = this.position;
    this.advance(); // 跳过开始的单引号

    while (this.currentChar !== null && this.currentChar !== "'") {
      this.advance();
    }

    if (this.currentChar === "'") {
      this.advance(); // 跳过结束的单引号
    }

    return {
      type: TokenType.String,
      value: this.formula.slice(start + 1, this.position - 1),
      position: start,
    };
  }

  /**
   * 读取标识符（单元格引用、函数名）
   */
  private readIdentifier(): Token {
    const start = this.position;

    // 读取字母部分（函数名或列标识）
    while (this.currentChar !== null &&
           (this.isLetter(this.currentChar) ||
            this.isDigit(this.currentChar) ||
            this.currentChar === '_' ||
            this.currentChar === '$')) {
      this.advance();
    }

    // 检查是否是单元格引用（A1, $A$1, A$1, $A1）
    if (this.currentChar !== null && this.isDigit(this.currentChar)) {
      // 读取数字部分（行号）
      while (this.currentChar !== null && this.isDigit(this.currentChar)) {
        this.advance();
      }

      // 检查是否有范围运算符（冒号）
      if (this.currentChar === ':') {
        // 这是一个范围引用，如 A1:B3
        this.advance();
        // 读取范围结束部分
        while (this.currentChar !== null &&
               (this.isLetter(this.currentChar) ||
                this.currentChar === '$')) {
          this.advance();
        }
        while (this.currentChar !== null && this.isDigit(this.currentChar)) {
          this.advance();
        }

        return {
          type: TokenType.Range,
          value: this.formula.slice(start, this.position),
          position: start,
        };
      }

      return {
        type: TokenType.CellRef,
        value: this.formula.slice(start, this.position),
        position: start,
      };
    }

    // 检查是否是布尔值
    const value = this.formula.slice(start, this.position).toUpperCase();
    if (value === 'TRUE') {
      return { type: TokenType.Boolean, value: 'TRUE', position: start };
    }
    if (value === 'FALSE') {
      return { type: TokenType.Boolean, value: 'FALSE', position: start };
    }

    // 否则就是函数名
    return {
      type: TokenType.Function,
      value: this.formula.slice(start, this.position),
      position: start,
    };
  }

  /**
   * 读取运算符
   */
  private readOperator(): Token {
    const pos = this.position;
    const char = this.currentChar!;
    this.advance();

    switch (char) {
      case '+':
      case '-':
      case '*':
      case '/':
      case '^':
      case '&':
        return { type: TokenType.Operator, value: char, position: pos };
      case '=':
        return { type: TokenType.Comparison, value: '=', position: pos };
      case '<':
        if (this.currentChar === '=') {
          this.advance();
          return { type: TokenType.Comparison, value: '<=', position: pos };
        }
        if (this.currentChar === '>') {
          this.advance();
          return { type: TokenType.Comparison, value: '<>', position: pos };
        }
        return { type: TokenType.Comparison, value: '<', position: pos };
      case '>':
        if (this.currentChar === '=') {
          this.advance();
          return { type: TokenType.Comparison, value: '>=', position: pos };
        }
        return { type: TokenType.Comparison, value: '>', position: pos };
      case '!':
        if (this.currentChar === '=') {
          this.advance();
          return { type: TokenType.Comparison, value: '!=', position: pos };
        }
        // 感叹号用于跨工作表引用，如 Sheet1!A1
        // 这里暂时返回感叹号，下一个Token将是单元格引用
        return { type: TokenType.Operator, value: char, position: pos };
      case '(':
        return { type: TokenType.LParen, value: char, position: pos };
      case ')':
        return { type: TokenType.RParen, value: char, position: pos };
      case ',':
        return { type: TokenType.Comma, value: char, position: pos };
      case ':':
        return { type: TokenType.Colon, value: char, position: pos };
      case ';':
        return { type: TokenType.Semicolon, value: char, position: pos };
      default:
        return { type: TokenType.Error, value: char, position: pos };
    }
  }

  /**
   * 向前移动一个字符
   */
  private advance(): void {
    this.position++;
    this.currentChar = this.position < this.formula.length
      ? this.formula[this.position]
      : null;
  }

  /**
   * 查看当前字符但不移动
   */
  private peek(): string | null {
    const nextPos = this.position + 1;
    return nextPos < this.formula.length ? this.formula[nextPos] : null;
  }

  /**
   * 检查是否是空白字符
   */
  private isWhitespace(char: string | null): boolean {
    return char === ' ' || char === '\t' || char === '\r' || char === '\n';
  }

  /**
   * 检查是否是数字
   */
  private isDigit(char: string | null): boolean {
    return char !== null && char >= '0' && char <= '9';
  }

  /**
   * 检查是否是字母
   */
  private isLetter(char: string | null): boolean {
    if (char === null) return false;
    const code = char.charCodeAt(0);
    return (code >= 65 && code <= 90) || (code >= 97 && code <= 122);
  }
}

/**
 * 语法分析器
 * 将Token流转换为AST
 */
class FormulaParser {
  private tokens: Token[] = [];
  private position: number = 0;
  private currentToken: Token;

  constructor(tokens: Token[]) {
    this.tokens = tokens;
    this.currentToken = tokens[0];
  }

  /**
   * 解析公式，返回AST根节点
   */
  parse(): ASTNode {
    if (this.currentToken.type === TokenType.EOF) {
      // 空公式
      return {
        type: ASTNodeType.NumberLiteral,
        value: 0,
      } as NumberLiteralNode;
    }
    return this.parseExpression();
  }

  /**
   * 获取下一个Token
   */
  private advance(): void {
    this.position++;
    this.currentToken = this.position < this.tokens.length
      ? this.tokens[this.position]
      : { type: TokenType.EOF, value: '', position: -1 };
  }

  /**
   * 查看当前Token
   */
  private peek(): Token {
    const nextPos = this.position + 1;
    return nextPos < this.tokens.length
      ? this.tokens[nextPos]
      : { type: TokenType.EOF, value: '', position: -1 };
  }

  /**
   * 解析表达式（处理加减运算）
   * 优先级：加减 < 乘除
   */
  private parseExpression(): ASTNode {
    let left = this.parseTerm();

    while (this.currentToken.type === TokenType.Operator &&
           (this.currentToken.value === '+' || this.currentToken.value === '-')) {
      const operator = this.currentToken.value;
      this.advance();
      const right = this.parseTerm();
      left = {
        type: ASTNodeType.BinaryOperation,
        operator,
        left,
        right,
      } as BinaryOperationNode;
    }

    return left;
  }

  /**
   * 解析项（处理乘除运算）
   */
  private parseTerm(): ASTNode {
    let left = this.parsePower();

    while (this.currentToken.type === TokenType.Operator &&
           (this.currentToken.value === '*' || this.currentToken.value === '/')) {
      const operator = this.currentToken.value;
      this.advance();
      const right = this.parsePower();
      left = {
        type: ASTNodeType.BinaryOperation,
        operator,
        left,
        right,
      } as BinaryOperationNode;
    }

    return left;
  }

  /**
   * 解析幂运算
   */
  private parsePower(): ASTNode {
    let left = this.parseUnary();

    if (this.currentToken.type === TokenType.Operator &&
        this.currentToken.value === '^') {
      this.advance();
      const right = this.parsePower(); // 幂运算右结合
      return {
        type: ASTNodeType.BinaryOperation,
        operator: '^',
        left,
        right,
      } as BinaryOperationNode;
    }

    return left;
  }

  /**
   * 解析一元运算
   */
  private parseUnary(): ASTNode {
    if (this.currentToken.type === TokenType.Operator &&
        (this.currentToken.value === '-' || this.currentToken.value === '+')) {
      const operator = this.currentToken.value;
      this.advance();
      const operand = this.parseUnary();
      return {
        type: ASTNodeType.UnaryOperation,
        operator,
        operand,
      } as UnaryOperationNode;
    }

    return this.parsePrimary();
  }

  /**
   * 解析基本元素
   */
  private parsePrimary(): ASTNode {
    switch (this.currentToken.type) {
      case TokenType.Number:
        const numValue = parseFloat(this.currentToken.value);
        this.advance();
        return {
          type: ASTNodeType.NumberLiteral,
          value: numValue,
        } as NumberLiteralNode;

      case TokenType.String:
        const strValue = this.currentToken.value;
        this.advance();
        return {
          type: ASTNodeType.StringLiteral,
          value: strValue,
        } as StringLiteralNode;

      case TokenType.Boolean:
        const boolValue = this.currentToken.value.toUpperCase() === 'TRUE';
        this.advance();
        return {
          type: ASTNodeType.BooleanLiteral,
          value: boolValue,
        } as BooleanLiteralNode;

      case TokenType.CellRef:
        return this.parseCellReference();

      case TokenType.Range:
        return this.parseRangeReference();

      case TokenType.LParen:
        this.advance(); // 跳过左括号
        const expr = this.parseExpression();
        if (this.currentToken.type === TokenType.RParen) {
          this.advance(); // 跳过右括号
        }
        return expr;

      case TokenType.Function:
        return this.parseFunctionCall();

      case TokenType.Error:
        this.advance();
        return {
          type: ASTNodeType.ErrorValue,
          errorType: this.currentToken.value,
        } as ErrorValueNode;

      default:
        // 未知 token，返回错误
        return {
          type: ASTNodeType.ErrorValue,
          errorType: '#VALUE!',
        } as ErrorValueNode;
    }
  }

  /**
   * 解析单元格引用
   */
  private parseCellReference(): ASTNode {
    const cellRef = this.currentToken.value;
    this.advance();

    // 检查是否有工作表引用（感叹号）
    let sheetName: string | undefined;
    if (this.currentToken.type === TokenType.Operator &&
        this.currentToken.value === '!') {
      this.advance(); // 跳过感叹号
      sheetName = cellRef;

      // 下一个 token 应该是单元格引用
      if (this.currentToken.type === TokenType.CellRef) {
        const newRef = this.currentToken.value;
        this.advance();
        return this.createCellReference(newRef, sheetName);
      }
    }

    return this.createCellReference(cellRef, sheetName);
  }

  /**
   * 创建单元格引用节点
   */
  private createCellReference(ref: string, sheetName?: string): CellReferenceNode {
    // 解析绝对引用
    const isAbsoluteCol = ref.includes('$');
    const isAbsoluteRow = ref.includes('$');

    // 移除 $ 符号
    const cleanRef = ref.replace(/\$/g, '');

    // 解析列和行
    const match = cleanRef.match(/^([A-Z]+)(\d+)$/i);
    if (!match) {
      return {
        type: ASTNodeType.ErrorValue,
        errorType: '#REF!',
      } as ErrorValueNode;
    }

    const colStr = match[1].toUpperCase();
    const row = parseInt(match[2], 10);

    return {
      type: ASTNodeType.CellReference,
      sheetName,
      column: colStr,
      columnIndex: this.columnLetterToIndex(colStr),
      row,
      isAbsoluteCol,
      isAbsoluteRow,
    };
  }

  /**
   * 解析范围引用
   */
  private parseRangeReference(): ASTNode {
    const rangeStr = this.currentToken.value;
    this.advance();

    const parts = rangeStr.split(':');
    if (parts.length !== 2) {
      return {
        type: ASTNodeType.ErrorValue,
        errorType: '#REF!',
      } as ErrorValueNode;
    }

    const startRef = this.createCellReference(parts[0]);
    const endRef = this.createCellReference(parts[1]);

    return {
      type: ASTNodeType.RangeReference,
      start: startRef,
      end: endRef,
    } as RangeReferenceNode;
  }

  /**
   * 解析函数调用
   */
  private parseFunctionCall(): ASTNode {
    const funcName = this.currentToken.value.toUpperCase();
    this.advance(); // 跳过函数名

    // 期望左括号
    if (this.currentToken.type !== TokenType.LParen) {
      return {
        type: ASTNodeType.ErrorValue,
        errorType: '#NAME?',
      } as ErrorValueNode;
    }
    this.advance(); // 跳过左括号

    // 解析参数列表
    const args: ASTNode[] = [];

    // 处理空参数列表
    if (this.currentToken.type !== TokenType.RParen) {
      args.push(this.parseExpression());

      // 解析后续参数
      while (this.currentToken.type === TokenType.Comma ||
             this.currentToken.type === TokenType.Semicolon) {
        // 某些本地化版本使用分号作为分隔符
        if (this.currentToken.type === TokenType.Semicolon) {
          // 将分号转换为逗号处理
        }
        this.advance(); // 跳过逗号或分号
        args.push(this.parseExpression());
      }
    }

    // 期望右括号
    if (this.currentToken.type === TokenType.RParen) {
      this.advance(); // 跳过右括号
    }

    return {
      type: ASTNodeType.FunctionCall,
      name: funcName,
      arguments: args,
    } as FunctionCallNode;
  }

  /**
   * 列字母转数字索引
   */
  private columnLetterToIndex(letters: string): number {
    let index = 0;
    for (let i = 0; i < letters.length; i++) {
      index = index * 26 + (letters.charCodeAt(i) - 64);
    }
    return index - 1;
  }
}

/**
 * 列索引转字母
 */
function columnIndexToLetter(index: number): string {
  let letters = '';
  let idx = index;
  while (idx >= 0) {
    letters = String.fromCharCode((idx % 26) + 65) + letters;
    idx = Math.floor(idx / 26) - 1;
  }
  return letters;
}
```

```typescript
/**
 * 公式计算上下文
 * 包含计算公式所需的所有信息
 */
interface EvaluationContext {

  sheet: Sheet;
  /** 所有工作表（用于跨工作表引用） */
  allSheets: Map<string, Sheet>;

/**
 * 公式引擎核心类
 */
class FormulaEngine {
  /** 内置函数注册表 */
  private functions: Map<string, Function> = new Map();
  /** 依赖图：单元格 -> 被它依赖的单元格 */
  private dependencyGraph: Map<string, Set<string>> = new Map();
  /** 反向依赖图：单元格 -> 依赖它的单元格 */
  private reverseDependencyGraph: Map<string, Set<string>> = new Map();
  /** 待重新计算的单元格队列 */
  private recalculationQueue: Set<string> = new Set();
  /** 当前计算版本 */
  private currentVersion: number = 0;

  constructor() {
    this.registerBuiltinFunctions();
  }

  /**

  /**
   * 注册函数
   */
  registerFunction(name: string, fn: Function): void {
    this.functions.set(name.toUpperCase(), fn);
  }

  /**
   * 解析公式字符串
   */
  parse(formula: string): ASTNode {
    const lexer = new FormulaLexer(formula);
    const tokens: Token[] = [];

    // 词法分析
    let token = lexer.nextToken();
    while (token.type !== TokenType.EOF) {
      tokens.push(token);
      token = lexer.nextToken();
    }

    // 语法分析
    const parser = new FormulaParser(tokens);
    return parser.parse();
  }

  /**
   * 计算公式
   */
  evaluate(
    formula: string,
    context: EvaluationContext
  ): { value: unknown; error?: string } {
    try {
      const ast = this.parse(formula);
      const value = this.evaluateNode(ast, context);
      return { value };
    } catch (error) {
      return {
        value: null,
        error: error instanceof Error ? error.message : '#VALUE!'
      };
    }
  }

  /**
   * 评估AST节点
   */
  private evaluateNode(node: ASTNode, context: EvaluationContext): unknown {
    switch (node.type) {
      case ASTNodeType.NumberLiteral:
        return (node as NumberLiteralNode).value;

      case ASTNodeType.StringLiteral:
        return (node as StringLiteralNode).value;

      case ASTNodeType.BooleanLiteral:
        return (node as BooleanLiteralNode).value;

      case ASTNodeType.CellReference: {
        const refNode = node as CellReferenceNode;
        return this.getCellValue(refNode, context);
      }

      case ASTNodeType.RangeReference: {
        const rangeNode = node as RangeReferenceNode;
        return this.getRangeValues(rangeNode, context);
      }

      case ASTNodeType.BinaryOperation: {
        const binNode = node as BinaryOperationNode;
        return this.evaluateBinaryOperation(binNode, context);
      }

      case ASTNodeType.UnaryOperation: {
        const unaryNode = node as UnaryOperationNode;
        const operand = this.evaluateNode(unaryNode.operand, context);
        switch (unaryNode.operator) {
          case '-':
            return -Number(operand);
          case '+':
            return +Number(operand);
          default:
            throw new Error(`未知一元运算符: ${unaryNode.operator}`);
        }
      }

      case ASTNodeType.FunctionCall: {
        const funcNode = node as FunctionCallNode;
        return this.evaluateFunctionCall(funcNode, context);
      }

      case ASTNodeType.ErrorValue:
        return (node as ErrorValueNode).errorType;

      default:
        return '#VALUE!';
    }
  }

  /**
   * 获取单元格值
   */
  private getCellValue(ref: CellReferenceNode, context: EvaluationContext): unknown {
    // 确定实际的行列位置（处理相对引用）
    let row = ref.row;
    let col = ref.columnIndex;

    // 如果是当前单元格使用相对引用，需要调整
    if (!ref.isAbsoluteRow) {
      row = context.currentRow + (ref.row - context.currentRow);
    }
    if (!ref.isAbsoluteCol) {
      col = context.currentCol + (ref.columnIndex - context.currentCol);
    }

    // 确定工作表
    const sheetName = ref.sheetName;
    const sheet = sheetName
      ? context.allSheets.get(sheetName)
      : context.sheet;

    if (!sheet) {
      return '#REF!';
    }

    // 获取单元格数据
    const rowData = sheet.rows.get(String(row));
    if (!rowData) {
      return null;
    }

    const cell = rowData.cells.get(String(col));
    if (!cell) {
      return null;
    }

    // 如果是公式单元格，递归计算
    if (cell.formula) {
      // 检查缓存
      const cacheKey = `${sheet.id}:${row}:${col}`;
      const cached = context.cache.get(cacheKey);

      if (cached && cached.version === context.version) {
        return cached.value;
      }

      // 创建子上下文进行计算
      const subContext: EvaluationContext = {
        ...context,
        sheet,
        currentRow: row,
        currentCol: col,
      };

      const result = this.evaluate(cell.formula, subContext);

      // 缓存结果
      context.cache.set(cacheKey, {
        value: result.value,
        version: context.version,
      });

      return result.value;
    }

    return cell.v;
  }

  /**
   * 获取范围值
   */
  private getRangeValues(range: RangeReferenceNode, context: EvaluationContext): unknown[][] {
    const values: unknown[][] = [];
    const startRow = Math.min(range.start.row, range.end.row);
    const endRow = Math.max(range.start.row, range.end.row);
    const startCol = Math.min(range.start.columnIndex, range.end.columnIndex);
    const endCol = Math.max(range.start.columnIndex, range.end.columnIndex);

    for (let row = startRow; row <= endRow; row++) {
      const rowValues: unknown[] = [];
      for (let col = startCol; col <= endCol; col++) {
        const cellRef: CellReferenceNode = {
          type: ASTNodeType.CellReference,
          sheetName: range.sheetName,
          column: columnIndexToLetter(col),
          columnIndex: col,
          row,
          isAbsoluteCol: false,
          isAbsoluteRow: false,
        };
        rowValues.push(this.getCellValue(cellRef, context));
      }
      values.push(rowValues);
    }

    return values;
  }

  /**
   * 评估二元运算
   */
  private evaluateBinaryOperation(
    node: BinaryOperationNode,
    context: EvaluationContext
  ): unknown {
    const left = this.evaluateNode(node.left, context);
    const right = this.evaluateNode(node.right, context);

    switch (node.operator) {
      case '+':
        // 支持字符串连接
        if (typeof left === 'string' || typeof right === 'string') {
          return String(left ?? '') + String(right ?? '');
        }
        return Number(left) + Number(right);
      case '-':
        return Number(left) - Number(right);
      case '*':
        return Number(left) * Number(right);
      case '/':
        if (Number(right) === 0) {
          return '#DIV/0!';
        }
        return Number(left) / Number(right);
      case '^':
        return Math.pow(Number(left), Number(right));
      case '&':
        return String(left ?? '') + String(right ?? '');
      case '=':
      case '==':
        return left === right;
      case '!=':
      case '<>':
        return left !== right;
      case '<':
        return Number(left) < Number(right);
      case '>':
        return Number(left) > Number(right);
      case '<=':
        return Number(left) <= Number(right);
      case '>=':
        return Number(left) >= Number(right);
      default:
        return '#VALUE!';
    }
  }

  /**
   * 评估函数调用
   */
  private evaluateFunctionCall(
    node: FunctionCallNode,
    context: EvaluationContext
  ): unknown {
    const func = this.functions.get(node.name);
    if (!func) {
      return '#NAME?';
    }

    // 评估参数
    const args = node.arguments.map(arg => this.evaluateNode(arg, context));

    try {
      return func(...args);
    } catch (error) {
      return '#VALUE!';
    }
  }

  /**
   * 标记单元格需要重新计算
   * 当单元格的值或公式发生变化时调用
   */
  markForRecalculation(sheetId: string, row: number, col: number): void {
    const cellId = `${sheetId}:${row}:${col}`;
    this.recalculationQueue.add(cellId);

    // 获取依赖该单元格的所有单元格
    const dependents = this.reverseDependencyGraph.get(cellId);
    if (dependents) {
      dependents.forEach(dep => {
        this.recalculationQueue.add(dep);
      });
    }
  }

  /**
   * 刷新并重新计算所有待处理的单元格
   */
  async recalculateAll(
    sheet: Sheet,
    allSheets: Map<string, Sheet>,
    onProgress?: (current: number, total: number) => void
  ): Promise<void> {
    this.currentVersion++;

    // 构建新的依赖图
    this.buildDependencyGraph(sheet, allSheets);

    // 使用拓扑排序确定计算顺序
    const sortedCells = this.topologicalSort();
    const total = sortedCells.length;

    for (let i = 0; i < sortedCells.length; i++) {
      const cellId = sortedCells[i];
      const [sheetId, rowStr, colStr] = cellId.split(':');
      const row = parseInt(rowStr, 10);
      const col = parseInt(colStr, 10);

      const targetSheet = allSheets.get(sheetId) || sheet;
      const rowData = targetSheet.rows.get(String(row));
      const cell = rowData?.cells.get(String(col));

      if (cell?.formula) {
        const context: EvaluationContext = {
          sheet: targetSheet,
          allSheets,
          currentRow: row,
          currentCol: col,
          cache: new Map(),
          version: this.currentVersion,
        };

        const result = this.evaluate(cell.formula, context);

        // 更新单元格显示值
        cell.displayValue = result.value;
        if (result.error) {
          cell.displayValue = result.error;
        }
      }

      if (onProgress) {
        onProgress(i + 1, total);
      }

      // 让出主线程，避免阻塞
      if (i % 100 === 0) {
        await new Promise(resolve => setTimeout(resolve, 0));
      }
    }
  }

  /**
   * 构建依赖图
   */
  private buildDependencyGraph(sheet: Sheet, allSheets: Map<string, Sheet>): void {
    this.dependencyGraph.clear();
    this.reverseDependencyGraph.clear();

    // 遍历所有单元格
    sheet.rows.forEach((rowData, rowId) => {
      const row = parseInt(rowId, 10);
      rowData.cells.forEach((cell, colId) => {
        const col = parseInt(colId, 10);

        if (cell.formula) {
          // 解析公式获取依赖
          const dependencies = this.extractDependencies(cell.formula);
          const cellId = `${sheet.id}:${row}:${col}`;

          dependencies.forEach(dep => {
            // 添加依赖关系
            if (!this.dependencyGraph.has(cellId)) {
              this.dependencyGraph.set(cellId, new Set());
            }
            this.dependencyGraph.get(cellId)!.add(dep);

            // 添加反向依赖关系
            if (!this.reverseDependencyGraph.has(dep)) {
              this.reverseDependencyGraph.set(dep, new Set());
            }
            this.reverseDependencyGraph.get(dep)!.add(cellId);
          });
        }
      });
    });
  }

  /**
   * 从公式中提取依赖的单元格
   */
  private extractDependencies(formula: string): string[] {
    const dependencies: string[] = [];

    // 使用正则表达式匹配单元格引用和范围引用
    const cellRefRegex = /\$?([A-Z]+)\$?(\d+)/g;
    const rangeRefRegex = /\$?([A-Z]+)\$?(\d+):\$?([A-Z]+)\$?(\d+)/g;

    // 匹配范围引用
    let match = formula.match(rangeRefRegex);
    if (match) {
      // 范围引用需要展开为多个单元格引用
      // 简化处理，实际实现中需要更复杂的逻辑
    }

    // 匹配单元格引用
    match = formula.match(cellRefRegex);
    if (match) {
      match.forEach(ref => {
        // 移除 $ 符号并解析
        const cleanRef = ref.replace(/\$/g, '');
        const colMatch = cleanRef.match(/([A-Z]+)(\d+)/);
        if (colMatch) {
          const colStr = colMatch[1];
          const row = parseInt(colMatch[2], 10);
          const col = this.columnLetterToIndex(colStr);
          // 默认添加到当前工作表，实际需要解析工作表名称
          dependencies.push(`default:${row}:${col}`);
        }
      });
    }

    return dependencies;
  }

  /**
   * 拓扑排序确定计算顺序
   */
  private topologicalSort(): string[] {
    const sorted: string[] = [];
    const visited = new Set<string>();
    const visiting = new Set<string>();

    const visit = (cellId: string): void => {
      if (visited.has(cellId)) return;
      if (visiting.has(cellId)) {
        // 检测到循环依赖
        console.warn(`循环依赖检测到: ${cellId}`);
        return;
      }

      visiting.add(cellId);

      // 先访问依赖
      const dependencies = this.dependencyGraph.get(cellId);
      if (dependencies) {
        dependencies.forEach(dep => visit(dep));
      }

      visiting.delete(cellId);
      visited.add(cellId);
      sorted.push(cellId);
    };

    this.dependencyGraph.forEach((_, cellId) => visit(cellId));

    return sorted;
  }

  /**
   * 列字母转数字索引
   */
  private columnLetterToIndex(letters: string): number {
    let index = 0;
    for (let i = 0; i < letters.length; i++) {
      index = index * 26 + (letters.charCodeAt(i) - 64);
    }
    return index - 1;
  }

    const diff = end.getTime() - start.getTime();
    const unitStr = String(unit).toUpperCase();

    switch (unitStr) {
      case 'Y':
        return end.getFullYear() - start.getFullYear();
      case 'M':
        return (end.getFullYear() - start.getFullYear()) * 12 + (end.getMonth() - start.getMonth());
      case 'D':
        return Math.floor(diff / (1000 * 60 * 60 * 24));
      case 'MD':
        return end.getDate() - start.getDate();
      case 'YM':
        return (end.getMonth() - start.getMonth() + 12) % 12;
      case 'YD':
        return Math.floor((end.getTime() - new Date(end.getFullYear(), start.getMonth(), start.getDate()).getTime()) / (1000 * 60 * 60 * 24));
      default:
        return diff;

    const col = Number(colIndex) - 1;
    const exact = !Boolean(rangeLookup);

    if (exact) {
      // 精确匹配
      for (const row of table) {
        if (row[0] === lookupValue) {
          return row[col];
        }

    } else {

      for (const row of table) {

          result = row[col];
        } else {

      return result;

  /**
   * HLOOKUP 函数 - 水平查找

    lookupValue: unknown,
    tableArray: unknown,
    rowIndex: unknown,
    rangeLookup: unknown = false
  ): unknown => {
    const table = this.ensureArray(tableArray);
    const row = Number(rowIndex) - 1;
    const exact = !Boolean(rangeLookup);

    if (table.length === 0 || table[0].length === 0) {
      return '#N/A';
    }

    if (exact) {
      // 精确匹配
      for (let i = 0; i < table[0].length; i++) {
        if (table[0][i] === lookupValue) {
          return table[row][i];
        }
      }
    } else {
      // 近似匹配
      let result: unknown;
      for (let i = 0; i < table[0].length; i++) {
        if (table[0][i] <= lookupValue) {
          result = table[row][i];
        } else {
          break;
        }
      }
      return result;
    }

    return '#N/A';
  };

   * INDEX 函数 - 返回指定位置的值

    array: unknown,
    rowNum: unknown,
    colNum: unknown = 1
  ): unknown => {
    const arr = this.ensureArray(array);
    const row = Number(rowNum) - 1;
    const col = Number(colNum) - 1;

    if (row < 0 || row >= arr.length || col < 0 || (arr[0] && col >= arr[0].length)) {
      return '#REF!';
    }

    return arr[row]?.[col] ?? '#REF!';

  /**
   * MATCH 函数 - 返回值的位置
   */
  private builtinMatch = (
    lookupValue: unknown,
    lookupArray: unknown,
    matchType: unknown = 0
  ): number => {
    const arr = this.ensureArray(lookupArray);
    const type = Number(matchType);

    if (type === 0) {
      // 精确匹配
      for (let i = 0; i < arr.length; i++) {
        if (arr[i][0] === lookupValue) {
          return i + 1;
        }
      }
    } else if (type === 1) {
      // 小于或等于（假设升序）
      let result = -1;
      for (let i = 0; i < arr.length; i++) {
        if (arr[i][0] <= lookupValue) {
          result = i;
        } else {
          break;
        }
      }
      return result + 1;
    } else if (type === -1) {
      // 大于或等于（假设降序）
      let result = -1;
      for (let i = 0; i < arr.length; i++) {
        if (arr[i][0] >= lookupValue) {
          result = i;
        } else {
          break;
        }
      }
      return result + 1;
    }

    return '#N/A';
  };

  /**
   * ROW 函数 - 返回行号

  private builtinRow = (cellRef?: unknown): number => {
    if (cellRef === undefined) {
      return 1; // 假设在第一行

    // 解析单元格引用获取行号
    const ref = String(cellRef).match(/\d+/);
    return ref ? parseInt(ref[0], 10) : 1;
  };

  /**
   * COLUMN 函数 - 返回列号
   */
  private builtinColumn = (cellRef?: unknown): number => {
    if (cellRef === undefined) {
      return 1; // 假设在第一列
    }
    // 解析单元格引用获取列号
    const ref = String(cellRef).match(/[A-Z]+/i);
    if (!ref) return 1;
    let col = 0;
    for (let i = 0; i < ref[0].length; i++) {
      col = col * 26 + (ref[0].charCodeAt(i) - 64);
    }
    return col;
  };

  /**
   * ADDRESS 函数 - 返回单元格地址字符串
   */
  private builtinAddress = (
    rowNum: unknown,
    colNum: unknown,
    absNum: unknown = 1,
    a1Style: unknown = true,
    sheetText?: unknown
  ): string => {
    const row = Number(rowNum);
    const col = Number(colNum);
    const abs = Number(absNum);
    const useA1 = Boolean(a1Style);

    let colStr = '';
    let colPart = '';
    let rowPart = '';

    // 计算列字母
    let c = col;
    while (c > 0) {
      colStr = String.fromCharCode(((c - 1) % 26) + 65) + colStr;
      c = Math.floor((c - 1) / 26);
    }

    // 根据绝对引用类型添加 $ 符号
    switch (abs) {
      case 1:
        colPart = `$${colStr}`;
        rowPart = `$${row}`;
        break;
      case 2:
        colPart = colStr;
        rowPart = `$${row}`;
        break;
      case 3:
        colPart = `$${colStr}`;
        rowPart = String(row);
        break;
      case 4:
      default:
        colPart = colStr;
        rowPart = String(row);
        break;
    }

    const address = `${colPart}${rowPart}`;
    const sheet = sheetText ? `${String(sheetText)}!` : '';

    return `${sheet}${address}`;
  };

  // ===== 辅助方法 =====

  /**
   * 扁平化数组
   * 处理多维数组和嵌套数组
   */
  private flattenArray(args: unknown[]): unknown[] {
    const result: unknown[] = [];

    const flatten = (arr: unknown[]): void => {
      for (const item of arr) {
        if (Array.isArray(item)) {
          flatten(item);
        } else {
          result.push(item);
        }
      }
    };

    for (const arg of args) {
      if (Array.isArray(arg)) {
        flatten(arg);
      } else {
        result.push(arg);
      }
    }

    return result;
  }

  /**
   * 确保值是数组
   */
  private ensureArray(value: unknown): unknown[][] {
    if (Array.isArray(value)) {
      if (Array.isArray(value[0])) {
        return value as unknown[][];
      }
      // 一维数组转为二维
      return [value as unknown[]];
    }
    // 单值转为单元素数组
    return [[value]];
  }

  /**
   * 检查值是否为空
   */
  private isEmpty(value: unknown): boolean {
    return value === null || value === undefined || value === '';
  }

  /**
   * 检查值是否是错误值
   */
  private isError(value: unknown): boolean {
    if (typeof value !== 'string') return false;
    return value.startsWith('#') && (value.endsWith('!') || value.endsWith('?'));
  }
}
```

### 4.1 筛选引擎设计

筛选功能允许用户根据条件过滤数据，是多维表格的核心功能之一。

```typescript
/**
 * 筛选条件
 */
interface FilterCondition {
  /** 列ID */
  columnId: string;
  /** 运算符 */
  operator: FilterOperator;
  /** 比较值 */
  value?: unknown;
  /** 是否区分大小写（用于文本） */
  caseSensitive?: boolean;
}

/**
 * 筛选运算符枚举
 */
enum FilterOperator {
  /** 等于 */
  Equal = 'equal',
  /** 不等于 */
  NotEqual = 'notEqual',
  /** 大于 */
  GreaterThan = 'greaterThan',
  /** 大于等于 */
  GreaterThanOrEqual = 'greaterThanOrEqual',
  /** 小于 */
  LessThan = 'lessThan',
  /** 小于等于 */
  LessThanOrEqual = 'lessThanOrEqual',
  /** 包含 */
  Contains = 'contains',
  /** 不包含 */
  NotContains = 'notContains',
  /** 开头是 */
  StartsWith = 'startsWith',
  /** 结尾是 */
  EndsWith = 'endsWith',
  /** 为空 */
  IsEmpty = 'isEmpty',
  /** 不为空 */
  IsNotEmpty = 'isNotEmpty',
  /** 在列表中 */
  In = 'in',
  /** 不在列表中 */
  NotIn = 'notIn',
  /** 介于 */
  Between = 'between',
}

/**
 * 筛选逻辑
 */
enum FilterLogic {
  /** AND - 所有条件都满足 */
  And = 'and',
  /** OR - 任一条件满足 */
  Or = 'or',
}

/**
 * 筛选器类
 *
 * 设计思路：
 * 1. 支持多种运算符和组合逻辑
 * 2. 使用索引优化筛选性能
 * 3. 支持虚拟滚动，只渲染可见行
 */
class FilterEngine {
  /** 索引缓存 */
  private indexCache: Map<string, Map<unknown, string[]>> = new Map();
  /** 索引版本 */
  private indexVersion: number = 0;
  /** 索引版本记录 */
  private indexVersionMap: Map<string, number> = new Map();

  /**
   * 执行筛选
   * @param sheet 工作表数据
   * @param conditions 筛选条件
   * @param logic 逻辑关系（AND/OR）
   * @returns 匹配的行ID列表
   */
  filter(
    sheet: Sheet,
    conditions: FilterCondition[],
    logic: FilterLogic = FilterLogic.And
  ): string[] {
    if (conditions.length === 0) {
      // 无条件，返回所有行
      return Array.from(sheet.rows.keys());
    }

    // 为每个条件计算匹配结果
    const results = conditions.map(condition =>
      this.filterByCondition(sheet, condition)
    );

    // 根据逻辑关系合并结果
    if (logic === FilterLogic.And) {
      // AND：取交集
      return this.intersect(results);
    } else {
      // OR：取并集
      return this.union(results);
    }
  }

  /**
   * 根据单个条件筛选
   */
  private filterByCondition(
    sheet: Sheet,
    condition: FilterCondition
  ): string[] {
    const { columnId, operator, value, caseSensitive } = condition;
    const matchedRows: string[] = [];

    // 遍历所有行
    sheet.rows.forEach((rowData, rowId) => {
      const cell = rowData.cells.get(columnId);
      const cellValue = cell?.v ?? null;

      if (this.evaluateCondition(cellValue, operator, value, caseSensitive)) {
        matchedRows.push(rowId);
      }
    });

    return matchedRows;
  }

  /**
   * 评估条件是否满足
   */
  private evaluateCondition(
    cellValue: unknown,
    operator: FilterOperator,
    conditionValue: unknown,
    caseSensitive?: boolean
  ): boolean {
    // 处理空值
    const isCellEmpty = cellValue === null || cellValue === undefined || cellValue === '';

    switch (operator) {
      case FilterOperator.IsEmpty:
        return isCellEmpty;

      case FilterOperator.IsNotEmpty:
        return !isCellEmpty;

      case FilterOperator.Equal:
        return this.compareEqual(cellValue, conditionValue, caseSensitive);

      case FilterOperator.NotEqual:
        return !this.compareEqual(cellValue, conditionValue, caseSensitive);

      case FilterOperator.GreaterThan:
        return !isCellEmpty && Number(cellValue) > Number(conditionValue);

      case FilterOperator.GreaterThanOrEqual:
        return !isCellEmpty && Number(cellValue) >= Number(conditionValue);

      case FilterOperator.LessThan:
        return !isCellEmpty && Number(cellValue) < Number(conditionValue);

      case FilterOperator.LessThanOrEqual:
        return !isCellEmpty && Number(cellValue) <= Number(conditionValue);

      case FilterOperator.Contains:
        return !isCellEmpty && String(cellValue).includes(String(conditionValue));

      case FilterOperator.NotContains:
        return !isCellEmpty && !String(cellValue).includes(String(conditionValue));

      case FilterOperator.StartsWith:
        return !isCellEmpty && String(cellValue).startsWith(String(conditionValue));

      case FilterOperator.EndsWith:
        return !isCellEmpty && String(cellValue).endsWith(String(conditionValue));

      case FilterOperator.In:
        if (Array.isArray(conditionValue)) {
          return conditionValue.some(v => this.compareEqual(cellValue, v, caseSensitive));
        }
        return this.compareEqual(cellValue, conditionValue, caseSensitive);

      case FilterOperator.NotIn:
        if (Array.isArray(conditionValue)) {
          return !conditionValue.some(v => this.compareEqual(cellValue, v, caseSensitive));
        }
        return !this.compareEqual(cellValue, conditionValue, caseSensitive);

      case FilterOperator.Between:
        if (Array.isArray(conditionValue) && conditionValue.length === 2) {
          const num = Number(cellValue);
          return !isCellEmpty && num >= Number(conditionValue[0]) && num <= Number(conditionValue[1]);
        }
        return false;

      default:
        return false;
    }
  }

  /**
   * 比较两个值是否相等
   */
  private compareEqual(
    a: unknown,
    b: unknown,
    caseSensitive?: boolean
  ): boolean {
    if (a === b) return true;

    const strA = String(a ?? '');
    const strB = String(b ?? '');

    if (caseSensitive) {
      return strA === strB;
    }
    return strA.toLowerCase() === strB.toLowerCase();
  }

  /**
   * 取交集（AND逻辑）
   */
  private intersect(results: string[][]): string[] {
    if (results.length === 0) return [];
    if (results.length === 1) return results[0];

    // 使用 Set 进行高效交集运算
    let intersection = new Set(results[0]);

    for (let i = 1; i < results.length; i++) {
      const set = new Set(results[i]);
      intersection = new Set(
        [...intersection].filter(x => set.has(x))
      );
    }

    return [...intersection];
  }

  /**
   * 取并集（OR逻辑）
   */
  private union(results: string[][]): string[] {
    if (results.length === 0) return [];
    if (results.length === 1) return results[0];

    const unionSet = new Set<string>();
    results.forEach(result => {
      result.forEach(id => unionSet.add(id));
    });

    return [...unionSet];
  }

  /**
   * 创建索引以加速筛选
   * @param sheet 工作表
   * @param columnId 列ID
   */
  createIndex(sheet: Sheet, columnId: string): void {
    const index = new Map<unknown, string[]>();

    sheet.rows.forEach((rowData, rowId) => {
      const cell = rowData.cells.get(columnId);
      const value = cell?.v ?? null;

      if (!index.has(value)) {
        index.set(value, []);
      }
      index.get(value)!.push(rowId);
    });

    this.indexCache.set(`${sheet.id}:${columnId}`, index);
    this.indexVersion++;
    this.indexVersionMap.set(`${sheet.id}:${columnId}`, this.indexVersion);
  }

  /**
   * 使用索引加速筛选
   */
  filterWithIndex(
    sheet: Sheet,
    columnId: string,
    value: unknown
  ): string[] {
    const cacheKey = `${sheet.id}:${columnId}`;
    const cachedIndex = this.indexCache.get(cacheKey);
    const currentVersion = this.indexVersionMap.get(cacheKey);

    // 检查索引是否过期
    if (!cachedIndex || currentVersion !== this.indexVersion) {
      // 索引不存在或过期，重新创建
      this.createIndex(sheet, columnId);
    }

    const index = this.indexCache.get(cacheKey);
    return index?.get(value) ?? [];
  }
}

/**
 * 排序器类
 */
class SortEngine {
  /**
   * 排序
   * @param sheet 工作表数据
   * @param sortColumns 排序列
   * @param rowIds 要排序的行ID列表（可以是筛选后的子集）
   * @returns 排序后的行ID列表
   */
  sort(
    sheet: Sheet,
    sortColumns: SortColumn[],
    rowIds?: string[]
  ): string[] {
    // 如果没有指定行，默认对所有行排序
    const idsToSort = rowIds ?? Array.from(sheet.rows.keys());

    if (sortColumns.length === 0) {
      return idsToSort;
    }

    // 转换为数组以便排序
    const rowsArray = idsToSort.map(rowId => ({
      id: rowId,
      rowData: sheet.rows.get(rowId),
    }));

    // 多级排序
    rowsArray.sort((a, b) => {
      for (const sortCol of sortColumns) {
        const { columnId, ascending } = sortCol;
        const cellA = a.rowData?.cells.get(columnId);
        const cellB = b.rowData?.cells.get(columnId);
        const valueA = cellA?.v ?? null;
        const valueB = cellB?.v ?? null;

        const result = this.compareValues(valueA, valueB);

        if (result !== 0) {
          return ascending ? result : -result;
        }
      }
      return 0;
    });

    return rowsArray.map(r => r.id);
  }

  /**
   * 比较两个值
   */
  private compareValues(a: unknown, b: unknown): number {
    // 空值排在最后
    const isEmptyA = a === null || a === undefined || a === '';
    const isEmptyB = b === null || b === undefined || b === '';

    if (isEmptyA && isEmptyB) return 0;
    if (isEmptyA) return 1;
    if (isEmptyB) return -1;

    // 数字比较
    if (typeof a === 'number' && typeof b === 'number') {
      return a - b;
    }

    // 布尔比较
    if (typeof a === 'boolean' && typeof b === 'boolean') {
      return a === b ? 0 : a ? -1 : 1;
    }

    // 日期比较
    if (a instanceof Date && b instanceof Date) {
      return a.getTime() - b.getTime();
    }

    // 字符串比较
    const strA = String(a);
    const strB = String(b);
    return strA.localeCompare(strB);
  }
}
```

## 五、分组与透视

### 5.1 分组功能实现

分组功能允许用户将相似的记录聚合在一起，并可对每个组进行汇总计算。

```typescript
/**
 * 分组信息
 */
interface GroupingInfo {
  /** 分组字段（列ID） */
  groupByField: string;
  /** 分组键值到组信息的映射 */
  groups: Map<unknown, GroupData>;
  /** 未分组的行（值为空的行） */
  nullGroup?: GroupData;
}

/**
 * 分组数据
 */
interface GroupData {
  /** 分组键值 */
  key: unknown;
  /** 该组包含的行ID */
  rowIds: string[];
  /** 是否展开 */
  expanded: boolean;
  /** 子分组 */
  subgroups?: Map<string, GroupData>;
  /** 汇总计算结果 */
  aggregates?: Record<string, unknown>;
}

/**
 * 汇总配置
 */
interface AggregateConfig {
  /** 源字段（列ID） */
  field: string;
  /** 汇总函数 */
  function: AggregateFunction;
  /** 结果字段名 */
  alias?: string;
}

/**
 * 汇总函数枚举
 */
enum AggregateFunction {
  Sum = 'sum',
  Average = 'average',
  Count = 'count',
  CountUnique = 'countUnique',
  Min = 'min',
  Max = 'max',
  Median = 'median',
  StdDev = 'stdDev',
  First = 'first',
  Last = 'last',
}

/**
 * 分组引擎类
 */
class GroupEngine {
  /**
   * 执行分组
   * @param sheet 工作表数据
   * @param groupByField 分组字段
   * @param rowIds 要分组的行ID（可以是筛选后的子集）
   * @returns 分组信息
   */
  group(
    sheet: Sheet,
    groupByField: string,
    rowIds?: string[]
  ): GroupingInfo {
    const idsToGroup = rowIds ?? Array.from(sheet.rows.keys());
    const groups = new Map<unknown, GroupData>();
    let nullGroup: GroupData | undefined;

    // 遍历每行，按分组字段的值进行聚合
    for (const rowId of idsToGroup) {
      const rowData = sheet.rows.get(rowId);
      if (!rowData) continue;

      const cell = rowData.cells.get(groupByField);
      const key = cell?.v ?? null;

      if (key === null || key === undefined || key === '') {
        // 放入空值组
        if (!nullGroup) {
          nullGroup = {
            key: null,
            rowIds: [],
            expanded: true,
          };
        }
        nullGroup.rowIds.push(rowId);
      } else {
        // 放入对应键值的组
        if (!groups.has(key)) {
          groups.set(key, {
            key,
            rowIds: [],
            expanded: true,
          });
        }
        groups.get(key)!.rowIds.push(rowId);
      }
    }

    return {
      groupByField,
      groups,
      nullGroup,
    };
  }

  /**
   * 执行多级分组
   * @param sheet 工作表数据
   * @param groupByFields 分组字段列表（从外到内）
   * @param rowIds 要分组的行ID
   * @returns 分组信息
   */
  groupMultiple(
    sheet: Sheet,
    groupByFields: string[],
    rowIds?: string[]
  ): GroupingInfo {
    if (groupByFields.length === 0) {
      return this.group(sheet, '', rowIds);
    }

    // 首先按第一个字段分组
    const firstLevelGrouping = this.group(sheet, groupByFields[0], rowIds);

    // 如果有更多字段，递归分组
    if (groupByFields.length > 1) {
      const remainingFields = groupByFields.slice(1);

      // 对每个组进行子分组
      const recursiveGroup = (
        groupData: GroupData,
        fields: string[]
      ): void => {
        if (fields.length === 0) return;

        const subField = fields[0];

        // 对该组的行进行分组
        const subgroupMap = new Map<unknown, GroupData>();

        for (const rowId of groupData.rowIds) {
          const rowData = sheet.rows.get(rowId);
          if (!rowData) continue;

          const cell = rowData.cells.get(subField);
          const key = cell?.v ?? null;

          if (!subgroupMap.has(key)) {
            subgroupMap.set(key, {
              key,
              rowIds: [],
              expanded: groupData.expanded,
            });
          }
          subgroupMap.get(key)!.rowIds.push(rowId);
        }

        groupData.subgroups = subgroupMap;

        // 递归处理子分组
        subgroupMap.forEach(subgroup => {
          recursiveGroup(subgroup, fields.slice(1));
        });
      };

      // 对所有顶级组进行递归分组
      firstLevelGrouping.groups.forEach(group => {
        recursiveGroup(group, remainingFields);
      });

      if (firstLevelGrouping.nullGroup) {
        recursiveGroup(firstLevelGrouping.nullGroup, remainingFields);
      }
    }

    return firstLevelGrouping;
  }

  /**
   * 计算分组汇总
   * @param sheet 工作表数据
   * @param groupingInfo 分组信息
   * @param aggregates 汇总配置
   */
  computeAggregates(
    sheet: Sheet,
    groupingInfo: GroupingInfo,
    aggregates: AggregateConfig[]
  ): void {
    const computeForRows = (rowIds: string[]): Record<string, unknown> => {
      const result: Record<string, unknown> = {};

      for (const config of aggregates) {
        const values = rowIds
          .map(rowId => sheet.rows.get(rowId)?.cells.get(config.field)?.v)
          .filter(v => v !== null && v !== undefined && v !== '');

        result[config.alias ?? `${config.field}_${config.function}`] =
          this.computeAggregate(values, config.function);
      }

      return result;
    };

    // 计算顶级组的汇总
    groupingInfo.groups.forEach(group => {
      group.aggregates = computeForRows(group.rowIds);

      // 递归计算子分组汇总
      if (group.subgroups) {
        group.subgroups.forEach(subgroup => {
          subgroup.aggregates = computeForRows(subgroup.rowIds);
        });
      }
    });

    // 计算空值组的汇总
    if (groupingInfo.nullGroup) {
      groupingInfo.nullGroup.aggregates = computeForRows(groupingInfo.nullGroup.rowIds);
    }
  }

  /**
   * 计算单个汇总值
   */
  private computeAggregate(
    values: unknown[],
    func: AggregateFunction
  ): unknown {
    if (values.length === 0) {
      return null;
    }

    const numbers = values.map(Number).filter(n => !isNaN(n));

    switch (func) {
      case AggregateFunction.Sum:
        return numbers.reduce((sum, n) => sum + n, 0);

      case AggregateFunction.Average:
        return numbers.length > 0
          ? numbers.reduce((sum, n) => sum + n, 0) / numbers.length
          : null;

      case AggregateFunction.Count:
        return values.length;

      case AggregateFunction.CountUnique:
        return new Set(values).size;

      case AggregateFunction.Min:
        return numbers.length > 0 ? Math.min(...numbers) : null;

      case AggregateFunction.Max:
        return numbers.length > 0 ? Math.max(...numbers) : null;

      case AggregateFunction.Median:
        if (numbers.length === 0) return null;
        const sorted = [...numbers].sort((a, b) => a - b);
        const mid = Math.floor(sorted.length / 2);
        return sorted.length % 2 !== 0
          ? sorted[mid]
          : (sorted[mid - 1] + sorted[mid]) / 2;

      case AggregateFunction.StdDev:
        if (numbers.length < 2) return null;
        const mean = numbers.reduce((sum, n) => sum + n, 0) / numbers.length;
        const squareDiffs = numbers.map(n => Math.pow(n - mean, 2));
        const avgSquareDiff = squareDiffs.reduce((sum, d) => sum + d, 0) / numbers.length;
        return Math.sqrt(avgSquareDiff);

      case AggregateFunction.First:
        return values[0];

      case AggregateFunction.Last:
        return values[values.length - 1];

      default:
        return null;
    }
  }

  /**
   * 展开或折叠分组
   */
  toggleGroupExpanded(
    groupingInfo: GroupingInfo,
    groupKey: unknown,
    expanded: boolean
  ): void {
    const group = groupingInfo.groups.get(groupKey);
    if (group) {
      group.expanded = expanded;
    }
  }
}
```

```typescript
/**
 * 协同编辑管理器

class CollaborationManager {
  /** WebSocket 连接 */
  private socket: WebSocket | null = null;
  /** 协同状态 */
  private state: CollaborationState;
  /** 事件总线 */
  private eventBus: EventBus;
  /** 变更缓冲区（用于批量处理） */
  private changeBuffer: Operation[] = [];
  /** 缓冲区刷新定时器 */
  private bufferFlushTimer: number | null = null;
  /** 重连定时器 */
  private reconnectTimer: number | null = null;
  /** 心跳定时器 */
  private heartbeatTimer: number | null = null;
  /** 重连尝试次数 */
  private reconnectAttempts: number = 0;
  /** 最大重连尝试次数 */
  private maxReconnectAttempts: number = 5;
  /** OT算法处理器 */
  private otHandler: OTSHandler;

  constructor(
    documentId: string,
    userId: string,
    room: string,
    eventBus: EventBus
  ) {
    this.state = {
      documentId,
      userId,
      room,
      activeUsers: [],
      cellLocks: new Map(),
      syncState: {
        synced: true,
        lastSyncedAt: Date.now(),
        pendingChanges: [],
        version: 0,
      },
    };
    this.eventBus = eventBus;
    this.otHandler = new OTSHandler();
  }

  /**
   * 连接到协同房间
   */
  connect(serverUrl: string): Promise<void> {
    return new Promise((resolve, reject) => {
      try {
        this.socket = new WebSocket(`${serverUrl}/collaboration/${this.state.room}`);

        this.socket.onopen = () => {
          console.log('协同连接已建立');
          this.startHeartbeat();
          this.reconnectAttempts = 0;

          // 发送加入房间消息
          this.send({
            type: 'join',
            documentId: this.state.documentId,
            userId: this.state.userId,
          });

          resolve();
        };

        this.socket.onmessage = (event) => {
          const message = JSON.parse(event.data);
          this.handleMessage(message);
        };

        this.socket.onerror = (error) => {
          console.error('协同连接错误:', error);
          reject(error);
        };

        this.socket.onclose = () => {
          console.log('协同连接已关闭');
          this.stopHeartbeat();
          this.scheduleReconnect();
        };
      } catch (error) {
        reject(error);
      }
    });
  }

  /**
   * 断开连接
   */
  disconnect(): void {
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
      this.reconnectTimer = null;
    }

    if (this.socket) {
      // 发送离开消息
      this.send({
        type: 'leave',
        documentId: this.state.documentId,
        userId: this.state.userId,
      });

      this.socket.close();
      this.socket = null;
    }
  }

  /**
   * 发送变更
   * @param type 操作类型
   * @param data 操作数据
   */
  sendChange(type: OperationType, data: unknown): void {
    const operation: Operation = {
      id: this.generateId(),
      type,
      data,
      userId: this.state.userId,
      timestamp: Date.now(),
      version: this.state.syncState.version + 1,
    };

    // 添加到缓冲区
    this.changeBuffer.push(operation);

    // 设置缓冲区刷新定时器（批量发送）
    if (!this.bufferFlushTimer) {
      this.bufferFlushTimer = window.setTimeout(() => {
        this.flushBuffer();
      }, 50); // 50ms 后刷新
    }

    // 更新本地状态（乐观更新）
    this.applyLocalChange(operation);
  }

  /**
   * 刷新缓冲区，发送批量变更
   */
  private flushBuffer(): void {
    if (this.changeBuffer.length === 0) return;

    const changes = [...this.changeBuffer];
    this.changeBuffer = [];
    this.bufferFlushTimer = null;

    this.send({
      type: 'changes',
      documentId: this.state.documentId,
      userId: this.state.userId,
      changes,
      baseVersion: this.state.syncState.version,
    });
  }

  /**
   * 应用本地变更（乐观更新）
   */
  private applyLocalChange(operation: Operation): void {
    // 根据操作类型更新本地状态
    switch (operation.type) {
      case OperationType.SetCellValue:
        this.applySetCellValue(operation.data as SetCellValueData);
        break;
      case OperationType.SetCellFormat:
        this.applySetCellFormat(operation.data as SetCellFormatData);
        break;
      // ... 其他操作类型
    }

    // 触发事件
    this.eventBus.emit({
      type: 'collaboration:localChange',
      data: operation,
      timestamp: operation.timestamp,
      sourceId: this.state.userId,
      undoable: false,
    });
  }

  /**
   * 处理接收到的消息
   */
  private handleMessage(message: Message): void {
    switch (message.type) {
      case 'join':
        this.handleUserJoin(message as UserJoinMessage);
        break;
      case 'leave':
        this.handleUserLeave(message as UserLeaveMessage);
        break;
      case 'cursor':
        this.handleCursorUpdate(message as CursorMessage);
        break;
      case 'selection':
        this.handleSelectionUpdate(message as SelectionMessage);
        break;
      case 'changes':
        this.handleRemoteChanges(message as ChangesMessage);
        break;
      case 'ack':
        this.handleAck(message as AckMessage);
        break;
      case 'sync':
        this.handleSync(message as SyncMessage);
        break;
      case 'lock':
        this.handleLock(message as LockMessage);
        break;
      case 'unlock':
        this.handleUnlock(message as UnlockMessage);
        break;
    }
  }

  /**
   * 处理用户加入
   */
  private handleUserJoin(message: UserJoinMessage): void {
    const user: ActiveUser = {
      id: message.userId,
      name: message.userName,
      color: message.color,
      avatar: message.avatar,
      lastActivity: Date.now(),
    };

    this.state.activeUsers.push(user);

    this.eventBus.emit({
      type: 'collaboration:userJoin',
      data: user,
      timestamp: Date.now(),
      sourceId: 'CollaborationManager',
      undoable: false,
    });
  }

  /**
   * 处理用户离开
   */
  private handleUserLeave(message: UserLeaveMessage): void {
    const index = this.state.activeUsers.findIndex(u => u.id === message.userId);
    if (index !== -1) {
      const user = this.state.activeUsers.splice(index, 1)[0];

      this.eventBus.emit({
        type: 'collaboration:userLeave',
        data: user,
        timestamp: Date.now(),
        sourceId: 'CollaborationManager',
        undoable: false,
      });
    }

    // 释放该用户持有的锁
    this.state.cellLocks.forEach((lock, cellId) => {
      if (lock.userId === message.userId) {
        this.state.cellLocks.delete(cellId);
      }
    });
  }

  /**
   * 处理光标更新
   */
  private handleCursorUpdate(message: CursorMessage): void {
    const user = this.state.activeUsers.find(u => u.id === message.userId);
    if (user) {
      user.cursor = message.position;
      user.lastActivity = Date.now();

      this.eventBus.emit({
        type: 'collaboration:cursorUpdate',
        data: { userId: message.userId, position: message.position },
        timestamp: Date.now(),
        sourceId: 'CollaborationManager',
        undoable: false,
      });
    }
  }

  /**
   * 处理选区更新
   */
  private handleSelectionUpdate(message: SelectionMessage): void {
    const user = this.state.activeUsers.find(u => u.id === message.userId);
    if (user) {
      user.selection = message.selection;
      user.lastActivity = Date.now();

      this.eventBus.emit({
        type: 'collaboration:selectionUpdate',
        data: { userId: message.userId, selection: message.selection },
        timestamp: Date.now(),
        sourceId: 'CollaborationManager',
        undoable: false,
      });
    }
  }

  /**
   * 处理远程变更
   */
  private handleRemoteChanges(message: ChangesMessage): void {
    // 使用 OT 算法处理并发变更
    const transformedChanges = this.otHandler.transform(
      message.changes,
      this.state.syncState.pendingChanges,
      message.baseVersion,
      this.state.syncState.version
    );

    // 应用变更
    transformedChanges.forEach(operation => {
      if (operation.userId !== this.state.userId) {
        this.applyRemoteChange(operation);
      }
    });

    // 更新版本号
    this.state.syncState.version = message.newVersion;
    this.state.syncState.synced = true;
    this.state.syncState.lastSyncedAt = Date.now();

    this.eventBus.emit({
      type: 'collaboration:sync',
      data: { changes: transformedChanges },
      timestamp: Date.now(),
      sourceId: 'CollaborationManager',
      undoable: false,
    });
  }

  /**
   * 应用远程变更
   */
  private applyRemoteChange(operation: Operation): void {
    // 与本地变更类似的处理逻辑
    switch (operation.type) {
      case OperationType.SetCellValue:
        this.applySetCellValue(operation.data as SetCellValueData);
        break;
      case OperationType.SetCellFormat:
        this.applySetCellFormat(operation.data as SetCellFormatData);
        break;
    }
  }

  /**
   * 处理确认消息
   */
  private handleAck(message: AckMessage): void {
    // 移除已确认的待处理变更
    this.state.syncState.pendingChanges = this.state.syncState.pendingChanges.filter(
      op => !message.acknowledgedIds.includes(op.id)
    );

    this.state.syncState.version = message.newVersion;
    this.state.syncState.synced = this.state.syncState.pendingChanges.length === 0;
  }

  /**
   * 处理同步消息
   */
  private handleSync(message: SyncMessage): void {
    // 全量同步状态
    this.state.syncState.version = message.version;
    this.state.syncState.synced = true;
    this.state.syncState.lastSyncedAt = Date.now();

    // 应用同步的状态
    this.applyFullState(message.state);

    this.eventBus.emit({
      type: 'collaboration:sync',
      data: { fullSync: true },
      timestamp: Date.now(),
      sourceId: 'CollaborationManager',
      undoable: false,
    });
  }

  /**
   * 发送消息到服务器
   */
  private send(message: object): void {
    if (this.socket && this.socket.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify(message));
    } else {
      console.warn('WebSocket 未连接，消息未发送');
    }
  }

  /**
   * 启动心跳
   */
  private startHeartbeat(): void {
    this.heartbeatTimer = window.setInterval(() => {
      this.send({ type: 'ping', userId: this.state.userId });
    }, 30000); // 每 30 秒发送一次心跳
  }

  /**
   * 停止心跳
   */
  private stopHeartbeat(): void {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer);
      this.heartbeatTimer = null;
    }
  }

  /**
   * 安排重连
   */
  private scheduleReconnect(): void {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      console.error('达到最大重连次数');
      this.eventBus.emit({
        type: 'collaboration:disconnected',
        data: { reason: 'maxReconnectAttempts' },
        timestamp: Date.now(),
        sourceId: 'CollaborationManager',
        undoable: false,
      });
      return;
    }

    const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempts), 30000);
    this.reconnectAttempts++;

    console.log(`${delay}ms 后尝试重连...`);

    this.reconnectTimer = window.setTimeout(() => {
      this.connect('').catch(() => {
        // 连接失败，scheduleReconnect 会被调用
      });
    }, delay);
  }

/**
 * OT (Operational Transformation) 算法处理器
 * 用于处理并发编辑冲突
 */
class OTSHandler {
  /**
   * 转换操作以解决冲突
   */
  transform(
    remoteOps: Operation[],
    localOps: Operation[],
    remoteBaseVersion: number,
    localVersion: number
  ): Operation[] {
    // 简化的 OT 实现
    // 实际实现需要根据操作类型进行复杂的转换

    const transformed: Operation[] = [];
    let localIndex = 0;

    for (const remoteOp of remoteOps) {
      // 跳过已经被本地操作包含的操作
      while (
        localIndex < localOps.length &&
        localOps[localIndex].timestamp < remoteOp.timestamp
      ) {
        transformed.push(localOps[localIndex]);
        localIndex++;
      }

      // 转换远程操作以适应本地变更
      const transformedOp = this.transformOperation(
        remoteOp,
        localOps.slice(0, localIndex)
      );
      transformed.push(transformedOp);
    }

    // 添加剩余的本地操作
    while (localIndex < localOps.length) {
      transformed.push(localOps[localIndex]);
      localIndex++;
    }

    return transformed;
  }

  /**
   * 转换单个操作
   */
  private transformOperation(
    operation: Operation,
    priorOperations: Operation[]
  ): Operation {
    // 根据操作类型和先前的操作进行转换
    // 这是一个简化实现，实际需要更复杂的逻辑

    const transformed = { ...operation };

    // 例如，如果远程操作是设置单元格 A1 的值，
    // 而本地操作已经删除了包含 A1 的行，则需要调整

    return transformed;
  }
```

## 七、性能优化策略

### 7.1 虚拟滚动

当表格数据量很大时，不可能渲染所有单元格，需要使用虚拟滚动技术只渲染可见区域的内容。

```typescript
/**
 * 虚拟滚动管理器
 *
 * 设计思路：
 * 1. 只渲染可见区域内的单元格
 * 2. 使用固定大小的行高和列宽计算可见范围
 * 3. 滚动时动态更新可见范围并重新渲染
 */
class VirtualScrollManager {
  /** 可视区域宽度 */
  private viewportWidth: number = 0;
  /** 可视区域高度 */
  private viewportHeight: number = 0;
  /** 滚动位置 */
  private scrollTop: number = 0;
  private scrollLeft: number = 0;
  /** 默认行高 */
  private defaultRowHeight: number = 32;
  /** 默认列宽 */
  private defaultColumnWidth: number = 100;
  /** 行高缓存 */
  private rowHeightCache: Map<number, number> = new Map();
  /** 列宽缓存 */
  private columnWidthCache: Map<number, number> = new Map();
  /** 总行数 */
  private totalRows: number = 0;
  /** 总列数 */
  private totalColumns: number = 0;
  /** 缓冲区大小（额外渲染的区域） */
  private bufferSize: number = 5;
  /** 回调函数 */
  private onVisibleRangeChange?: (range: VisibleRange) => void;

  constructor(config: VirtualScrollConfig) {
    this.viewportWidth = config.viewportWidth;
    this.viewportHeight = config.viewportHeight;
    this.totalRows = config.totalRows;
    this.totalColumns = config.totalColumns;
    this.defaultRowHeight = config.defaultRowHeight ?? 32;
    this.defaultColumnWidth = config.defaultColumnWidth ?? 100;
    this.bufferSize = config.bufferSize ?? 5;
    this.onVisibleRangeChange = config.onVisibleRangeChange;
  }

  /**
   * 更新可视区域尺寸
   */
  setViewportSize(width: number, height: number): void {
    this.viewportWidth = width;
    this.viewportHeight = height;
    this.notifyVisibleRangeChange();
  }

  /**
   * 更新总行数
   */
  setTotalRows(totalRows: number): void {
    this.totalRows = totalRows;
    this.notifyVisibleRangeChange();
  }

  /**
   * 更新总列数
   */
  setTotalColumns(totalColumns: number): void {
    this.totalColumns = totalColumns;
    this.notifyVisibleRangeChange();
  }

  /**
   * 设置滚动位置
   */
  setScrollPosition(scrollTop: number, scrollLeft: number): void {
    const oldVisibleRange = this.getVisibleRange();

    this.scrollTop = scrollTop;
    this.scrollLeft = scrollLeft;

    const newVisibleRange = this.getVisibleRange();

    // 只有可见范围变化时才通知
    if (!this.isRangeEqual(oldVisibleRange, newVisibleRange)) {
      this.notifyVisibleRangeChange();
    }
  }

  /**
   * 获取可见范围
   */
  getVisibleRange(): VisibleRange {
    // 计算可见的起始行
    let startRow = 0;
    let accumulatedHeight = 0;

    for (let row = 0; row < this.totalRows; row++) {
      const rowHeight = this.getRowHeight(row);
      if (accumulatedHeight + rowHeight > this.scrollTop) {
        startRow = row;
        break;
      }
      accumulatedHeight += rowHeight;
    }

    // 计算可见的结束行
    let endRow = startRow;
    accumulatedHeight = 0;
    for (let row = 0; row <= startRow; row++) {
      accumulatedHeight += this.getRowHeight(row);
    }

    for (let row = startRow; row < this.totalRows; row++) {
      accumulatedHeight += this.getRowHeight(row);
      if (accumulatedHeight >= this.scrollTop + this.viewportHeight) {
        endRow = row;
        break;
      }
      if (row === this.totalRows - 1) {
        endRow = row;
      }
    }

    // 添加缓冲区
    startRow = Math.max(0, startRow - this.bufferSize);
    endRow = Math.min(this.totalRows - 1, endRow + this.bufferSize);

    // 计算可见的起始列
    let startCol = 0;
    let accumulatedWidth = 0;

    for (let col = 0; col < this.totalColumns; col++) {
      const colWidth = this.getColumnWidth(col);
      if (accumulatedWidth + colWidth > this.scrollLeft) {
        startCol = col;
        break;
      }
      accumulatedWidth += colWidth;
    }

    // 计算可见的结束列
    let endCol = startCol;
    accumulatedWidth = 0;
    for (let col = 0; col <= startCol; col++) {
      accumulatedWidth += this.getColumnWidth(col);
    }

    for (let col = startCol; col < this.totalColumns; col++) {
      accumulatedWidth += this.getColumnWidth(col);
      if (accumulatedWidth >= this.scrollLeft + this.viewportWidth) {
        endCol = col;
        break;
      }
      if (col === this.totalColumns - 1) {
        endCol = col;
      }
    }

    // 添加缓冲区
    startCol = Math.max(0, startCol - this.bufferSize);
    endCol = Math.min(this.totalColumns - 1, endCol + this.bufferSize);

    return { startRow, endRow, startCol, endCol };
  }

  /**
   * 获取行高
   */
  private getRowHeight(row: number): number {
    return this.rowHeightCache.get(row) ?? this.defaultRowHeight;
  }

  /**
   * 获取列宽
   */
  private getColumnWidth(col: number): number {
    return this.columnWidthCache.get(col) ?? this.defaultColumnWidth;
  }

  /**
   * 设置行高
   */
  setRowHeight(row: number, height: number): void {
    this.rowHeightCache.set(row, height);
    this.notifyVisibleRangeChange();
  }

  /**
   * 设置列宽
   */
  setColumnWidth(col: number, width: number): void {
    this.columnWidthCache.set(col, width);
    this.notifyVisibleRangeChange();
  }

  /**
   * 通知可见范围变化
   */
  private notifyVisibleRangeChange(): void {
    if (this.onVisibleRangeChange) {
      const range = this.getVisibleRange();
      this.onVisibleRangeChange(range);
    }
  }

  /**
   * 比较两个范围是否相等
   */
  private isRangeEqual(a: VisibleRange, b: VisibleRange): boolean {
    return (
      a.startRow === b.startRow &&
      a.endRow === b.endRow &&
      a.startCol === b.startCol &&
      a.endCol === b.endCol
    );
  }

  /**
   * 计算总内容高度
   */
  getTotalContentHeight(): number {
    let total = 0;
    for (let row = 0; row < this.totalRows; row++) {
      total += this.getRowHeight(row);
    }
    return total;
  }

  /**
   * 计算总内容宽度
   */
  getTotalContentWidth(): number {
    let total = 0;
    for (let col = 0; col < this.totalColumns; col++) {
      total += this.getColumnWidth(col);
    }
    return total;
  }
}

/**
 * 可见范围
 */
interface VisibleRange {
  startRow: number;
  endRow: number;
  startCol: number;
  endCol: number;
}

/**
 * 虚拟滚动配置
 */
interface VirtualScrollConfig {
  viewportWidth: number;
  viewportHeight: number;
  totalRows: number;
  totalColumns: number;
  defaultRowHeight?: number;
  defaultColumnWidth?: number;
  bufferSize?: number;
  onVisibleRangeChange?: (range: VisibleRange) => void;
}

/**
 * React 虚拟滚动 Hook
 */
function useVirtualScroll(
  containerRef: React.RefObject<HTMLDivElement>,
  totalRows: number,
  totalColumns: number,
  rowHeight: number = 32,
  columnWidth: number = 100
): {
  visibleRange: VisibleRange;
  scrollTop: number;
  scrollLeft: number;
  totalHeight: number;
  totalWidth: number;
  handleScroll: (event: React.UIEvent<HTMLDivElement>) => void;
} {
  const [visibleRange, setVisibleRange] = useState<VisibleRange>({
    startRow: 0,
    endRow: 20,
    startCol: 0,
    endCol: 10,
  });
  const [scrollTop, setScrollTop] = useState(0);
  const [scrollLeft, setScrollLeft] = useState(0);

  const scrollManagerRef = useRef<VirtualScrollManager | null>(null);

  // 初始化滚动管理器
  useEffect(() => {
    if (!containerRef.current) return;

    const container = containerRef.current;
    const { width, height } = container.getBoundingClientRect();

    scrollManagerRef.current = new VirtualScrollManager({
      viewportWidth: width,
      viewportHeight: height,
      totalRows,
      totalColumns,
      defaultRowHeight: rowHeight,
      defaultColumnWidth: columnWidth,
      onVisibleRangeChange: setVisibleRange,
    });

    // 初始可见范围
    setVisibleRange(scrollManagerRef.current.getVisibleRange());

    // 监听容器尺寸变化
    const resizeObserver = new ResizeObserver(entries => {
      for (const entry of entries) {
        const { width, height } = entry.contentRect;
        scrollManagerRef.current?.setViewportSize(width, height);
      }
    });

    resizeObserver.observe(container);

    return () => {
      resizeObserver.disconnect();
    };
  }, [containerRef, totalRows, totalColumns, rowHeight, columnWidth]);

  // 处理滚动事件
  const handleScroll = useCallback((event: React.UIEvent<HTMLDivElement>) => {
    const target = event.currentTarget;
    const newScrollTop = target.scrollTop;
    const newScrollLeft = target.scrollLeft;

    setScrollTop(newScrollTop);
    setScrollLeft(newScrollLeft);

    scrollManagerRef.current?.setScrollPosition(newScrollTop, newScrollLeft);
  }, []);

  // 计算总尺寸
  const totalHeight = totalRows * rowHeight;
  const totalWidth = totalColumns * columnWidth;

  return {
    visibleRange,
    scrollTop,
    scrollLeft,
    totalHeight,
    totalWidth,
    handleScroll,
  };
}
```

## 八、总结

- **增量计算**：通过拓扑排序确定最优计算顺序

### 8.4 筛选与排序
