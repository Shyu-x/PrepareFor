# 数据大屏性能优化与布局方案

## 前言

数据大屏（Data Dashboard / Data Screen）是企业级数据可视化中最常见的应用场景之一。它通常用于展示实时监控数据、业务统计报表、KPI 指标等信息，需要在各种尺寸的显示设备上呈现出专业、美观的效果。

本章将从实战角度出发，详细讲解数据大屏的开发技术选型、响应式布局方案、性能优化策略以及首屏加载优化技巧。无论你是要开发一个企业级数据监控中心，还是一个简单的业务数据展示页面，本指南都将提供全面的技术指导。

### 1.2 技术选型对比

| 技术方案 | 适用场景 | 优点 | 缺点 |
|----------|---------|------|------|
| **ECharts** | 中小规模图表、仪表盘 | 学习成本低、配置方便、性能良好 | 复杂图表能力有限 |
| **ECharts + WebGL** | 大数据量、复杂可视化 | 性能极强、支持 3D | 学习成本高 |
| **D3.js** | 完全自定义可视化 | 灵活性最高 | 开发效率低 |
| **Three.js / R3F** | 3D 可视化场景 | 3D 能力强大 | 学习曲线陡峭 |

| **rem / vw 方案** | 简单适配 | 简单直接 | 复杂布局适配困难 |
| **Scale 方案** | 精确适配 | 等比缩放、清晰度高 | 可能需要微调 |

### 2.3 Scale 缩放方案

```typescript
/**
 * Scale 缩放方案
 *
 * 核心思想：
 * 1. 按照设计稿的基准分辨率开发
 * 2. 运行时会计算当前屏幕与设计稿的比例
 * 3. 使用 CSS transform 进行等比缩放
 *
 * 优点：严格等比，不会出现布局错乱
 * 缺点：可能超出屏幕，需要处理溢出
 */

/**
 * Scale 计算 Hook
 */
import { useState, useEffect, useCallback } from 'react';

interface ScaleConfig {
  /** 设计稿宽度 */
  designWidth: number;
  /** 设计稿高度 */
  designHeight: number;
  /** 是否启用缩放 */
  enabled: boolean;
  /** 缩放模式：'scale' | 'width' | 'height' */
  mode: 'scale' | 'width' | 'height';
}

interface ScaleResult {
  /** 缩放比例 */
  scale: number;
  /** 缩放样式 */
  style: {
    transform: string;
    width: string;
    height: string;
    overflow: string;
  };
  /** 是否溢出 */
  isOverflow: boolean;
}

/**
 * 使用 scale 进行响应式缩放
 *
 * @param config 配置
 * @returns 缩放样式
 */
function useScale(config: ScaleConfig): ScaleResult {
  const { designWidth, designHeight, enabled, mode } = config;

  const [result, setResult] = useState<ScaleResult>({
    scale: 1,
    style: {
      transform: 'scale(1)',
      width: '100%',
      height: '100%',
      overflow: 'hidden',
    },
    isOverflow: false,
  });

  const calculateScale = useCallback(() => {
    if (!enabled) {
      return {
        scale: 1,
        style: {
          transform: 'none',
          width: '100%',
          height: '100%',
          overflow: 'auto',
        },
        isOverflow: false,
      };
    }

    // 获取当前窗口尺寸
    const windowWidth = window.innerWidth;
    const windowHeight = window.innerHeight;

    // 计算宽高比例
    const scaleX = windowWidth / designWidth;
    const scaleY = windowHeight / designHeight;

    let scale: number;
    let isOverflow = false;

    if (mode === 'width') {
      // 固定宽度缩放
      scale = scaleX;
    } else if (mode === 'height') {
      // 固定高度缩放
      scale = scaleY;
    } else {
      // 自动选择（取较小值，确保不溢出）
      scale = Math.min(scaleX, scaleY);
      isOverflow = scaleX < 1 || scaleY < 1;
    }

    // 计算缩放后的尺寸
    const scaledWidth = designWidth * scale;
    const scaledHeight = designHeight * scale;

    return {
      scale,
      style: {
        transform: `scale(${scale})`,
        width: `${designWidth}px`,
        height: `${designHeight}px`,
        // 居中定位
        position: 'absolute' as const,
        left: `${(windowWidth - scaledWidth) / 2}px`,
        top: `${(windowHeight - scaledHeight) / 2}px`,
        transformOrigin: 'center center' as const,
        overflow: isOverflow ? 'auto' : 'hidden',
      },
      isOverflow,
    };
  }, [designWidth, designHeight, enabled, mode]);

  useEffect(() => {
    // 初始计算
    setResult(calculateScale());

    // 监听窗口变化
    const handleResize = () => {
      setResult(calculateScale());
    };

    window.addEventListener('resize', handleResize);

    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, [calculateScale]);

  return result;
}

/**
 * Scale 容器组件
 */
interface ScaleContainerProps {
  children: React.ReactNode;
  designWidth?: number;
  designHeight?: number;
  enabled?: boolean;
  mode?: 'scale' | 'width' | 'height';
}

const ScaleContainer: React.FC<ScaleContainerProps> = ({
  children,
  designWidth = 1920,
  designHeight = 1080,
  enabled = true,
  mode = 'scale',
}) => {
  const { style } = useScale({
    designWidth,
    designHeight,
    enabled,
    mode,
  });

  return (
    <div
      style={{
        width: '100vw',
        height: '100vh',
        position: 'relative',
        overflow: 'hidden',
        background: '#0a1929',
      }}
    >
      <div
        style={{
          ...style,
          display: 'flex',
          flexDirection: 'column',
        }}
      >
        {children}
      </div>
    </div>
  );
};

export default ScaleContainer;
```

class DataCache {
  private cache: Map<string, { data: any; timestamp: number }> = new Map();
  private maxSize: number;

  constructor(maxSize: number = 100) {
    this.maxSize = maxSize;
  }

  set(key: string, data: any): void {
    // 如果缓存已满，删除最旧的条目
    if (this.cache.size >= this.maxSize) {
      const oldestKey = this.cache.keys().next().value;
      this.cache.delete(oldestKey);
    }

    this.cache.set(key, {
      data,
      timestamp: Date.now(),
    });
  }

    const entry = this.cache.get(key);
    if (!entry) return null;

    // 检查是否过期（5分钟）
    if (Date.now() - entry.timestamp > 5 * 60 * 1000) {
      this.cache.delete(key);
      return null;
    }

    return entry.data;
  }

  clear(): void {
    this.cache.clear();
  }

/**

class RenderPerformanceMonitor {
  private marks: Map<string, number> = new Map();

  /**
   * 记录性能标记
   */
  mark(name: string): void {
    this.marks.set(name, performance.now());
  }

  /**
   * 测量两个标记之间的耗时
   */
  measure(name: string, startMark: string, endMark: string): number {
    const start = this.marks.get(startMark);
    const end = this.marks.get(endMark);

    if (start && end) {
      const duration = end - start;
      console.log(`[Performance] ${name}: ${duration.toFixed(2)}ms`);
      return duration;
    }

    return -1;
  }

  /**
   * 测量组件渲染时间
   */
  measureRender(componentName: string, renderFn: () => void): void {
    const start = performance.now();
    renderFn();
    const end = performance.now();

    console.log(`[Render] ${componentName}: ${(end - start).toFixed(2)}ms`);
  }

  /**
   * 监测 FPS
   */
  monitorFPS(callback: (fps: number) => void): () => void {
    let frameCount = 0;
    let lastTime = performance.now();

    const tick = () => {
      frameCount++;
      const currentTime = performance.now();

      if (currentTime - lastTime >= 1000) {
        const fps = frameCount;
        frameCount = 0;
        lastTime = currentTime;
        callback(fps);

        // FPS 过低时警告
        if (fps < 30) {
          console.warn(`[Performance] Low FPS: ${fps}`);
        }
      }

      requestAnimationFrame(tick);
    };

    const frameId = requestAnimationFrame(tick);

    // 返回停止函数
    return () => {
      cancelAnimationFrame(frameId);
    };
  }
}

    // 模拟数据加载
    const loadData = async () => {
      await new Promise((resolve) => setTimeout(resolve, 1000));
      setLoading(false);
    };

    loadData();
  }, []);

  // 更新当前时间
  useEffect(() => {
    const timer = setInterval(() => {
      setCurrentTime(new Date());
    }, 1000);

    return () => clearInterval(timer);
  }, []);

  // FPS 监测
  useEffect(() => {
    let frameCount = 0;
    let lastTime = performance.now();

    const tick = () => {
      frameCount++;
      const now = performance.now();

      if (now - lastTime >= 1000) {
        setFPS(frameCount);
        frameCount = 0;
        lastTime = now;
      }

      requestAnimationFrame(tick);
    };

    const frameId = requestAnimationFrame(tick);

    return () => cancelAnimationFrame(frameId);
  }, []);

  // 渲染加载状态
  if (loading) {
    return <LoadingSkeleton />;
  }

  return (
    <ScaleContainer designWidth={1920} designHeight={1080}>
      <div className="dashboard">

const LineChart: React.FC<{ data: any[]; title: string }> = ({ data, title }) => {
  const chartRef = useRef<HTMLDivElement>(null);
  const chartInstanceRef = useRef<echarts.ECharts | null>(null);

  useEffect(() => {
    if (!chartRef.current) return;

    // 初始化图表
    const chart = echarts.init(chartRef.current);
    chartInstanceRef.current = chart;

    // 配置
    const option = {
      title: {
        text: title,
        textStyle: { color: '#fff', fontSize: 16 },
        left: 20,
        top: 10,
      },
      tooltip: {
        trigger: 'axis',
        backgroundColor: 'rgba(0, 20, 40, 0.9)',
        borderColor: '#00BFFF',
        textStyle: { color: '#fff' },
      },
      grid: {
        left: 50,
        right: 20,
        top: 50,
        bottom: 30,
      },
      xAxis: {
        type: 'category',
        data: data.map((d) => d.name),
        axisLine: { lineStyle: { color: '#1E3A5F' } },
        axisLabel: { color: '#7B9BBF' },
      },
      yAxis: {
        type: 'value',
        axisLine: { show: false },
        splitLine: { lineStyle: { color: '#1E3A5F', type: 'dashed' } },
        axisLabel: { color: '#7B9BBF' },
      },
      series: [
        {
          type: 'line',
          data: data.map((d) => d.value),
          smooth: true,
          lineStyle: { color: '#00BFFF', width: 2 },
          areaStyle: {
            color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
              { offset: 0, color: 'rgba(0, 191, 255, 0.5)' },
              { offset: 1, color: 'rgba(0, 191, 255, 0.1)' },
            ]),
          },
        },
      ],
    };

    chart.setOption(option);

    // 响应窗口大小变化
    const handleResize = () => chart.resize();
    window.addEventListener('resize', handleResize);

    // 清理
    return () => {
      window.removeEventListener('resize', handleResize);
      chart.dispose();
    };
  }, [data, title]);

  return <div ref={chartRef} style={{ width: '100%', height: '100%' }} />;
};
