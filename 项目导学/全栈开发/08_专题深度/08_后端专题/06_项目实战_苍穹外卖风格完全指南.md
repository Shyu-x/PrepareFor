> 本文将带你完整实现一个外卖配送平台，涵盖商户端、用户端、骑手端三大业务场景，订单系统、支付系统、分布式Session等核心模块。通过这个项目，你将掌握Spring Boot + MySQL + Redis + RedisSession的完整企业级开发技能。

### 3.2 公共模块

```java
/**
 * 统一返回结果封装
 */
public class Result<T> implements Serializable {

    private Integer code;      // 状态码
    private String message;    // 消息
    private T data;           // 数据

    // 快速返回成功
    public static <T> Result<T> success() {
        return success(null);
    }

    public static <T> Result<T> success(T data) {
        Result<T> result = new Result<>();
        result.setCode(ResultCode.SUCCESS.getCode());
        result.setMessage(ResultCode.SUCCESS.getMessage());
        result.setData(data);
        return result;
    }

    // 快速返回失败
    public static <T> Result<T> error(String message) {
        Result<T> result = new Result<>();
        result.setCode(ResultCode.ERROR.getCode());
        result.setMessage(message);
        return result;
    }

    public static <T> Result<T> error(Integer code, String message) {
        Result<T> result = new Result<>();
        result.setCode(code);
        result.setMessage(message);
        return result;
    }

    // Getter和Setter
    public Integer getCode() {
        return code;
    }

    public void setCode(Integer code) {
        this.code = code;
    }

    public String getMessage() {
        return message;
    }

    public void setMessage(String message) {
        this.message = message;
    }

    public T getData() {
        return data;
    }

    public void setData(T data) {
        this.data = data;
    }
}

/**
 * 状态码枚举
 */

@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(BusinessException.class)
    public Result<?> handleBusinessException(BusinessException e) {
        return Result.error(e.getCode(), e.getMessage());
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public Result<?> handleValidationException(MethodArgumentNotValidException e) {
        String message = e.getBindingResult().getFieldErrors().stream()
            .map(FieldError::getDefaultMessage)
            .collect(Collectors.joining(","));
        return Result.error(ResultCode.VALIDATION_ERROR.getCode(), message);
    }

    @ExceptionHandler(Exception.class)
    public Result<?> handleException(Exception e) {
        log.error("系统异常", e);
        return Result.error(ResultCode.SYSTEM_ERROR.getCode(), e.getMessage());
    }
}

/**
 * Token工具类

public class JwtUtils {

    // JWT配置
    private static final String SECRET = "your-secret-key";
    private static final long EXPIRATION = 24 * 60 * 60 * 1000; // 24小时

    /**
     * 生成Token
     */
    public static String generateToken(Long userId, String role) {
        Date now = new Date();
        Date expiration = new Date(now.getTime() + EXPIRATION);

        return Jwts.builder()
            .setSubject(String.valueOf(userId))
            .claim("role", role)
            .setIssuedAt(now)
            .setExpiration(expiration)
            .signWith(SignatureAlgorithm.HS256, SECRET)
            .compact();
    }

    /**
     * 解析Token
     */
    public static Claims parseToken(String token) {
        return Jwts.parser()
            .setSigningKey(SECRET)
            .parseClaimsJws(token)
            .getBody();
    }

    /**
     * 从Token获取用户ID
     */
    public static Long getUserIdFromToken(String token) {
        Claims claims = parseToken(token);
        return Long.parseLong(claims.getSubject());
    }
}
```

```java
/**
 * 订单状态枚举

public enum OrderStatus {

    PENDING_PAYMENT(0, "待支付"),
    PAID(1, "已支付"),
    PENDING_ACCEPT(2, "待接单"),
    PENDING_PREPARE(3, "待备餐"),
    DELIVERING(4, "配送中"),
    COMPLETED(5, "已完成"),
    CANCELLED(6, "已取消"),
    REFUNDING(7, "退款中"),
    REFUNDED(8, "已退款");

    private final Integer code;
    private final String description;

    OrderStatus(Integer code, String description) {
        this.code = code;
        this.description = description;
    }

    public Integer getCode() {
        return code;
    }

    public String getDescription() {
        return description;
    }

    public static OrderStatus fromCode(Integer code) {
        for (OrderStatus status : values()) {
            if (status.code.equals(code)) {
                return status;
            }
        }
        throw new BusinessException(ResultCode.ORDER_STATUS_ERROR);
    }
}

 * 订单状态转换服务

@Service
public class OrderStatusService {

    /**
     * 订单状态转换校验
     */
    public void validateStatusTransition(Order order, OrderStatus newStatus) {
        OrderStatus currentStatus = OrderStatus.fromCode(order.getStatus());

        // 状态转换规则
        Map<OrderStatus, Set<OrderStatus>> allowedTransitions = new HashMap<>();
        allowedTransitions.put(OrderStatus.PENDING_PAYMENT,
            EnumSet.of(OrderStatus.PAID, OrderStatus.CANCELLED));
        allowedTransitions.put(OrderStatus.PAID,
            EnumSet.of(OrderStatus.PENDING_ACCEPT, OrderStatus.REFUNDING));
        allowedTransitions.put(OrderStatus.PENDING_ACCEPT,
            EnumSet.of(OrderStatus.PENDING_PREPARE, OrderStatus.CANCELLED));
        allowedTransitions.put(OrderStatus.PENDING_PREPARE,
            EnumSet.of(OrderStatus.DELIVERING, OrderStatus.REFUNDING));
        allowedTransitions.put(OrderStatus.DELIVERING,
            EnumSet.of(OrderStatus.COMPLETED, OrderStatus.REFUNDING));
        allowedTransitions.put(OrderStatus.COMPLETED,
            EnumSet.of(OrderStatus.REFUNDING));
        allowedTransitions.put(OrderStatus.REFUNDING,
            EnumSet.of(OrderStatus.REFUNDED, OrderStatus.COMPLETED));

        Set<OrderStatus> allowed = allowedTransitions.get(currentStatus);
        if (allowed == null || !allowed.contains(newStatus)) {
            throw new BusinessException(ResultCode.ORDER_STATUS_ERROR,
                "订单状态不允许从 " + currentStatus.getDescription() +
                " 转换到 " + newStatus.getDescription());
        }
    }

    /**
     * 更新订单状态
     */
    @Transactional
    public void updateOrderStatus(Long orderId, OrderStatus newStatus) {
        Order order = orderMapper.selectById(orderId);
        if (order == null) {
            throw new BusinessException(ResultCode.ORDER_NOT_FOUND);
        }

        // 校验状态转换
        validateStatusTransition(order, newStatus);

        OrderStatus oldStatus = OrderStatus.fromCode(order.getStatus());

        // 更新状态
        order.setStatus(newStatus.getCode());
        order.setUpdateTime(new Date());

        // 处理状态变化的后续逻辑
        handleStatusChange(order, oldStatus, newStatus);

        orderMapper.updateById(order);

        // 发送状态变更事件
        applicationEventPublisher.publishEvent(
            new OrderStatusChangeEvent(orderId, oldStatus, newStatus));
    }

    /**
     * 处理状态变更的后续逻辑
     */
    private void handleStatusChange(Order order, OrderStatus oldStatus, OrderStatus newStatus) {
        switch (newStatus) {
            case PAID:
                // 支付成功后，通知商家有新订单
                notificationService.notifyMerchantNewOrder(order.getMerchantId(), order.getId());
                // 启动超时未支付检查
                orderTimeoutScheduler.scheduleTimeoutCheck(order.getId());
                break;

            case PENDING_ACCEPT:
                // 商家接单，启动备餐超时检查
                orderTimeoutScheduler.schedulePrepareTimeoutCheck(order.getId());
                break;

            case DELIVERING:
                // 骑手取餐，发送骑手位置
                startLocationTracking(order.getId());
                break;

            case COMPLETED:
                // 订单完成，结算商家货款、骑手收入
                settleMerchantPayment(order);
                settleRiderPayment(order);
                // 检查是否需要发放优惠券
                couponService.checkAndGrantCoupon(order.getUserId());
                break;

            case CANCELLED:
                // 取消订单，恢复库存
                restoreStock(order);
                // 如果已支付，退款
                if (oldStatus.getCode() >= OrderStatus.PAID.getCode()) {
                    refundService.refund(order);
                }
                break;

            default:
                break;
        }
    }
}
```

```java
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) {
        // 放行OPTIONS请求
        if ("OPTIONS".equals(request.getMethod())) {
            return true;
        }

        // 从Header或Cookie获取Token
        String token = request.getHeader("Authorization");
        if (token == null || token.isEmpty()) {
            token = request.getHeader("token");
        }
        if (token == null || token.isEmpty()) {
            Cookie[] cookies = request.getCookies();
            if (cookies != null) {
                for (Cookie cookie : cookies) {
                    if ("token".equals(cookie.getName())) {
                        token = cookie.getValue();
                        break;
                    }
                }
            }
        }

        if (token == null || token.isEmpty()) {
            response.setStatus(401);
            return false;
        }

        // 验证Token
        try {
            Claims claims = JwtUtils.parseToken(token);
            Long userId = Long.parseLong(claims.getSubject());
            String role = (String) claims.get("role");

            // 存储到ThreadLocal
            UserContext.setUserId(userId);
            UserContext.setUserRole(role);
            UserContext.setToken(token);

            return true;

        } catch (JwtException e) {
            response.setStatus(401);
            return false;
        }
    }

    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response,
                               Object handler, Exception ex) {
        // 清理ThreadLocal
        UserContext.clear();
    }
```

```java
@Slfined

        if (cartItems == null || cartItems.isEmpty()) {
            throw new BusinessException(ResultCode.ERROR, "购物车为空");
        }

        // 4. 计算价格
        PriceCalculationResult priceResult = calculatePrice(cartItems, createDTO);

        // 5. 生成订单号
        String orderNo = generateOrderNo();

        // 6. 创建订单
        Order order = new Order();
        order.setOrderNo(orderNo);
        order.setUserId(userId);
        order.setMerchantId(createDTO.getMerchantId());
        order.setAddressId(address.getId());
        order.setConsignee(address.getConsignee());
        order.setPhone(address.getPhone());
        order.setProvince(address.getProvince());
        order.setCity(address.getCity());
        order.setDistrict(address.getDistrict());
        order.setDetailAddress(address.getDetailAddress());
        order.setTotalAmount(priceResult.getTotalAmount());
        order.setDiscountAmount(priceResult.getDiscountAmount());
        order.setPayAmount(priceResult.getPayAmount());
        order.setStatus(OrderStatus.PENDING_PAYMENT.getCode());
        order.setRemark(createDTO.getRemark());
        order.setEstimateDeliveryTime(calculateDeliveryTime());
        order.setCreateTime(new Date());

        orderMapper.insert(order);

        // 7. 创建订单明细
        List<OrderItem> orderItems = createOrderItems(order.getId(), cartItems);
        orderItemMapper.insertBatch(orderItems);

        // 8. 清理购物车
        cartService.clearCart(userId, createDTO.getMerchantId());

        // 9. 锁定库存（预留）
        productService.reserveStock(cartItems);

        // 10. 使用优惠券
        if (createDTO.getCouponId() != null) {
            couponService.useCoupon(userId, createDTO.getCouponId(), order.getId());
        }

        // 11. 发送订单创建消息（异步更新商家销量等）
        rabbitTemplate.convertAndSend(
            "order.exchange",
            "order.created",
            new OrderCreatedEvent(order.getId())
        );

        // 12. 启动超时取消
        scheduleOrderTimeoutCancel(order.getId());

        // 13. 返回结果
        OrderCreateVO vo = new OrderCreateVO();
        vo.setOrderId(order.getId());
        vo.setOrderNo(orderNo);
        vo.setPayAmount(priceResult.getPayAmount());
        vo.setEstimatedDeliveryTime(order.getEstimateDeliveryTime());

        return vo;
    }

    /**
     * 计算价格
     */
    private PriceCalculationResult calculatePrice(List<CartItem> cartItems,
                                                  OrderCreateDTO createDTO) {
        BigDecimal totalAmount = BigDecimal.ZERO;
        BigDecimal discountAmount = BigDecimal.ZERO;

        // 计算商品小计
        for (CartItem item : cartItems) {
            BigDecimal itemAmount = item.getPrice().multiply(BigDecimal.valueOf(item.getQuantity()));
            totalAmount = totalAmount.add(itemAmount);
        }

        // 促销活动优惠
        BigDecimal promotionDiscount = productService.calculatePromotionDiscount(
            createDTO.getMerchantId(), totalAmount);
        discountAmount = discountAmount.add(promotionDiscount);

        // 优惠券优惠
        if (createDTO.getCouponId() != null) {
            BigDecimal couponDiscount = couponService.calculateDiscount(
                createDTO.getCouponId(), totalAmount);
            discountAmount = discountAmount.add(couponDiscount);
        }

        // 新客立减
        BigDecimal newUserDiscount = checkAndApplyNewUserDiscount(
            createDTO.getUserId(), createDTO.getMerchantId());
        discountAmount = discountAmount.add(newUserDiscount);

        // 防止优惠金额超过总价
        if (discountAmount.compareTo(totalAmount) > 0) {
            discountAmount = totalAmount;
        }

        BigDecimal payAmount = totalAmount.subtract(discountAmount);

        return new PriceCalculationResult(totalAmount, discountAmount, payAmount);
    }

    /**
     * 生成订单号
     * 规则：时间戳(14位) + 随机数(6位) + 业务标识(2位)
     */
    private String generateOrderNo() {
        String timestamp = new SimpleDateFormat("yyyyMMddHHmmss").format(new Date());
        String random = String.format("%06d", new Random().nextInt(999999));
        String bizCode = "01"; // 外卖订单
        return timestamp + random + bizCode;
    }

    /**
     * 计算预计送达时间
     * = 当前时间 + 商家备餐时间 + 配送时间
     */
    private Date calculateDeliveryTime() {
        Merchant merchant = merchantService.getCurrentMerchant();

        int prepareTime = merchant.getAvgPrepareTime(); // 商家平均备餐时间（分钟）
        int deliveryTime = 30; // 固定配送时间（分钟）

        return DateUtils.addMinutes(new Date(), prepareTime + deliveryTime);
    }

    /**
     * 订单超时取消
     */
    private void scheduleOrderTimeoutCancel(Long orderId) {
        // 15分钟未支付自动取消
        rabbitTemplate.schedule(
            "order.timeout.cancel",
            new OrderTimeoutCancelEvent(orderId),
            Duration.ofMinutes(15)
        );
    }
```

```java
/**

@Service
@Slfined
public class PaymentService {

    @Autowired
    private OrderMapper orderMapper;

    @Autowired
    private PaymentMapper paymentMapper;

    @Autowired
    private WechatPayService wechatPayService;

    @Autowired
    private AlipayService alipayService;

    /**
     * 创建支付
     */
    public PaymentVO createPayment(Long orderId, Integer payType) {
        // 1. 校验订单
        Order order = orderMapper.selectById(orderId);
        if (order == null) {
            throw new BusinessException(ResultCode.ORDER_NOT_FOUND);
        }

        if (!OrderStatus.PENDING_PAYMENT.getCode().equals(order.getStatus())) {
            throw new BusinessException(ResultCode.ORDER_PAID);
        }

        // 2. 校验订单金额
        if (order.getPayAmount().compareTo(BigDecimal.ZERO) <= 0) {
            // 金额为0，直接跳转支付成功
            return PaymentVO.builder()
                .payAmount(BigDecimal.ZERO)
                .build();
        }

        // 3. 根据支付方式创建支付
        Payment payment = new Payment();
        payment.setOrderId(orderId);
        payment.setOrderNo(order.getOrderNo());
        payment.setPayType(payType);
        payment.setPayAmount(order.getPayAmount());
        payment.setStatus(PaymentStatus.PENDING.getCode());
        payment.setCreateTime(new Date());

        paymentMapper.insert(payment);

        // 4. 调用第三方支付
        PaymentResult result;
        if (payType.equals(PayType.WECHAT.getCode())) {
            result = wechatPayService.createOrder(order, payment);
        } else if (payType.equals(PayType.ALIPAY.getCode())) {
            result = alipayService.createOrder(order, payment);
        } else {
            throw new BusinessException(ResultCode.PAYMENT_FAILED);
        }

        // 5. 更新支付记录
        payment.setTransactionId(result.getTransactionId());
        paymentMapper.updateById(payment);

        // 6. 返回支付信息
        return PaymentVO.builder()
            .paymentId(payment.getId())
            .payAmount(order.getPayAmount())
            .qrCode(result.getQrCode())
            .payUrl(result.getPayUrl())
            .build();
    }

    /**
     * 微信支付回调
     */
    public void handleWechatPayCallback(String xmlData) {
        // 1. 解析回调数据
        WechatPayCallback callback = wechatPayService.parseCallback(xmlData);

        // 2. 验签
        if (!wechatPayService.verifySignature(callback)) {
            log.warn("微信支付回调验签失败");
            return;
        }

        // 3. 处理支付结果
        if ("SUCCESS".equals(callback.getResultCode())) {
            handlePaySuccess(callback.getOutTradeNo(), callback.getTransactionId());
        } else {
            handlePayFailed(callback.getOutTradeNo(), callback.getErrCode());
        }
    }

    /**
     * 支付成功处理
     */
    @Transactional(rollbackFor = Exception.class)
    public void handlePaySuccess(String orderNo, String transactionId) {
        // 1. 查询支付记录
        Payment payment = paymentMapper.selectByOrderNo(orderNo);
        if (payment == null) {
            log.warn("支付记录不存在: {}", orderNo);
            return;
        }

        if (PaymentStatus.SUCCESS.getCode().equals(payment.getStatus())) {
            log.info("订单已支付: {}", orderNo);
            return;
        }

        // 2. 更新支付记录
        payment.setStatus(PaymentStatus.SUCCESS.getCode());
        payment.setTransactionId(transactionId);
        payment.setCallbackTime(new Date());
        paymentMapper.updateById(payment);

        // 3. 查询订单
        Order order = orderMapper.selectById(payment.getOrderId());
        if (order == null) {
            log.warn("订单不存在: {}", payment.getOrderId());
            return;
        }

        // 4. 更新订单状态
        order.setStatus(OrderStatus.PENDING_ACCEPT.getCode());
        order.setPayType(payment.getPayType());
        order.setUpdateTime(new Date());
        orderMapper.updateById(order);

        // 5. 发送订单支付成功消息
        rabbitTemplate.convertAndSend(
            "order.exchange",
            "order.paid",
            new OrderPaidEvent(order.getId())
        );

        // 6. 通知商家有新订单
        notificationService.notifyMerchantNewOrder(order.getMerchantId(), order.getId());

        log.info("订单支付成功: {}, 交易号: {}", orderNo, transactionId);
    }

    public void handleAlipayCallback(String xmlData) {

        if ("TRADE_SUCCESS".equals(callback.getTradeStatus()) ||
            "TRADE_FINISHED".equals(callback.getTradeStatus())) {
```

```java
/**
 * 订单配送服务
 */
@Service
@Slfined
public class DeliveryService {

    @Autowired
    private OrderMapper orderMapper;

    @Autowired
    private RiderMapper riderMapper;

    @Autowired
    private LocationService locationService;

    @Autowired
    private NotificationService notificationService;

    @Autowired
    private RabbitTemplate rabbitTemplate;

    /**
     * 分配骑手
     * 策略：距离优先 + 负载均衡
     */
    public void assignRider(Long orderId) {
        Order order = orderMapper.selectById(orderId);
        Merchant merchant = merchantService.getById(order.getMerchantId());
        Address address = addressService.getById(order.getAddressId());

        // 1. 获取可用骑手列表
        List<Rider> availableRiders = riderMapper.selectAvailableRiders(
            DeliveryStatus.IDLE.getCode()
        );

        if (availableRiders.isEmpty()) {
            // 没有可用骑手，通知用户等待
            log.warn("无可用骑手，订单: {}", orderId);
            notificationService.notifyUserRiderWaiting(order.getUserId(), orderId);
            return;
        }

        // 2. 计算距离并排序
        Map<Long, Double> riderDistances = new HashMap<>();
        for (Rider rider : availableRiders) {
            double distance = calculateDistance(
                rider.getCurrentLatitude(), rider.getCurrentLongitude(),
                merchant.getLatitude(), merchant.getLongitude()
            );
            riderDistances.put(rider.getId(), distance);
        }

        // 3. 选择最优骑手（距离近 + 配送单少）
        Rider bestRider = selectBestRider(availableRiders, riderDistances);

        // 4. 更新订单
        order.setRiderId(bestRider.getId());
        order.setStatus(OrderStatus.DELIVERING.getCode());
        order.setUpdateTime(new Date());
        orderMapper.updateById(order);

        // 5. 更新骑手状态
        bestRider.setStatus(DeliveryStatus.DELIVERING.getCode());
        bestRider.setCurrentOrderId(orderId);
        bestRider.setUpdateTime(new Date());
        riderMapper.updateById(bestRider);

        // 6. 通知骑手
        notificationService.notifyRiderNewOrder(bestRider.getId(), orderId);

        // 7. 开启位置追踪
        locationService.startTracking(orderId, bestRider.getId());

        // 8. 计算预计送达时间
        calculateAndUpdateDeliveryTime(order, bestRider, merchant, address);

        log.info("订单分配骑手完成: 订单={}, 骑手={}", orderId, bestRider.getId());
    }

    /**
     * 选择最优骑手
     */
    private Rider selectBestRider(List<Rider> riders, Map<Long, Double> distances) {
        return riders.stream()
            .min(Comparator
                .comparingDouble((Rider r) -> distances.get(r.getId()))  // 距离优先
                .thenComparingInt(Rider::getTotalOrders))  // 再考虑配送经验
            .orElse(riders.get(0));
    }

    /**
     * 计算两点之间的距离（米）
     */
    private double calculateDistance(Double lat1, Double lon1, Double lat2, Double lon2) {
        if (lat1 == null || lon1 == null || lat2 == null || lon2 == null) {
            return Double.MAX_VALUE;
        }

        // 使用Haversine公式计算球面距离
        double R = 6371000; // 地球半径（米）
        double dLat = Math.toRadians(lat2 - lat1);
        double dLon = Math.toRadians(lon2 - lon1);
        double a = Math.sin(dLat/2) * Math.sin(dLat/2) +
                   Math.cos(Math.toRadians(lat1)) * Math.cos(Math.toRadians(lat2)) *
                   Math.sin(dLon/2) * Math.sin(dLon/2);
        double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        return R * c;
    }

    /**
     * 计算预计送达时间
     */
    private void calculateAndUpdateDeliveryTime(Order order, Rider rider,
                                                Merchant merchant, Address address) {
        // 商家到用户地址的距离
        double deliveryDistance = calculateDistance(
            merchant.getLatitude(), merchant.getLongitude(),
            address.getLatitude(), address.getLongitude()
        );

        // 骑手当前位置到商家的距离
        double riderToMerchant = calculateDistance(
            rider.getCurrentLatitude(), rider.getCurrentLongitude(),
            merchant.getLatitude(), merchant.getLongitude()
        );

        // 估算时间（假设平均速度30km/h）
        double speed = 30.0 / 3.6; // m/s
        int deliveryMinutes = (int) ((riderToMerchant + deliveryDistance) / speed / 60) + 5;

        Date estimatedTime = DateUtils.addMinutes(new Date(), deliveryMinutes);
        order.setEstimateDeliveryTime(estimatedTime);
        orderMapper.updateById(order);

        // 通知用户预计送达时间
        notificationService.notifyUserDeliveryTime(order.getUserId(), orderId, estimatedTime);
    }

    /**
     * 骑手位置更新
     */
    public void updateRiderLocation(Long riderId, Double latitude, Double longitude) {
        // 1. 更新骑手位置
        Rider rider = riderMapper.selectById(riderId);
        rider.setCurrentLatitude(latitude);
        rider.setCurrentLongitude(longitude);
        rider.setUpdateTime(new Date());
        riderMapper.updateById(rider);

        // 2. 如果有正在配送的订单，更新订单路径
        if (rider.getCurrentOrderId() != null) {
            locationService.recordLocation(rider.getCurrentOrderId(), latitude, longitude);
        }
    }

    /**
     * 骑手确认送达
     */
    @Transactional(rollbackFor = Exception.class)
    public void riderConfirmDelivery(Long orderId, Long riderId) {
        Order order = orderMapper.selectById(orderId);

        // 1. 校验骑手
        if (!order.getRiderId().equals(riderId)) {
            throw new BusinessException(ResultCode.ERROR, "无权操作此订单");
        }

        // 2. 校验订单状态
        if (!OrderStatus.DELIVERING.getCode().equals(order.getStatus())) {
            throw new BusinessException(ResultCode.ORDER_STATUS_ERROR);
        }

        // 3. 更新订单状态
        order.setStatus(OrderStatus.COMPLETED.getCode());
        order.setActualDeliveryTime(new Date());
        order.setUpdateTime(new Date());
        orderMapper.updateById(order);

        // 4. 更新骑手状态
        Rider rider = riderMapper.selectById(riderId);
        rider.setStatus(DeliveryStatus.IDLE.getCode());
        rider.setCurrentOrderId(null);
        rider.setTotalOrders(rider.getTotalOrders() + 1);
        rider.setUpdateTime(new Date());
        riderMapper.updateById(rider);

        // 5. 停止位置追踪
        locationService.stopTracking(orderId);

        // 6. 计算骑手收入
        calculateRiderIncome(order, rider);

        // 7. 通知用户
        notificationService.notifyUserOrderDelivered(order.getUserId(), orderId);

        // 8. 通知商家
        notificationService.notifyMerchantOrderDelivered(order.getMerchantId(), orderId);

        // 9. 发送订单完成事件
        rabbitTemplate.convertAndSend(
            "order.exchange",
            "order.completed",
            new OrderCompletedEvent(orderId)
        );

        log.info("骑手确认送达: 订单={}, 骑手={}", orderId, riderId);
    }

    /**
     * 计算骑手收入
     */
    private void calculateRiderIncome(Order order, Rider rider) {
        // 基础配送费
        BigDecimal baseFee = new BigDecimal("5.00");

        // 距离补贴
        Address address = addressService.getById(order.getAddressId());
        Merchant merchant = merchantService.getById(order.getMerchantId());
        double distance = calculateDistance(
            merchant.getLatitude(), merchant.getLongitude(),
            address.getLatitude(), address.getLongitude()
        );

        BigDecimal distanceFee = BigDecimal.ZERO;
        if (distance > 3000) { // 超过3公里
            distanceFee = new BigDecimal((distance - 3000) / 1000 * 1.5);
        }

        // 订单金额补贴
        BigDecimal orderFee = order.getPayAmount().multiply(new BigDecimal("0.05"));

        // 总收入
        BigDecimal totalIncome = baseFee.add(distanceFee).add(orderFee);

        // 更新骑手余额
        rider.setBalance(rider.getBalance().add(totalIncome));
        riderMapper.updateById(rider);

        // 记录收入明细
        incomeRecordService.record(rider.getId(), order.getId(), totalIncome);
    }
}
```
