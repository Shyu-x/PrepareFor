```go
    // 3. 创建负载均衡器
    lb := cfg.LoadBalancer

    // 4. 创建连接池
    pool := NewConnPool(cfg.Pool)

    // 5. 组装客户端
    return &kitexClient{
        serviceName:  serviceName,
        transporter:  transporter,
        loadBalancer: lb,
        connPool:     pool,
        options:      cfg,
    }, nil

// KitexClient 客户端实现
type kitexClient struct {
    serviceName  string
    transporter  Transporter
    loadBalancer LoadBalancer
    connPool     ConnPool
    options      Options
}

// Call 同步调用实现
func (c *kitexClient) Call(ctx context.Context, req, resp interface{}) error {
    // 1. 选择实例（负载均衡）
    inst := c.loadBalancer.Pick(ctx, c.getInstances())

    // 2. 获取连接
    conn, err := c.connPool.Get(ctx, inst)
    if err != nil {
        return err
    }

    // 3. 编码请求
    encReq, err := c.options.Codec.Encode(req)
    if err != nil {
        c.connPool.Put(conn, err) // 归还连接
        return err
    }

    // 4. 发送请求（通过Netpoll高性能网络）
    err = c.transporter.Send(ctx, conn, encReq)
    if err != nil {
        c.connPool.Put(conn, err)
        return err
    }

    // 5. 接收响应
    encResp, err := c.transporter.Recv(ctx, conn)
    if err != nil {
        c.connPool.Put(conn, err)
        return err
    }

    // 6. 解码响应
    err = c.options.Codec.Decode(encResp, resp)
    if err != nil {
        c.connPool.Put(conn, err)
        return err
    }

    // 7. 归还连接
    c.connPool.Put(conn, nil)

    return nil
}
```

### 2.2 自研网络库Netpoll

Netpoll是字节跳动自研的高性能网络库，相比标准库net/http有显著性能提升。

```go
// Netpoll 核心设计

// NetworkEventLoop 网络事件循环
// 使用epoll实现高性能I/O多路复用
type NetworkEventLoop struct {
    // epoll文件描述符
    epfd int

    // 事件通道
    // 用于接收就绪的事件
    eventCh chan *PollEvent

    // 连接管理
    connections map[int]*Connection

    // 锁
    mu sync.RWMutex

    // 调度器
    scheduler *Scheduler
}

// Poller epoll封装
type Poller struct {
    // epoll fd
    epfd int

    // 事件列表
    events []epollEvent

    // 连接映射
    conns map[int]*Connection

    // 读/写处理器
    readHandler  func(*Connection)
    writeHandler func(*Connection)
}

// 创建新的Poller
func NewPoller(num int) (*Poller, error) {
    // 创建epoll实例
    epfd, err := unix.EpollCreate1(unix.EPOLL_CLOEXEC)
    if err != nil {
        return nil, err
    }

    return &Poller{
        epfd:    epfd,
        events:  make([]unix.EpollEvent, 1024),
        conns:   make(map[int]*Connection),
    }, nil
}

// 注册连接
func (p *Poller) AddConnection(fd int, conn *Connection) error {
    // 设置为非阻塞
    unix.SetNonBlocking(fd)

    // 添加到epoll
    ev := unix.EpollEvent{
        Events: unix.EPOLLIN | unix.EPOLLOUT | unix.EPOLLHUP,
        Fd:     int32(fd),
    }

    if err := unix.EpollCtl(p.epfd, unix.EPOLL_CTL_ADD, fd, &ev); err != nil {
        return err
    }

    p.conns[fd] = conn
    return nil
}

// 事件循环
func (p *Poller) Wait() {
    for {
        // 等待事件
        n, err := unix.EpollWait(p.epfd, p.events, -1)
        if err != nil {
            if err == unix.EINTR {
                continue
            }
            log.Printf("epoll wait error: %v", err)
            continue
        }

        // 处理就绪事件
        for i := 0; i < n; i++ {
            ev := &p.events[i]
            fd := int(ev.Fd)

            conn := p.conns[fd]
            if conn == nil {
                continue
            }

            // 处理事件
            if ev.Events&unix.EPOLLIN != 0 {
                p.handleRead(conn)
            }

            if ev.Events&unix.EPOLLOUT != 0 {
                p.handleWrite(conn)
            }

            if ev.Events&(unix.EPOLLHUP|unix.EPOLLERR) != 0 {
                p.handleClose(conn)
            }
        }
    }
}

// Connection 连接封装
type Connection struct {
    // 文件描述符
    fd int

    // 读写缓冲区
    readBuf  *netpoll.ByteBuffer
    writeBuf *netpoll.ByteBuffer

    // 状态
    state int32 // reading, writing, closed

    // 回调
    onRead  func(*Connection, []byte)
    onWrite func(*Connection)
    onClose func(*Connection)

    // 上下文
    ctx context.Context
}

// 处理读取
func (p *Poller) handleRead(conn *Connection) {
    // 从fd读取数据到缓冲区
    n, err := unix.Read(conn.fd, conn.readBuf.Free())

    if err != nil {
        if err == unix.EAGAIN {
            return // 资源暂时不可用
        }
        // 错误，关闭连接
        conn.onClose(conn)
        return
    }

    if n == 0 {
        // 对端关闭
        conn.onClose(conn)
        return
    }

    // 推进缓冲区位置
    conn.readBuf.AddWritten(n)

    // 触发读回调
    data := conn.readBuf.Bytes()
    conn.onRead(conn, data)

    // 重置缓冲区
    conn.readBuf.Reset()
}
```

```go
    if err != nil {
        return nil, err

    if err != nil {
        p.releaseConn(conn, err)
        return nil, err

    p.releaseConn(conn, nil)

type KitexServer struct {
    // 服务信息
    svcInfo *remote.ServiceInfo

    // 监听器
    listener net.Listener

    // 传输层

    // 编解码器

    // 中间件
    mw []remote.Middleware

    // 运行状态
    status int32
}

// Run 启动服务
func (s *KitexServer) Run() error {
    // 1. 初始化网络库
    listener, err := netpoll.NewListener("tcp", s.addr)
    if err != nil {
        return err
    }

    // 2. 创建事件循环
    eventLoop := &EventLoop{
        // 读取请求
        onRead: s.handleRequest,
        // 错误处理
        onError: s.handleError,
    }

    // 3. 启动网络事件循环
    go eventLoop.Run(listener)

    // 4. 等待信号
    sig := make(chan os.Signal, 1)
    signal.Notify(sig, syscall.SIGINT, syscall.SIGTERM)
    <-sig

    // 5. 优雅关闭
    return s.GracefulStop()
}

// handleRequest 处理请求
func (s *KitexServer) handleRequest(ctx context.Context, conn *netpoll.Conn) error {
    // 1. 读取请求数据
    reqData, err := conn.Read()
    if err != nil {
        return err
    }

    // 2. 解码请求
    req, err := s.codec.Decode(reqData)
    if err != nil {
        return err
    }

    // 3. 构建上下文
    ctx = remote.NewContext(ctx, conn)

    // 4. 执行中间件链
    handler := s.getHandler()
    for i := len(s.mw) - 1; i >= 0; i-- {
        handler = s.mw[i](handler)
    }

    // 5. 执行处理器
    resp, err := handler(ctx, req)
    if err != nil {
        // 错误处理
        resp = s.errorResp(err)
    }

    // 6. 编码响应
    respData, err := s.codec.Encode(resp)
    if err != nil {
        return err
    }

    // 7. 发送响应
    _, err = conn.Write(respData)

    return err
}
```

```go
// 字节跳动超时控制实践

// Timeout 中间件
// 用于RPC调用的超时控制
type TimeoutMiddleware struct {
    // 默认超时时间
    DefaultTimeout time.Duration

    // 每方法超时配置
    MethodTimeouts map[string]time.Duration
}

func (m *TimeoutMiddleware) Middleware(next remote.Handler) remote.Handler {
    return func(ctx context.Context, conn net.Conn, req, resp interface{}) error {
        // 1. 获取方法名
        method := remote.GetMethodName(ctx)

        // 2. 确定超时时间
        timeout := m.DefaultTimeout
        if mt, ok := m.MethodTimeouts[method]; ok {
            timeout = mt
        }

        // 3. 检查context是否已超时
        if d, ok := ctx.Deadline(); ok {
            // 使用较短的截止时间
            remaining := time.Until(d)
            if remaining < timeout {
                timeout = remaining
            }
        }

        // 4. 创建超时context
        ctx, cancel := context.WithTimeout(ctx, timeout)
        defer cancel()

        // 5. 使用channel实现超时等待
        errCh := make(chan error, 1)

        go func() {
            errCh <- next(ctx, conn, req, resp)
        }()

        // 6. 等待结果或超时
        select {
        case err := <-errCh:
            return err
        case <-ctx.Done():
            // 超时处理
            return ErrRPCTimeout
        }
    }
}

// 重试超时策略
type RetryTimeoutStrategy struct {
    // 最大重试次数
    MaxRetry int

    // 重试间隔
    RetryInterval time.Duration

    // 退避策略
    Backoff BackoffFunc
}

func (s *RetryTimeoutStrategy) DoWithRetry(ctx context.Context, fn func() error) error {
    var lastErr error

    for attempt := 0; attempt < s.MaxRetry; attempt++ {
        // 检查context是否取消
        if ctx.Err() != nil {
            return ctx.Err()
        }

        // 执行
        err := fn()
        if err == nil {
            return nil
        }

        lastErr = err

        // 检查是否应该重试
        if !s.ShouldRetry(err) {
            return err
        }

        // 计算退避时间
        delay := s.Backoff(attempt)

        // 等待
        select {
        case <-time.After(delay):
        case <-ctx.Done():
            return ctx.Err()
        }
    }

    return lastErr
}
```

```go
// 字节跳动限流熔断实践

// TokenBucketLimiter 令牌桶限流器
// Kitex内置的限流实现
type TokenBucketLimiter struct {
    // 桶容量
    capacity int64

    // 当前令牌数
    tokens int64

    // 填充速率（令牌/秒）
    rate int64

    // 最后填充时间
    lastFillTime int64

    // 锁
    mu sync.Mutex
}

func (l *TokenBucketLimiter) Allow() bool {
    l.mu.Lock()
    defer l.mu.Unlock()

    // 计算应该填充的令牌
    now := time.Now().UnixNano()
    elapsed := now - l.lastFillTime

    // 按速率填充
    refill := (elapsed * l.rate) / int64(time.Second)
    l.tokens += refill

    if l.tokens > l.capacity {
        l.tokens = l.capacity
    }

    l.lastFillTime = now

    // 消费令牌
    if l.tokens > 0 {
        l.tokens--
        return true
    }

    return false
}

// CircuitBreaker 熔断器
// 灵感来自Hystrix但针对Go进行了优化
type CircuitBreaker struct {
    // 熔断器名称
    name string

    // 状态
    state int32 // closed=0, open=1, half-open=2

    // 计数器
    success int64
    failure int64

    // 阈值
    failureThreshold int64 // 失败多少次后打开
    successThreshold int64 // 成功多少次后关闭

    // 超时时间
    timeout time.Duration

    // 锁
    mu sync.Mutex

    // 状态变更回调
    onStateChange func(name string, from, to CircuitState)
}

const (
    StateClosed CircuitState = iota
    StateOpen
    StateHalfOpen
)

func (cb *CircuitBreaker) Do(ctx context.Context, fn func() error) error {
    // 1. 检查状态
    state := cb.getState()

    switch state {
    case StateOpen:
        // 检查是否应该转为半开
        if cb.shouldTryHalfOpen() {
            cb.setState(StateHalfOpen)
        } else {
            return ErrCircuitOpen
        }

    case StateHalfOpen:
        // 半开状态，只允许一个请求通过
        if !cb.tryAcquire() {
            return ErrTooManyRequests
        }
    }

    // 2. 执行
    err := fn()

    // 3. 记录结果
    if err != nil {
        cb.onFailure()
    } else {
        cb.onSuccess()
    }

    return err
}

func (cb *CircuitBreaker) onFailure() {
    atomic.AddInt64(&cb.failure, 1)

    // 检查是否应该打开
    if atomic.LoadInt64(&cb.failure) >= cb.failureThreshold {
        cb.setState(StateOpen)
    }
}

func (cb *CircuitBreaker) onSuccess() {
    atomic.AddInt64(&cb.success, 1)

    // 检查是否应该关闭
    if atomic.LoadInt64(&cb.success) >= cb.successThreshold {
        cb.setState(StateClosed)
    }
}
```

```go
// 字节跳动链路追踪实践

// TraceInfo 追踪信息
type TraceInfo struct {
    // 追踪ID
    TraceID string

    // 跨度ID
    SpanID string

    // 父跨度ID
    ParentSpanID string

    // 调用层级
    Depth int
}

// Middleware 追踪中间件
func TracingMiddleware(next remote.Handler) remote.Handler {
    return func(ctx context.Context, conn net.Conn, req, resp interface{}) error {
        // 1. 从context获取或生成TraceInfo
        traceInfo := GetTraceInfo(ctx)
        if traceInfo == nil {
            traceInfo = &TraceInfo{
                TraceID:    generateTraceID(),
                SpanID:     generateSpanID(),
                ParentSpanID: "",
                Depth:      0,
            }
        } else {
            // 创建子span
            traceInfo = &TraceInfo{
                TraceID:      traceInfo.TraceID,
                SpanID:       generateSpanID(),
                ParentSpanID: traceInfo.SpanID,
                Depth:        traceInfo.Depth + 1,
            }
        }

        // 2. 将TraceInfo放入context
        ctx = WithTraceInfo(ctx, traceInfo)

        // 3. 记录开始时间
        start := time.Now()

        // 4. 添加trace头到请求
        if p, ok := req.(Packet); ok {
            p.SetHeader("trace-id", traceInfo.TraceID)
            p.SetHeader("span-id", traceInfo.SpanID)
            p.SetHeader("depth", strconv.Itoa(traceInfo.Depth))
        }

        // 5. 执行调用
        err := next(ctx, conn, req, resp)

        // 6. 记录追踪数据
        duration := time.Since(start)
        RecordSpan(&SpanData{
            TraceID:   traceInfo.TraceID,
            SpanID:    traceInfo.SpanID,
            ParentID:  traceInfo.ParentSpanID,
            Service:   getServiceName(ctx),
            Method:    getMethodName(ctx),
            StartTime: start,
            Duration:  duration,
            Status:    getStatus(err),
            Peer:      conn.RemoteAddr().String(),
        })

        return err
    }
}
```

```go
    ID string

    Host string

    // 端口

// EtcdResolver etcd实现
type EtcdResolver struct {
    // etcd客户端
    client *clientv3.Client

    // 缓存
    cache map[string][]*Instance

    // 缓存锁
    mu sync.RWMutex

    // 订阅回调
    watchers map[string][]func([]*Instance)
}

func (r *EtcdResolver) Resolve(ctx context.Context, serviceName string) []*Instance {
    // 先从缓存获取
    r.mu.RLock()
    instances, ok := r.cache[serviceName]
    r.mu.RUnlock()

    if ok {
        return instances
    }

    // 缓存未命中，从etcd获取
    instances, err := r.fetchFromEtcd(serviceName)
    if err != nil {
        // 返回缓存（可能是过期的）
        return instances
    }

    // 更新缓存
    r.mu.Lock()
    r.cache[serviceName] = instances
    r.mu.Unlock()

    // 触发订阅回调
    r.notifySubscribers(serviceName, instances)

    return instances
}

func (r *EtcdResolver) fetchFromEtcd(serviceName string) ([]*Instance, error) {
    // 构建key
    key := fmt.Sprintf("/bytedance/services/%s", serviceName)

    // 查询
    resp, err := r.client.Get(context.Background(), key, clientv3.WithPrefix())
    if err != nil {
        return nil, err
    }

    // 解析
    instances := make([]*Instance, 0, len(resp.Kvs))
    for _, kv := range resp.Kvs {
        var inst Instance
        if err := json.Unmarshal(kv.Value, &inst); err != nil {
            continue
        }
        instances = append(instances, &inst)
    }

    return instances, nil
}
```

```go
// 负载均衡器接口
type LoadBalancer interface {
    // 选择一个实例
    Pick(ctx context.Context, instances []*Instance) *Instance
}

// WeightedRoundRobin 加权轮询
type WeightedRoundRobin struct {
    // 当前索引
    current int

    // 有效权重（会动态调整）
    effectiveWeight map[string]int

    // 权重
    weight map[string]int

    // 锁
    mu sync.Mutex
}

func (lb *WeightedRoundRobin) Pick(ctx context.Context, instances []*Instance) *Instance {
    lb.mu.Lock()
    defer lb.mu.Unlock()

    if len(instances) == 0 {
        return nil
    }

    // 简单轮询
    inst := instances[lb.current%len(instances)]
    lb.current = (lb.current + 1) % len(instances)

    return inst
}

// LeastPending 最少连接
type LeastPending struct {
    // 连接计数
    pending map[string]int

    // 锁
    mu sync.Mutex
}

func (lb *LeastPending) Pick(ctx context.Context, instances []*Instance) *Instance {
    lb.mu.Lock()
    defer lb.mu.Unlock()

    var best *Instance
    var minPending = math.MaxInt

    for _, inst := range instances {
        count := lb.pending[inst.ID]
        if count < minPending {
            minPending = count
            best = inst
        }
    }

    if best != nil {
        lb.pending[best.ID]++
    }

    return best
}

// ConsistentHash 一致性哈希
type ConsistentHash struct {
    // 虚拟节点数
    virtualNodes int

    // 哈希环
    circle map[uint32]string

    // 已排序的哈希值
    keys []uint32

    // 锁
    mu sync.Mutex
}

func NewConsistentHash(virtualNodes int) *ConsistentHash {
    return &ConsistentHash{
        virtualNodes: virtualNodes,
        circle:       make(map[uint32]string),
        keys:         make([]uint32, 0),
    }
}

func (h *ConsistentHash) Add(instance *Instance) {
    h.mu.Lock()
    defer h.mu.Unlock()

    // 为每个真实节点添加虚拟节点
    for i := 0; i < h.virtualNodes; i++ {
        key := hash(fmt.Sprintf("%s:%d:%d", instance.ID, instance.Host, i))
        h.circle[key] = instance.ID
        h.keys = append(h.keys, key)
    }

    // 排序
    sort.Slice(h.keys, func(i, j int) bool {
        return h.keys[i] < h.keys[j]
    })
}

func (h *ConsistentHash) Pick(ctx context.Context, reqID string) *Instance {
    h.mu.Lock()
    defer h.mu.Unlock()

    if len(h.keys) == 0 {
        return nil
    }

    // 计算请求的哈希值
    key := hash(reqID)

    // 二分查找第一个大于等于key的位置
    idx := sort.Search(len(h.keys), func(i int) bool {
        return h.keys[i] >= key
    })

    // 如果超过最后一个，循环到第一个
    if idx == len(h.keys) {
        idx = 0
    }

    // 获取实例ID
    instanceID := h.circle[h.keys[idx]]

    // 返回实例（实际应用中需要查询实例详情）
    return &Instance{ID: instanceID}
}
```

```go
// Router 路由接口
type Router interface {
    // 根据请求选择目标实例
    Route(ctx context.Context, instances []*ServiceInstance, req *Request) *ServiceInstance
}

// VersionRouter 版本路由
type VersionRouter struct {
    // 默认版本
    DefaultVersion string
}

func (r *VersionRouter) Route(ctx context.Context, instances []*ServiceInstance, req *Request) *ServiceInstance {
    // 从请求获取目标版本
    targetVersion := r.getVersionFromRequest(req)

    // 过滤版本匹配的实例
    candidates := make([]*ServiceInstance, 0)
    for _, inst := range instances {
        if inst.Version == targetVersion {
            candidates = append(candidates, inst)
        }
    }

    // 如果没有匹配的，使用默认版本
    if len(candidates) == 0 && targetVersion != r.DefaultVersion {
        for _, inst := range instances {
            if inst.Version == r.DefaultVersion {
                candidates = append(candidates, inst)
            }
        }
    }

    // 随机选择一个
    if len(candidates) > 0 {
        return candidates[rand.Intn(len(candidates))]
    }

    return nil
}

func (r *VersionRouter) getVersionFromRequest(req *Request) string {
    // 从header获取版本
    if v := req.Header.Get("x-service-version"); v != "" {
        return v
    }

    // 从cookie获取
    if v := req.Cookie.Get("service_version"); v != "" {
        return v
    }

    // 从context获取
    if v := req.Context.Value("target_version"); v != "" {
        return v.(string)
    }

    return r.DefaultVersion
}

// CanaryRouter 金丝雀路由
type CanaryRouter struct {
    // 金丝雀流量比例（0-100）
    CanaryPercent int

    // 金丝雀版本
    CanaryVersion string
}

func (r *CanaryRouter) Route(ctx context.Context, instances []*ServiceInstance, req *Request) *ServiceInstance {
    // 稳定版本实例
    stable := make([]*ServiceInstance, 0)
    // 金丝雀版本实例
    canary := make([]*ServiceInstance, 0)

    for _, inst := range instances {
        if inst.Version == r.CanaryVersion {
            canary = append(canary, inst)
        } else {
            stable = append(stable, inst)
        }
    }

    // 根据比例决定是否路由到金丝雀
    if len(canary) > 0 && rand.Intn(100) < r.CanaryPercent {
        return canary[rand.Intn(len(canary))]
    }

    // 默认路由到稳定版本
    if len(stable) > 0 {
        return stable[rand.Intn(len(stable))]
    }

    return nil
}
```

```go
// internal/data/user.go

package data

import (
    "context"
    "database/sql"
    "fmt"

    "github.com/bilibili/kratos/v2/log"

    "myservice/internal/biz"
)

// UserRepo 用户仓储实现
type UserRepo struct {
    // 数据库连接
    db *sql.DB

    // 缓存
    cache *RedisCache

    // 日志
    logger *log.Helper
}

// NewUserRepo 创建仓储
func NewUserRepo(db *sql.DB, cache *RedisCache, logger log.Logger) biz.UserRepo {
    return &UserRepo{
        db:    db,
        cache: cache,
        logger: log.NewHelper(logger),
    }
}

func (r *UserRepo) FindByID(ctx context.Context, id int64) (*biz.User, error) {
    // 1. 先查缓存
    key := fmt.Sprintf("user:%d", id)
    cached, err := r.cache.Get(ctx, key)
    if err == nil && cached != nil {
        return cached.(*biz.User), nil
    }

    // 2. 缓存未命中，查数据库
    user := &biz.User{}
    query := `SELECT id, name, email, phone, created_at, updated_at FROM users WHERE id = ?`
    row := r.db.QueryRowContext(ctx, query, id)

    err = row.Scan(&user.ID, &user.Name, &user.Email, &user.Phone, &user.CreatedAt, &user.UpdatedAt)
    if err != nil {
        if err == sql.ErrNoRows {
            return nil, biz.ErrUserNotFound
        }
        r.logger.WithContext(ctx).Errorf("QueryRow failed: %v", err)
        return nil, err
    }

    // 3. 更新缓存
    r.cache.Set(ctx, key, user, time.Hour)

    return user, nil
}

func (r *UserRepo) FindByIDs(ctx context.Context, ids []int64) ([]*biz.User, error) {
    if len(ids) == 0 {
        return nil, nil
    }

    // 批量查询
    query := `SELECT id, name, email, phone, created_at, updated_at FROM users WHERE id IN (`
    args := make([]interface{}, len(ids))
    for i, id := range ids {
        if i > 0 {
            query += ","
        }
        query += "?"
        args[i] = id
    }
    query += ")"

    rows, err := r.db.QueryContext(ctx, query, args...)
    if err != nil {
        r.logger.WithContext(ctx).Errorf("Query failed: %v", err)
        return nil, err
    }
    defer rows.Close()

    // 遍历结果
    users := make([]*biz.User, 0)
    for rows.Next() {
        user := &biz.User{}
        err := rows.Scan(&user.ID, &user.Name, &user.Email, &user.Phone, &user.CreatedAt, &user.UpdatedAt)
        if err != nil {
            r.logger.WithContext(ctx).Errorf("Scan failed: %v", err)
            continue
        }
        users = append(users, user)
    }

    return users, nil
}

func (r *UserRepo) Create(ctx context.Context, user *biz.User) (*biz.User, error) {
    // 检查邮箱是否存在
    existing, err := r.FindByEmail(ctx, user.Email)
    if err == nil && existing != nil {
        return nil, biz.ErrUserEmailExisted
    }

    // 插入
    query := `INSERT INTO users (name, email, phone, created_at, updated_at) VALUES (?, ?, ?, ?, ?)`
    now := time.Now()

    result, err := r.db.ExecContext(ctx, query, user.Name, user.Email, user.Phone, now, now)
    if err != nil {
        r.logger.WithContext(ctx).Errorf("Insert failed: %v", err)
        return nil, err
    }

    id, err := result.LastInsertId()
    if err != nil {
        return nil, err
    }

    user.ID = id
    user.CreatedAt = now
    user.UpdatedAt = now

    return user, nil
}

func (r *UserRepo) Update(ctx context.Context, user *biz.User) error {
    query := `UPDATE users SET name = ?, email = ?, phone = ?, updated_at = ? WHERE id = ?`
    user.UpdatedAt = time.Now()

    _, err := r.db.ExecContext(ctx, query, user.Name, user.Email, user.Phone, user.UpdatedAt, user.ID)
    if err != nil {
        r.logger.WithContext(ctx).Errorf("Update failed: %v", err)
        return err
    }

    // 删除缓存
    key := fmt.Sprintf("user:%d", user.ID)
    r.cache.Del(ctx, key)

    return nil
}

func (r *UserRepo) Delete(ctx context.Context, id int64) error {
    query := `DELETE FROM users WHERE id = ?`
    _, err := r.db.ExecContext(ctx, query, id)
    if err != nil {
        r.logger.WithContext(ctx).Errorf("Delete failed: %v", err)
        return err
    }

    // 删除缓存
    key := fmt.Sprintf("user:%d", id)
    r.cache.Del(ctx, key)

    return nil
}

func (r *UserRepo) FindByEmail(ctx context.Context, email string) (*biz.User, error) {
    query := `SELECT id, name, email, phone, created_at, updated_at FROM users WHERE email = ?`
    row := r.db.QueryRowContext(ctx, query, email)

    err := row.Scan(&user.ID, &user.Name, &user.Email, &user.Phone, &user.CreatedAt, &user.UpdatedAt)
    if err != nil {
        if err == sql.ErrNoRows {
            return nil, biz.ErrUserNotFound
        }
        return nil, err
    }
```
