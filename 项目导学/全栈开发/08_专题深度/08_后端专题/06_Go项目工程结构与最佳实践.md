# Go项目工程结构与最佳实践

## 概述

Go语言以其简洁的语法和强大的工具链著称，但Go的生态系统在项目结构方面相对灵活，没有强制的约定。本文档介绍Go项目的标准布局、错误处理模式、日志规范、配置管理、依赖注入以及测试实践，帮助开发者构建高质量、可维护的Go项目。

### 1.2 核心目录说明

#### cmd/ - 应用入口

```go
// cmd/myapp/main.go

package main

import (
    "context"
    "flag"
    "fmt"
    "os"
    "os/signal"
    "syscall"

    "myproject/internal/biz"
    "myproject/internal/conf"
    "myproject/internal/data"
    "myproject/internal/server"
)

func main() {
    // 1. 解析命令行参数
    configPath := flag.String("config", "configs/config.yaml", "配置文件路径")
    flag.Parse()

    // 2. 加载配置
    cfg, err := conf.Load(*configPath)
    if err != nil {
        fmt.Fprintf(os.Stderr, "加载配置失败: %v\n", err)
        os.Exit(1)
    }

    // 3. 初始化依赖
    // 依赖注入：按照 data -> biz 的顺序初始化
    db, err := data.NewMySQL(cfg.Database)
    if err != nil {
        fmt.Fprintf(os.Stderr, "连接数据库失败: %v\n", err)
        os.Exit(1)
    }
    defer db.Close()

    // 创建仓储
    userRepo := data.NewUserRepo(db)
    // 创建业务用例
    userBiz := biz.NewUserUsecase(userRepo)

    // 4. 创建服务器
    httpServer := server.NewHTTPServer(cfg.Server, userBiz)
    grpcServer := server.NewGRPCServer(cfg.Server, userBiz)

    // 5. 启动服务器
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    go func() {
        if err := httpServer.Start(ctx); err != nil {
            fmt.Fprintf(os.Stderr, "HTTP服务启动失败: %v\n", err)
            cancel()
        }
    }()

    go func() {
        if err := grpcServer.Start(ctx); err != nil {
            fmt.Fprintf(os.Stderr, "gRPC服务启动失败: %v\n", err)
            cancel()
        }
    }()

    // 6. 等待信号
    sig := make(chan os.Signal, 1)
    signal.Notify(sig, syscall.SIGINT, syscall.SIGTERM)
    <-sig

    fmt.Println("正在关闭...")

    // 7. 优雅关闭
    shutdownCtx, shutdownCancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer shutdownCancel()

    if err := httpServer.Stop(shutdownCtx); err != nil {
        fmt.Fprintf(os.Stderr, "HTTP服务关闭失败: %v\n", err)
    }

    if err := grpcServer.Stop(shutdownCtx); err != nil {
        fmt.Fprintf(os.Stderr, "gRPC服务关闭失败: %v\n", err)
    }

    fmt.Println("服务已关闭")
}
```

```go
        return nil
    }
    return fmt.Errorf("%s: %w", message, err)
}

// WithMessagef 格式化包装错误信息
func WithMessagef(err error, format string, args ...interface{}) error {
    if err == nil {
        return nil
    }
    return fmt.Errorf("%s: %w", fmt.Sprintf(format, args...), err)
}

func Is(err, target error) bool {
    return errors.Is(err, target)
}

func As(err error, target interface{}) bool {
    return errors.As(err, target)
}
```

### 1.3 领域驱动分层

```go
// internal/biz/user.go

// 领域层 - 核心业务逻辑
// 纯业务逻辑，不依赖任何外部框架

package biz

import (
    "context"
    "errors"
    "time"
)

// 领域错误
var (
    ErrUserNotFound   = errors.New("用户不存在")
    ErrInvalidEmail   = errors.New("无效的邮箱格式")
    ErrEmailExists    = errors.New("邮箱已被使用")
)

// User 领域模型
type User struct {
    ID        int64
    Name      string
    Email     Email
    Phone     Phone
    Status    UserStatus
    CreatedAt time.Time
    UpdatedAt time.Time
}

// Email 值对象
type Email struct {
    value string
}

func NewEmail(v string) (*Email, error) {
    if !isValidEmail(v) {
        return nil, ErrInvalidEmail
    }
    return &Email{value: v}, nil
}

func (e Email) String() string {
    return e.value
}

// UserStatus 用户状态
type UserStatus int

const (
    UserStatusActive   UserStatus = 1
    UserStatusInactive UserStatus = 2
    UserStatusBanned   UserStatus = 3
)

// UserRepository 接口 - 仓储模式
// 定义领域层对数据的操作契约
type UserRepository interface {
    FindByID(ctx context.Context, id int64) (*User, error)
    FindByEmail(ctx context.Context, email Email) (*User, error)
    Save(ctx context.Context, user *User) error
    Delete(ctx context.Context, id int64) error
    List(ctx context.Context, offset, limit int) ([]*User, error)
}

// UserUsecase 用户业务用例
type UserUsecase struct {
    repo UserRepository // 依赖接口而非实现
}

// NewUserUsecase 构造函数注入
func NewUserUsecase(repo UserRepository) *UserUsecase {
    return &UserUsecase{repo: repo}
}

// CreateUser 创建用户
func (uc *UserUsecase) CreateUser(ctx context.Context, name, email, phone string) (*User, error) {
    // 1. 创建邮箱值对象
    e, err := NewEmail(email)
    if err != nil {
        return nil, err
    }

    // 2. 检查邮箱是否已存在
    existing, err := uc.repo.FindByEmail(ctx, *e)
    if err != nil && !errors.Is(err, ErrUserNotFound) {
        return nil, err
    }
    if existing != nil {
        return nil, ErrEmailExists
    }

    // 3. 创建用户
    user := &User{
        Name:      name,
        Email:      *e,
        Phone:      Phone{value: phone},
        Status:     UserStatusActive,
        CreatedAt:  time.Now(),
        UpdatedAt:  time.Now(),
    }

    // 4. 保存
    if err := uc.repo.Save(ctx, user); err != nil {
        return nil, err
    }

    return user, nil
}

// GetUser 获取用户
func (uc *UserUsecase) GetUser(ctx context.Context, id int64) (*User, error) {
    user, err := uc.repo.FindByID(ctx, id)
    if err != nil {
        return nil, err
    }
    return user, nil
}
```

```go
// internal/data/user.go

// 数据层 - 实现仓储接口
package data

import (
    "context"
    "database/sql"
    "fmt"

    "myproject/internal/biz"
)

// UserRepository 实现biz.UserRepository
type UserRepository struct {
    db *sql.DB
}

// NewUserRepository 创建仓储
func NewUserRepository(db *sql.DB) biz.UserRepository {
    return &UserRepository{db: db}
}

// FindByID 实现接口
func (r *UserRepository) FindByID(ctx context.Context, id int64) (*biz.User, error) {
    query := `SELECT id, name, email, phone, status, created_at, updated_at
              FROM users WHERE id = ?`

    var user biz.User
    err := r.db.QueryRowContext(ctx, query, id).Scan(
        &user.ID,
        &user.Name,
        &user.Email.value,
        &user.Phone.value,
        &user.Status,
        &user.CreatedAt,
        &user.UpdatedAt,
    )

    if err == sql.ErrNoRows {
        return nil, biz.ErrUserNotFound
    }
    if err != nil {
        return nil, err
    }

    return &user, nil
}

// FindByEmail 实现接口
func (r *UserRepository) FindByEmail(ctx context.Context, email biz.Email) (*biz.User, error) {
    query := `SELECT id, name, email, phone, status, created_at, updated_at
              FROM users WHERE email = ?`

    var user biz.User
    err := r.db.QueryRowContext(ctx, query, email.String()).Scan(
        &user.ID,
        &user.Name,
        &user.Email.value,
        &user.Phone.value,
        &user.Status,
        &user.CreatedAt,
        &user.UpdatedAt,
    )

    if err == sql.ErrNoRows {
        return nil, biz.ErrUserNotFound
    }
    if err != nil {
        return nil, err
    }

    return &user, nil
}

// Save 保存用户
func (r *UserRepository) Save(ctx context.Context, user *biz.User) error {
    query := `INSERT INTO users (name, email, phone, status, created_at, updated_at)
              VALUES (?, ?, ?, ?, ?, ?)`

    result, err := r.db.ExecContext(ctx, query,
        user.Name,
        user.Email.String(),
        user.Phone.value,
        user.Status,
        user.CreatedAt,
        user.UpdatedAt,
    )
    if err != nil {
        return err
    }

    id, err := result.LastInsertId()
    if err != nil {
        return err
    }

    user.ID = id
    return nil
}

// Delete 删除用户
func (r *UserRepository) Delete(ctx context.Context, id int64) error {
    query := `DELETE FROM users WHERE id = ?`
    _, err := r.db.ExecContext(ctx, query, id)
    return err
}

// List 列表查询
func (r *UserRepository) List(ctx context.Context, offset, limit int) ([]*biz.User, error) {
    query := `SELECT id, name, email, phone, status, created_at, updated_at
              FROM users ORDER BY id LIMIT ? OFFSET ?`

    rows, err := r.db.QueryContext(ctx, query, limit, offset)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    var users []*biz.User
    for rows.Next() {
        var user biz.User
        if err := rows.Scan(
            &user.ID,
            &user.Name,
            &user.Email.value,
            &user.Phone.value,
            &user.Status,
            &user.CreatedAt,
            &user.UpdatedAt,
        ); err != nil {
            return nil, err
        }
        users = append(users, &user)
    }

    return users, nil
}
```

## 二、错误处理规范

### 2.1 Go错误处理哲学

Go的错误处理强调显式处理而非异常机制，这要求开发者认真对待每一个可能的错误。

```
┌─────────────────────────────────────────────────────────────────┐
│                    Go 错误处理原则                               │
├─────────────────────────────────────────────────────────────────┤
│  1. 错误是值 - 错误只是普通的返回值                              │
│  2. 显式处理 - 不允许忽略错误（除非显式声明）                    │
│  3. 快速失败 - 发现错误立即处理，不要延迟                        │
│  4. 包装信息 - 提供足够的上下文信息                              │
│  5. 分类处理 - 根据错误类型采取不同措施                          │
└─────────────────────────────────────────────────────────────────┘
```

### 2.2 错误类型与模式

```go
// errors/error_types.go

package main

import (
    "errors"
    "fmt"
)

// 1. Sentinel Errors（哨兵错误）
// 预定义的错误值，用于公开的API
var (
    ErrNotFound          = errors.New("resource not found")
    ErrPermissionDenied  = errors.New("permission denied")
    ErrAlreadyExists     = errors.New("resource already exists")
    ErrInvalidInput      = errors.New("invalid input")
)

// 2. Error Wrapping（错误包装）
// 使用fmt.Errorf或%w提供更多上下文
func readFile(path string) error {
    data, err := os.ReadFile(path)
    if err != nil {
        // 使用%w包装原始错误，保留错误链
        return fmt.Errorf("读取文件失败 %s: %w", path, err)
    }
    return nil
}

// 3. 自定义错误类型
// 用于携带更多结构化信息
type ValidationError struct {
    Field   string
    Message string
}

func (e *ValidationError) Error() string {
    return fmt.Sprintf("%s: %s", e.Field, e.Message)
}

// 4. 错误接口
// 实现error接口的自定义类型
type MyError struct {
    Code    int
    Message string
    Cause   error
}

func (e *MyError) Error() string {
    return fmt.Sprintf("error %d: %s (caused by: %v)", e.Code, e.Message, e.Cause)
}

func (e *MyError) Unwrap() error {
    return e.Cause
}

// 5. 错误断言与判断
func handleError(err error) {
    // 使用errors.Is检查错误链
    if errors.Is(err, ErrNotFound) {
        // 处理资源未找到
        fmt.Println("资源未找到")
        return
    }

    // 使用errors.As获取底层类型
    var valErr *ValidationError
    if errors.As(err, &valErr) {
        fmt.Printf("验证错误: %s - %s\n", valErr.Field, valErr.Message)
        return
    }

    // 检查原始错误
    if errors.Is(err, os.ErrNotExist) {
        fmt.Println("文件不存在")
        return
    }

    // 未知错误
    fmt.Printf("未知错误: %v\n", err)
}
```

### 2.3 业务错误处理

```go
// internal/biz/errors/errors.go

// 业务错误定义
package bizerrors

// 业务错误码
type BizCode int

const (
    // 客户端错误 1000-1999
    BizCodeInvalidParams BizCode = 1001
    BizCodeNotFound     BizCode = 1002
    BizCodeUnauthorized BizCode = 1003
    BizCodeForbidden    BizCode = 1004
    BizCodeConflict     BizCode = 1005

    // 服务端错误 5000-5999
    BizCodeInternal     BizCode = 5001
    BizCodeDBError      BizCode = 5002
    BizCodeTimeout      BizCode = 5003
)

// BizError 业务错误
type BizError struct {
    Code    BizCode   `json:"code"`
    Message string    `json:"message"`
    Cause   error     `json:"-"`
    Details map[string]interface{} `json:"details,omitempty"`
}

func (e *BizError) Error() string {
    if e.Cause != nil {
        return fmt.Sprintf("%s: %v", e.Message, e.Cause)
    }
    return e.Message
}

func (e *BizError) Unwrap() error {
    return e.Cause
}

// 错误构造方法
func NewBizError(code BizCode, message string) *BizError {
    return &BizError{
        Code:    code,
        Message: message,
    }
}

func NewBizErrorWithCause(code BizCode, message string, cause error) *BizError {
    return &BizError{
        Code:    code,
        Message: message,
        Cause:   cause,
    }
}

func NewBizErrorWithDetails(code BizCode, message string, details map[string]interface{}) *BizError {
    return &BizError{
        Code:    code,
        Message: message,
        Details: details,
    }
}

// 便捷构造函数
func InvalidParams(field, reason string) *BizError {
    return NewBizErrorWithDetails(BizCodeInvalidParams, "参数错误",
        map[string]interface{}{"field": field, "reason": reason})
}

func NotFound(resource string) *BizError {
    return NewBizError(BizCodeNotFound, fmt.Sprintf("%s不存在", resource))
}

func Internal(err error) *BizError {
    return NewBizErrorWithCause(BizCodeInternal, "内部错误", err)
}

// Is 方法用于errors.Is
func (e *BizError) Is(target error) bool {
    if t, ok := target.(*BizError); ok {
        return e.Code == t.Code
    }
    return false
}

// As 方法用于errors.As
func (e *BizError) As(target interface{}) bool {
    if t, ok := target.(*BizError); ok {
        *t = *e
        return true
    }
    return false
}
```

## 三、日志规范

### 3.1 日志级别与使用场景

```
┌─────────────────────────────────────────────────────────────────┐
│                    日志级别规范                                  │
├─────────────────────────────────────────────────────────────────┤
│  DEBUG   - 调试信息，开发环境使用                               │
│           例如：入参、出参、变量值                              │
│                                                                 │
│  INFO    - 一般信息，正常运行时的重要事件                       │
│           例如：服务启动、请求处理、成功操作                    │
│                                                                 │
│  WARN    - 警告信息，潜在问题但不影响功能                       │
│           例如：资源使用率较高、重试操作、配置缺失              │
│                                                                 │
│  ERROR   - 错误信息，功能受损但可降级                           │
│           例如：请求失败、数据库错误、超时                       │
│                                                                 │
│  FATAL   - 致命错误，导致服务不可用                             │
│           例如：启动失败、无法恢复的错误                        │
│           注意：FATAL日志后程序通常会退出                       │
└─────────────────────────────────────────────────────────────────┘
```

### 3.2 日志格式规范

```go
// pkg/logger/logger.go

package logger

import (
    "fmt"
    "os"
    "time"
)

// LogLevel 日志级别
type LogLevel int

const (
    DEBUG LogLevel = iota
    INFO
    WARN
    ERROR
    FATAL
)

func (l LogLevel) String() string {
    switch l {
    case DEBUG: return "DEBUG"
    case INFO:  return "INFO"
    case WARN:  return "WARN"
    case ERROR: return "ERROR"
    case FATAL: return "FATAL"
    default:    return "UNKNOWN"
    }
}

// LogEntry 日志条目
type LogEntry struct {
    Time      time.Time
    Level     LogLevel
    Message   string
    Fields    map[string]interface{}
    Caller    string
    TraceID   string
    SpanID    string
}

// Logger 接口
type Logger interface {
    Debug(msg string, fields ...map[string]interface{})
    Info(msg string, fields ...map[string]interface{})
    Warn(msg string, fields ...map[string]interface{})
    Error(msg string, fields ...map[string]interface{})
    Fatal(msg string, fields ...map[string]interface{})
}

// Formatter 格式化器接口
type Formatter interface {
    Format(entry *LogEntry) []byte
}

// JSONFormatter JSON格式
type JSONFormatter struct{}

func (f *JSONFormatter) Format(entry *LogEntry) []byte {
    // 简化实现
    return []byte(fmt.Sprintf(
        `{"time":"%s","level":"%s","message":"%s","trace_id":"%s","span_id":"%s","caller":"%s"}`,
        entry.Time.Format(time.RFC3339),
        entry.Level.String(),
        entry.Message,
        entry.TraceID,
        entry.SpanID,
        entry.Caller,
    ))
}
```

```go
        cfg.Server.HTTP.Port = p
    }
```

```go
        code = "USER_NOT_FOUND"

        status = http.StatusInternalServerError
        code = "INTERNAL_ERROR"
    }

        Code:    code,
        Message: err.Error(),
    })
```
