```typescript
// 方案一：哈希加碰撞检测
function generateShortCode(url: string): string {
  const hash = crypto.createHash('md5').update(url).digest('hex');
  return hash.substring(0, 8); // 8位短码
}

// 方案二：自增ID加进制转换（推荐）
class ShortCodeGenerator {
  private counter = 0;
  private base62Chars = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';

  // 将自增ID转换为62进制字符串
  generate(id: number): string {
    let result = '';
    while (id > 0) {
      result = this.base62Chars[id % 62] + result;
      id = Math.floor(id / 62);
    }
    return result.padStart(6, '0');
  }

  // 62进制字符串转回ID
  decode(code: string): number {
    let result = 0;
    for (let i = 0; i < code.length; i++) {
      result = result * 62 + this.base62Chars.indexOf(code[i]);
    }
    return result;
  }
}

// 方案三：分布式ID生成器（雪花算法）
class SnowflakeIDGenerator {
  private epoch = 1609459200000n; // 2021-01-01
  private workerId: bigint;
  private sequence = 0n;

  constructor(workerId: number) {
    this.workerId = Bigint(workerId);
  }

  nextId(): bigint {
    const timestamp = Bigint(Date.now()) - this.epoch;
    return (timestamp << 22n) |
           (this.workerId << 12n) |
           (this.sequence++);
  }
}
```

#### 3. Redis库存扣减实现

```typescript
// 使用Redis Lua脚本保证原子性
const DECREASE_STOCK_SCRIPT = `
local stock = redis.call('GET', KEYS[1])
if not stock then
  return -1  -- 库存不存在
end
stock = tonumber(stock)
if stock <= 0 then
  return 0   -- 库存不足
end
redis.call('DECR', KEYS[1])
return stock - 1
`;

class SeckillService {
  private redis: Redis;

  async decreaseStock(ticketId: string, userId: string): Promise<boolean> {
    // 1. 用户限购检查（使用Redis Set）
    const userKey = `seckill:user:${ticketId}`;
    const isFirst = await this.redis.sadd(userKey, userId);
    if (!isFirst) {
      throw new Error('您已参与过本次抢购');
    }
    await this.redis.expire(userKey, 86400); // 24小时过期

    // 2. 库存扣减（原子操作）
    const stockKey = `seckill:stock:${ticketId}`;
    const result = await this.redis.eval(
      DECREASE_STOCK_SCRIPT,
      1,
      stockKey
    );

    if (result === 0) {
      throw new Error('库存不足');
    }

    if (result === -1) {
      throw new Error('活动不存在');
    }

    // 3. 发送订单消息（异步）
    await this.orderQueue.send({
      type: 'CREATE_ORDER',
      ticketId,
      userId,
      timestamp: Date.now()
    });

    return true;
  }
}
```

```typescript
// 多层限流防护
class RateLimiter {
  // 1. IP级别限流
  async checkIPLimit(ip: string): Promise<boolean> {
    const key = `rate:ip:${ip}`;
    const count = await this.redis.incr(key);

    if (count === 1) {
      await this.redis.expire(key, 1); // 1秒窗口
    }

    return count <= 100; // 每秒最多100请求
  }

  // 2. 用户级别限流
  async checkUserLimit(userId: string): Promise<boolean> {
    const key = `rate:user:${userId}`;
    const count = await this.redis.incr(key);

    if (count === 1) {
      await this.redis.expire(key, 60); // 1分钟窗口
    }

    return count <= 10; // 每分钟最多10次抢购
  }

  // 3. 验证码校验
  async verifyCaptcha(userId: string, captcha: string): Promise<boolean> {
    const key = `captcha:${userId}`;
    const valid = await this.redis.get(key);

    if (valid !== captcha) {
      return false;
    }

    await this.redis.del(key); // 一次性使用
    return true;
  }
}
```

#### 4. WebSocket连接管理

```typescript
class ConnectionManager {
  // 用户ID 到 WebSocket连接映射
  private connections: Map<string, Set<WebSocket>> = new Map();

  // 设备ID 到 用户ID映射
  private deviceToUser: Map<string, string> = new Map();

  // 建立连接
  handleConnection(ws: WebSocket, userId: string, deviceId: string) {
    // 1. 保存连接
    if (!this.connections.has(userId)) {
      this.connections.set(userId, new Set());
    }
    this.connections.get(userId)!.add(ws);
    this.deviceToUser.set(deviceId, userId);

    // 2. 更新在线状态
    this.updateUserStatus(userId, 'online');

    // 3. 广播用户上线
    this.broadcastUserStatus(userId, 'online');

    // 4. 发送离线消息
    this.sendOfflineMessages(userId);
  }

  // 发送消息给指定用户
  async sendToUser(userId: string, message: Message) {
    const connections = this.connections.get(userId);

    if (connections && connections.size > 0) {
      // 在线用户：直接发送
      connections.forEach(ws => {
        if (ws.readyState === WebSocket.OPEN) {
          ws.send(JSON.stringify(message));
        }
      });
    } else {
      // 离线用户：存储离线消息
      await this.storeOfflineMessage(userId, message);
    }
  }

  // 群发消息
  async broadcastToConversation(conversationId: string, message: Message, excludeUserId?: string) {
    const members = await this.getConversationMembers(conversationId);

    await Promise.all(
      members
        .filter(memberId => memberId !== excludeUserId)
        .map(memberId => this.sendToUser(memberId, message))
    );
  }

  // 离线消息存储
  async storeOfflineMessage(userId: string, message: Message) {
    const key = `offline:msg:${userId}`;
    await this.redis.lpush(key, JSON.stringify(message));
    // 限制离线消息数量
    await this.redis.ltrim(key, 0, 99);
  }

  // 获取离线消息
  async getOfflineMessages(userId: string): Promise<Message[]> {
    const key = `offline:msg:${userId}`;
    const messages = await this.redis.lrange(key, 0, -1);
    await this.redis.del(key); // 获取后删除
    return messages.map(m => JSON.parse(m));
  }
}
```

```typescript
class MessageSync {

  async syncMessages(userId: string, since: number): Promise<Message[]> {

      .where('receiverId', userId)
```

```typescript
class PresenceService {
  // 心跳检测
  async handleHeartbeat(userId: string, deviceId: string) {
    const key = `presence:${userId}`;
    await this.redis.hset(key, deviceId, Date.now().toString());
    await this.redis.expire(key, 300); // 5分钟过期
  }

  // 批量获取在线状态
  async getOnlineStatus(userIds: string[]): Promise<Map<string, boolean>> {
    const pipeline = this.redis.pipeline();

    userIds.forEach(userId => {
      pipeline.exists(`presence:${userId}`);
    });

    const results = await pipeline.exec();
    const statusMap = new Map<string, boolean>();

    userIds.forEach((userId, index) => {
      statusMap.set(userId, results[index][1] === 1);
    });

    return statusMap;
  }

  // 订阅用户状态变化
  subscribeToPresence(userId: string, callback: (status: boolean) => void) {
    const subscriber = this.redis.duplicate();

    subscriber.subscribe(`presence:${userId}`, (err) => {
      if (err) {
        console.error('订阅失败:', err);
        return;
      }

      subscriber.on('message', (channel, message) => {
        callback(message === 'online');
      });
    });

    return () => {
      subscriber.unsubscribe(`presence:${userId}`);
      subscriber.quit();
    };
  }
}
```

### 2.2 如何设计一个高效的数据库索引？

**题目描述**：如何设计数据库索引来优化查询性能？有哪些注意事项？

**思路引导**：
1. 先理解索引的底层原理：B+树、哈希索引
2. 考虑查询场景：等值查询、范围查询、模糊查询
3. 分析索引的代价：存储空间、写入性能

**完整解答**：

#### 1. 索引底层原理

```sql
-- MySQL InnoDB 使用B+树作为索引结构
-- B+树特点：
-- 1. 所有数据都在叶子节点
-- 2. 叶子节点之间用双向链表连接
-- 3. 查询复杂度O(log n)，范围查询高效

-- 示例：给users表的email字段建索引
CREATE INDEX idx_email ON users(email);

-- B+树结构示意：
--                    [root节点]
--                   /    |    \
--            [内节点]  [内节点]  [内节点]
--              |        |        |
--         [叶节点]  [叶节点]  [叶节点]
--         email    email    email
--         ->id     ->id     ->id
--         数据页    数据页    数据页
```

#### 2. 索引设计原则

```sql
-- 原则一：选择区分度高的列建索引
-- 错误：性别区分度低（只有男女）
CREATE INDEX idx_gender ON users(gender);  -- 区分度 ≈ 2

-- 正确：邮箱区分度高
CREATE INDEX idx_email ON users(email);    -- 区分度 ≈ 用户数

-- 原则二：覆盖索引减少回表
-- 低效：需要回表查询
CREATE INDEX idx_user_id ON orders(user_id);  -- 只索引了user_id

-- 高效：覆盖索引，索引包含所有查询字段
CREATE INDEX idx_user_id_cover ON orders(user_id, status, created_at);
-- 查询：SELECT status, created_at FROM orders WHERE user_id = 1;
--       无需回表，直接从索引中获取数据

-- 原则三：遵循最左前缀原则（复合索引）
CREATE INDEX idx_name_age_city ON users(name, age, city);

-- 生效的查询
SELECT * FROM users WHERE name = '张三';           -- 使用name
SELECT * FROM users WHERE name = '张三' AND age = 25;  -- 使用name, age
SELECT * FROM users WHERE name = '张三' AND age = 25 AND city = '北京'; -- 全部使用

-- 部分生效
SELECT * FROM users WHERE age = 25;                -- 不使用（跳过name）
SELECT * FROM users WHERE name = '张三' AND city = '北京'; -- 只使用name

-- 原则四：范围查询放最后
CREATE INDEX idx_status_created ON orders(status, created_at);

-- 正确：先等值，再范围
SELECT * FROM orders WHERE user_id = 1 AND status = 'paid' AND created_at > '2024-01-01';

-- 错误：范围查询导致后续索引失效
SELECT * FROM orders WHERE created_at > '2024-01-01' AND status = 'paid';
-- created_at是范围查询，status索引失效
```

#### 3. 实战索引优化案例

```sql
-- 案例一：用户登录查询优化
-- 原始查询（无索引）
SELECT * FROM users WHERE email = 'test@example.com' AND password_hash = 'xxx';

-- 优化：为高频等值查询字段建索引
CREATE INDEX idx_email ON users(email);
CREATE INDEX idx_phone ON users(phone);  -- 手机号登录也常用

-- 案例二：订单列表查询优化
-- 原始查询
SELECT id, user_id, status, total_amount, created_at
FROM orders
WHERE user_id = 1
  AND status IN ('paid', 'shipped', 'delivered')
  AND created_at BETWEEN '2024-01-01' AND '2024-12-31'
ORDER BY created_at DESC
LIMIT 20 OFFSET 100;

-- 优化索引设计
CREATE INDEX idx_user_status_created ON orders(user_id, status, created_at DESC);
-- 覆盖索引包含所有查询字段，避免回表

-- 案例三：商品分类查询优化
-- 原始查询
SELECT p.*, c.name as category_name
FROM products p
JOIN categories c ON p.category_id = c.id
WHERE p.category_id = 10 AND p.status = 'active'
ORDER BY p.sales_count DESC
LIMIT 20;

-- 优化：分解为两个索引
CREATE INDEX idx_category_status ON products(category_id, status, sales_count DESC);
CREATE INDEX idx_category_id ON categories(id);
```

#### 4. 索引分析工具

```sql

-- 结果解读：

-- key: 实际使用的索引

  AND table_name = 'orders';
```

#### 5. 索引注意事项

```sql
-- 1. 索引不是越多越好
-- 索引会占用磁盘空间
-- 每次INSERT/UPDATE/DELETE都需要维护索引
-- 写入性能下降：INSERT变慢（需要更新索引）

-- 2. 避免在索引列上使用函数
-- 错误：导致索引失效
SELECT * FROM orders WHERE YEAR(created_at) = 2024;
SELECT * FROM users WHERE SUBSTRING(name, 1, 1) = '张';

-- 正确：使用范围查询或表达式索引
SELECT * FROM orders WHERE created_at >= '2024-01-01' AND created_at < '2025-01-01';

-- 3. 前导模糊查询无法使用索引
-- 错误：'%张%'无法使用索引
SELECT * FROM users WHERE name LIKE '%张三%';

-- 正确：使用后置模糊或全文索引
SELECT * FROM users WHERE name LIKE '张三%';  -- 可以使用索引
-- 或使用全文索引
ALTER TABLE users ADD FULLTEXT INDEX ft_name (name);
SELECT * FROM users WHERE MATCH(name) AGAINST('张三');

-- 4. 字符串索引长度选择
-- 原则：区分度足够即可，不要过长
-- 验证不同长度的区分度
SELECT
  LEFT(email, 10) as prefix,
  COUNT(DISTINCT LEFT(email, 10)) / COUNT(*) as selectivity
FROM users;
-- 选择 selectivity 接近 1 的最短长度
```

#### 6. PostgreSQL vs MySQL 索引差异

| 特性 | PostgreSQL | MySQL (InnoDB) |
|------|------------|----------------|
| **索引类型** | B-tree, Hash, GiST, GIN, BRIN | B-tree, Hash |
| **表达式索引** | 支持 | 不支持 |
| **部分索引** | 支持 | 支持 |
| **覆盖索引** | 支持 | 支持 |
| **NULL索引** | 索引中包含NULL | 只索引NOT NULL列 |
| **并发写入** | MVCC，写入性能好 | 行级锁，高并发有竞争 |

```sql
-- PostgreSQL 特有功能
-- 1. 表达式索引
CREATE INDEX idx_lower_email ON users(LOWER(email));

-- 2. 部分索引（只索引需要的数据）
CREATE INDEX idx_active_users ON users(email)
WHERE status = 'active';

-- 3. 组合索引顺序不严格（可以任意顺序）
-- PostgreSQL 可以利用索引的任何列组合
```

- B+树索引原理

- 索引失效场景

### 2.3 如何保证数据库事务隔离？

**题目描述**：数据库事务的隔离级别有哪些？分别能解决哪些问题？脏读、不可重复读、幻读是如何产生的？

**思路引导**：
1. 先理解ACID中隔离性的含义
2. 理解四种隔离级别：读未提交、读已提交、可重复读、串行化
3. 理解脏读、不可重复读、幻读的概念和产生原因

**完整解答**：

#### 1. 事务隔离级别概览

| 隔离级别 | 脏读 | 不可重复读 | 幻读 | 并发度 |
|----------|------|------------|------|--------|
| **读未提交 (Read Uncommitted)** | 可能 | 可能 | 可能 | 最高 |
| **读已提交 (Read Committed)** | 不可能 | 可能 | 可能 | 中 |
| **可重复读 (Repeatable Read)** | 不可能 | 不可能 | MySQL不可能 | 较低 |
| **串行化 (Serializable)** | 不可能 | 不可能 | 不可能 | 最低 |

#### 2. 并发问题详解

```sql
-- 脏读：读取到其他事务未提交的数据
-- 时间线：
-- T1: 事务A - UPDATE users SET balance = balance - 100 WHERE id = 1
-- T2: 事务B - SELECT balance FROM users WHERE id = 1  -- 读到900（脏数据）
-- T3: 事务A - ROLLBACK  -- 回滚操作
-- 结果：事务B读到了不存在的数据（脏读）

-- 不可重复读：同一事务中两次读取同一行数据，结果不同
-- 时间线：
-- T1: 事务A - SELECT balance FROM users WHERE id = 1  -- 读到1000
-- T2: 事务B - UPDATE users SET balance = 900 WHERE id = 1 -- 提交
-- T3: 事务A - SELECT balance FROM users WHERE id = 1  -- 读到900
-- 结果：同一事务中重复读取，结果不一致

-- 幻读：同一事务中两次查询，结果集不同（如多了或少了行）
-- 时间线：
-- T1: 事务A - SELECT COUNT(*) FROM orders WHERE user_id = 1  -- 查到10条
-- T2: 事务B - INSERT INTO orders (user_id, ...) VALUES (1, ...) -- 提交
-- T3: 事务A - SELECT COUNT(*) FROM orders WHERE user_id = 1  -- 查到11条
-- 结果：同一事务中重复查询，结果集数量不同（幻读）
```

#### 3. MySQL InnoDB 隔离级别实现

```sql
-- 查看当前隔离级别
SHOW VARIABLES LIKE 'transaction_isolation';

-- 设置隔离级别（会话级/全局）
SET SESSION TRANSACTION ISOLATION LEVEL READ COMMITTED;
SET GLOBAL TRANSACTION ISOLATION LEVEL REPEATABLE READ;

-- MySQL REPEATABLE READ 实现原理

-- 1. MVCC（多版本并发控制）
-- 每行数据都有两个隐藏列：创建版本号、删除版本号
-- READ COMMITTED：每次读取都生成新快照
-- REPEATABLE READ：第一次读取时生成快照，之后复用

-- 事务A的READ视图
{
  "transaction_id": 100,
  "active_txn_ids": [98, 99],  // 当前活跃事务ID列表
  "min_active_txn_id": 98,
  "max_txn_id": 101,
  "snapshots": {
    // 每行数据的可见版本
    "users:1": { "balance": 1000, "version": 95 }
  }
}

-- 2. Next-Key Lock（临键锁）防止幻读
-- 锁定索引区间（左闭右开）
-- 例如：SELECT * FROM orders WHERE id BETWEEN 10 AND 20 FOR UPDATE
-- 会锁定 [10, 20) 这个区间

-- 3. Gap Lock（间隙锁）
-- 锁定两条记录之间的间隙
-- 表数据：id = 1, 5, 10
-- 锁定：(-∞, 1), (1, 5), (5, 10), (10, +∞)
```

#### 4. 事务使用示例

```typescript

      // 扣除库存

        const product = await queryRunner.findOne(Product, {
          where: { id: item.productId },
          lock: { mode: 'pessimistic_write' }  // 行级锁
        });

        if (product.stock < item.quantity) {
          throw new Error(`商品${product.name}库存不足`);
        }

        await queryRunner.update(
          Product,
          { id: item.productId },
          { stock: product.stock - item.quantity }
        );
      }

      // 创建订单
      const order = queryRunner.create(Order, {
        userId,
        items,
        status: 'pending',
        totalAmount: this.calculateTotal(items)
      });
      await queryRunner.save(order);

      await queryRunner.commitTransaction();
      return order;

    } catch (error) {
      await queryRunner.rollbackTransaction();
      throw error;
    } finally {
      await queryRunner.release();
    }
  }

  // 方式二：装饰器事务（更简洁）
  @Transactional()
  async createOrderDecorator(userId: string, items: OrderItem[]) {
    // 扣除库存
    for (const item of items) {
      const product = await this.productRepo
        .createQueryBuilder('product')
        .setLock('pessimistic_write')
        .where('product.id = :id', { id: item.productId })
        .getOne();

      if (product.stock < item.quantity) {
        throw new Error(`商品${product.name}库存不足`);
      }

      await this.productRepo.decrement(
        { id: item.productId },
        'stock',
        item.quantity
      );
    }

    // 创建订单
    const order = this.orderRepo.create({
      userId,
      items,
      status: 'pending'
    });

    return this.orderRepo.save(order);
  }
}

// 分布式事务解决方案：Saga模式
class OrderSaga {
  async execute(orderRequest: OrderRequest) {
    const steps = [
      { name: '预留库存', execute: this.reserveStock.bind(this) },
      { name: '创建订单', execute: this.createOrder.bind(this) },
      { name: '发送通知', execute: this.sendNotification.bind(this) },
    ];

    const completedSteps = [];

    try {
      for (const step of steps) {
        await step.execute(orderRequest);
        completedSteps.push(step.name);
      }
    } catch (error) {
      // 补偿已完成的步骤（回滚）
      for (const step of completedSteps.reverse()) {
        await this.compensate(step, orderRequest);
      }
      throw error;
    }
  }
}
```

#### 5. 隔离级别选择建议

```sql
-- 不同场景的隔离级别选择

-- 场景一：金融转账（最高安全要求）
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
-- 或使用悲观锁
SELECT balance FROM accounts WHERE id = 1 FOR UPDATE;
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;

-- 场景二：订单查询（平衡性能和一致性）
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
-- MySQL默认REPEATABLE READ，也适合

-- 场景三：日志记录（允许一定脏读）
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
-- 只需要最终一致性的场景

-- 场景四：报表统计（避免幻读）
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
-- 或使用快照读 + 写操作时加锁
SELECT COUNT(*) FROM orders WHERE ...;  -- 快照读
INSERT INTO orders ... FOR UPDATE;      -- 写操作加锁
```

#### 6. PostgreSQL MVCC vs MySQL

| 特性 | PostgreSQL | MySQL (InnoDB) |
|------|------------|----------------|
| **并发模型** | MVCC + SSI | MVCC + 锁 |
| **隔离级别** | 4种（都完整实现） | 4种（RR有细微区别） |
| **REPEATABLE READ** | SSI（快照隔离+谓词锁） | 快照隔离 |
| **幻读处理** | 谓词锁防止幻读 | Gap Lock防止幻读 |
| **UPDATE处理** | 写偏斜（Write Skew）可能 | 依赖Gap Lock |

```sql
-- PostgreSQL  Serializable Snapshot Isolation (SSI)
-- 可以检测并防止写偏斜问题

-- 示例：两个账户总余额必须为1000
-- 事务A：读取账户1=600，账户2=400
-- 事务B：读取账户1=600，账户2=400
-- 事务A：更新账户1=700，账户2=300（满足约束）
-- 事务B：更新账户1=500，账户2=500（也满足约束）
-- 结果：总余额=1000，但两个事务都是合法的
-- SSI会检测这种写偏斜并回滚其中一个事务
```

```typescript
    const instances = await this.consul.health.service({
      service: serviceName,
      passing: true // 只返回健康的实例
    });

    // 负载均衡策略：加权随机
    const weights = instances.map(i => i.Service.Meta?.weight || 1);
    const totalWeight = weights.reduce((a, b) => a + b, 0);
    let random = Math.random() * totalWeight;

    for (let i = 0; i < instances.length; i++) {
      random -= weights[i];
      if (random <= 0) {
        return instances[i].Service;
      }
    }

    return instances[0].Service;
  }
```

```typescript
// 熔断器模式实现
class CircuitBreaker {
  private state: 'CLOSED' | 'OPEN' | 'HALF_OPEN' = 'CLOSED';
  private failureCount = 0;
  private lastFailureTime: number | null = null;

  constructor(
    private threshold = 5,        // 失败次数阈值
    private timeout = 60000,      // 熔断恢复时间(ms)
    private halfOpenRequests = 3  // 半开状态允许的请求数
  ) {}

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    // 检查熔断状态
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime! > this.timeout) {
        this.state = 'HALF_OPEN';
        this.failureCount = 0;
      } else {
        throw new Error('Circuit is OPEN - fallback triggered');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private onSuccess() {
    this.failureCount = 0;
    if (this.state === 'HALF_OPEN') {
      this.state = 'CLOSED';
    }
  }

  private onFailure() {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.failureCount >= this.threshold) {
      this.state = 'OPEN';
    }
  }
}

// 实际应用

  async getProduct(id: string): Promise<Product> {

    });
  }
```

### 4.1 如何防止SQL注入攻击？

**题目描述**：请解释什么是SQL注入攻击，以及如何在开发中防止SQL注入。

**思路引导**：
1. 先理解SQL注入的原理：用户输入被当作SQL代码执行
2. 分析常见注入方式：字符串拼接、ORM误用
3. 掌握正确的防护措施

**完整解答**：

#### 1. SQL注入原理

```sql
-- 原始查询（有漏洞）
SELECT * FROM users
WHERE username = '${username}' AND password = '${password}';

-- 正常输入
-- username: admin
-- password: 123456
-- 执行: SELECT * FROM users WHERE username = 'admin' AND password = '123456'

-- 注入攻击
-- username: admin' --
-- password: 任意
-- 执行: SELECT * FROM users WHERE username = 'admin' --' AND password = 'xxx'
-- -- 后的内容被注释掉，密码验证被绕过

-- 另一种注入：联合查询
-- username: ' UNION SELECT * FROM users --
-- 执行: SELECT * FROM users WHERE username = '' UNION SELECT * FROM users --'
-- 获取所有用户数据

-- 布尔盲注
-- username: admin' AND (SELECT COUNT(*) FROM users) > 0 --
-- 通过返回结果判断条件真假

-- 时间盲注
-- username: admin' AND SLEEP(5) --
-- 通过响应时间判断条件
```

#### 2. 安全防护措施

```typescript

// 安全：参数化查询

// NestJS + TypeORM 示例

  constructor(
    @InjectRepository(User)
    private repo: Repository<User>
  ) {}

  async findById(id: string): Promise<User | null> {
    return this.repo.createQueryBuilder('user')
      .where('user.id = :id', { id })  // 参数化查询
      .getOne();
  }

  // 安全：使用内置方法
  async findByEmail(email: string): Promise<User | null> {
    return this.repo.findOne({ where: { email } });
  }

  // 危险：动态字符串拼接

      .where(`user.${field} = :value`, { value })
```

#### 3. CSRF攻击原理

```

1. 用户已登录bank.com，浏览器保存了cookie
2. 用户访问evil.com，页面包含：

关键点：
- 攻击发生在evil.com（跨站）
- 浏览器自动携带目标网站的cookie
```

```typescript
// 防护一：CSRF Token
class CSRFMiddleware {
  // 生成Token
  static generateToken(): string {
    return crypto.randomBytes(32).toString('hex');
  }

  // 验证Token
  static validateToken(sessionToken: string, requestToken: string): boolean {
    if (!sessionToken || !requestToken) {
      return false;
    }
    return crypto.timingSafeEqual(
      Buffer.from(sessionToken),
      Buffer.from(requestToken)
    );
  }
}

// NestJS实现
import { Injectable } from '@nestjs/common';
import { Request, Response } from 'express';

@Injectable()
export class CSRFCookieService {
  // 为表单生成CSRF Token
  generateToken(req: Request, res: Response): string {
    const token = CSRFMiddleware.generateToken();

    // 存储到session
    req.session.csrfToken = token;

    // 设置到Cookie（HttpOnly防XSS，但前端仍可通过cookie读取）
    res.cookie('csrfToken', token, {
      httpOnly: true,    // 防止XSS读取
      secure: true,      // HTTPS only
      sameSite: 'strict' // 防止CSRF
    });

    return token;
  }
}

// 验证中间件
@Injectable()
export class CSRFMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction) {
    // 只验证非GET请求
    if (['GET', 'HEAD', 'OPTIONS'].includes(req.method)) {
      return next();
    }

    const sessionToken = req.session.csrfToken;
    const bodyToken = req.body._csrf;
    const headerToken = req.headers['x-csrf-token'];

    const requestToken = bodyToken || headerToken;

    if (!CSRFMiddleware.validateToken(sessionToken, requestToken)) {
      return res.status(403).json({ error: 'CSRF token mismatch' });
    }

    next();
  }
}
```

```typescript
class ConsistentHash {
  private ring = new Map<number, string>(); // hash -> node
  private nodes = new Set<string>();         // 所有节点
  private virtualNodes = 150;                // 虚拟节点数

  constructor(nodes: string[]) {
    nodes.forEach(node => this.addNode(node));
  }

  // 添加节点
  addNode(node: string) {
    this.nodes.add(node);

    // 创建虚拟节点
    for (let i = 0; i < this.virtualNodes; i++) {
      const virtualNode = `${node}#VN${i}`;
      const hash = this.hash(virtualNode);
      this.ring.set(hash, node);
    }

    // 保持环有序
    this.sortRing();
  }

  // 移除节点
  removeNode(node: string) {
    if (!this.nodes.has(node)) return;

    // 移除所有虚拟节点
    for (let i = 0; i < this.virtualNodes; i++) {
      const virtualNode = `${node}#VN${i}`;
      const hash = this.hash(virtualNode);
      this.ring.delete(hash);
    }

    this.nodes.delete(node);
  }

  // 获取节点
  getNode(key: string): string {
    if (this.ring.size === 0) {
      throw new Error('No nodes available');
    }

    const hash = this.hash(key);

    // 找到第一个大于等于hash的节点
    for (const [nodeHash, node] of this.ring) {
      if (nodeHash >= hash) {
        return node;
      }
    }

    // 环回起点
    return this.ring.values().next().value;
  }

  // 虚拟节点映射到物理节点
  getPhysicalNode(virtualNode: string): string {
    const index = virtualNode.lastIndexOf('#VN');
    return index !== -1 ? virtualNode.substring(0, index) : virtualNode;
  }

  private hash(key: string): number {
    // MurmurHash3算法（均匀分布）
    return Math.abs(this.murmurhash3(key));
  }

  private murmurhash3(key: string): number {
    // 简化实现，实际应使用专业库
    let h = 0;
    for (let i = 0; i < key.length; i++) {
      h = Math.imul(31, h) + key.charCodeAt(i) | 0;
    }
    return h ^ h >>> 16;
  }

  private sortRing() {
    // 按hash排序
    const sorted = new Map([...this.ring.entries()].sort((a, b) => a[0] - b[0]));
    this.ring = sorted;
  }
}
```

```typescript
// 缓存淘汰算法实现

// 1. LRU (Least Recently Used)
class LRUCache<K, V> {
  private cache = new Map<K, V>();

  constructor(private capacity: number) {}

  get(key: K): V | undefined {
    if (!this.cache.has(key)) return undefined;

    // 移到最前面
    const value = this.cache.get(key)!;
    this.cache.delete(key);
    this.cache.set(key, value);
    return value;
  }

  set(key: K, value: V) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else if (this.cache.size >= this.capacity) {
      // 删除最老的（Map的迭代顺序是插入顺序）
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    this.cache.set(key, value);
  }
}

// 2. LFU (Least Frequently Used)
class LFUCache<K, V> {
  private cache = new Map<K, { value: V; freq: number; lastAccess: number }>();
  private minFreq = 0;

  constructor(private capacity: number) {}

  get(key: K): V | undefined {
    const item = this.cache.get(key);
    if (!item) return undefined;

    item.freq++;
    item.lastAccess = Date.now();
    return item.value;
  }

  set(key: K, value: V) {
    if (this.cache.has(key)) {
      this.cache.get(key)!.value = value;
      this.get(key); // 更新频率
      return;
    }

    if (this.cache.size >= this.capacity) {
      // 找到最小频率的键删除
      let minFreqKey: K | null = null;
      let minFreq = Infinity;

      for (const [k, v] of this.cache) {
        if (v.freq < minFreq) {
          minFreq = v.freq;
          minFreqKey = k;
        }
      }

      if (minFreqKey !== null) {
        this.cache.delete(minFreqKey);
      }
    }

    this.cache.set(key, { value, freq: 1, lastAccess: Date.now() });
  }
}

// 3. TTL (Time To Live) 过期策略
class TTLCache<K, V> {
  private cache = new Map<K, { value: V; expireAt: number }>();
  private cleanupInterval: NodeJS.Timeout;

  constructor(private defaultTTL: number = 3600000) {
    // 每分钟清理过期数据
    this.cleanupInterval = setInterval(() => this.cleanup(), 60000);
  }

  set(key: K, value: V, ttl?: number) {
    const expireAt = Date.now() + (ttl || this.defaultTTL);
    this.cache.set(key, { value, expireAt });
  }

  get(key: K): V | undefined {
    const item = this.cache.get(key);
    if (!item) return undefined;

    if (Date.now() > item.expireAt) {
      this.cache.delete(key);
      return undefined;
    }

    return item.value;
  }

  private cleanup() {
    const now = Date.now();
    for (const [key, item] of this.cache) {
      if (now > item.expireAt) {
        this.cache.delete(key);
      }
    }
  }
}
```

```typescript
class RedisMigration {
  // 1. 预先建立连接
  async prepareMigration(
    source: Redis,
    target: Redis,
    keys: string[]
  ) {
    // 2. 使用SCAN遍历（而非KEYS）
    const stream = source.scanStream({
      match: keys,
      count: 100
    });

    for await (const keys of stream) {
      // 批量获取数据
      const values = await source.mget(keys);

      // 批量写入新集群
      const pipeline = target.pipeline();
      keys.forEach((key, i) => {
        pipeline.set(key, values[i]);
      });
      await pipeline.exec();
    }
  }

  // 3. 增量同步（双写期间）
  async incrementalSync(
    source: Redis,
    target: Redis,
    keyPattern: string
  ) {
    // 使用PSUBSCRIBE订阅keyspace notifications
    const subscriber = source.duplicate();

    await subscriber.pSubscribe('__keyevent@0__:set', (key) => {
      // 同步更新到目标集群
      this.syncKey(source, target, key);
    });

    await subscriber.pSubscribe('__keyevent@0__:del', (key) => {
      // 删除同步到目标集群
      target.del(key);
    });
  }

  private async syncKey(source: Redis, target: Redis, key: string) {
    const value = await source.get(key);
    if (value !== null) {
      const ttl = await source.ttl(key);
      if (ttl > 0) {
        await target.setex(key, ttl, value);
      } else {
        await target.set(key, value);
      }
    }
  }
}
```

```typescript
class RetryableConsumer {
  private maxRetries = 3;
  private retryDelays = [1000, 5000, 30000]; // 1s, 5s, 30s

  async handleMessage(message: Message) {
    try {
      await this.processMessage(message);
    } catch (error) {
      const retryCount = this.getRetryCount(message);

      if (retryCount < this.maxRetries) {
        // 延迟重试
        const delay = this.retryDelays[retryCount];
        await this.scheduleRetry(message, delay);
      } else {
        // 超过最大重试次数，发送到死信队列
        await this.sendToDLQ(message, error);
      }
    }
  }

  private async scheduleRetry(message: Message, delay: number) {
    // 使用延时消息或定时器
    await this.kafka.send({
      topic: 'order.events.retry',
      messages: [{
        key: message.id,
        value: JSON.stringify(message),
        headers: {
          'x-retry-count': String(this.getRetryCount(message) + 1),
          'x-original-offset': String(message.offset)
        }
      }],
      delay: {
        topic: 'order.events.retry',
        delayMs: delay
      }
    });
  }
}
```

```typescript
class TrieNode {
  children: Map<string, TrieNode> = new Map();
  isEndOfWord = false;
  frequency = 0;  // 搜索频率，用于排序
}

class SearchSuggestion {
  private root = new TrieNode();

  // 添加搜索词
  insert(word: string) {
    let node = this.root;
    for (const char of word.toLowerCase()) {
      if (!node.children.has(char)) {
        node.children.set(char, new TrieNode());
      }
      node = node.children.get(char)!;
    }
    node.isEndOfWord = true;
    node.frequency++;
  }

  // 获取前缀匹配的词
  getSuggestions(prefix: string, limit = 10): string[] {
    let node = this.root;
    for (const char of prefix.toLowerCase()) {
      if (!node.children.has(char)) {
        return [];
      }
      node = node.children.get(char)!;
    }

    // DFS收集所有匹配的词
    const suggestions: string[] = [];
    this.collectSuggestions(node, prefix.toLowerCase(), suggestions);

    // 按频率排序
    suggestions.sort((a, b) => {
      const freqA = this.getWordFrequency(a);
      const freqB = this.getWordFrequency(b);
      return freqB - freqA;
    });

    return suggestions.slice(0, limit);
  }

  private collectSuggestions(node: TrieNode, current: string, results: string[]) {
    if (results.length >= 100) return; // 限制收集数量

    if (node.isEndOfWord) {
      results.push(current);
    }

    for (const [char, childNode] of node.children) {
      this.collectSuggestions(childNode, current + char, results);
    }
  }
}
```

```typescript
// 方案一：Redis有序集合 + 轮询
class RedisDelayQueue {
  async addTask(taskId: string, executeAt: number, payload: any) {
    const score = executeAt; // 过期时间戳作为score
    await this.redis.zadd('delay:tasks', score, JSON.stringify({ taskId, payload }));
  }

  async pollTasks() {
    const now = Date.now();
    const minScore = 0;
    const maxScore = now;

    // 取出所有到期的任务
    const tasks = await this.redis.zrangebyscore(
      'delay:tasks',
      minScore,
      maxScore
    );

    for (const task of tasks) {
      const { taskId, payload } = JSON.parse(task);

      // 删除任务（使用Lua保证原子性）
      const removed = await this.redis.zrem('delay:tasks', task);
      if (removed === 1) {
        await this.executeTask(taskId, payload);
      }
    }
  }

  @Interval(1000) // 每秒轮询
  pollTasks() {
    this.pollTasks().catch(console.error);
  }
}

// 方案二：多层时间轮
class TimingWheel {
  private wheels: Map<number, Map<string, Task>> = new Map();
  private currentTime = Date.now();

  // 添加任务到对应的时间槽
  addTask(task: Task, delayMs: number) {
    const executeAt = Date.now() + delayMs;
    const slot = Math.floor(executeAt / 1000) % 3600; // 小时槽

    if (!this.wheels.has(slot)) {
      this.wheels.set(slot, new Map());
    }

    this.wheels.get(slot)!.set(task.id, task);
  }

  // 推进时间轮
  tick() {
    const now = Math.floor(Date.now() / 1000) % 3600;
    const slotTasks = this.wheels.get(now);

    if (slotTasks) {
      for (const [taskId, task] of slotTasks) {
        this.executeTask(task);
      }
      this.wheels.delete(now);
    }
  }

  @Interval(1000)
  tick() {
    this.tick();
  }
}
```
