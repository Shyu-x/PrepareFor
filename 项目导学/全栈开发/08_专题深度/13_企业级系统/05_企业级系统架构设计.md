# 企业级系统架构设计

## 一、系统架构设计基础

### 1.1 什么是企业级系统？

想象一下你要建一座写字楼。从外面看，它是一个整体；但内部有很多楼层，每个楼层有很多房间，每间房有不同的用途（会议室、办公室、卫生间等）。还要有电梯、空调、消防系统等基础设施把这些连接起来。

**企业级系统**就是这样的"写字楼"：
- 它要服务很多人（成千上万的员工、客户、合作伙伴）
- 它要处理很多复杂的事情（审批、报表、数据分析）
- 它要稳定可靠（7x24小时运行，不能动不动就崩溃）
- 它要安全（数据不能泄露，权限不能混乱）
- 它要可以扩展（业务增长了，系统要能跟着扩展）

### 1.2 企业级系统 vs 小型系统

| 维度 | 小型系统 | 企业级系统 |
|------|----------|------------|
| **用户数** | 几十到几百 | 几千到几十万 |
| **并发量** | 很低 | 很高 |
| **数据量** | 几百GB | 几十TB甚至PB |
| **复杂度** | 单模块 | 跨多个业务域 |
| **可用性** | 99% | 99.9%甚至99.99% |
| **扩展性** | 垂直扩展 | 水平扩展 |
| **团队规模** | 几个开发 | 几十甚至几百开发 |
| **开发周期** | 几个月 | 几年甚至更长 |

## 二、SaaS多租户架构

### 2.1 什么是多租户？

**多租户（Multi-tenancy）** 就像一个大型商场，里面有很多店铺。每个店铺都是独立的，但共享商场的基础设施（电梯、空调、安保）。

在软件系统中：
- **单租户**：每个客户单独部署一套系统（就像独栋别墅）
- **多租户**：多个客户共享一套系统（就像公寓大楼）

### 2.2 多租户的优势

| 优势 | 说明 |
|------|------|
| **成本低** | 共享基础设施，维护成本分摊 |
| **部署快** | 新客户上线只需配置，不用部署 |
| **统一升级** | 一次升级，所有租户受益 |
| **易于管理** | 统一的监控、备份、安全策略 |
| **弹性扩展** | 根据整体负载扩缩容 |

```sql
-- ============================================
-- 多租户数据隔离 - 方案对比
-- ============================================

-- 方案一：共享数据库，共享表（通过tenant_id隔离）
-- 这是最常见的SaaS多租户方案

-- 员工表（带租户ID）
CREATE TABLE employees (
    id              BIGSERIAL PRIMARY KEY,
    tenant_id       BIGINT NOT NULL,                    -- 租户ID
    employee_name   VARCHAR(100) NOT NULL,
    email           VARCHAR(100),

    -- 关键：在所有表上都加tenant_id索引
    UNIQUE(tenant_id, email)
);

CREATE INDEX idx_employees_tenant ON employees(tenant_id);

-- 订单表
CREATE TABLE orders (
    id              BIGSERIAL PRIMARY KEY,
    tenant_id       BIGINT NOT NULL,
    order_no        VARCHAR(50) NOT NULL,
    customer_id     BIGINT,
    amount          DECIMAL(15,2),

    -- 确保同一租户内订单号唯一
    UNIQUE(tenant_id, order_no)
);

CREATE INDEX idx_orders_tenant ON orders(tenant_id);
CREATE INDEX idx_orders_customer ON orders(tenant_id, customer_id);

-- ============================================
-- 数据查询示例
-- ============================================

-- 查询所有订单（会自动带上tenant_id条件）

-- JOIN employees e ON o.customer_id = e.id AND o.tenant_id = e.tenant_id

-- 方案二：独立Schema（PostgreSQL Schema隔离）

-- tenant_b_db
```

```typescript
import { Request } from 'express';

/**
 * 租户上下文
 *
 * 在请求处理过程中，保存当前请求所属的租户信息
 * 这个信息会一直传递下去，用于数据过滤
 */
interface TenantContext {
  tenantId: number;           // 租户ID
  tenantCode: string;        // 租户编码
  tenantName: string;        // 租户名称
  plan: 'free' | 'pro' | 'enterprise';  // 租户套餐
  features: string[];        // 启用的功能
  limits: {                  // 使用限制
    maxUsers: number;
    maxStorage: number;
    maxApiCalls: number;
  };
}

@Injectable({ scope: Scope.REQUEST })
export class TenantService {
  private context: TenantContext | null = null;

  constructor(@Inject(REQUEST) private request: Request) {}

  /**
   * 获取当前租户ID
   */
  getTenantId(): number {
    if (!this.context) {
      throw new Error('Tenant context not initialized');
    }
    return this.context.tenantId;
  }

  /**
   * 获取当前租户上下文
   */
  getContext(): TenantContext {
    if (!this.context) {
      throw new Error('Tenant context not initialized');
    }
    return this.context;
  }

  /**
   * 设置租户上下文
   * 通常在拦截器中调用
   */
  setContext(context: TenantContext): void {
    this.context = context;
  }

  /**
   * 检查是否超限
   */
  checkLimits(limitType: keyof TenantContext['limits']): boolean {
    if (!this.context) return false;

    const limit = this.context.limits[limitType];
    if (limit === undefined) return true;  // 无限制

    // 实际应该查询当前使用量进行对比
    // 这里简化处理
    return true;
  }
}

// src/tenancy/interceptors/tenant.interceptor.ts
import {
  Injectable,
  NestInterceptor,
  ExecutionContext,
  CallHandler,
} from '@nestjs/common';
import { Observable } from 'rxjs';
import { TenantService } from '../tenant.service';

/**
 * 租户上下文拦截器
 *
 * 在请求处理之前，从请求中提取租户信息
 * 并设置到TenantService中
 *
 * 租户信息可以从以下来源获取：
 * 1. Header中的 tenant-id
 * 2. JWT Token中的租户信息
 * 3. 子域名
 * 4. URL参数
 */
@Injectable()
export class TenantInterceptor implements NestInterceptor {
  constructor(private tenantService: TenantService) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const request = context.switchToHttp().getRequest();

    // 1. 从JWT Token中获取租户信息（最常见的方式）
    // 假设我们已经在其他地方验证了token的合法性
    const user = request.user;

    if (user?.tenantId) {
      // 设置租户上下文
      this.tenantService.setContext({
        tenantId: user.tenantId,
        tenantCode: user.tenantCode,
        tenantName: user.tenantName,
        plan: user.plan,
        features: user.features || [],
        limits: user.limits || { maxUsers: 10, maxStorage: 1000000000, maxApiCalls: 10000 },
      });
    }

    return next.handle();
  }
}

// src/tenancy/typeorm/tenant.plugin.ts
import { DataSource, EntitySubscriberInterface } from 'typeorm';
import { TenantService } from '../tenant.service';

/**
 * TypeORM多租户查询插件
 *
 * 自动在所有查询中添加tenant_id条件
 * 开发者无需在每个查询中手动添加
 */
export function setupTenantPlugin(
  dataSource: DataSource,
  tenantService: TenantService
) {
  // 使用QueryRunner钩子自动添加租户过滤
  dataSource.queryRunner.addListener('beforeQuery', (event) => {
    // 跳过DDL操作
    if (event.query.startsWith('ALTER') ||
        event.query.startsWith('CREATE') ||
        event.query.startsWith('DROP')) {
      return;
    }

    // 尝试从event中获取query和parameters
    const { query, parameters } = event;

    try {
      // 检查是否需要添加租户过滤
      const needsFilter = shouldAddTenantFilter(query);

      if (needsFilter && parameters) {
        const tenantId = tenantService.getTenantId();

        // 简单处理：在WHERE条件中添加 tenant_id = ?
        // 实际实现需要更复杂的SQL解析
        if (!query.includes('tenant_id')) {
          const modifiedQuery = addTenantCondition(query, tenantId);
          event.query = modifiedQuery.query;
          event.parameters = modifiedQuery.parameters;
        }
      }
    } catch (error) {
      // 忽略错误，继续执行
      console.error('Tenant filter error:', error);
    }
  });
}

/**
 * 判断是否应该添加租户过滤
 */
function shouldAddTenantFilter(query: string): boolean {
  // SELECT查询需要过滤
  if (query.trim().toUpperCase().startsWith('SELECT')) {
    return true;
  }

  // UPDATE和DELETE需要过滤
  const upperQuery = query.trim().toUpperCase();
  if (upperQuery.startsWith('UPDATE') || upperQuery.startsWith('DELETE')) {
    return true;
  }

  return false;
}

/**
 * 添加租户条件
 */
function addTenantCondition(
  query: string,
  tenantId: number
): { query: string; parameters: any[] } {
  // 简化实现
  // 实际需要解析SQL语法，正确添加条件

  const upperQuery = query.toUpperCase();

  if (upperQuery.includes('WHERE')) {
    // 已有WHERE条件，AND添加
    const insertIndex = query.toUpperCase().indexOf('WHERE') + 5;
    const modifiedQuery =
      query.slice(0, insertIndex) +
      ` tenant_id = ${tenantId} AND` +
      query.slice(insertIndex);

    return { query: modifiedQuery, parameters: [] };
  } else if (upperQuery.includes('ORDER') || upperQuery.includes('LIMIT')) {
    // 没有WHERE但有ORDER/LIMIT
    const insertIndex = Math.min(
      upperQuery.includes('ORDER') ? query.toUpperCase().indexOf('ORDER') : query.length,
      upperQuery.includes('LIMIT') ? query.toUpperCase().indexOf('LIMIT') : query.length
    );

    const modifiedQuery =
      query.slice(0, insertIndex) +
      ` WHERE tenant_id = ${tenantId} ` +
      query.slice(insertIndex);

    return { query: modifiedQuery, parameters: [] };
  } else {
    // 简单追加WHERE
    return {
      query: `${query} WHERE tenant_id = ${tenantId}`,
      parameters: [],
    };
  }
}
```

```typescript
// src/app.module.ts

@Module({

      .apply(TenantMiddleware)
      .forRoutes('*');  // 应用到所有路由

// src/middleware/tenant.middleware.ts
import { Injectable, NestMiddleware } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';

/**
 * 租户解析中间件
 *
 * 从请求中解析租户信息：
 * 1. 子域名：xxx.tenant.com -> tenant = xxx
 * 2. URL路径：/tenant/{tenantCode}/api/...
 * 3. Header：X-Tenant-ID
 */
@Injectable()
export class TenantMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction) {
    let tenantCode: string | undefined;

    // 方式一：从子域名解析
    // 例如：acme.saasplatform.com -> tenantCode = "acme"
    const hostname = req.hostname;
    const parts = hostname.split('.');

    if (parts.length >= 3 && parts[0] !== 'www') {
      tenantCode = parts[0];
    }

    // 方式二：从URL路径解析
    // 例如：/api/tenants/acme/...
    if (!tenantCode) {
      const pathMatch = req.path.match(/\/tenants\/([^/]+)/);
      if (pathMatch) {
        tenantCode = pathMatch[1];
      }
    }

    // 方式三：从Header解析
    // 例如：X-Tenant-Code: acme
    if (!tenantCode) {
      tenantCode = req.headers['x-tenant-code'] as string;
    }

    // 解析成功后，挂载到request对象上
    if (tenantCode) {
      req.tenantCode = tenantCode;
    }

    next();
  }
}
```

### 3.2 RBAC权限模型实现

```sql
-- ============================================
-- RBAC权限模型数据库设计
-- ============================================

-- 用户表
CREATE TABLE users (
    id              BIGSERIAL PRIMARY KEY,
    username        VARCHAR(50) UNIQUE NOT NULL,
    password_hash   VARCHAR(200) NOT NULL,
    email           VARCHAR(100),
    status          VARCHAR(20) DEFAULT 'active',
    tenant_id       BIGINT NOT NULL,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 角色表
CREATE TABLE roles (
    id              BIGSERIAL PRIMARY KEY,
    role_code       VARCHAR(50) UNIQUE NOT NULL,     -- 角色编码：admin/manager/user
    role_name       VARCHAR(100) NOT NULL,           -- 角色名称：管理员/经理/普通用户
    role_type       VARCHAR(20),                     -- 角色类型：system/custom
    description     VARCHAR(500),
    tenant_id       BIGINT NOT NULL,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 权限表
CREATE TABLE permissions (
    id              BIGSERIAL PRIMARY KEY,
    permission_code VARCHAR(100) UNIQUE NOT NULL,     -- 权限编码：user:create/article:read
    permission_name VARCHAR(200) NOT NULL,            -- 权限名称：创建用户/阅读文章
    resource_type   VARCHAR(50),                      -- 资源类型：user/article/order
    action          VARCHAR(50),                      -- 操作类型：create/read/update/delete

    -- 权限分类
    category        VARCHAR(50),                     -- 所属分类：用户管理/订单管理
    description     VARCHAR(500),
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 角色-权限关联表（多对多）
CREATE TABLE role_permissions (
    role_id         BIGINT NOT NULL,
    permission_id   BIGINT NOT NULL,

    -- 额外的权限约束条件（JSON格式）
    -- 例如：只能查看自己部门的订单
    -- {"field": "department_id", "operator": "=", "value": "${user.department_id}"}
    conditions      JSONB DEFAULT '{}',

    PRIMARY KEY (role_id, permission_id)
);

-- 用户-角色关联表（多对多）
CREATE TABLE user_roles (
    user_id         BIGINT NOT NULL,
    role_id         BIGINT NOT NULL,
    -- 如果用户有多个角色，可以设置主角色
    is_primary      BOOLEAN DEFAULT false,

    PRIMARY KEY (user_id, role_id)
);
```

```typescript
import { Injectable, CanActivate, ExecutionContext } from '@nestjs/common';
import { PermissionService } from '../permission.service';

/**
 * 权限检查装饰器使用的参数
 */
interface RequirePermissions {
  resources: string[];
  actions: string[];
  mode?: 'all' | 'any';  // 是需要全部权限还是任一权限即可
}

/**
 * 权限守卫
 *
 * 用于保护API端点，确保调用者有相应权限
 *
 * 使用示例：
 * @UseGuards(PermissionGuard)
 * @RequirePermissions({ resources: ['order'], actions: ['create'] })
 * @Post()
 * async createOrder() {}
 */
@Injectable()
export class PermissionGuard implements CanActivate {
  constructor(
    private permissionService: PermissionService,
    private reflector: Reflector,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    // 1. 获取路由上定义的权限要求
    const requiredPermissions = this.reflector.get<RequirePermissions>(
      'requiredPermissions',
      context.getHandler()
    );

    // 如果没有定义权限要求，放行
    if (!requiredPermissions) {
      return true;
    }

    // 2. 获取当前用户
    const request = context.switchToHttp().getRequest();
    const user = request.user;

    if (!user) {
      throw new UnauthorizedException('请先登录');
    }

    // 3. 逐个检查权限
    const mode = requiredPermissions.mode || 'any';
    const results = [];

    for (const resource of requiredPermissions.resources) {
      for (const action of requiredPermissions.actions) {
        const result = await this.permissionService.checkPermission(user, {
          resource,
          action,
        });
        results.push(result.allowed);

        // any模式：任意一个通过即可
        if (mode === 'any' && result.allowed) {
          return true;
        }
      }
    }

    // all模式：必须全部通过
    if (mode === 'all') {
      return results.every(r => r);
    }

    // any模式：必须至少有一个通过
    return results.some(r => r);
  }
}
```

```typescript
    }

    // 聚合计算
    if (options.aggregates) {
      const aggParts = [];
      for (const [func, fields] of Object.entries(options.aggregates)) {
        for (const field of fields) {

        sql += ', ' + aggParts.join(', ');

    if (options.tenantIds && options.tenantIds.length > 0) {
      sql += ` WHERE tenant_id IN (${options.tenantIds.join(', ')})`;
    }

    if (options.groupBy && options.groupBy.length > 0) {
      sql += ' GROUP BY ' + options.groupBy.join(', ');

      sql += ` LIMIT ${options.limit}`;
    }
    if (options.offset) {

            error: error.message,
          });
        }
      }
```

```typescript
  pricing: {
    billingCycle: 'monthly' | 'yearly';  // 计费周期
    price: number;             // 价格
    currency: string;          // 币种
    discount?: number;         // 折扣（如年付8折）
  };

  // 包含的资源
  included: {
    maxUsers: number;         // 最大用户数
    maxStorage: number;       // 存储空间（字节）
    maxApiCalls: number;      // API调用次数
    features: string[];        // 包含的功能列表
  };

  // 超额费率
  overage: {
    perUser?: number;         // 超出用户的单价
    perStorage?: number;       // 超出存储的单价
    perApiCall?: number;       // 超出API的单价
  };
}

/**
 * 计费服务

@Injectable()
class BillingService {
  constructor(
    @InjectRepository(Tenant)
    private tenantRepository: Repository<Tenant>,

    @InjectRepository(UsageRecord)
    private usageRepository: Repository<UsageRecord>,

    @InjectRepository(BillingEvent)
    private eventRepository: Repository<BillingEvent>,

    @InjectRepository(Invoice)
    private invoiceRepository: Repository<Invoice>,
  ) {}

  /**
   * 获取租户的当前套餐
   */
  async getTenantPlan(tenantId: number): Promise<SubscriptionPlan> {
    const tenant = await this.tenantRepository.findOne({
      where: { id: tenantId }
    });

    return this.getPlanById(tenant.planId);
  }

  /**
   * 记录用量
   *
   * 每次用户使用服务时调用
   * 例如：API被调用、文件被上传、用户被创建
   */
  async recordUsage(
    tenantId: number,
    metric: string,
    value: number,
    metadata?: Record<string, any>
  ): Promise<void> {
    // 1. 记录原始用量
    const usage = this.usageRepository.create({
      tenantId,
      metric,
      value,
      unit: this.getMetricUnit(metric),
      timestamp: new Date(),
      metadata,
    });

    await this.usageRepository.save(usage);

    // 2. 检查是否超过限制
    const plan = await this.getTenantPlan(tenantId);
    const currentUsage = await this.getCurrentUsage(tenantId, metric);
    const included = this.getIncludedLimit(plan, metric);

    // 3. 如果超限，触发告警
    if (included && currentUsage > included) {
      await this.notifyOverage(tenantId, metric, currentUsage, included);
    }

    // 4. 如果超限严重（超过150%），可以触发限流
    if (included && currentUsage > included * 1.5) {
      await this.triggerRateLimit(tenantId);
    }
  }

  /**
   * 计算月度账单
   */
  async calculateMonthlyBill(
    tenantId: number,
    billingMonth: string  // 格式：YYYY-MM
  ): Promise<{
    subscription: number;      // 订阅费用
    usage: number;             // 用量费用
    total: number;
    currency: string;
    details: {
      planFee: number;
      userOverage: number;
      storageOverage: number;
      apiCallOverage: number;
    };
  }> {
    const plan = await this.getTenantPlan(tenantId);

    // 1. 计算订阅费用（按日折算）
    const subscriptionFee = await this.calculateSubscriptionFee(
      tenantId, plan, billingMonth
    );

    // 2. 计算超额费用
    const usageDetails = await this.calculateOverageFees(tenantId, plan, billingMonth);

    // 3. 汇总
    const total = subscriptionFee + usageDetails.userOverage +
                  usageDetails.storageOverage + usageDetails.apiCallOverage;

    return {
      subscription: subscriptionFee,
      usage: usageDetails.total,
      total,
      currency: 'CNY',
      details: {
        planFee: subscriptionFee,
        userOverage: usageDetails.userOverage,
        storageOverage: usageDetails.storageOverage,
        apiCallOverage: usageDetails.apiCallOverage,
      },
    };
  }

  /**
   * 计算订阅费用
   */
  private async calculateSubscriptionFee(
    tenantId: number,
    plan: SubscriptionPlan,
    billingMonth: string
  ): Promise<number> {
    // 订阅费用基于计费周期
    const monthlyPrice = plan.pricing.billingCycle === 'yearly'
      ? plan.pricing.price / 12
      : plan.pricing.price;

    // 检查当月是否有套餐变更
    const planChanges = await this.getPlanChanges(tenantId, billingMonth);

    if (planChanges.length === 0) {
      return monthlyPrice;
    }

    // 有套餐变更，按天折算
    let totalFee = 0;
    let currentDate = new Date(billingMonth + '-01');
    const monthEnd = new Date(currentDate.getFullYear(), currentDate.getMonth() + 1, 0);

    for (const change of planChanges) {
      const changeDate = new Date(change.effectiveDate);

      // 变更前的天数费用
      const daysBeforeChange = Math.floor(
        (changeDate.getTime() - currentDate.getTime()) / (1000 * 60 * 60 * 24)
      );

      totalFee += (monthlyPrice / monthEnd.getDate()) * daysBeforeChange;

      // 更新当前日期和价格
      currentDate = changeDate;
    }

    // 最后一段时期的费用
    const lastDays = monthEnd.getDate() - currentDate.getDate() + 1;
    totalFee += (monthlyPrice / monthEnd.getDate()) * lastDays;

    return Math.round(totalFee * 100) / 100;
  }

  /**
   * 计算超额费用
   */
  private async calculateOverageFees(
    tenantId: number,
    plan: SubscriptionPlan,
    billingMonth: string
  ): Promise<{
    userOverage: number;
    storageOverage: number;
    apiCallOverage: number;
    total: number;
  }> {
    const result = {
      userOverage: 0,
      storageOverage: 0,
      apiCallOverage: 0,
      total: 0,
    };

    // 用户数超额
    const currentUsers = await this.getUserCount(tenantId);
    if (plan.included.maxUsers > 0 && currentUsers > plan.included.maxUsers) {
      const overageUsers = currentUsers - plan.included.maxUsers;
      result.userOverage = overageUsers * (plan.overage.perUser || 10);
    }

    // 存储超额
    const currentStorage = await this.getCurrentUsage(tenantId, 'storage');
    if (plan.included.maxStorage > 0 && currentStorage > plan.included.maxStorage) {
      const overageStorage = currentStorage - plan.included.maxStorage;
      const overageGB = overageStorage / (1024 * 1024 * 1024);
      result.storageOverage = overageGB * (plan.overage.perStorage || 0.1);
    }

    // API调用超额
    const currentApiCalls = await this.getCurrentUsage(tenantId, 'api_calls');
    if (plan.included.maxApiCalls > 0 && currentApiCalls > plan.included.maxApiCalls) {
      const overageCalls = currentApiCalls - plan.included.maxApiCalls;
      result.apiCallOverage = overageCalls * (plan.overage.perApiCall || 0.001);
    }

    result.total = result.userOverage + result.storageOverage + result.apiCallOverage;

    return result;
  }

  /**
   * 生成发票
   */
  async generateInvoice(
    tenantId: number,
    billingMonth: string
  ): Promise<Invoice> {
    // 1. 计算账单
    const bill = await this.calculateMonthlyBill(tenantId, billingMonth);

    // 2. 创建发票记录
    const invoice = this.invoiceRepository.create({
      invoiceNo: await this.generateInvoiceNo(),
      tenantId,
      billingMonth,
      subscriptionAmount: bill.subscription,
      usageAmount: bill.usage,
      totalAmount: bill.total,
      currency: bill.currency,
      status: 'pending',
      dueDate: new Date(billingMonth + '-28'),  // 28日到期
    });

    return this.invoiceRepository.save(invoice);
  }

  /**
   * 处理支付
   */
  async processPayment(
    invoiceId: number,
    paymentMethod: 'credit_card' | 'bank_transfer' | 'alipay' | 'wechat'
  ): Promise<{ success: boolean; transactionId?: string; error?: string }> {
    const invoice = await this.invoiceRepository.findOne({
      where: { id: invoiceId }
    });

    if (!invoice) {
      return { success: false, error: '发票不存在' };
    }

    if (invoice.status === 'paid') {
      return { success: false, error: '发票已支付' };
    }

    // 模拟支付处理
    try {
      // 实际应该调用支付网关
      const transactionId = await this.callPaymentGateway({
        amount: invoice.totalAmount,
        currency: invoice.currency,
        method: paymentMethod,
        description: `Invoice ${invoice.invoiceNo}`,
      });

      // 更新发票状态
      invoice.status = 'paid';
      invoice.paidAt = new Date();
      invoice.transactionId = transactionId;
      await this.invoiceRepository.save(invoice);

      // 记录计费事件
      await this.eventRepository.save({
        eventType: 'invoice_generated',
        tenantId: invoice.tenantId,
        amount: invoice.totalAmount,
        currency: invoice.currency,
        timestamp: new Date(),
      });

      return { success: true, transactionId };

    } catch (error) {
      return { success: false, error: error.message };
    }
  }

  /**
   * 升级套餐
   */
  async upgradePlan(
    tenantId: number,
    newPlanId: string,
    effectiveDate?: Date
  ): Promise<void> {
    const effective = effectiveDate || new Date();

    // 记录套餐变更
    await this.savePlanChange(tenantId, newPlanId, effective);

    // 更新租户当前套餐
    await this.tenantRepository.update(tenantId, { planId: newPlanId });

    // 记录事件
    await this.eventRepository.save({
      eventType: 'subscription_upgraded',
      tenantId,
      planId: newPlanId,
      amount: 0,
      currency: 'CNY',
      timestamp: new Date(),
    });
  }
```

```typescript
        startMonth: options.startMonth
      });

      query.andWhere('invoice.billingMonth <= :endMonth', {
        endMonth: options.endMonth
      });

    const [list, total] = await query
      .orderBy('invoice.billingMonth', 'DESC')
      .skip((options?.page - 1) * (options?.pageSize || 20))
      .take(options?.pageSize || 20)
      .getManyAndCount();

    const totalAmount = await query
      .select('SUM(invoice.totalAmount)', 'total')
      .getRawOne();

    return {
      list,
      total,
      totalAmount: parseFloat(totalAmount?.total || '0'),
    };
  }
```

### 6.1 企业级架构核心要点回顾

1. **SaaS多租户**：通过租户ID实现数据隔离，共享基础设施降低成本

### 6.3 架构设计原则

| 原则 | 说明 |
|------|------|
| **最小权限** | 每个组件只拥有完成任务所需的最小权限 |
| **纵深防御** | 多层安全防护，单层失效不影响整体 |
| **默认安全** | 安全配置开箱即用，无需额外配置 |
| **可观测性** | 日志、监控、追踪全面覆盖 |
| **故障隔离** | 单个租户问题不影响其他租户 |
| **弹性扩展** | 根据负载自动扩缩容 |
| **数据安全** | 加密存储、传输加密、定期备份 |
