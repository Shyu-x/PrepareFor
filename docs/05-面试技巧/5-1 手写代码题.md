## 一、事件循环与异步编程

### 1.1 带并发限制的异步调度器

**题目要求**：实现一个Scheduler类，使同时运行的任务不超过limit个。

**考察点**：队列控制、Promise状态流转、异步编程能力。

**实现思路**：

```javascript
class Scheduler {
    constructor(limit = 2) {
        this.limit = limit;
        this.running = 0;
        this.queue = [];
    }

    async add(promiseCreator) {
        // 如果当前运行任务已满，加入等待队列
        if (this.running >= this.limit) {
            await new Promise(resolve => this.queue.push(resolve));
        }
        
        this.running++;
        
        try {
            return await promiseCreator();
        } finally {
            this.running--;
            // 任务完成后，从队列取出下一个任务执行
            if (this.queue.length > 0) {
                const nextResolve = this.queue.shift();
                nextResolve();
            }
        }
    }
}

// 测试用例
const timeout = (ms) => new Promise(resolve => setTimeout(resolve, ms));

const scheduler = new Scheduler(2);

const addTask = (time, order) => {
    scheduler.add(() => timeout(time)).then(() => console.log(order));
};

addTask(1000, '1');
addTask(500, '2');
addTask(300, '3');
addTask(400, '4');

// 预期输出：2, 3, 1, 4
// 解释：1和2先并发执行，2完成(500ms)后3开始，3完成(300ms)后4开始，1最后完成(1000ms)
```

**关键点解析**：

- 使用队列管理等待的任务
- Promise的resolve/reject控制任务执行时机
- finally块确保任务完成后释放执行槽位
- 支持高并发场景下的任务调度

### 1.2 实现Promise.all

**题目要求**：手写Promise.all函数。

**考察点**：Promise API理解、错误处理、边界情况。

```javascript
function promiseAll(promises) {
    // 边界情况：空数组
    if (!Array.isArray(promises)) {
        return Promise.reject(new TypeError('参数必须是数组'));
    }
    
    // 边界情况：空数组直接resolve
    if (promises.length === 0) {
        return Promise.resolve([]);
    }
    
    return new Promise((resolve, reject) => {
        const results = [];
        let completed = 0;
        
        // 使用Promise.resolve处理非Promise值
        promises.forEach((promise, index) => {
            Promise.resolve(promise).then(
                value => {
                    results[index] = value;
                    completed++;
                    
                    // 所有Promise都完成后resolve
                    if (completed === promises.length) {
                        resolve(results);
                    }
                },
                reason => {
                    // 任意一个Promise reject就reject
                    reject(reason);
                }
            );
        });
    });
}

// 测试
const p1 = Promise.resolve(1);
const p2 = Promise.resolve(2);
const p3 = new Promise((resolve) => setTimeout(() => resolve(3), 100));

promiseAll([p1, p2, p3]).then(console.log); // [1, 2, 3]
promiseAll([]).then(console.log); // []
```

```javascript
function promiseAllSettled(promises) {

        return Promise.reject(new TypeError('参数必须是数组'));

    if (promises.length === 0) {

                    completed++;
                    if (completed === promises.length) {
                        resolve(results);
                    }
                }
            );
        });
    });
}
```

### 1.5 实现红交换（Promise重试机制）

**题目要求**：实现一个retry函数，支持指定重试次数和间隔时间。

```javascript
async function retry(fn, maxRetries = 3, delay = 1000) {
    let lastError;
    
    for (let i = 0; i <= maxRetries; i++) {
        try {
            return await fn();
        } catch (error) {
            lastError = error;
            if (i < maxRetries) {
                await new Promise(resolve => setTimeout(resolve, delay));
            }
        }
    }
    
    throw lastError;
}

// 使用示例
async function fetchData() {
    // 模拟可能失败的请求
    if (Math.random() > 0.5) {
        throw new Error('请求失败');
    }
    return '数据获取成功';
}

// 测试
retry(fetchData, 3, 1000)
    .then(console.log)
    .catch(console.error);
```

```javascript
function Animal(name) {

Dog.prototype = Object.create(Animal.prototype);

// 修复构造函数指向
```

```
dog → Dog.prototype → Animal.prototype → Object.prototype → null
     ↑                   ↑                   ↑
     constructor         constructor         constructor
     bark()              speak()             toString()
```

### 2.3 instanceof实现

**题目要求**：手写instanceOf函数。

```javascript
function myInstanceOf(left, right) {
    // 获取右边的原型对象
    let rightPrototype = right.prototype;
    
    // 获取左边的原型对象
    let leftPrototype = Object.getPrototypeOf(left);
    
    // 沿着原型链向上查找
    while (leftPrototype !== null) {
        if (leftPrototype === rightPrototype) {
            return true;
        }
        leftPrototype = Object.getPrototypeOf(leftPrototype);
    }
    
    return false;
}

// 测试

    this.name = name;
}

myInstanceOf(person, Object); // true
myInstanceOf(person, Array); // false
```

## 三、闭包与函数

### 3.1 防抖函数

**题目要求**：实现防抖函数，支持立即执行和取消功能。

**考察点**：闭包、this指向、定时器管理。

```javascript
function debounce(fn, delay, options = {}) {
    let timer = null;
    const { immediate = false } = options;
    
    const debounced = function(...args) {
        // 清除之前的定时器
        if (timer) clearTimeout(timer);
        
        if (immediate) {
            // 立即执行模式
            const callNow = !timer;
            
            timer = setTimeout(() => {
                timer = null;
            }, delay);
            
            if (callNow) {
                fn.apply(this, args);
            }
        } else {
            // 延迟执行模式
            timer = setTimeout(() => {
                fn.apply(this, args);
            }, delay);
        }
    };
    
    // 取消功能
    debounced.cancel = function() {
        if (timer) {
            clearTimeout(timer);
            timer = null;
        }
    };
    
    return debounced;
}

// 使用示例
const handleSearch = debounce((keyword) => {
    console.log('搜索:', keyword);
}, 300, { immediate: true });

// 连续调用只执行一次
handleSearch('a');
handleSearch('ab');
handleSearch('abc');
// 输出: 搜索: abc (300ms后，如果是立即模式则立即输出)
```

### 3.2 节流函数

**题目要求**：实现节流函数，限制函数执行频率。

```javascript
function throttle(fn, delay) {
    let lastTime = 0;
    let timer = null;
    
    const throttled = function(...args) {
        const now = Date.now();
        const remaining = delay - (now - lastTime);
        
        if (remaining <= 0) {
            // 到了执行时间
            if (timer) {
                clearTimeout(timer);
                timer = null;
            }
            lastTime = now;
            fn.apply(this, args);
        } else if (!timer) {
            // 设置定时器执行
            timer = setTimeout(() => {
                lastTime = Date.now();
                timer = null;
                fn.apply(this, args);
            }, remaining);
        }
    };
    
    // 取消功能
    throttled.cancel = function() {
        if (timer) {
            clearTimeout(timer);
            timer = null;
        }
    };
    
    return throttled;
}

// 使用示例
const handleScroll = throttle(() => {
    console.log('滚动位置:', window.scrollY);
}, 100);

window.addEventListener('scroll', handleScroll);
```

### 3.3 函数柯里化

**题目要求**：实现通用柯里化函数。

```javascript
function curry(fn) {
    return function curried(...args) {
        // 如果参数足够，执行原函数
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        }
        
        // 参数不够，返回接收剩余参数的函数
        return function(...args2) {
            return curried.apply(this, args.concat(args2));
        };
    };
}

// 测试

    return a + b + c;
}

curriedAdd(1, 2)(3); // 6
curriedAdd(1)(2, 3); // 6
```

## 四、深拷贝与数据类型

### 4.1 基础版深拷贝

**题目要求**：实现深拷贝函数，处理基本数据类型和对象。

```javascript
function deepClone(target) {
    // 处理原始类型和null
    if (target === null || typeof target !== 'object') {
        return target;
    }
    
    // 处理Date
    if (target instanceof Date) {
        return new Date(target);
    }
    
    // 处理RegExp
    if (target instanceof RegExp) {
        return new RegExp(target.source, target.flags);
    }
    
    // 处理数组和对象
    const cloneTarget = Array.isArray(target) ? [] : {};
    
    for (const key in target) {
        if (target.hasOwnProperty(key)) {
            cloneTarget[key] = deepClone(target[key]);
        }
    }
    
    return cloneTarget;
}

// 测试
const obj = {
    name: 'Tom',
    age: 25,
    date: new Date(),
    hobby: ['reading', 'coding'],
    address: { city: 'Beijing' }
};

const cloned = deepClone(obj);
console.log(cloned);
console.log(cloned.date instanceof Date); // true
console.log(cloned.hobby === obj.hobby); // false
```

### 4.2 进阶版深拷贝（处理循环引用）

**题目要求**：在深拷贝中处理循环引用。

```javascript
function deepCloneAdvanced(target, cache = new WeakMap()) {
    // 处理原始类型
    if (target === null || typeof target !== 'object') {
        return target;
    }
    
    // 处理Date
    if (target instanceof Date) {
        return new Date(target);
    }
    
    // 处理RegExp
    if (target instanceof RegExp) {
        return new RegExp(target.source, target.flags);
    }
    
    // 处理循环引用
    if (cache.has(target)) {
        return cache.get(target);
    }
    
    // 创建新对象/数组
    const cloneTarget = Array.isArray(target) ? [] : {};
    cache.set(target, cloneTarget);
    
    // 遍历所有属性
    for (const key in target) {
        if (target.hasOwnProperty(key)) {
            cloneTarget[key] = deepCloneAdvanced(target[key], cache);
        }
    }
    
    return cloneTarget;
}

obj.self = obj; // 循环引用
```

### 5.1 事件委托

**题目要求**：实现事件委托函数。

**考察点**：DOM事件模型、事件委托原理、性能优化。

```javascript
function delegate(parent, selector, event, handler) {
    parent.addEventListener(event, function(e) {
        // 获取事件目标
        const target = e.target;
        
        // 查找匹配选择器的最近祖先元素
        const matchedElement = target.closest(selector);
        
        // 如果找到匹配元素且在父元素内部
        if (matchedElement && parent.contains(matchedElement)) {
            handler.call(matchedElement, e);
        }
    });
}

// 使用示例

delegate(list, 'li', 'click', function(e) {
    console.log('点击了:', this.textContent);
});
```

```javascript
Array.prototype.myMap = function(callback, thisArg) {
    if (typeof callback !== 'function') {
        throw new TypeError(callback + 'is not a function');
    }
    
    const arr = this;
    const result = [];
    
    for (let i = 0; i < arr.length; i++) {

        if (i in arr) {
```

### 6.3 reduce实现

**题目要求**：手写Array.prototype.reduce。

```javascript
Array.prototype.myReduce = function(callback, initialValue) {
    if (typeof callback !== 'function') {
        throw new TypeError(callback + 'is not a function');
    }
    
    const arr = this;
    let accumulator, currentIndex;
    
    // 确定初始值
    if (initialValue !== undefined) {
        accumulator = initialValue;
        currentIndex = 0;
    } else {
        // 找到第一个有效值作为初始值
        for (let i = 0; i < arr.length; i++) {
            if (i in arr) {
                accumulator = arr[i];
                currentIndex = i + 1;
                break;
            }
        }
        
        // 如果没有有效值，抛出错误
        if (currentIndex === undefined) {
            throw new TypeError('Reduce of empty array with no initial value');
        }
    }
    
    // 遍历执行回调
    for (let i = currentIndex; i < arr.length; i++) {
        if (i in arr) {
            accumulator = callback(accumulator, arr[i], i, arr);
        }
    }
    
    return accumulator;
};

// 测试
[1, 2, 3, 4].myReduce((acc, cur) => acc + cur, 0); // 10
[1, 2, 3].myReduce((acc, cur) => acc.concat([cur * 2]), []); // [2, 4, 6]
```

### 7.3 实现LRU缓存

**题目要求**：实现一个LRU（最近最少使用）缓存。

```javascript
class LRUCache {
    constructor(capacity) {
        this.capacity = capacity;
        this.cache = new Map();
    }
    
    get(key) {
        if (!this.cache.has(key)) {
            return -1;
        }
        
        // 将该元素移到末尾（最近使用）
        const value = this.cache.get(key);
        this.cache.delete(key);
        this.cache.set(key, value);
        
        return value;
    }
    
    put(key, value) {
        // 如果key已存在，删除
        if (this.cache.has(key)) {
            this.cache.delete(key);
        } else if (this.cache.size >= this.capacity) {
            // 如果容量已满，删除最旧的（第一个）
            const firstKey = this.cache.keys().next().value;
            this.cache.delete(firstKey);
}
        
        // 添加到末尾
        this.cache.set(key, value);
    }
}

// 测试
const cache = new LRUCache(2);
cache.put(1, 1);
cache.put(2, 2);
cache.get(1); // 1
cache.put(3, 3); // 删除key 2
cache.get(2); // -1
cache.put(4, 4); // 删除key 1
cache.get(1); // -1
cache.get(3); // 3
```

### 8.2 常见错误及避免

