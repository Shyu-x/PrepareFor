┃                                                                  ┃
   
┃  5. 数组/对象：                                                  ┃

6.2.3 JSDoc 注释规范

参考答案：

/**
 * 计算两个数的和
 * @param {number} a - 第一个数
 * @param {number} b - 第二个数
 * @returns {number} 两数之和
 * @example
 * add(1, 2) // returns 3
 */
function add(a, b) {
  return a + b;
}

/**
 * 用户类
 * @class
 * @constructor
 * @param {string} name - 用户名
 * @param {string} email - 用户邮箱
 */
class User {
  /**
   * 创建用户
   * @param {string} name
   * @param {string} email
   */
  constructor(name, email) {
    this.name = name;
    this.email = email;
  }

  /**
   * 获取用户信息
   * @returns {Object} 用户信息对象
   */
  getInfo() {
    return {
      name: this.name,
      email: this.email,
    };
  }
}

  return list.filter(item => item.active);
}, [list]);
