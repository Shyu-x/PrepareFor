第5章 算法与数据结构

4.1 数组

数组是最基本的数据结构，元素在内存中连续存储，支持随机访问。

特点：

  • 随机访问：O(1)
  • 插入/删除：O(n)
  • 动态数组（ArrayList）：插入/删除均摊 O(1)

JavaScript 数组方法：

// 常用方法
const arr = [1, 2, 3];

// 添加元素
arr.push(4);        // 末尾添加
arr.unshift(0);      // 开头添加

// 删除元素
arr.pop();           // 末尾删除
arr.shift();         // 开头删除

// 查找
arr.indexOf(2);
arr.includes(2);
arr.find(x => x > 1);
arr.findIndex(x => x > 1);

// 转换
arr.map(x => x * 2);
arr.filter(x => x > 1);
arr.reduce((acc, x) => acc + x, 0);

// 其他
arr.slice(1, 3);
arr.splice(1, 2, 'a', 'b');
arr.sort((a, b) => a - b);
arr.reverse();

4.2 链表

链表是一种线性数据结构，每个节点包含数据和指向下一个节点的指针。

类型：

  • 单向链表
  • 双向链表
  • 循环链表

实现：

// 单向链表节点
class ListNode {
    constructor(val) {
        this.val = val;
        this.next = null;
    }
}

// 单向链表
class LinkedList {
    constructor() {
        this.head = null;
        this.size = 0;
    }

    // 尾部添加
    append(val) {
        const node = new ListNode(val);
        if (!this.head) {
            this.head = node;
        } else {
            let curr = this.head;
            while (curr.next) {
                curr = curr.next;
            }
            curr.next = node;
        }
        this.size++;
    }

    // 头部添加
    prepend(val) {
        const node = new ListNode(val);
        node.next = this.head;
        this.head = node;
        this.size++;
    }

    // 删除指定值
    remove(val) {
        if (!this.head) return;
        if (this.head.val === val) {
            this.head = this.head.next;
            this.size--;
            return;
        }
        let curr = this.head;
        while (curr.next) {
            if (curr.next.val === val) {
                curr.next = curr.next.next;
                this.size--;
                return;
            }
            curr = curr.next;
        }
    }

    // 查找
    find(val) {
        let curr = this.head;
        while (curr) {
            if (curr.val === val) return curr;
            curr = curr.next;
        }
        return null;
    }
}

第5章 栈与队列

5.1 栈（Stack）

栈是一种后进先出（LIFO）的数据结构。

class Stack {
    constructor() {
        this.items = [];
    }

    push(element) {
        this.items.push(element);
    }

    pop() {
        return this.items.pop();
    }

    peek() {
        return this.items[this.items.length - 1];
    }

    isEmpty() {
        return this.items.length === 0;
    }

    size() {
        return this.items.length;
    }
}

应用场景：

  • 函数调用栈
  • 表达式求值
  • 括号匹配
  • 深度优先搜索

5.2 队列（Queue）

队列是一种先进先出（FIFO）的数据结构。

class Queue {
    constructor() {
        this.items = [];
    }

    enqueue(element) {
        this.items.push(element);
    }

    dequeue() {
        return this.items.shift();
    }

    front() {
        return this.items[0];
    }

    isEmpty() {
        return this.items.length === 0;
    }

    size() {
        return this.items.length;
    }
}

变体：

  • 双端队列（Deque）
  • 优先队列（Priority Queue）

应用场景：

  • 任务调度
  • 广度优先搜索
  • 消息队列

第6章 树与图

6.1 二叉树

二叉树是每个节点最多有两个子树的树结构。

class TreeNode {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

// 二叉树遍历
function preorder(root) {
    if (!root) return [];
    return [root.val, ...preorder(root.left), ...preorder(root.right)];
}

function inorder(root) {
    if (!root) return [];
    return [...inorder(root.left), root.val, ...inorder(root.right)];
}

function postorder(root) {
    if (!root) return [];
    return [...postorder(root.left), ...postorder(root.right), root.val];
}

// 层序遍历
function levelOrder(root) {
    if (!root) return [];
    const result = [];
    const queue = [root];

    while (queue.length) {
        const level = [];
        const levelSize = queue.length;

        for (let i = 0; i < levelSize; i++) {
            const node = queue.shift();
            level.push(node.val);
            if (node.left) queue.push(node.left);
            if (node.right) queue.push(node.right);
        }
        result.push(level);
    }
    return result;
}

6.2 二叉搜索树（BST）

二叉搜索树是一种特殊的二叉树，左子树所有节点值小于根节点，右子树所有节点值大于根节点。

class BST {
    constructor() {
        this.root = null;
    }

    insert(val) {
        const node = new TreeNode(val);
        if (!this.root) {
            this.root = node;
            return;
        }
        let curr = this.root;
        while (true) {
            if (val < curr.val) {
                if (!curr.left) {
                    curr.left = node;
                    break;
                }
                curr = curr.left;
            } else {
                if (!curr.right) {
                    curr.right = node;
                    break;
                }
                curr = curr.right;
            }
        }
    }

    search(val) {
        let curr = this.root;
        while (curr) {
            if (val === curr.val) return curr;
            if (val < curr.val) {
                curr = curr.left;
            } else {
                curr = curr.right;
            }
        }
        return null;
    }
}

6.3 堆

堆是一棵完全二叉树，分为最大堆和最小堆。

class MinHeap {
    constructor() {
        this.heap = [];
    }

    parent(i) {
        return Math.floor((i - 1) / 2);
    }

    left(i) {
        return 2 * i + 1;
    }

    right(i) {
        return 2 * i + 2;
    }

    insert(val) {
        this.heap.push(val);
        this.bubbleUp(this.heap.length - 1);
    }

    bubbleUp(i) {
        while (i > 0) {
            const p = this.parent(i);
            if (this.heap[p] > this.heap[i]) {
                [this.heap[p], this.heap[i]] = [this.heap[i], this.heap[p]];
                i = p;
            } else {
                break;
            }
        }
    }

    extractMin() {
        if (this.heap.length === 0) return null;
        if (this.heap.length === 1) return this.heap.pop();

        const min = this.heap[0];
        this.heap[0] = this.heap.pop();
        this.bubbleDown(0);
        return min;
    }

    bubbleDown(i) {
        while (this.left(i) < this.heap.length) {
            let smallest = this.left(i);
            if (this.right(i) < this.heap.length &&
                this.heap[this.right(i)] < this.heap[smallest]) {
                smallest = this.right(i);
            }
            if (this.heap[i] > this.heap[smallest]) {
                [this.heap[i], this.heap[smallest]] = [this.heap[smallest], this.heap[i]];
                i = smallest;
            } else {
                break;
            }
        }
    }
}

6.4 哈希表

哈希表通过哈希函数将键映射到数组索引，实现 O(1) 查找。

class HashTable {
    constructor(size = 100) {
        this.size = size;
        this.table = new Array(size);
    }

    hash(key) {
        let hash = 0;
        for (let i = 0; i < key.length; i++) {
            hash = (hash + key.charCodeAt(i)) % this.size;
        }
        return hash;
    }

    set(key, value) {
        const index = this.hash(key);
        if (!this.table[index]) {
            this.table[index] = [];
        }
        for (let i = 0; i < this.table[index].length; i++) {
            if (this.table[index][i][0] === key) {
                this.table[index][i][1] = value;
                return;
            }
        }
        this.table[index].push([key, value]);
    }

    get(key) {
        const index = this.hash(key);
        if (!this.table[index]) return undefined;
        for (let [k, v] of this.table[index]) {
            if (k === key) return v;
        }
        return undefined;
    }

    delete(key) {
        const index = this.hash(key);
        if (!this.table[index]) return false;
        for (let i = 0; i < this.table[index].length; i++) {
            if (this.table[index][i][0] === key) {
                this.table[index].splice(i, 1);
                return true;
            }
        }
        return false;
    }
}

6.5 图

图由顶点和边组成，分为有向图和无向图。

// 邻接表表示
class Graph {
    constructor() {
        this.adjacencyList = new Map();
    }

    addVertex(vertex) {
        if (!this.adjacencyList.has(vertex)) {
            this.adjacencyList.set(vertex, []);
        }
    }

    addEdge(v1, v2) {
        this.adjacencyList.get(v1).push(v2);
        this.adjacencyList.get(v2).push(v1);
    }

    // 深度优先搜索
    dfs(start) {
        const visited = new Set();
        const result = [];

        const dfsHelper = (vertex) => {
            visited.add(vertex);
            result.push(vertex);
            for (let neighbor of this.adjacencyList.get(vertex)) {
                if (!visited.has(neighbor)) {
                    dfsHelper(neighbor);
                }
            }
        };

        dfsHelper(start);
        return result;
    }

    // 广度优先搜索
    bfs(start) {
        const visited = new Set();
        const queue = [start];
        const result = [];

        visited.add(start);
        while (queue.length) {
            const vertex = queue.shift();
            result.push(vertex);
            for (let neighbor of this.adjacencyList.get(vertex)) {
                if (!visited.has(neighbor)) {
                    visited.add(neighbor);
                    queue.push(neighbor);
                }
            }
        }

        return result;
    }
}

第7章 递归与回溯

7.1 递归

递归是函数调用自身的编程技术。

// 阶乘
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

// 斐波那契数列
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

// 链表反转
function reverseList(head) {
    if (!head || !head.next) return head;
    const newHead = reverseList(head.next);
    head.next.next = head;
    head.next = null;
    return newHead;
}

7.2 回溯

回溯是一种通过试错来解决问题的算法。

// 全排列
function permutations(nums) {
    const result = [];
    const used = new Set();

    function backtrack(path) {
        if (path.length === nums.length) {
            result.push([...path]);
            return;
        }

        for (let i = 0; i < nums.length; i++) {
            if (used.has(i)) continue;
            path.push(nums[i]);
            used.add(i);
            backtrack(path);
            path.pop();
            used.delete(i);
        }
    }

    backtrack([]);
    return result;
}

// 子集
function subsets(nums) {
    const result = [];

    function backtrack(start, path) {
        result.push([...path]);
        for (let i = start; i < nums.length; i++) {
            path.push(nums[i]);
            backtrack(i + 1, path);
            path.pop();
        }
    }

    backtrack(0, []);
    return result;
}

第8章 动态规划

动态规划是一种通过将问题分解为子问题来解决复杂问题的方法。

// 斐波那契数列（DP）
function fibonacciDP(n) {
    if (n <= 1) return n;
    const dp = new Array(n + 1);
    dp[0] = 0;
    dp[1] = 1;
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

// 背包问题
function knapsack(weights, values, capacity) {
    const n = weights.length;
    const dp = new Array(n + 1).fill(0).map(() => new Array(capacity + 1).fill(0));

    for (let i = 1; i <= n; i++) {
        for (let w = 0; w <= capacity; w++) {
            if (weights[i - 1] <= w) {
                dp[i][w] = Math.max(
                    dp[i - 1][w],
                    dp[i - 1][w - weights[i - 1]] + values[i - 1]
                );
            } else {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    return dp[n][capacity];
}

// 最长公共子序列
function longestCommonSubsequence(s1, s2) {
    const m = s1.length, n = s2.length;
    const dp = new Array(m + 1).fill(0).map(() => new Array(n + 1).fill(0));

    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (s1[i - 1] === s2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }
    return dp[m][n];
}

第9章 高级数据结构

9.1 完全二叉树

完全二叉树是一种效率很高的数据结构，堆就是一种完全二叉树。完全二叉树由满二叉树引出。

定义：若设二叉树的深度为h，除第h层外，其他各层（1~h-1）的结点数都达到最大个数，第h层所有的结点都连续集中在最左边，这就是完全二叉树。

特点：

  • 叶子结点只能出现在最下层和次下层
  • 最下层的叶子结点集中在树的左部
  • 倒数第二层若存在叶子结点，一定在右部连续位置
  • 如果结点度为1，则该结点只有左孩子
  • 同样结点数目的二叉树，完全二叉树深度最小
  • 满二叉树一定是完全二叉树，但完全二叉树不一定是满二叉树

数组存储：完全二叉树通常用数组而不是链表存储

// 完全二叉树的数组表示
// 对于下标i的节点（i从0开始）：
// - 父节点下标：Math.floor((i - 1) / 2)
// - 左孩子下标：2 * i + 1
// - 右孩子下标：2 * i + 2

class CompleteBinaryTree {
    constructor() {
        this.tree = [];
    }

    // 获取父节点索引
    parent(i) {
        return Math.floor((i - 1) / 2);
    }

    // 获取左孩子索引
    leftChild(i) {
        return 2 * i + 1;
    }

    // 获取右孩子索引
    rightChild(i) {
        return 2 * i + 2;
    }

    // 判断是否为完全二叉树（层序遍历判断）
    isCompleteBinaryTree() {
        if (this.tree.length === 0) return true;

        let foundNull = false;
        for (let i = 0; i < this.tree.length; i++) {
            if (this.tree[i] === null) {
                foundNull = true;
            } else if (foundNull) {
                // 如果发现null之后还有非null节点，则不是完全二叉树
                return false;
            }
        }
        return true;
    }
}

完全二叉树的节点数计算：

  • 第i层最多有 2^(i-1) 个节点（i从1开始）
  • 深度为k的完全二叉树最多有 2^k - 1 个节点

9.2 AVL树（平衡二叉搜索树）

AVL树是最先发明的自平衡二叉查找树，由苏联数学家Adelson-Velsky和Landis于1962年提出。

定义：

  • 本身是一棵二叉搜索树
  • 带有平衡条件：每个结点的左右子树的高度之差（平衡因子）绝对值最多为1
  • 平衡因子 = 左子树高度 - 右子树高度

特点：

  • 查找、插入、删除在平均和最坏情况下的时间复杂度都是 O(log n)
  • 任何节点的两个子树的高度最大差别为1

旋转操作：

  • 左旋（LL型）：右子树过高
  • 右旋（RR型）：左子树过高
  • 左右旋（LR型）：先左旋后右旋
  • 右左旋（RL型）：先右旋后左旋
class AVLNode {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
        this.height = 1;
    }
}

class AVLTree {
    getHeight(node) {
        return node ? node.height : 0;
    }

    getBalanceFactor(node) {
        return node ? this.getHeight(node.left) - this.getHeight(node.right) : 0;
    }

    updateHeight(node) {
        node.height = 1 + Math.max(this.getHeight(node.left), this.getHeight(node.right));
    }

    // 右旋
    rightRotate(y) {
        const x = y.left;
        const T2 = x.right;

        x.right = y;
        y.left = T2;

        this.updateHeight(y);
        this.updateHeight(x);

        return x;
    }

    // 左旋
    leftRotate(x) {
        const y = x.right;
        const T2 = y.left;

        y.left = x;
        x.right = T2;

        this.updateHeight(x);
        this.updateHeight(y);

        return y;
    }

    insert(node, val) {
        if (!node) return new AVLNode(val);

        if (val < node.val) {
            node.left = this.insert(node.left, val);
        } else if (val > node.val) {
            node.right = this.insert(node.right, val);
        } else {
            return node;
        }

        this.updateHeight(node);

        const balance = this.getBalanceFactor(node);

        // LL型 - 右旋
        if (balance > 1 && val < node.left.val) {
            return this.rightRotate(node);
        }

        // RR型 - 左旋
        if (balance < -1 && val > node.right.val) {
            return this.leftRotate(node);
        }

        // LR型 - 左右旋
        if (balance > 1 && val > node.left.val) {
            node.left = this.leftRotate(node.left);
            return this.rightRotate(node);
        }

        // RL型 - 右左旋
        if (balance < -1 && val < node.right.val) {
            node.right = this.rightRotate(node.right);
            return this.leftRotate(node);
        }

        return node;
    }
}

9.3 红黑树

红黑树是一种自平衡的二叉查找树，1972年由Rudolf Bayer发明。

五大性质：

  1. 每个节点要么是黑色，要么是红色
  2. 根节点是黑色
  3. 每个叶子节点（NIL）是黑色
  4. 每个红色结点的两个子结点一定都是黑色
  5. 任意一结点到每个叶子结点的路径都包含数量相同的黑结点

与AVL树的比较：

  • AVL树更严格平衡，查找效率更高
  • 红黑树插入删除效率更高（旋转次数少）
  • AVL树适合查找频繁的场景，红黑树适合插入删除频繁的场景
class RBNode {
    constructor(val) {
        this.val = val;
        this.color = 'RED';  // 新插入的节点默认为红色
        this.left = null;
        this.right = null;
        this.parent = null;
    }
}

class RedBlackTree {
    constructor() {
        this.NIL = new RBNode(null);
        this.NIL.color = 'BLACK';
        this.root = this.NIL;
    }

    leftRotate(x) {
        const y = x.right;
        x.right = y.left;
        if (y.left !== this.NIL) {
            y.left.parent = x;
        }
        y.parent = x.parent;
        if (x.parent === null) {
            this.root = y;
        } else if (x === x.parent.left) {
            x.parent.left = y;
        } else {
            x.parent.right = y;
        }
        y.left = x;
        x.parent = y;
    }

    // 插入后的修复操作
    insertFixup(node) {
        while (node.parent && node.parent.color === 'RED') {
            if (node.parent === node.parent.parent?.left) {
                const uncle = node.parent.parent.right;
                if (uncle && uncle.color === 'RED') {
                    node.parent.color = 'BLACK';
                    uncle.color = 'BLACK';
                    node.parent.parent.color = 'RED';
                    node = node.parent.parent;
                } else {
                    if (node === node.parent.right) {
                        node = node.parent;
                        this.leftRotate(node);
                    }
                    node.parent.color = 'BLACK';
                    node.parent.parent.color = 'RED';
                    this.rightRotate(node.parent.parent);
                }
            } else {
                // 对称情况
                const uncle = node.parent.parent?.left;
                if (uncle && uncle.color === 'RED') {
                    node.parent.color = 'BLACK';
                    uncle.color = 'BLACK';
                    node.parent.parent.color = 'RED';
                    node = node.parent.parent;
                } else {
                    if (node === node.parent.left) {
                        node = node.parent;
                        this.rightRotate(node);
                    }
                    node.parent.color = 'BLACK';
                    node.parent.parent.color = 'RED';
                    this.leftRotate(node.parent.parent);
                }
            }
        }
        this.root.color = 'BLACK';
    }
}

9.4 B树与B+树

B树和B+树广泛应用于数据库和文件系统的索引结构。

B树特点：

  • 是一种平衡的多路查找树
  • 每个节点可以有多个子节点
  • 所有关键字分布在整棵树中
  • 非叶子节点也包含数据

B+树特点：

  • 是B树的变体
  • 所有数据都存储在叶子节点
  • 叶子节点之间通过指针连接，形成有序链表
  • 非叶子节点只存储索引（关键字）
  • 更适合范围查询和顺序访问
// B树节点
class BTreeNode {
    constructor(t) {
        this.t = t;  // 最小度数
        this.keys = [];  // 关键字数组
        this.children = [];  // 子节点数组
        this.leaf = true;  // 是否为叶子节点
    }
}

class BTree {
    constructor(t) {
        this.root = new BTreeNode(t);
        this.t = t;
    }

    // 搜索
    search(k, node = this.root) {
        let i = 0;
        while (i < node.keys.length && k > node.keys[i]) {
            i++;
        }
        if (i < node.keys.length && k === node.keys[i]) {
            return { node, index: i };
        }
        if (node.leaf) {
            return null;
        }
        return this.search(k, node.children[i]);
    }

    // 分裂子节点
    splitChild(parent, index) {
        const t = this.t;
        const child = parent.children[index];
        const newNode = new BTreeNode(t);

        // 将child的后t-1个关键字移动到newNode
        newNode.keys = child.keys.splice(t);
        // 将child的后t个子节点移动到newNode
        if (!child.leaf) {
            newNode.children = child.children.splice(t);
            newNode.leaf = false;
        }

        // 将child的第t个关键字提升到父节点
        parent.keys.splice(index, 0, child.keys[t - 1]);
        child.keys.splice(t - 1, 1);

        // 将newNode插入父节点的children
        parent.children.splice(index + 1, 0, newNode);
    }

    // 插入
    insert(k) {
        const root = this.root;
        if (root.keys.length === 2 * this.t - 1) {
            const newRoot = new BTreeNode(this.t);
            newRoot.children.push(root);
            newRoot.leaf = false;
            this.splitChild(newRoot, 0);
            this.root = newRoot;
            this.insertNonFull(newRoot, k);
        } else {
            this.insertNonFull(root, k);
        }
    }

    insertNonFull(node, k) {
        let i = node.keys.length - 1;
        if (node.leaf) {
            while (i >= 0 && k < node.keys[i]) {
                i--;
            }
            node.keys.splice(i + 1, 0, k);
        } else {
            while (i >= 0 && k < node.keys[i]) {
                i--;
            }
            i++;
            if (node.children[i].keys.length === 2 * this.t - 1) {
                this.splitChild(node, i);
                if (k > node.keys[i]) {
                    i++;
                }
            }
            this.insertNonFull(node.children[i], k);
        }
    }
}

9.5 线段树

线段树是一种高效处理区间查询和动态修改问题的数据结构。

特点：

  • 每次查询和更新时间复杂度：O(log n)
  • 构建时间复杂度：O(n)
  • 空间复杂度：O(n)
class SegmentTree {
    constructor(arr) {
        this.n = arr.length;
        this.tree = new Array(4 * this.n);
        this.build(arr, 0, 0, this.n - 1);
    }

    build(arr, node, start, end) {
        if (start === end) {
            this.tree[node] = arr[start];
        } else {
            const mid = Math.floor((start + end) / 2);
            const leftNode = 2 * node + 1;
            const rightNode = 2 * node + 2;

            this.build(arr, leftNode, start, mid);
            this.build(arr, rightNode, mid + 1, end);

            // 区间求和
            this.tree[node] = this.tree[leftNode] + this.tree[rightNode];
        }
    }

    // 区间查询
    query(node, start, end, qL, qR) {
        if (qR < start || qL > end) {
            return 0;  // 不在查询范围内
        }
        if (qL <= start && end <= qR) {
            return this.tree[node];  // 完全在查询范围内
        }

        const mid = Math.floor((start + end) / 2);
        const leftResult = this.query(2 * node + 1, start, mid, qL, qR);
        const rightResult = this.query(2 * node + 2, mid + 1, end, qL, qR);

        return leftResult + rightResult;
    }

    // 单点更新
    update(node, start, end, idx, value) {
        if (start === end) {
            this.tree[node] = value;
            return;
        }

        const mid = Math.floor((start + end) / 2);
        if (idx <= mid) {
            this.update(2 * node + 1, start, mid, idx, value);
        } else {
            this.update(2 * node + 2, mid + 1, end, idx, value);
        }

        this.tree[node] = this.tree[2 * node + 1] + this.tree[2 * node + 2];
    }
}

9.6 树状数组（Fenwick Tree）

树状数组（Fenwick Tree）是一种高效处理前缀和查询及单点更新的数据结构。

核心操作：

  • 更新：O(log n)
  • 前缀和查询：O(log n)
  • 区间和查询：O(log n)
class FenwickTree {
    constructor(size) {
        this.size = size;
        this.tree = new Array(size + 1).fill(0);
    }

    // lowbit函数：获取最低位1
    lowbit(x) {
        return x & (-x);
    }

    // 更新：给位置i增加delta
    update(i, delta) {
        while (i <= this.size) {
            this.tree[i] += delta;
            i += this.lowbit(i);
        }
    }

    // 查询：返回1到i的前缀和
    query(i) {
        let sum = 0;
        while (i > 0) {
            sum += this.tree[i];
            i -= this.lowbit(i);
        }
        return sum;
    }

    // 区间和查询
    rangeQuery(left, right) {
        return this.query(right) - this.query(left - 1);
    }
}

9.7 字典树（Trie）

字典树，又称前缀树或Trie树，是一种用于高效存储和检索字符串的数据结构。

特点：

  • 根节点不包含字符
  • 从根节点到某个节点的路径上的字符连接起来为该节点对应的字符串
  • 每个节点的所有子节点包含的字符都不相同
class TrieNode {
    constructor() {
        this.children = new Map();  // 字符到节点的映射
        this.isEnd = false;  // 是否为单词结尾
    }
}

class Trie {
    constructor() {
        this.root = new TrieNode();
    }

    // 插入单词
    insert(word) {
        let node = this.root;
        for (const char of word) {
            if (!node.children.has(char)) {
                node.children.set(char, new TrieNode());
            }
            node = node.children.get(char);
        }
        node.isEnd = true;
    }

    // 搜索单词
    search(word) {
        const node = this.searchPrefix(word);
        return node !== null && node.isEnd;
    }

    // 搜索前缀
    startsWith(prefix) {
        return this.searchPrefix(prefix) !== null;
    }

    searchPrefix(prefix) {
        let node = this.root;
        for (const char of prefix) {
            if (!node.children.has(char)) {
                return null;
            }
            node = node.children.get(char);
        }
        return node;
    }

    // 获取所有以prefix为前缀的单词
    getWordsWithPrefix(prefix) {
        const result = [];
        const node = this.searchPrefix(prefix);
        if (!node) return result;

        const dfs = (currNode, currentWord) => {
            if (currNode.isEnd) {
                result.push(currentWord);
            }
            for (const [char, childNode] of currNode.children) {
                dfs(childNode, currentWord + char);
            }
        };

        dfs(node, prefix);
        return result;
    }
}

应用场景：

  • 字符串检索
  • 前缀匹配
  • 自动补全
  • IP路由（最长前缀匹配）

9.8 单调栈

单调栈是一种特殊的栈，保持栈内元素单调递增或递减。

应用场景：

  • 下一个更大/更小元素
  • 股票价格跨度
  • 每日温度
  • 去除重复字母
// 单调递减栈 - 寻找下一个更大元素
function nextGreaterElement(nums) {
    const n = nums.length;
    const result = new Array(n).fill(-1);
    const stack = [];  // 存储索引

    for (let i = 0; i < n; i++) {
        // 当前元素大于栈顶元素，找到下一个更大元素
        while (stack.length && nums[i] > nums[stack[stack.length - 1]]) {
            const idx = stack.pop();
            result[idx] = nums[i];
        }
        stack.push(i);
    }

    return result;
}

// 单调递增栈 - 寻找下一个更小元素
function nextSmallerElement(nums) {
    const n = nums.length;
    const result = new Array(n).fill(-1);
    const stack = [];

    for (let i = 0; i < n; i++) {
        while (stack.length && nums[i] < nums[stack[stack.length - 1]]) {
            const idx = stack.pop();
            result[idx] = nums[i];
        }
        stack.push(i);

}

// 柱状图中最大的矩形
function largestRectangleArea(heights) {
    const n = heights.length;
    const stack = [];
    let maxArea = 0;

    for (let i = 0; i <= n; i++) {
        const h = i === n ? 0 : heights[i];
        while (stack.length && h < heights[stack[stack.length - 1]]) {
            const height = heights[stack.pop()];
            const width = stack.length ? i - stack[stack.length - 1] - 1 : i;
            maxArea = Math.max(maxArea, height * width);
        }
        stack.push(i);
    }

    return maxArea;
}

9.9 单调队列

单调队列是一种特殊的队列，常用于滑动窗口最大值问题。

// 滑动窗口最大值
function maxSlidingWindow(nums, k) {
    const result = [];
    const deque = [];  // 存储索引

    for (let i = 0; i < nums.length; i++) {
        // 移除超出窗口的索引
        while (deque.length && deque[0] <= i - k) {
            deque.shift();
        }

        // 移除比当前元素小的元素（保持递减）
        while (deque.length && nums[deque[deque.length - 1]] <= nums[i]) {
            deque.pop();
        }

        deque.push(i);

        // 记录窗口最大值
        if (i >= k - 1) {
            result.push(nums[deque[0]]);
        }
    }

    return result;
}

9.10 并查集（Union-Find）

并查集是一种高效处理集合合并和查询的数据结构。

核心操作：

  • 查询（find）：查找元素所属集合
  • 合并（union）：合并两个集合
  • 路径压缩：优化查询效率
  • 按秩合并：优化合并效率
class UnionFind {
    constructor(n) {
        this.parent = new Array(n);
        this.rank = new Array(n);  // 按秩合并
        for (let i = 0; i < n; i++) {
            this.parent[i] = i;
            this.rank[i] = 0;
        }
    }

    // 查找（带路径压缩）
    find(x) {
        if (this.parent[x] !== x) {
            this.parent[x] = this.find(this.parent[x]);  // 路径压缩
        }
        return this.parent[x];
    }

    // 按秩合并
    union(x, y) {
        const rootX = this.find(x);
        const rootY = this.find(y);

        if (rootX === rootY) return;

        if (this.rank[rootX] < this.rank[rootY]) {
            this.parent[rootX] = rootY;
        } else if (this.rank[rootX] > this.rank[rootY]) {
            this.parent[rootY] = rootX;
        } else {
            this.parent[rootY] = rootX;
            this.rank[rootX]++;
        }
    }

    // 判断是否连通
    connected(x, y) {
        return this.find(x) === this.find(y);
    }
}

应用场景：

  • 朋友圈
  • 岛屿数量
  • 连通分量
  • Kruskal最小生成树算法

9.11 布隆过滤器（Bloom Filter）

布隆过滤器是一种空间效率极高的概率型数据结构，用于判断一个元素是否在集合中。

特点：

  • 空间效率高
  • 查询时间 O(k)，k为哈希函数个数
  • 可能存在误判（可能把不存在的元素判断为存在）
  • 不存在误判（判断存在可能不存在，但判断不存在一定不存在）
class BloomFilter {
    constructor(size = 100000, hashCount = 3) {
        this.size = size;
        this.hashCount = hashCount;
        this.bitArray = new Array(size).fill(false);
    }

    // 简单的哈希函数
    hash(value, seed) {
        let hash = 0;
        const str = String(value);
        for (let i = 0; i < str.length; i++) {
            hash = (hash * 31 + str.charCodeAt(i) + seed) % this.size;
        }
        return hash;
    }

    // 添加元素
    add(value) {
        for (let i = 0; i < this.hashCount; i++) {
            const index = this.hash(value, i);
            this.bitArray[index] = true;
        }
    }

    // 判断元素是否存在
    contains(value) {
        for (let i = 0; i < this.hashCount; i++) {
            const index = this.hash(value, i);
            if (!this.bitArray[index]) {
                return false;
            }
        }
        return true;
    }
}

应用场景：

  • 缓存穿透问题
  • 垃圾邮件过滤
  • URL去重
  • 大数据去重

9.12 LRU缓存（最近最少使用）

LRU是一种常用的缓存淘汰策略。

class LRUCache {
    constructor(capacity) {
        this.capacity = capacity;
        this.cache = new Map();
    }

    get(key) {
        if (!this.cache.has(key)) {
            return -1;
        }
        // 移到末尾（最近使用）
        const value = this.cache.get(key);
        this.cache.delete(key);
        this.cache.set(key, value);
        return value;
    }

    put(key, value) {
        if (this.cache.has(key)) {
            this.cache.delete(key);
        } else if (this.cache.size >= this.capacity) {
            // 删除最旧的（第一个）
            const firstKey = this.cache.keys().next().value;
            this.cache.delete(firstKey);
        }
        this.cache.set(key, value);
    }
}

// 使用双向链表+哈希表实现（更高效）
class LRUCache2 {
    constructor(capacity) {
        this.capacity = capacity;
        this.cache = new Map();

        // 伪头部和伪尾部
        this.head = { key: null, value: null, prev: null, next: null };
        this.tail = { key: null, value: null, prev: this.head, next: null };
        this.head.next = this.tail;
    }

    // 移动到末尾
    moveToTail(key) {
        const node = this.cache.get(key);
        // 移除节点
        node.prev.next = node.next;
        node.next.prev = node.prev;
        // 插入尾部
        node.prev = this.tail.prev;
        node.next = this.tail;
        this.tail.prev.next = node;
        this.tail.prev = node;
    }

    get(key) {
        if (!this.cache.has(key)) {
            return -1;
        }
        this.moveToTail(key);
        return this.cache.get(key).value;
    }

    put(key, value) {
        if (this.cache.has(key)) {
            this.cache.get(key).value = value;
            this.moveToTail(key);
        } else {
            if (this.cache.size >= this.capacity) {
                const first = this.head.next;
                this.cache.delete(first.key);
                this.head.next = first.next;
                first.next.prev = this.head;
            }
            const node = { key, value, prev: this.tail.prev, next: this.tail };
            this.cache.set(key, node);
            this.tail.prev.next = node;
            this.tail.prev = node;
        }
    }
}

10.1 Dijkstra最短路径算法

// 迪杰斯特拉算法 - 单源最短路径
function dijkstra(graph, start) {
    const dist = new Array(graph.length).fill(Infinity);
    const visited = new Array(graph.length).fill(false);
    const parent = new Array(graph.length).fill(-1);

    dist[start] = 0;

    for (let i = 0; i < graph.length; i++) {
        // 找到未访问的最小距离节点
        let minDist = Infinity;
        let minNode = -1;
        for (let j = 0; j < graph.length; j++) {
            if (!visited[j] && dist[j] < minDist) {
                minDist = dist[j];
                minNode = j;
            }
        }

        if (minNode === -1) break;
        visited[minNode] = true;

        // 更新邻居节点的距离
        for (let j = 0; j < graph.length; j++) {
            if (graph[minNode][j] > 0 && !visited[j]) {
                const newDist = dist[minNode] + graph[minNode][j];
                if (newDist < dist[j]) {
                    dist[j] = newDist;
                    parent[j] = minNode;
                }
            }
        }
    }

    return { dist, parent };
}

// 重建最短路径
function getPath(parent, start, end) {
    const path = [];
    let curr = end;
    while (curr !== -1) {
        path.unshift(curr);
        curr = parent[curr];
    }
    return path[0] === start ? path : [];
}

10.2 拓扑排序

拓扑排序用于处理有向无环图（DAG）的排序。

// Kahn算法 - 基于入度
function topologicalSortKahn(numCourses, prerequisites) {
    const graph = new Array(numCourses).fill(null).map(() => []);
    const inDegree = new Array(numCourses).fill(0);

    for (const [course, prereq] of prerequisites) {
        graph[prereq].push(course);
        inDegree[course]++;
    }

    const queue = [];
    const result = [];

    // 入度为0的节点入队
    for (let i = 0; i < numCourses; i++) {
        if (inDegree[i] === 0) {
            queue.push(i);
        }
    }

    while (queue.length) {
        const node = queue.shift();
        result.push(node);

        for (const next of graph[node]) {
            inDegree[next]--;
            if (inDegree[next] === 0) {
                queue.push(next);
            }
        }
    }

    return result.length === numCourses ? result : [];
}

// DFS拓扑排序
function topologicalSortDFS(numCourses, prerequisites) {
    const graph = new Array(numCourses).fill(null).map(() => []);
    const visited = new Array(numCourses).fill(0);  // 0:未访问 1:访问中 2:已完成
    const result = [];

    for (const [course, prereq] of prerequisites) {
        graph[prereq].push(course);
    }

    const dfs = (node) => {
        if (visited[node] === 1) return false;  // 有环
        if (visited[node] === 2) return true;

        visited[node] = 1;
        for (const next of graph[node]) {
            if (!dfs(next)) return false;
        }
        visited[node] = 2;
        result.push(node);
        return true;
    };

    for (let i = 0; i < numCourses; i++) {
        if (visited[i] === 0 && !dfs(i)) {
            return [];
        }
    }

    return result.reverse();
}
