1. CSS 盒模型

面试官问：CSS 盒模型是什么？标准的和 IE 的有啥不一样？

标准盒模型：

  • box-sizing: content-box（默认值）
  • 元素 width/height 只包含内容区域
  • 总宽度 = width + padding + border + margin

IE 盒模型：

  • box-sizing: border-box
  • 元素 width/height 包含内容 + padding + border
  • 总宽度 = width + margin
/* 标准盒模型 */
.box-standard {
  box-sizing: content-box;
  width: 100px;
  padding: 10px;
  border: 5px solid #333;
  /* 实际占用宽度 = 100 + 20 + 10 = 130px */
}

/* IE 盒模型 */
.box-ie {
  box-sizing: border-box;
  width: 100px;
  padding: 10px;
  border: 5px solid #333;
  /* 内容区域 = 100 - 20 - 10 = 70px */
}

实际开发建议：

/* 全局设置盒模型 */
* {
  box-sizing: border-box;
}

2. React useEffect

面试官问：React 里那个 useEffect，依赖数组空着、不写、写具体值的时候，效果不一样吧？还问为啥依赖不能漏写。

依赖数组的三种情况：

// 1. 不写依赖数组 - 每次渲染都执行
useEffect(() => {
  console.log('每次渲染都执行');
});

// 2. 空数组 - 只在挂载时执行一次
useEffect(() => {
  console.log('只在挂载时执行');
  // 相当于 componentDidMount
}, []);

// 3. 数组中有值 - 依赖变化时执行
useEffect(() => {
  console.log('userId 变化时执行');
}, [userId]);

依赖不能漏写的原因：

// ❌ 错误：漏写依赖，导致闭包陷阱
function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    // 这里的 timer 会记住创建时的 count 值（0）
    const timer = setInterval(() => {
      console.log(count); // 永远是 0
    }, 1000);
    // 没有返回清理函数，导致内存泄漏
  }, []); // 缺少 count 依赖
}

// ✅ 正确：添加依赖
function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      // 使用函数式更新，不依赖外部 count 值
      setCount(c => c + 1);
    }, 1000);

    // ✅ 返回清理函数
    return () => clearInterval(timer);
  }, []); // 或者不需要依赖，因为使用了函数式更新
}

常见 useEffect 问题：

问题 原因 解决方案
无限循环 依赖是对象/数组，引用变化 使用 useMemo 或 useCallback
内存泄漏 没有清理定时器/订阅 返回清理函数
闭包陷阱 依赖项缺失 使用函数式更新

3. JavaScript 闭包

面试官问：JS 闭包是什么？能用到哪些地方？用的时候要注意什么问题？

闭包定义：

闭包是指能够访问自由变量的函数。自由变量是指在函数中使用的，但既不是函数参数也不是函数局部变量的变量。

// 经典闭包示例
function outer() {
  const a = 1;

  function inner() {
    console.log(a); // inner 访问了 outer 的变量 a
  }

  return inner;
}

const fn = outer(); // fn 就是一个闭包
fn(); // 输出 1

闭包的三大作用：

// 1. 封装私有变量
function createCounter() {
  let count = 0; // 私有变量

  return {
    getCount() {
      return count;
    },
    increment() {
      count++;
    },
    decrement() {
      count--;
    }
  };
}

const counter = createCounter();
counter.increment();
console.log(counter.getCount()); // 1

// 2. 函数柯里化
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function(...args2) {
      return curried.apply(this, args.concat(args2));
    };
  };
}

const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);
curriedAdd(1)(2)(3); // 6

// 3. 模拟块级作用域
for (var i = 0; i < 3; i++) {
  // 使用闭包保存 i 的值
  (function(j) {
    setTimeout(() => console.log(j), 100);
  })(i); // 输出 0, 1, 2
}

// 或者使用 let
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100); // 输出 0, 1, 2
}

使用闭包需要注意的问题：

// 1. 内存泄漏
function badExample() {
  const largeData = new Array(1000000).fill('*');
  return function() {
    console.log(largeData);
  };
}
// largeData 不会被释放，导致内存泄漏

// 2. this 指向问题
const obj = {
  name: 'obj',
  getName() {
    // ❌ 错误：闭包中的 this 不指向 obj
    return function() {
      console.log(this.name); // undefined
    };
  },
  getNameCorrect() {
    // ✅ 正确：保存 this 引用
    const that = this;
    return function() {
      console.log(that.name); // obj
    };
  },
  getNameArrow() {
    // ✅ 正确：使用箭头函数
    return () => {
      console.log(this.name); // obj
    };
  }
};

// 3. 变量提升问题
function example() {
  console.log(a); // undefined（变量提升）
  if (false) {
    var a = 1; // var 提升到函数顶部
  }
}

4. 前端跨域解决方案

面试官问：前端跨域有哪些解决办法？每种咋回事，适合什么情况？

跨域原因：浏览器同源策略（Same-Origin Policy）限制

七种解决方案：

// 1. CORS（跨域资源共享）- 推荐
// 后端设置响应头
// Access-Control-Allow-Origin: http://example.com
// Access-Control-Allow-Methods: GET, POST, PUT, DELETE
// Access-Control-Allow-Headers: Content-Type, Authorization
// Access-Control-Allow-Credentials: true

// 前端简单请求
fetch('http://api.example.com/data', {
  credentials: 'include' // 携带 cookie
});

// 2. JSONP（只支持 GET）
function jsonp(url, callback) {
  const script = document.createElement('script');
  script.src = `${url}?callback=${callback}`;
  document.body.appendChild(script);
}

function handleResponse(data) {
  console.log(data);
}

jsonp('http://api.example.com/data', 'handleResponse');

// 3. postMessage
// 父窗口
window.postMessage({ message: 'hello' }, 'http://child.com');

// 子窗口
window.addEventListener('message', (event) => {
  if (event.origin === 'http://parent.com') {
    console.log(event.data);
  }
});

// 4. WebSocket
const ws = new WebSocket('ws://api.example.com');

// 5. nginx 反向代理
// nginx.conf
// server {
//   location /api/ {
//     proxy_pass http://api.example.com/;
//   }
// }

// 6. node 中间代理
// server.js
const express = require('express');
const app = express();

app.get('/api/data', (req, res) => {
  const result = await fetch('http://api.example.com/data');
  const data = await result.json();
  res.json(data);
});

// 7. document.domain（主域名相同，子域名不同）
// a.example.com 和 b.example.com
document.domain = 'example.com';

方案对比：

方案 优点 缺点 适用场景
CORS 支持所有请求 需要后端配合 前后端都可控
JSONP 兼容性好 只支持 GET 兼容老浏览器
postMessage 安全 语法复杂 iframe 通信
WebSocket 双向通信 需要服务器支持 实时通信
nginx 简单 需要 nginx 配置 前端可控
node 中间件 灵活 增加复杂度 前端可控

5. Vue3 响应式原理

面试官问：Vue3 和 Vue2 的响应式原理不一样吧？为什么 Vue3 用 Proxy 不用 Object.defineProperty 了？

Vue2 响应式（Object.defineProperty）：

// Vue2 源码简化版
function defineReactive(obj, key, val) {
  // 递归监听子属性
  if (typeof val === 'object') {
    new Observer(val);
  }

  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get() {
      // 依赖收集
      if (Dep.target) {
        dep.depend();
      }
      return val;
    },
    set(newVal) {
      if (newVal === val) return;
      // 通知更新
      val = newVal;
      dep.notify();
    }
  });
}

Vue2 的问题：

// 1. 无法监听新增属性
const obj = {};
new Vue({ data: { obj } });

obj.newProp = 'new'; // ❌ 不触发响应式

// 解决方案：Vue.set(obj, 'newProp', 'new')

// 2. 无法监听数组变化
const arr = [];
new Vue({ data: { arr } });

arr.push(1); // ❌ 不触发响应式
arr[0] = 1; // ❌ 不触发响应式

// 解决方案：Vue.set(arr, 0, 1) 或 Vue.set(arr, 'length', arr.length)

// 3. 需要递归遍历所有属性

Vue3 响应式（Proxy）：

// Vue3 响应式简化实现
function reactive(target) {
  return new Proxy(target, {
    get(target, key, receiver) {
      // 依赖收集
      track(target, key);
      const value = Reflect.get(target, key, receiver);
      // 深度响应式
      if (typeof value === 'object' && value !== null) {
        return reactive(value);
      }
      return value;
    },
    set(target, key, value, receiver) {
      const oldValue = Reflect.get(target, key, receiver);
      const result = Reflect.set(target, key, value, receiver);
      // 触发更新
      if (oldValue !== value) {
        trigger(target, key);
      }
      return result;
    },
    deleteProperty(target, key) {
      const result = Reflect.deleteProperty(target, key);
      if (result) {
        trigger(target, key);
      }
      return result;
    },
    has(target, key) {
      track(target, key);
      return Reflect.has(target, key);
    },
    ownKeys(target) {
      track(target, Symbol.iterator);
      return Reflect.ownKeys(target);
    }
  });
}

Vue3 Proxy 的优势：

特性 Vue2 (defineProperty) Vue3 (Proxy)
新增属性 需用 Vue.set 自动响应
删除属性 需用 Vue.delete 自动响应
数组索引 无法监听 自动响应
数组方法 重写数组方法 无需重写
性能 递归遍历，性能差 懒代理，性能好
浏览器支持 IE11+ 现代浏览器 
// Vue3 响应式的使用
import { reactive, ref, computed } from 'vue';

// ref - 基础类型
const count = ref(0);
count.value++; // 需要 .value

// reactive - 对象
const state = reactive({
  user: { name: '张三' }
});
state.user.name = '李四'; // 直接修改

// computed - 计算属性
const doubleCount = computed(() => count.value * 2);

// watch - 监听
watch(count, (newVal, oldVal) => {
  console.log(`count 从 ${oldVal} 变为 ${newVal}`);
});

6. 浏览器回流和重绘

面试官问：浏览器的回流和重绘是啥意思？怎么能少发生点？

定义：

  • 回流（Reflow）：当 DOM 的几何属性（位置、大小）发生变化时，浏览器需要重新计算布局
  • 重绘（Repaint）：当 DOM 的外观（颜色、背景色）发生变化时，浏览器重新绘制
// 触发回流的操作
element.style.width = '100px'; // 宽度变化
element.style.height = '100px'; // 高度变化
element.style.padding = '10px'; // 内边距变化
element.style.margin = '10px'; // 外边距变化
element.style.display = 'none'; // 显示隐藏
element.appendChild(child); // 添加子元素
element.innerHTML = '...'; // 修改内容

// 只触发重绘的操作
element.style.color = 'red'; // 颜色变化
element.style.backgroundColor = 'blue'; // 背景色变化
element.style.visibility = 'hidden'; // 隐藏但占空间

优化策略：

// 1. 批量修改 DOM
// ❌ 多次回流
for (let i = 0; i < 10; i++) {
  div.style.width = (i * 10) + 'px';
}

// ✅ 一次性修改
div.style.cssText = 'width: 100px; height: 100px;';
div.classList.add('new-class');

// 2. 使用 CSS transform 代替 top/left
// ❌ 触发回流
element.style.top = '100px';
element.style.left = '100px';

// ✅ 只触发合成
element.style.transform = 'translate(100px, 100px)';

// 3. 使用 documentFragment
// ❌ 多次回流
for (let i = 0; i < 100; i++) {
  ul.innerHTML += `<li>${i}</li>`;
}

// ✅ 一次回流
const fragment = document.createDocumentFragment();
for (let i = 0; i < 100; i++) {
  const li = document.createElement('li');
  li.textContent = i;
  fragment.appendChild(li);
}
ul.appendChild(fragment);

// 4. 缓存布局信息
// ❌ 每次读取都触发回流
const width = div.offsetWidth; // 触发
div.style.width = '100px';
const width2 = div.offsetWidth; // 再次触发

// ✅ 缓存后使用
const width = div.offsetWidth;
div.style.width = (width * 2) + 'px';

// 5. 使用 will-change
.element {
  will-change: transform; // 提示浏览器提前优化
}

// 6. 使用 CSS 变量
:root {
  --primary-color: blue;
}
.element {
  color: var(--primary-color); // 只触发重绘
}

7. Promise 状态与方法

面试官问：Promise 有三种状态是什么？还有 all、race、allSettled 这几个，区别和用在哪？

Promise 三种状态：

// pending - 进行中
// fulfilled - 已成功
// rejected - 已失败

const promise = new Promise((resolve, reject) => {
  // pending 状态
  setTimeout(() => {
    resolve('成功'); // 变为 fulfilled
    // 或 reject('失败'); // 变为 rejected
  }, 1000);
});

promise.then(result => {
  console.log(result); // '成功'
}).catch(error => {
  console.error(error);
});

Promise.all：

// 所有 Promise 都成功才成功，有一个失败就失败
const p1 = Promise.resolve(1);
const p2 = Promise.resolve(2);
const p3 = Promise.resolve(3);

Promise.all([p1, p2, p3])
  .then(values => console.log(values)); // [1, 2, 3]

// 有一个失败
const p4 = Promise.reject('error');
Promise.all([p1, p4, p3])
  .then(values => console.log(values))
  .catch(error => console.log(error)); // 'error'

Promise.race：

// 返回最先完成（无论成功或失败）的结果
const p1 = new Promise(r => setTimeout(() => r('p1'), 100));
const p2 = new Promise(r => setTimeout(() => r('p2'), 50));
const p3 = new Promise(r => setTimeout(() => r('p3'), 80));

Promise.race([p1, p2, p3])
  .then(value => console.log(value)); // 'p2'

// 应用：请求超时
function fetchWithTimeout(url, timeout = 3000) {
  return Promise.race([
    fetch(url),
    new Promise((_, reject) =>
      setTimeout(() => reject(new Error('timeout')), timeout)
    )
  ]);
}

Promise.allSettled：

// 所有 Promise 完成后返回所有结果（无论成功或失败）
const p1 = Promise.resolve(1);
const p2 = Promise.reject('error');
const p3 = Promise.resolve(3);

Promise.allSettled([p1, p2, p3])
  .then(results => {
    console.log(results);
    // [
    //   { status: 'fulfilled', value: 1 },
    //   { status: 'rejected', reason: 'error' },
    //   { status: 'fulfilled', value: 3 }
    // ]
  });

// 应用：确保所有请求完成后再处理
async function loadAllData() {
  const results = await Promise.allSettled([
    fetch('/api/users'),
    fetch('/api/posts'),
    fetch('/api/comments')
  ]);

  const users = results[0].status === 'fulfilled' ? await results[0].value.json() : [];
  const posts = results[1].status === 'fulfilled' ? await results[1].value.json() : [];
  const comments = results[2].status === 'fulfilled' ? await results[2].value.json() : [];

  return { users, posts, comments };
}

三者对比：

方法 成功条件 失败条件 返回值
all 全部成功 任意一个失败 成功值的数组
race 任意一个完成 无 最先完成的值
allSettled 无 无 全部结果对象

8. Webpack Loader 和 Plugin

面试官问：webpack 里 loader 和 plugin 不一样吧？举几个常用的，说说干啥的。

Loader：

  • 用于处理非 JavaScript 文件（如 CSS、图片、字体）
  • 链式调用，从右到左执行
  • 本质是函数
// webpack.config.js
module.exports = {
  module: {
    rules: [
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader', 'postcss-loader']
      },
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: {
          loader: 'babel-loader',
          options: {
            presets: ['@babel/preset-env']
          }
        }
      },
      {
        test: /\.(png|jpg|gif)$/,
        use: {
          loader: 'file-loader',
          options: {
            name: 'images/[name].[hash:8].[ext]'
          }
        }
      }
    ]
  }
};

常用 Loader：

Loader 功能
babel-loader 转译 ES6+ 代码
ts-loader 处理 TypeScript
css-loader 处理 CSS 文件
style-loader 注入 CSS 到 DOM
postcss-loader PostCSS 处理
sass-loader 处理 Sass/SCSS
file-loader 处理文件资源
url-loader 处理 base64
vue-loader 处理 Vue 组件

Plugin：

  • 用于扩展 Webpack 功能
  • 在构建过程中插入自定义行为
  • 本质是类（拥有 apply 方法）
// 自定义 Plugin 示例
class MyPlugin {
  constructor(options) {
    this.options = options;
  }

  apply(compiler) {
    // 监听钩子
    compiler.hooks.emit.tap('MyPlugin', (compilation) => {
      console.log('构建开始');

      // 生成额外文件
      compilation.assets['custom.txt'] = {
        source: () => '自定义内容',
        size: () => 4
      };
    });

    compiler.hooks.done.tap('MyPlugin', (stats) => {
      console.log('构建完成');
    });
  }
}

module.exports = {
  plugins: [
    new MyPlugin({ option: true }),
    new HtmlWebpackPlugin({
      template: './src/index.html'
    }),
    new MiniCssExtractPlugin({
      filename: '[name].css'
    })
  ]
};

常用 Plugin：

Plugin 功能
HtmlWebpackPlugin 生成 HTML 文件
MiniCssExtractPlugin 提取 CSS 到独立文件
DefinePlugin 定义全局常量
UglifyJsPlugin 压缩 JS 代码
CleanWebpackPlugin 清理输出目录
CopyWebpackPlugin 复制静态文件
HotModuleReplacementPlugin 热更新

Loader vs Plugin 对比：

特性 Loader Plugin
角色 翻译官 扩展器
执行时机 单文件处理 构建生命周期
输入输出 源文件 → 转换文件 编译对象
配置方式 module.rules plugins 数组

9. HTML meta 标签

面试官问：HTML 里 meta 标签有啥用？比如 viewport 那个，作用是啥，一般咋配置？

meta 标签作用：提供元数据信息，不显示在页面上

<!-- 字符编码 -->
<meta charset="UTF-8">

<!-- 页面描述 -->
<meta name="description" content="这是一个前端面试题汇总页面">

<!-- 关键词 -->
<meta name="keywords" content="前端,面试,JavaScript,React">

<!-- 作者 -->
<meta name="author" content="张三">

<!-- 兼容模式 -->
<meta http-equiv="X-UA-Compatible" content="IE=edge,chrome=1">

<!-- 刷新和重定向 -->

<!-- 缓存控制 -->

<!-- 社交媒体分享 (Open Graph) -->

完整深拷贝（考虑循环引用和特殊类型）：

function deepClone(target, map = new WeakMap()) {
  // 1. 处理基本类型和 null
  if (target === null || typeof target !== 'object') {
    return target;
  }

  // 2. 处理日期
  if (target instanceof Date) {
    return new Date(target.getTime());
  }

  // 3. 处理正则
  if (target instanceof RegExp) {
    return new RegExp(target.source, target.flags);
  }

  // 4. 处理函数（一般不深拷贝）
  if (typeof target === 'function') {
    return target;
  }

  // 5. 处理 Map
  if (target instanceof Map) {
    const cloneMap = new Map();
    target.forEach((value, key) => {
      cloneMap.set(key, deepClone(value, map));
    });
    return cloneMap;
  }

  // 6. 处理 Set
  if (target instanceof Set) {
    const cloneSet = new Set();
    target.forEach(value => {
      cloneSet.add(deepClone(value, map));
    });
    return cloneSet;
  }

  // 7. 处理数组
  if (Array.isArray(target)) {
    const cloneArr = [];
    map.set(target, cloneArr);
    target.forEach((item, index) => {
      cloneArr[index] = deepClone(item, map);
    });
    return cloneArr;
  }

  // 8. 处理对象
  const cloneObj = {};
  map.set(target, cloneObj);

  for (const key in target) {
    if (target.hasOwnProperty(key)) {
      // 处理 Symbol 作为 key
      cloneObj[key] = deepClone(target[key], map);
    }
  }

  // 处理 Symbol 作为键名
  const symbolKeys = Object.getOwnPropertySymbols(target);
  symbolKeys.forEach(sym => {
    cloneObj[sym] = deepClone(target[sym], map);
  });

  return cloneObj;
}

  let inThrottle;
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

5.3 深拷贝

function deepClone(target, map = new WeakMap()) {
  if (target === null || typeof target !== 'object') {
    return target;
  }

  if (target instanceof Date) {
    return new Date(target.getTime());
  }

  if (target instanceof RegExp) {
    return new RegExp(target.source, target.flags);
  }

  if (map.get(target)) return map.get(target);

  const clone = Array.isArray(target) ? [] : {};
  map.set(target, clone);

  for (const key in target) {
    if (target.hasOwnProperty(key)) {
      clone[key] = deepClone(target[key], map);
    }
  }

  return clone;
}

class DoublyLinkedList {
  constructor() {
    this.head = null;
    this.tail = null;
    this.length = 0;
  }

  // 追加节点
  append(val) {
    const node = { val, prev: this.tail, next: null };
    if (this.tail) {
      this.tail.next = node;
    } else {
      this.head = node;
    }
    this.tail = node;
    this.length++;
    return this;
  }

  // 头部插入
  prepend(val) {
    const node = { val, prev: null, next: this.head };
    if (this.head) {
      this.head.prev = node;
    } else {
      this.tail = node;
    }
    this.head = node;
    this.length++;
    return this;
  }

  // 删除节点
  delete(val) {
    let current = this.head;
    while (current) {
      if (current.val === val) {
        if (current.prev) current.prev.next = current.next;
        else this.head = current.next;
        if (current.next) current.next.prev = current.prev;
        else this.tail = current.prev;
        this.length--;
        return true;
      }
      current = current.next;
    }
    return false;
  }

  // 查找
  find(val) {
    let current = this.head;
    while (current) {
      if (current.val === val) return current;
      current = current.next;
    }
    return null;
  }
}

10.3 大文件上传

// 分片上传 + 断点续传
class Uploader {
  constructor(file, options = {}) {
    this.file = file;
    this.chunkSize = options.chunkSize || 2 * 1024 * 1024; // 2MB
    this.uploadedChunks = this.loadProgress();
  }

  // 加载本地存储的进度
  loadProgress() {
    const saved = localStorage.getItem(`upload_${this.file.name}`);
    return saved ? JSON.parse(saved) : [];
  }

  // 保存进度
  saveProgress(chunks) {
    localStorage.setItem(`upload_${this.file.name}`, JSON.stringify(chunks));
  }

  // 上传
  async upload() {
    const totalChunks = Math.ceil(this.file.size / this.chunkSize);

    for (let i = 0; i < totalChunks; i++) {
      if (this.uploadedChunks.includes(i)) continue;

      const start = i * this.chunkSize;
      const end = Math.min(start + this.chunkSize, this.file.size);
      const chunk = this.file.slice(start, end);

      await this.uploadChunk(chunk, i);

      this.uploadedChunks.push(i);
      this.saveProgress(this.uploadedChunks);
    }

    await this.mergeChunks();
  }

  async uploadChunk(chunk, index) {
    const formData = new FormData();
    formData.append('chunk', chunk);
    formData.append('index', index);
    formData.append('filename', this.file.name);

    return fetch('/api/upload', {
      method: 'POST',
      body: formData
    });
  }
}

    }

      }
      return true;
    },

      if (!handlers.has(prop)) {
        handlers.set(prop, []);
      }
      handlers.get(prop).push(fn);
    }
  });
}

// 接雨水 - 动态规划
function trap(height) {
  if (!height || height.length === 0) return 0;

  const n = height.length;
  const left = new Array(n);
  const right = new Array(n);

  // 左边最高
  left[0] = height[0];
  for (let i = 1; i < n; i++) {
    left[i] = Math.max(left[i - 1], height[i]);
  }

  // 右边最高
  right[n - 1] = height[n - 1];
  for (let i = n - 2; i >= 0; i--) {
    right[i] = Math.max(right[i + 1], height[i]);
  }

  // 计算积水
  let result = 0;
  for (let i = 0; i < n; i++) {
    result += Math.min(left[i], right[i]) - height[i];
  }

  return result;
}

// 双指针优化
function trap2(height) {
  let left = 0, right = height.length - 1;
  let leftMax = 0, rightMax = 0;
  let result = 0;

  while (left < right) {
    if (height[left] < height[right]) {
      if (height[left] >= leftMax) {
        leftMax = height[left];
      } else {
        result += leftMax - height[left];
      }
      left++;
    } else {
      if (height[right] >= rightMax) {
        rightMax = height[right];
      } else {
        result += rightMax - height[right];
      }
      right--;
    }
  }

  return result;
}

34.3 岛屿数量 (LeetCode 200)

// 岛屿数量 - BFS
function numIslands(grid) {
  if (!grid || grid.length === 0) return 0;

  const rows = grid.length;
  const cols = grid[0].length;
  let count = 0;

  function bfs(r, c) {
    const queue = [[r, c]];
    grid[r][c] = '0';

    while (queue.length) {
      const [row, col] = queue.shift();
      const directions = [[-1, 0], [1, 0], [0, -1], [0, 1]];

      for (const [dr, dc] of directions) {
        const nr = row + dr;
        const nc = col + dc;

        if (nr >= 0 && nr < rows && nc >= 0 && nc < cols && grid[nr][nc] === '1') {
          grid[nr][nc] = '0';
          queue.push([nr, nc]);
        }
      }
    }
  }

  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (grid[i][j] === '1') {
        count++;
        bfs(i, j);
      }
    }
  }

  return count;
}

// DFS
function numIslandsDFS(grid) {
  if (!grid || grid.length === 0) return 0;

  let count = 0;

  function dfs(grid, i, j) {
    const rows = grid.length;
    const cols = grid[0].length;

    if (i < 0 || i >= rows || j < 0 || j >= cols || grid[i][j] === '0') {
      return;
    }

    grid[i][j] = '0';
    dfs(grid, i - 1, j);
    dfs(grid, i + 1, j);
    dfs(grid, i, j - 1);
    dfs(grid, i, j + 1);
  }

  for (let i = 0; i < grid.length; i++) {
    for (let j = 0; j < grid[0].length; j++) {
      if (grid[i][j] === '1') {
        count++;
        dfs(grid, i, j);
      }
    }
  }

  return count;
}

34.4 最大正方形 (LeetCode 221)

// 最大正方形 - 动态规划
function maximalSquare(matrix) {
  if (!matrix || matrix.length === 0) return 0;

  const rows = matrix.length;
  const cols = matrix[0].length;
  let maxSide = 0;

  // dp[i][j] 表示以(i,j)为右下角的最大正方形边长
  const dp = new Array(rows + 1).fill(0).map(() => new Array(cols + 1).fill(0));

  for (let i = 1; i <= rows; i++) {
    for (let j = 1; j <= cols; j++) {
      if (matrix[i - 1][j - 1] === '1') {
        dp[i][j] = Math.min(
          dp[i - 1][j],
          dp[i][j - 1],
          dp[i - 1][j - 1]
        ) + 1;
        maxSide = Math.max(maxSide, dp[i][j]);
      }
    }
  }

  return maxSide * maxSide;
}

// 最长回文子串 - 中心扩展
function longestPalindrome(s) {
  if (!s || s.length < 1) return "";

  let start = 0, end = 0;

  function expandAroundCenter(left, right) {
    while (left >= 0 && right < s.length && s[left] === s[right]) {
      left--;
      right++;
    }
    return right - left - 1;
  }

  for (let i = 0; i < s.length; i++) {
    const len1 = expandAroundCenter(i, i);     // 奇数
    const len2 = expandAroundCenter(i, i + 1); // 偶数
    const len = Math.max(len1, len2);

    if (len > end - start) {
      start = i - Math.floor((len - 1) / 2);
      end = i + Math.floor(len / 2);
    }
  }

  return s.substring(start, end + 1);
}

function findKthLargest(nums, k) {
  return quickSelect(nums, 0, nums.length - 1, nums.length - k);
}

function quickSelect(nums, left, right, k) {
  const pivotIndex = partition(nums, left, right);

  if (pivotIndex === k) {
    return nums[pivotIndex];
  } else if (pivotIndex < k) {
    return quickSelect(nums, pivotIndex + 1, right, k);
  } else {
    return quickSelect(nums, left, pivotIndex - 1, k);
  }
}

function partition(nums, left, right) {
  const pivot = nums[right];
  let i = left;

  for (let j = left; j < right; j++) {
    if (nums[j] <= pivot) {
      [nums[i], nums[j]] = [nums[j], nums[i]];
      i++;
    }
  }

  [nums[i], nums[right]] = [nums[right], nums[i]];
  return i;
}

class EventEmitter {
  private events: Map<string, Function[]> = new Map();

  on(event: string, listener: Function) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event)!.push(listener);
  }

  off(event: string, listener: Function) {
    const listeners = this.events.get(event);
    if (listeners) {
      const index = listeners.indexOf(listener);
      if (index > -1) listeners.splice(index, 1);
    }
  }

  emit(event: string, ...args: any[]) {
    const listeners = this.events.get(event);
    if (listeners) {
      listeners.forEach(listener => listener(...args));
    }
  }

  once(event: string, listener: Function) {
    const wrapper = (...args: any[]) => {
      listener(...args);
      this.off(event, wrapper);
    };
    this.on(event, wrapper);
  }
}

class PubSub {
  private topics: Map<string, Array<{ handler: Function; once?: boolean }>> = new Map();
  private ids: number = 0;

  subscribe(topic: string, handler: Function, once = false): number {
    const id = ++this.ids;
    if (!this.topics.has(topic)) {
      this.topics.set(topic, []);
    }
    this.topics.get(topic)!.push({ handler, once, id });
    return id;
  }

  unsubscribe(topic: string, id: number) {
    const handlers = this.topics.get(topic);
    if (handlers) {
      const index = handlers.findIndex(h => h.id === id);
      if (index > -1) handlers.splice(index, 1);
    }
  }

  publish(topic: string, data?: any) {
    const handlers = this.topics.get(topic);
    if (handlers) {
      handlers.forEach(({ handler, once }) => {
        handler(data);
        if (once) this.unsubscribe(topic, handlers.find(h => h.handler === handler)?.id!);
      });
    }
  }
}

function diff(oldTree, newTree) {
  const patches = [];

  function walk(oldNode, newNode, index = 0) {
    if (!oldNode) {
      // 新增节点
      patches.push({ type: 'INSERT', node: newNode });
    } else if (!newNode) {
      // 删除节点
      patches.push({ type: 'REMOVE', index });
    } else if (oldNode.type !== newNode.type) {
      // 类型不同，替换
      patches.push({ type: 'REPLACE', oldNode, newNode });
    } else if (oldNode.type === 'TEXT') {
      // 文本节点，比较内容
      if (oldNode.content !== newNode.content) {
        patches.push({ type: 'TEXT', content: newNode.content, index });
      }
    } else {
      // 相同类型，比较属性
      const propsPatch = diffProps(oldNode.props, newNode.props);
      if (Object.keys(propsPatch).length > 0) {
        patches.push({ type: 'PROPS', props: propsPatch, index });
      }
      // 递归比较子节点
      diffChildren(oldNode.children, newNode.children);
    }
  }

  return patches;
}

    !Object.is(state[currentIdx], deps);

  if (hasChanged) {
    // 清理旧 effect
    if (state[currentIdx]) {
      state[currentIdx]();
    }
    // 注册新 effect
    state[currentIdx] = callback;
  }

  idx++;  // 保持 useState 和 useEffect 使用不同的 idx
}

// useRef 实现
function useRef(initialValue) {
  return useState({ current: initialValue })[0];
}

// useMemo 实现
function useMemo(factory, deps) {
  const currentIdx = idx;
  const hasChanged = !deps ||
    deps.some((dep, i) => !Object.is(dep, state[currentIdx][i]));

  if (hasChanged || state[currentIdx] === undefined) {
    state[currentIdx] = [factory(), [...deps]];
  }

  return state[currentIdx][0];
}

      document.documentElement;

      loadMore();
    }

    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, [handleScroll]);

    </div>
  );
};

// 使用 Intersection Observer 的版本
export const InfiniteScrollObserver: React.FC<InfiniteScrollProps> = ({
  loadMore,
  hasMore,
  loading,
  children
}) => {
  const loaderRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!hasMore || loading) return;

    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          loadMore();
        }
      },
      { threshold: 0.1 }
    );

    if (loaderRef.current) {
      observer.observe(loaderRef.current);
    }

    return () => observer.disconnect();
  }, [loadMore, hasMore, loading]);

  return (
    <div>
      {children}
      <div ref={loaderRef} className="loader">
        {loading && '加载中...'}
      </div>
    </div>
  );
};

class JSBridge {
  constructor() {
    this.callbacks = new Map();
    this.callbackId = 0;
    this.init();
  }

  init() {
    // 等待 Native 注入
    if (window.JSBridge) {
      window.JSBridge.onMessage = (data) => this.handleMessage(data);
    }

    // 兼容老版本
    window.JSBridge = window.JSBridge || {};

    // 挂载方法到 window
    window.JSBridge.call = (module, method, params) =>
      this.callNative(module, method, params);

    window.JSBridge.on = (event, callback) => this.on(event, callback);
  }

  generateCallbackId() {
    return `cb_${++this.callbackId}_${Date.now()}`;
  }

  callNative(module, method, params = {}) {
    return new Promise((resolve, reject) => {
      const callbackId = this.generateCallbackId();

      this.callbacks.set(callbackId, { resolve, reject });

      // 设置超时
      setTimeout(() => {
        if (this.callbacks.has(callbackId)) {
          this.callbacks.delete(callbackId);
          reject(new Error('Native call timeout'));
        }
      }, 10000);

      // 发送消息给 Native
      const message = {
        module,
        method,
        params,
        callbackId
      };

      // iOS
      if (window.webkit && window.webkit.messageHandlers[module]) {
        window.webkit.messageHandlers[module].postMessage(message);
      }
      // Android
      else if (window.JSBridgeAndroid) {
        window.JSBridgeAndroid.postMessage(JSON.stringify(message));
      }
      // 兜底 URL Scheme
      else {
        const url = `jsbridge://${module}/${method}?${this.encodeParams({ ...params, callbackId })}`;
        this.invokeURLScheme(url);
      }
    });
  }

  handleMessage(data) {
    const { callbackId, result, error } = data;

    if (callbackId && this.callbacks.has(callbackId)) {
      const { resolve, reject } = this.callbacks.get(callbackId);
      this.callbacks.delete(callbackId);

      if (error) {
        reject(new Error(error));
      } else {
        resolve(result);
      }
    }
  }

  on(event, callback) {
    if (!this.events) this.events = new Map();
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event).push(callback);
  }

  emit(event, data) {
    if (this.events && this.events.has(event)) {
      this.events.get(event).forEach(cb => cb(data));
    }
  }

  encodeParams(params) {
    return Object.entries(params)
      .map(([k, v]) => `${k}=${encodeURIComponent(JSON.stringify(v))}`)
      .join('&');
  }

  invokeURLScheme(url) {
    const iframe = document.createElement('iframe');
    iframe.style.display = 'none';
    iframe.src = url;
    document.body.appendChild(iframe);
    setTimeout(() => iframe.remove(), 100);
  }
}

class LRUCache {
  constructor(maxSize = 5) {
    this.maxSize = maxSize;
    this.cache = new Map();
    this.init();
  }

  init() {
    // 从 LocalStorage 恢复
    const stored = localStorage.getItem('lru_cache');
    if (stored) {
      const data = JSON.parse(stored);
      this.cache = new Map(data);
    }
  }

  save() {
    // 保存到 LocalStorage
    const data = Array.from(this.cache.entries());
    try {
      localStorage.setItem('lru_cache', JSON.stringify(data));
    } catch (e) {
      // 空间不足，清理一半
      this.trim(Math.floor(this.maxSize / 2));
      this.save();
    }
  }

  get(key) {
    if (!this.cache.has(key)) return null;

    // 移到末尾（最新）
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);

    return value;
  }

  set(key, value) {
    // 已存在，删除后重新设置
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }

    // 超过最大容量，删除最老的
    if (this.cache.size >= this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    this.cache.set(key, value);
    this.save();
  }

  trim(maxSize) {
    while (this.cache.size > maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    this.save();
  }

  clear() {
    this.cache.clear();
    localStorage.removeItem('lru_cache');
  }
}

class StorageManager {
  constructor(options = {}) {
    this.storage = options.storage || localStorage;
    this.maxSize = options.maxSize || 4 * 1024 * 1024; // 4MB
    this.warningThreshold = options.warningThreshold || 0.8;
  }

  // 检查空间
  checkSpace() {
    let used = 0;
    for (let key in this.storage) {
      if (this.storage.hasOwnProperty(key)) {
        used += this.storage[key].length * 2; // UTF-16 字符占2字节
      }
    }
    return {
      used,
      available: this.maxSize - used,
      percent: used / this.maxSize
    };
  }

  // 自动清理
  autoClean(targetPercent = 0.5) {
    const { percent } = this.checkSpace();

    if (percent > this.warningThreshold) {
      // 按访问时间排序，删除最老的
      const keys = Object.keys(this.storage)
        .map(key => ({
          key,
          time: parseInt(this.storage.getItem(`__${key}_time__`) || 0)
        }))
        .sort((a, b) => a.time - b.time);

      const targetSize = this.maxSize * targetPercent;
      let currentSize = this.checkSpace().used;

      for (const item of keys) {
        if (currentSize <= targetSize) break;

        const key = item.key;
        const size = this.storage[key].length * 2;
        this.storage.removeItem(key);
        this.storage.removeItem(`__${key}_time__`);
        currentSize -= size;
      }
    }
  }

  // 安全存储
  safeSet(key, value) {
    const data = JSON.stringify(value);
    const size = data.length * 2;

    // 检查单条数据是否过大
    if (size > this.maxSize * 0.5) {
      console.warn('Data too large for localStorage');
      return false;
    }

    try {
      this.autoClean();
      this.storage.setItem(key, data);
      this.storage.setItem(`__${key}_time__`, Date.now().toString());
      return true;
    } catch (e) {
      if (e.name === 'QuotaExceededError') {
        // 强制清理
        this.clear();
        try {
          this.storage.setItem(key, data);
          return true;
        } catch (e) {
          console.error('Storage failed:', e);
          return false;
        }
      }
      return false;
    }
  }
}

async function* streamAIResponse(url, body) {
  const response = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(body)
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();

    if (done) break;

    const chunk = decoder.decode(value, { stream: true });
    const lines = chunk.split('\n').filter(line => line.trim());

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = line.slice(6);
        if (data === '[DONE]') return;
        yield JSON.parse(data);
      }
    }
  }
}

  }

function sortedSquares(nums) {
  const n = nums.length;
  const result = new Array(n);
  let left = 0;
  let right = n - 1;
  let pos = n - 1;

  while (left <= right) {
    const leftSquare = nums[left] * nums[left];
    const rightSquare = nums[right] * nums[right];

    if (leftSquare > rightSquare) {
      result[pos] = leftSquare;
      left++;
    } else {
      result[pos] = rightSquare;
      right--;
    }
    pos--;
  }

  return result;
}

46.2 有效括号（带优先级）

// 带优先级的括号匹配
function isValid(s) {
  const stack = [];
  const pairs = {
    ')': '(',
    ']': '[',
    '}': '{'
  };
  const priority = { '(': 1, '[': 2, '{': 3 };

  for (const char of s) {
    if ('([{'.includes(char)) {
      // 新括号入栈时，检查优先级
      if (stack.length > 0) {
        const top = stack[stack.length - 1];
        if (priority[char] > priority[top]) {
          // 优先级高的不能嵌套在优先级低的中
          return false;
        }
      }
      stack.push(char);
    } else if (')]}'.includes(char)) {
      if (!stack.length || stack[stack.length - 1] !== pairs[char]) {
        return false;
      }
      stack.pop();
    }
  }

  return stack.length === 0;
}

// 示例
console.log(isValid('{[()]}')); // true
console.log(isValid('{[(])}')); // false

function buildTree(arr) {
  const map = new Map();
  const roots = [];

  // 创建节点
  arr.forEach(item => {
    map.set(item.id, { ...item, children: [] });
  });

  // 建立关系
  arr.forEach(item => {
    const node = map.get(item.id);
    if (item.parentId === null || item.parentId === undefined) {
      roots.push(node);
    } else {
      const parent = map.get(item.parentId);
      if (parent) {
        parent.children.push(node);
      } else {
        roots.push(node);
      }
    }
  });

  return roots;
}

特性 普通 script defer async
执行时机 阻塞 HTML 解析 HTML 解析完成后执行 下载完成后立即执行

<script defer src="app.js"></script>
