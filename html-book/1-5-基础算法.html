第5章 算法与数据结构

1.1 什么是算法

算法（Algorithm）是解决特定问题的一系列明确的计算步骤，是计算机科学的核心概念之一。一个好的算法应该具备以下特点：

  1. 正确性：算法能够正确解决问题
  2. 可读性：算法逻辑清晰，易于理解
  3. 健壮性：能够处理异常输入
  4. 时间效率：执行时间合理
  5. 空间效率：占用内存合理

1.2 算法的分类

按计算方式分类

  1. 穷举法：遍历所有可能的情况
  2. 分治法：将大问题分解为小问题
  3. 动态规划：最优子结构+重叠子问题
  4. 贪心算法：局部最优策略
  5. 回溯算法：尝试搜索解空间
  6. 分支限界法：广度优先搜索解空间

按问题类型分类

  1. 排序算法：将无序数据变为有序
  2. 查找算法：在数据集中寻找目标
  3. 图算法：处理图结构数据
  4. 字符串算法：处理字符串问题
  5. 数学算法：数学计算相关

1.3 算法复杂度概述

时间复杂度

时间复杂度表示算法执行时间随输入规模增长的变化趋势。常用表示法为大O记号。

常见时间复杂度（从低到高）：

  • O(1)：常数时间
  • O(log n)：对数时间
  • O(n)：线性时间
  • O(n log n)：线性对数时间
  • O(n^2)：平方时间
  • O(n^3)：立方时间
  • O(2^n)：指数时间

空间复杂度

空间复杂度表示算法占用内存空间随输入规模增长的变化趋势。

// O(1) - 常数时间
function constantTime(arr, index) {
    return arr[index];
}

// O(n) - 线性时间
function linearTime(arr) {
    let sum = 0;
    for (let i = 0; i < arr.length; i++) {
        sum += arr[i];
    }
    return sum;
}

// O(n^2) - 平方时间
function quadraticTime(arr) {
    let count = 0;
    for (let i = 0; i < arr.length; i++) {
        for (let j = 0; j < arr.length; j++) {
            if (arr[i] === arr[j]) count++;
        }
    }
    return count;
}

// O(log n) - 对数时间
function logarithmicTime(n) {
    let count = 0;
    while (n > 1) {
        n = Math.floor(n / 2);
        count++;
    }
    return count;
}

第2章 排序算法详解

2.1 排序算法概述

排序算法是将一组数据按照特定顺序排列的算法。排序是计算机科学中最基本也是最重要的算法之一。

排序算法的分类

  1. 比较排序：通过比较元素大小进行排序

    • 冒泡排序
    • 选择排序
    • 插入排序
    • 希尔排序
    • 归并排序
    • 快速排序
    • 堆排序

  2. 非比较排序：不通过比较进行排序

    • 计数排序
    • 桶排序
    • 基数排序

排序算法的稳定性

稳定排序：如果两个相等的元素，排序前后它们的相对位置不变，则该排序算法是稳定的。

2.2 冒泡排序 (Bubble Sort)

冒泡排序是最简单的排序算法之一，通过不断比较相邻元素并交换位置，将最大的元素"冒泡"到数组末尾。

/**
 * 冒泡排序 - 基础版本
 * 时间复杂度: O(n^2)
 * 空间复杂度: O(1)
 * 稳定性: 稳定
 */
function bubbleSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    return arr;
}

2.3 选择排序 (Selection Sort)

选择排序是一种简单直观的排序算法，每次从未排序部分选择最小（或最大）元素放到已排序部分的末尾。

/**
 * 选择排序
 * 时间复杂度: O(n^2)
 * 空间复杂度: O(1)
 * 稳定性: 不稳定
 */
function selectionSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        let minIndex = i;
        for (let j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        if (minIndex !== i) {
            [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]];
        }
    }
    return arr;
}

2.4 插入排序 (Insertion Sort)

插入排序类似于整理扑克牌，将数组分为已排序区和未排序区，每次将未排序区的第一个元素插入到已排序区的正确位置。

/**
 * 插入排序
 * 时间复杂度: O(n^2) 平均/最坏, O(n) 最好（已排序）
 * 空间复杂度: O(1)
 * 稳定性: 稳定
 */
function insertionSort(arr) {
    const n = arr.length;
    for (let i = 1; i < n; i++) {
        const key = arr[i];
        let j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
    return arr;
}

2.5 归并排序 (Merge Sort)

归并排序是分治思想的典型应用，将数组分成若干子序列，分别排序后再合并。

/**
 * 归并排序
 * 时间复杂度: O(n log n)
 * 空间复杂度: O(n)
 * 稳定性: 稳定
 */
function mergeSort(arr) {
    if (arr.length <= 1) return arr;

    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));

    return merge(left, right);
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;

    while (i < left.length && j < right.length) {
        if (left[i] <= right[j]) {
            result.push(left[i++]);
        } else {
            result.push(right[j++]);
        }
    }

    return result.concat(left.slice(i)).concat(right.slice(j));
}

2.6 快速排序 (Quick Sort)

快速排序是应用最广泛的排序算法，采用分治策略。

/**
 * 快速排序
 * 时间复杂度: O(n log n) 平均, O(n^2) 最坏
 * 空间复杂度: O(log n)
 * 稳定性: 不稳定
 */
function quickSort(arr, left = 0, right = arr.length - 1) {
    if (left < right) {
        const pivotIndex = partition(arr, left, right);
        quickSort(arr, left, pivotIndex - 1);
        quickSort(arr, pivotIndex + 1, right);
    }
    return arr;
}

function partition(arr, left, right) {
    const pivot = arr[right];
    let i = left - 1;

    for (let j = left; j < right; j++) {
        if (arr[j] < pivot) {
            i++;
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
    }

    [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
    return i + 1;
}

function binarySearch(arr, target) {
    let left = 0;
    let right = arr.length - 1;

    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (arr[mid] === target) {
            return mid;

            left = mid + 1;
        } else {

3.3 二分查找变体

// 查找第一个等于目标的位置
function findFirst(arr, target) {
    let left = 0, right = arr.length - 1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (arr[mid] < target) {
            left = mid + 1;
        } else if (arr[mid] > target) {
            right = mid - 1;
        } else {
            if (mid === 0 || arr[mid - 1] !== target) {
                return mid;
            }
            right = mid - 1;
        }
    }
    return -1;
}

// 查找最后一个等于目标的位置
function findLast(arr, target) {
    let left = 0, right = arr.length - 1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (arr[mid] < target) {
            left = mid + 1;
        } else if (arr[mid] > target) {
            right = mid - 1;
        } else {
            if (mid === arr.length - 1 || arr[mid + 1] !== target) {
                return mid;
            }
            left = mid + 1;
        }
    }
    return -1;
}

// 有序数组去重
function removeDuplicates(nums) {
    let slow = 0;
    for (let fast = 1; fast < nums.length; fast++) {
        if (nums[fast] !== nums[slow]) {
            slow++;
            nums[slow] = nums[fast];
        }

}

// 两数之和（有序数组）
function twoSum(nums, target) {
    let left = 0, right = nums.length - 1;
    while (left < right) {
        const sum = nums[left] + nums[right];
        if (sum === target) {
            return [left, right];
        } else if (sum < target) {
            left++;
        } else {
            right--;
        }
    }
    return [];
}

// 最大子数组和
function maxSubArraySum(arr, k) {
    let maxSum = -Infinity;
    let windowSum = 0;

    for (let i = 0; i < arr.length; i++) {
        windowSum += arr[i];
        if (i >= k - 1) {
            maxSum = Math.max(maxSum, windowSum);
            windowSum -= arr[i - k + 1];
        }
    }
    return maxSum;
}

第5章 双指针技巧详解

5.1 相向双指针

相向双指针从数组两端向中间遍历，常用于有序数组的两数之和、三数之和等问题。

5.1.1 两数之和 II（LeetCode 167）

/**
 * 有序数组的两数之和 II
 * 输入: numbers = [2,7,11,15], target = 9
 * 输出: [1,2]
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function twoSum(numbers, target) {
    let left = 0;
    let right = numbers.length - 1;

    while (left < right) {
        const sum = numbers[left] + numbers[right];
        if (sum === target) {
            return [left + 1, right + 1];  // 题目要求1-indexed
        } else if (sum < target) {
            left++;   // 太小，移动左指针
        } else {
            right--;  // 太大，移动右指针
        }
    }
    return [];
}

5.1.2 三数之和（LeetCode 15）

/**
 * 三数之和
 * 要求返回不重复的三元组
 *
 * 时间复杂度: O(n^2)
 */
function threeSum(nums) {
    const result = [];
    nums.sort((a, b) => a - b);

    for (let i = 0; i < nums.length - 2; i++) {
        // 跳过重复的i
        if (i > 0 && nums[i] === nums[i - 1]) continue;

        let left = i + 1;
        let right = nums.length - 1;

        while (left < right) {
            const sum = nums[i] + nums[left] + nums[right];
            if (sum === 0) {
                result.push([nums[i], nums[left], nums[right]]);
                // 跳过重复的left和right
                while (left < right && nums[left] === nums[left + 1]) left++;
                while (left < right && nums[right] === nums[right - 1]) right--;
                left++;
                right--;
            } else if (sum < 0) {
                left++;
            } else {
                right--;
            }
        }
    }
    return result;
}

5.1.3 盛最多水的容器（LeetCode 11）

/**
 * 盛最多水的容器
 * 面积 = min(height[i], height[j]) * (j - i)
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function maxArea(height) {
    let left = 0;
    let right = height.length - 1;
    let maxArea = 0;

    while (left < right) {
        const width = right - left;
        const h = Math.min(height[left], height[right]);
        maxArea = Math.max(maxArea, width * h);

        // 移动较短的那一边
        if (height[left] < height[right]) {
            left++;
        } else {
            right--;
        }
    }
    return maxArea;
}

5.1.4 接雨水（LeetCode 42）

/**
 * 接雨水
 * 动态规划解法：预处理每个位置的左边最大值和右边最大值
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(n)
 */
function trap(height) {
    if (height.length === 0) return 0;

    const n = height.length;
    const leftMax = new Array(n);
    const rightMax = new Array(n);

    // 预处理左边最大值
    leftMax[0] = height[0];
    for (let i = 1; i < n; i++) {
        leftMax[i] = Math.max(leftMax[i - 1], height[i]);
    }

    // 预处理右边最大值
    rightMax[n - 1] = height[n - 1];
    for (let i = n - 2; i >= 0; i--) {
        rightMax[i] = Math.max(rightMax[i + 1], height[i]);
    }

    // 计算雨水
    let water = 0;
    for (let i = 0; i < n; i++) {
        water += Math.min(leftMax[i], rightMax[i]) - height[i];
    }
    return water;
}

/**
 * 双指针优化：空间O(1)
 */
function trap2(height) {
    let left = 0, right = height.length - 1;
    let leftMax = 0, rightMax = 0;
    let water = 0;

    while (left < right) {
        if (height[left] < height[right]) {
            if (height[left] >= leftMax) {
                leftMax = height[left];
            } else {
                water += leftMax - height[left];
            }
            left++;
        } else {
            if (height[right] >= rightMax) {
                rightMax = height[right];
            } else {
                water += rightMax - height[right];
            }
            right--;
        }
    }
    return water;
}

5.2 滑动窗口

滑动窗口通过维护一个窗口来高效处理子数组/子串问题。

5.2.1 长度最小的子数组（LeetCode 209）

/**
 * 长度最小的子数组（和 >= target）
 *
 * 时间复杂度: O(n)
 */
function minSubArrayLen(target, nums) {
    let left = 0;
    let sum = 0;
    let minLength = Infinity;

    for (let right = 0; right < nums.length; right++) {
        sum += nums[right];

        while (sum >= target) {
            minLength = Math.min(minLength, right - left + 1);
            sum -= nums[left];
            left++;
        }
    }

    return minLength === Infinity ? 0 : minLength;
}

5.2.2 滑动窗口最大值（LeetCode 239）

/**
 * 滑动窗口最大值（单调队列解法）
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(k)
 */
function maxSlidingWindow(nums, k) {
    const result = [];
    const deque = [];  // 存储索引，保持递减

    for (let i = 0; i < nums.length; i++) {
        // 移除超出窗口的索引
        while (deque.length && deque[0] <= i - k) {
            deque.shift();
        }

        // 移除比当前元素小的索引
        while (deque.length && nums[deque[deque.length - 1]] <= nums[i]) {
            deque.pop();
        }

        deque.push(i);

        // 记录结果
        if (i >= k - 1) {
            result.push(nums[deque[0]]);
        }
    }

    return result;
}

5.2.3 无重复字符的最长子串（LeetCode 3）

/**
 * 无重复字符的最长子串
 *
 * 时间复杂度: O(n)
 */
function lengthOfLongestSubstring(s) {
    const charIndex = new Map();
    let maxLength = 0;
    let left = 0;

    for (let right = 0; right < s.length; right++) {
        if (charIndex.has(s[right]) && charIndex.get(s[right]) >= left) {
            left = charIndex.get(s[right]) + 1;
        }
        charIndex.set(s[right], right);
        maxLength = Math.max(maxLength, right - left + 1);
    }

    return maxLength;
}

5.3 二分查找

5.3.1 红蓝染色法（二分查找模板）

/**
 * 红蓝染色法 - 二分查找通用模板
 *
 * 核心思想：
 * - 红色：不符合条件
 * - 蓝色：符合条件
 * - 寻找蓝色区域的左边界
 */
function binarySearch(nums, target) {
    let left = -1;   // 红色边界（不符合条件）
    let right = nums.length;  // 蓝色边界（符合条件）

    while (left + 1 < right) {
        const mid = left + Math.floor((right - left) / 2);
        if (isBlue(mid, nums, target)) {
            right = mid;  // 蓝色区域在左边
        } else {
            left = mid;   // 蓝色区域在右边
        }
    }

    return right;  // 返回蓝色区域左边界
}

// 判断是否为蓝色（符合条件）
function isBlue(mid, nums, target) {
    return nums[mid] >= target;
}

5.3.2 在排序数组中查找元素位置（LeetCode 34）

/**
 * 在排序数组中查找元素的第一个和最后一个位置
 *
 * 时间复杂度: O(log n)
 */
function searchRange(nums, target) {
    const leftBorder = findLeftBorder(nums, target);
    const rightBorder = findRightBorder(nums, target);

    if (leftBorder === -1 || rightBorder === -1) {
        return [-1, -1];
    }
    return [leftBorder, rightBorder];
}

function findLeftBorder(nums, target) {
    let left = 0, right = nums.length - 1;
    let border = -1;

    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] >= target) {
            border = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    return border;
}

function findRightBorder(nums, target) {
    let left = 0, right = nums.length - 1;
    let border = -1;

    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] <= target) {
            border = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return border;
}

5.3.3 搜索旋转排序数组（LeetCode 33, 153, 162）

/**
 * 搜索旋转排序数组（LeetCode 33）
 * 数组原本是升序的，但被旋转了
 */
function search(nums, target) {
    let left = 0, right = nums.length - 1;

    while (left <= right) {
        const mid = Math.floor((left + right) / 2);

        if (nums[mid] === target) return mid;

        // 判断哪半边是有序的
        if (nums[left] <= nums[mid]) {
            // 左边是有序的
            if (target >= nums[left] && target < nums[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        } else {
            // 右边是有序的
            if (target > nums[mid] && target <= nums[right]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    }
    return -1;
}

/**
 * 搜索旋转排序数组 II（LeetCode 33）- 有重复值
 */
function searchWithDuplicates(nums, target) {
    let left = 0, right = nums.length - 1;

    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] === target) return true;

        // 处理重复值的情况
        if (nums[left] === nums[mid] && nums[mid] === nums[right]) {
            left++;
            right--;
        } else if (nums[left] <= nums[mid]) {
            if (target >= nums[left] && target < nums[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        } else {
            if (target > nums[mid] && target <= nums[right]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    }
    return false;

/**
 * 搜索旋转排序数组中的最小值（LeetCode 153）
 */
function findMin(nums) {
    let left = 0, right = nums.length - 1;

    while (left < right) {
        const mid = Math.floor((left + right) / 2);
        if (nums[mid] > nums[right]) {
            // 最小值在右半边
            left = mid + 1;
        } else {
            // 最小值在左半边或就是mid
            right = mid;
        }
    }
    return nums[left];
}

 * 反转链表
 *
 * 时间复杂度: O(n)
 * 空间复杂度: O(1)
 */
function reverseList(head) {
    let prev = null;
    let curr = head;

    while (curr) {
        const next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }

    return prev;
}

// 递归版本
function reverseListRec(head) {
    if (!head || !head.next) return head;

    const newHead = reverseListRec(head.next);
    head.next.next = head;
    head.next = null;
    return newHead;
}

/**
 * K个一组翻转链表
 */
function reverseKGroup(head, k) {
    const dummy = new ListNode(0);
    dummy.next = head;

    let prevGroupEnd = dummy;

    while (true) {
        const kth = getKth(prevGroupEnd, k);
        if (!kth) break;

        const nextGroupStart = kth.next;

        // 反转当前组
        let prev = kth.next;
        let curr = prevGroupEnd.next;

        while (curr !== nextGroupStart) {
            const next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }

        const groupStart = prevGroupEnd.next;
        prevGroupEnd.next = kth;
        prevGroupEnd = groupStart;
    }

    return dummy.next;
}

function getKth(node, k) {
    for (let i = 0; i < k && node; i++) {
        node = node.next;
    }
    return node;
}

/**
 * 检测环形链表（LeetCode 141）
 */
function hasCycle(head) {
    let slow = head;
    let fast = head;

    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow === fast) return true;
    }
    return false;
}

/**
 * 环形链表 II（LeetCode 142）
 * 返回环的起点
 */
function detectCycle(head) {
    let slow = head;
    let fast = head;

    // 第一步：找到相遇点
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow === fast) break;
    }

    if (!fast || !fast.next) return null;

    // 第二步：找到环的起点
    slow = head;
    while (slow !== fast) {
        slow = slow.next;
        fast = fast.next;
    }

    return slow;
}

/**

function removeNthFromEnd(head, n) {

    dummy.next = head;

    let fast = dummy;
    let slow = dummy;

    // fast指针先走n步
    for (let i = 0; i <= n; i++) {
        fast = fast.next;
    }

        fast = fast.next;
        slow = slow.next;

        if (curr.val === curr.next.val) {
            curr.next = curr.next.next;
        } else {
            curr = curr.next;
        }
    }

}

/**
 * 删除所有重复的节点（LeetCode 82）
 */
function deleteAllDuplicates(head) {
    const dummy = new ListNode(0);
    dummy.next = head;
    let prev = dummy;
    let curr = head;

    while (curr) {
        let isDuplicate = false;
        while (curr.next && curr.val === curr.next.val) {
            isDuplicate = true;
            curr = curr.next;
        }

        if (isDuplicate) {
            prev.next = curr.next;
        } else {
            prev = curr;
        }
        curr = curr.next;
    }

    return dummy.next;
}

        const level = [];

        for (let i = 0; i < levelSize; i++) {
            const node = queue.shift();
            level.push(node.val);
            if (node.left) queue.push(node.left);
            if (node.right) queue.push(node.right);
        }

        result.push(level);

        return root;

    const right = lowestCommonAncestor(root.right, p, q);

    if (left && right) return root;

}

/**
 * 子集 - LeetCode 78
 */
function subsets(nums) {
    const result = [];

    function backtrack(start, path) {
        result.push([...path]);

        for (let i = start; i < nums.length; i++) {
            path.push(nums[i]);
            backtrack(i + 1, path);
            path.pop();
        }
    }

    backtrack(0, []);
    return result;
}

/**
 * 分割回文串 - LeetCode 131
 */
function partition(s) {
    const result = [];
    const path = [];

    function isPalindrome(str, l, r) {
        while (l < r) {
            if (str[l] !== str[r]) return false;
            l++;
            r--;
        }
        return true;
    }

    function backtrack(start) {
        if (start === s.length) {
            result.push([...path]);
            return;
        }

        for (let end = start; end < s.length; end++) {
            if (isPalindrome(s, start, end)) {
                path.push(s.substring(start, end + 1));
                backtrack(end + 1);
                path.pop();
            }
        }
    }

    backtrack(0);
    return result;
}

/**
 * 电话号码的字母组合 - LeetCode 17
 */
function letterCombinations(digits) {
    if (!digits) return [];

    const map = {
        '2': 'abc', '3': 'def', '4': 'ghi', '5': 'jkl',
        '6': 'mno', '7': 'pqrs', '8': 'tuv', '9': 'wxyz'
    };

    const result = [];
    const path = [];

    function backtrack(index) {
        if (index === digits.length) {
            result.push(path.join(''));
            return;
        }

        for (const char of map[digits[index]]) {
            path.push(char);
            backtrack(index + 1);
            path.pop();
        }
    }

    backtrack(0);
    return result;
}

7.2 组合型回溯（带剪枝）

/**
 * 组合总和 III - LeetCode 216
 */
function combinationSum3(k, n) {
    const result = [];
    const path = [];

    function backtrack(start, sum) {
        if (path.length === k) {
            if (sum === n) {
                result.push([...path]);
            }
            return;
        }

        // 剪枝：剩余数字不够或和太大
        const remaining = k - path.length;
        for (let i = start; i <= 9; i++) {
            if (i > n - sum) break;  // 剪枝
            path.push(i);
            backtrack(i + 1, sum + i);
            path.pop();
        }
    }

    backtrack(1, 0);
    return result;
}

/**
 * 括号生成 - LeetCode 22
 */
function generateParenthesis(n) {
    const result = [];
    const path = [];

    function backtrack(open, close) {
        if (open === n && close === n) {
            result.push(path.join(''));
            return;
        }

        if (open < n) {
            path.push('(');
            backtrack(open + 1, close);
            path.pop();
        }

        if (close < open) {
            path.push(')');
            backtrack(open, close + 1);
            path.pop();
        }
    }

    backtrack(0, 0);
    return result;
}

7.3 排列型回溯

/**
 * 全排列 - LeetCode 46
 */
function permute(nums) {
    const result = [];
    const used = new Array(nums.length).fill(false);
    const path = [];

    function backtrack() {
        if (path.length === nums.length) {
            result.push([...path]);
            return;
        }

        for (let i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            used[i] = true;
            path.push(nums[i]);
            backtrack();
            path.pop();
            used[i] = false;
        }
    }

    backtrack();
    return result;
}

/**
 * N皇后 - LeetCode 51
 */
function solveNQueens(n) {
    const result = [];
    const cols = new Set();
    const diag1 = new Set();  // row - col
    const diag2 = new Set();  // row + col
    const board = Array(n).fill(null).map(() => Array(n).fill('.'));

    function backtrack(row) {
        if (row === n) {
            result.push(board.map(row => row.join('')));
            return;
        }

        for (let col = 0; col < n; col++) {
            if (cols.has(col) || diag1.has(row - col) || diag2.has(row + col)) {
                continue;
            }

            board[row][col] = 'Q';
            cols.add(col);
            diag1.add(row - col);
            diag2.add(row + col);

            backtrack(row + 1);

            board[row][col] = '.';
            cols.delete(col);
            diag1.delete(row - col);
            diag2.delete(row + col);
        }
    }

    backtrack(0);
    return result;
}

/**
 * 打家劫舍 - LeetCode 198
 */
function rob(nums) {
    if (!nums.length) return 0;
    if (nums.length === 1) return nums[0];

    // dp[i] = max(dp[i-1], dp[i-2] + nums[i])
    const dp = new Array(nums.length);
    dp[0] = nums[0];
    dp[1] = Math.max(nums[0], nums[1]);

    for (let i = 2; i < nums.length; i++) {
        dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
    }

    return dp[nums.length - 1];
}

// 空间优化
function robOptimized(nums) {
    if (!nums.length) return 0;

    let prev2 = 0;
    let prev1 = nums[0];

    for (let i = 1; i < nums.length; i++) {
        const curr = Math.max(prev1, prev2 + nums[i]);
        prev2 = prev1;
        prev1 = curr;
    }

    return prev1;
}

8.2 背包问题

/**
 * 0-1背包
 */
function knapsack(weights, values, capacity) {
    const n = weights.length;
    const dp = Array(n + 1).fill(null).map(() => Array(capacity + 1).fill(0));

    for (let i = 1; i <= n; i++) {
        for (let w = 0; w <= capacity; w++) {
            if (weights[i - 1] <= w) {
                dp[i][w] = Math.max(
                    dp[i - 1][w],
                    dp[i - 1][w - weights[i - 1]] + values[i - 1]
                );
            } else {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }

    return dp[n][capacity];
}

/**
 * 目标和 - LeetCode 494
 */
function findTargetSumWays(nums, target) {
    const sum = nums.reduce((a, b) => a + b, 0);

    // 转换为子集和问题
    // (sum - target) 必须是偶数
    if (Math.abs(target) > sum) return 0;
    if ((sum - target) % 2 !== 0) return 0;

    const bagSize = (sum - target) / 2;

    const dp = new Array(bagSize + 1).fill(0);
    dp[0] = 1;

    for (const num of nums) {
        for (let j = bagSize; j >= num; j--) {
            dp[j] += dp[j - num];
        }
    }

    return dp[bagSize];
}

/**
 * 零钱兑换 - LeetCode 322
 */
function coinChange(coins, amount) {
    const dp = new Array(amount + 1).fill(amount + 1);
    dp[0] = 0;

    for (let i = 1; i <= amount; i++) {
        for (const coin of coins) {
            if (coin <= i) {
                dp[i] = Math.min(dp[i], dp[i - coin] + 1);
            }
        }
    }

    return dp[amount] > amount ? -1 : dp[amount];
}

// 二分优化 O(n log n)
function lengthOfLISBinary(nums) {
    const piles = [];

    for (const num of nums) {
        let left = 0, right = piles.length;

        while (left < right) {
            const mid = Math.floor((left + right) / 2);
            if (piles[mid] < num) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        if (left === piles.length) {
            piles.push(num);
        } else {
            piles[left] = num;
        }
    }

    return piles.length;
}

 * 买卖股票的最佳时机（冷冻期）- LeetCode 309

    const n = prices.length;
    // dp[i][0]: 持有股票
    // dp[i][1]: 不持有股票，处于冷冻期
    // dp[i][2]: 不持有股票，不在冷冻期
    const dp = Array(n).fill(null).map(() => Array(3).fill(0));

            dp[i - 1][0],           // 继续持有
            dp[i - 1][2] - prices[i]  // 买入
        );
        dp[i][1] = dp[i - 1][0] + prices[i];  // 卖出（进入冷冻期）
        dp[i][2] = Math.max(
            dp[i - 1][1],  // 继续不在冷冻期
            dp[i - 1][2]   // 之前也不在冷冻期
        );

    return Math.max(dp[n - 1][1], dp[n - 1][2]);
}

        }

}

9.1 树的直径（LeetCode 543）

/**
 * 二叉树的直径
 */
function diameterOfBinaryTree(root) {
    let maxDiameter = 0;

    function depth(node) {
        if (!node) return 0;

        const leftDepth = depth(node.left);
        const rightDepth = depth(node.right);

        // 更新最大直径
        maxDiameter = Math.max(maxDiameter, leftDepth + rightDepth);

        return 1 + Math.max(leftDepth, rightDepth);
    }

    depth(root);
    return maxDiameter;
}

function rob(root) {
    function dfs(node) {
        if (!node) return [0, 0];  // [不偷，偷]

        const left = dfs(node.left);
        const right = dfs(node.right);

        // 当前节点不偷：取子节点的最大值
        const notRob = Math.max(left[0], left[1]) +
                       Math.max(right[0], right[1]);

        // 当前节点偷：子节点不能偷
        const rob = node.val + left[0] + right[0];

        return [notRob, rob];
    }

    return Math.max(...dfs(root));
}

9.3 监控二叉树（LeetCode 968）

/**
 * 监控二叉树
 * 0: 无覆盖
 * 1: 有摄像头
 * 2: 有覆盖
 */
function minCameraCover(root) {
    let result = 0;

    function dfs(node) {
        if (!node) return 2;  // 空节点视为有覆盖

        const left = dfs(node.left);
        const right = dfs(node.right);

        // 子节点有未覆盖的，需要放摄像头
        if (left === 0 || right === 0) {
            result++;
            return 1;
        }

        // 子节点有摄像头，当前节点被覆盖
        if (left === 1 || right === 1) {
            return 2;
        }

        // 子节点都被覆盖，返回未覆盖
        return 0;
    }

    const status = dfs(root);
    // 根节点未覆盖，需要额外摄像头
    if (status === 0) result++;

    return result;
}

/**
 * 每日温度 - LeetCode 739
 */
function dailyTemperatures(T) {
    const n = T.length;
    const result = new Array(n).fill(0);
    const stack = [];  // 存储索引

    for (let i = 0; i < n; i++) {
        while (stack.length && T[i] > T[stack[stack.length - 1]]) {
            const prevIndex = stack.pop();
            result[prevIndex] = i - prevIndex;
        }
        stack.push(i);
    }

    return result;
}

/**
 * 滑动窗口最小值
 */
function minSlidingWindow(nums, k) {
    const result = [];
    const deque = [];  // 存储索引，保持递增

    for (let i = 0; i < nums.length; i++) {
        // 移除超出窗口的
        while (deque.length && deque[0] <= i - k) {
            deque.shift();
        }

        // 移除比当前元素大的
        while (deque.length && nums[deque[deque.length - 1]] >= nums[i]) {
            deque.pop();
        }

        deque.push(i);

        if (i >= k - 1) {
            result.push(nums[deque[0]]);
        }
    }

    return result;
}
