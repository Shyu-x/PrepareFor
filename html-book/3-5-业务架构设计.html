5.1 TOB 与 TOC 业务特点与架构差异

5.1.1 TOB 业务特点

TOB（To Business）业务主要面向企业用户，具有以下特点：

// TOB 业务特征
const TOBCharacteristics = {
  // 复杂业务流程
  complexWorkflow: {
    多角色协同: '管理员、普通用户、审计员等',
    权限体系: '细粒度的功能和数据权限',
    审批流程: '多级审批、条件分支'

  // 数据量大
  largeData: {
    表格数据: '大量列表、报表',
    导入导出: 'Excel、CSV 等批量操作',
    统计分析: '多维度数据聚合'

5.2 设计模式在前端中的应用

5.2.1 创建型模式

5.2.1.1 单例模式
// 单例模式 - 状态管理
class Store {
  constructor() {
    if (Store.instance) {
      return Store.instance;
    }
    this.state = {};
    Store.instance = this;
  }

  getState() {
    return this.state;
  }

  setState(newState) {
    this.state = { ...this.state, ...newState };
  }
}

// 使用
const store = new Store(); // 全局唯一实例
5.2.1.2 工厂模式
// 工厂模式 - 组件创建
class ComponentFactory {
  static create(type, props) {
    switch (type) {
      case 'button':
        return new ButtonComponent(props);
      case 'input':
        return new InputComponent(props);
      case 'modal':
        return new ModalComponent(props);
      default:
        throw new Error(`Unknown component type: ${type}`);
    }
  }
}

    if (isLoading) {

5.2.2.2 代理模式
// 代理模式 - 数据请求
class DataProxy {
  constructor(source) {
    this.source = source;
    this.cache = new Map();
  }

  async get(key) {
    if (this.cache.has(key)) {
      return this.cache.get(key);
    }

    const data = await this.source.get(key);
    this.cache.set(key, data);
    return data;
  }
}

5.2.3 行为型模式

5.2.3.1 观察者模式
// 观察者模式 - 事件系统
class EventEmitter {
  constructor() {
    this.events = {};
  }

  on(event, callback) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    this.events[event].push(callback);
  }

  off(event, callback) {
    if (!this.events[event]) return;
    this.events[event] = this.events[event].filter(cb => cb !== callback);
  }

  emit(event, ...args) {
    if (!this.events[event]) return;
    this.events[event].forEach(cb => cb(...args));
  }
}
5.2.3.2 策略模式
// 策略模式 - 表单验证
class ValidationStrategy {
  constructor() {
    this.strategies = {};
  }

  addStrategy(name, validator) {
    this.strategies[name] = validator;
  }

  validate(value, strategyName) {
    const strategy = this.strategies[strategyName];
    if (!strategy) {
      throw new Error(`Unknown strategy: ${strategyName}`);
    }
    return strategy.validate(value);
  }
}

// 使用
const validator = new ValidationStrategy();
validator.addStrategy('email', {
  validate: (value) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)
});

5.4 设计理论与原则

5.4.1 SOLID 原则

原则 描述 前端应用
单一职责 一个类/函数只做一件事 组件、Hook 单一功能
开闭原则 对扩展开放，对修改封闭 组件可扩展、不可修改
里氏替换 子类可替换父类 继承体系
接口隔离 使用专门接口 TypeScript 接口
依赖倒置 依赖抽象 依赖注入

5.4.2 DRY 原则

Don’t Repeat Yourself - 不要重复自己

// 不好 - 重复代码
function formatName1(name) {
  return name.trim().toLowerCase();
}

function formatName2(name) {
  return name.trim().toLowerCase();
}

// 好 - 抽取公共函数
function formatName(name) {
  return name.trim().toLowerCase();
}

5.4.3 KISS 原则

Keep It Simple, Stupid - 保持简单

// 不好 - 过度设计
class ComplexCalculator {
  constructor() {
    this.history = [];
  }

  add(a, b) {
    const result = a + b;
    this.history.push({ operation: 'add', result });
    return result;
  }
}

// 好 - 简单直接
function add(a, b) {
  return a + b;
}

5.6.1 CI 流程

# GitHub Actions CI 配置
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
      - run: npm ci
      - run: npm run lint

  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
      - run: npm ci
      - run: npm test

  build:
    runs-on: ubuntu-latest
    needs: [lint, test]
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
      - run: npm ci
      - run: npm run build
