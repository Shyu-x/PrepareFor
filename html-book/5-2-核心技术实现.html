第2章 FastDocument

详细讲解 FastDocument 项目的具体代码实现，包括富文本编辑器、实时协作、视频会议、状态管理、权限管理等核心模块

1.1 块编辑器架构

技术选型理由: 项目采用自定义块编辑器架构而非直接使用 TipTap，核心原因是需要更精细的块级控制、虚拟滚动支持以及与协作系统的深度集成。自定义编辑器采用 Block 架构，每个内容块独立管理，便于实现拖拽排序、分块加载等高级功能。

核心编辑器位于 D:\Develeping\FastDocument\frontend\src\components\Editor.tsx，采用块架构设计：

"use client";

import React, { useRef, useEffect, useState, useCallback } from "react";
import { useDocumentStore, Block } from "@/store/documentStore";
import { socketClient } from "@/lib/socket";
import { exportDocument, importDocument } from "@/lib/export";
import { useTheme } from "@/components/ThemeProvider";
import { DocumentMenuBar } from "@/components/DocumentMenuBar";
import { DocumentOutline } from "@/components/DocumentOutline";
import { BubbleMenu } from "@/components/BubbleMenu";
import { VirtualEditor } from "./VirtualEditor";
import { motion, AnimatePresence } from "framer-motion";

/**
 * 文档编辑器主组件
 * 根据文档大小自动选择普通编辑器或虚拟滚动编辑器
 *
 * 业务场景:
 * - 小型文档 (<100 块): 使用普通渲染，完整的动画效果
 * - 大型文档 (>=100 块): 使用虚拟滚动，优化性能
 */
export const Editor: React.FC = () => {
  const {
    id: docId,
    blocks,
    title,
    updateTitle,
    addBlock,
    remoteUpdateBlock,
    setUserTyping,
    removeUserTyping,
    fetchDocument,
    setOnlineStatus,
    setOnlineUsers,
    onlineUsers
  } = useDocumentStore();
  const { isMounted, theme } = useTheme();
  const isDark = isMounted && theme.isDark;
  const [showOutline, setShowOutline] = useState(false);

  // 判断是否有协作者（仅显示聊天和评论功能）
  const hasCollaborators = onlineUsers && onlineUsers.length > 0;

  // 大文件阈值：超过 100 个块使用虚拟滚动编辑器
  const VIRTUAL_EDITOR_THRESHOLD = 100;
  const useVirtualEditor = blocks.length > VIRTUAL_EDITOR_THRESHOLD;

  // 初始数据获取 & Socket 连接
  useEffect(() => {
    fetchDocument(docId || "demo-doc");
    const currentUserName = typeof window !== 'undefined'
      ? (localStorage.getItem('username') || "Guest")
      : "Guest";

    // 建立 WebSocket 连接
    socketClient.connect((status) => setOnlineStatus(status));
    socketClient.joinDocument(docId, currentUserName);

    // 监听远程块更新
    socketClient.onBlockUpdated((data: any) => {
      if (data.docId === docId) {
        remoteUpdateBlock(data.blockId, data.content, data.properties, data.type);
      }
    });

    return () => { socketClient.disconnect(); };
  }, [docId]);

  // 大文件使用虚拟滚动编辑器
  if (useVirtualEditor) {
    return <VirtualEditor />;
  }

  return (
    <div className={`relative min-h-screen ${isDark ? 'bg-[#0B0F1A]' : 'bg-white'}`}>
      <DocumentMenuBar onToggleOutline={() => setShowOutline(!showOutline)} showOutline={showOutline} />

      <main className={`mx-auto max-w-4xl px-12 pt-32 pb-40 transition-all duration-300 ${showOutline ? 'pr-80' : ''}`}>
        <EditorToolbar />

        {/* Document Title */}
        <input
          type="text"
          className={`w-full text-5xl font-bold border-none outline-none mb-12 bg-transparent placeholder:opacity-40 ${isDark ? 'text-white' : 'text-gray-900'}`}
          placeholder="Untitled"
          value={title}
          onChange={(e) => updateTitle(e.target.value)}
        />

        {/* Blocks Area - 使用 AnimatePresence 实现块添加/删除动画 */}
        <div className="space-y-1">
          <AnimatePresence mode="popLayout">
            {blocks.map((block, index) => (
              <BlockRenderer key={block.id} block={block} index={index} />
            ))}
          </AnimatePresence>
        </div>

        {/* Empty State / Add Block Area */}
        <div
          className="mt-4 opacity-50 hover:opacity-100 transition-opacity cursor-text"
          onClick={() => addBlock("text")}
        >
          <div className="flex items-center gap-2 text-gray-400">
            <Plus size={18} />
            <span>Click to add a block</span>
          </div>
        </div>

        {/* 协作功能 - 仅在有协作者时显示 */}
        {hasCollaborators && (
          <>
            <ChatDrawer />
            <CommentPanel />
          </>
        )}

        {/* 浮动气泡菜单用于文本选择 */}
        <BubbleMenu onFormat={handleFormat} />
      </main>

      <DocumentOutline isVisible={showOutline} onClose={() => setShowOutline(false)} />
    </div>
  );
};

1.3 Markdown 触发器

编辑器支持 Markdown 快捷输入，自动将文本转换为对应块类型：

// 在 BlockRenderer 组件中
const checkMarkdownTrigger = (content: string) => {
  const patterns: Record<string, Block['type']> = {
    "# ": "h1",           // # 标题 -> H1
    "## ": "h2",          // ## 标题 -> H2
    "### ": "h3",         // ### 标题 -> H3
    "- ": "todo",         // - 待办 -> 待办事项
    "[] ": "todo",        // [ ] 待办 -> 待办事项
    "> ": "callout",      // > 引用 -> 提示框
    "---": "divider",     // --- -> 分隔线
    "```": "code"         // ``` -> 代码块
  };

  // 检查精确匹配（如 ---）
  if (patterns[content]) {
    transformBlock(block.id, patterns[content]);
    return true;
  }

  // 检查前缀匹配
  for (const [prefix, type] of Object.entries(patterns)) {
    if (content.startsWith(prefix) && content.length > prefix.length) {
      const newContent = content.slice(prefix.length);
      updateBlock(block.id, newContent, {}, type);
      return true;
    }
  }
  return false;
};

2.1 Socket.io 客户端

// lib/socket.ts
class SocketClient {
  private socket: io.Socket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;

  connect(onStatusChange: (status: 'connected' | 'disconnected' | 'reconnecting') => void) {
    this.socket = io(SOCKET_URL, {
      transports: ['websocket'],
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionDelayMax: 5000,
      reconnectionAttempts: this.maxReconnectAttempts,
    });

    this.socket.on('connect', () => {
      this.reconnectAttempts = 0;
      onStatusChange('connected');
    });

    this.socket.on('disconnect', () => {
      onStatusChange('disconnected');
    });

    this.socket.on('reconnecting', () => {
      this.reconnectAttempts++;
      onStatusChange('reconnecting');
    });
  }

  joinDocument(docId: string, userName: string) {
    this.socket?.emit('joinDocument', { docId, userName });
  }

  updateBlock(docId: string, blockId: string, content: string, type?: string, properties?: any) {
    this.socket?.emit('updateBlock', { docId, blockId, content, type, properties });
  }

  onBlockUpdated(callback: (data: any) => void) {
    this.socket?.on('blockUpdated', callback);
  }

  // ... 更多方法
}

export const socketClient = new SocketClient();

2.2 WebSocket 网关（后端）

// documents.gateway.ts
@WebSocketGateway()
export class DocumentsGateway {
  @WebSocketServer()
  server: Server;

  handleConnection(client: Socket) {
    console.log(`Client connected: ${client.id}`);
  }

  @SubscribeMessage('joinDocument')
  handleJoinDocument(@MessageBody() data: { docId: string; userName: string }, @ConnectedSocket() client: Socket) {
    client.join(data.docId);
    // 通知其他用户
    client.to(data.docId).emit('userJoined', { userName: data.userName });
  }

  @SubscribeMessage('updateBlock')
  handleUpdateBlock(@MessageBody() data: any) {
    // 广播给房间内其他用户
    this.server.to(data.docId).emit('blockUpdated', data);
  }
}

5.1 权限级别定义

enum Permission {
  READ = 'read',
  WRITE = 'write',
  ADMIN = 'admin',
}

interface AccessControl {
  ownerId: string;
  permissions: {
    [userId: string]: Permission[];
  };
  publicLevel: Permission;
}

function checkAccess(ac: AccessControl, userId: string, required: Permission): boolean {
  if (ac.ownerId === userId) return true;
  if (ac.permissions[userId]?.includes(required)) return true;
  if (ac.permissions[userId]?.includes(Permission.ADMIN)) return true;
  if (ac.publicLevel === required) return true;
  return false;
}

6.1 虚拟滚动实现

// VirtualEditor 组件核心逻辑
const VirtualEditor: React.FC = () => {
  const [scrollTop, setScrollTop] = useState(0);
  const containerRef = useRef<HTMLDivElement>(null);

  const ITEM_HEIGHT = 60;
  const OVERSCAN = 5;

  const { visibleItems, totalHeight, startIndex, endIndex } = useMemo(() => {
    const totalHeight = blocks.length * ITEM_HEIGHT;
    const startIdx = Math.max(0, Math.floor(scrollTop / ITEM_HEIGHT) - OVERSCAN);
    const endIdx = Math.min(
      blocks.length - 1,
      Math.ceil((scrollTop + containerHeight) / ITEM_HEIGHT) + OVERSCAN
    );

    return {
      totalHeight,
      startIndex: startIdx,
      endIndex: endIdx,
      visibleItems: blocks.slice(startIdx, endIdx + 1),
    };
  }, [blocks, scrollTop]);

  return (
    <div
      ref={containerRef}
      onScroll={(e) => setScrollTop(e.currentTarget.scrollTop)}
      style={{ height: '100%', overflow: 'auto' }}
    >
      <div style={{ height: totalHeight, position: 'relative' }}>
        {visibleItems.map((block, index) => (
          <div
            key={block.id}
            style={{
              position: 'absolute',
              top: (startIndex + index) * ITEM_HEIGHT,
              height: ITEM_HEIGHT,
            }}
          >
            <BlockRenderer block={block} />
          </div>
        ))}
      </div>
    </div>
  );
};

7.1 文档版本历史

// 版本历史服务
class VersionService {
  async createVersion(docId: string, blocks: Block[]): Promise<Version> {
    const version = await this.versionRepository.save({
      docId,
      content: JSON.stringify(blocks),
      createdAt: new Date(),
    });
    return version;
  }

  async rollback(versionId: string): Promise<Block[]> {
    const version = await this.versionRepository.findById(versionId);
    return JSON.parse(version.content);
  }

  async getVersionHistory(docId: string): Promise<Version[]> {
    return this.versionRepository.find({
      where: { docId },
      order: { createdAt: 'DESC' },
    });
  }
}
